        """
        Set line-in encoding on zp but do not wait for the setting to replicate
        on all HH zps by default
        """

                raise e

    def _get_cap_filter(self, src=None, dst=None, trans=None, recv=None,
                        frame_len=None, min_frame_len=None, max_frame_len=None):
        """
        Return capture filter to be used for packet capture based on source,
        destination, transmitter or receiver mac addresses as well as frame
        length parameters

        :param src: sonos zone component (source)
        :type src: :class:`~sonos.client.internal.SonosZoneComponent`
        :param dst: sonos zone component (destination)
        :type dst: :class:`~sonos.client.internal.SonosZoneComponent`
        :param trans: sonos zone component (transmitter)
        :type trans: :class:`~sonos.client.internal.SonosZoneComponent`
        :param recv: sonos zone component (receiver)
        :type recv: :class:`~sonos.client.internal.SonosZoneComponent`
        :param frame_len: size of frames to capture
        :type frame_len: :obj:'int'
        :param min_frame_len: minimum  size of frames to capture
        :type min_frame_len: :obj:'int'
        :param max_frame_len: max size of frames to capture
        :type max_frame_len: :obj:'int'
        :return: capture fitler generated by he input parameters
        :rtype: :obj:'str'
        """
        filter_ = ''
        prev_len = len(filter_)
        if src is not None:
            filter_, prev_len = self.conditionaly_add_space_and(filter_,
                                                                prev_len)
            filter_ += 'wlan src {}'.format(
                                src.direct_routing_mac_address.replace(':', ''))
        if dst is not None:
            filter_, prev_len = self.conditionaly_add_space_and(filter_,
                                                                prev_len)
            filter_ += 'wlan dst {}'.format(
                                dst.direct_routing_mac_address.replace(':', ''))
        if trans is not None:
            filter_, prev_len = self.conditionaly_add_space_and(filter_,
                                                                prev_len)
            filter_ += 'wlan ta {}'.format(
                            trans.direct_routing_mac_address.replace(':', ''))
        if recv is not None:
            filter_, prev_len = self.conditionaly_add_space_and(filter_,
                                                                prev_len)
            filter_ += 'wlan ra {}'.format(
                            recv.direct_routing_mac_address.replace(':', ''))
        if frame_len is not None:
            filter_, prev_len = self.conditionaly_add_space_and(filter_,
                                                                prev_len)
            # this is confusing because greater and less equivalent to >=, <=
            filter_ += 'greater {} && less {}'.format(frame_len, frame_len)
        if min_frame_len is not None:
            filter_, prev_len = self.conditionaly_add_space_and(filter_,
                                                                prev_len)
            filter_ += 'greater {}'.format(min_frame_len)
        if max_frame_len is not None:
            filter_, prev_len = self.conditionaly_add_space_and(filter_,
                                                                prev_len)
            filter_ += 'less {}'.format(max_frame_len)
        self.logger.info('Capture filter: '.format(filter_))
        return filter_


[docs]    def conditionaly_add_space_and(self, filter_, prev_len):
        """
        Helper method to allow appending filter strings

        :param: filter: filter to conditionally append to
        :type: :obj:'str'
        :param: filter: filter to conditionally append to
        :type: :obj:'str'
        :return: tuple of capture filter with the suffix required to append
            further filter rules, and the resulting length of the filter
        :rtype: :obj:`tuple`
        """
        if len(filter_) > prev_len:
            filter_ += ' && '
        return filter_, len(filter_)

                           qos=None, frame_len=None, max_frame_len=None,
                           min_frame_len=None):
        """
        Return display filter to be used for packet capture based on source,
        destination, transmitter or receiver mac addresses as well as frame
        length parameters in addition to appending any fixture display filter

        :param src: sonos zone component (source)
        :type src: :class:`~sonos.client.internal.SonosZoneComponent`
        :param dst: sonos zone component (destination)
        :type dst: :class:`~sonos.client.internal.SonosZoneComponent`
        :param trans: sonos zone component (transmitter)
        :type trans: :class:`~sonos.client.internal.SonosZoneComponent`
        :param recv: sonos zone component (receiver)
        :type recv: :class:`~sonos.client.internal.SonosZoneComponent`
        :param qos: qos value to filter for
        :type qos: :obj:'int'
        :param frame_len: size of frames to capture
        :type frame_len: :obj:'int'
        :param min_frame_len: minimum  size of frames to capture
        :type min_frame_len: :obj:'int'
        :param max_frame_len: max size of frames to capture
        :type max_frame_len: :obj:'int'
        :return: display fitler generated by he input parameters
        :rtype: :obj:'str'
        """
        filter_ = ''
        filter_len = len(filter_)
        if src is not None:
            filter_, filter_len = self.conditionaly_add_space_and(
                                                            filter_, filter_len)
            filter_ += 'wlan.sa == {}'.format(src.direct_routing_mac_address)
        if dst is not None:
            filter_, filter_len = self.conditionaly_add_space_and(
                                                            filter_, filter_len)
            filter_ += 'wlan.da == {}'.format(dst.direct_routing_mac_address)
        if trans is not None:
            filter_, filter_len = self.conditionaly_add_space_and(
                                                            filter_, filter_len)
            filter_ += 'wlan.ta == {}'.format(trans.direct_routing_mac_address)
        if recv is not None:
            filter_, filter_len = self.conditionaly_add_space_and(
                                                            filter_, filter_len)
            filter_ += 'wlan.ra == {}'.format(recv.direct_routing_mac_address)
        if qos is not None:
            filter_, filter_len = self.conditionaly_add_space_and(
                                                            filter_, filter_len)
            filter_ += 'wlan.qos == {}'.format(qos)
        if frame_len is not None:
            filter_, filter_len = self.conditionaly_add_space_and(
                                                            filter_, filter_len)
            filter_ += 'frame.len == {}'.format(frame_len)
        if max_frame_len is not None:
            filter_, filter_len = self.conditionaly_add_space_and(
                                                            filter_, filter_len)
            filter_ += 'frame.len <= {}'.format(max_frame_len)
        if min_frame_len is not None:
            filter_, filter_len = self.conditionaly_add_space_and(
                                                            filter_, filter_len)
            filter_ += 'frame.len >= {}'.format(min_frame_len)
        if hasattr(self, 'DISPLAY_FILTER'):
            if self.DISPLAY_FILTER is not None:
                filter_, filter_len = self.conditionaly_add_space_and(
                                                            filter_, filter_len)
                filter_ += self.DISPLAY_FILTER
        self.logger.info('Display filter: {}'.format(filter_))
        return filter_



[docs]    def get_qos_filter(self, qos, min_frame_len=None, max_frame_len=None):
        """
        Simple helper method to make qos filters based on expected qos value
        and size of frames independent of devices

        :param qos: value of qos priority flag to filter for
        :type qos: :obj:'int'
        :param min_frame_len: minimum size of frames to filter for
        :type min_frame_len: :obj:'int'
        :param max_frame_len: maximum size of frames to filter for
        :type max_frame_len: :obj:'int'
        :return: qos display fitler generated by he input parameters
        :rtype: :obj:'str'
        """
        if all(frame_len is not None for frame_len in (min_frame_len,
                                                       max_frame_len)):
            self.verifyGreaterOrStop(max_frame_len, min_frame_len,
                            'max frame len must be greater than min frame len')
        qos_filter = 'wlan.qos.priority == {}'.format(qos)
        if min_frame_len is not None:
            qos_filter += ' && frame.cap_len > {}'.format(min_frame_len)
        if max_frame_len is not None:
            qos_filter += ' && frame.cap_len < {}'.format(max_frame_len)
        return qos_filter



[docs]    def determine_compression(self, hh_mode, compression, group_members):
        """
        Determines the expected compression mode and returns the correct filter
        to capture it based on the hh mode, the compression setting and the
        number of group members.

        :param hh_mode: mode of HH, either station or sonosnet
        :type hh_mode: :obj:'str'
        :param compression: compression mode. one of AUTO, COMPRESSED or
            UNCOMPRESSED
        :type compression: :obj:'srt'
        :param group_members: list of group members of the DUT
        :type group_members: :class:`~sonos.client.internal.SonosZoneComponent`
        :return: tuple of expected compression mode and the corresponding qos
            filter
        :rtype: :obj:`tuple`
        """
        if hh_mode == 'station':
            if compression != 'UNCOMPRESSED':
                return ('COMPRESSED', self.COMPRESSED_QOS_FILTER)
            else:
                return ('UNCOMPRESSED', self.UNCOMPRESSED_QOS_FILTER)
        if compression == 'AUTO':
            if len(group_members) > 2:
                qos_filter = self.COMPRESSED_QOS_FILTER
                expected_mode = 'COMPRESSED'
            else:
                qos_filter = self.UNCOMPRESSED_QOS_FILTER
                expected_mode = 'UNCOMPRESSED'
        else:
            expected_mode = compression
            if compression == 'COMPRESSED':
                qos_filter = self.COMPRESSED_QOS_FILTER
            else:
                qos_filter = self.UNCOMPRESSED_QOS_FILTER
        return (expected_mode, qos_filter)

            frame_len=None, min_frame_len=None, max_frame_len=None):
        """

        receiver address will be the mac of the first wire

        :param cap_file: the name of the capture file to filter with the qos

        :type cap_file: :obj:`str`
        :param src: sonos zone component (source)
        :type src: :class:`~sonos.client.internal.SonosZoneComponent`
        :param dst: sonos zone component (destination)
        :type dst: :class:`~sonos.client.internal.SonosZoneComponent`
        :param qos_value: qos value to filter for
        :type qos_value: :obj:'int'
        :param frame_len: size of frames to capture
        :type frame_len: :obj:'int'
        :param min_frame_len: minimum  size of frames to capture
        :type min_frame_len: :obj:'int'
        :param max_frame_len: max size of frames to capture
        :type max_frame_len: :obj:'int'
        :param display_filter: additional filter to apply to the parsed qos
            filter
        :type display_filter: :obj:'str'
        :return: The resulting capture with applied display [qos] filter
        :rtype: pyshark capture
        """
        disp_filter = self.get_display_filter(src=src,
                                              dst=dst,
                                              qos=qos_value,
                                              frame_len=frame_len,
                                              min_frame_len=min_frame_len,
                                              max_frame_len=max_frame_len)
        file_cap = pyshark.FileCapture(input_file=cap_file,
                                       display_filter=disp_filter)
        file_cap.load_packets(timeout=self.FILE_CAP_TIMEOUT)
        file_cap.close()
        return file_cap

            src=None, dst=None, frame_len=None, min_frame_len=None,
            max_frame_len=None):
        """
        Verifies the cap_file meets the capture threshold with positive or
        negative verification per the given parameters

        :param cap_file: the name of the capture file to check
        :type cap_file: :obj:`str`
        :param negative_test: is test positive or negative
        :type negative_test: :obj:`bool`
        :param qos_value: qos value to filter for
        :type qos_value: :obj:'int'
        :param src: sonos zone component (source)
        :type src: :class:`~sonos.client.internal.SonosZoneComponent`
        :param dst: sonos zone component (destination)
        :type dst: :class:`~sonos.client.internal.SonosZoneComponent`
        :param frame_len: size of frames to capture
        :type frame_len: :obj:'int'
        :param min_frame_len: minimum  size of frames to capture
        :type min_frame_len: :obj:'int'
        :param max_frame_len: max size of frames to capture
        :type max_frame_len: :obj:'int'
        """
        if hasattr(self, 'MIN_CAP_LEN') == False:
            self.logger.error('verify_capture_threshold needs to be called by '
                'a fixture which has specified MIN_CAP_LEN else the capture '
                'length threshold defualts to 1')
            self.MIN_CAP_LEN = 1
        parsed_cap = self.parse_cap_for_qos(cap_file, qos_value, src, dst)
        if nagative_test:
            self.verifyLessOrStop(len(parsed_cap), self.MIN_CAP_LEN,
                'Negative test: {} should be less than min session cap '
                'size {}'.format(parsed_cap, self.MIN_CAP_LEN))
        else:
            self.verifyGreaterEqualOrStop(len(parsed_cap), self.MIN_CAP_LEN,
                'Positive test: {} should be greater than min session cap '
                'size {}'.format(parsed_cap, self.MIN_CAP_LEN))



[docs]    def run_qos_capture(self, hh_mode, line_in_zp, group_members, compression):
        """
        Takes input parameters and determine the expected qos flag to be used
        between devices, runs a packet capture with the expected filters based
        on the inputs and either pass or fail if the expected qos flag and src
        and dsr is found or not found

        :param hh_mode: mode of HH, either station or sonosnet
        :type hh_mode: :obj:str
        :param line_in_zp: line in DUT device which will send audio to the GMs
        :type line_in_zp: :class:`~sonos.client.internal.SonosZoneComponent`
        :param group_members: list of group members of the DUT
        :type group_members: :class:`~sonos.client.internal.SonosZoneComponent`
        :param compression: compression mode. one of AUTO, COMPRESSED or
            UNCOMPRESSED
        :type compression: :obj'str'
        """
        cap_filter = self._get_cap_filter(line_in_zp, group_members[0])
        [zp.AVTransport.join_group(line_in_zp) for zp in group_members]
        # pause here because stopping playback allows the
        # compression state to be recalculated based on grouping
        # see https://confluence.sonos.com/display/CS/Line-in+Audio+Compression
        if line_in_zp.AVTransport.is_playing():
            line_in_zp.AVTransport.Pause_and_wait()
        line_in_zp.AVTransport.Play_and_wait()
        wait_until_true(lambda: all([gm.raw_udn in
            line_in_zp.ZoneGroupTopology.get_zone_player_uuids_in_group() for
            gm in group_members]))
        wait_until_true(lambda: all([line_in_zp.raw_udn in
            gm.AVTransport.get_avtransport_uri() for gm in group_members]))
        wait_until_true(lambda: all([gm.AVTransport.is_playing for gm in
                                     [line_in_zp] + group_members]),
            reason='line_in_zp and all group_members should be playing audio')
        expected_mode, qos_filter = self.determine_compression(hh_mode,
                                                    compression, group_members)
        self.logger.info('Testing {} mode {} compression with'
            ' {} group members (should be {} with qos filter {})'.format(
                hh_mode, compression, len(group_members), expected_mode,
                qos_filter))
        self._run_sniffer_trace(
            'test_{}_{}_line_in_with_{}_gms'.format(
                                    hh_mode, compression, len(group_members)),
            line_in_zp, cap_filter, qos_filter)

    def _ungroup_all(self):
        """
        Helper function to make sure all ZPs used in test
        are no longer part of a group.
        """
        self.logger.debug('Ungrouping all ZPs in testbed')
        ungroup_zones = []
        for zp in self.my_devices:
            if zp.is_playback_device():
                ungroup_zones.append(zp)
        run_concurrent_processes(lambda zp:
                                 zp.AVTransport.leave_group_and_wait(),
                                 ungroup_zones)

    def _redo_packet_capture(self, cap_filter, capture_file):
        """
        Helper function to toggle wireless channel and redo packet capture
        in case initial capture failed

        :param cap_filter: BPF filter to be applied to LiveCapture
        :type cap_filter: :obj:'str'
        :param capture_file: path to capture file for outputting capture data
        :type capture_file: :obj:'str'
        """
        self.logger.info('Failed to capture any packets. Toggling '
                         'wireless channel may fix the problem')
        if self.cap_channel in [1, 11]:
            temp_chan = 6
        else:
            temp_chan = self.cap_channel + 1
        self._set_wireless_channel(temp_chan)
        self._set_wireless_channel(self.cap_channel)

        self.logger.info('Retrying live capture for {} '
                         'secs with capture filter set to '
                         '{}'.format(self.SNIFF_SAMPLE_TIME,
                                    cap_filter))
        cap = pyshark.LiveCapture(interface=self.WIRELESS_CAP_ID,
                                  output_file=capture_file,
                                  bpf_filter=cap_filter,
                                  only_summaries=True)
        cap.sniff(timeout=self.SNIFF_SAMPLE_TIME)
        return cap

    def _run_sniffer_trace(self, testname, dut, cap_filter,
            disp_filter, verify_none=False):
        """
        Set up wireless channel and start packet capture with pyshark

        :param testname: name of test to be used as name of capture file
        :type testname: :obj:'str'
        :param dut: sonos zone component
        :type dut: :class:`~sonos.client.internal.SonosZoneComponent`
        :param cap_filter: BPF filter to be applied to LiveCapture
        :type cap_filter: :obj:'str'
        :param disp_filter: display filter to be applied to FileCapture
        :type disp_filter: :obj:'str'
        :param verify_none: if True, verify that packets subject to the given
            filters were NOT captured
        :type verify_none: :obj:'bool'
        """
        freq = dut.cli.get_wireless_channel('ath0')
        self.cap_channel = channel_frequency_to_number(freq)
        temp_chan = [x for x in [1, 6, 11] if x != self.cap_channel][0]
        self._set_wireless_channel(temp_chan)
        self._set_wireless_channel(self.cap_channel)
        capture_file = '/tmp/{}.pcap'.format(testname)
        self.logger.info('Starting live capture for {} '
                         'secs with capture filter set to '
                         '{}'.format(self.SNIFF_SAMPLE_TIME,
                                    cap_filter))
        cap = pyshark.LiveCapture(interface=self.WIRELESS_CAP_ID,
                                  output_file=capture_file,
                                  bpf_filter=cap_filter)
        cap.sniff(timeout=self.SNIFF_SAMPLE_TIME)
        cap.close()
        if len(cap) == 0:
            cap = self._redo_packet_capture(cap_filter, capture_file)
            cap.close()

        self.logger.info('Capture completed. File {} created. '
            'Starting capture file analysis.'.format(capture_file))

        file_cap = pyshark.FileCapture(input_file=capture_file,
                                       display_filter=disp_filter)
        file_cap.load_packets(timeout=self.FILE_CAP_TIMEOUT)
        file_cap.close()
        self.logger.info('Filtered capture file {} with filter set to '
                         '{}'.format(capture_file, disp_filter))

        if verify_none:
            self.verifyEqualOrStop(0, len(file_cap),
                "Verify no packets were captured")
        else:
            self.verifyGreaterOrStop(len(file_cap), 0,
                "Verify expected data packets were captured")

    def _capture_dut_frames(self, cap_file, dut=None, dut_interface=None,

        name(s) to collect a sniffer trace and store it in the corersponding
        trace file

        if dut == None:
            dut = self.dut
        freq = dut.cli.get_wireless_channel(dut_interface or 'ath0')
        self.cap_channel = channel_frequency_to_number(freq)
        temp_chan = [x for x in [1, 6, 11] if x != self.cap_channel][0]
        self._set_wireless_channel(temp_chan)
        self._set_wireless_channel(self.cap_channel)
        if hasattr(self, 'cap_filter'):
            cap_filter = self.cap_filter
        self.logger.debug('performing packet capture: outfile: '
                          '{}, bpf: {}, duration {}'.format(
                                cap_file, cap_filter, self.SNIFF_SAMPLE_TIME))
        cap = pyshark.LiveCapture(interface=self.WIRELESS_CAP_ID,
                                  output_file=cap_file,
                                  bpf_filter=cap_filter,
                                  only_summaries=True)
        cap.sniff(timeout=self.SNIFF_SAMPLE_TIME)
        self.logger.debug('end packet capture: '
                          'outfile: {}, bpf: {}, duration {}'.format(
                                cap_file, cap_filter, self.SNIFF_SAMPLE_TIME))
        cap.close()
