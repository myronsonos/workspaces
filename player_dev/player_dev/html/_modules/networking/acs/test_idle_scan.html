[docs]    def wait_until_zone_on_ht_chan(self, primary, zone, timeout=45):
        """
        Waits until zone is on the ht channel of the primary

        :param primary: the primary to wait for zone to be on ht channel of
        :type zone: :class:`~sonos.client.internal.SonosZoneComponent`
        :param zone: the zone to wait for on primary ht channel
        :type zone: :class:`~sonos.client.internal.SonosZoneComponent`
        """
        wait_until_true(
            lambda: self.athconfigs[zone].getchannel(INTERFACE_2GHZ) ==
            self.athconfigs[primary].getchannel(INTERFACE_5GHZ),
            timeout_seconds=timeout, iteration_delay=5,
            reason='waited {} s for sat chan {} == ht master\' ht '
            'chan {}'.format(
                timeout,
                self.athconfigs[zone].getchannel(INTERFACE_2GHZ),
                self.athconfigs[primary].getchannel(INTERFACE_5GHZ)))



[docs]    def wait_for_nol_timeout(self, zone, channel):
        """
        Waits until the zone is no longer blacklisting channel due to radar and
        performs a scan before the black list ends to prove the interval is
        honored, keeps the device scanning

        :param zone: the zone to wait for nol on channel
        :type zone: :class:`~sonos.client.internal.SonosZoneComponent`
        :param channel: wireless channel to wait for return from nol
        :type channel: :obj:`int`
        """
        # disable cac to regulate scan iteration times also if cac abort cac
        self.set_cac_enable(zone, 0)
        start_time = time.time()
        try:
            wait_until_true(lambda: self.is_channel_in_acs_table(channel, zone),
                iteration_delay=self.nol_timeout * 0.2,
                timeout_seconds=(self.nol_timeout + (self.MIN_SCAN_INTERVAL +
                                 self.TOTAL_ACS_SCAN_TIME + self.cac_interval)),
                retry=lambda: self.scan_if_not_scanning(zone),
                reason='channel {} should not be in zone {} acs table '
                    'following radar for {} seconds plus one following '
                    'completion time'.format(channel, zone, self.nol_timeout))
        except TimeoutError:
            self.set_cac_enable(zone, 1)
            self.fail('channel {} should not be in zone {} acs table following '
                'radar for {} seconds plus one following completion '
                'time'.format(channel, zone, self.nol_timeout))
        nol_time = time.time() - (start_time)
        # return cac to operation
        self.set_cac_enable(zone, 1)
        self.verifyGreaterEqualOrFailCase(nol_time, self.nol_timeout,
            'channel {} should not be in ACS channels list before noltimeout {}'
            ' expires'.format(channel, self.nol_timeout))
        self.verifyAlmostEqualOrStop(nol_time, self.nol_timeout,
            'channel {} should return to zone\'s {} acs table after for '
            'noltimeout {} and subsequent scan time {}'.format(channel, zone,
                self.nol_timeout, self.TOTAL_ACS_SCAN_TIME),
            delta=(self.MIN_INITIAL_SCAN_INTERVAL +
                   self.TOTAL_ACS_SCAN_TIME * 2))



[docs]    def verify_ht_quickly_flees_radar(self, ht_master, sats, dfs_channel):
        """
        Verifies that zones which are on a DFS channel leave quickly in response
        to a radar event. The test is rather simplistic only verifying our leave
        channel time is less than the allowed data bucket interval of 200 ms.
        Compliance does more minute timing test for regulatory so we are just
        making sure we leave pretty quickly with master being the last to leave

        .. note::

            * zones must leave the channel by 10 seconds
            * zones must stop transmitting data frames within 200 ms of the
              event
            * zones must not transmit for more than 60 ms worth of management
              frames within the 10 seconds after the initial radar event

        :param ht_master: the primary zone to check ht for post radar dwell
            time
        :type ht_master: :class:`~sonos.client.internal.SonosZoneComponent`
        :param sats: list of the satellites of primary which must flee before
            primary
        :type sats: :obj:'list'
        :param dfs_channel: wireless DFS channel verify zones flee time against
        :type channel: :obj:`int` or :obj:`str`
        """
        # create and object for the connection to shunt connection manager as
        # it increases latency
        mastercli = ht_master.cli
        # the nol must be enabled for the DFS radar avoidance logic to kick in
        self.set_nol_enable(ht_master, 1)
        radar_event_time = time.time()
        mastercli.command(
            '/wifi/athconfig bangradar ath1', wait_for_response=False)
        wait_until_true(lambda: dfs_channel not in
                mastercli.command('/wifi/athconfig getchannel ath1'),
            timeout_seconds=1, iteration_delay=0.02,
            reason=('channel switch should happen in under 200 ms, waited '
                '1 s'))
        flee_time = time.time() - radar_event_time
        self.verifyTrueOrFailCase(flee_time < self.fcc_data_tx_radar_time and
                all([str(dfs_channel) != self.athconfigs[sat].getchannel(
                                            INTERFACE_2GHZ) for sat in sats]),
            'primary and its satellites should have fled radar channel {} '
            'quickly by {} time: {} sat chans {}'.format(dfs_channel,
                self.fcc_data_tx_radar_time, flee_time,
                [self.athconfigs[sat].getchannel(
                                            INTERFACE_2GHZ) for sat in sats]))




[docs]    def verify_nol_timeout(self, channel, zone, event_time):
        """
        Verifies that when a DFS channel has been blacklisted by a detected
        radar event that it is excluded from use for the nol_timeout which is
        30 minutes by default

        :param channel: wireless channel to verify nol interval on
        :type channel: :obj:`int`
        :param zone: the zone the nol interval against on channel
        :type zone: :class:`~sonos.client.internal.SonosZoneComponent`
        :param event_time: at which the radar event occured on zone
        :type event_time: :obj:`flaot`
        :return: whether or not the zone tx while CAC
        :rtype: :obj:`bool`
        """
        wait_until_true(lambda: channel != int(self.athconfigs[
                zone].getchannel(INTERFACE_5GHZ)),
            iteration_delay=2,
            timeout_seconds=10)
        fled_radar_time = time.time() - event_time
        self.verifyLessEqualOrFailCase(fled_radar_time,
            self.flee_radar_interval, message='zone {} must leave the DFS '
            'channel within {} seconds when radar is detected, took {} '
            'seconds'.format(zone, self.flee_radar_interval,
                             fled_radar_time))
        self.verifyFalseOrFailCase(
            self.is_channel_in_acs_table(channel, zone),
            'channel :{} should be on nol and not in acs table for use '
            '{}'.format(channel, zone))
        self.wait_for_nol_timeout(zone, channel)
        self.is_post_nol_channel(zone, channel)

    def test_scan_once_idle(self, ht_master):
        """
        Verifies the operation of ACS given the control signal of idle manager's
        client 6 state (5GHz idle state). When client 6 is active we don't idle
        scan, when it is idle we idle scan

        The idle manager client 6 for HT masters without satellites is never
        so they are always idle, this is implicitly verified by test single
        interval

        :param ht_master: the zone to test
        :type zone: :class:`~sonos.client.internal.SonosZoneComponent`
        """
        self.transition_testbed_sats_to_wireless()
        if not ht_master.get_satellites():
            create_ht_50([ht_master, self.sats[0], self.sats[1]])
        ht_master.AVTransport.Play_buzzer_and_wait()
        for sat in self.sats:
            self.wait_until_zone_on_ht_chan(ht_master, sat)
        wait_until_true(lambda: ht_master.idlemgr_client[6],
            iteration_delay=ITERATION_DELAY, timeout_seconds=10,
            reason='waiting for {} client 6 to become active'.format(ht_master))
        wait_until_true(
            lambda:(self.get_scan_active_state(ht_master) !=
                    ht_master.idlemgr_client[6]),
            iteration_delay=ITERATION_DELAY, timeout_seconds=3,
            reason='waiting for {} scan_active != idlemgr_client[6]'.format(
                ht_master))
        ht_master.AVTransport.Stop_and_wait()
        wait_until_true(lambda: not ht_master.idlemgr_client[6],
            iteration_delay=ITERATION_DELAY, timeout_seconds=10,
            reason='waiting for {} client 6 to become idle'.format(ht_master))
        wait_until_true(
            lambda:(self.get_scan_active_state(ht_master) !=
                    ht_master.idlemgr_client[6]),
            iteration_delay=ITERATION_DELAY,
            timeout_seconds=self.MIN_INITIAL_SCAN_INTERVAL + 3,
            reason='waiting for {} scan_active {} != idlemgr_client[6] '
                '{}'.format(
                    ht_master,
                    self.get_scan_active_state(ht_master),
                    ht_master.idlemgr_client[6]))
        wait_until_true(lambda: self.is_zone_scanning(ht_master),
            iteration_delay=ITERATION_DELAY,
            timeout_seconds=3,
            reason='waiting for {} to scan while idle'.format(ht_master))

    def test_cac_with_radar_nol(self, ht_master):
        """
        Verifies that when we do detect radar that we leave the channel within
        10 seconds from the radar detection and when a channel detects radar is
        it added to the no occupancy list and is blacklisted for 30 minutes
        after which it will require CAC before use

        :param ht_master: the zone to test
        :type zone: :class:`~sonos.client.internal.SonosZoneComponent`
        """
        # Don't rely on the DUT CACing at the start of the test as it may be
        # just about to finish the running CAC, which is 60 seconds long,
        # instead force a new scan and CAC to guarantee we can get the 3 cli
        # commands to the DUT before the CAC time runs out. Catching the DUT
        # in an already running scan gives us at least 60 seconds to do this.
        self.scan_if_not_scanning(ht_master)
        self.wait_for_next_idle_scan_to_start(ht_master)
        self.wait_for_cac_begin(ht_master,
            interval_delay=self.TOTAL_ACS_SCAN_TIME * 0.4,
            timeout_seconds=(self.TOTAL_ACS_SCAN_TIME * 1.3))
        cac_channel = int(self.athconfigs[ht_master].getchannel(INTERFACE_5GHZ))
        self.set_nol_enable(ht_master, 1)
        self.set_nol_timeout(ht_master, self.nol_timeout)
        radar_event_time = time.time()
        self.athconfigs[ht_master].bangradar(INTERFACE_5GHZ)
        self.verify_nol_timeout(cac_channel, ht_master, radar_event_time)

    def test_radar_flee_time_with_sats(self, ht_master):
        """
        Verifies that when a radar event is detected by HT with satellites that
        the primary and satellites stop transmitting data frames within 200 ms
        and leave the channel within 10 seconds. During that 10 seconds, after
        the initial 200 ms, only management frames with an aggregate
        transmission time of 60 ms may be sent. This test does not verify such
        details and instead sets the pass criteria at a channel switch at less
        than 200 ms. This is because if we do utilize the time post 200 ms all
        satellites would drop out until going to a new channel.

        :param ht_master: the zone to test
        :type zone: :class:`~sonos.client.internal.SonosZoneComponent`
        """
        self.transition_testbed_sats_to_wireless()
        if not ht_master.get_satellites():
            create_ht_50([ht_master, self.sats[0], self.sats[1]])
        acs_channel_table = self.get_acs_channel_table(ht_master)
        if not self._get_cleared_dfs_channels(acs_channel_table):
            self.logger.debug('ht_master has no cleared DFS channels {}'.format(
                ht_master))
            ht_master.AVTransport.Play_buzzer_and_wait()
            ht_master.AVTransport.Stop_and_wait()
            self.wait_for_cac_begin(ht_master,
                interval_delay=self.TIME_TO_START_CAC_POST_PLAY * 0.2,
                timeout_seconds=self.TIME_TO_START_CAC_POST_PLAY + 3)
        wait_until_true(lambda: self._get_cleared_dfs_channels(
                self.get_acs_channel_table(ht_master)),
            iteration_delay=self.cac_interval * 0.2,
            timeout_seconds=self.cac_interval + 2,
            reason='zone {} should finish CAC of one channel'.format(ht_master))
        acs_channel_table = self.get_acs_channel_table(ht_master)
        dfs_chans_cleared = self._get_cleared_dfs_channels(acs_channel_table)
        best_cm_dfs_chan = str(self._get_channels_sorted_by_metric(
            dfs_chans_cleared)[0])
        ht_master.AVTransport.Play_buzzer_and_wait()
        self.athconfigs[ht_master].setchannel(INTERFACE_5GHZ, best_cm_dfs_chan)
        try:
            wait_until_true(
                lambda:(
                    self.athconfigs[self.sats[0]].getchannel(INTERFACE_2GHZ) ==
                    best_cm_dfs_chan),
                timeout_seconds=10,
                reason='sat {} chan should be best dfs chan {}'.format(
                    self.sats[0], best_cm_dfs_chan))
        except TimeoutError as e:
            self.fail(
                'sat should be on best dfs chan {}, {}'.format(
                    best_cm_dfs_chan, e))
        try:
            self.verify_ht_quickly_flees_radar(ht_master, self.sats,
                                           best_cm_dfs_chan)
        except TimeoutError as e:
            self.stop(
                'primary and sats should have left dfs chan quickly (less than '
                '200 ms) after radar event. {}'.format(e))
        ht_master.AVTransport.Stop_and_wait()

    Verifies that ht masters do not try to clear cac during an idle scan
    after all DFS channels have already been cleared

    .. seealso::

        QMetry `P-8674: All DFS Channels Cleared CAC \
        <https://qmetry.sonos.com/qmetry/Main.php?type=tm&do=mytestCaseDetailsMain&type=tm&linkRedirect=yes&typeId=11010_2>`_

    def _clear_all_dfs_channels(self, zone):
        """
        Clears all dfs channels of zone

        :param ht_master: the zone to test
        :type zone: :class:`~sonos.client.internal.SonosZoneComponent`
        """
        self.logger.info("Clearing CAC on all DFS channels for {}".format(
            zone))
        for x in range(2):
            channels_needing_cac = len(self._get_needscac_dfs_channels(
                zone.diag.get_primary().acs.channels_table))
            for cac_left in range(channels_needing_cac, 0, -1):
                self.logger.info("Clearing CAC on {} more channels".format(
                    cac_left))
                self.athconfigs[zone].acs(INTERFACE_5GHZ, 1)
                guaranteed_sleep(
                    self.TOTAL_ACS_SCAN_TIME + self.cac_interval + 10)
