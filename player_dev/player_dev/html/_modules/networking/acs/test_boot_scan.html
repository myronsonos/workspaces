# All relevant 5GHz channels we use as per IEEE 802.11
SUPERSET_5GHZ = [36, 40, 44, 48, 52, 56, 60, 64,
                 100, 104, 108, 112, 116, 120, 124, 128, 132, 136, 140, 144,
                 149, 153, 157, 161, 165]

# Used for negative cases
NON_PERMITTED = [32, 68, 170]

# NOTE: We currently do not use ch 144 as it was recently made available other
# than region 3: china we also don't support 120, 124, 128 in EU because we
# don't have them in US as a result of the overlap with Doppler weather radar
# and we only support the higher non DFS channels for china because the effort
# does not yet merit otherwise
# see https://jira.sonos.com/browse/SWPBL-19744
# see https://jira.sonos.com/browse/SWPBL-21319
USED_CHANNELS = FrozenAttributeDict((
    (MdpRegion.USCA, [36, 40, 44, 48, 52, 56, 60, 64,
        100, 104, 108, 112, 116,
        132, 136, 140,
        149, 153, 157, 161, 165]),
    (MdpRegion.EU, [36, 40, 44, 48, 52, 56, 60, 64,
        100, 104, 108, 112, 116, 132, 136, 140]),
    (MdpRegion.CHINA, [149, 153, 157, 161, 165]),
    (MdpRegion.JAPAN, [36, 40, 44, 48, 52, 56, 60, 64,
        100, 104, 108, 112, 116,
        132, 136, 140])))

DFS_CHANNELS = FrozenAttributeDict((
    (MdpRegion.USCA, [52, 56, 60, 64,
        100, 104, 108, 112, 116, 132, 136, 140, 144]),
    (MdpRegion.EU, [52, 56, 60, 64,
        100, 104, 108, 112, 116, 120, 124, 128, 132, 136, 140]),
    (MdpRegion.CHINA, [52, 56, 60, 64]),
    (MdpRegion.JAPAN, [52, 56, 60, 64,
        100, 104, 108, 112, 116,
        132, 136, 140])))

[docs]    def setUpFixture(self):
        super(AcsBootScanWorkflowFixture, self).setUpFixture()
        self.master_configure_dev_files = dict()
        for master in self.ht_masters[:]:
            if not master.cli.does_configure_dev_exist():
                master.cli.set_configure_dev_file()
                self.master_configure_dev_files[master] = ''
            else:
                self.master_configure_dev_files[master] = master.cli.command(
                    'cat /jffs/system/Configure.dev')
            master.cli.command(
                'echo "/wifi/athconfig nolenable ath1 0" >> '
                '/jffs/system/Configure.dev')
            master.cli.command('chmod +x /jffs/system/Configure.dev')
            self.verifyFalseOrStop(master.HTControl.get_toslink_connected(),
                "Verify HT Master {} toslink input is not active".format(
                    master))
            if master.cli.mdp.region != MdpRegion.USCA:
                self.logger.info("MDP region on {} was not set to USA, "
                    "resetting it prior to running tests".format(master))
                master.cli.mdp.region = MdpRegion.USCA
            master.force_serial_cli = True
        # prevent new serial connections from logging us out after new
        # UPnP device is detected post reboot
        self.remove_serial_port_callback_from_devices_list()

[docs]    def verify_channels_scanned(self, channel_list, region):
        """
        Verifies that the number of channels passed in is almost equal to the
        total USA ACS channels (20).

        .. note::

            Almost equals is used here because during boot we must use a serial
            Cli connection which has a longer polling timeout to prevent false
            detection of being "prompt ready" during boot.  This longer time
            causes us to consistently miss ~3 channels every boot scan.

        :param channel_list: a list of channels that were scanned
        :type channel_list: :obj:`list`
        """
        self.verifyAlmostEqualOrStop(region['expected_channels'],
            len(channel_list),
            "Verify number of ACS channels scanned during boot",
            delta=4)



[docs]    def verify_acs_chans_per_region(self, ht_master, region_data):
        """
        Verifies both positive and negative cases for available HT channels in
        each region for all primaries based on IEEE 802.11 lists and performs a
        single smoke test boot scan against each to verify operation one each
        region

        :param ht_master: the primary to verify the channels of against region
        :type ht_master: :class:`~sonos.client.internal.SonosZoneComponent`
        :param region_data: region in which to verify the channels of
        ht_master in
        :type region_data: :class:`~custom_dicts.FrozenAttributeDict`
        """
        for channel in SUPERSET_5GHZ + NON_PERMITTED:
            radar_mitigation_attempts = 2
            for attempt in range(radar_mitigation_attempts + 1):
                self.logger.info(
                    "setting {} channel to {} for {!r}".format(
                        ht_master, channel, region_data.region))
                res = ht_master.cli.command('wifi/athconfig setchannel {} '
                        '{}'.format(INTERFACE_5GHZ, channel))
                if res == 'ioctl failed':
                    self.logger.error('{} setchannel to {} returned '
                            '{}'.format(ht_master, channel, res))
                    # check if there was unexpected radar on channel and retry
                    if channel in USED_CHANNELS[region_data.region]:
                        if attempt == radar_mitigation_attempts:
                            raise StopRunningTestException(
                                'suspect unexpected radar detected while '
                                'verifying chans per region on {}'.format(
                                    ht_master))
                        radar_logs = ht_master.cli.command(
                            "dmesg | grep 'dfs: radar detected on {}'".format(
                                channel))
                        if len(radar_logs) > 0:
                            ht_master.cli.reboot()
                    else:
                        self.verifyTrueOrFailCase(
                            channel not in USED_CHANNELS[region_data.region],
                            'ioctl failure expected for unused channel {} in '
                            '{!r}'.format(channel, region_data.region))
            # sleep as continuously hitting the ioctls can fail
            guaranteed_sleep(0.2)
        # verify all the channels in the table match their expectations
        self.logger.info(
            'verifying valid channels and flags for {}'.format(ht_master))
        verified_channels = []
        acs_channels_table = self.get_acs_channel_table(ht_master)
        for channel in acs_channels_table:
            channel_number = channel_frequency_to_number(
                channel.frequency)
            self.verifyTrueOrFailCase(
                channel_number in SUPERSET_5GHZ,
                'channel {} of master should be in 5GHz channel '
                'superset'.format(channel_number))
            self.verifyTrueOrFailCase(
                channel_number in USED_CHANNELS[region_data.region],
                'channel {} of master should be in used_channel for '
                '{!r}'.format(channel_number, region_data.region))
            self.verifyTrueOrFailCase(
                bool(channel.flags & AcsFlags.DFS) ==
                (channel_number in DFS_CHANNELS[region_data.region]),
                'channel {} dfs status should match dfs_channels'
                ' per region {!r}'.format(channel, region_data.region))
            verified_channels.append(channel_frequency_to_number(
                channel.frequency))
        # verify all expected channels are in the table and as expected
        self.verifyTrueOrFailCase(
            sorted(verified_channels) == USED_CHANNELS[region_data.region],
            'zone {} 5GHz channels reported should match expected, '
            'difference: {}'.format(ht_master,
                sorted(list(set(USED_CHANNELS[region_data.region]) -
                    set(verified_channels)))))



[docs]    @combinatorial('ht_masters')
    def test_scan(self, ht_master):
        for region_info in REGION_DATA:
            expect_cac = (False if region_info.region == MdpRegion.CHINA else
                True)

            ht_master.cli.mdp.region = region_info.region
            ht_master.cli.reboot(wait=True)

            self.verify_acs_scan(ht_master, expect_cac, region_info,
                is_boot_scan=True)
            self.verify_acs_chans_per_region(ht_master, region_info)
