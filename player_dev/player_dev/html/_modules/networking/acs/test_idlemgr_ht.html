def play(fixture, *args, **kwargs):
    av = fixture.get_relevant_avt()
    fixture.master.RenderingControl.SetVolume(00)

    try:
        av.zp.RenderingControl.SetVolume(00)
        av.zp.GroupRenderingControl.SetGroupVolume(00)
        av.Play_and_wait()

    except UPnPError as e:
        # If it turns out that there is no AVT set, or that there are
        # no tracks in the queue, then I don't want to error out. Random
        # event sequences won't always have something ready to play.
        if e.upnp_error == UPNP_ERROR_701_TRANSITION_UNAVAILABLE:
            fixture.logger.warning("nothing to play")
        else:
            raise

[docs]def toslink_autoplay(fixture, *args, **kwargs):
    av = fixture.master.AVTransport
    # This is a bit of a hack, but the input must have been silent for
    # at least 8 seconds for autoplay to work reliably, so we'll sleep
    # until that condition is satisfied.
    if time.time() - last_silence_start < 8:
        guaranteed_sleep(8 - (time.time() - last_silence_start))
    fixture.master.diag.testpoint.ht_spdif_override = "signal"
    wait_until_true(
        lambda: 'x-sonos-htastream' in av.get_avtransport_uri(),
        iteration_delay=.5, timeout_seconds=15, info_threshold=15)
    # We expect toslink_autoplay to ungroup as long as it's set, which is
    # the default.
    wait_until_true(
        lambda: not fixture.is_master_grouped(),
        iteration_delay=.5, timeout_seconds=15, info_threshold=15)

        """
        You can also generate a random set of event sequences:
        self.event_sequences = generate_random_sequences(events, 10, 100)

        Or take a test name and generate an event_sequence (e.g. to re-run a

        self.event_sequences = \
            [recover_sequence_from_name("become_gm__set_avt_24__play")] * 10
        """

        for i, event_sequence in enumerate(self.event_sequences):
            method_name = \
                "__".join(["test"] +
                          [x.__name__ for x in event_sequence if x] +
                          [str(i), str(len(self.event_sequences))])
            self.attach_test(method_name,
                             self.template_for_tests,
                             event_sequence)

        self.already_tested = set()

        newfunc = partial(self.template_for_tests, *args, **kwargs)
        newfunc.__name__ = name
        setattr(self, name, newfunc)
        self._testMethodNames[type(self)].append(name)

[docs]    def template_for_tests(self, events):
        expected = None
        for i, event in enumerate(events):
            if not event:
                continue

            self.logger.info("doing event {}".format(event))

            event(self)

            if self.get_accumulated_events(events, i) in self.already_tested:
                self.logger.info(
                    "Already tested {}, skipping verification".format(
                        self.get_accumulated_events(events, i)))
                continue

            play_state = self.get_play_state()
            self.logger.debug("play state: {}".format(play_state))
            expected = self.expected_scan_state(play_state)
            self.logger.debug("expected scan state: {}".format(expected))

            self.logger.debug(
                "result of {}: {}".format(self.get_accumulated_events(events,
                                                                      i),
                                          expected))
            self.verify_powersave(expected['powersave_ath1'])
            self.verify_scan_5GHz(expected['scan5'])
            self.verify_client(expected['client'], 0)
            self.verify_client(expected['client'], 6)

            self.already_tested.add(self.get_accumulated_events(events, i))

[docs]    def verify_powersave(self, expected):
        powersave_map = {True: "sleep", False: "awake"}

        def check():
            return not self.get_acs(self.master).scan.active

        # Wait until we're not actively ACS scanning to get a truthful
        # powersave reading.
        wait_until_true(check, iteration_delay=0.5,
                        timeout_seconds=90, info_threshold=5)

        self.actual_ps_state = None
        def check():
            self.actual_ps_state = (self.master.diag.get_fullstatus('ath1').
                powersave.state)
            return (powersave_map[expected] == self.actual_ps_state)

        try:
            wait_until_true(check, iteration_delay=0.5,
                            timeout_seconds=10, info_threshold=5)
        except TimeoutError:
            pass

        self.verifyEqualOrFailCase(
            powersave_map[expected], self.actual_ps_state,
            "should have correct power save state on ath1")

[docs]    def get_play_state(self):
        state = AttributeDict(grouped=False,
                              playing_24=False,
                              playing_toslink=False)

        state.grouped = self.is_master_grouped()

        av = self.master.AVTransport
        if av.is_group_member():
            state.playing_24 = (self.get_gc(self.master).
                                AVTransport.is_playing())
        elif av.is_playing():
            uri = av.get_avtransport_uri()
            if 'x-sonos-htastream' in uri:
                state.playing_toslink = True
            else:
                state.playing_24 = True
        return state



[docs]    def expected_scan_state(self, state):
        """
        Given a play/grouping/satellite state, what is the expected scan state?
        """

        scan_state = AttributeDict(scan24=True,
                                   scan5=True,
                                   powersave_ath1=True,
                                   client={})

        master_is_playing = state.playing_24 or state.playing_toslink

        if master_is_playing:
            if self.master.get_ht_satellites():
                scan_state.scan5 = False
                scan_state.client[6] = True
            else:
                scan_state.scan5 = True
                scan_state.client[6] = False
        else:
            scan_state.scan5 = True
            scan_state.client[6] = False

        if state.playing_24 or (state.playing_toslink and state.grouped):
            scan_state.client[0] = True
        else:
            scan_state.client[0] = False

        if self.master.get_ht_satellites():
            scan_state.powersave_ath1 = False

            if self.ath1_in_powersave:
                # When sats are first configured, the master's ath1 powersave
                # stays in the state it was in until changed (music starts
                # playing).
                if not master_is_playing:
                    scan_state.powersave_ath1 = True
                else:
                    self.ath1_in_powersave = False
        else:
            scan_state.powersave_ath1 = True

        return scan_state




[docs]def generate_random_sequences(events, length, number):

    events = [play, play, play, play,
              toslink_autoplay, toslink_autoplay,
              stop, pause,
              toslink_autostop_silence, toslink_autostop_nosignal,
              add_to_queue, clear_queue, set_avt_24, clear_avt,
              become_gc, become_gm, ungroup,
              ]
    sequences = []
    for i in range(number):
        sequence = []
        for j in range(length):
            sequence.append(random.choice(events))
        sequences.append(sequence)
    return sequences
