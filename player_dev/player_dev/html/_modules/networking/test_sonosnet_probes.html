    def _validate_packet_values(self, packet, field_val_dict, layer):
        """
        Helper function to validate the fields values found in a
        particular layer of a packet

        :param capture: pyshark packet
        :type capture: :class:`pyshark.Packet`
        :param field_val_dict: dictionary of pyshark packet field to
            expected value mapping.
            For example {'ssid':'Sonos_FbraA8FDSCsgcDLngd0mAzNFaF'}
        :type field_val_dict: :obj:`dict`
        :param layer: pyshark packet layer name, ie. radiotap, wlan,
            wlan_mgt, etc.
        :type layer: :obj:`str`
        """
        # The following IEs need to have their byte order reformatted
        little_endian_ies = ['sonos_ie_boot_seqnum', 'sonos_ie_htchan']
        packet_layer = getattr(packet, layer)
        for field, val in field_val_dict.items():
            actual_val = getattr(packet_layer, field, '')
            if actual_val:
                if field in little_endian_ies:
                    actual_val = self._reformat_ie(actual_val)
                elif 'uuid' in field:
                    actual_val = self._reformat_uuid(actual_val)
            verify_msg = ("Verify {} is {}found in {} layer {} in packet"
                          " {{}}".format(field, "not " if not val else "",
                                         layer,
                                         "with val of {} ".format(val) if val else ""))
            self.verifyEqualOrStop(val, actual_val,
                                   verify_msg.format(packet.wlan.seq))

    def _verify_basic_probe(self, zc, packet_list):
        """
        Test to verify some basic properties of probes

        :type zc: :class:`~sonos.client.internal.SonosZoneComponent`
        :param zc: sonos zone
        :param packet_list: list of pyshark packets
        :type packet_list: :obj:`list` of :class:`pyshark.Packet`

            routing_flags = ''
        else:
            routing_flags = '01'

            packet_list)

    def _verify_sequence_and_antenna_ie(self, zc, packet_list):
        """
        Verify seq num and ant of each packet. The seq nums should be incrementing
        and the ant should alternate between 0 and 1 on 11g devices. This test
        assumes that there are 11g devices in the HH. As a result, devices
        send each packet twice, with same sequence number.

        :type zc: :class:`~sonos.client.internal.SonosZoneComponent`
        :param zc: sonos zone
        :param packet_list: list of pyshark packets (filtered for probes and DUT
                            source address)
        :type packet_list: :obj:`list` of :class:`pyshark.Packet`
        """
        first_packet = packet_list[0]
        ie = getattr(getattr(first_packet, 'wlan_mgt'),
                     'sonos_ie_seq_and_ant', '')
        last_seq_num, last_ant = [int(item, 16) for item in ie.split(':')]
        self.logger.debug("Starting seq num: {} Starting ant: {}"
                          .format(last_seq_num, last_ant))
        same_seq_num_count = 0
        wrap_count = 0
        for packet in packet_list[1:]:
            ie = getattr(getattr(packet, 'wlan_mgt'), 'sonos_ie_seq_and_ant', '')
            seq_num, ant = [int(item, 16) for item in ie.split(':')]
            self.logger.debug("IE: {}".format(ie))
            self.logger.debug("Last_seq_num: {} seq_num: {}".format(last_seq_num,
                                                                    seq_num))
            """
            The sequence number is a uint8. As such, the sequence number will
            wrap after 255. In this case, it is possible that the current
            sequence number is less than the last. Given the length of the sniff
            time, this should, also, only happen once.
            """
            diff = seq_num - last_seq_num
            if diff < 0:
                diff = seq_num
                same_seq_num_count = 0
                wrap_count += 1
                """
                A failure here would indicate that the sequence numbers are not
                increasing
                """
                self.verifyLessEqualOrFailCase(wrap_count, 1, "The sequence "
                                               "number should wrap no more "
                                               "than once")
            elif diff == 0:
                same_seq_num_count += 1
                self.verifyLessOrFailCase(same_seq_num_count, 2, "A packet with"
                                          " the same sequence number should be"
                                          " seen no more than twice")
            else:
                same_seq_num_count = 0
            if zc.is_bg_only():
                if same_seq_num_count:
                    self.verifyEqualOrFailCase(int(not last_ant), ant, "If we've"
                                               " seen a packet with the seq num"
                                               " twice, the ant should have switched")
                else:
                    self.verifyLessEqualOrFailCase(ant, 1, "The current antenna"
                                                   " should be either 0 or 1")
                last_ant = ant
            else:
                self.verifyEqualOrFailCase(0, ant, "The antenna should be 0")
            self.verifyLessEqualOrFailCase(diff, 5, "The difference in "
                                           "sequence numbers should be small")
            last_seq_num = seq_num

    def _reformat_ie(self, value):
        """
        Reformats the value given by sonos_ie_boot_seqnum. When taken from
        pyshark, this value is a string of the form xx:xx:xx:xx, which represents
        a little endian byte string. This method removes the colons and
        converts the string to an int

        :param value: The value received from wlan_mgt.sonos_ie_boot_seqnum
        :type value: :obj:`str`
        :return: The value converted to an int
        :rtype: :obj:`int`
        """
        # Strip colon from value
        value = value.replace(':', '')
        # Change endianness and convert to int
        value = ''.join(reversed([value[i:i + 2] for i in range(0, len(value), 2)]))
        return int(value, 16)

    def _reformat_uuid(self, value):
        """
        Wireshark/tshark returns the uuid as string of 24 hex digits separated
        by a colon. This function strips the colons and converts the resulting
        string from hex digits to their ASCII equivelent.

        :param value: The value received from wlan_mgt.sonos_ie_uuid
        :type value: :obj:`str`
        :return: The UUID
        :rtype: :obj:`str`
        """
        uuid = ''
        value = value.replace(':', '')
        for i in range(0, len(value), 2):
            uuid += str(unichr(int(value[i:i + 2], 16)))
        return uuid

    def start_scan_and_count_probes(self, chan, zc):
        """
        Helper function to trigger scan and capture resulting probes
        on a specific channel

        :param chan: channel to sniff on
        :type chan: :obj:`int`
        :param zc: sonos zone that will be scanning
        :type zc: :class:`~sonos.client.internal.SonosZoneComponent`
        :rtype: :obj:`int`
        """
        SNIFF_SEC = 3
        SNIFFER_PREP_TIME_SEC = 1.2  # ensure we're monitoring before scanstart

        self.set_wireless_channel(WIRELESS_CAP_ID, chan)

        output_file = '{}_{}.pcap'.format(zc.modelNumber, chan)
        bpf_filter = 'wlan addr2 {}'.format(zc.cli.get_mac_address(INTERFACE_2GHZ))

        def capture_func(**kwargs):
            # Remove existing pcap file to prevent PCAP corruption issues
            if os.path.exists(kwargs['output_file']):
                os.remove(kwargs['output_file'])
            timeout = kwargs.pop('timeout')
            capture = pyshark.LiveCapture(**kwargs)
            capture.set_debug()
            capture.sniff(timeout=timeout)
            sleep(0.250)  # yield in order to prevent broken pipe errors
            capture.close()
            del capture

        background_capture = Process(target=capture_func,
                                     kwargs={'interface': WIRELESS_CAP_ID,
                                             'timeout': SNIFF_SEC,
                                             'output_file': output_file,
                                             'bpf_filter': bpf_filter})

        self.logger.info('Capture for {}s on chan {}'.format(SNIFF_SEC, chan))
        background_capture.start()
        sleep(SNIFFER_PREP_TIME_SEC)
        self.athconfigs[zc].scanstart(INTERFACE_2GHZ, 0)
        background_capture.join()
        self.logger.info('Capture completed.')

        display_filter = (
            '''
            {} and
            frame.time_relative < {} and
            wlan.da == ff:ff:ff:ff:ff:ff and
            wlan_mgt.ds.current_channel == {} and
            radiotap.datarate == 1 and
            wlan.fc.retry == 0 and
            wlan.fcs.status == 1 and
            wlan_mgt.ssid == ""
            '''.format(PROBE_REQ_FILTER, SNIFF_SEC, chan))

        try:
            cap = self.post_process_capture(input_file=output_file,
                                            display_filter=display_filter)
        except:
            raise SonosnetProbesException("PCAP file error!")

        n_probes = len(cap)
        cap.close()
        if n_probes > 1:
            return n_probes
        else:
            raise SonosnetProbesException(
                "{} failed to capture probes on channel {}".format(zc, chan))



[docs]    def run_basic_probe_capture(self, zp, interface, output_file=None):
        """
        Performs a wireless packet capture and filters the results to return
        probes only.

        :param zp: sonos zone
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`
        :param interface: The zp's wifi interface (Used to determine channel)
        :type interface: :obj:`str`
        :return: the post processed packets
        :rtype: :class:`pyshark.FileCapture`
        """
        channel = channel_frequency_to_number(zp.cli.get_wireless_channel(interface))
        self.logger.info("Capture probes to {} for {} secs".format(
            output_file, SNIFF_TIME))

        self.run_wireless_monitor_capture(WIRELESS_CAP_ID,
                                          channel,
                                          timeout=SNIFF_TIME,
                                          output_file=output_file)

        self.logger.info('Capture completed.')

        return self.post_process_capture(interface=WIRELESS_CAP_ID,
                                         display_filter=PROBE_REQ_FILTER,
                                         input_file=output_file)

    def test_vendor_ies_for_ht(self, ht_master, zp_l, zp_r):
        """
        Test to verify probes sent from HT masters and satellites the correct
        vendor specific IEs. The test configures 5.0 system, sniffs for probes,
        and verifies the IEs contain the correct values.

        :param ht_master: The master device for the HT configuration
        :type ht_master: :class:`~sonos.client.internal.SonosZoneComponent`
        :param zp_l: The left satellite for the HT configuration
        :type zp_l: :class:`~sonos.client.internal.SonosZoneComponent`
        :param zp_r: The right satellite for the HT configuration
        :type zp_r: :class:`~sonos.client.internal.SonosZoneComponent`
        """
        ht_master.DeviceProperties.create_50_config(zp_l, zp_r)
        ht_master.cli.command('/wifi/athconfig scanenable ath1 0')
        capture_2ghz = self.run_basic_probe_capture(ht_master, INTERFACE_2GHZ,
            output_file='test_vendor_ies_for_{}_2ghz.pcap'.format(
                ht_master.serialNumber))
        master_mac_2ghz = ht_master.direct_routing_mac_address.lower()
        master_packets_2ghz = filter(lambda pkt: pkt.wlan.sa == master_mac_2ghz,
                                     capture_2ghz)
        self.verifyGreaterOrStop(
            len(master_packets_2ghz), 0,
            "The test should have captured packets from the master on 2ghz")
        """
        When surrounds are wired connect:amp or hideout, ath0 will be disabled
        on the surround SAT. SWPBL-23315 and SWPBL-96426.
        Therefore, HT master will not have an HT channel on its ath1.
        So we will skip validating probes of ht master on its ath1 and
        probes of SATs on their ath0.
        """
        if not zp_l.is_zp120() and not zp_l.is_s16():
            capture = self.run_basic_probe_capture(ht_master, INTERFACE_5GHZ,
                output_file='test_vendor_ies_for_{}_5ghz.pcap'.format(
                ht_master.modelNumber))

            channel = ht_master.cli.get_wireless_channel(INTERFACE_5GHZ)
            master_mac = ht_master.mac_address_5ghz.lower()
            self.logger.debug("Master: {} 5Ghz MAC: {}".format(ht_master,
                                                           master_mac))
            sat_mac = zp_l.direct_routing_mac_address.lower()
            master_packets = filter(lambda pkt: pkt.wlan.sa == master_mac,
                                    capture)

            self.verifyGreaterOrStop(
                len(master_packets), 0,
                "The test should have captured packets from the master")

            """
            Since the devices are all wired, they should all advertise an empty
            string for the routing flags.
            """
            routing_flags = ''
            master_ie_val_dict = {
                'ssid': self.hhid,
                'sonos_ie_bridge_addr': ht_master.mac_address.lower(),
                'sonos_ie_routing_flags': routing_flags,
                'sonos_ie_uuid': ht_master.raw_udn,
                'sonos_ie_sat_data': SAT_DATA_IE,
                'sonos_ie_htchan': channel}
            map(lambda pkt: self._validate_packet_values(
                    pkt, master_ie_val_dict, "wlan_mgt"),
                master_packets)
            """
            The htmac IE is included in probes sent on the 2GHz interface,
            if the ht master is bonded with satellites. This allows satellites that
            are rebooted, or fall off of 5GHz, to find the master and move to the
            correct channel
            """
            map(lambda pkt: self._validate_packet_values(
                    pkt, {'sonos_ie_htmac': master_mac}, "wlan_mgt"),
                master_packets_2ghz)
            sat_packets = filter(lambda pkt: pkt.wlan.sa == sat_mac, capture)
            self.verifyGreaterOrStop(
                len(sat_packets), 0,
                "The test should have captured packets from the satellite")
            sat_ie_val_dict = {
                'ssid': self.hhid,
                'sonos_ie_bridge_addr': zp_l.mac_address.lower(),
                'sonos_ie_routing_flags': routing_flags,
                'sonos_ie_primary_uuid': ht_master.raw_udn}
            map(lambda pkt: self._validate_packet_values(
                    pkt, sat_ie_val_dict, "wlan_mgt"),
                sat_packets)
            capture.clear()
            capture.close()
        ht_master.DeviceProperties.cleanup_bonded_configuration()
        ht_master.cli.command('/wifi/athconfig scanenable ath1 1')
        capture_2ghz.clear()
        capture_2ghz.close()

                                                         {'datarate':"24"},

            probes_11g = filter(lambda pkt: pkt.radiotap.datarate == "24",
                                packet_list)
            self.verifyInRangeOrFailCase(len(probes_11n),
                                             PASSING_MIN_PROBE_COUNT_11N,
                                             PASSING_MAX_PROBE_COUNT_11N,
                                              "Verify number of 11n probes sent.")

                                              PASSING_MIN_PROBE_COUNT_11N,
                                             PASSING_MAX_PROBE_COUNT_11N,
                                              "Verify number of 11g probes sent.")
            self.verifyEqualOrFailCase(len(packet_list),
                                       len(probes_11n) + len(probes_11g),
                                       "Verify all probes are sent at "

    def _validate_infra_data(self, packet, ant0, ant1, neighbor):
        """
        Validates that the packet contains the infra data IE and that the
        RSSI value for the given neighbor is within the range of values listed
        in /proc/ath_rincon/status for the interface under test.

        :param packet: The packet to verify
        :type packet: :class:`pyshark.Packet`
        :param high_rssi: The top end of the RSSI range for the neighbor
        :type high_rssi: :obj:`int`
        :param low_rssi: The bottom end or the RSSI range for the neighbor
        :type low_rssi: :obj:`int`
        :param neighbor: The mac address of the neighbor in question
        :type neighbor: :obj:`str`
        """
        actual_val = getattr(getattr(packet, "wlan_mgt"), 'sonos_ie_infra_data', '')
        if not actual_val:
            self.fail('Failed to find infra data in probe')
            return
        result = re.search(neighbor.lower() + ':(?P<ant0>[0-9 | a-f]{2}):(?P<ant1>[0-9 | a-f]{2}).*',
                           actual_val)
        if result:
            actual_ant0_val = int(result.group('ant0'), 16)
            actual_ant1_val = int(result.group('ant1'), 16)
            self.verifyEqualOrFailCase(ant0, actual_ant0_val, 'ant0 RSSI from probe'
                                       ' should match RSSI value found in'
                                       ' /proc/ath_rincon_<netif>/status for'
                                       ' given neighbor ({})'.format(neighbor))
            self.verifyEqualOrFailCase(ant1, actual_ant1_val, 'ant1 RSSI from probe'
                                       ' should match RSSI value found in'
                                       ' /proc/ath_rincon_<netif>/status for'
                                       ' given neighbor ({})'.format(neighbor))
        else:
            self.fail("Failed to get RSSI for {} (IE: {})".format(neighbor,
                                                                  actual_val))

    def test_infra_data_ie(self, zc):
        """
        Test infra data IE

        :type zc: :class:`~sonos.client.internal.SonosZoneComponent`
        :param zc: sonos zone
        """
        if zc.is_br100():
            self.skip('Skipping BRIDGE due to test points being unsupported')

        captures = {}
        fullstats = {}
        brctl = Brctl(zc)
        if zc.is_ht_master_capable():
            zc.DeviceProperties.create_50_config(self.zp_l, self.zp_r)
            interfaces.append(INTERFACE_5GHZ)
            zc.cli.command('/wifi/athconfig scanenable ath1 0')
        self.set_neighbor_rssi(zc, INTERFACE_2GHZ)
        # Wait for stale RSSI values to be aged out of fullstatus
        sleep(33)
        for interface in interfaces:
            capture = self.run_basic_probe_capture(zc, interface,
                output_file='test_infra_data_ie_{}.pcap'.format(zc.modelNumber))

            fullstats[interface] = Fullstatus(zc, interface)
            if interface == INTERFACE_2GHZ:
                zc_mac = zc.direct_routing_mac_address.lower()
            else:
                zc_mac = zc.mac_address_5ghz.lower()
            captures[interface] = filter(lambda pkt: pkt.wlan.sa == zc_mac,
                                         capture)
            capture.clear()
            capture.close()

        # Get neighbor/showports
        neighbor_info = brctl.showports()
        """
        Verify that each testbed device is found in the DUTs bridge table
        """
        neighbor_macs = [neighbor.mac for neighbor in neighbor_info]
        for device in self.my_devices:
            if device == zc:
                continue
            self.verifyTrueOrStop(device.direct_routing_mac_address in neighbor_macs,
                                  "{} should be found in DUTs ({}) bridge table"
                                  .format(device.direct_routing_mac_address, zc))

        # Check neighbors RSSI against IE
        for neighbor in neighbor_info:
