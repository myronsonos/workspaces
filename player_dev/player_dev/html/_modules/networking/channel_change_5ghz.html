[docs]    def change_home_region(self, zps, region):
        """
        Sets the home region of the Sonos device

        Arguments:
        zps -- list of SonosZoneComponents
        region --MdpRegion, region code.  1 - USA, 2 - Europe, 3 - China, 4 - Japan

        Keyword Arguments:
        None

        Returns:
        None
        """
        for i in range(len(zps)):
            self.logger.info("ZP: %s - Setting home mdp region to: %s" % (zps[i], region))
            if zps[i].cli.mdp.set_region(region):
                # reboot needed after changing mdp region
                self.logger.info("Rebooting %s" % zps[i])
                self._reboot(zps[i])
            else:
                self.logger.error("Failed to change region code to %s on %s" % (region, zps[i]))


    def _start_chsnk(self):
        """
        Private method for starts chsnk recording

        Arguments:
        None

        Keyword Arguments:
        None

        Returns:
        boolean
        """
        is_recording = False
        for i in range(len(self.ht_devices)):
            if self.ht_devices[i].diag.start_recording_chsnk_stats():
                self.logger.info("Recording chsnk stats on %s" % self.ht_devices[i])
                is_recording = True
        return is_recording

    def _stop_chsnk(self):
        """
        Private method that stops chsnk recording

        Arguments:
        None

        Keyword Arguments:
        None

        Returns:
        boolean
        """
        is_stopped = False
        for i in range(len(self.ht_devices)):
            if self.ht_devices[i].diag.stop_recording_chsnk_stats():
                self.logger.info("Stopped recording chsnk stats on %s" % self.ht_devices[i])
                is_stopped = True
        return is_stopped

    def _review_chsnk(self):
        """
        Private method that reviews chsnk stats and evaluates pass/fail criteria for:
            'elapsed_secs': number of seconds since recording began
            'LSE': number of Large Sync Errors
            'missed_packets': number of missed packets
            'num_out_of_order_packets': number of out of order packets
            'num_NAKed_packets': number of NAK'ed packets

        Arguments:
        None

        Keyword Arguments:
        None

        Returns:
        boolean
        """
        review = False
        for i in range(len(self.ht_devices)):
            stats = self.ht_devices[i].diag.get_chsnk_stats()
            elapsed_secs = stats.get('elapsed_secs')
            num_lse = stats.get('LSE')
            self.verifyEqualOrFailCase('0', num_lse, "%s - Large sync error(s) reported after %s seconds: %s"
                                          % (self.ht_devices[i], elapsed_secs, num_lse))

            missed_packets = stats.get('missed_packets')
            self.verifyGreaterEqualOrFailCase(MISSED_THRESHOLD, missed_packets, "%s - missed packets: %s" % (self.ht_devices[i], missed_packets))

            # Want rest of misc packets reported
            del stats['elapsed_secs'], stats['LSE'], stats['missed_packets']
            for key in stats.iterkeys():
                for value in stats.itervalues():
                    self.verifyGreaterEqualOrWarn(MISC_THRESHOLD, value, "%s - misc packets: %s: %s" % (self.ht_devices[i], key, value))
            review = True
        return review

    def _do_ht_config_and_playback(self, htconfigs):
        """
        Private method that creates a 5.1 Home Theater configuration and initiate SPDIF (TV) playback

        Arguments:
        htconfigs -- WorkFlowExecutionConfiguration list

        Keyword Arguments:
        None

        Returns:
        boolean
        """
        self.logger.info("Adding HT Satellites")
        uri = 'x-sonos-htastream:%s:spdif' % self.s9.raw_udn
        for htconfig in htconfigs:
                self.helpers.zones.setup_device_configuration(htconfig, self.ht_devices)
                if not self.helpers.playback.is_playing(self.s9):
                    self.logger.info("Loading spdif uri and starting tv playback")
                    self.helpers.playback.load_playback_resource(self.s9, uri)
                    self.helpers.playback.start_playback(self.s9)
                else:
                    self.logger.info("HT system is already playing")

    def _wait_for_cac(self, freq, timeout=65):
        """
        Private method that waits for Channel Availablility Check (CAC).  The maximum CAC time is 60 sec.
        Anything after 65 sec indicates the channel was forcefully changed during CAC.
        Notifies if radar is detected.

        Arguments:
        freq -- str, channel frequency

        Keyword Arguments:
        None

        Returns:
        None
        """
        start_time = time.time()
        stop_time = start_time
        while self.s9.cli.get_advertising_channel() is None:
            advertising_channel = self.s9.cli.get_advertising_channel()
            self.logger.debug(advertising_channel)
            self.logger.debug("CAC in progress...")
            time.sleep(5)
            if self.s9.cli.detect_radar(freq):
                self.logger.warn("Radar detected on {}".format(freq))
                break
            if time.time() - start_time > timeout:
                now_channel = self.s9.cli.get_wireless_channel('ath1')
                self.verifyEqualOrFailCase(freq, now_channel,
                    "Verify channel does not change during CAC")
                break

    def _verify_channel_change(self, freq):
        """
        Private method that verifies channel change

        Arguments:
        freq -- str, channel frequency

        Keyword Arguments:
        None

        Returns:
        None
        """
        advertising_channel = self.s9.cli.get_advertising_channel()
        sat_sub_channels = [self.s3_left.cli.get_wireless_channel('ath0'),
            self.s3_right.cli.get_wireless_channel('ath0'),
            self.sub.cli.get_wireless_channel('ath0')]

        for channel in sat_sub_channels:
            if advertising_channel is not None:
                self.verifyEqualOrFailCase(freq == advertising_channel,
                    freq == channel,
                    'Channel should change across all satellites: '
                    '{} == {}'.format(advertising_channel, channel))

    def _set_wifi_channels(self, freq):
        """
        Private method that sets desired channel, waits for CAC, detect if radar is present and verifies
        channel change

        Arguments:
        freq -- str, channel frequency

        Keyword Arguments:
        None

        Returns:
        None
        """
        # change from unicode to str as change_wireless_channel requires str
        self.verifyEqualOrStop('PLAYING',
            self.helpers.playback.get_current_transport_state(self.s9),
            "TV playback should be initiated")
        if self._start_chsnk():
            self.logger.info("ZP: {} - Setting wifi channel to: {}".format(
                self.s9, freq))
            if self.s9.cli.change_wireless_channel('ath1', freq):
                # allow 2 seconds to capture any sync errors after channel change
                time.sleep(2)
                self._wait_for_cac(freq)
                self._verify_channel_change(freq)
            else:
                self.logger.warn("{} is on NOL".format(freq))
        else:
            self.logger.error("Failed to record chsnk stats")

        if self._review_chsnk() and self._stop_chsnk():
                self.logger.info("Reviewing chsnk stats completed for channel: "
                    "{}".format(freq))
        else:
            self.logger.error("Failed to evaluate chsnk stats for channel: "
                "{}".format(freq))
