[docs]    def test_encrypt_keys_basic(self):
        """
        This test verifies the following encryption key properties on
        a zp given the output of dumpregs on each dut ZP:
        1. Since the dut zps are in station mode with ap configured
        for wpa2 psk aes by default, key slot 4 should have mac of ap
        with AES-CCM as key type
        2. key slots 0-3 are reserved for GTK from the AP
        3. key slots 5 until the end are reserved for direct routed
        sonos peers and they should be the same key for all peers
        since they are in the same HH.
        4. all encryption keys are hexidecimal and 8 bytes long
        5. inherent in checking for points 1-4, we verify that
        there is some key in slots 0-4 for station mode capable zps.
        """
        for zc in self.zps:
            self.logger.info("====== Verifying {} DUT".format(zc.modelName))
            peers = set([peer.direct_routing_mac_address.lower()
                         for peer in self.zps if peer is not zc])
            res = filter(None, self.dumpregs[zc].get_dump_regs(option='-k'))
            keys_0_3 = [x.split()[self.key_col[zc]] for x in res[0:4]]
            s = set()
            duplicates = set(x for x in keys_0_3 if x in s or s.add(x))
            self.verifyEqualOrFailCase(self.ap_bssid.lower(),
                                       res[4].split()[2].lower(),
                                       "Verify mac of ap in slot index 4")
            self.verifyEqualOrFailCase("AES-CCM",
                                       res[4].split()[3],
                                       "Verify encryption type "
                                       "in ptk slot index 4")
            res_index = zc.cli.command('cat /proc/ath_rincon/station |'
                                 ' grep "type: AES"')
            gtk_index_used = list(set(re.findall('GTKState([0,1,2,3])', res_index)))
            for gtk in gtk_index_used:
                self.verifyNotInOrFailCase(res[int(gtk)].split()[self.key_col[zc]],
                                           duplicates,
                                           "Verify GTK is unique")
            actual_peers = set([x.split()[2].lower() for x in res[5:]])
            self.verifyEqualOrFailCase(peers,
                                       actual_peers,
                                       "Verify key slot entry for each expected peer.")
            peers_keys = set([x.split()[self.key_col[zc]].lower() for x in res[5:]])
            all_keys = set([x.split()[self.key_col[zc]].lower().replace('-', '') for x in res])
            self.verifyEqualOrFailCase(1,
                                       len(peers_keys),
                                       "Verify all peers have the same encryption key.")
            peers_key_type = set([x.split()[3] for x in res[5:]])
            self.verifyEqualOrFailCase(set(["AES-CCM"]),
                                       peers_key_type,
                                       "Verify all peers have AES-CCM encryption key type.")
            for key in all_keys:
                self.verifyTrueOrFailCase(all(c in string.hexdigits for c in key),
                                          "verify encryption key {} "
                                          "is hexidecimal".format(key))
                self.verifyEqualOrFailCase(32,
                                           len(key),
                                           "Verify encryption key {} "
                                           "is 8 byes long".format(key))

    def test_encrypted_data_ap(self, zc):
        """
        Test to validate the proper bits set in the encrypted
        data frames sent to an AP:
        - ZP sends encrypted data to AP with fc protected bit set
        - ZP sends encrypted data to AP with wep key id 0 in 802.11 header
        - initialization vector is incremented per frame for non-tries
        - When ZP disconnects from the AP, the PTK and GTK keys entries
        are reset

        :type zc: :class:`~sonos.client.internal.SonosZoneComponent`
        :param zc: sonos zone
        """

        self.run_basic_data_encrypt_test(zc,
                                         WIRELESS_CAP_ID,
                                         CHANNEL,
                                         zc.cli.get_mac_address('ath0'),
                                         self.ap_eth0_mac,
                                         self.ap.ap_info['ip'],
                                         ping_count=10,
                                         ping_size=1000,
                                         sniff_time=10,
                                         key_id=0,
                                         wlan_fc_ds='0x01')
        res_before = filter(None, self.dumpregs[zc].get_dump_regs(option='-k'))[:5]
        ptk_before = res_before[4].split()[self.key_col[zc]].lower()
        res = zc.cli.command('cat /proc/ath_rincon/station |'
                                 ' grep "type: AES"')
        gtk_index_used = list(set(re.findall('GTKState([0,1,2,3])', res)))
        try:
            self.add_wire(zc)
            res_after = filter(None, self.dumpregs[zc].get_dump_regs(option='-k'))
            self.verifyEqualOrFailCase('00:00:00:00:00:00',
                                       res_after[4].split()[2].lower(),
                                       "Verify AP  mac in slot index 4 is cleared")
            ptk_after = res_after[4].split()[self.key_col[zc]].lower()
            self.verifyNotEqualOrFailCase(ptk_before,
                                          ptk_after,
                                          "Verify AP PTK is reset")
            for gtk in gtk_index_used:
                gtk_before = res_before[int(gtk)].split()[self.key_col[zc]].lower()
                gtk_after = res_after[int(gtk)].split()[self.key_col[zc]].lower()
                self.verifyNotEqualOrFailCase(gtk_before,
                                              gtk_after,
                                              "Verify AP GTK is reset")
        finally:
            self.remove_wire(zc, wait=False)

    def test_encrypted_data_peer(self, zc):
        """
        Test to validate the proper bits set in the encrypted
        data frames sent to a zc peer in the same HH:
        - ZP sends encrypted data to peer with fc protected bit set
        - ZP sends encrypted data to peer with wep key id 2 in 802.11 header
        - initialization vector is incremented per frame for non-tries
        - When peer is removed from bridge table, peer encryption key is
        removed as well.

        :type zc: :class:`~sonos.client.internal.SonosZoneComponent`
        :param zc: sonos zone
        """
        # get some other peer that is not the dut zc
        peer = filter(lambda x: x is not zc, self.zps)[0]
        self.run_basic_data_encrypt_test(zc,
                                         WIRELESS_CAP_ID,
                                         CHANNEL,
                                         zc.direct_routing_mac_address,
                                         peer.direct_routing_mac_address,
                                         peer.ip,
                                         ping_count=10,
                                         ping_size=1000,
                                         sniff_time=10,
                                         key_id=2,
                                         wlan_fc_ds='0x00')
        try:
            peer_mac = peer.direct_routing_mac_address
            self.disable_zones(*[peer])
            sleep(45)
            res = filter(None, self.dumpregs[zc].get_dump_regs(option='-k'))
            active_peers = set([x.split()[2].lower() for x in res[5:]])
            self.verifyNotInOrFailCase(peer_mac,
                                       active_peers,
                                       "Verify key has been removed for peer that is gone")
        finally:
            self.enable_zones(*[peer])

    def test_nfwssid_deauth_at_bootup(self, zc):
        """
        Test to validate that a zc will scan for the
        ssid specified in its nfwssid.txt during bootup
        and send a deauth to it.
        Need to test this for station mode and sonosnet
        mode bootup.

        :type zc: :class:`~sonos.client.internal.SonosZoneComponent`
        :param zc: sonos zone
        """
        output_file = '{}_reboot.pcap'.format(zc.modelName)
        bpf_filter = 'wlan addr2 {}'.format(zc.cli.get_mac_address('ath0'))
        display_filter = ("(wlan.fc.type_subtype == 0x0024 and "
                          "wlan.da == {}) and "
                          "(wlan.fc.type_subtype == 0x000c or "
                          "wlan.fc.type_subtype == 0x000a".format(self.ap_bssid))
        verify_msg = "Verify deauth sent at bootup for {}".format(zc)
        zc.cli.reboot(wait=False)
        self.run_basic_sniffer_test(WIRELESS_CAP_ID, CHANNEL,
                                    output_file, bpf_filter,
                                    display_filter,
                                    1,
                                    2,
                                    verify_msg,
                                    timeout=150)
        zc.wire.enable()
        zc.wait_for_sonosnet_mode()
        zc.cli.reboot(wait=False)
        self.run_basic_sniffer_test(WIRELESS_CAP_ID, CHANNEL,
                                    output_file, bpf_filter,
                                    display_filter,
                                    1,
                                    2,
                                    verify_msg,
                                    timeout=150)
        zc.wire.disable()
        zc.wait_for_station_mode()
