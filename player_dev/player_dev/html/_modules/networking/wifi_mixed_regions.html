    def _reboot(self, device):
        # Reboot and block until it detects that all services have been added back to the underlying
        # Coherence RootDevice object.  The default timeout = 45 sec
        device.reboot()

    def _ht_config(self, htconfigs):
        """
        Private method that creates a 5.1 Home Theater configuration

        Arguments:
        htconfigs -- WorkFlowExecutionConfiguration list

        Keyword Arguments:
        None

        Returns:
        boolean
        """
        self.logger.info("Adding HT Satellites")
        for zp in htconfigs:
            self.logger.debug("setting up configuration: %s" % zp)
            if zp.is_bonded():
                self.helpers.zones.setup_device_configuration(zp, self.my_devices)
            else:
                self.logger.error("Incorrect bonded config")

    def _wait_for_cac(self, freq):
        """
        Private method that waits for Channel Availablility Check (CAC).  The maximum CAC time is 60 sec.
        Anything after 65 sec indicates the channel was forcefully changed during CAC.
        Notifies if radar is detected.

        Arguments:
        freq -- str, channel frequency

        Keyword Arguments:
        None

        Returns:
        None
        """
        timeout = 65
        start_time = time.time()
        stop_time = start_time
        while self.s9.cli.get_advertising_channel() is None:
            advertising_channel = self.s9.cli.get_advertising_channel()
            self.logger.debug(advertising_channel)
            self.logger.debug("CAC in progress...")
            time.sleep(5)
            if self.s9.cli.detect_radar(freq):
                self.logger.warn("Radar detected on %s" % freq)
                break
            new_time = time.time()
            delta = new_time - stop_time
            if delta > timeout:
                now_channel = self.s9.cli.get_wireless_channel('ath1')
                self.verifyLessOrFailCase(delta, timeout, "Channel should not change during CAC. Should be: %s, but is now %s" %
                                          (freq, now_channel))
                break

    def _verify_channel_change(self, freq):
        """
        Private method that verifies channel change

        Arguments:
        freq -- str, channel frequency

        Keyword Arguments:
        None

        Returns:
        None
        """
        advertising_channel = self.s9.cli.get_advertising_channel()
        sat_sub_channels = dict(
            lr=self.sat_lr.cli.get_wireless_channel('ath0'),
            rr=self.sat_rr.cli.get_wireless_channel('ath0'),
            sub=self.sub.cli.get_wireless_channel('ath0'))
        if advertising_channel is not None:
            self.verifyEqualOrFailCase(set([freq]),
                set(advertising_channel),
                'Verify expected Playbar advertising channel')
            self.verifyEqualOrFailCase(set([freq]),
                 set(sat_sub_channels.values()),
                 'Verify Satellite & Sub channels {} equal Playbar '
                 'channel {}'.format(sat_sub_channels, advertising_channel))

    def _set_wifi_channels(self, region_ch_freqs):
        """
        Private method that sets desired channel, waits for CAC, detect if radar is present and verifies
        channel change

        Arguments:
        region_ch_freqs -- str, list of the available channel frequencies for partiular region

        Keyword Arguments:
        None

        Returns:
        None
        """
        for region_freq in region_ch_freqs:
            self.logger.info("Setting wifi channel to: {}".format(region_freq))
            # change from unicode to str as change_wireless_channel requires str
            if self.s9.cli.change_wireless_channel('ath1', region_freq):
                # allow 2 seconds to ensure all sats change channel as well
                time.sleep(2)
                self._wait_for_cac(region_freq)
                self._verify_channel_change(region_freq)
            else:
                self.logger.warn("{} is on NOL".format(region_freq))

    def _change_satellite_region(self, region_ch_freqs):
        """
        Private method that sets the region of the satellite

        Arguments:
        :param region_ch_freqs: list of the available channel frequencies for
            partiular region
        :type region_ch_freqs: :obj:`str`
        """
        for region in MdpRegion:
            for sat in self.satellites:
                self.logger.info("Setting satellite mdp region to {} on "
                    "{}".format(region, sat))
                if sat.cli.mdp.set_region(region):
                    # reboot needed after changing mdp region
                    self.logger.info("Rebooting {}".format(sat))
                    self._reboot(sat)
                else:
                    self.logger.error("Failed to change region code to {} "
                        "on {}".format(region, self.satellites))
            self._set_wifi_channels(region_ch_freqs)
            self.logger.info("Completed region: {} on satellites".format(
                region))
            self.logger.info("Rebooting PLAYBAR to reset CAC and NOL for DFS "
                "channels")
            self._reboot(self.s9)

    def _change_home_region(self, region):
        """
        Private method that sets the home region of the Sonos device

        Arguments:
        region --str, region code.  1 - USA, 2 - Europe, 3 - China, 4 - Japan

        Keyword Arguments:
        None

        Returns:
        None
        """
        self.logger.info("Setting home mdp region to {} on {}".format(region,
            self.s9))
        if self.s9.cli.mdp.set_region(region):
            # reboot needed after changing mdp region
            self.logger.info("Rebooting {}".format(self.s9))
            self._reboot(self.s9)
        else:
            self.logger.error("Failed to change region code to {} on {}".format(
                region, self.s9))
