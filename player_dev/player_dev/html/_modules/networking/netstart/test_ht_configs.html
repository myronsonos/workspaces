    def _cleanup_devices(self):
        """
        Cleans up the HT configuration of the devices in the list passed or
        cleans up all devices in my_devices. Not using .base version for time
        savings on factory reset and reboot.
        """
        [zp.DeviceProperties.cleanup_bonded_configuration()
         for zp in self.my_devices if zp.modelNumber in PLAYBACK_DEVICES
         or zp.modelNumber == 'Sub']


[docs]    def verify_sats_in_hh(self, ht_config):
        """
        Verifies that the satellites in the given ht config have the same HHID
        as the given home theater master or fails the test.

        :param ht_config: The HT config used for the test
        :type ht_config:
            :optuple:class`~sonos.client.internal.SonosZoneComponent`
        """
        master_hhid = ht_config.master.hhid
        sats = [zp for zp in ht_config if zp not in (ht_config.master, None)]
        [self.verifyEqualOrFailCase(master_hhid, sat.hhid,
            "Sat HHID should equal to HT master HHID: {}".
            format(sat, sat.hhid, master_hhid)) for sat in sats]



[docs]    def verify_ht_sat_def(self, ht_config):
        """
        Verifies that the expected HTSatelliteDefn matches the actual on both
        the HT master as well as its expected satellites.

        :param ht_config: The HT config used for the test
        :type ht_config:
            :optuple:class`~sonos.client.internal.SonosZoneComponent`
        """
        mstr_map_uuids = re.findall("RINCON\w+", ht_config.master.
            get_ht_channelmap())
        mstr_map = ht_config.master.get_ht_channelmap()
        mstr_sat_def = mstr_map.split(';')
        ordered_ht_uuids = [zp.raw_udn for zp in ht_config if zp]
        self.verifyTrueOrFailCase(mstr_map_uuids == ordered_ht_uuids,
            "HT master ht sat defn uuids should match uuids of zp in ht_config"
            .format(mstr_map_uuids, ordered_ht_uuids))
        sats = [zp for zp in ht_config if zp not in (ht_config.master, None)]
        for sat in sats:
            try:
                wait_until_true(lambda: all(x in mstr_sat_def for x in sat.
                    get_ht_channelmap().split(';')), .25, 10)
            except TimeoutError:
                self.stop("Timed out waiting for sat defn {} to be in "
                    " ht's {}".format(sat.get_ht_channelmap(), mstr_map))

[docs]    def verify_sats_on_ht_channel(self, ht_config):
        """
        Verifies that the satellites of the given ht config have the same
        operating channel as the HT master's Home Theater (HT) channel within
        10 seconds or fails the test.

        .. note::

            This method was used instead of a simple assert due to DFS on
            5GHz bands.  10 seconds was chosen because it is the MAX change
            time defined by DFS requirements. For CAC channel clear may take
            up to 60 seconds...

        :param ht_config: The HT config used for the test
        :type ht_config:
            :optuple:class`~sonos.client.internal.SonosZoneComponent`
        """
        mstr = ht_config.master
        sats = [sat for sat in ht_config if sat not in (ht_config.master, None)]
        self.logger.info("Verifying satellites wireless interface are "
                "operating on 5Ghz Home Theater channel")
        for sat in sats:
            try:
                wait_until_true(lambda: int(mstr.diag.get_fullstatus().
                    home_theater_channel) > 5000)
            except TimeoutError:
                ht_chan = mstr.diag.get_fullstatus().home_theater_channel
                self.fail('Expected ht master to have 5GHz channel, channel '
                    'was {}'.format(ht_chan))
            try:
                wait_until_true(lambda: mstr.diag.get_fullstatus()
                    .home_theater_channel == sat.diag.get_fullstatus().
                    operating_channel, timeout_seconds=60, iteration_delay=0.5)
            except TimeoutError:
                ht_chan = mstr.diag.get_fullstatus().home_theater_channel
                sat_chan = sat.diag.get_fullstatus().operating_channel
                self.fail("Expected channel {} but got {} for satellite".
                    format(ht_chan, sat_chan))
            else:
                ht_chan = mstr.diag.get_fullstatus().home_theater_channel
                sat_chan = sat.diag.get_fullstatus().operating_channel
                self.logger.info("Satellite Operating Channel {} == PLAYBAR "
                    "HT Channel {}".format(sat_chan, ht_chan))

[docs]    def verify_ht_playback(self, bonded_zones):
        """
        Verifies the playback ability of the given zp and its bonded members.

        :param bonded_zones: the zone players to verify bonded playback
        :type bonded_zones:
            :optuple:class`~sonos.client.internal.SonosZoneComponent`
        """
        bonded_zones.master.AVTransport.use_buzzer()
        try:
            bonded_zones.master.AVTransport.Play_and_wait()
        except TimeoutError as e:
            self.fail('Timed out waiting for {} to play with exception {}'.
                format(bonded_zones.master, e))
        self.verifyTrueOrFailCase(bonded_zones.master.AVTransport.is_playing(),
            'HT should be playing buzzer')
        bonded_zones.master.AVTransport.Stop()



[docs]    def config_ht(self, ht_config):
        """
        Configures the current ht config using the devices within.

        .. note::

            RINCON_000E58BA0DCB01400:LF,RF;RINCON_000E58984D9201400:SW

        :param ht_config:
            :optuple:class`~sonos.client.internal.SonosZoneComponent`
        :raises UPnPError: error calling AddHTSatellite
        :raises TimeoutError: timed out waiting for device to bond or reconnect
        """
        if ht_config.sw:
            ht_config.master.DeviceProperties.add_ht_satellite(ht_config.sw,
                HTPositions.SW)
        if ht_config.lr:
            ht_config.master.DeviceProperties.add_ht_satellite(ht_config.lr,
                HTPositions.LR)
        if ht_config.rr:
            ht_config.master.DeviceProperties.add_ht_satellite(ht_config.rr,
                HTPositions.RR)

    def test_ht_config(self, test_ht_config):
        """
        Configures the HH HT to match the input config and verifies the config
        was successful.

        :param test_ht_config: the configuration to test
        :type test_ht_config:
            :optuple:class`~sonos.client.internal.SonosZoneComponent`
        """
        # We only have one test case for now so store it here, going forward we
        # will have many either from a generator or in another json
        ht_transactions = ['BECOME_SATELLITE_ALSO_WIRE_NO_NFW']
        for ht_transaction in ht_transactions:
            for zp in test_ht_config:
                if zp not in (test_ht_config.master, None):
                    zp.log_inspector = NetstartdLogInspector(zp)
                    zp.log_inspector.mark_last_log_entry()
            self.config_ht(test_ht_config)
            self.verify_sats_in_hh(test_ht_config)
            self.verify_ht_sat_def(test_ht_config)
            self.verify_sats_primary_uuid(test_ht_config)
            self.verify_sats_on_ht_channel(test_ht_config)
            self.verify_ht_netstart_states(test_ht_config, ht_transaction)
            # TODO: method by which zones store proxy ip/MAC data is changing as
            # per https://jira.sonos.com/browse/MCS-1376 we want this because
            # when the time comes that we test ht config in station mode we
            # will have distinct verification points in the proxy's table
            # self.verify_ht_arp_tables(test_ht_config)
            self.verify_ht_playback(test_ht_config)
