    """
    Verifies Netstartd and NetManager state machines through inspection of
    the /opt/log/netstartd.log file.  This is done through setting up and
    configuring wired and wireless households and adding wired and/or wireless
    players to them.

    def _setup_nfw_creds_from_test_data(self):
        """
        Configures nfw_ssid and nfw_pwd instance variables from the execution
        configuration used for the test fixture.  If there is a "nfw_creds"
        section then this method will attempt to set values from "nfwssid" and
        "nfwpwd" section attributes.  If the expected attribute does not exist
        or there is no section in the execution configuration file then the
        variables will be set to None.
        """
        self.nfw_ssid = None
        self.nfw_pwd = None
        if hasattr(self.test_data, 'nfw_creds'):
            self.nfw_ssid = self.test_data.nfw_creds.as_dict().get('nfwssid')
            self.nfw_pwd = self.test_data.nfw_creds.as_dict().get('nfwpwd')

        """

        :rtype: :obj:`tuple`
        """
        return tuple([dev for dev in self.test_zones

        :rtype: :obj:`tuple`
        """

            if dev.is_n_capable() and dev.is_playback_device()])


[docs]    def verify_netstartd_log_states(self, zone, state_transition):
        """
        Checks /opt/log/netstartd.log on zone for one or more log sequenced
        log entries as defined by the state_transition entry in one of the
        test json files that contains expected results

        :param zone: the zone player whose log to check
        :type zone: :class:`~sonos.client.internal.SonosZoneComponent`
        :param state_transition: the identifier used to locate expected results
        :type state_transition: :class:`~.StateTransition`
        """
        self.verify_netstart_states(self.log_inspectors[zone].netstart,
            state_transition.value)
        self.verify_netmanager_states_for_transaction_type(
            self.log_inspectors[zone].netmanager,
            state_transition.value)


    def _create_and_verify_new_sonos_hh(self):
        """
        Creates a new SonosNet HH from the first test zone
        """
        create_zone = self.test_zones.first_zone
        hh_created = self.setup_first_zone(create_zone)
        self.verify_netstartd_log_states(create_zone,
            StateTransition.SONOSNET_SETUP)
        query_result = self.netstart_test.query(create_zone)
        self.verifyEqualOrStop(hh_created, query_result.hhid,
            "Verify HouseholdID from netstart QUERY equals HHID created "
            "via SONOSNET_SETUP netstart_test transaction")

    def _add_device_to_sonos_hh_and_verify(self):
        """
        Adds the second test zone to the SonosNet HH the first zone is in
        """
        hh_zone = self.test_zones.first_zone
        add_zone = self.test_zones.second_zone
        self.netstart_test.add_component(hh_zone, add_zone)
        self.verify_netstartd_log_states(add_zone,
            StateTransition.SN_ADD_WIRED_ZONE)
        query_result = self.netstart_test.query(add_zone)
        self.verifyEqualOrStop(hh_zone.hhid, query_result.hhid,
            "Verify <{}> is added to household".format(add_zone))


[docs]    def setUpFixture(self):
        """
        Parses no first wire credentials and transforms string values to
        enums in the execution config
        """
        super(NetstartStateMachineTestFixture, self).setUpFixture()
        self.test_zones = [zone for zone in
            self.generators.generate_testbed_unique_devices()]
        self._setup_nfw_creds_from_test_data()
        self.verifyGreaterEqualOrStop(
            len(self._get_g_only_playback_devices()),
            1,
            'Verify one or more G only non-bridge devices in testbed')
        self.verifyGreaterEqualOrStop(
            len(self._get_n_capable_playback_devices()),
            1,
            'Verify one or more G only non-bridge devices in testbed')
        g_device = self._get_g_only_playback_devices()[0]
        n_device = self._get_n_capable_playback_devices()[0]
        self.test_zones = TestZones(g_device, n_device)
        [zone.factory_reset() for zone in self.test_zones]
        self.log_inspectors = {
            dev: NetstartLogInspectors(
                NetstartdLogInspector(dev),
                NetstartdLogInspector(dev))
            for dev in self.test_zones}
        self._create_and_verify_new_sonos_hh()
        self._add_device_to_sonos_hh_and_verify()

        """
        Resets the testbed to a single HH registered to automation user
        """
        devices_to_reset = [zc for zc in self.my_devices if zc.hhid]
        [zc.factory_reset() for zc in devices_to_reset]
        associated_zone = self._get_first_non_dock()
        devices_to_reset.remove(associated_zone)
        self.netstart_test.sonosnet_setup(associated_zone)
        for new_zone in devices_to_reset:
            self.netstart_test.add_component(associated_zone, new_zone)
        associated_zone.SystemProperties.register_device_to_test_user()



[docs]    def test_sonosnet_both_to_sonosnet_path(self):
        """
        Makes sure a Sonos device can successfully transition from a
        SonosNet with Both state to a SonosNet with Path state
        """
        inspector = self.log_inspectors[self.test_zones.second_zone]
        if not inspector.netmanager.is_current_marker(
                state=NetManagerState.SONOSNET_BOTH):
            [zone.wire.enable() for zone in self.test_zones]
            inspector.netmanager.wait_for_next(
                state=NetManagerState.SONOSNET_BOTH)
        self.test_zones.second_zone.wire.disable()
        self.verify_netstartd_log_states(self.test_zones.second_zone,
            StateTransition.SN_B_TO_SN_P)
        self.test_zones.second_zone.wire.enable()
        [zone.wait_for_sonosnet_mode() for zone in self.test_zones]

[docs]    def setup_first_zone(self, first_zone):
        """
        Configures a new sonosnet HH from first_zone and then adds No First
        Wire credentials to the HH

        :param first_zone: the zone component to create a HH with
        :type first_zone: :class:`~sonos.client.internal.SonosZoneComponent`
        :returns: the new HH id
        :rtype: :obj:`str`
        """
        new_hhid = self.netstart_test.sonosnet_setup(first_zone)
        # make sure we are in SonosNet before adding wireless credentials
        log_inspector = NetstartdLogInspector(first_zone)
        self.verifyTrueOrStop(
            log_inspector.wait_for_next(ur"Set household {}.*".format(
                new_hhid)),
            "FirstZP {} never made it to a configured SonosNet HH".format(
                first_zone))
        self.verifyTrueOrStop(
            log_inspector.wait_for_next(state=NetstartState.IDLE),
            "Failed to find {} on FirstZP {} after being configured for a new "
            "HH".format(NetstartState.IDLE, first_zone))
        self.netstart_test.nfw_update(first_zone, self.nfw_ssid, self.nfw_pwd)
        return new_hhid

class _StateMachineFixtureSelectorAction(argparse.Action):
    """
    Custom argparse action used to convert string class name values to their
    respective fixture list, so we can run a specific fixture or all of them
    """

    def __call__(self, parser, namespace, values, option_string=None):
        """
        Callable action implementation required by :class:`argparse.Action`
        objects
        """
        fixtures = [SimpleUpgradeTestFixture]
        for value in values:
            fixtures.append(getattr(sys.modules[self.__module__], value))
        setattr(namespace, self.dest, fixtures)


TEST_FIXTURES = (SonosNetStateMachine, SonosNetNfwStateMachine)


def _parse_arguments():
    """
    Parse command line argument when this module is runnable.  This add the
    -f/--fixture option that allows users to run any single test fixture
    instead of running them all.
    """
    parser = argparse.ArgumentParser("NetstartStateMachineTestSuite")
    parser.add_argument("-f", "--fixtures", dest="fixtures",
        choices=[klazz.__name__ for klazz in TEST_FIXTURES],
        default=(SimpleUpgradeTestFixture,) + TEST_FIXTURES, nargs='+',
        help="specify one of the fixtures to run or run all",
        action=_StateMachineFixtureSelectorAction)
    known_args, unknown_args = parser.parse_known_args()
    sys.argv[1:] = unknown_args
    return known_args


if __name__ == '__main__':
    args = _parse_arguments()
    suite = WorkflowTestSuite('NetstartStateMachineTestSuite')
    suite.run([fixture() for fixture in args.fixtures])
