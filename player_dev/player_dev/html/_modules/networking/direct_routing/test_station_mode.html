[docs]    def setUpFixture(self):
        """
        Remove other module types that share the same architecture.
        Only need to test Fenway(S3), LL(S9), SH4(ZP80), PPC(S5),
        encore for GC-GM pairs.

        AP hw mode configured for DR disabled case in
        test_station_mode_tx_rate_to_peers

        Selects the wireless protocol to use, possible values are 11b, 11bg, 11g,
        11gdt (G + dynamic turbo, madwifi only), 11gst (G turbo, broadcom only), 11a,
        11adt (A + dynamic turbo, madwifi only), 11ast (A + static turbo, madwifi only),
        11fh (frequency hopping), 11lrs (LRS mode, broadcom only), 11ng (11N+11G, 2.4GHz,
        mac80211 only), 11na (11N+11A, 5GHz, mac80211 only) or auto.
        Note: AP default =  driver default
        """
        super(DirectRoutingSTA, self).setUpFixture()
        self._config_ap()
        self.ap_dev = '0'
        self.ap.set_hwmode(self.ap_dev, '11ng')
        self.ap.commit_wifi_config_and_restart_radios()
        ap_hwmode = self.ap.get_hwmode(self.ap_dev)
        self.ap_mac = self.ap.mac_address_2ghz
        self.verifyEqualOrStop(
           ap_hwmode, '11ng',
           'AP should be in hwmode 11ng, actual {}'.format(ap_hwmode))
        self.logger.info('removing_wireless_credentials on self.my_devices')
        self.remove_wireless_credentials(*self.my_devices)
        self.logger.info("Configuring test ZPs for station mode.")
        if self.gc.is_bg_only():
            self.tx_rate_peers = self.gms
        else:
            n_zcs_from_gms = [zc for zc in self.gms if zc.is_n_capable()]
            bg_zcs_from_gms = [zc for zc in self.gms if zc.is_bg_only()]
            self.tx_rate_peers = []
 
            if len(n_zcs_from_gms) > 0:
                self.tx_rate_peers.append(n_zcs_from_gms[0])
            else:
                self.tx_rate_peers.append([
                    zc for zc in self.subnet.get_playback_devices() if
                    zc.is_n_capable() and zc != self.gc][0])
            if len(bg_zcs_from_gms) > 0:
                self.tx_rate_peers.append(bg_zcs_from_gms[0])
            else:
                self.tx_rate_peers.append([
                    zc for zc in self.subnet.get_playback_devices() if
                    zc.is_bg_only() and zc != self.gc][0])
            shuffle(self.tx_rate_peers)
        self.add_wireless_credentials(*list(set(self.zps) | set(self.tx_rate_peers)),
            ssid=self.get_ap_ssid(),
            password=AP_PWD)
        self.transition_zones_to_station_mode(*self.zps)

    def _get_ath0_dr_hex(self, mac_hex):
        """
        Helper function to return the station mode
        direct routing mac address.
        For given serial 00:0E:58:00:00:00, return
        02:0E:58:00:00:00 if the last byte of serial
        is odd, otherwise 00:0E:58:00:00:01

        :param mac_hex: mac address as hex string
        :type mac_hex: :obj:`str`
        :return: ath0 mac address
        :rtype: :obj:`str`
        """
        if mac_hex & 0xFF % 2 == 0:
            return super(DirectRoutingSTA, self)._get_ath0_dr_hex(mac_hex)
        else:
            return mac_hex + 0x020000000000

    def _get_zp_serial_mac(self, zp):
        """
        Helper function to return the serial number
        mac address for a given zp of the form:
        00:0E:58:00:00:00

        :param zp: zone component
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`
        :return: serial number mac address
        :rtype: :obj:`str`
        """
        zp_sn_mac = ':'.join(a + b for a, b in
                             zip(zp.serialNumber[::2],
                                 zp.serialNumber[1::2])).upper()
        return zp_sn_mac

    def _count_data_sent(self, capture, source_mac, dest_mac,
                         qos, bssid, min_packet_size=1000):
        """
        Helper function to count the number of qos data packets
        found in the capture that matches the source & destination mac
        and bssid supplied and meets the min_packet_size.

        :param capture: pyshark FileCapture class with
            only_summaries == False (default)
        :type capture: :class: `pyshark.capture.file_capture.FileCapture`
        :param source_mac: source mac address
        :type source_mac: :obj:`str`
        :param dest_mac: destination mac address
        :type dest_mac: :obj:`str`
        :param qos: flag to indicate to check qos data vs. regular data
        :type qos: :obj:`bool`
        :param bssid: wlan.bssid in packet
        :type bssid: :obj:`str`
        :param min_packet_size: minimum packet size for packet to be
            counted. For group audio, packet size is about 1168.
        :type min_packet_size: :obj:`int` or 1000
        :return: number of packets found sent to dest_mac and is
            at least min_packet_size in byte length
        :rtype: :obj:`int`
        """
        count = 0
        for packet in capture:
            if (packet.wlan.da.lower() == dest_mac.lower() and
                    packet.wlan.sa.lower() == source_mac.lower() and
                    packet.wlan.bssid.lower() == bssid.lower() and
                    int(packet.length) > min_packet_size):
                # data packets routed through AP do not have qos
                if not qos:
                    count += 1
                else:
                    if (int(packet.wlan.fc_type_subtype) == int(0x28) and
                            int(packet.wlan.qos_priority) == 1):
                        count += 1
        self.logger.info('Qos set to {} in counting packets'.format(qos))
        self.logger.info('Found {} data packets with size greater than '
                         '{} bytes sent to {} from {} '
                         'via bssid {}'.format(count,
                                               min_packet_size,
                                               dest_mac,
                                               source_mac,
                                               bssid))
        return count

    def _get_dr_enable_lists(self, gms):
        """
        Helper function to return the send_list, no_send_list,
        and via_ap_no_send_list when direct routing is enabled.

        :param gms: zp group members
        :type gms: :obj: `list` of
            :class:`~sonos.client.internal.SonosZoneComponent`
        :return: dictionary of lists
        :rtype: :obj:`dict`
        """
        send_list = []
        no_send_list = []
        via_ap_no_send_list = []
        for gm in gms:
            send_list.append(self._get_ath0_dr_mac(gm))
            no_send_list.append(self._get_zp_serial_mac(gm))
            via_ap_no_send_list.append(self._get_ath0_dr_mac(gm))
            via_ap_no_send_list.append(self._get_zp_serial_mac(gm))
        return {'send_list': send_list,
                'no_send_list': no_send_list,
                'via_ap_no_send_list': via_ap_no_send_list}

        """

        no_send_list = []
        via_ap_no_send_list = []
        via_ap_send_list = []
        for gm in gms:
            no_send_list.append(self._get_ath0_dr_mac(gm))
            no_send_list.append(self._get_zp_serial_mac(gm))
            via_ap_no_send_list.append(self._get_ath0_dr_mac(gm))
            via_ap_send_list.append(self._get_zp_serial_mac(gm))
        return {'send_list': [],
                'no_send_list': no_send_list,
                'via_ap_no_send_list': via_ap_no_send_list,
                'via_ap_send_list': via_ap_send_list}

    def _run_sniffer_analysis(self, name, gc_mac, send_list, no_send_list,
            sniff_start_delay, via_ap_send_list, via_ap_no_send_list, gc):
        """
        Helper function to run analysis of generic multicast or
        unicast group playback test with wireless sniffer verification points.
        Start sniffer capture, filtering on packets sent only by the GC
        via the AP or via direct route
        Then verify that GC is sending audio packets wirelessly to the
        send_list of destination mac ZPs, and not sending audio packets
        wirelessly to the no_send_list of destination mac ZPs via direct
        route.
        Then verify that GC is sending audio packets wirelessly to the
        send_list of destination mac ZPs, and not sending audio packets
        wirelessly to the no_send_list of destination mac ZPs via AP.

        :param name: name of test. also used for name of capture file.
        :type name: :obj:`str`
        :param gc_mac: group coordinator mac address
        :type gc_mac: :obj:`str`
        :param send_list: list of mac addresses sent by GC
        :type send_list: :obj:`list`
        :param no_send_list: list of mac addresses not sent by GC
        :type no_send_list: :obj:`list`
        :param sniff_start_delay: additional delay in seconds to start
            sniffing after playback has started
        :type sniff_start_delay: :obj:`int`
        :param via_ap_send_list: list of mac addresses sent by GC through
            the AP
        :type via_ap_send_list: :obj:`list`
        :param via_ap_no_send_list: list of mac addresses not sent by GC
            through the AP
        :type via_ap_no_send_list: :obj:`list`
        :param gc: dut group coordinator
        :type gc: :class:`~sonos.client.internal.SonosZoneComponent`
        """
        """
        set capture filter to grab data packets sent by GC serial mac
        (sent via AP) or by GC direct routing ath0 mac
        """
        gc_sn_mac = self._get_zp_serial_mac(gc)
        cap_filter = ('type data and '
                      '(wlan addr2 {} '
                      'or wlan addr2 {})').format(gc_mac, gc_sn_mac)
        cap_file = self._start_sniffer(name, sniff_start_delay, cap_filter)
        capture = pyshark.FileCapture(cap_file)
        self.logger.info('Filtered capture file {} with filter set to '
                         '{}'.format(cap_file, cap_filter))
        # verify that station mode GC direct routing mac is not sending audio
        # to the zp macs listed in the no_send_list
        for zp_mac in no_send_list:
            self.verifyTrueOrFailCase(
                (self._count_data_sent(capture, gc_mac, zp_mac, True, zp_mac) <
                    self.NON_PLAYBACK_PCK_THRESHOLD),
                'Verify GC {} is NOT sending wirelessly to ZP {}'.format(
                    gc_mac, zp_mac))
        # verify that station mode GC direct routing mac is sending audio
        # to the zp macs listed in the no_send_list
        for zp_mac in send_list:
            self.verifyTrueOrFailCase(self._count_data_sent(capture,
                gc_mac, zp_mac, True, zp_mac) > self.PLAYBACK_PCKT_THRESHOLD,
                'Verify GC {} is sending wirelessly to ZP {}'.format(
                    gc_mac, zp_mac))
        # verify that station mode GC is not sending audio
        # via the AP to the zp macs listed in the via_ap_no_send_list
        for zp_mac in via_ap_no_send_list:
            self.verifyTrueOrFailCase(
                self._count_data_sent(capture, gc_sn_mac, zp_mac, False,
                    self.ap_mac) < self.NON_PLAYBACK_PCK_THRESHOLD,
                ('Verify GC {} is NOT sending wirelessly to ZP {} via AP '
                    '{}'.format(gc_sn_mac, zp_mac, self.ap_mac)))
        # verify that station mode GC is sending audio
        # via the AP to the zp macs listed in the via_ap_no_send_list
        for zp_mac in via_ap_send_list:
            self.verifyTrueOrFailCase(
                self._count_data_sent(capture, gc_sn_mac, zp_mac, False,
                    self.ap_mac) > self.PLAYBACK_PCKT_THRESHOLD,
                'Verify GC {} is sending wirelessly to ZP {} via AP {}'.format(
                    gc_sn_mac, zp_mac, self.ap_mac))
        capture.close()

    def _run_group_playback(self, gms, test_name, wire_list, send_list,
            no_send_list, dynamic_dr_list=[], sniff_start_delay=20,
            via_ap_send_list=[], via_ap_no_send_list=[]):
        """
        Run basic group playback test.

        :param gms: zp group memebers
        :type gms: :obj:`list` of
            :class:`~sonos.client.internal.SonosZoneComponent`
        :param test_name: name of test. also used for name of capture file.
        :type test_name: :obj:`str`
        :param zp_wire_list: list of ZPs whose ethernet wire will be
            enabled. All other ZPs in the testbeds/HH will have disabled
            ethernet.
        :type zp_wire_list: :obj:`list` of
            :class:`~sonos.client.internal.SonosZoneComponent`
        :param send_list: list of mac addresses sent by GC
        :type send_list: :obj:`list`
        :param no_send_list: list of mac addresses not sent by GC
        :type no_send_list: :obj:`list`
        :param dynamic_dr_list: list of :data:`DRInfo`
            :class:`collections.namedtuple`. this is used to specify changing
            the direct routing state of the group after playback has started.
        :type dynamic_dr_list: :obj:`list` or []
        :param sniff_start_delay: additional delay in seconds to start
            sniffing after playback has started
        :type sniff_start_delay: :obj:`int` or 0
        """
        zp_test_group = [self.gc] + gms
        self._wait_for_devices_ready(zp_test_group)
        self.begin_watching_for_sync_error(zp_test_group)
        self._group_and_start_playback(self.gc, gms)
        for dr_set in dynamic_dr_list:
            self._verify_dr(dr_set.settestpoint_cmds,
                            self.gc,
                            dr_set.peer,
                            dr_set.is_dr_enabled,
                            timeout=180)
        gc_mac = self._get_ath0_dr_mac(self.gc)
        self._run_sniffer_analysis(test_name, gc_mac,
                                   send_list, no_send_list,
                                   sniff_start_delay,
                                   via_ap_send_list=via_ap_send_list,
                                   via_ap_no_send_list=via_ap_no_send_list,
                                   gc=self.gc)
        self.stop_watching_for_sync_error(zp_test_group)

    def test_station_mode_tx_rate_to_peers(self):
        """
        Verify an 11n devices uses the correct tx rates when sending data to
        station mode peers with direct routing enabled or disabled. Also
        verifies that the correct mcs index is selcted for 11n to 11n gms given
        the corresponding number of chains (2x2, 3x3)

        Cases which could be covered but are left out for simplicity:
        - 11n GC uses the highest sopported rate (26 mbps for 11n peers and 24
            for 11g) of a peer if DR is enabled
        - 11n GC uses mcs 3 for 3x3 11n GMs and mcs 9 for 2x2 11n GMs
        - 11g GC uses 24 mbps for all peers if DR is enabled
        - if DR is not enabled all GCs use the AP rate
        """
        sniff_time = 5
        gc_dr_mac = self._get_ath0_dr_mac(self.gc)
        gc_no_dr_mac = self.gc.mac_address
        gc_type = '11bg' if self.gc.is_bg_only() else '11n'
        dr_capture_name = 'station_mode_tx_rate_to_peers'
        no_dr_capture_name = 'station_mode_tx_rate_to_ap'
        cap_filter = ('type data and len > 400 and wlan addr2 {} or '
                      'wlan addr2 {}'.format(gc_no_dr_mac, gc_dr_mac))
        base_display_filter = ('radiotap.datarate == {} and '
                               'frame.len >= 1000 and '
                               'wlan.da == {}')
        # format alias to stay under 80 chars below
        cds = self._count_data_sent
        rate_11bg = 24
        rate_11n = 26
        mcs3 = 3
        mcs9 = 9
        mcs_index_to_2x2_11n_peer = mcs3
        mcs_index_to_3x3_11n_peer = mcs9
        rate_ap_high_rssi = rate_11bg
        rate_ap_low_rssi = 12
        # one 11n and one 11g device covers both cases intended to be tested
        # if also checking DR0 and DR1 for a single gm, in the even the gc is
        # bg only it limits the rate and we have few bg only devices in the
        # testbed so pair it with 11n gms
        gm_pair = self.tx_rate_peers
        self.transition_zones_to_station_mode(*list(set(self.zps) | set(gm_pair)))
        self.logger.info('Wireless GMs with RSSI '
                         'greater than 25: {}'.format(gm_pair))
        cmd_list = ['ath0 fixedrssi {}'.format(DIRECT_ROUTING_ENABLED_RSSI)]
        self._verify_dr(cmd_list, self.gc, gm_pair[0], True, timeout=180)
        self._verify_dr(cmd_list, self.gc, gm_pair[1], True, timeout=180)
        self._group_and_start_playback(self.gc, gm_pair)
        both_gm_dr_enabled_capture = self._start_sniffer(
            dr_capture_name, 0, cap_filter, sniff_time)
        info_str = ('with direct routing {}, should have captured at least {} '
                    'frames at tx rate {} mbps, from GC {} {} to GM {} {}')
        for gm in gm_pair:
            gm_mac = self._get_ath0_dr_mac(gm)
            gm_type = '11bg' if gm.is_bg_only() else '11n'
            if gc_type == '11bg' or gm.is_bg_only():
                rate = rate_11bg
            else:
                rate = rate_11n
                chaininfo = self.athconfigs[gm].getrfchains(INTERFACE_2GHZ)
                if chaininfo["tx"]["count"] == 2:
                    mcs_index = mcs_index_to_2x2_11n_peer
                else:
                    mcs_index = mcs_index_to_3x3_11n_peer
            da_of_audio = gm_mac
            display_filter = base_display_filter.format(rate, da_of_audio)
            if rate == rate_11n:
                display_filter += ' and radiotap.mcs.index == {}'.format(
                                                                    mcs_index)
                verification_str = info_str + ' at mcs index {}'.format(
                                                                    mcs_index)
            else:
                verification_str = info_str
            capture = pyshark.FileCapture(input_file=both_gm_dr_enabled_capture,
                                          display_filter=display_filter)
            num_expected_rate_frames_captured = cds(capture,
                                                    gc_dr_mac,
                                                    gm_mac,
                                                    0,
                                                    gm_mac)
            min_expected_num_frames = (
                GROUPED_PLAYBACK_FRAME_RATE * sniff_time * 0.7)
            self.verifyGreaterOrFailCase(
                num_expected_rate_frames_captured,
                min_expected_num_frames,
                verification_str.format('enabled', min_expected_num_frames,
                                        rate, gc_type, self.gc, gm_type, gm))
        self.gc.AVTransport.Stop_and_wait()
        cmd_list.append('ath0 {} 8'.format(self._get_ath0_dr_mac(gm_pair[1])))
        self.logger.info(
            'Wireless GM with RSSI less than 19: {}'.format(gm_pair[1]))
        self._verify_dr(cmd_list, self.gc, gm_pair[0], True, timeout=180)
        self._verify_dr(cmd_list, self.gc, gm_pair[1], False, timeout=180)
        self._group_and_start_playback(self.gc, gm_pair)
        one_gc_dr_enabled_capture = self._start_sniffer(
            no_dr_capture_name, 0, cap_filter.format(gc_dr_mac, gc_no_dr_mac),
            sniff_time)
        for gm in gm_pair:
            gm_mac = self._get_ath0_dr_mac(gm)
            gm_type = '11bg' if gm.is_bg_only() else '11n'
            if gc_type == '11bg' or gm.is_bg_only():
                rate = rate_11bg
            else:
                rate = rate_11n
                chaininfo = self.athconfigs[gm].getrfchains(INTERFACE_2GHZ)
                if chaininfo["tx"]["count"] == 2:
                    mcs_index = mcs_index_to_2x2_11n_peer
                else:
                    mcs_index = mcs_index_to_3x3_11n_peer
            if gm == gm_pair[1]:
                dr_enabled = 'disabled'
                ta_of_audio = gc_no_dr_mac
                da_of_audio = gm.mac_address
                bssid = self.ap_mac
                rate = rate_11bg
                verification_str = info_str
            else:
                dr_enabled = 'enabled'
                ta_of_audio = gc_dr_mac
                da_of_audio = gm_mac
                bssid = gm_mac
            display_filter = base_display_filter.format(rate, da_of_audio)
            if rate == rate_11n:
                display_filter += ' and radiotap.mcs.index == {}'.format(
                                                                    mcs_index)
                verification_str = info_str + ' at mcs index {}'.format(
                                                                    mcs_index)
            capture = pyshark.FileCapture(one_gc_dr_enabled_capture,
                                          display_filter=display_filter)
            num_expected_rate_frames_captured = cds(capture,
                                                    ta_of_audio,
                                                    da_of_audio,
                                                    0,
                                                    bssid)
            min_expected_num_frames = (
                GROUPED_PLAYBACK_FRAME_RATE * sniff_time * 0.7)
            self.verifyGreaterOrFailCase(
                num_expected_rate_frames_captured,
                min_expected_num_frames,
                verification_str.format(dr_enabled, min_expected_num_frames,
                                        rate, gc_type, self.gc, gm_type, gm))
        ap_dump = self.ap.get_connected_stations(self.ap_dev)
        gc_station_info = [station_info for station_info in ap_dump if
                           self.gc.mac_address.lower() in station_info][0]
        ap_gc_rx_rate = int(float(gc_station_info.rx_bitrate.split(' ')[0]))
        gc_reported_tx_rate = self.gc.cli.get_station()._sta_tx_rate
        self.verifyEqualOrFailCase(

        self.verifyEqualOrFailCase(

            'when direct routing is disabled gc {} should use tx rate {} mbps, '
            'gc station page reporting tx rate {} mbps'.format(
                            self.gc, rate_ap_high_rssi, gc_reported_tx_rate))
        self.gc.AVTransport.Stop_and_wait()
        self.gc.AVTransport.BecomeCoordinatorOfStandaloneGroup()
        cmd_list = ['ath0 fixedrssi {}'.format(DIRECT_ROUTING_DISABLED_RSSI)]

        wait_until_true(

            display_filter = base_display_filter.format(rate, da_of_audio)
            capture = pyshark.FileCapture(low_ap_rssi_capture,

            num_expected_rate_frames_captured = cds(capture,

            min_expected_num_frames = (
