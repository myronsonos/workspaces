    def _get_dr_enable_lists(self, gms):
        """
        Helper function to return the send_list, no_send_list,
        and via_ap_no_send_list when direct routing is enabled.

        :param self.gms: zp group members
        :type self.gms: :obj: `list` of
            :class:`~sonos.client.internal.SonosZoneComponent`
        :return: dictionary of lists
        :rtype: :obj:`dict`
        """
        send_list = []
        for gm in gms:
            send_list.append(self._get_ath0_dr_mac(gm))
        return {'send_list': send_list,
                'no_send_list': []}

    def _get_dr_disable_lists(self, gms, zp_wire_list):
        """
        Helper function to return the send_list, no_send_list,
        via_ap_no_send_list, via_ap_send_list, and zb_send_list
        when direct routing is disabled.

        :param self.gms: zp group members
        :type self.gms: :obj: `list` of
            :class:`~sonos.client.internal.SonosZoneComponent`
        :param self.gc: group coordinator
        :type self.gc: :class:`~sonos.client.internal.SonosZoneComponent`
        :param zp_wire_list: zp group members
        :type zp_wire_list: :obj: `list` of
            :class:`~sonos.client.internal.SonosZoneComponent`
        :return: dictionary of lists
        :rtype: :obj:`dict`
        """
        no_send_list = []
        zb_send_list = []
        for gm in gms:
            no_send_list.append(self._get_ath0_dr_mac(gm))
            if gm not in zp_wire_list:
                zb_send_list.append(self._get_ath0_dr_mac(gm))
        if self.gc in zp_wire_list and self.bridge in zp_wire_list:
            no_send_list.append(self._get_ath0_dr_mac(self.bridge))
        return {'send_list': [],
                'no_send_list': no_send_list,
                'zb_send_list': zb_send_list}

    def _count_qos_data_sent(self, capture, source_mac, dest_mac,
                             min_packet_size=1000):
        """
        Helper function to count the number of qos data packets
        found in the capture that matches the source & destination mac
        supplied and meets the min_packet_size.

        :param capture: pyshark FileCapture class with only_summaries == True
        :type capture: :class: `pyshark.capture.file_capture.FileCapture`
        :param source_mac: source mac address
        :type source_mac: :obj:`str`
        :param dest_mac: destination mac address
        :type dest_mac: :obj:`str`
        :param min_packet_size: minimum packet size for packet to be
            counted. For group audio, packet size is about 1168.
        :type min_packet_size: :obj:`int` or 1000
        :return: number of packets found sent to dest_mac and is
            at least min_packet_size in byte length
        :rtype: :obj:`int`
        """
        count = 0
        for packet in capture:
            if (packet.destination.lower() == dest_mac.lower() and
                    packet.source.lower() == source_mac.lower() and
                    int(packet.length) > min_packet_size and
                    'QoS Data' in packet.info):
                count += 1
        self.logger.info('Found {} qos data packets with size greater than '
            '{} bytes sent to {} from {}'.format(
                count,
                min_packet_size,
                dest_mac,
                source_mac))
        return count

    def _run_sniffer_analysis(self, name, gc_mac, send_list, no_send_list,
              zb_send_list, sniff_start_delay, **kwargs):
        """
        Helper function to run analysis of generic multicast or
        unicast group playback test with wireless sniffer verification points.
        Start sniffer capture, filtering on packets sent only by the GC,
        or both the GC and wired bridge ZP.
        Then verify that GC is sending audio packets wirelessly to the
        send_list of destination mac ZPs, and not sending audio packets
        wirelessly to the no_send_list of destination mac ZPs.
        Verfiy that the zone bridge is sending audio packets wirelessly to
        the zb_send_list of destination mac ZPs.

        :param name: name of test. also used for name of capture file.
        :type name: :obj:`str`
        :param gc_mac: group coordinator mac address
        :type gc_mac: :obj:`str`
        :param send_list: list of mac addresses sent by GC
        :type send_list: :obj:`list`
        :param no_send_list: list of mac addresses not sent by GC
        :type no_send_list: :obj:`list`
        :param zb_send_list: list of mac addresses sent by wired zone bridge
        :type zb_send_list: :obj:`list`
        :param sniff_start_delay: additional delay in seconds to start
            sniffing after playback has started
        :type sniff_start_delay: :obj:`int`
        """
        # set capture filter to grab data packets sent by GC, or both GC & ZB
        zb_mac = self._get_ath0_dr_mac(self.bridge)
        gc_mac = self._get_ath0_dr_mac(self.gc)
        if not zb_send_list:
            cap_filter = 'type data and wlan addr2 {}'.format(gc_mac)
        else:
            cap_filter = ('type data and '
                          '(wlan addr2 {} '
                          'or wlan addr2 {})').format(gc_mac, zb_mac)
        cap_file = self._start_sniffer(name, sniff_start_delay, cap_filter)
        capture = pyshark.FileCapture(cap_file,
                                      display_filter=self.GROUP_AUDIO_FILTER,
                                      only_summaries=True)
        self.logger.info('Filtered capture file {} with filter set to '
                         '{}'.format(cap_file, cap_filter))
        # verify that GC is not sending audio to the zp macs listed in
        # the no_send_list
        for zp_mac in no_send_list:
            self.verifyGreaterOrStop(self.NON_PLAYBACK_PCK_THRESHOLD,
                    self._count_qos_data_sent(capture, gc_mac,
                        zp_mac),
                    'Verify GC {} is NOT sending wirelessly to ZP {}'.format(
                        gc_mac, zp_mac))
        # verify that GC is sending audio to the zp macs listed in
        # the send_list
        for zp_mac in send_list:
            self.verifyLessOrStop(self.PLAYBACK_PCKT_THRESHOLD,
                    self._count_qos_data_sent(capture, gc_mac,
                            zp_mac),
                    'Verify GC {} is sending wirelessly to ZP {}'.format(
                        gc_mac, zp_mac))
        # verify that wired bridge is sending audio to the zp macs listed in
        # the send_list
        for zp_mac in zb_send_list:
            self.verifyLessOrStop(self.PLAYBACK_PCKT_THRESHOLD,
                    self._count_qos_data_sent(capture, zb_mac,
                        zp_mac),
                    'Verify wired zone bridge {} is sending wirelessly to '
                    'ZP {}'.format(zb_mac, zp_mac))
        capture.close()

    def _config_eth_ports(self, enable_list):
        """
        Configure the managed switch ethernet ports for the testbed.
        If ZP is in the enable_list, then enable the eth switch port
        for that ZP, else disable it.

        :param enable_list: list of ZPs to enable its managed switch eth port
        :type enable_list: :obj:`list` of
            :class:`~sonos.client.internal.SonosZoneComponent`
        """
        self.logger.debug('Configuring testbed managed ethernet switch ports.'
                          '  Enable only: {}'.format(enable_list))
        zones_touched = []
        for zp in self.my_devices:
            if zp not in enable_list:
                needs_wire_toggled = zp.wire.is_enabled
                perform_wire_operation = zp.wire.disable
            else:
                needs_wire_toggled = zp.wire.is_disabled
                perform_wire_operation = zp.wire.enable
            if needs_wire_toggled():
                zones_touched.append(zp)
                perform_wire_operation()

        def _ensure_valid(zone):
            zone.diag.wait_for_status_page(wait_timeout=60)

        run_concurrent_processes(_ensure_valid, zones_touched)

    def _wait_for_all_stp_stable(self):
        for zp_brctl in self.brctls.values():
            zp_brctl.wait_for_stp_stable()

    def _force_zc_with_lowest_path_cost(self):
        """
        From the GC's perspective, the GM RSSIs will be 25, and the
        ZC will have a true RSSI (normally in the 60-70 range).
        This will ensure that the GC will have lower path cost to the
        wired bridge than the wired GMs. We want the DUT GC to prefer
        to forward STP-routed traffic to wired bridge vs. wired GM. This is
        done so that we can differentiate when the DUT GC is forwarding via
        STP path vs. direct routing to a GM.
        """
        LOW_RSSI = 25
        STP_CONVERGE_SEC = 60

        for gm in self.gms:
            self.athconfigs[gm].settestpoint('ath0', 'fixedrssi', LOW_RSSI)

        self.brctls[self.gc].wait_for_stp_stable(start_delay=STP_CONVERGE_SEC)

    def _wait_for_devices_ready(self, zps):
        """Once STP stablizes, ensure that we can hit all diag pages"""
        self.logger.info('Wait for devices: {}'.format(zps))
        self._wait_for_all_stp_stable()
        super(DirectRoutingSN, self)._wait_for_devices_ready(zps)

    def _run_group_playback(self, gms, test_name, zp_wire_list, send_list,
            no_send_list, zb_send_list=[], dynamic_dr_list=[],
            dynamic_eth_list=[], sniff_start_delay=0, via_ap_send_list=[],
            via_ap_no_send_list=[], force_stp_path=False):
        """
        Run basic group playback test.

        :param gc: dut group coordinator
        :type self.gc: :class:`~sonos.client.internal.SonosZoneComponent`
        :param gms: zp group members
        :type gms: :obj:`list` of
            :class:`~sonos.client.internal.SonosZoneComponent`
        :param test_name: name of test. also used for name of capture file.
        :type test_name: :obj:`str`
        :param zp_wire_list: list of ZPs whose ethernet wire will be
            enabled. All other ZPs in the testbeds/HH will have disabled
            ethernet.
        :type zp_wire_list: :obj:`list` of
            :class:`~sonos.client.internal.SonosZoneComponent`
        :param send_list: list of mac addresses sent by GC
        :type send_list: :obj:`list`
        :param no_send_list: list of mac addresses not sent by GC
        :type no_send_list: :obj:`list`
        :param zb_send_list: list of mac addresses sent by wired zone bridge
        :type zb_send_list: :obj:`list`
        :param dynamic_dr_list: list of :data:`DRInfo`
            :class:`collections.namedtuple` this is used to specify changing
            the direct routing state of the group after playback has started.
        :type dynamic_dr_list: :obj:`list` or []
        :param dynamic_eth_list: list of
            :data:`EthInfo` :class:`collections.namedtuple`
            this is used to specify changing the ethernet state of the group
            after playback has started.
        :type dynamic_eth_list: :obj:`list` or []
        :param sniff_start_delay: additional delay in seconds to start
            sniffing after playback has started
        :type sniff_start_delay: :obj:`int` or 0
        :param force_stp_path: Force STP path using software attenuation
        :type force_stp_path: :obj:`bool`
        """
        zp_test_group = [self.gc] + gms
        self._config_eth_ports(zp_wire_list)

        if force_stp_path:
            self._force_zc_with_lowest_path_cost()

        self._wait_for_devices_ready(zp_test_group)
        for eth_set in dynamic_eth_list:
            self._verify_dr([], self.gc, eth_set.zp, eth_set.dr_state,
                timeout=180)
        self.begin_watching_for_sync_error(zp_test_group)
        self._group_and_start_playback(self.gc, gms)
        for dr_set in dynamic_dr_list:
            self._verify_dr(dr_set.settestpoint_cmds,
                            self.gc,
                            dr_set.peer,
                            dr_set.is_dr_enabled,
                            timeout=180)
        for eth_set in dynamic_eth_list:
            if eth_set.enable_wire:
                eth_set.zp.wire.enable()
            else:
                eth_set.zp.wire.disable()
            self.subnet.clear_and_wait_for_devices([eth_set.zp],
                                                   timeout_seconds=300)
            self._verify_dr([], self.gc, eth_set.zp, eth_set.dr_state,
                            timeout=180)
        gc_mac = self._get_ath0_dr_mac(self.gc)
        self._run_sniffer_analysis(test_name, gc_mac,
                                   send_list, no_send_list,
                                   zb_send_list, sniff_start_delay,
                                   via_ap_send_list=via_ap_send_list,
                                   via_ap_no_send_list=via_ap_no_send_list,
                                   gc=self.gc)
        self.stop_watching_for_sync_error(zp_test_group)

    def _run_playback_dr_disabled(self, gc, gms, test_name,
            zp_wire_list=[]):
        super(DirectRoutingSN, self)._run_playback_dr_disabled(gc, gms,
            test_name)
        self._run_group_playback(gms, test_name, zp_wire_list,
            **self._get_dr_disable_lists(gms, zp_wire_list))

        Verify that a wireless DUT GC when grouped with
        2 wireless GMs where RSSI is less than 19 on one GM for playback
        will use STP to GM with less than 19 RSSI and direct route to
        other GM with higher/normal RSSI (SWPBL-121431)

        :type self.gc: :class:`~sonos.client.internal.SonosZoneComponent`
        :param self.gms: zp group memebers
        :type self.gms: :obj:`list` of
            :class:`~sonos.client.internal.SonosZoneComponent`

                                                       self.gms[0].serialNumber,
                                                       self.gms[1].serialNumber)
        """
        gms[0] RSSI == 18, and gms[1] will have RSSI where DR is enabled.
        this is done in the below call to _run_playback_dr_disabled.

                                   send_list=[self._get_ath0_dr_mac(gm_dr)],

    def _get_wireless_capture(self, cap_filter, display_filter, timeout):
        """
        A self-contained function to create a capture object.

        :param string cap_filter Capture filter to use on the capture.
        :param string display_filter Display filter to use on the packets.
        :param number timeout Number of seconds to capture frames.
        :return: The resulting capture
        :rtype: pyshark capture
        """
        with NamedTemporaryFile() as f:
            sniffer = pyshark.LiveCapture(interface=self.WIRELESS_CAP_ID,
                                          bpf_filter=cap_filter,
                                          output_file=f.name)
            sniffer.sniff(timeout=timeout)
            sniffer.close()

            # sniffer post-processing when sniffer is done capturing
            self.logger.info('Capture completed. Starting capture file '
                             'analysis.')
            # check that sniffer captured the expected data by via the
            # file size.  don't bother running rest of test if it's not
            # the expected size.
            self._verify_capture_file(f.name, cap_filter)

            capture = pyshark.FileCapture(f.name, display_filter)
            capture.load_packets()
            return capture

    def _test_probes(self, wired_devs, tests):
        """
        Test probes received using a given set of wired devices using
        a given callback.

        :param list wired_devs: list of Sonos devices to have wires attached
        :param function tests: a function callback that will be run on a probe
            from each unique ZP.
        """
        self._config_eth_ports(wired_devs)
        self._wait_for_devices_ready(self.zps)

        all_zp_ath0s = set([self._get_ath0_dr_mac(zp).lower()
            for zp in self.gms])
        ath0_bridge_mac_map = dict([(self._get_ath0_dr_mac(zp).lower(),
           zp.cli.get_mac_address('br0').lower()) for zp in self.zps])
        cap_filter = " or ".join(["wlan addr2 {}".format(x)
            for x in all_zp_ath0s])
        capture = self._get_wireless_capture(cap_filter,
            "wlan.fc.type_subtype == 0x4", 5)
        zps_seen = set()
        for pkt in capture:
            try:
                if pkt.wlan.ta in zps_seen:
                    continue
                tests(pkt, ath0_bridge_mac_map)
                zps_seen.add(pkt.wlan.ta)
            except AttributeError:
                pass
        self.verifyEqualOrFailCase(all_zp_ath0s, zps_seen,
           "Expect to verify probes from all ZPs in household")

        detected_gm_macs = set([entry.member.lower() for entry in entries])

                self.gms[-1].cli.get_mac_address('br0').lower(),
                detected_gm_macs,

                "in the multicast forwarding table after the multicast "
                "ageing timeout interval")

                                             display_filter,
                                             self.GROUP_AUDIO_SAMPLE_TIME)

        self.verifyGreaterEqualOrFailCase(len(list(capture)), 1,

            "the remaining group member")

        self.stop_watching_for_sync_error(zp_test_group)

    def _eth_fwd_mc_dynamic_playback(self, gc, gms, test_name, enable_wire,
            dr_state, zp_wire_list, send_list, no_send_list, zb_send_list=[]):
        """
        Helper function to run the ethernet forwarding multicast
        direct routing dynamic enable or disable test cases.

        :param self.gc: dut group coordinator
        :type self.gc: :class:`~sonos.client.internal.SonosZoneComponent`
        :param self.gms: zp group members
        :type self.gms: :obj:`list` of
            :class:`~sonos.client.internal.SonosZoneComponent`
        :param test_name: name of test. also used for name of capture file.
        :type test_name: :obj:`str`
        :param enable_wire: enable wire on self.gms?
        :type enable_wire: :obj:`bool`
        :param dr_state: direct routing state
        :type dr_state: :obj:`bool`
        :param zp_wire_list: list of ZPs whose ethernet wire will be
            enabled. All other ZPs in the testbeds/HH will have disabled
            ethernet.
        :type zp_wire_list: :obj:`list` of
            :class:`~sonos.client.internal.SonosZoneComponent`
        :param send_list: list of mac addresses sent by GC
        :type send_list: :obj:`list`
        :param no_send_list: list of mac addresses not sent by GC
        :type no_send_list: :obj:`list`
        :param zb_send_list: list of mac addresses sent by wired zone bridge
        :type zb_send_list: :obj:`list` or []
        """
        eth_list = [EthInfo(gm, enable_wire, dr_state) for gm in gms]
        """
        We need to wait for the gm to age out of multicast table
        before starting sniffer.
        """
        sleep_time = (self.MULTICAST_AGEING_TIME_SECS +
                      self.MULTICAST_BUFFER_TIME_SECS)
        """
        Start group playback and change ethernet state on self.gms which
        should dynamically enable or disable multicast direct routing
        from self.gc to self.gms.
        """
        self._run_group_playback(gms, test_name,
                                 zp_wire_list, send_list, no_send_list,
                                 dynamic_eth_list=eth_list,
                                 zb_send_list=zb_send_list,
                                 sniff_start_delay=sleep_time,
                                 force_stp_path=True)
