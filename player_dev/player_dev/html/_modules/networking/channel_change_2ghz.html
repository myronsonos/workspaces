[docs]    def supported_channels(self, zp, freq, interface):
        """
        Verifies expected channel frequencies match the channel frequencies in the wifi driver

        Arguments:
        zp -- SonosZoneComponent, DUT
        freq -- channel's frequency
        interface -- str, wifi interface

        Keyword Arguments:
        None

        Returns:
        boolean
        """
        res = False
        self.logger.info("\n\nZP: %s - Verifying supported channels" % zp)
        supported_channels = zp.cli.get_supported_channels(interface)
        if freq in supported_channels:
            res = True
        else:
            res = False
        return res



[docs]    def check_support_ch12_and_ch13(self, zp, interface):
        """
        Verifies channel 12 and 13 frequencies are supported in the wifi driver

        Arguments:
        zp -- SonosZoneComponent, DUT
        interface -- str, wifi interface

        Keyword Arguments:
        None

        Returns:
        boolean
        """
        res = False
        self.logger.info("\n\nZP: %s - Verifying channels 12 and 13" % zp)
        ch12 = '2467'
        ch13 = '2472'
        supported_channels = zp.cli.get_supported_channels(interface)
        self.logger.debug(supported_channels)
        if ch12 and ch13 in supported_channels:
            res = True
        else:
            return res
        return res

        """
        Verifies if channel change was successful

        Arguments:
        zp -- SonosZoneComponent, DUT
        interface -- str, wifi interface

        Keyword Arguments:
        None

        Returns:
        None
        """
        self.verifyEqualOrFailCase(channel,
            zp.cli.get_wireless_channel(INTERFACE_2GHZ),
            "Verify channel change")



[docs]    def set_channel(self, zp, region_ch_freq):
        """
        Sets the desired channel frequency

        Arguments:
        zp -- SonosZoneComponent, DUT
        region_ch_freq -- str, channel's frequency

        Keyword Arguments:
        None

        Returns:
        None
        """
        # change from unicode to str as change_wireless_channel requires str
        self.logger.info("\nZP: %s - Setting wifi channel to: %s" % (zp, region_ch_freq))
        if zp.cli.change_wireless_channel('ath0', region_ch_freq):
            # allow 2 sec for channel change to happen
            time.sleep(2)
            self.verify_channel_change(zp, region_ch_freq)
        else:
            self.fail("%s channel is not available" % region_ch_freq)



[docs]    def change_home_region(self, zp, region):
        """
        Sets the home region of the Sonos device

        Arguments:
        zp -- SonosZoneComponent, DUT
        region --MdpRegion, region code.  1 - USA, 2 - Europe, 3 - China, 4 - Japan

        Keyword Arguments:
        None

        Returns:
        None
        """
        self.logger.info("ZP: %s - Setting home mdp region to: %s" % (zp, region))
        if zp.cli.mdp.set_region(region):
            # reboot needed after changing mdp region
            self.logger.info("Rebooting %s" % zp)
            self._reboot(zp)
        else:
            self.logger.error("Failed to change region code to %s on %s" % (region, zp))

[docs]    def test_check_support_for_ch12_ch13(self):
        self.logger.info("\n\nInitiating CH12 and CH13 support test")
        for i in range(len(self.zps)):
            for j in range(len(CH12_CH13_REGIONS)):
                self.change_home_region(self.zps[i], CH12_CH13_REGIONS[j])
                if self.verifyTrueOrFailCase(self.check_support_ch12_and_ch13(self.zps[i], 'ath0'), "Should support channel 12 and 13"):
                    self.logger.info("Channel 12 and 13 are supported on region: %s" % CH12_CH13_REGIONS[j])
            for k in range(len(CH12_CH13_NOT_SUPPORTED)):
                self.change_home_region(self.zps[i], CH12_CH13_NOT_SUPPORTED[k])
                if self.verifyFalseOrFailCase(self.check_support_ch12_and_ch13(self.zps[i], 'ath0'), "Should NOT support channel 12 and 13"):
                    self.logger.info("Channel 12 and 13 are NOT supported as expected for %s" % CH12_CH13_NOT_SUPPORTED[k])
