[docs]    def install_tools_on_zps(self, zps, tools):
        """
        Install a list of tools onto multiple zps
        :param zps: zps to install tool(s) on
        :type zps :obj:`list` of class:~sonos.client.internal.SonosZoneComponent
        :param tools: binary tools to install
        :type tools: :obj:`list` of `str`
        """
        ret = True

        for zp in zps:
            if not all([self.zp_install_tool(zp, t) for t in tools]):
                ret = False

        return ret



[docs]    def zp_install_tool(self, zp, tool_name, force_install=False):
        """
        Install a binary tool on a zp
        :param zp: zp to install binary on
        :type zp :class:~sonos.client.internal.SonosZoneComponent
        :param tool_name: binary name of tool to install
        :type interface :obj:`str`
        :param force_install: If binary exists, overwrite it
        :type interface :obj:`bool`
        :return: installation was successful
        :rtype: :obj:`bool`
        """
        nv_path = '/jffs/'
        chk_cmd = '[ -x {}{} ] && echo $?'.format(nv_path, tool_name)
        get_cmd = 'wget -O {} {}../{}/{}/out/{} && chmod +x {}'.format(
                  nv_path + tool_name,
                  self.suite.get_test_update_url(),
                  'platforms-out',
                  zp.cli.get_arch(),
                  tool_name,
                  nv_path + tool_name)

        if force_install or zp.cli.command(chk_cmd) != "0":
            zp.cli.command(get_cmd)
            if zp.cli.command(chk_cmd) != "0":
                return False

        return True

[docs]    def are_satellites_on_interface(self, interface):
        """
        Get whether all satellites settled on given interface
        :param interface: interface that will be checked
        :type interface :obj:`str`
        :return: whether all satellites settled on given interface
        :rtype: :obj:`bool`
        """
        satellites_to_check = []
        self.prepare_brctl_for_zp(self.ht_mstr)
        for zp in self.ht_sats:
            satellites_to_check.append(self.mac_to_int(zp.serialNumber) + 1)
        ret = all([(entry.dev == interface) for entry in
            self.brctls[self.ht_mstr.serialNumber].showports()
            if self.mac_to_int(entry.mac) in satellites_to_check])
        return ret

[docs]    def booted_up_after(self, zp, start):
        """
        This function can be used to check whether ZP booted up
        after time point 'start'(datetime type).
        :param zp: reboot checking DUT
        :type zp :class:~sonos.client.internal.SonosZoneComponent
        :param start: time when reboot command issued to ZP
        :type start :class: datetime.datetime
        :return: whether ZP rebooted
        :rtype: :obj:`bool`
        """
        try:
            # current_zp is timedelta type
            current_zp = zp.cli.get_uptime()
            # current_local is datetime type
            current_local = datetime.now()
        except:
            return False
        self.logger.error("{} time since reboot: {} zp uptime: {}".format(
            zp, current_local - start, current_zp))
        # (current time) - (time when reboot issued to ZP) > (ZP uptime)
        return (current_local - start) > current_zp

        """
        Check whether mode running on ZP matches 'mode'.
        Possible modes are: 'sonosnet', 'station' and 'satellite'
        """
        mode_t = zp.diag.get_fullstatus().mode
        ret = (mode_t == mode)
        if not ret:
            self.logger.error("Mode dismatch {} {}", mode_t, zp.serialNumber)
        return ret



[docs]    def get_freshed_zp(self, target_zp, ip):
        """
        Try to ZP automaticly refreshed by framework. 'target_zp' is
        original object of that ZP. 'ip' is original IP.
        'target_zp.ip' may be updated by framework, but other part of
        it won't be updated after refreshing. Thats's why original IP is
        needed here as input.
        :return: refreshed ZP object
        :rtype: :obj:`SonosZoneComponent`
        """
        ret = None
        for zp in self.my_devices:
            if zp.serialNumber == target_zp.serialNumber:
                ret = zp
        if ret is None or id(target_zp) == id(ret) or ret.ip == ip:
            return None
        # Returned new ZP if it is not None and is not original one
        return ret



[docs]    def refresh_zp(self, target_zp, ip, wait_for_new=False, timeout=120):
        """
        Get ZP automaticly refreshed by framework.
        If 'wait_for_new' is true, this function will wait for ZP really
        get refreshed. If 'wait_for_new' is false, when framework is
        still holding original ZP object, None will be returned.
        :return: refreshed ZP object
        :rtype: :obj:`SonosZoneComponent`
        """
        self.logger.debug("Original ZP id {} {}".format(id(target_zp),
                                                        target_zp))
        if wait_for_new:
            wait_until_true(
                lambda: self.get_freshed_zp(target_zp, ip) is not None,
                iteration_delay=5, timeout_seconds=timeout)
        ret = self.get_freshed_zp(target_zp, ip)
        self.suite.all_testbed_devices.remove(target_zp)
        self.suite.all_testbed_devices.append(ret)
        # Update already cached info
        if self._is_among_proxies(target_zp):
            self.ht_mstr = ret
        if self._is_among_satellites(target_zp):
            tmp_zps = [zp for zp in self.ht_sats
                       if zp.serialNumber != ret.serialNumber]
            self.ht_sats = tmp_zps + [ret]
        # Update brctl entries that may already got cached
        if ret.serialNumber in self.brctls:
            self.brctls[ret.serialNumber] = Brctl(ret)
        self.logger.debug("Newly refreshed ZP {!r}".format(ret))
        self.logger.debug("Current self.my_device {}".format(self.my_devices))
        # Update DUT
        tmp_zps = [zp for zp in self.devices_to_test
            if zp.serialNumber != target_zp.serialNumber]
        self.devices_to_test = tmp_zps + [ret]
        ret.subscribe_all_services()
        return ret


    def _is_among_proxies(self, zp):

    def _is_among_satellites(self, zp):

    def _interface_and_sn_matches(self, interface, if_addr, mac_addr):
        """
        This function checks whether address of a interface and a
        serial number of a MAC of a ZP matches. 'sn' in function name
        means serial number.
        """
        if_addr_int = self.mac_to_int(if_addr)
        mac_addr_int = self.mac_to_int(mac_addr)
        if interface == "ath0":
            return if_addr_int == (mac_addr_int + 1)
        elif interface == "ath1":
            return if_addr_int == (mac_addr_int + 2)
        elif interface == "eth0":
            return if_addr_int == mac_addr_int
        else:
            return False

    def _interface_addr_was_learned_by_zp(self, zp, interface, expected_zp,
            satip):
        """
        Check whether address of interface that belongs to expected_zp is
        learned by zp.
        'interface' could be 'ath0', 'ath1' or 'eth0'.
        """
        entries = self.brctls[zp.serialNumber].showports()
        if not entries:
            return False
        expected_mac = expected_zp.serialNumber
        for entry in entries:
            if_mac = entry.mac
            if self._interface_and_sn_matches(interface, if_mac,
                    expected_mac) and entry.satip == satip:
                return True
        return False

        """
        Convert mac address to integer format. This function can be used
        when you want comparison between MAC adresses of different
        interfaces to be easier.
        :return: integer format of a MAC address
        :rtype: :obj:`int`
        """
        return int(mac.replace(":", ""), 16)

        """

[docs]    def verify_p2p_tunnel_entries(self):
        """
        Verify that on each satellite, MAC address of primary is already
        learned; and verify that, on primary, MAC and IP address of each
        satellite is already learned.
        MAC of primary learned by satellites should be MAC of its ath1
        MAC of satellite learned by satellites should be MAC of ath0
        """
        for zp in self.devices_to_test:
            self.prepare_brctl_for_zp(zp)
        for zp in self.ht_sats:
            valid = self._interface_addr_was_learned_by_zp(self.ht_mstr,
                                                           "ath0", zp, zp.ip)
            self.verifyTrueOrFailCase(
                valid, "Verifying primary {} "
                "learned satellite {} in bridge.".format(self.ht_mstr, zp))
            valid = self._interface_addr_was_learned_by_zp(zp, "ath1",
                                                           self.ht_mstr, None)
            self.verifyTrueOrFailCase(
                valid, "Verifying satellites {} "
                "learned primary {} in bridge.".format(zp, self.ht_mstr))



[docs]    def verify_stp_forwarding_state_for_zp(self, local_zp, other_zp):
        """
        Verify that, in STP status of local_zp, state of link between
        other_zp and itself is 'forwarding'
        """
        self.prepare_brctl_for_zp(local_zp)
        stp_entries = self.brctls[local_zp.serialNumber].showstp_tunnels()
        forwarding = False
        for entry in stp_entries:
            if (self._interface_and_sn_matches(
                    "ath1", entry.mac, other_zp.serialNumber) and
                    entry.state == "forwarding"):
                forwarding = True
                break
        self.verifyTrueOrFailCase(forwarding, "Verifying STP forwarding state "
                                  "for {} and {}.".format(local_zp, other_zp))



[docs]    def clean_up_setup(self):
        """
        Ensure testbed devices are in a clean state, ie. freshly rebooted
        and Telnet-connectable.

        Raises sonos.exception.TimeoutError
        """
        self.logger.info("Re-enable PHY and wait for IP-connectivity")
        for zp in self.my_devices:
            zp.wire.enable()

        wait_until_true(
            lambda: all([self.is_layer3_connected(zp) for zp in self.my_devices]),
            iteration_delay=(10),
            timeout_seconds=(60 * 2))

        self.logger.info("Reboot ZPs")
        run_concurrent_processes(
            lambda zp: zp.cli.reboot(True),
            self.my_devices)

        check_telnet_list = self.my_devices
        expiry_time = time.time() + (60 * 5)
        self.logger.info("Wait for ZPs to become Telnet-connectable")
        while len(check_telnet_list) and time.time() <= expiry_time:
            time.sleep(15)
            check_telnet_list = [zp for zp in check_telnet_list
                                 if not self.is_telnet_cli_available(zp)]

        if len(check_telnet_list):
            raise TimeoutError(
                "Could not establish Telnet connection to the following: " +
                "\n".join([repr(zp) for zp in check_telnet_list]))
