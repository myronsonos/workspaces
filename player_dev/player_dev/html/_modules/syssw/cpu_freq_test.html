    """
    Checks CPU freq transitions
    Steps:
    1) Reboot
    2) Verify that freq goes to 996000 Hz shortly after boot
    3) Let device idle
    4) Verify that freq goes to 396000 Hz when idle
    5) Play audio
    6) Verify that the freq goes to 996000 Hz while playing
    7) Mute the player
    8) Stop audio
    9) Immediately verify freq is 996000 Hz
    10) Verify that freq drops to 396000 Hz shortly after audio stops.
    """

        """
        Turns the ampctl mute setting off, empties the queue, and clears the

        for zp in self.devices_in_test:

            zp.cli.remove_amp_power_off()


    def _cleanup(self, zp):
        """
        Clears the play queue, play state, unmutes the player, and removes any
        ampctl overrides

        :param zp: The zone player under test
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent
        """
        # clear the queue
        zp.AVTransport.clear_queue_and_wait()
        # set mute off
        if zp.cli.can_override_amp_interface():
            self.verifyTrueOrFailCase(zp.cli.set_ampctl('clear-override'),
                                      'Ampctl settings should be '
                                      'back to their default values')
        # clear the play state - sets AVTransportURI = "", previously
        # self.BUZZER_URI
        zp.AVTransport.clear_play_state()
        zp.RenderingControl.SetMute(False)
        for diag, level in DIAG_LEVELS:
            zp.diag.set_diag_level(diag, 0)


[docs]    def wait_for_speed(self,
                       zp,
                       speed,
                       iteration,
                       wait_time,
                       msg,
                       ignore_timeout=False):
        """
        Waits for the CPU to go to the given frequency

        :param zp: The zone player under test
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent
        :param speed: The desired speed to wait for
        :type speed: :obj:`int`
        :param iteration: The current test iteration
        :type iteration: :obj:`int`
        :param wait_time: The amount of time to wait before timing out if the
                            correct speed is not set.
        :type wait_time: :obj:`int`
        :param msg: Message to append to timeout reason. Usefull for logging
                    fail location
        :type msg: :obj:`str`
        :param ignore_timeout: Whether or not to ignore timeout errors while
                                waiting for the speed to be set.
        :type ignore_timeout: :obj:`bool`
        """
        """
        The following line reduces noise in the log by increasing the
        iteration delay for checks that expect a long wait time
        """
        delay = 1 if wait_time <= 10 else 20
        try:
            wait_until_true(
                lambda: speed == zp.cli.get_cpu_frequency_hz() \
                and not ignore_timeout,
                iteration_delay=delay,
                timeout_seconds=wait_time,
                reason='Timed out while waiting for {} '
                'on {}: {}'.format(speed, zp, msg))
        except TimeoutError as e:
            if ignore_timeout:
                self.logger.info("{} did not set CPU to {}: {}"
                                 .format(zp, speed, msg))
                expected = (zp.high_cpu_speed if speed == zp.idle_cpu_speed
                           else zp.idle_cpu_speed)
                self.verifyEqualOrFailCase(expected,
                                           zp.cli.get_cpu_frequency_hz(),
                                           "{} cpu should be running at {}"
                                           .format(zp, expected))
            else:
                self.log_errors(zp, e, iteration)



[docs]    def log_errors(self, zp, e, iteration):
        """
        Method to log errors found while testing a re-raise exceptions captured

        :param zp: The zone player under test
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent
        :param e: The exception that was caught
        :type: :obj:`Exception`
        :param iteration: The iteration number of the test
        :type: :obj:`int`
        :raises e: The exception passed in
        """
        self.logger.debug("Exception hit on {} after {} iterations of test"
                          .format(zp, iteration))
        self.logger.debug(''.join('{} {}\n'.format(tup['t_stamp'],
                                    tup['log']) for tup in zp.cli.get_dmesg()))
        zp.cli.get_anacapa_trace()
        zp.cli.command('cat /proc/blackbox/log')
        if zp.voice_capable:
            zp.cli.command('cat /proc/driver/micctl')
        raise e



[docs]    def perform_test(self, zp, voice_enabled=False):
        """
        Checks whether the appropriate CPU frequency is returned
        Note: Devices without amps still change CPU frequency during playback.
              Anacapa can still schedule to "power off" amps
              on devices with no amps.

            1. after reboot
            2. when the player enters the idle state after reboot
            3. while playing music
            4. when the player is muted
            5. after stopping the music
            6. when the player enters the idle state after stopping the music
            7. Repeats for the given number of iterations

        :param zp: The zone player to run the test on
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent
        :param voice_enabled: Whether or not the test is running against a \
        voice capable devices with voice enabled
        :type voice_enabled: :obj:`bool`
        """
        try:
            self._cleanup(zp)
            zp.cli.set_amp_power_off(2, False)
            log = AnacapaTraceLogInspector(zp)
            for i in range(ITERATIONS):
                wait_time = (DEFAULT_AMP_OFF if zp.is_s12() or
                                zp.is_s13() else FAST_AMP_OFF)
                # TODO: Reduce this buffer after SWPBL-86458
                wait_time += 30  # add small buffer to threshold
                self.logger.info('*' * 80)
                self.logger.info('Starting iteration {} on {}'.format(i + 1,
                                                                      zp))

                # REBOOT --- 1GHz (HIGH)
                zp.reboot()
                zp.cli.set_blackbox_mask(BLACKBOX_AMPCTL_MASK)

                # WAITING FOR IDLE --- 400MHz (LOW)
                """
                Wait for anacapa to start and then attempt to turn the amps off.
                Once the amps are off, speed should drop to IDLE
                Devices without amps are idle when there's no playback.
                """
                wait_until_true(lambda: zp.cli.is_anacapa_running(),
                                iteration_delay=5,
                                reason="Waiting for anacapa to start")
                for diag, level in DIAG_LEVELS:
                    zp.diag.set_diag_level(diag, level)

                if zp in self.voice_devices:
                    """
                    Assumes these devices have a voice accout enabled
                    """
                    if not voice_enabled:
                        """
                        Voice enabled devices only scale CPU
                        when voice is disabled
                        """
                        self.logger.info("Pressing zone M to mute mic")
                        zp.cli.mute_mic()
                    else:
                        self.logger.info("Pressing zone M to unmute mic")
                        zp.cli.unmute_mic()
                self.logger.info('Waiting for first IDLE after boot')
                """
                Voice enabled players should not drop to IDLE speed so the
                above listed call to wait_for_speed() should fail. We should,
                therefore, catch this failure and move on.
                """
                self.wait_for_speed(zp, zp.idle_cpu_speed, i + 1, wait_time,
                                    "Waiting for first IDLE after boot",
                                    ignore_timeout=voice_enabled)

                # MUTE + IDLE
                self.logger.info('Mute {}'.format(zp))
                zp.RenderingControl.SetMute(True)
                self.logger.info("Waiting for return to "
                                 "IDLE after being muted on"
                                 " {}".format(zp))
                self.wait_for_speed(zp, zp.idle_cpu_speed, i + 1,
                                    IDLE_SPEED_TIMEOUT,
                                    "Waiting for return to IDLE after mute",
                                    ignore_timeout=voice_enabled)
                zp.RenderingControl.SetMute(False)
                """
                Sol will jump to HIGH SPEED when unmuted so wait for speed to
                settle back to IDLE
                """
                self.logger.info("Waiting for return to IDLE after unmute of {}"
                                 .format(zp))
                # Testing revealed that it can take 5 secs from
                # unmute to drop speed
                self.wait_for_speed(zp, zp.idle_cpu_speed, i + 1,
                                    IDLE_SPEED_TIMEOUT,
                                    "Waiting for return to IDLE after unmute",
                                    ignore_timeout=voice_enabled)
                zp.RenderingControl.SetVolume(1)

                # PLAY MUSIC --- 1GHz (HIGH)
                self.logger.info('Play Music on {}'.format(zp))
                zp.diag.post_diagmsg(
                    "**************Starting Buzzer*************")
                zp.AVTransport.Play_buzzer_and_wait(timeout_seconds=10)
                self.verifyEqualOrFailCase(zp.high_cpu_speed,
                                           zp.cli.get_cpu_frequency_hz(),
                                           "{} CPU frequency should be {} Hz"
                                           .format(zp, zp.high_cpu_speed))
                zp.RenderingControl.SetMute(True)
                self.wait_for_speed(zp, zp.idle_cpu_speed, i + 1, wait_time,
                                    "CPU frequency should not be impacted by "
                                    "muting player", ignore_timeout=True)
                zp.RenderingControl.SetMute(False)
                self.wait_for_speed(zp, zp.high_cpu_speed, i + 1, wait_time,
                                    "CPU frequency should not be impacted by "
                                    "muting player")

                # STOP MUSIC --- 1GHz (HIGH) still because just stopped
                self.logger.info('Stop Music on {}'.format(zp))
                zp.diag.post_diagmsg(
                    "**************Stopping Buzzer*************")
                zp.AVTransport.Stop_and_wait()
                wait_until_true(
                    lambda: log.find_next(".*[scheduling|amp\s+turn]\s+off.*"),
                    iteration_delay=10, timeout_seconds=IDLE_SPEED_TIMEOUT,
                    reason="Waiting for anacapa to schedule amp off "
                    "after stopping audio on {}".format(zp))

                # IDLE --- 4MHz
                self.logger.info('Waiting for IDLE after stop on {}'.format(zp))
                self.wait_for_speed(zp, zp.idle_cpu_speed, i+1, wait_time,
                                    "Waiting for IDLE after stopping audio",
                                    ignore_timeout=voice_enabled)
        except Exception as e:
            # Additional except block added for debugging jenkins failures
            self.logger.info("{} hit exception. Dumping anacapa.trace\n{}".
                             format(zp, zp.cli.get_anacapa_trace()))
            raise e
