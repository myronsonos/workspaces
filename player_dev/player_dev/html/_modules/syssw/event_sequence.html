    def __init__(self, zp, run_as_fast_as_possible=False):
        """
        Initialize the EventSequence with a specific zoneplayer
        This object is used to execute a set of actions sequentially on the
        associate zoneplayer object

        :param zp: The zoneplayer to associate with this sequence
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`

        :param run_as_fast_as_possible: If True, when calling run, all delays
            in the event sequence will be ignored. This is useful for running
            button press events in extremely rapid succession, but will break
            the expected result for actions that require a delay, like
            orientation changes or holding buttons
        :type run_as_fast_as_possible: :obj: `bool`
        """
        self.run_as_fast_as_possible = run_as_fast_as_possible
        self.action_sequence = []
        self.expected_sources = []
        self.expected_infos = []
        self.logger = logging.getLogger(
            make_log_name([__name__, self.__class__.__name__]))
        self.zp = zp
        self.thresholds = zp.cli.get_temperature_thresholds()
        self.logger.info(
            "Temperature thresholds for {} are {}".format(zp, self.thresholds))


[docs]    def sleep(self, amt):
        if self.run_as_fast_as_possible:
            self.logger.debug("Skipping {} second sleep.".format(amt))
            return
        else:
            self.logger.debug("Sleeping {} seconds.".format(amt))
            sl(amt)



[docs]    def run(self):
        """
        Iterate through self.action_sequence, executing each action on the
        associated zoneplayer
        """
        self.logger.info(
            "On zp {}, executing event sequence...\n{}".format(self.zp, self))
        for action in self.action_sequence:
            try:
                action(self.zp)
            except TimeoutError as e:
                self.logger.error("Timeout while evaluating {}".format(action))
                self.logger.error("TimeoutError: \n {}".format(e))
            self.sleep(0.1)  # an extra buffer between events

                self.logger.error(
                    "{} cannot measure cpu warn temperature.".format(self.zp))
                return
            self.expected_sources.append(Evt.Source.CPU.value)
            self.expected_infos.append(Evt.Info.TEMP_WARNING.value)
            self.expected_sources.append(Evt.Source.CPU.value)
            self.expected_infos.append(Evt.Info.TEMP_OK.value)
            self.action_sequence.append(
                lambda pl: pl.cli.spoof_cpu_temperature(
                    self.thresholds.CPUWarn + 1))
            self.action_sequence.append(lambda garbage: sleep(6))
            self.action_sequence.append(
                lambda pl: pl.cli.clear_spoofed_cpu_temperature(False))
            self.action_sequence.append(lambda garbage: sleep(6))

        # temperature events need a 6 second pause for events to boil upwards
        elif action == self.actions.amp_temp_fault_normal:
