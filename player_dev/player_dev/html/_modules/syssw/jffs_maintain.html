[docs]    def generate_unique_fenway_limelight(self):
        """
        Generator which yields only one instance of Play:3, Play:1, SUB, and
        Limelight

        :return: a playback device in the testbed
        :rtype: :class:`~sonos.client.internal.SonosZoneComponent`

        """
        models = {}
        for zp in [zp for zp in self.my_devices if zp.cli.is_fenway() or
                   zp.cli.is_playbar()]:
            reboot_count = 0
            # If we're already seen a device of this model, skip this device.
            # Note: zp.modelNumber is actually a string, such as "ZP100"!
            if zp.modelNumber in models:
                continue
            models[zp.modelNumber] = 1
            self.logger.debug('About to check for fix on {}'.format(zp))
            """
            Devices that previously erased blocks that have already been erased
            need to be rebooted twice for the fix to take effect.
            """
            while(self.get_erased_block_count(zp) > LOW_ERASE_THRESHOLD):
                if reboot_count > 2:
                    self.stop('Unable to reset erased block count')
                self.subnet.reboot_devices([zp])
                reboot_count += 1
            if reboot_count == 0:
                # Reboot the DUT to so that the erased block count is low
                self.subnet.reboot_devices([zp])
            yield zp



[docs]    def get_files_from_server(self, zp):
        """
        On each ZP in the testbed, uses wget to download each of the test files.
        It will name these files test<some index>, where some index is a number
        between 0 and one less than the number of files.

        :param zp: The receiver ZP
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`
        """
        zp.cli.command('mkdir {}'.format(TEMP_ZP_LOCATION))
        for f in self.file_list:
            file_location = 'http://{}/{}'.format(self.host.host(), f.name)
            zp.cli.command('wget -O {}{} {}'.format(TEMP_ZP_LOCATION, f.name,
                                                    file_location))
            # Due to limited space in JFFS, DOCK should only grab the first file
            if zp.is_dock():
                break



[docs]    def create_files(self, file_count, max_file_size):
        """
        On the test runner, creates each of the files needed for testing. The
        first file should be 512kB in size, the others will be of random size
        between 1 and MAX_SIZE.

        :return: A list of named tuples of type 'files' with the field names
            'size', 'name', and 'file'. 'file' is a file object, 'size' is the
            size of the file in kilobytes, 'name' is the file name
        :rtype: :obj:`list` of :obj:`collections.namedtuple`
        """
        file_list = []
        file_list.append(self.make_file(KNOWN_FILE_KB))
        for index in range(1, file_count):
            file_list.append(self.make_file(random.randrange(1, max_file_size)))
        self.verifyTrueOrStop(len(file_list) > 0, 'At least one file should be '
                              'available')
        self.logger.debug('Created {} length list'.format(len(file_list)))
        return file_list



[docs]    def make_file(self, file_size):
        """
        Creates a test file of size file_size

        :param file_size: The size of the file to be created, in kilobytes
        :return: A named tuple of type 'files' with the field names
            'size', 'name', and 'file'. 'file' is a file object, 'size' is the
            size of the file in kilobytes, 'name' is the file name
        :rtype: :obj:`'collections.namedtuple`
        """
        files = namedtuple('files', 'size name file')
        f = tempfile.NamedTemporaryFile(prefix=TEST_FILE_PREFIX,
                                        dir=TEST_FILE_LOCATION)
        f.write(bytearray(file_size * ONE_KILO))
        f.flush()
        return files(file_size, f.name.lstrip(TEST_FILE_LOCATION), f)

[docs]    def remove_files(self, zp, n_files, file_list):
        """
        Removes files from the ZP and the list of files

        :param zp: The receiver ZP
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`
        :param n_files: The number of files to remove from the zp
        :type n_files: :obj:'int'
        :param file_list: The list of files to check for removal
        :type file_list: :obj:'list' of :obj:`collections.namedtuple`
        :returns: The updated list of files, after removing n_files
        :rtype: :obj:'list' of :obj:`collections.namedtuple`
        """
        try:
            for index in range(n_files):
                f = file_list[index]
                zp.cli.command('rm {}{}'.format(TEMP_ZP_LOCATION, f.name))
                file_list.remove(f)
                return file_list
        except:
            self.logger.error('Caught exception at file index {}'.format(index))
            self.logger.info('length of file list: {}'.format(len(self.file_list)))



[docs]    def verify_files(self, zp, file_list):
        """
        Checks for the existence of each test file. If each file is found, it
        checks the size of the file against the list of file sizes.

        :param zp: The receiver ZP
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`
        :param file_list: The list of files to check for removal
        :type file_list: :obj:'list' of :obj:`collections.namedtuple`
        """
        for f in file_list:
            file_found = zp.cli.does_file_exist('{}{}'.format(TEMP_ZP_LOCATION,
                                                              f.name))
            self.verifyTrueOrFailCase(file_found,
                                      '{} should exist'.format(f.name))
            if not file_found:
                continue
            file_size = self.get_file_size(zp, '{}{}'.format(TEMP_ZP_LOCATION,
                                                             f.name))
            self.verifyEqualOrFailCase(f.size * 1024, file_size,
                                       '{} should be {}kB'.format(f.name,
                                                                  f.size))
            # Due to limited space in JFFS, DOCK should only grab and verify
            # the first file
            if zp.is_dock():
                break



[docs]    def get_file_size(self, zp, path_to_file):
        """
        Checks for the existence of the file. If the file is found, it
        gets and returns the size of the file.

        :param zp: The receiver ZP
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`
        :param list sizes: A list of the expected file sizes
        """
        file_found = zp.cli.does_file_exist('{}'.format(path_to_file))
        self.verifyTrueOrFailCase(file_found,
                                  '{} should exist'.format(path_to_file))
        if not file_found:
            return None
        file_info = zp.cli.command('ls -l {}'.format(path_to_file))
        file_info = file_info.split()
        file_size = int(file_info[4])
        return file_size



[docs]    def get_erased_block_count(self, zp):
        """
        Cats proc/vnb on Fenway products or PLAYBAR and gets the number of
        erased blocks

        :param zp: The receiver ZP
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`
        :returns: The number of erased blocks or None for unsupported devices
        :rtype: :obj:'int'
        """
        if zp.cli.is_fenway() or zp.cli.is_playbar():
            results = zp.cli.command('cat /proc/vnb')
            return int(re.search(ur'Blocks read/written/erased:\s+\d+/\d+/(\d+)',
                                 results).group(1))
        else:
            return None



[docs]    def verify_vnb_erases(self, zp, erased, threshold, erase_expected=False):
        """
        This method is written to test SWPBL-40647. It verifies that, on
        Fenway products, JFFS is only erased when expected. In most cases,
        catting /proc/vnb should only show a small number of erases. If files
        have been added and removed from JFFS, it should show a much larger
        number of erases. This test uses a loose set of criteria to verify this
        is the case.

        :param zp: The receiver ZP
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`
        :param erase_expected: This shold be True in the case that a larger
                                number of erases is expected
        :type erase_expected: :obj:'bool'
        """
        if erase_expected:
            self.verifyGreaterOrFailCase(erased, threshold, 'A larger number of'
                                         ' erases is expected')
        else:
            self.verifyLessEqualOrFailCase(erased, threshold, 'A small number of'
                                           ' erases is expected')

[docs]    @combinatorial("generate_testbed_unique_devices")
    def test_maintain_jffs(self, zp):
        """
        Combinatorial test that gets the test files from the sever, verifies the
        DUT has the correct files and file sizes. It then reboots the DUT and
        rechecks each of the files.

        :param zp: The receiver ZP
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`
        """
        files_to_check = self.file_list[:]
        self.get_files_from_server(zp)
        self.verify_files(zp, files_to_check)
        self.subnet.reboot_devices([zp])
        self.verify_files(zp, files_to_check)
        files_to_check = self.remove_files(zp, 2, files_to_check)
        self.subnet.reboot_devices([zp])
        self.verify_files(zp, files_to_check)
        zp.cli.command('rm {}*'.format(TEMP_ZP_LOCATION))



[docs]    @combinatorial("generate_unique_fenway_limelight")
    def test_vnb_erase(self, zp):
        """
        SWPBL-40647 -- This test verifies that already erased blocks in JFFS are
        not re-erased during boot up. The number of erases should be very low,
        unless there have been several files deleted from JFFS

        :param zp: The receiver ZP
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`
        """
        erased = self.get_erased_block_count(zp)
        self.verify_vnb_erases(zp, erased, LOW_ERASE_THRESHOLD)
        # Create and delete a 4MB random file in JFFS\
        zp.cli.command('dd if=/dev/urandom of={}dummy bs={} count={}'
                       .format(TEMP_ZP_LOCATION, FILE_BLOCK_SIZE_BYTES,
                               FILE_BLOCK_COUNT))
        wait_until_true(lambda: zp.cli.does_file_exist('{}dummy'.
                                                   format(TEMP_ZP_LOCATION)),
                            timeout_seconds=30, reason='Creating dummy file: ')
        zp.cli.command('rm {}dummy'.format(TEMP_ZP_LOCATION))
        """
        Pause to not check proc before it's updated. 90% of the time
        the 5 second sleep is long enough, wait_until_true catches the other
        10%. Using wait_until_true without the sleep fails 100% of the
        time
        """
        try:
            sleep(5)
            wait_until_true(lambda: self.get_erased_block_count(zp) > erased,
                            iteration_delay=5, timeout_seconds=20)
        except:
            self.fail('Timed out waiting for the number of erased blocks to'
                      ' increase.')
