[docs]    def remove_duplicate_models_from_list(self, zones):
        """
        Arguments:
        zones -- list, list of SonosZoneComponent objects

        Returns:
        list -- SonosZoneComponents with duplicate models removed

        """
        self.logger.debug("WARNING - DO NOT USE THIS METHOD OR ANYOTHER LOCATED IN ZONES.PY")
        unique_device_collection = {}
        for zp in zones:
            if zp.modelNumber not in unique_device_collection:
                unique_device_collection[zp.modelNumber] = zp
                continue
            self.logger.debug("found duplicate device: %s" % zp)
        return unique_device_collection.values()



[docs]    def get_zone_from_model(self, model, zones, omit_channel_map_devices=False):
        """
        Return one SonosZoneComponent of the specified model from this list
        Arguments:
        model -- string, model number of device to return
        zone_list -- list, list of SonosZoneComponent objects to find device in

        Returns:
        SonosZoneComponent -- device matching requested model
        None -- no SonosZoneComponent found which matches requested model

        """
        self.logger.debug("WARNING - DO NOT USE THIS METHOD OR ANYOTHER LOCATED IN ZONES.PY")
        zone = None
        if omit_channel_map_devices:
            matching_zones = [ zp for zp in zones if zp.modelNumber == model and self.get_channel_map_from_device(zp) == None ]
        else:
            matching_zones = [ zp for zp in zones if zp.modelNumber == model ]
        if matching_zones:
            zone = matching_zones[0]
        else:
            self.logger.error("could not find model %s" % model)
        return zone

[docs]    def get_zones_from_channel_map(self, channel_map, zone_list):
        """
        Return SonosZoneComponent objects specified in this channel map

        Arguments:
        channel_map -- string, channel map used by AddBondedZones
                       list, channel map used by AddHTSatellite

        zone_list -- list, list of SonosZoneComponents to search for devices from

        Returns:
        list -- SonosZoneComponents found in the channel map

        """
        self.logger.debug("WARNING - DO NOT USE THIS METHOD OR ANYOTHER LOCATED IN ZONES.PY")
        udn_list = [ zp.raw_udn for zp in zone_list ]
        matching_zones = []

        # if our channnel map was created for a HTSatellite configuration, it's going
        # to be a list. We'll turn it into a string below so we can search for UDN substrings
        if isinstance(channel_map, collections.Iterable):
            channel_map = str().join(channel_map)

        for udn in udn_list:
            if udn in channel_map:
                zone = [ zp for zp in zone_list if zp.raw_udn == udn and zp not in matching_zones ]
                if zone:
                    matching_zones.append(zone.pop())

        return matching_zones

[docs]    def cleanup_device_configuration(self, zp):
        """
        Cleanup bonded/satellite configurations for this device

        Arguments:
        zp -- SonosZoneComponent

        """
        self.logger.debug("WARNING - DO NOT USE THIS METHOD OR ANYOTHER LOCATED IN ZONES.PY")
        if len(zp.services) == 0:
            self.logger.info("%s (%s) has 0 services - reconnecting..." % (zp.modelNumber, zp.serialNumber))
        localsettings_dict = zp.diag.get_localsettings_as_dict()
        self.logger.info("satellite/bonded cleanup for %s" % zp.modelNumber)
        satellite_chmap = localsettings_dict.get('HTSatelliteDefn')
        self.logger.debug("satellite_chmap (%s): %s" % (zp.serialNumber, satellite_chmap))
        bonded_chmap = localsettings_dict.get('ChannelMapSet')
        self.logger.debug("bonded_chmap (%s): %s" % (zp.serialNumber, bonded_chmap))
        if zp.is_bonded():
             zp.DeviceProperties.cleanup_bonded_configuration()
        else:
            self.logger.debug("found no satellite/bonded configuration on %s" % zp.modelNumber)



[docs]    def wait_for_addition_of_subnet_devices(self, zones, timeout_seconds=60):
        """
        Wait for the specified zones to be detected by the UPnP framework

        Arguments:
        zones -- list of SonosZoneComponents

        Keyword Arguments:
        timeout_seconds -- integer, default=60, number of seconds to wait for removal of specified devices

        """
        self.logger.debug("WARNING - DO NOT USE THIS METHOD OR ANYOTHER LOCATED IN ZONES.PY")
        self.logger.debug("waiting %d seconds for the addition of %d subnet devices" % (timeout_seconds, len(zones)))
        subnet_handle = self.get_subnet_handle_from_device(zones[0])
        wait_until_true(lambda:\
                False not in [ 'uuid:' + zp.raw_udn in subnet_handle.device_udn_list for zp in zones ], timeout_seconds=timeout_seconds)
        self.logger.debug("%d devices added to subnet" % len(zones))

        """
        Get all SonosZoneComponents with the HHID of this device

        Arguments:
        hhid_zp -- SonosZoneComponent, zone to get HHID from

        Returns:
        list of SonosZoneComponents with same HHID as hhid_zp

        self.logger.debug("WARNING - DO NOT USE THIS METHOD OR ANYOTHER LOCATED IN ZONES.PY")

        if include_hhid_zp:
            hhid_devices.append(hhid_zp)
        return hhid_devices



[docs]    def build_device_channel_map(self, zp, chmap_config):
        self.logger.debug("WARNING - DO NOT USE THIS METHOD OR ANYOTHER LOCATED IN ZONES.PY")
        chmap = None
        chmap_strings = []
        chmap_response = None
        used_primary = False
        # don't consider the device available if it's already in a bonded/HT configuration
        available_zones = [ hhid_zp for hhid_zp in self.get_all_devices_in_this_household(zp) if not self.get_channel_map_from_device(hhid_zp) ]
        used_satellites = []
        chmap = ''
        if zp.is_soundbar():
            # LL is always going to be these three channels in a satellite config
            chmap = "%s:LF,RF;" % zp.raw_udn
            used_primary = True
        for satellite_data in chmap_config:
            satellite_channel, satellite_model = satellite_data
            if not used_primary and satellite_model == zp.modelNumber:
                satellite_zone = zp
                used_satellites.append(satellite_zone)
                used_primary = True
                self.logger.info("using primary zp: %s" % zp.modelNumber)
            else:
                candidate_zones = [ candidate for candidate in self.get_testbed_device_by_model([satellite_model]) if candidate not in used_satellites ]
                satellite_zone = candidate_zones[0]
                used_satellites.append(satellite_zone)
            chmap_strings.append("%s:%s;" % (satellite_zone.raw_udn, satellite_channel))
        if chmap_strings:
            if zp.is_soundbar():
                chmap_response = [ chmap + chmap_string for chmap_string in chmap_strings ]
            else:
                chmap_response = chmap + str().join(chmap_strings)
        self.logger.info("produced channel map: %s" % chmap_response)
        return chmap_response

[docs]    def wait_for_num_topology_devices(self, zp, num_expected_devices, timeout_seconds=60):
        """
        Wait for the status/topology diag page to show the expected number of devices (or more)

        Arguments:
        zp -- SonosZoneComponent
        num_expected_devices -- integer, number of devices to wait for

        Returns:
        boolean -- does status/topology page contain expected number of devices (or more)

        """
        self.logger.debug("WARNING - DO NOT USE THIS METHOD OR ANYOTHER LOCATED IN ZONES.PY")
        self.logger.debug("waiting for %d devices to appear in topology for %s" % (num_expected_devices, zp))
        try:
            wait_until_true(lambda: self.get_num_devices_in_topology(zp) >= num_expected_devices, timeout_seconds=timeout_seconds)
        except TimeoutError as e:
            topology = zp.diag.get_topology()
            topology_devices = topology.ZonePlayer
            self.logger.error("caught TimeoutError waiting for %d devices to appear in topology for %s (found %d)" % (num_expected_devices, zp, len(topology_devices)))
            for d in topology_devices:
                self.logger.debug(str(d))
            raise e

[docs]    def setup_device_configuration(self, chmap_config, zones):
        """
        Setup a satellite or bonded configuration for a test

        Arguments:
        chmap_config -- list specifying the satellite/bonded configuration

                    list contents:
                        string, 'model description': description of bonded/satellite configuration
                        string, 'playback device': model number of device to playback from during test
                        string, 'monitor device': model number of device to monitor during test
                        list of tuples, 'channel map specification':
                            string, 'channel': desired channel
                            string, 'device model': model of device to use in this channel

                    ex:
                        ["S9 + Sub + 2 S3s", "S9","S9",["SW","Sub"],["LR","S3"],["RR","S3"]] ]
                        S9 in 5.1 config, S9 as both playback and monitoring device

        zones -- list of SonosZoneComponents to create bonded/satellite configuration from

        Returns:
        tuple --
            string, model description
            SonosZoneComponent, playback device
            SonosZoneComponent, monitoring device

        """
        self.logger.debug("WARNING - DO NOT USE THIS METHOD OR ANYOTHER LOCATED IN ZONES.PY")
        model_description, playback_model, monitor_model, satellite_data = self.parse_bonded_config(chmap_config)
        playback_zp = self.get_zone_from_model(playback_model, zones, omit_channel_map_devices=True)
        monitor_zp = self.get_zone_from_model(monitor_model, zones, omit_channel_map_devices=True)
        wait_until_true(lambda: hasattr(playback_zp, "DeviceProperties"))
        chmap_data = self.build_device_channel_map(playback_zp, satellite_data)
        self.logger.info("got channel map: %s" % chmap_data)
        if isinstance(chmap_data, list):
            self.setup_htsatellite_config(playback_zp, chmap_data)
        else:
            self.logger.info("calling AddBondedZones(%s) on %s" % (chmap_data, playback_zp))
            self.wait_for_devices_to_have_same_number_of_topology_devices(zones)
            self.util.call_and_retry(lambda: playback_zp.DeviceProperties.AddBondedZones(chmap_data),
                                        allowable_upnp_exceptions=[1034],
                                        error_callable=lambda: time.sleep(15))
        return (model_description, playback_zp, monitor_zp)



[docs]    def is_zone_in_topologies(self, test_zp, zones):
        """
        Is this zone UUID listed in the status/topology page of
        the other specified zones?

        Arguments:
        test_zp -- SonosZoneComponent, zone whose UUID will be searched for
        zones -- list of SonosZoneComponents, zones whose status/topology
                 pages will be searched

        Returns:
        boolean -- is test_zp in topology for all zones?

        """
        self.logger.debug("WARNING - DO NOT USE THIS METHOD OR ANYOTHER LOCATED IN ZONES.PY")
        self.logger.info("checking %d zones topology for %s" %
                         (len(zones), test_zp.raw_udn))
        is_in_topology = False
        topology_results = []
        for zp in zones:
            topology = zp.diag.get_topology()
            topology_data = topology.ZonePlayer
            topology_results.append(True in \
                [ test_zp.raw_udn == group.uuid for group in topology_data ])
        is_in_topology = False not in topology_results
        return is_in_topology

        """

        self.logger.debug("WARNING - DO NOT USE THIS METHOD OR ANYOTHER LOCATED IN ZONES.PY")

                            iteration_delay=1)
