[docs]    def load_playback_resource(self, zp, resource_uri, resource_metadata='', timeout_seconds=120):
        """
        Clear the current AVTransportURI of zp, and set the AVTransportURI the specified resource

        If multiple resource URIs are specified, stop after the first successful URI

        Arguments:
        zp -- SonosZoneComponent
        resource_uri -- string, URI for resource
                        list of strings, URIs for resource

        Keyword Arguments:
        resource_metadata -- string, defualt=None, metadata for resource
        
        """
        self.logger.info("loading playback resource '%s' (md=%s) on %s" % (resource_uri, resource_metadata, zp.modelNumber))
        is_loaded = False
        if not isinstance(resource_uri, list):
            resource_uri = [resource_uri]
        try:
            self.clear_play_state(zp)
        except UPnPError:
            self.exceptions.log_upnp_error(zp)
        except TimeoutError:
            self.logger.error("caught TimeoutError waiting for AVTransport.BecomeCoordinatorOfStandaloneGroup to clear AVTransportURI")
        self.logger.debug("setting %s AVTransportURI to: %s" % (zp.modelNumber, resource_uri))
        for uri in resource_uri:
            try:
                zp.AVTransport.SetAVTransportURI(uri, resource_metadata)
                wait_until_true(lambda: uri in zp.AVTransport.GetMediaInfo().get('CurrentURI'), timeout_seconds=timeout_seconds)
                is_loaded = True
                self.logger.info("loaded resource URI '%s' on %s" % (uri, zp.modelNumber))
                break
            except UPnPError:
                self.exceptions.log_upnp_error(zp)
            except TimeoutError:
                self.logger.error("caught TimeoutError waiting for '%s' to appear in get_avtransport_uri() on %s" % (uri, zp.modelNumber))
        return is_loaded



[docs]    def start_playback(self, zp):
        """
        Start playback of the current AVTransportURI on this device

        Arguments:
        zp -- SonosZoneComponent

        Returns:
        boolean -- is device playing?

        """
        is_playing = False
        self.logger.info("starting playback on %s" % zp.modelNumber)
        try:
            zp.AVTransport.Play_and_wait()
            is_playing = True
        except UPnPError:
            self.exceptions.log_upnp_error(zp)
        except TimeoutError:
            self.logger.error("caught TimeoutError waiting for transport state to become PLAYING on %s" % zp.modelNumber)
        return is_playing

        """
        Stop playback on this device, and clear the current AVTransportURI

        Arguments:
        zp -- SonosZoneComponent

        """
        self.logger.debug("stopping playback on %s" % zp.modelNumber)
        try:
            self.set_playmode_normal(zp)
            self.clear_play_state(zp)
            wait_until_true(lambda: self.is_zone_playing(zp) == False)
        except UPnPError:
            self.exceptions.log_upnp_error(zp)
        except TimeoutError:
            self.logger.error("caught TimeoutError waiting for %s AVTransportURI to be empty" % (zp.modelNumber))



[docs]    def wait_for_track_to_end(self, zp, max_wait_seconds, iteration_delay=0.5):
        """
        Wait for transport state on device to not be PLAYING

        Arguments:
        zp -- SonosZoneComponent
        max_wait_seconds -- integer, maximum number of seconds to wait for playback to stop

        Keyword Arguments:
        iteration_delay -- float, delay between each poll to see if track has ended

        Returns:
        Boolean -- device stopped playback

        """
        self.logger.info("waiting for transport state to not be PLAYING on %s (max wait=%d)" % (zp.modelNumber, max_wait_seconds))
        playback_stopped = False
        try:
            wait_until_true(lambda: zp.AVTransport.GetTransportInfo().get('CurrentTransportState') != 'PLAYING',
                            timeout_seconds=max_wait_seconds,
                            iteration_delay=iteration_delay)
            playback_stopped = True
        except TimeoutError:
            self.logger.error("caught TimeoutError waiting for %s transport state to not be PLAYING" % zp.modelNumber)
        except UPnPError:
            self.exceptions.log_upnp_error(zp)
        return playback_stopped

[docs]    def seek_to_position_before_track_end(self, zp, num_seconds):
        """
        Seek to a specified number of seconds before the playing track ends

        Arguments:
        zp -- SonosZoneComponent
        num_seconds -- integer, number of seconds before track ends to seek to

        """
        num_seconds = int(num_seconds)
        track_duration = zp.AVTransport.get_current_track_duration()
        track_duration_seconds = self.util.reltime_str_to_seconds(track_duration)
        desired_track_position = track_duration_seconds - num_seconds
        desired_track_position_str = self.util.seconds_to_reltime_str(desired_track_position)
        zp.AVTransport.Seek('REL_TIME', desired_track_position_str)

[docs]    def wait_for_track_transition(self, zp, max_wait_seconds=300):
        """
        Wait for the currently playing track to transition to the next track on this device

        Arguments:
        zp -- SonosZoneComponent

        Keyword Arguments:
        max_wait_seconds -- integer, default=300, max number of seconds to wait for track transition

        """
        self.logger.info("waiting for track to transition to next track")
        if self.is_playing(zp):
            current_track = self.get_current_track_number(zp)
            wait_until_true(lambda: self.get_current_track_number(zp) != current_track, 
                                    timeout_seconds=max_wait_seconds,
                                    iteration_delay=0.5)
        else:
            self.logger.error("device '%s' is not currently playing!" % zp.modelNumber)

[docs]    def clear_play_state(self, zp):
        """
        Make the specified device become coordinator of a standalone group, and clear the device queue

        Arguments:
        zp -- SonosZoneComponent

        """
        try:
            if self.get_avtransport_uri(zp):
                self.clear_avtransport_uri(zp)
                wait_until_true(lambda: zp.GroupManagement.is_group_coordinator_local() == True)
            if zp.ContentDirectory.get_num_tracks_in_queue() > 0:
                zp.AVTransport.clear_queue_and_wait()
            if not zp.GroupManagement.is_group_coordinator_local():
                zp.AVTransport.BecomeCoordinatorOfStandaloneGroup()
                wait_until_true(lambda: zp.GroupManagement.is_group_coordinator_local() or \
                                        not zp.AVTransport.GetMediaInfo().get("CurrentURI"),
                                        iteration_delay=1)
        except UPnPError:
            self.exceptions.log_upnp_error(zp)
        except TimeoutError:
            self.logger.error("caught TimeoutError clearing play state for %s" % zp.serialNumber)



[docs]    def clear_play_states(self, zones):
        """
        Clear the play state for multiple devices

        Arguments:
        zones -- list of SonosZoneComponents

        """
        for zp in zones:
            if not hasattr(zp, "AVTransport") or zp.is_dock() or (zp.is_sub() and zp.is_bonded()):
                self.logger.info("not clearing play state on dock, bonded Sub, or unsupported devices: {}".format(zp.serialNumber))
                continue
            self.clear_play_state(zp)
