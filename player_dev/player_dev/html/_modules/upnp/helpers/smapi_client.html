    def __init__(self, svcName, wsdl, insecureURI, secureURI, flags, authModel):
        self.svcName = svcName
        self.insecureURI = insecureURI
        self.secureURI = secureURI
        self.authModel = authModel
        self.flags = flags
        self.client = Client(wsdl)
        transport = HttpTransport()
        transport.options.compression='yes'
        transport.options.timeout = SMAPIClient.SONOS_SVC_CONNECT_TIMEOUT / 1000
        self.client.set_options(location=self.insecureURI, transport=transport)
        self.credentials = self.client.factory.create('credentials')
        self.credentials.deviceId = '00-00-00-00-00-00:Z'
        self.credentials.deviceProvider = 'Sonos'
        if self.authModel == 'Stateless':
            self.credentials.login = self.client.factory.create('login')
        elif self.authModel == 'DeviceLink':
            self.credentials.loginToken = self.client.factory.create('loginToken')
        self.stats = {}

    def __call(self, func, *args):
        if func.method.name in self.secureMethods:
            self.client.set_options(location=self.secureURI)
        else:
            self.client.set_options(location=self.insecureURI)
        logger.debug(u'{0}{1}'.format(func.method.name, args))
        stats = None
        if func.method.name in self.stats:
            stats = self.stats[func.method.name]
        else:
            stats = CallTimeStats(func.method.name)
        stats.start()
        ret = func(*args)
        stats.stop()
        self.stats[func.method.name] = stats
        return ret

    def __withLoginRetry(self, func, *args):
        ret = None
        try:
            ret = self.__call(func, *args)
        except WebFault, f:
            if str.find(str(f.fault.faultstring), 'Client.SessionIdInvalid') >= 0 \
            or str.find(str(f.fault.faultcode), 'Client.SessionIdInvalid') >= 0:
                self.login()
                ret = self.__call(func, *args)
            # TODO: this could be more elegant
            elif self.authModel == 'DeviceLink':
                self.login()
                ret = self.__call(func, *args)
            else:
                raise f
        return ret

[docs]    def login(self):
        if self.authModel == 'UserId':
            self.credentials.sessionId = self.getSessionId(self.username, self.password)
            self.client.set_options(soapheaders=self.credentials)
        elif self.authModel == 'Stateless':
            self.credentials.login.username = self.username
            self.credentials.login.password = self.password
            self.client.set_options(soapheaders=self.credentials)
        elif self.authModel == 'DeviceLink':
            self.credentials.loginToken.token = self.auth_token
            self.credentials.loginToken.key = self.private_key
            self.credentials.loginToken.householdId = self.household_id
            self.client.set_options(soapheaders=self.credentials)
        return self.credentials

    def __validateList(self, res, reqIndex, reqCount):
        def warn(warnStr, info):
            if warnStr is None:
                warnStr = 'Invalid mediaList [request:(index={0},count={1}) reply:(index={2},count={3},total={4},len(Items)={5})]'\
                .format(reqIndex, reqCount, res.index, res.count, res.total, len(res.Items))
            warnStr = warnStr + ', ' + info
            return warnStr
        warnStr = None
        if 'count' in res and res.count == 0:
            return None
        if len(res.Items) == 0:
            warnStr = warn(warnStr, 'empty list')
        if res.count < 0 or res.index < 0 or res.total < 0:
            warnStr = warn(warnStr, 'negative value(s)')
        if res.count != len(res.Items):
            warnStr = warn(warnStr, 'count != actual')
        if res.index+res.count > res.total:
            warnStr = warn(warnStr, 'segment length exceeds total')
        if res.count > reqCount:
            warnStr = warn(warnStr, 'result count exceeds request')
        if res.index != reqIndex:
            warnStr = warn(warnStr, 'result index different than request')
        if warnStr is not None:
            raise Exception(warnStr)
        return None
