    def wait_for_transport_state_change(avts, max_wait, user_defined_start_states=None):
        """
        Wait max_wait for AVTransport to change from user_defined_start_states.

        :param avts: AVTransport service object
        :type avts: :class:`~sonos.services.avtransport.AVTransportService`

        :param max_wait: Maximum wait time for the transition change to occur in seconds
        :type max_wait: :int:

        :param user_defined_start_states: Start startes allowed to check for transition changes.  Any state not matching
                                          the start states will be considered a transition.
        :type user_defined_start_states: :list:

        :return: Tuple of (initial transport start states, new transport state)
        :rtype: :tuple:
        """
        if user_defined_start_states:
            initial_tss = list(set(user_defined_start_states))
        else:
            initial_tss = [avts.get_state_variable('TransportState', wait=False)]
        new_ts = None
        start = end = datetime.datetime.now()
        while ((new_ts.upper() in map(lambda x: x.upper(), initial_tss)) if new_ts else True) and (
                    end - start).total_seconds() < max_wait:
            new_ts = avts.get_state_variable('TransportState', wait=True,
                                             timeout=max_wait - (end - start).total_seconds())
            end = datetime.datetime.now()
        return initial_tss, new_ts



[docs]    def start(self, max_wait_for_transition, min_duration_before_transition, user_defined_start_states=None):
        """
        Start the timer and wait for max_wait_for_transition for TransportState transition.
        If user passed in user_defined_start_state, the state_changed function will it as
        the initial TransportState for comparison with the TranpsortState when timer expired.

        :param max_wait_for_transition: Max time to wait for TransportState transition.
        :type max_wait_for_transition: :obj:`float`

        :param min_duration_before_transition: Minimum duration we expect the AVTransportState to NOT change
        :type min_duration_before_transition: :obj:`float`

        :param user_defined_start_states: User-defined start state for state_changed comparison.
        :type user_defined_start_states: :obj:`list`

        :return: None
        """
        if max_wait_for_transition < min_duration_before_transition:
            raise ValueError(
                "max_wait_for_transition {} is less than min_duration_before_transition {} -> timer not started".format(
                    max_wait_for_transition,
                    min_duration_before_transition))

        self.__expiry = False
        self.__elapsed = None

        # If timer expires (i.e. after min_duration_before_transition), expiry will be set to True
        self.__timer = threading.Timer(min_duration_before_transition, self.__timer_expiry)
        self.__timer.start()

        start = datetime.datetime.now()
        self.__start_state, self.__stop_state = \
            self.wait_for_transport_state_change(self.__avts,
                                                 max_wait=max_wait_for_transition,
                                                 user_defined_start_states=user_defined_start_states)

        # If transport state did not change within max_wait, timer would not be canceled and expiry would be set
        # Hence asseting expiry asserts that "transport state did not change within max_wait"

        # If transport state did change within max_wait, timer would have been canceled and expiry would not be set
        # Hence asseting non-expiry asserts that "transport state did change within max_wait"
        self.__elapsed = (datetime.datetime.now() - start).total_seconds()
        self.__timer.cancel()
        self.__timer.join()


    def __timer_expiry(self):
        self.__expiry = True    # Means timer wasn't canceled.

    @property
    def expiry(self):
        if self.__expiry is None:
            raise RuntimeWarning("Timer Not Started")
        return self.__expiry

    @property
    def state_changed(self):
        if None in (self.__start_state, self.__stop_state):
            raise RuntimeWarning("Timer Not Started")
        return self.__stop_state.upper() not in map(lambda x: x.upper(), self.__start_state)

    @property
    def elapsed(self):
        if self.__elapsed is None:
            raise RuntimeWarning("Timer Not Started")
        return self.__elapsed




[docs]def start_playback_with_source_selector(
        zp,
        (source, source_meta)=('queue', 1),
        timeout_seconds=3.0
):
    """
    Set AVTransport URI, metadata and start playback on zp

    :param zp: Sonos Zone Component object
    :type zp: :class:`~sonos.client.internal.SonosZoneComponent`

    :param source: Source - 'queue' or 'spdif'
    :type source: :obj:`str`

    :param source_meta: Source-specific metadata
    :type source_meta: Depends on source.  If queue, will be :int:.

    :param timeout_seconds: Timeout in seconds to be passed to Play_and_wait().
    :type timeout_seconds: :float:

    :return: Elapsed time of when is_playing() condition is met.
    :rtype: :obj:`float`
    """
    if not hasattr(zp, "AVTransport"):
        raise RuntimeError("{} has no 'AVTransport' attribute".format(zp))
    if source == 'queue':
        zp.AVTransport.use_queue(queue_id=0)
        zp.AVTransport.set_queue_position(source_meta, instance_id=0)
    elif source == 'spdif':
        zp.AVTransport.use_spdif()

    return zp.AVTransport.Play_and_wait(timeout_seconds=timeout_seconds)




[docs]def add_files_as_uri_to_queue(zp, files, uri_base_url=None, queue_instance_id=0):
    """
    Add music tracks as URIs to zp's music queue

    :param zp: Sonos Zone Component object
    :type zp: :class:`~sonos.client.internal.SonosZoneComponent`

    :param files: List of media tracks
    :type files: :obj: `list`

    :param uri_base_url: Base URL.  If list, must match length of files
    :type uri_base_url: :obj: `str` or :list:/:set:/:tuple: of :str:

    :param queue_instance_id: Queue instance id
    :type queue_instance_id: :int:

    :return: URIs list
    :rtype: :list: of :str:
    """
    track_uris = []
    if not hasattr(zp, "AVTransport"):
        raise RuntimeError("{} has no 'AVTransport' attribute".format(zp))
    if isinstance(uri_base_url, (list, set, tuple)):
        track_uris = map(lambda x: "/".join(x),
                         zip(uri_base_url, [urllib.quote(f) for f in files]))
    elif isinstance(uri_base_url, (str, unicode)):
        track_uris = ["{}/{}".format(uri_base_url, urllib.quote(f)) for f in
                      files]
    elif uri_base_url is None:
        track_uris = files
    [zp.AVTransport.add_to_queue(track,
                                 instance_id=queue_instance_id)
     for track in track_uris]
    return track_uris




[docs]def start_playback_and_assert_playback_duration(
        avts,
        zp,
        (media_source, media_source_meta),
        max_wait_for_transition,
        min_duration_before_transition,
        expected_expiry,
        expected_state_changed,
        assert_media_source_meta_and_duration_only=False
):
    """
    :param avts: AVTransportStateTransitionTimer service object
    :type avts: :class:`~helpers.avtransport.AVTransportStateTransitionTimer`

    :param media_source: Valid media sources for
                            :method:`~upnp.helpers.avtransport.start_playback_with_source_selector`
    :type media_source: :obj:`str`

    :param media_source_meta: Valid media sources meta-data for
                                :method:`~upnp.helpers.avtransport.start_playback_with_source_selector`
    :type media_source_meta: Depends on media_source.

    :param zp: Sonos Zone Component object
    :type zp: :class:`~sonos.client.internal.SonosZoneComponent`

    :param max_wait_for_transition: Max time to wait for TransportState transition.
    :type max_wait_for_transition: :obj:`float`

    :param min_duration_before_transition: Minimum duration we expect the AVTransportState to NOT change
    :type min_duration_before_transition: :obj:`float`

    :param expected_expiry: Is timer expiry expected
    :type expected_expiry: :bool:

    :param expected_state_changed: Is transport state changed expected
    :type expected_state_changed: :bool:

    :param assert_media_source_meta_and_duration_only: No need to start playback with source selector.
    :type assert_media_source_meta_and_duration_only: :bool:

    :return: Elapsed time from function entry to function exit
    (i.e. prolouge elapsed plus timer elapsed).
    :rtype: :float:
    """
    initial_playback_elapsed = 0.0
    if not assert_media_source_meta_and_duration_only:
        # initial_playback_elapsed is approximation of when Play() was
        # triggered to when is_playing() is True.
        try:
            initial_playback_elapsed = start_playback_with_source_selector(
                zp,
                (media_source, media_source_meta),
                timeout_seconds=3.0
            )
        except TimeoutError as te:
            err = "{} did not start {} playback, exception: {}".format(
                zp,
                media_source,
                te
            )
            logger.error(err)
            raise TimeoutError(err)

    logger.info("initial_playback_elapsed: {}".format(
        initial_playback_elapsed)
    )

    media_source_check_start_time = datetime.datetime.now()
    if media_source == 'spdif':
        wait_until_true(
            lambda: zp.diag.get_toslink_stream_type() ==
                    media_source_meta['StreamInfo'],
            iteration_delay=0.5,
            timeout_seconds=3.0,
            reason="Expected SPDIF StreamInfo: {}, "
                   "Actual SPDIF StreamInfo: {}".format(
                media_source_meta['StreamInfo'],
                zp.diag.get_toslink_stream_type()))
        wait_until_true(
            lambda: zp.diag.get_toslink_sample_rate() ==
                    media_source_meta['SampleRate'],
            iteration_delay=0.5,
            timeout_seconds=3.0,
            reason="Expected SampleRate: {}, "
                   "Actual SPDIF SampleRate: {}".format(
                media_source_meta['SampleRate'],
                zp.diag.get_toslink_sample_rate()))

    media_source_check_elapsed_secs = \
        (datetime.datetime.now() -
         media_source_check_start_time).total_seconds()

    logger.info("media_source_check_elapsed_secs: {}".format(
        media_source_check_elapsed_secs)
    )

    # Buffer time for setting up timer for assert_playback_duration
    assertion_duration_buffer_secs = 0.5

    if not (min_duration_before_transition - initial_playback_elapsed >=
            media_source_check_elapsed_secs + assertion_duration_buffer_secs):
        err = "Failed to assert min_duration_before_transition ({} secs)"
        " as media_source_check_elapsed_secs ({} secs) "
        "doesn't have enough buffer ({} secs) after "
        "offsetting initial_playback_elapsed ({} secs).".format(
            min_duration_before_transition,
            media_source_check_elapsed_secs,
            assertion_duration_buffer_secs,
            initial_playback_elapsed
        )
        logger.error(err)
        raise TimeoutError(err)

    if not (max_wait_for_transition - initial_playback_elapsed >=
            media_source_check_elapsed_secs + assertion_duration_buffer_secs):
        err = "Failed to assert max_wait_for_transition ({} secs)"
        " as media_source_check_elapsed_secs ({} secs) "
        "doesn't have enough buffer ({} secs) after "
        "offsetting initial_playback_elapsed ({} secs).".format(
            max_wait_for_transition,
            media_source_check_elapsed_secs,
            assertion_duration_buffer_secs,
            initial_playback_elapsed)
        logger.error(err)
        raise TimeoutError(err)

    effective_min_duration_before_transition =\
        min_duration_before_transition -\
        initial_playback_elapsed -\
        media_source_check_elapsed_secs

    effective_max_wait_for_transition = max_wait_for_transition - \
                                        initial_playback_elapsed - \
                                        media_source_check_elapsed_secs

    logger.info("effective_min_duration_before_transition: {}, "
                 "effective_max_wait_for_transition: {}".format(
        effective_min_duration_before_transition,
        effective_max_wait_for_transition)
    )
    return initial_playback_elapsed + \
           media_source_check_elapsed_secs + \
           assert_playback_duration(avts, zp,
                                    effective_max_wait_for_transition,
                                    effective_min_duration_before_transition,
                                    expected_expiry,
                                    expected_state_changed)




[docs]def assert_playback_duration(avts, zp,
                             max_wait_for_transition,
                             min_duration_before_transition,
                             expected_expiry,
                             expected_state_changed):
    """
    :param avts: AVTransportStateTransitionTimer service object
    :type avts: :class:`~helpers.avtransport.AVTransportStateTransitionTimer`

    :param zp: Sonos Zone Component object
    :type zp: :class:`~sonos.client.internal.SonosZoneComponent`

    :param max_wait_for_transition: Max time to wait for TransportState transition.
    :type max_wait_for_transition: :obj:`float`

    :param min_duration_before_transition: Minimum duration we expect the AVTransportState to NOT change
    :type min_duration_before_transition: :obj:`float`

    :param expected_expiry: Is timer expiry expected
    :type expected_expiry: :bool:

    :param expected_state_changed: Is transport state changed expected
    :type expected_state_changed: :bool:

    :return: Timer elapsed duration
    :rtype: :float:
    """
    avts.start(max_wait_for_transition, min_duration_before_transition,
               user_defined_start_states=["PLAYING", "TRANSITIONING"])
    assert avts.expiry == expected_expiry, \
        '{} {} in "Playing" state for at least {} secs (elapsed: {})'.format(
            zp,
            "did not stay" if expected_expiry else "stayed",
            min_duration_before_transition,
            avts.elapsed
        )
    assert avts.state_changed == expected_state_changed, \
        "{}'s TransportState {} after {} secs (max_wait_for_transition:" \
        " {}.)".format(
            zp,
            "did not change as expected" if expected_state_changed
            else "changed unexpectedly",
            avts.elapsed,
            max_wait_for_transition,
        )
    return avts.elapsed
