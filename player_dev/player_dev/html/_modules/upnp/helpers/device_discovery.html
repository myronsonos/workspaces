    """
    This class contains a single device discovery message 
    """
    def __init__(self, host, port, data):
        self.host = host
        self.port = port
        self.data = data
        self.isParsed = False
        self.res_cachectl = None
        self.res_ext = None
        self.res_location = None
        self.res_server = None
        self.res_st = None
        self.res_usn = None
    
    def _parse(self):
        if self.isParsed:
            return
        
        # HTTP response fields for a given host in the test
        cmd, headers = utils.parse_http_response(self.data)
        if 'cache-control' in headers:
            self.res_cachectl = headers['cache-control']
        if 'ext' in headers:
            self.res_ext = headers['ext']
        if 'location' in headers:
            self.res_location = headers['location']
        if 'server' in headers:
            self.res_server = headers['server']
        if 'st' in headers:
            self.res_st = headers['st']
        if 'usn' in headers:
            self.res_usn = headers['usn']
        self.isParsed = True

        # clone the header response for later processing
        self.allHeaders = dict(headers)
        

[docs]    def from_udn(self, udn):
        self._parse() # parses only if not yet parsed
        
        if self.res_usn:
            # All messages from the given UUID start with the UUID
            return self.res_usn.startswith(udn)
        else:
            return False

        

[docs]    def from_root_udn(self, udn):
        self._parse() # parses only if not yet parsed
        
        if self.res_usn:
            return (self.res_usn == udn) or (self.res_usn == ("%s::upnp:rootdevice" % udn)) or \
                    self.res_usn.startswith("%s::urn:schemas-upnp-org:device:ZonePlayer:")
        else:
            return False

        

[docs]    def from_root_udn_exactly(self, udn):
        self._parse() # parses only if not yet parsed
        
        if self.res_usn:
            return self.res_usn == udn
        else:
            return False

[docs]    def get_msgs(self, udn):
        if udn in self.res_discovered_devices: 
            host = self.res_discovered_devices[udn]
            if host in self.msgs:
                # filter out all messages from the given UDN
                return filter(lambda msg: msg.from_udn(udn), self.msgs[host])
        return None

[docs]    def report_data(self, host, port, data):
        if host not in self.msgs:
            self.msgs[host] = []

        # add new device discovery response from this host        
        self.msgs[host].append(DevDiscMessage(host, port, data))    

        

[docs]    def report_root_device(self, host, udn):
        isUnknown = True 
        if host in self.msgs:
            for msg in self.msgs[host]:
                if msg.from_udn(udn):
                    self.res_discovered_devices[udn] = host
                    isUnknown = False
                    break
                
        if isUnknown:
            self.res_unknown_root_devices[host] = udn

    def _disconnect(self):
        logger.info("DeviceDiscovery._disconnect")
        if self.isListening:
            louie.disconnect(self.add_root_device, 
                             sonos.upnp.COHERENCE_ADD_DEVICE_SIGNAL, 
                             louie.Any)
            louie.disconnect(self.remove_root_device, 
                             sonos.upnp.COHERENCE_REMOVE_DEVICE_SIGNAL, 
                             louie.Any)
            louie.disconnect(self.ssdp_packet_received, 
                             COHERENCE_SSDP_PACKET_RECEIVED, 
                             louie.Any)
            louie.disconnect(self.ssdp_new_device, 
                             COHERENCE_SSDP_NEW_DEVICE, 
                             louie.Any)
            louie.disconnect(self.add_device, 
                             COHERENCE_DEVICE_DETECTED, 
                             louie.Any)
            louie.disconnect(self.add_service, 
                             COHERENCE_SERVICE_DETECTED, 
                             louie.Any)
            louie.disconnect(self.add_service_failed, 
                             COHERENCE_SERVICE_DETECTED_FAILED, 
                             louie.Any)
            louie.disconnect(self.group_coordinator_changed, 
                             COHERENCE_GROUP_COORDINATOR_CHANGED, 
                             louie.Any) # TODO: just GroupManagement
            louie.disconnect(self.invisible_changed, 
                             COHERENCE_INVISIBLE_CHANGED, 
                             louie.Any) # TODO: just DeviceProperties
            self.isListening = False

        # reset M-SEARCH params
        if self.origParams:
            coherence.upnp.core.msearch.MSearch.setParameters(*self.origParams)
            self.origParams = None


[docs]    def setup(self, searchTarget, extraHeaders=[]):
        ''' 
        Configures M-SEARCH and starts listening to Coherence signals
        '''
        
        # save original M-SEARCH parameters
        self.origParams = coherence.upnp.core.msearch.MSearch.getParameters()

        # set up M-SEARCH to mimic Sonos controllers
        coherence.upnp.core.msearch.MSearch.setParameters(searchTarget, maximumWaitTimeInSecs="1",
                                                          extraHeaders=extraHeaders)
        
        # set up our callbacks before starting Coherence
        if not self.isListening:
            louie.connect(self.add_root_device, 
                          sonos.upnp.COHERENCE_ADD_DEVICE_SIGNAL, 
                          louie.Any)
            louie.connect(self.remove_root_device, 
                          sonos.upnp.COHERENCE_REMOVE_DEVICE_SIGNAL, 
                          louie.Any)
            louie.connect(self.ssdp_packet_received, 
                          COHERENCE_SSDP_PACKET_RECEIVED, 
                          louie.Any)
            louie.connect(self.ssdp_new_device, 
                          COHERENCE_SSDP_NEW_DEVICE, 
                          louie.Any)
            louie.connect(self.add_device, 
                          COHERENCE_DEVICE_DETECTED, 
                          louie.Any)
            louie.connect(self.add_service, 
                          COHERENCE_SERVICE_DETECTED, 
                          louie.Any)
            louie.connect(self.add_service_failed,

            louie.connect(self.group_coordinator_changed,

    def format_serial_from_udn(udn):
        sn = None
        if udn and udn.startswith("uuid:RINCON_"):
            sn = udn.split("uuid:RINCON_")[1][:-5]

    def __is_udn_in_test(self, udn):
        found = filter(lambda s: s == self.format_serial_from_udn(udn),
