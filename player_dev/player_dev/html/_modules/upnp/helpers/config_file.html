[docs]    def make_config_file(self, invoke_path):
        """ Create a default config file path + name based on where the calling
            script was invoked from.  Simply pass sys.argv[0] as the argument.
        """

        # Got to have something in the invocation path
        if invoke_path == None or invoke_path == '':
            raise(AssertionError('ERROR - make_config_file() requires the invocation path as an argument.'))

        # Remove calling script name and leave just the invocation path
        caller = inspect.getmoduleinfo(invoke_path)[0] + \
                 inspect.getmoduleinfo(invoke_path)[1]
        just_path = re.sub(caller, '', invoke_path)

        # Default configfile subdirectory and name
        default_config_dir = 'config'
        default_config_name = 'tests.cfg'

        # Altogether now
        config_file = os.path.join(just_path, default_config_dir, default_config_name)

        return config_file
