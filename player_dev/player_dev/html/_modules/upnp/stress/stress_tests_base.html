[docs]    def read_in_error_config_file(self):
        """
        read in the error configuration file so that we know how many errors a
        given model should have seen in order to hit an error case (trend
        analysis). Also, initialize all of the counter values so that
        they're at 0 for each relevant model and module.
        """
        filename = os.path.join(os.path.dirname(__file__),
                                "error_threshold_config_jsons/config.json")
        class_name = self.get_child_class_name()
        with open(filename) as fr:
            error_config_dict = json.loads(fr.read())
        if class_name not in error_config_dict:
            self.logger.info("Class name: {} not found in dictionary: "
                             "{}".format(class_name, error_config_dict))
            return
        self.threshold_dictionary_exists_for_class = True
        if 'max_underflows' not in error_config_dict[class_name]:
            self.logger.info("max_underflows field not found in dictionary"
                             ": {}".format(error_config_dict))
            self.threshold_dictionary_exists_for_class = False
            return
        self.error_thresholds = error_config_dict[class_name]['max_underflows']

        # initialize the counters for the appropriate model numbers and
        for mn in self.error_thresholds:
            if mn == 'default' or mn == 'groups':
                continue
            self.detected_errors_counters[mn] = {}
            for mod in self.error_thresholds[mn]:
                self.detected_errors_counters[mn][mod] = 0



[docs]    def get_error_threshold(self, model_number, module):
        """
        get the error threshold from the member dictionary if it exists for
        the given model number and module
        :param str model_number: a ZP model number
        :param str module: the module ('lla', 'chsnk', etc.)
        :return: the threshold (uint)
        """

        # First, check to see if there's a model number explicitly specified
        if model_number in self.error_thresholds:
            if module in self.error_thresholds[model_number]:
                self.logger.info(
                    "Error threshold explicitly specified for "
                    "{} - {}: {}".format(model, module, self.error_thresholds[
                        model_number][module
                    ]
                                                      ))
                return self.error_thresholds[model_number][module]

        # Next, check if there's a group of players that specifies the given
        if 'groups' in self.error_thresholds:
            for group in self.error_thresholds['groups']:
                if model_number in group['zps']:
                    if module in group:
                        self.logger.info("Error threshold for {} - {} "
                                         "specified in group".format(
                                            model_number, module
                                         ))
                        return group[module]

        # Finally, check for a default
        if 'default' in self.error_thresholds:
            if module in self.error_thresholds['default']:
                self.logger.info("Error threshold specified via default: "
                                 "{} - {}".format(model_number, module))
                return self.error_thresholds['default'][module]

        # if nothing is found, return 0
        return 0

[docs]    def get_test_devices(self):
        """
        Get test devices after factoring in command line params 'exclude_model'
        or 'allow_missing_models'
        :return test_devices: List of test devices- Sonos Zone Components
        :rtype test_devices: :`list`
        """
        test_devices = []
        available_zps = self.get_zp_with_models(TEST_MODELS)
        for model in TEST_MODELS:
            if model not in available_zps:
                if self.allow_missing_models:
                    self.warn("Model {} not available for test"
                              .format(model))
                else:
                    self.stop_suite("Test stopped due to missing Model: {}"
                                    .format(model))
        if self.allow_missing_models or self.exclude_models:
            for model in available_zps:
                if model in self.exclude_models:
                    self.logger.info("Excluding {} from test"
                                     .format(available_zps[model]))
                    continue
                test_devices.append(available_zps[model])
        else:
            [test_devices.append(available_zps[model])
             for model in available_zps]
        return test_devices

[docs]    def get_msg_module(self, msg):
        """
        get the module from the anacapa log message
        :param msg str: the log message string
        :return str: the module stripped of all other parts of the message
        """
        regex = r"[<][a-zA-Z_]*[,]"
        results = re.findall(regex, msg)
        if len(results) != 1:
            self.logger.info("Detected unexpected number of regex matches"
                             "in msg string: {} - found {}".format(
                                    msg, len(results)))
            return None
        # strip away the preceding '<' and the terminating ','
        return results[0][1:-1]



[docs]    def update_error_counters(self, mn, mod):
        """
        given the model number and the module under which the error
        occurred, update the error counter.
        if we've exceeded the max threshold of errors for the given model
        and module, then set error_detected to True
        the error counters
        :param mn: the model number of the zone player under test
        :param mod: the module (i.e. 'htap', 'chsnk', etc.) under which the
        error occurred
        """

        # if the calling class does not have a specified set of thresholds
        # in the config file, revert to flagging the test case for any
        # underflow errors.
        if not self.threshold_dictionary_exists_for_class:
            self.error_detected = True
            return

        # get the threshold value for the given model and module
        error_threshold = self.get_error_threshold(mn, mod)

        # create a dictionary for the model if it does not exist in the
        # detected errors counters
        if mn not in self.detected_errors_counters:
            self.detected_errors_counters[mn] = {}

        # if the module of the reported error is not in our set of thresholds,
        # we add it to the detected error counters
        if mod not in self.detected_errors_counters[mn]:
            self.detected_errors_counters[mn][mod] = 1

        # increment counters
        self.detected_errors_counters[mn][mod] += 1

        # check to see if the counters have exceeded the specified thresholds
        if self.detected_errors_counters[mn][mod] > error_threshold:
            self.error_detected = True



[docs]    def cb_detected_error(self, msg, zp, (m, expr, l, counters)):
        """
        Callback on detecting error string
        :param str msg: error message

        :type zp: :class:`~sonos.client.SonosZoneComponent`
        """
        self.error_msg = "Detected error: {} on <{}>".format(msg, zp)

        module = self.get_msg_module(msg)
        if module is not None:
            self.update_error_counters(zp.modelNumber, module)

        self.logger.critical(self.error_msg)
        self.error_counter += 1
        self.error_stats[zp].append(msg)
        if 'watchdog' in msg:
            self.collect_threadinfo(zp)
            self.continue_on_errors = False
            self.watchdog_fired = True

[docs]    def start_playback(self,
                       zp,
                       source=QUEUE,
                       play_mode='NORMAL',
                       crossfade=False,
                       qpos=1):
        """
        Start playback. For playing from queue, tracks should be added to
        queue already, provice position in queue to start playback from. For
        radio play from Tunein. For radio and TV avt is set in this function.
        :param zp: Sonos zone component
        :type zp: :`~sonos.client.SonosZoneComponent`
        :param str source: Source of music queue/share or radio or tv
        :param str play_mode: Play mode
        :param bool crossfade: Enable/disable crossfadel. Default False
        :param int qpos: Start position in queue for playback
        """
        self.logger.info("Starting playback from {} on {}"
                         .format(source, zp))
        if source in [QUEUE, SHARE]:
            zp.AVTransport.use_queue()
            self.logger.info(
                "Queue position: {} Playmode: {} Crossfade: {}"
                .format(qpos, play_mode, crossfade))
            zp.AVTransport.set_queue_position(qpos)
            if crossfade:
                zp.AVTransport.SetCrossfadeMode(crossfade)
            zp.AVTransport.SetPlayMode(play_mode)
        elif source == TV:
            uri = TV_URI.format(zp.raw_udn)
            zp.AVTransport.SetAVTransportURI_and_wait(uri, "")
        elif source == RADIO:
            zp.AVTransport.SetAVTransportURI_and_wait(RADIO_URI, RADIO_MD)
        try:
            zp.AVTransport.Play_and_wait()
        except TimeoutError:
            self.stop("Timed out waiting for <{}> to start play".format(zp))

[docs]    def get_track_length(self, track_name):
        """
        Get length of track.
        :param str file_name: DD track name
        :return int track_length
        """
        command = 'mediainfo'
        commandline = "{} '{}'".format(command, track_name)
        file_info = commands.getoutput(commandline).split('\n')
        info_dict = {}
        for line_info in file_info:
            param_value = [x.strip() for x in line_info.split(':')]
            if len(param_value) == 2:
                if param_value[0] not in info_dict:
                    info_dict[param_value[0]] = param_value[1]
        track_duration = info_dict['Duration']
        # parse hr, min seconds and ms from track duration, apply appropriate
        # multiplier to get total seconds.
        pattern = '(\d{1,3})\s*([a-z]{1,3})\s*(\d{1,3})\s*([a-z]{1,3})'
        result = re.search(pattern, track_duration)
        if result:
            if result.group(2) in ['hr', 'h']:
                major = 3600
            elif result.group(2) in ['min', 'mn']:
                major = 60
            elif result.group(2) == 's':
                major = 1
            elif result.group(2) == 'ms':
                major = 0
            if result.group(4) in ['min', 'mn']:
                minor = 60
            elif result.group(4) == 's':
                minor = 1
            elif result.group(4) == 'ms':
                minor = 0
            track_length = (int(result.group(1)) * major +
                            int(result.group(3)) * minor)
        return track_length
