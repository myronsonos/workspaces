[docs]    def generate_ht_devices_and_source(self):
        """
        Generate line in capable player and its playback source
        """
        for ht_master in self.ht_masters:
            for ht_sat in self.ht_sats:
                zp_l, zp_r = ht_sat
                for source in self.MUSIC_SOURCE:
                    zps = ht_master, zp_l, zp_r
                    yield (zps, source)

    def _play(self, zp, source):
        """
        Set AVTransport URI, metadata and start playback on zp
        :param zp: Sonos zone component on which error string was caught
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`
        :param str source: Source for audio on zp
        """
        self.logger.info("{}:Starting playback. Source: {}".format(zp, source))
        if source == SHARE:
            zp.AVTransport.use_queue()
            uri = zp.ContentDirectory.get_track_uri_for_filename(SHARE_TRACK)
            zp.AVTransport.add_to_queue(uri)
            # resetting to QUEUE as this is common for share or loading
            # some tracks from say spotify/internet - non programmed raddio
            source = QUEUE
        self.start_playback(zp, source=source)

    def _add_test_share(self, zp):
        """
        Add test share
        :param zp: Sonos zone component on which to perform share action
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`
        """
        (test_share, test_share_credentials) = (self.execution_configuration
                                                .get_share_by_key("cpu"))
        zp.ContentDirectory.remove_all_shares()
        res = zp.ContentDirectory.add_share(test_share,
                                            *test_share_credentials)
        self.verifyTrueOrStop(res, "Expects the test share to be added")

    def _add_sats(self, zp, chmap):
        """
        Do parallel addition of satellites
        :param zp: Sonos zone component on which to perform share action
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`
        :param str chmap: Channel map
        """
        self.logger.info("Performing satellite configuration. Chmap: {}"
                         .format(chmap))
        zp.DeviceProperties.AddHTSatellite(chmap)

    def _are_chmaps_consistent(self, ht_master, sat_l, sat_r, sub):
        """
        Are channel maps consistent accross the players ?
        :param zp: Sonos zone component on which to perform share action
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`
        :param zp: Sonos zone component ,ht_master
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`
        :param zp: Sonos zone component ,left surround
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`
        :param zp: Sonos zone component ,right surround
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`
        :param zp: Sonos zone component , Sub
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`
        :return bool:
        """
        master_state = ht_master.DeviceProperties.get_state_variable(
            'ChannelMapSet')
        sat_l_state = sat_l.DeviceProperties.get_state_variable(
            'ChannelMapSet')
        sat_r_state = sat_r.DeviceProperties.get_state_variable(
            'ChannelMapSet')
        sub_state = self.sub.DeviceProperties.get_state_variable(
            'ChannelMapSet')
        return master_state==sat_l_state==sat_r_state==sub_state

    def _run_parallel_htconfig_test(self, zps, source):
        """
        Run ht config tests using delayed (parallel) chmap setting
        :param zps: tuple of sonos zone components-htmaster, left and right sat.
        :type zps: :`tuple`
        :param str source: Type of source to be played on zps
        """
        (ht_master, sat_l, sat_r) = zps
        self.config_devices = [ht_master, sat_l, sat_r, self.sub]
        iteration = 1
        self._add_test_share(sat_l)
        while iteration <= self.num_iterations:
            self.logger.info("Iteration # {}/{}"
                             .format(iteration, self.num_iterations))
            # for zp in self.config_devices:
            #     self.setup_udp_logger(zp)
            self.verifyTrueOrStop(
                self._are_chmaps_consistent(ht_master, sat_l, sat_r, self.sub),
                "Chmaps should be consistent")
            #start playback on zps
            self._play(ht_master, TV)
            [self._play(zp, source) for zp in [sat_l, sat_r]]
            self.logger.info("Allow playback for {} seconds".format(PLAY_TIME))
            time.sleep(PLAY_TIME)
            # do parallel ht config with retries
            if sat_l.is_zp120():
                chmap = ZP120_CHMAP.format(ht_master.raw_udn,
                                           sat_l.raw_udn,
                                           self.sub.raw_udn)
            else:
                chmap = CHAMP.format(ht_master.raw_udn,sat_l.raw_udn,
                                     sat_r.raw_udn,self.sub.raw_udn)
            try:
                self._add_sats(ht_master, chmap)
                self.logger.info("Waiting for {} secs to confirm no errors"
                                 .format(WATCH_TIME))
                time.sleep(WATCH_TIME)
                self.verifyTrueOrFailCase(
                    ht_master.AVTransport.is_playing(),
                    "Expect {} to be playing".format(ht_master))
                for zp in self.config_devices:
                    self.verifyTrueOrFailCase(
                        zp.is_bonded(), "Expect {} to be bonded".format(zp))
            except Exception as e:
                # check if the exception is raised due to watchdog being fired
                try:
                    wait_until_true(
                                    lambda:self.watchdog_fired,
                                    iteration_delay=5, timeout_seconds=120)
                except TimeoutError:
                    # no watchdog fired. config failure
                    diagid = ht_master.ZoneGroupTopology.SubmitDiagnostics()
                    self.error_msg = ("HT Config Failure on {}. "
                        "Error: {} DIAG: {}.".format(ht_master, e, diagid))
                    self.error_detected = True
                else:
                    # watchdog fired, abort test by raising signal
                    # !!! note there is no cleaup done in this case
                    os.kill(os.getpid(), signal.SIGUSR1)
            if self.error_detected:
                if self.continue_on_errors:
                    self.fail(self.error_msg)
                else:
                    self.stop_suite(self.error_msg)
                    break
            for zp in self.config_devices:
                self.cleanup(zp)
            iteration += 1

    def test_parallel_htconfig(self, zps, source):
        """
        Perform delayed bonding and unbonding ITERATION times and verify
        no bonding failure, underflows or anacapa hang

        :type zps: :`tuple`

        """

        with MplayerAc3Playback(TV_TRACK_PATH):
            self.logger.info("Started playing TV source")
