[docs]    def build_audio_source_sequence(self, audio_sources):
        """
        Build non repeating sequence of audio sources using permutations of
        AUDIO_SOURCES
        :return source_list: Sequence of audio sources
        :rtype: :obj:`list`
        """
        # master list of all permutation sets, that is updated as repeating
        # sequences are being removed.
        master_list = []
        source_permutations = [source_pair for source_pair in
                               self.source_generator(audio_sources)]
        for source in audio_sources:
            source_permutations = self._remove_duplicate_sequences(
                source, source_permutations, master_list)
        # list of all sources from master permutations set list.
        source_sequence = []
        for i in range(len(master_list)):
            first, second = master_list[i - 1]
            source_sequence.append(first)
            if i < len(master_list):
                next_first, next_seconds = master_list[i]
                if second == next_first:
                    continue
                else:
                    source_sequence.append(second)
        return source_sequence

[docs]    def source_generator(self, source_list):
        """
        Generate all permutation of start and end audio source to be played
        from source_list
        :param list source_list: List of http uris
        :return: Tuple of start and end http uris
        :rtype: :obj:`tuple`
        """
        for start, end in itertools.permutations(source_list, 2):
            yield (start, end)


    def _remove_duplicate_sequences(self,
                                    source,
                                    source_permutations,
                                    master_list):
        """
        Remove duplicate sequence of sources from set of all permutations. Get
        all permutations that has starting source = given source, store that
        set, and go over entire permutation list, if there is any sequence that
        matches (example second source of a set and first source of next set),
        remove that list and update master list of all permutations.
        :param str source: audio source
        :param list source_permutations: List of tuples of source permutations
        :param list master_list: master list of all tuples of source set
        :return temp_list: list of tuples of sources
        :rtype: :obj:`list`
        """
        temp_list = []
        source_list = []
        delete_list = []
        for i in range(len(source_permutations)):
            start, end = source_permutations[i]
            if source == start:
                source_list.append(source_permutations[i])
                master_list.append(source_permutations[i])
        for i in range(len(source_list)):
            start, end = source_list[i]
            if i + 1 < len(source_list):
                next_start, next_end = source_list[i + 1]
                delete_set = (end, next_start)
                delete_list.append(delete_set)
        temp_list[:] = [a_set for a_set in source_permutations
                        if a_set not in master_list]
        for a_set in delete_list:
            if a_set in temp_list:
                temp_list.remove(a_set)
        return temp_list


[docs]    def cb_playing_event(self, msg, zp, (m, expr, l, counters)):
        """
        Callback on detecting tv or linein (sbc) or music playback
        :param str msg: anacapa trace line containing detected string
        :param zp: Sonos zone component from which to submit
        :type zp: :class:`~sonos.client.SonosZoneComponent`
        """
        self.logger.info("Detected playing: {} on {}".format(msg, zp))
        if re.search(PLAYING_CHSNK,msg):
            self.playing_chsnk = True
        elif re.search(PLAYING_SPDIF_FROM_NONE, msg):
            self.playing_spdif_from_none = True
        elif re.search(PLAYING_SPDIF_FROM_SPDIF, msg):
            self.playing_spdif_from_spdif = True
        elif re.search(PLAYING_SPDIF, msg):
            self.playing_spdif = True
        elif re.search(PLAYING_SBC,msg):
            self.playing_sbc[zp] = True
            self.logger.debug("Playing sbc values detected so far: {}"
                              .format(self.playing_sbc))
        elif re.search(PLAYING_MUSIC, msg):
            self.playing_music[zp] = True
            self.logger.debug("Playing music values detected so far: {}"
                              .format(self.playing_music))


    def _verify_all_zps_linked(self, gc, gms):
        """
        Verify expected number of gms linked to gc
        :param gc: Sonos zone component - Group coordinator
        :type gc: :class:`~sonos.client.SonosZoneComponent`
        :param list gms: List of sonos zone components which are group members
        :return: all zps linked ?
        :rtype: :obj:`bool`
        """
        num_gms = len(gms)
        try:
            wait_until_true(lambda: gc.ZoneGroupTopology.get_num_linked_zones()
                            == num_gms)
        except TimeoutError:
            expected_uuids = [zp.raw_udn for zp in self.test_devices]
            actual_uuids = gc.ZoneGroupTopology.get_zone_player_uuids_in_group()
            missing_uuids = [uuid for uuid in expected_uuids
                             if uuid not in actual_uuids]
            missing_zps = [self.get_testbed_device_by_raw_udn(raw_udn) for
                           raw_udn in missing_uuids]
            self.logger.critical(
                "Current UUID's in group: {}"
                .format(gc.ZoneGroupTopology.get_zone_player_uuids_in_group()))
            self.logger.critical("Missing UUIDs from group: <{}>"
                                 .format(missing_uuids))
            self.logger.critical("Missing ZPs from group: <{}>"
                                 .format(missing_zps))

            return False
        else:
            return True

    def _verify_all_playing(self, source, zp, gms, transition=None):
        """
        Verify devices are playing expected source
        :param str source: Type of source expected to be playing
        :param zp: Sonos zone component - Group coordinator
        :type zp: :class:`~sonos.client.SonosZoneComponent`
        :param list gms: List of sonos zone components which are group members
        :param str transition : Applicable to ht_master, did transition
            occur to chsnk or to spdif or no transition
        """
        if transition is not None:
            gc_result = self._verify_ht_master_playing(transition)
            gms_result = True
            if gms:
                devices = gms
                gms_result = self._verify_devices_playing(source,devices)
            if not (gc_result and gms_result):
                self.error_detected = True
                return
        else:
            devices = gms + [zp]
            result = self._verify_devices_playing(source, devices)
            if not result:
                self.error_detected = True
                return
        self.logger.log(AssertionLevel.PASS,
                        "Playing expected source: '{}' on all players"
                        .format(source))

    def _verify_devices_playing(self, source, devices):
        """
        Verify devices are playing expected source
        :param str source: Type of source expected to be playing
        :param list devices: List of sonos zone components
        :return: all devices played expected source ?
        :rtype: :obj:`bool`
        """
        try:
            wait_until_true(lambda: self._did_all_devices_play(source, devices),
                            iteration_delay=1,
                            timeout_seconds=SRC_DETECT_TIMEOUT)
        except TimeoutError:
            self.error_msg = 'Did not play {} on all devices'.format(source)
            for zp in devices:
                if source in [LINEIN, SPDIF]:
                    if not self.playing_sbc[zp]:
                        self.logger.critical(
                            "Did not detect sbc playing on <{}>".format(zp))
                if source in [SHARE, INTERNET_HLS, INTERNET]:
                    if not self.playing_music[zp]:
                        self.logger.critical(
                            "Did not detect music playing on <{}>".format(zp))
            return False
        return True

    def _verify_ht_master_playing(self, transition):
        """
        Verify playbar is playing expected source
        :param str transition: True if playbar source is chsnk
        :return: ht_master played from expected source ?
        :rtype: :obj:`bool`
        """
        try:
            wait_until_true(lambda: self._did_src_transition_occur(transition),
                            iteration_delay=1,
                            timeout_seconds=SRC_DETECT_TIMEOUT)
        except TimeoutError:
            self.error_msg = ('Did not play from {} on ht_master'
                              .format(transition))
            return False
        return True

    def _did_src_transition_occur(self, transition):
        """
        Check if expected transition occur on ht_master between chsnk and spdif
        :param str transition: did transition occur to chsnk or to spdif
        :return: ht_master played expected source ?
        :rtype: :obj:`bool`
        """
        if transition == SPDIF_TO_CHSNK:
            return self.playing_chsnk
        elif transition == CHSNK_TO_SPDIF:
            return self.playing_spdif
        elif transition == NONE_TO_SPDIF:
            return (self.playing_spdif_from_none or
                    self.playing_spdif_from_spdif)

    def _did_all_devices_play(self, source, devices):
        """
        Check if all gms played TV
        :param str source: expected source of audio data : linein, music or tv
        :param list devices: List of sonos zone component
        :return: Did all devices play expected source ?
        :rtype: :obj:`bool`
        """
        if source in [LINEIN, SPDIF]:
            self.logger.info("Played {} state values: {}"
                             .format(source, self.playing_sbc))
            return all([self.playing_sbc[zp] for zp in devices])
        if source in [SHARE, INTERNET, INTERNET_HLS]:
            self.logger.info("Played {} state values: {}"
                             .format(source, self.playing_music))
            return all([self.playing_music[zp] for zp in devices])

    def _play(self, zp, play_from=None):
        """
        Play 'play_from' source on zp
        :param zp: Sonos zone component - on which to play source
        :type zp: :class:`~sonos.client.SonosZoneComponent`
        :param str play_from: Audio source to play from
        """
        if play_from == SHARE:
            uri = SHARE_TRACK_URI.format(zp.raw_udn)
            zp.AVTransport.use_queue()
            zp.AVTransport.set_queue_position(SHARE_TRACK_POSITION)
        elif play_from == INTERNET:
            zp.AVTransport.use_queue()
            uri = INTERNET_TRACK_URI.format(zp.raw_udn)
            zp.AVTransport.set_queue_position(INTERNET_TRACK_POSITION)
        elif play_from == SPDIF:
            uri = SPDIF_URI.format(zp.raw_udn)
            zp.AVTransport.SetAVTransportURI(uri, "")
        elif play_from == LINEIN:
            uri = LINEIN_URI.format(self.line_in_device.raw_udn)
            zp.AVTransport.SetAVTransportURI(uri, "")
        elif play_from == INTERNET_HLS:
            uri = HLS_URI.format(self.websrv.host(), MASTER)
            zp.AVTransport.SetAVTransportURI(uri, "")
        zp.AVTransport.Play_and_wait()

    def _run_transitions_test(self, zp, gms):
        """
        Do the transition test
        :param zp: Sonos zone component - Group coordinator
        :type zp: :class:`~sonos.client.SonosZoneComponent`
        :param list gms: List of sonos zone components which are group members
        """
        i = 0
        while i < self.num_iterations:
            self.logger.info("Iteration #: {}".format(i))
            for j in range(len(self.source_sequence)):
                transition = None
                source = self.source_sequence[j]
                if j > 0:
                    previous_source = self.source_sequence[j-1]
                else:
                    previous_source = 'idle'
                if j+1 < len(self.source_sequence):
                    next_source = self.source_sequence[j+1]
                else:
                    next_source = None
                self.logger.info("Switching source from '{}' to '{}'"
                                 .format(previous_source, source))
                self._play(zp, play_from=source)
                if source == INTERNET_HLS:
                    if next_source == SPDIF:
                        # SWPBL-56559 It was noticed that waiting 60-80
                        # seconds when playing hls and then transitioning to TV
                        # caused watchdog to fire.
                        time.sleep(random.randint(*HLS_TO_TV_INTERVAL_RANGE))
                    else:
                        time.sleep(PLAYBACK_TIME)
                if zp.is_ht_master_capable:
                    if source == SPDIF and previous_source == IDLE:
                        transition = NONE_TO_SPDIF
                    elif source == SPDIF and previous_source not in [IDLE,
                                                                     SPDIF]:
                        transition = CHSNK_TO_SPDIF
                    elif source != SPDIF and previous_source == SPDIF:
                        transition = SPDIF_TO_CHSNK
                self._verify_all_playing(source, zp, gms,
                                         transition=transition)
                time.sleep(PLAYBACK_TIME)
                self.reset_states()
                if self.error_detected:
                    if self.continue_on_errors:
                        self.fail("Error detected when transitioning from "
                                  "{} to {}. GC: {}: {}"
                                  .format(previous_source, source,
                                          zp.modelNumber, self.error_msg))
                        self.error_detected = False
                    else:
                        self.stop("Error detected when transitioning from "
                                  "{} to {}: {}".format(previous_source,
                                                        source, self.error_msg))
                        break
                if self.last_iteration:
                    break
            if self.last_iteration:
                break
            i += 1

    def test_playback_source_transitions(self, zp, gms):
        """
        Test playback all audio source transitions on devices between
            a. TV (applicable only if zp is ht_master_capable)
            b. linein
            c. internet
            d. hls
            e. share
        :param zp: Sonos zone component - Group coordinator
        :type zp: :class:`~sonos.client.SonosZoneComponent`
        :param list gms: List of sonos zone components which are group members
        """
        if zp.is_ht_master_capable():
            audio_sources = HT_AUDIO_SOURCES
        else:
            audio_sources = NON_HT_AUDIO_SOURCES
        self.source_sequence = self.build_audio_source_sequence(
            audio_sources=audio_sources)
        self.add_tracks_to_queue(zp, self.internet_track_uris)
        self.add_tracks_to_queue(zp, self.share_track_uris, track_num=1)
        if gms:
            zp.AVTransport.link_zones(gms, timeout=120)
            if not self._verify_all_zps_linked(zp, gms):
                self.stop("All devices not linked")
        signal.alarm(self.test_run_time - 30)
        with MplayerAc3Playback(TV_TRACK_PATH):
            self.logger.info("Started playing TV source")
            with MplayerMp3LineOutPlayback(LINEIN_TRACK_PATH):
                self.logger.info("Started playing line in source")
                self._run_transitions_test(zp, gms)
