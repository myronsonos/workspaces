        """
        Generate all combinations of zp pairs
        :return: A pair of Sonos zone components
        :rtype: :obj:`tuple`

        for zp_pair in itertools.combinations(self.test_devices, 2):

[docs]    def generate_gc_and_gm(self):
        """
        Generate gc and gm. Regardless of who is the GC, each device should get
        to be a group member
        :return: Sonos zone components - group coordinator and group member
        :rtype: :obj:`tuple`
        """
        for device in self.test_devices:
            gc = [zp for zp in self.test_devices if zp != device][0]
            yield (gc, device)



[docs]    def generate_gc_and_state(self):
        """
        Generate group coordinator other than ht_master
        :return: Sonos zone component - group coordinator other than ht_master
        :rtype: :`class:`~sonos.client.SonosZoneComponent`
        """
        for master in self.ht_masters:
            for device in self.test_devices:
                if not device == master:
                    for state in self.AUTOPLAY_STATE_VECTORS:
                        yield (master, device, state)

        self.num_group_members = args.num_group_members
        self.error_watch_time = args.error_watch_time
        self.logger.info("Number of iterations: {}"
                         .format(self.num_iterations))
        self.logger.info("Error watch time: {}"
                         .format(self.error_watch_time))
        self.logger.info("Number of group members: {}"
                         .format(self.num_group_members))

        """
        Setup test
        """
        self.logger.info("Setup test")
        self.logger.debug("Test started at : {}".format(datetime.now()))
        self.error_detected = False
        self.error_msg = ''
        self.delegation_or_grouping_error = False
        self.delegation_or_grouping_error_msg = ''
        for zp in self.test_devices:
            self.cleanup(zp)
        # variables used to check if device played tv or music content
        self.playing_tv = {}
        self.playing_music = {}
        self.playing_spdif = False
        for zp in self.test_devices:
            self.playing_tv[zp] = False
            self.playing_music[zp] = False
        # turn off autoplay on start of TV content
        [zp.DeviceProperties.SetAutoplayRoomUUID('') for zp in self.ht_masters]
        super(DelegationStressTest, self).setUpTest(
            trigger_events=self.PLAYING_EVENT,
            trigger_cb=self.cb_playing_event)

[docs]    def cb_playing_event(self, msg, zp, (m, expr, l, counters)):
        """
        Callback on detecting tv or linein (sbc) or music playback
        :param str msg: anacapa trace line containing detected string
        :param zp: Sonos zone component from which to submit
        :type zp: :class:`~sonos.client.SonosZoneComponent`
        """
        self.logger.info("Detected playing: {} on {}".format(msg, zp))
        if PLAYING_TV in msg:
            self.playing_tv[zp] = True
            self.logger.debug("Playing TV values detected so far: {}"
                              .format(self.playing_tv))
        if PLAYING_MUSIC in msg:
            self.playing_music[zp] = True
            self.logger.debug("Playing music values detected so far: {}"
                              .format(self.playing_music))
        if PLAYING_SPDIF in msg:
            self.playing_spdif = True


    def _verify_all_zps_linked(self, gc, gms):
        """
        Verify expected number of gms linked to gc
        :param gc: Sonos zone component - Group coordinator
        :type gc: :class:`~sonos.client.SonosZoneComponent`
        :param list gms: List of sonos zone component that are group members
        """
        self.logger.info("Verifying all gms linked to gc:{}".format(gc))
        expected_uuids = [zp.raw_udn for zp in gms] + [gc.raw_udn]

        def verify_gms_linked_to_gc():
            exp_num_linked_zones = len(expected_uuids)
            actual_linked_zones = (
                     len(gc.ZoneGroupTopology.get_zone_player_uuids_in_group()))
            if exp_num_linked_zones != actual_linked_zones:
                self.delegation_or_grouping_error_msg = "Devices not yet linked"
                self.logger.warn(self.delegation_or_grouping_error_msg)
                return False
            for gm in gms:
                if gc.raw_udn not in gm.AVTransport.get_current_uri():
                    self.delegation_or_grouping_error_msg = (
                        "GM: {} not pointing to GC: {}".format(gm, gc))
                    self.logger.warn(self.delegation_or_grouping_error_msg)
                    return False
            return True
        try:
            wait_until_true(
                lambda: verify_gms_linked_to_gc(), iteration_delay=2)
        except TimeoutError:
            self.logger.critical(self.delegation_or_grouping_error)
            self.delegation_or_grouping_error = True
            return False
        else:
            self.logger.log(AssertionLevel.PASS,
                            "All zps linked to: <{}>".format(gc))
            return True

    def _verify_all_playing_source(self, source, gc, gms):
        """
        Verify all devices are playing expected media source.
        :param str source: media source
        :param gc: Sonos zone component - Group coordinator
        :type gc: :class:`~sonos.client.SonosZoneComponent`
        :param list gms: List of sonos zone component that are group members
        :return: Did all devices play expected source ?
        :rtype: :obj:`bool`
        """
        self.logger.info("Verifying all devices playing source: {}. GC: {}"
                         .format(source, gc))
        result = True
        if source == TV:
            if not self._verify_ht_master_playing_tv():
                self.delegation_or_grouping_error_msg = (
                    "Did not play TV on ht_master: {}".format(gc))
                return False
            devices = gms
        if source == MUSIC:
            devices = gms + [gc]
        if result:
            try:
                wait_until_true(
                    lambda: self._did_all_devices_play(source, devices),
                    iteration_delay=1,
                    timeout_seconds=SRC_DETECT_TIMEOUT)
            except TimeoutError:
                self.delegation_or_grouping_error_msg = (
                    'Did not play {} on all devices'.format(source))
                for zp in devices:
                    if source == TV:
                        if not self.playing_tv[zp]:
                            self.logger.critical(
                                "Did not detect sbc playing on <{}>".format(zp))
                    if source == MUSIC:
                        if not self.playing_music[zp]:
                            self.logger.critical(
                                "Did not detect music playing on <{}>".format(
                                    zp))
                self.delegation_or_grouping_error = True
                return False
            else:
                self.logger.log(
                    AssertionLevel.PASS,
                    "Playing {} on all devices: <{}>".format(source, devices))
                return True

    def _verify_ht_master_playing_tv(self):
        """
        Verify ht_master is playing expected source
        :return: ht_master played expected source ?
        :rtype: :obj:`bool`
        """
        try:
            wait_until_true(lambda: self.playing_spdif,
                            iteration_delay=1,
                            timeout_seconds=SRC_DETECT_TIMEOUT)
        except TimeoutError:
            self.delegation_or_grouping_error = ('Did not play from spdif on '
                                                 'ht_master')
            self.delegation_or_grouping_error = True
            return False
        return True

    def _did_all_devices_play(self, source, devices):
        """
        Check if all gms played TV
        :param str source: expected source of audio data : linein, music or tv
        :param list devices: List of sonos zone component
        :return: Did all devices play expected source ?
        :rtype: :obj:`bool`
        """
        if source == TV:
            self.logger.info("Played {} state values: {}"
                             .format(source, self.playing_tv))
            return all([self.playing_tv[zp] for zp in devices])
        if source == MUSIC:
            self.logger.info("Played {} state values: {}"
                             .format(source, self.playing_music))
            return all([self.playing_music[zp] for zp in devices])

    def _delegate_to_new_member(self, current_gc, new_member):
        """
        Do delegation by first adding a member to group and immidiatelty,
        remove gc
        :param current_gc: Sonos zone component - Group coordinator
        :type current_gc: :class:`~sonos.client.SonosZoneComponent`
        :param new_member: Sonos zone component - member to be added to group
        :type new_member: :class:`~sonos.client.SonosZoneComponent`
        """
        self.logger.info("GM: {} joining group".format(new_member))
        new_member.AVTransport.join_group(current_gc)
        time.sleep(1)
        self.logger.info("GC: {} leaving group".format(current_gc))
        current_gc.AVTransport.BecomeCoordinatorOfStandaloneGroup()

    def _verify_delegation(self, source, gc, gms):
        """
        Verify delegation to gc occurred and gms linked to gc
        :param gc: Sonos zone component - Group coordinator
        :type gc: :class:`~sonos.client.SonosZoneComponent`
        :param list gms: List of sonos zone components as gms
        """
        self.logger.info("Verifying delegation to {}".format(gc))
        if self._verify_all_zps_linked(gc, gms):
            if self._verify_all_playing_source(source, gc, gms):
                return True
            else:
                self.delegation_or_grouping_error_msg = ("Delegation failed: "
                    "All devices not playing post delegation")
                self.delegation_or_grouping_error = True
                return False
        else:
            self.delegation_or_grouping_error_msg = ("Delegation failed: All "
                                                    "devices not linked")
            self.delegation_or_grouping_error = True
            return False

    def test_delegation(self, gc, gm):
        """
        Test delegation for a 2 zone player group. When gc player leaves
        group,gm should continue playing as standalone with avt inherited from
        gc. Allow playback to continue for 15 minutes and monitor for LSEs or
        other playback error. See SWPBL-40732
        """
        self.add_tracks_to_queue(gc, track_uris=self.http_uris)
        self.start_playback(gc, play_mode="REPEAT_ALL")
        # Allow playback for PLAYBACK_TIME seconds
        self.logger.debug("Allowing <{}> seconds for playback to continue"
                          .format(PLAYBACK_TIME))
        time.sleep(PLAYBACK_TIME)
        gc.AVTransport.link_zones([gm])
        self.logger.debug("Allowing <{}> seconds for playback to continue"
                          .format(PLAYBACK_TIME))
        time.sleep(PLAYBACK_TIME)
        gc.AVTransport.BecomeCoordinatorOfStandaloneGroup()
        self.verifyTrueOrFailCase(gm.AVTransport.is_playing,
                                  "Expect {} to continue playing".format(gm))
        played_time = 0
        self.logger.info("Playing for {} secs to look for errors"
                         .format(self.error_watch_time))
        while played_time < self.error_watch_time:
            time.sleep(1)
            played_time += 1
            if self.error_detected:
                self.fail(self.error_msg)
                break

    def test_delegation_to_ht_master(self, ht_master, gc, state):
        """
        Test delegation to ht_master when ht_master is a group member and
        has TV content to play. The original GC should delegate to ht_master and
        becomes GM of the new group whose GC is ht_master playing TV
        Case #1: Auto Play : ON
            1. Start music playback on GC that is not ht_master
            2. Group all members
            3. Verify all grouped and playing music
            4. Start HDMI playback
            5. Verify delegation: ht_master as GC all others as GMs
            6. Verify all playing TV
        Case #2: Auto Play : OFF
            1, 2, 3, 4 : as above
            5. Verify no delegation.
            6. Turn auto play
            7. Verify as in 5 and 6 of case 1
        Case #3: Auto Play : OFF
            1, 2,3, 4 : as above
            5 : as in case 2
            6. SetAVTransport URI on non ht_master GC to that of TV
            7. Verify as in 5 and 6 of case 1
        """

        (auto_play_state, delegation_reason) = state
        tv_uri = 'x-sonos-htastream:{}:spdif'.format(ht_master.raw_udn)
        if auto_play_state == 'ON':
            ht_master.DeviceProperties.SetAutoplayRoomUUID(ht_master.raw_udn)
        if auto_play_state == 'OFF':
            ht_master.DeviceProperties.SetAutoplayRoomUUID('')
        ht_master.DeviceProperties.SetAutoplayLinkedZones(True)
        if self.num_group_members == 1:
            gms = [ht_master]
        else:
            gms = [zp for zp in self.test_devices
                   if zp not in [gc, ht_master]][:self.num_group_members-1]
            gms.append(ht_master)
        # step 1
        self.add_tracks_to_queue(gc, track_uris=self.http_uris)
        self.start_playback(gc, play_mode="REPEAT_ALL")
        # step 2
        gc.AVTransport.link_zones(gms)
        # step 3
        if self._verify_all_zps_linked(gc, gms):
            if not self._verify_all_playing_source(MUSIC, gc, gms):
                self.stop(self.delegation_or_grouping_error_msg)
        else:
            self.stop(self.delegation_or_grouping_error_msg)
        # step 4
        self.logger.info("Waiting for {} secs of silence on spdif"
                         .format(SPDIF_SILENCE_TIME))
        time.sleep(SPDIF_SILENCE_TIME)
        with MplayerAc3Playback(TV_TRACK_PATH):
            self.logger.info("Started playback on HDMI/SPDIF")
            post_delegation_gc = ht_master
            post_delegation_gms = [zp for zp in gms
                                   if not zp == ht_master] + [gc]
            if auto_play_state == 'OFF':
                if not self._verify_all_zps_linked(gc, gms):
                    self.stop(self.delegation_or_grouping_error_msg)
                if delegation_reason == 'TURN_ON_AUTOPLAY':
                    # Step #6
                    ht_master.DeviceProperties.SetAutoplayRoomUUID(
                        ht_master.raw_udn)
                elif delegation_reason == 'SELECT_TV':
                    # step #6
                    gc.AVTransport.SetAVTransportURI(tv_uri, "")
            # verify delegation
            # steps 5, 6 if AUTO PLAY already ON
            # step 7 if AUTO PLAY is OFF initially.
            if not self._verify_delegation(
                    TV, post_delegation_gc, post_delegation_gms):
                self.fail(self.delegation_or_grouping_error_msg)
        if self.error_detected:
            self.fail("Failed with error: {}".format(self.error_msg))

    def test_fast_delegation(self, zp1, zp2):
        """
        Verify no audio playback issues when performing quick multiple
        delegations. See SWPBL-50481, SWPBL-50278.
        :param zp1: Sonos zone component
        :type zp1::class:`~sonos.client.SonosZoneComponent`
        :param zp2: Sonos zone component
        :type zp2::class:`~sonos.client.SonosZoneComponent`
        """
        gms = [zp for zp in self.test_devices
               if zp not in [zp1, zp2]][:self.num_group_members]
        iteration = 0
        self.add_tracks_to_queue(zp1, track_uris=self.http_uris)
        self.start_playback(zp1)
        # Allow playback for PLAYBACK_TIME seconds
        self.logger.debug("Allowing <{}> seconds for playback to continue"
                          .format(PLAYBACK_TIME))
        time.sleep(PLAYBACK_TIME)
        zp1.AVTransport.link_zones(gms)
        self.logger.debug("Allowing <{}> seconds for playback to continue"
                          " after grouping".format(PLAYBACK_TIME))
        time.sleep(PLAYBACK_TIME)
        current_gc = zp1
        new_member = zp2
        while iteration < self.num_iterations:
            iteration += 1
            self.logger.info("Now doing iteration #{} of {}"
                             .format(iteration, self.num_iterations))
            self._delegate_to_new_member(current_gc, new_member)
            time.sleep(DELEGATION_GAP)
            if not self._verify_delegation(MUSIC, new_member, gms):
                break
            # interchange gc and gm and redo delegation
            current_gc, new_member = new_member, current_gc
            self._delegate_to_new_member(current_gc, new_member)
            if not self._verify_delegation(MUSIC, new_member, gms):
                break
            # interchange gc and gm and redo delegation
            current_gc, new_member = new_member, current_gc
            self.logger.debug("Waiting <{}> seconds to look for errors"
                              .format(WAIT_TIME))
            played_time = 0
            while played_time < WAIT_TIME:
                time.sleep(1)
                played_time += 1
                if self.error_detected:
                    break
            if self.delegation_or_grouping_error:
                self.stop(self.delegation_or_grouping_error_msg)
            if self.error_detected:
                if self.continue_on_errors:
                    self.fail("Failed with error: {}".format(self.error_msg))
                else:
                    self.stop("Stopping with error: {}".format(self.error_msg))
                    break
