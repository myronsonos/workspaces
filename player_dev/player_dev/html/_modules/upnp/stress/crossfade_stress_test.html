    """
    Purpose of this test is to test crossfade for all codec transitions in 
    grouped environment as that is most likely to produce failures if any. Also
    serve tracks via http as it exercises the audio pipeline more fully then via
    cifs. In case of spotify tracks, play full track as it was found to create
    a 'buildup' of state which could potentially cause a failure such as LSE
    """
    CROSSFADE_EVENTS = {'chsnk': {2: [CROSSFADE_ON, CROSSFADE_END]}}

    def __init__(self):
        StressTestsBaseNew.__init__(self)
        self.crossfading = {}
        self.all_tracks_dict = {}


[docs]    def parse_test_args(self):
        """
        Parse command line args
        """
        parser = argparse.ArgumentParser()
        parser.add_argument("--gc_models", required=False, type=str,
                            default='all',
                            help="Run test on comma separated list of gcs"
                                 "default = All test models.")
        parser.add_argument("--num_group_members", required=False, type=int,
                            default=NUM_GROUP_MEMBERS,
                            help=("Number of group members"
                                  "Default: {}".format(NUM_GROUP_MEMBERS)))
        args, unknown_args = parser.parse_known_args()
        sys.argv[1:] = unknown_args
        self.gc_models = []
        test_devices_models = [zp.modelNumber for zp in self.test_devices]
        if args.gc_models == 'all':
            self.gc_models = test_devices_models
        else:
            temp_models = re.split(',', args.gc_models)
            for temp_model in temp_models:
                model = temp_model.upper()
                if model in test_devices_models:
                    self.gc_models.append(model)
                else:
                    self.logger.warn("{} not in expected list of {}"
                                     .format(model, test_devices_models))
        self.num_group_members = args.num_group_members
        self.logger.info("Number of GMs: {}".format(self.num_group_members))
        self.logger.info("GC Models: {}".format(self.gc_models))



[docs]    def gc_gms_generator(self):
        """
        Generate gc.
        :return: Sonos zone component which will serve as a group coordinator
        :rtype: :obj:`~sonos.client.internal.SonosZoneComponent`
        """
        for gc in self.test_devices:
            gms = [zp for zp in self.test_devices if zp != gc][:self.num_group_members]
            if gc.modelNumber in self.gc_models:
                yield gc, gms

[docs]    def crossfade_uri_generator(self, http_uris):
        """
        Generate all permutation of start and end track from non spotify uris
        :param list http_uris: List of http uris
        :return: Tuple of start and end http uris
        :rtype: :obj:`tuple`
        """
        for start, end in itertools.permutations(http_uris, 2):
            yield (start, end)


    def _remove_duplicate_sequences(self, codec, tracks_set, master_list):
        """
        Remove duplicate sequence of tracks from set of all permutations
        :param str codec: track codec 
        :param list tracks_set: List of tuples of tracks
        :param list master_list: master list of all tuples of track set
        :return temp_list: list of tuples of tracks
        :rtype: :obj:`list`
        """
        temp_list = []
        codec_list = []
        delete_list = []
        for i in range(len(tracks_set)):
            # delete_list = []
            start, end = tracks_set[i]
            if codec == start:
                codec_list.append(tracks_set[i])
                master_list.append(tracks_set[i])
        for i in range(len(codec_list)):
            start, end = codec_list[i]
            if i + 1 < len(codec_list):
                next_start, next_end = codec_list[i + 1]
                delete_set = (end, next_start)
                delete_list.append(delete_set)
        temp_list[:] = [a_set for a_set in tracks_set
                        if a_set not in master_list]
        for a_set in delete_list:
            if a_set in temp_list:
                temp_list.remove(a_set)
        return temp_list

    def _add_tracks_to_queue(self, zp):
        """
        Add tracks to queue as per all permutation generated.
        :param zp: Sonos zone component 
        :type zp: :`~sonos.client.SonosZoneComponent`
        """
        self.logger.info("Adding tracks to queue")
        # add spotify tracks
        for spotify_track in self.spotify_http_uris:
            zp.AVTransport.add_to_queue(spotify_track)
        # add all codec tracks, all permutations from codec to codec
        http_uri_permutations = [http_uri_pair for http_uri_pair in
                                 self.crossfade_uri_generator(self.http_uris)]
        all_uris_list = []
        # progressively remove repeating sequences to save test time
        # example : from  these permutation, (ogg, mp4) (flac, wav) (mp4 flac)
        # remove (mp4 flac) transition as this sequence is already obtained
        # between first and second permutation.
        for uri in self.http_uris:
            http_uri_permutations = self._remove_duplicate_sequences(
                uri, http_uri_permutations, all_uris_list)
        # add tracks for transition between same codecs
        for http_uri in self.http_uris:
            all_uris_list.append((http_uri, http_uri))
        for http_uri_pair in all_uris_list:
            first_track, second_track = http_uri_pair
            zp.AVTransport.add_to_queue(first_track)
            zp.AVTransport.add_to_queue(second_track)
        num_tracks = zp.AVTransport.get_number_of_tracks_in_queue()
        return num_tracks

    def _did_all_devices_crossfade(self, device_list):
        """
        Check is all devices crossfade
        :param list device_list: List of Sonos zone components under test
        :return: Did all devices crossfade ?
        :rtype: :obj:`bool`
        """
        self.logger.info("Crossfade state values: {}".format(self.crossfading))
        return all([self.crossfading[zp.raw_udn] for zp in device_list])

    def _did_next_track_start(self, zp, current_track):
        """
        Check if next track playback started
        :param zp: Sonos zone component
        :type zp: :`~sonos.client.SonosZoneComponent`
        :param int current_track: Current track number
        :return: Did next track start ?
        :rtype: :obj:`bool`
        """
        self.logger.info("Current position info: {}"
                         .format(zp.AVTransport
                                 .GetPositionInfo().get('RelTime')))
        return zp.AVTransport.get_current_track() != current_track

    def test_crossfade(self, gc, gms):
        if gms:
            gc.AVTransport.link_zones(gms)
        # list of devices under test
        device_list = [gc] + gms
        device_udn_list = [zp.raw_udn for zp in device_list]
        self.num_tracks = self._add_tracks_to_queue(gc)
        self.start_playback(gc, crossfade=True)
        self.verifyTrueOrFailCase(lambda: gc.AVTransport.GetCrossfadeMode(),
                                  "Expect crossfade on")
        self.playback_log = PLAYBACK_LOG.format(gc.modelNumber)
        # write http uri sequence to file as it is played
        fh = open(self.playback_log, 'w')
        current_track = -1
        while current_track < self.num_tracks:
            self.error_detected = False
            self.error_msg = ''
            # not_crossfading_zps = []
            # reset crossfading states to False
            for zp in device_list:
                self.crossfading[zp.raw_udn] = False
            track_duration = reltime_str_to_seconds(
                                    gc.AVTransport.get_current_track_duration())
            current_track_uri = gc.AVTransport.get_current_track_uri()
            current_track = gc.AVTransport.get_current_track()
            fh.write(current_track_uri + '\n')
            self.logger.info("Now playing : {}/{} - {}"
                             .format(current_track, self.num_tracks,
                                     current_track_uri))
            # wait for track transition
            self.logger.info("Waiting for track transtion on <{}>"
                             .format(gc))
            try:
                wait_until_true(lambda: self._did_next_track_start(gc,
                                current_track),
                                iteration_delay=5,
                                timeout_seconds=track_duration + 20)
            except TimeoutError:
                if current_track < self.num_tracks:
                    self.error_msg = 'Timeout waiting for next track to start'
                    self.error_detected = True
                    # if we are not in playing state, stop test else will go
                    # in endless loop.
                    if not gc.AVTransport.is_playing():
                        self.error_msg = ("{}: Playback stopped !!!! "
                                          .format(gc))
                        self.continue_on_errors = False
            # verify cross fade occurred
            self.logger.info("Checking if all players crossfaded")
            try:
                wait_until_true(
                    lambda: self._did_all_devices_crossfade(device_list),
                    iteration_delay=1,
                    timeout_seconds=30)
            except TimeoutError:
                if current_track < self.num_tracks:
                    self.error_msg = 'Crossfade did not occur on:'
                    self.error_detected = True
                    for udn in device_udn_list:
                        if not self.crossfading[udn]:
                            zone = self.get_testbed_device_by_raw_udn(udn)
                            if zone.raw_udn == gc.raw_udn:
                                role = 'GC'
                            else:
                                role = 'GM'
                            self.logger.critical("{}: <{}> did not crossfade"
                                                 .format(role, zone))
                            self.error_msg += ", {} - <{}>".format(role, zone)

            if self.error_detected:
                if self.continue_on_errors:
                    self.fail(self.error_msg)
                else:
                    fh.close()
                    gc.AVTransport.Stop()
                    self.stop("Stopping test on detecting error: {}"
                              .format(self.error_msg))
                    break
