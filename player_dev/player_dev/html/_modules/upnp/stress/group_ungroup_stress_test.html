[docs]class GroupUnGroupStressTest(StressTestsBaseNew):
    """
    Purpose of this test is to stress grouping and ungrouping. GC is stressed
    when zones are grouped and ungrouped. In this test, single gm is ungrouped
    from a group and after waiting few seconds, is re-added to group. This
    sequence is rotated amongst all the gms for 'num_iterations'
    """

    PLAYING_MUSIC_EVENT = {'chsnk': {2: [PLAYING_MUSIC]}}
    DEFAULT_ITERATIONS = 5


[docs]    def gc_gms_and_state_generator(self):
        """
        Generate gc.
        :return: Tuple of Sonos zone component which will serve as a
            group coordinator and is new group member already playing ?
        :rtype: :obj:`~sonos.client.internal.SonosZoneComponent
        :rtype: :obj:`bool`
        """
        for device in self.test_devices:
            gms = [gms for gms in self.test_devices
                   if gms != device][:self.num_group_members]
            for new_member_playing in NEW_MEMBER_PLAYING_STATE:
                yield device, gms, new_member_playing

        """
        Parse command line args
        """
        parser = argparse.ArgumentParser()
        parser.add_argument("--num_iterations", required=False, type=int,
                            default=self.DEFAULT_ITERATIONS,

                            help="Run on Spotify playlist - requires account")
        parser.add_argument("--num_group_members", required=False, type=int,
                            default=len(self.test_devices) - 1,
                            help=("Number of group members Default: {}"
                                  .format(len(self.test_devices) - 1)))
        args, unknown_args = parser.parse_known_args()
        sys.argv[1:] = unknown_args
        self.num_iterations = args.num_iterations
        self.num_group_members = args.num_group_members
        self.spotify = args.spotify
        self.logger.info("Number of iterations: {}"
                         .format(self.num_iterations))
        self.logger.info("Number of group members: {}"
                         .format(self.num_group_members))
        self.logger.info("Test with Spotify : {}".format(self.spotify))

        """
        Setup test
        """
        self.logger.info("Setup test")

        for zp in self.test_devices:

    def _add_tracks_to_queue(self, zp):
        """
        Add tracks to queue on this zp
        :param test_zone: Sonos zone component
        :type test zone: :`~sonos.client.SonosZoneComponent`
        """
        self.logger.info("Adding tracks to queue on <{}>".format(zp))
        if self.spotify:
            metadata = zp.AVTransport.generate_didl_lite(
                title=ROCK_BALLADS.title,
                svc_id=self.svc_acct.service_id,
                acct_id=self.svc_acct.username)
            self.num_tracks = zp.AVTransport.add_to_queue(
                ROCK_BALLADS.uri, metadata)
            self.num_tracks = zp.AVTransport.remove_restricted_tracks()
        else:
            for uri in self.http_uris:
                zp.AVTransport.add_to_queue(uri)
            self.num_tracks = zp.AVTransport.get_number_of_tracks_in_queue()
            self.logger.info("{}: Number of tracks in queue: {}"
                             .format(zp, self.num_tracks))


[docs]    def cb_playing_event(self, msg, zp, (m, expr, l, counters)):
        """
        Callback on detecting music playback
        :param zp: Sonos zone component from which to submit
        :type zp: :class:`~sonos.client.SonosZoneComponent`
        """
        self.logger.info("Detected playing music on {}".format(zp))
        self.playing_music[zp] = True


    def _did_all_gms_play(self, gms):
        """
        Check if all gms played TV
        :param list gms: List of Sonos zone components as group members
        :return: Did all devices play tv ?
        :rtype: :obj:`bool`
        """
        self.logger.info("Played music state values: {}"
                         .format(self.playing_music))
        return all([self.playing_music[zp] for zp in gms])

    def _verify_all_gms_playing_music(self, gms):
        """
        Wait until all gms are playing on timeout, stop the test
        :param list gms: List of Sonos zone components as group members
        """
        try:
            wait_until_true(lambda: self._did_all_gms_play(gms),
                            iteration_delay=1,
                            timeout_seconds=MUSIC_DETECT_TIMEOUT)
        except TimeoutError:
            self.error_msg = 'Did not play music on all players'
            for zp in self.playing_music:
                if not self.playing_music[zp]:
                    self.logger.critical("Did not detect music playing on <{}>"
                                         .format(zp))
            self.stop(self.error_msg)
        else:
            self.logger.log(AssertionLevel.PASS,
                            "Playing music on all gms: <{}>".format(gms))

    def _verify_all_zps_linked(self, gc, num_gms):
        """
        Verify expected number of gms linked to gc
        :param zp: Sonos zone component - Group coordinator
        :type zp: :class:`~sonos.client.SonosZoneComponent`
        :param int num_gms: expected number of gms
        """
        try:
            wait_until_true(lambda: gc.ZoneGroupTopology.get_num_linked_zones()
                            == num_gms)
        except TimeoutError:
            self.error_msg = "All zps not linked"
            expected_uuids = [zp.raw_udn for zp in self.test_devices]
            actual_uuids = gc.ZoneGroupTopology.get_zone_player_uuids_in_group()
            missing_uuids = [uuid for uuid in expected_uuids
                             if uuid not in actual_uuids]
            self.logger.critical(
                "Current UUID's in group: {}"
                .format(gc.ZoneGroupTopology.get_zone_player_uuids_in_group()))
            self.logger.critical("Missing UUIDs: <{}>".format(missing_uuids))
            self.stop(self.error_msg)
        else:
            self.logger.log(AssertionLevel.PASS,
                            "All zps linked to: <{}>".format(gc))

    def _run_test(self, gc, gms, new_gm_playing):
        """
        :param zp: Sonos zone component - Group coordinator
        :type zp: :class:`~sonos.client.SonosZoneComponent`
        :param bool new_member_playing;is new added member playing
        """
        i = 0
        self.logger.info("Linking zones to gc: <{}>".format(gc))
        for zp in gms:
            self.playing_music[zp] = False
        gc.AVTransport.link_zones(gms, timeout=120)
        self._add_tracks_to_queue(gc)
        qpos = 1 if not self.spotify else randint(1, self.num_tracks)
        self.start_playback(gc, qpos=qpos, play_mode="REPEAT_ALL")
        # verify all gms playing music
        self.logger.info("Checking if all GM players is playing music")
        self._verify_all_gms_playing_music(gms)
        q = deque(gms)
        self.logger.info("Running test for: {} rotations"
                         .format(self.num_iterations))
        while i < (self.num_iterations * len(gms)):
            gm = q[0]
            self.logger.info("<{}> leaving group".format(gm))
            gm.AVTransport.leave_group_and_wait(
                timeout_seconds=LEAVE_GROUP_WAIT_TIME)
            if new_gm_playing == 'new_gm_playing':
                # new gm in playing state
                self.logger.info("Starting playback on new gm candidate {}"
                                 "before joining to group with GC: {}"
                                 .format(gm, gc))
                self._add_tracks_to_queue(gm)
                qpos = (i % self.num_tracks) + 1
                self.start_playback(gm, qpos=qpos)
            self.playing_music[gm] = False
            self.verifyTrueOrStop(gc.AVTransport.is_playing(),
                                  "Expect gc <{}> in playing state"
                                  .format(gc))
            time.sleep(GROUPING_WAIT_TIME)
            self.logger.info("<{}> joining group".format(gm))
            gm.AVTransport.join_group(gc)
            self._verify_all_zps_linked(gc, len(gms))
            self.verifyTrueOrStop(gc.AVTransport.is_playing(),
                                  "Expect gc <{}> in playing state"
                                  .format(gc))
            self._verify_all_gms_playing_music(gms)
            time.sleep(GROUPING_WAIT_TIME)
            if self.error_detected:
                if self.continue_on_errors:
                    self.fail(self.error_msg)
                    self.error_detected = False
                else:
                    self.stop(self.error_msg)
            q.rotate()
            i += 1

    def test_grouping_ungrouping(self, gc, gms, new_gm_playing):
        """
        Form a group, remove a player, add a player rotate thru the gms for
        'ITERATIONS'. New GM playing"
        """
        self.logger.info("GC: {}. New GM playing state -Playing ?: {}"

        self._run_test(gc, gms, new_gm_playing)
