        """

        * One line in player
        * Home theater masters

        :return: Line in ZP, HT master, HT satellite
        :rtype: :class:`~sonos.client.internal.SonosZoneComponent`
        """
        ht_devices = (
            self.generators.
            generate_testbed_ht_master_satellite_capable_devices())
        for line_in_zp in self.line_in_devices:
            if line_in_zp.is_encore():
                for ht_master, left, right in ht_devices:
                    if line_in_zp.serialNumber not in [left.serialNumber,
                                                        right.serialNumber]:
                        yield line_in_zp, ht_master, left, right

[docs]    def linein_playback_compression(self, zp):
        """
        Verifies the zone player can play line-in source without stopping
        while the input audio compression mode is changed.
        SWPBL-66922, SWPBL-41803
        1. Select zone X with line-in
        2. Start playback on a different zone Y using line-in
        3. While playing line-in on zone Y change the compression mode
        4. Verify zone Y playing line-in (AVT state)
        5. Confirm zone Y did not attempt to restart line-in playback
        6. Repeat steps 3-5 until playback stops unexpectedly or test
        duration expired.
        """
        linein_zp = [zp for zp in self.line_in_devices if zp.is_encore()][0]
        encoding = LINE_IN_COMPRESSION_TYPES[0]
        self.logger.info(
                "Initial line-in playback using {} encoding".format(encoding))
        zp.SystemProperties.set_line_in_encoding(encoding)
        zp.AVTransport.use_line_in(linein_zp)
        zp.AVTransport.Play_and_wait()
        playing = self.playing_line_in(zp)
        self.verifyTrueOrFailCase(
            playing,
            "Expect to be playing initial line-in {}".format(encoding))
        for compression in LINE_IN_COMPRESSION_TYPES:
            zp.SystemProperties.set_line_in_encoding(compression)
            encoding = zp.SystemProperties.GetString("R_AudioInEncodeType")
            self.verifyEqualOrFailCase(
                compression,
                encoding,
                "Is expected encoding correct ?")
            self.logger.info(
                "Playback using {} encoding".format(encoding))
            playing = self.playing_line_in(zp)
            if not playing:
                self.stop(
                    "Failed to play line-in {} encoding".format(encoding))
            self.logger.info(
                "PASS playing line-in {} encoding".format(encoding))
            self.logger.info("Sleeping 5 seconds")
            time.sleep(SLEEP_BETWEEN_LINE_IN_ENCODINGS)

    def test_linein(self, zp, state, playback_devices):
        """
        Play line while line is src is playing music or line-in to itself
        or nothing. Look for audio playback errors while runing the test.
        """
        self.add_tracks_to_queue(zp, self.track_uris)
        with MplayerMp3LineOutPlayback(LINE_IN_TRACK):
            line_in_device = zp
            # line-in src is not playing anything
            if state == 'none':
                line_in_device.AVTransport.clear_play_state()
            # line-in src is playing music locally
            if state == 'music':
                self.start_playback(line_in_device, play_mode="REPEAT_ALL")
            # line-in src is playing line-in locally
            if state == 'linein':
                line_in_device.AVTransport.use_line_in(line_in_device)
                line_in_device.AVTransport.Play_and_wait()
            for zp in playback_devices:
                zp.AVTransport.use_line_in(line_in_device)
                zp.AVTransport.Play_and_wait()

            test_duration = TEST_DURATION
            while test_duration > 0:
                time.sleep(1)
                if self.error_detected:
                    if not self.continue_on_errors:
                        break
                    else:
                        self.fail("Detected error: {}".format(self.error_msg))
                test_duration -= test_duration
            for zp in playback_devices:
                self.verifyTrueOrFailCase(
                    self.playing_line_in(zp),
                    "Expect {} to be in playing state".format(zp))
            if self.error_detected:
                self.fail("Detected error: {}".format(self.error_msg))



[docs]    def cb_playing_event(self, msg, zp, (m, expr, l, counters)):
        """
        Callback on detecting tv or linein (sbc) or music playback and
        when we detect player attempt to play line-in
        :param str msg: anacapa trace line containing detected string
        :param zp: Sonos zone component from which to submit
        :type zp: :class:`~sonos.client.SonosZoneComponent`
        """
        self.logger.info("Detected playing: {} on {}".format(msg, zp))
        if CHSNK_SYNC_PLAY_MSG in msg:
            self.stopping += self.stopping
        elif COMPRESSED_LINEIN_PLAYBACK in msg:
            self.playing_sbc[zp.serialNumber] = True
            self.logger.debug("Playing sbc values detected so far: {}"
                              .format(self.playing_sbc))
        elif UNCOMPRESSED_LINEIN_PLAYBACK in msg:
            self.playing_wav[zp.serialNumber] = True
            self.logger.debug("Playing wav codec on {}".format(zp))



[docs]    @combinatorial('generate_line_in_player_and_ht_players')
    def test_linein_ht_grouped(self, line_in_device, ht_master, left, right):
        """
        Verifies enabled home theater satellites do not "fail to play
        samples" when grouped with a line-in device playing compressed audio
        (SBC)
        :param :class:`sonos.client.internal.SonosZoneComponent` line_in_zp:
        :param :class:`sonos.client.internal.SonosZoneComponent` ht_master:
        """
        ht_master.SystemProperties.set_line_in_encoding("COMPRESSED")
        ht_master.DeviceProperties.create_50_config(left, right)
        ht_master.RenderingControl.set_surround_mode('full')
        with MplayerMp3LineOutPlayback(LINE_IN_TRACK):
            self.logger.info("Now playing line track: {} using mplayer"
                                 .format(LINE_IN_TRACK))
            # limiting the use of playback_devices to only ht_master as the
            # intent of test is just grouping ht master
            line_in_device.AVTransport.use_line_in(line_in_device)
            line_in_device.AVTransport.Play_and_wait()
            ht_master.AVTransport.join_group(line_in_device)
            wait_until_true(
                lambda: all([self.playing_sbc[line_in_device.serialNumber],
                             self.playing_sbc[ht_master.serialNumber]]),
                iteration_delay=1,
                timeout_seconds=30)
            failed_to_play_samples_diagc = 'failed to play samples 4 buff -1'
            failed_to_play_samples = False
            try:
                left.diag.wait_for_anacapa_log_statement(
                    failed_to_play_samples_diagc, 10)
                failed_to_play_samples = True
            except TimeoutError:
                pass
            else:
                self.fail('<{}> DIAGC should not occur'.format(
                          failed_to_play_samples_diagc))
            if self.error_detected:
                self.fail("Detected Errors. {}.".format(self.error_msg))
