[docs]class SeekStressTest(StressTestsBaseNew):
    """
    Purpose of this test is to verify for all player and codec types:
    1. We do not make excessive http partial requests when seeking (SWPBL-53169)
        when playing from the internet
    2. No playback errors.
    """

    # The TrackSeekVector namedtuple fields are:
    # 'track': name of track, 'file_length': length of track in secs
    # 'threshold': For a track, threshold for number of http request for seek
    # play_to_[1..5] and seek_to_[1..5] : For the test, start playback, play
    # till play_to_x seconds in track, then Seek/scrub to seek_to_y secs, then
    # do verifications, continue playback till next play_to_x and so on for
    # 5 sets of play_to and seek_to
    TrackSeekVector = namedtuple('TrackSeekVector',
                                 ['track', 'file_length','threshold',
                                  'play_to_1', 'seek_to_1',
                                  'play_to_2', 'seek_to_2',
                                  'play_to_3', 'seek_to_3',
                                  'play_to_4', 'seek_to_4',
                                  'play_to_5', 'seek_to_5']
                                 )
    HTTP_REQUEST_EVENT = {'dataio': {3: [HTTP_PARTIAL_REQUEST]}}

    def _build_tracks_dict(self):
        tracks_dict = {}
        f = open(TRACKS_JSON)
        json_str = f.read()
        test_dict = json.loads(json_str)
        for codec in test_dict:
            if 'enable' in test_dict[codec] and not test_dict[codec]['enable']:
                self.logger.warning("Skipping codec \"{}\"{}".format(codec, ", Reason: {}".format(
                    test_dict[codec]['reason']) if 'reason' in test_dict[codec] else ""))
                continue
            tracks_data = dict(test_dict[codec])
            for track_data in tracks_data:
                tracks_dict[codec+track_data] = tracks_data[track_data]['name']
        self.logger.info("tracks_dict: {}".format(tracks_dict))
        return tracks_dict

    def _build_test_vectors(self):
        """
        Build test vector after reading data from TRACKS_JSON.
        :return: List of test vector of type TrackSeekVector
        :rtype: :obj:`list`
        """
        test_vectors = []
        f = open(TRACKS_JSON)
        json_str = f.read()
        tracks_dict = json.loads(json_str)
        for codec in tracks_dict:
            if 'enable' in tracks_dict[codec] and not tracks_dict[codec]['enable']:
                self.logger.warning("Skipping codec \"{}\"{}".format(codec, ", Reason: {}".format(
                    tracks_dict[codec]['reason']) if 'reason' in tracks_dict[codec] else ""))
                continue
            # we now have codec key
            tracks_data = dict(tracks_dict[codec])
            for track_data in tracks_data:
                track = tracks_data[track_data]['name']
                threshold = tracks_data[track_data]['http_request_threshold']
                file_length = tracks_data[track_data]['duration']
                seek_vector = tracks_data[track_data]['seek_vector']
                temp_vector = self.TrackSeekVector(track, file_length,
                                                   threshold, *seek_vector)
                test_vectors.append(temp_vector)
        return test_vectors

[docs]    def cb_http_event(self, msg, zp, (m, expr, l, counters)):
        """
        Callback on detecting http partial request. Count the number of
        requests.
        :param str msg: error message
        :param zp: Sonos zone component on which http request was detected
        :type zp: :class:`~sonos.client.SonosZoneComponent`
        """
        self.num_http_requests_made += 1


    def _is_num_requests_within_threshold(self, threshold):
        """
        Check the number of http requests made on a seek against threshold
        :return: Were total number of request on seek less than threshold ?
        :rtype: :obj:`bool`
        """
        self.logger.info("Number of http requests made: {}"
                         .format(self.num_http_requests_made))
        return self.num_http_requests_made < threshold

    def _are_all_players_linked(self, gc, gms):
        """
        Check if all players are linked
        :param gc: Sonos zone component
        :type gc: :`~sonos.client.SonosZoneComponent`
        :param gms: Group members
        :type gms: :`list`
        """
        result = True
        gc_uri = gc.AVTransport.get_avtransport_uri()
        if not gc_uri.startswith("x-rincon-queue"):
            self.logger.critical("AVTransport uri: {} on GC: {} not as expected"
                                .format(gc, gc_uri))
            result = False
        for zp in gms:
            uri = zp.AVTransport.get_avtransport_uri()
            if gc.serialNumber not in uri:
                self.logger.critical("AVTransport uri: {} on GM: {} not as "
                                     "expected. GC:{}".format(zp, uri, gc))
                result = False
        return result

    def _do_seek_test(self, zp, test_vector, track_source):
        """
        Do seek test
        :param zp: Sonos zone component
        :type zp: :`~sonos.client.SonosZoneComponent`
        :param test_vector: Test vector containing seek specifications
        :type test_vector: :`tuple`
        """
        track = test_vector.track
        # split the test vector in 5 seeks. The sequence is play_to x sec in
        # track, then seek to y sec in track, then next seek set.
        seek_sets = [(test_vector.play_to_1, test_vector.seek_to_1),
                     (test_vector.play_to_2, test_vector.seek_to_2),
                     (test_vector.play_to_3, test_vector.seek_to_3),
                     (test_vector.play_to_4, test_vector.seek_to_4),
                     (test_vector.play_to_5, test_vector.seek_to_5),
                     ]
        track_uri = self.get_track_uris([track], source=track_source)
        self.add_tracks_to_queue(zp, track_uri)
        self.start_playback(zp)
        # iterate thru 5 seek sets
        for seek_set in seek_sets:
            # reset counters
            self.num_http_requests_made = 0
            self.logger.info("Now seeking from {} to {} position in {}"
                             .format(seek_set[PLAY_TO],
                                     seek_set[SEEK_TO],
                                     track))
            # play till seek_set[PLAY_TO] position in track
            try:
                wait_until_true(
                    lambda:convert_reltime_to_seconds(zp) >= seek_set[PLAY_TO],
                    iteration_delay=1, timeout_seconds=30)
            except TimeoutError:
                if self.error_detected:
                    self.logger.critical("{} : Playback Errors detected {} "
                      .format(self.error_msg, zp))
                self.stop("{}: Player in stopped state.".format(zp))
            # seek to seek_set[SEEK_TO] position in track
            seek_to = seconds_to_reltime_str(seek_set[SEEK_TO])
            zp.AVTransport.Seek('REL_TIME', seek_to)
            # wait until after we've played past the seek time to be sure
            # we've actually transitioned

            def check_reltime(sec):
                return (seek_set[SEEK_TO] <
                        sec <
                        (seek_set[SEEK_TO]+SEEK_NEAR_DELTA))

            def is_zp_playing():
                return (zp.AVTransport.get_current_transport_state()==
                        zp.AVTransport.PLAYING_STATE)
            try:
                wait_until_true(
                    lambda: is_zp_playing() and
                    check_reltime(convert_reltime_to_seconds(zp)),
                    timeout_seconds=10, iteration_delay=.5)
            except TimeoutError:
                # let the code following double check
                pass
            self.verifyTrueOrFailCase(
                is_zp_playing(),
                "Expect {} to be in playing state".format(zp))
            curr_reltime = convert_reltime_to_seconds(zp)
            self.verifyTrueOrFailCase(check_reltime(curr_reltime),
                                      "Expect {} to seeked to {} but is {}"
                                      .format(zp, seek_set[SEEK_TO],
                                              curr_reltime))
            if track_source == HTTP:
                self.verifyTrueOrFailCase(
                self._is_num_requests_within_threshold(test_vector.threshold),
                    "Expect number of http requests less than {}"
                    .format(test_vector.threshold))
            # playback errors detected while seeking
            if self.error_detected:
                if self.continue_on_errors:
                    self.fail("{}. GC: {}Track played: {}"
                              .format(self.error_msg, zp, test_vector.track))
                else:
                    self.stop("{}. GC: {}Track played: {}"
                              .format(self.error_msg, zp, test_vector.track))
