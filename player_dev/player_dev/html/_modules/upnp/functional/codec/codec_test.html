    def _add_to_queue(self, zp, files):
        """
        Add music tracks to zp's music queue

        :param zp: Sonos Zone Component object
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`

        :param files: List of media tracks
        :type files: :obj: 'dict'

        :return: None
        """
        track_uris = ["http://{}/{}".format(self.__websrv.host(), file_name)
                      for file_name in files.values()]
        tracks_added = zp.AVTransport.add_to_queue(track_uris, instance_id=0)
        assert tracks_added == len(track_uris)

[docs]    def playback_timer(self, zp, files):
        """
        Tests playback using AVTransport Transition Timer. Asserts that playback
        starts, continues for a fixed play time, and stops without errors.

        :param zp: Sonos Zone Component Object
        :param files: playback files that are added to queue in each test function
        :return: None
        """
        avts = AVTransportStateTransitionTimer(zp.AVTransport)
        file_num = 1
        while file_num <= len(files):
            # assert playback works
            self.start_playback(zp, file_num)
            avts.start(
                CodecAudioTests.MAXIMUM_PLAY_TIME,
                CodecAudioTests.MINIMUM_PLAY_TIME)
            self.logger.info("Elapsed: {}".format(avts.elapsed))
            assert avts.expiry, '{} did not stay in "Playing" state while playing for at least {} secs.'.format(
                zp, CodecAudioTests.MAXIMUM_PLAY_TIME)
            assert not avts.state_changed, "{}'s TransportState changed unexpectedly expected after {} secs.".format(
                zp, avts.elapsed)
            self.logger.info("ZP {} stop local music playback.".format(zp))
            zp.AVTransport.Stop_and_wait()
            file_num += 1

[docs]    def verify_file(
            self,
            bit_rate,
            sample_rate,
            channels,
            bit_mode,
            format,
            file_path):
        """
        Verify an audio files' parameters using the media info command line tool.
        :param: bit_rate: bit rate of audio file
        :type: bit_rate: :obj: `str`
        :param: sample_rate: sample rate of audio file
        :type: sample_rate: :obj: `str`
        :param: channels: number of channels of file (mono or stereo)
        :param: bit_mode: variable or constant bit rate mode
        :param: format: audio file format
        :param: file_path: path to the file in the _test_data/test_files directory.
        :return: Boolean value representing if every parameter matches the output in mediainfo
        """
        # retrieve real values from media info command line
        bit_rate_output = subprocess.Popen(
            'mediainfo --Inform="Audio;%BitRate%" ' +
            file_path,
            stdout=subprocess.PIPE,
            shell=True)
        mediainfo_bit_rate = bit_rate_output.stdout.read()

        chan_output = subprocess.Popen(
            'mediainfo --Inform="Audio;%Channel(s)%" ' +
            file_path,
            stdout=subprocess.PIPE,
            shell=True)
        mediainfo_channels = chan_output.stdout.read()

        bit_mode_output = subprocess.Popen(
            'mediainfo --Inform="Audio;%BitRate_Mode%" ' +
            file_path,
            stdout=subprocess.PIPE,
            shell=True)
        mediainfo_bit_mode = bit_mode_output.stdout.read()

        sample_rate_output = subprocess.Popen(
            'mediainfo --Inform="Audio;%SamplingRate%" ' +
            file_path,
            stdout=subprocess.PIPE,
            shell=True)
        mediainfo_sample_rate = sample_rate_output.stdout.read()

        format_output = subprocess.Popen(
            'mediainfo --Inform="Audio;%Format%" ' +
            file_path,
            stdout=subprocess.PIPE,
            shell=True)
        mediainfo_format = format_output.stdout.read()

        format_profile_output = subprocess.Popen(
            'mediainfo --Inform="Audio;%Format_Profile%" ' +
            file_path,
            stdout=subprocess.PIPE,
            shell=True)

        mediainfo_format_profile = format_profile_output.stdout.read()

        if bit_mode == 'CBR':
            if format == 'LC' or format == 'HE-AAC / LC' or format == 'HE-AACv2 / HE-AAC / LC' or format == 'PCM':
                pass
            else:
                if bit_rate + '\n' != mediainfo_bit_rate:
                    print file_path + ' did not return the correct bit rate.'
                    return False
        if sample_rate + '\n' != mediainfo_sample_rate:
            print file_path + 'did not return the correct sample rate: '
            return False
        if channels + '\n' != mediainfo_channels:
            print file_path + 'did not return the correct number of channels.'
            return False
        if bit_mode + '\n' != mediainfo_bit_mode:
            print file_path + 'did not return the correct bit rate mode.'
            return False
        if format == 'LC' or format == 'HE-AAC / LC' or format == 'HE-AACv2 / HE-AAC / LC':
            if format + '\n' != mediainfo_format_profile:
                print file_path + 'did not return the correct format'
                return False
        else:
            if format + '\n' != mediainfo_format:
                print file_path + 'did not return the correct format.'
                return False
        return True

    def test_mp3_playback(self, zp):
        """
        Test mp3 playback for all supported sample rates in mono and stereo. Constant bitrate and variable
        bitrate. Uses LAME encoder.
        For variable bitrate, default quality is 4. LAME converts to joint-stereo (mid-side).

        :param zp: Sonos Zone Component object
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`

        :return: None
        """

        dir_path = CodecAudioTests.TEST_PATH
        mp3_test_files = {}

        # constant bit rate
        for sample_rate in CodecAudioTests.SAMPLE_RATES:
            for bit_rate in CodecAudioTests.MP3_BIT_RATES:
                #if the sample rate is lower than 32000, libmp3lame only encodes up to 160kpbs
                #found through trial and error... no documentation for this
                if int(bit_rate) >= 192000 and int(sample_rate) <= 24000:
                    pass
                else:
                    cmd = '{} -i {}/original_mono.wav -codec:a libmp3lame -b:a {} -ar {} mono_{}{}.mp3'.format(self.FFMPEG_PATH,
                        dir_path, bit_rate, sample_rate, sample_rate, bit_rate)
                    subprocess.check_call(cmd, shell=True)
                    self.verifyTrueOrStop(
                        self.verify_file(
                            bit_rate,
                            sample_rate,
                            '1',
                            'CBR',
                            'MPEG Audio',
                            'mono_' +
                            sample_rate +
                            bit_rate +
                            '.mp3'),
                        "mono_{}{} should have {} sample rate and {} bit rate".format(
                            sample_rate,
                            bit_rate, sample_rate, bit_rate))
                    mp3_test_files['mp3monocbr {} {}'.format(
                        sample_rate, bit_rate)] = 'test_files/mono_{}{}.mp3'.format(sample_rate, bit_rate)

                                   bit_rate] = 'test_files/stereo_' + sample_rate + bit_rate + '.mp3'
        # variable bit rate
        for sample_rate in CodecAudioTests.SAMPLE_RATES:
            for quality in CodecAudioTests.MP3_VBR_QUALITY:

                cmd = '{} -i {}/original_mono.wav -codec:a libmp3lame -ar {} -q:a {} mono_{}{}.mp3'.format(self.FFMPEG_PATH,

                    sample_rate, ext, sample_rate))

            cmd = '{} -i {}/original_stereo.wav{}-ar {} stereo_{}{}'.format(self.FFMPEG_PATH,

    def test_aac_playback(self, zp):
        """
        verifies playback for local AAC files.

        Uses ffmpeg libfdk-aac and fdkaac encoders.

        ffmpeg with libfdk-aac only support CBR at specific sampling rate + bit rate
        combinations. It also supports very few parameters for VBR, so I used a command line
        front-end of libfdk-aac to handle VBR. I used the same method for heaac and heaacv2.

        :param zp: Sonos Zone Component object
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`

        :return: None
        """

        dir_path = CodecAudioTests.TEST_PATH
        aac_test_files = {}

        # constant bit rate
        for bit_rate in CodecAudioTests.AAC_BIT_RATES:

            # libfdk aac only supports certain sample rates with certain
            # bitrates.

            sample_rate = ''

            if int(bit_rate) < 32000:
                sample_rate = '16000'
            elif 32000 <= int(bit_rate) < 40000:
                sample_rate = '22050'
            elif 40000 <= int(bit_rate) < 96000:
                sample_rate = '32000'
            elif 96000 <= int(bit_rate):
                sample_rate = '44100'

            if int(bit_rate) <= 256000:

                #I found that when encoding from mono to stereo with libfdk aac, bit rate and bit rate mode
                # does not come back as suspected if bit rate is larger than 256000.

                cmd = '{} -i {}/original_mono.wav -codec:a libfdk_aac -b:a {} -ar {} mono_{}{}.mp4'.format(self.FFMPEG_PATH,
                    dir_path, bit_rate, sample_rate, sample_rate, bit_rate)
                subprocess.check_call(cmd, shell=True)
                self.verifyTrueOrStop(
                    self.verify_file(
                        bit_rate, sample_rate, '2', 'CBR', 'LC', 'mono_{}{}.mp4'.format(
                            sample_rate, bit_rate)), 'mono_{}{}.mp4 should have {} bit rate and  {} sample rate'.format(
                        sample_rate, bit_rate, bit_rate, sample_rate))
                aac_test_files['aacmonocbr ' + sample_rate + \
                    bit_rate] = 'test_files/mono_' + sample_rate + bit_rate + '.mp4'

            cmd = '{} -i {}/original_stereo.wav -codec:a libfdk_aac -b:a {} -ar {} stereo_{}{}.mp4'.format(self.FFMPEG_PATH,
                dir_path, bit_rate, sample_rate, sample_rate, bit_rate)
            subprocess.check_call(cmd, shell=True)
            self.verifyTrueOrStop(
                self.verify_file(
                    bit_rate, sample_rate, '2', 'CBR', 'LC', 'stereo_{}{}.mp4'.format(
                        sample_rate, bit_rate)), 'stereo_{}{}.mp4 should have {} bit rate and {} sample rate'.format(
                    sample_rate, bit_rate, bit_rate, sample_rate))
            aac_test_files['aacstereocbr ' + sample_rate + \
                bit_rate] = 'test_files/stereo_' + sample_rate + bit_rate + '.mp4'

        # variable bit rate
        for sample_rate in CodecAudioTests.SAMPLE_RATES:

            # fdkaac doesn't have a sample rate parameter, so I am resampling
            # PCM files to use as source file
            cmd = '{} -i {}/original_mono.wav -ar {} mono_{}.wav'.format(self.FFMPEG_PATH,
                dir_path, sample_rate, sample_rate)
            subprocess.check_call(cmd, shell=True)
            cmd = '{} -i {}/original_stereo.wav -ar {} stereo_{}.wav'.format(self.FFMPEG_PATH,
                dir_path, sample_rate, sample_rate)
            subprocess.check_call(cmd, shell=True)

            for quality in CodecAudioTests.AAC_VBR_QUALITY:

                cmd = '{} -m{} mono_{}.wav -o mono_{}.mp4'.format(self.FDKAAC_PATH,
                    quality, sample_rate, quality)
                subprocess.check_call(cmd, shell=True)
                self.verifyTrueOrStop(
                    self.verify_file(
                        '',
                        sample_rate,
                        '1',
                        'VBR',
                        'LC',
                        'mono_{}.mp4'.format(quality)),
                    'mono_{}.mp4 should have {} sample rate'.format(quality, sample_rate))
