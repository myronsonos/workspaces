    def _set_up_test(self, config, ht_master, left, right):
        """
        1. Cleans up bonded configs
        2. Creates HT config
        3. Disables speaker's audio
        4. Enable sub and surrounds before disable test (see SWPBL-126879)
        :param str config: 5.0 or 5.1
        :param :class:`sonos.client.internal.SonosZoneComponent` ht_master:
        :param :class:`sonos.client.internal.SonosZoneComponent` left:
        :param :class:`sonos.client.internal.SonosZoneComponent` right:
        """
        for zp in [ht_master, left, right, self.sub]:
            zp.DeviceProperties.cleanup_bonded_configuration()
        if config == '5.0':
            ht_master.DeviceProperties.create_50_config(left, right)
        elif config == '5.1':
            ht_master.DeviceProperties.create_51_config(self.sub, left, right)
        for zp in [ht_master, self.sub, left, right]:
            zp.diag.wait_for_status_page()
            zp.disable_speaker_audio()
        track = ht_master.ContentDirectory.get_track_uri_for_filename(
            'MaggieFarm.mp3')
        ht_master.AVTransport.add_track_multiple_times_to_queue(track, 10)
        ht_master.AVTransport.use_queue()
        ht_master.RenderingControl.set_sub_enable(True)
        ht_master.RenderingControl.set_surround_enable(True)

    @retry(attempts=3, expected_exception=TimeoutError, retry_delay=5)
    def _verify_start_playback_via_sat_button(self, ht_master, sat):
        """
        Verify start of playback via play-pause/mute button press on satellite.
        Retry added for case where playback may not start if HT master has
        not yet reconnected to satellite, in which case, the button press on
        satellite results in no action.
        :param :class:`sonos.client.internal.SonosZoneComponent` ht_master:
        ht master device
        :param :class:`sonos.client.internal.SonosZoneComponent` sat:
        satellite on which button press is performed
        """
        sat.cli.volume_mute()
        wait_until_true(
            lambda: ht_master.AVTransport.is_playing(), iteration_delay=1,
            timeout_seconds=10)

    def _verify_sat_buttons(self, ht_master, sat, sat_enabled):
        """
        Verifies satellite buttons can play/pause/resume/next/previous
        :param :class:`sonos.client.internal.SonosZoneComponent` sat:
        """
        # Play
        self._verify_start_playback_via_sat_button(ht_master, sat)
        pos = ht_master.AVTransport.get_queue_position()
        self.verifyEqualOrFailCase(1, pos, 'Queue position should be 1')
        # Pause
        guaranteed_sleep(WAIT_BETWEEN_BUTTON_PRESSES)
        sat.cli.volume_mute()
        wait_until_true(
            lambda: ht_master.AVTransport.is_paused(), iteration_delay=1,
            timeout_seconds=10)
        self.verifyFalseOrFailCase(
            ht_master.AVTransport.is_playing(), 'Should be paused')
        # Resume
        guaranteed_sleep(WAIT_BETWEEN_BUTTON_PRESSES)
        sat.cli.volume_mute()
        wait_until_true(
            lambda: ht_master.AVTransport.is_playing(), iteration_delay=1,
            timeout_seconds=10)
        self.verifyTrueOrFailCase(
            ht_master.AVTransport.is_playing(), 'Should be playing')
        if sat_enabled:
            # Next
            guaranteed_sleep(WAIT_BETWEEN_BUTTON_PRESSES)
            sat.cli.spoof_track_changes('next')
            wait_until_true(
                lambda: ht_master.AVTransport.get_queue_position() == 2,
                iteration_delay=1, timeout_seconds=10)
            pos = ht_master.AVTransport.get_queue_position()
            self.verifyEqualOrFailCase(2, pos, 'Queue position should be 2')
            # Previous
            guaranteed_sleep(WAIT_BETWEEN_BUTTON_PRESSES)
            sat.cli.spoof_track_changes('previous')
            wait_until_true(
                lambda: ht_master.AVTransport.get_queue_position() == 1,
                iteration_delay=1, timeout_seconds=10)
            pos = ht_master.AVTransport.get_queue_position()
            self.verifyEqualOrFailCase(1, pos, 'Queue position should be 1')
        # Pause
        guaranteed_sleep(WAIT_BETWEEN_BUTTON_PRESSES)
        sat.cli.volume_mute()
        wait_until_true(
            lambda: ht_master.AVTransport.is_paused(), iteration_delay=1,
            timeout_seconds=10)
        self.verifyFalseOrFailCase(
            ht_master.AVTransport.is_playing(), 'Should be paused')

    def _is_remote_requested(self, ht_master, block):
        """
        Is remoteRequested on LRSwitcher, RRSwitcher, or SWSwitcher?
        (Are satellites enabled? Query left rear satellite / right rear /
        subwoofer)
        :param :class:`sonos.client.internal.SonosZoneComponent` ht_master:
        :param str block: LRSwitcher, RRSwitcher, SWSwitcher
        :return: bool is remoteRequested?
            True = satellite enabled
            False = satellite disabled
        """
        switcher_block = ht_master.diag.get_dsp_block(block)
        remote_requested = switcher_block.find(
            './/parameter/param[1]').attrib['value']
        return remote_requested == 'Yes'

    def _is_satellite_dsp_enabled(self, ht_master, sat):
        """
        Is satellite DSP enabled?
        :param :class:`sonos.client.internal.SonosZoneComponent` ht_master:
        :param :class:`sonos.client.internal.SonosZoneComponent` sat:
        :return: bool is satellite enabled?
        """
        if sat.is_left_channel():
            state = self._is_remote_requested(ht_master, 'LRSwitcher')
        # Did not use is_right_channel because it requires the left ZP
        if not sat.is_left_channel() and not sat.is_sub():
            state = self._is_remote_requested(ht_master, 'RRSwitcher')
        if sat.is_sub():
            state = self._is_remote_requested(ht_master, 'SWSwitcher')
        return state

    def _is_satellite_disabled(self):
        """
        Has the satellite_enabled event not been set?
        :return: bool is satellite disabled?
        """
        return not self.satellite_enabled.isSet()

    def _enable_udp_log(self, sats):
        """
        Start or stop udp log on satellites
        :param class:`sonos.client.internal.SonosZoneComponent` sat:
        :param mode: True - to enable udp log
                     False - to disable udp log
        """
        for sat in sats:
            # SWPBL-107281
            for module in DIAG_LEVELS:
                sat.diag.set_diag_level(module, DIAG_LEVELS[module])
            sat.diag.start_udp_log()
        guaranteed_sleep(WAIT_AFTER_UDP_LOG_START)

    def _disable_udp_log(self, sats):
        """
        Stop udp log on satellites
        :param class:`sonos.client.internal.SonosZoneComponent` sat:
        """
        for sat in sats:
            sat.diag.stop_udp_log()

    def _verify_satellite_enable(self, ht_master, sats, sat_enabled):
        """
        Sets sat or SUB state & verifies using DSP block & anacapa.trace
        1. Set watch string
        2. Enable/disable sat or SUB
        3. Verify remoteRequested DSP value
        4. Verify watch string was logged
        :param :class:`sonos.client.internal.SonosZoneComponent` sat:
        """
        for sat in sats:
            # Set watch string
            if sat_enabled:
                self.satellite_enabled.clear()
                watch_string = 'satellite audio disabled 0'
                sat.diag.watch_udp_log_for_string(
                    watch_string,
                    lambda msg: self.satellite_enabled.set())
            else:
                self.satellite_enabled.set()
                watch_string = 'satellite audio disabled 1'
                sat.diag.watch_udp_log_for_string(
                    watch_string,
                    lambda msg: self.satellite_enabled.clear())
        # Start playback to verify DSP block
        ht_master.AVTransport.Play_and_wait()
        # Disable sats/SUB
        for sat in sats:
            if sat.is_sub():
                ht_master.RenderingControl.set_sub_enable(sat_enabled)
        ht_master.RenderingControl.set_surround_enable(sat_enabled)

        for sat in sats:
            dsp_state = self._is_satellite_dsp_enabled(ht_master, sat)
            # B. Dizon: for the SUB the `remoteRequested` value is always `Yes` if
            # a SUB is bonded, and when disabled from the UI I zero that input by
            # configuring the SUB bass manager to output zeros
            if not sat.is_sub():
                self.verifyEqualOrFailCase(
                    sat_enabled, dsp_state,
                    'DSP enable should be <{}> on <{}>'.format(sat_enabled, sat))
            # Verify watch string was logged
            if sat_enabled:
                wait_until_true(
                    lambda: self.satellite_enabled.isSet(), iteration_delay=1,
                    timeout_seconds=60)
                self.verifyTrueOrStop(
                    self.satellite_enabled.isSet(),
                    '<{}> Satellite enabled should be True'.format(sat))
            else:
                wait_until_true(
                    lambda: self._is_satellite_disabled(), iteration_delay=1,
                    timeout_seconds=60,
                    reason='Timeout waiting for <{}> on <{}>'.format(
                        watch_string, sat))
                self.verifyFalseOrStop(
                    self.satellite_enabled.isSet(),
                    '<{}> Satellite enabled should be False'.format(sat))

            sat.diag.stop_watching_udp_log_for_string(watch_string)

        ht_master.AVTransport.Stop_and_wait()

    def test_disable_enable_sats_sub(self, ht_master, left, right):
        """
        Verifies disabled/enabled satellites & SUB

        2. Verify Play/Pause/Resume button press on satellites

        self.verify_sat_enable_disable(ht_master, [self.sub, left, right], True)
        self.verify_sat_buttons(ht_master, [left, right], True)
