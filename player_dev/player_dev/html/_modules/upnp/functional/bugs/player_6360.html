    """
    Verifies the Sonos device can recover from a segmentation violation
    signal or an abort signal.
    The expected behavior should be that when an anacapa thread encounters
    or receives one of those signals, anacapa should recover and restart
    with minimum disruption in service.
    The test verifies PLAYER-6360 - intermittent zombie processes on PPC
    """
    TEST_TYPE = ["SETTINGS"]
    """
    Platforms which don't have backtrace support
    """
    NO_BACKTRACE_SUPPORT = (ZB100, BR200, WD100,)
    """
    Zone player platforms (ppc, sh) running Linux 2.4.x
    """
    PPC_SH_ZP = (S5, ZP80, ZP120, ZP100, ZP90,)
    """
    Vector signals to send to these threads.
    """
    TIMEDOBJ_THREADS = ('timedjobmgr', 'timedjobmgr_btn', 'timedjobmgr_ac',)
    """
    WD100, BOOST don't not have timedjobmgr_btn and timedjobmgr_oc threads
    """
    WD100_BOOST = (WD100, BR200,)
    WD100_TIMEDOBJ_THREADS = ('timedjobmgr',)

    THREADINFO_ANACAPA_THREADS = ('anacapa',)

    TERM_SIGNALS = ('-ABRT', '-SEGV',)

        server = AutomationServer(os.environ["AUTOMATION_USER"], os.environ["AUTOMATION_TOKEN"])

        node.set_offline(cgi.escape(reason))



[docs]    def termination_signal_generator(self):
        """
        Generates a test vector based on the platform architecture.
        This generator will generate the zone player, the thread name,
        the signal which will be used to kill the thread and if it has backtrace support.

        .. note::

        Note that each device should only ever be sent a signal once as subsequent
        signal doesn't create a new backtrace file unless:
        1. ip:1400/support/aggregate was called
        2. backtrace file is zero bytes or not present when anacapad starts up
        Hence all we do here is randomized the signal type and the process to kill
        per device.
        """
        for zp in self.generators.generate_testbed_unique_devices():
            if zp.modelNumber in ZombieProcesses.WD100_BOOST:
                yield (zp, random.choice(ZombieProcesses.WD100_TIMEDOBJ_THREADS),
                       random.choice(ZombieProcesses.TERM_SIGNALS))
            elif zp.modelNumber in ZombieProcesses.PPC_SH_ZP:
                if not zp.cli.is_sh4():
                    yield (zp, random.choice(ZombieProcesses.TIMEDOBJ_THREADS),
                           random.choice(ZombieProcesses.TERM_SIGNALS))
            elif zp.modelNumber == ZB100:
                yield (zp, 'anacapad', random.choice(ZombieProcesses.TERM_SIGNALS))
            else:
                yield (zp, random.choice(ZombieProcesses.THREADINFO_ANACAPA_THREADS),
                       random.choice(ZombieProcesses.TERM_SIGNALS))


    def __backtrace_manual_remove(self, devices):
        for zp in devices:
            self.logger.debug("Clearing backtrace for {} manually.".format(zp))
            zp.cli.clear_backtrace()

    def __forced_exit_and_disable_node(self, msg):
        try:
            self.logger.critical(msg)
            # We really should be exposing all command line options in any test script and fixtures.
            self.disable_node(os.environ["NODE_NAME"], msg)
        finally:
            # Intentional exit with no teardown whatsoever and we also want to disable the test bed.
            # This is because there will not be any /jffs/backtrace on the device whereby anacapad ignored
            # the signal...hence downstream project that checks for backtraces will not catch it.
            raise SystemExit("Stopping test and exiting from all discovery.")

    def __clean_dmesg_and_backtrace(self):
        """
        Clears the dmesg output and removes backtrace files.
        Called on test tear down.
        """
        # SWPBL-54687: Clear all dmesg's for all ZPs
        for zp in self.my_devices:
            zp.cli.clear_dmesg()
        self.__backtrace_manual_remove(self.my_devices)

[docs]    def tearDownFixture(self):
        """
        By sumbitting a diagnostic the devices which support
        backtrace, clear the contents of the backtrace file
        as part of the submission.
        This is done at the end of the test so that subsequent
        tests will be able to write to the backtrace file in the
        event anacapad generates a backtrace.
        """
        cfg = load_workflow_execution_configuration_from_EXECUTION_ENVIRONMENT_or_None()
        exec_env_uuids = map(lambda x: x.serial_number, cfg.get_testbed_zone_players())

        # We allow retry 2 times at 10 secs apart
        i = 0
        retry = 2
        retry_wait_secs = 10
        while i < len(self.my_devices):
            d = self.my_devices[i]
            try:
                zgs = d.ZoneGroupTopology.GetZoneGroupState()
                self.logger.debug("GetZoneGroupState of {}: {}".format(d, zgs))
                zgs_tree = ET.fromstring(zgs['ZoneGroupState'])
                missing, _ = unorderable_list_difference(exec_env_uuids,
                                                         map(lambda x: x.attrib['Coordinator'][len('RINCON_'):-5],
                                                             zgs_tree.find('ZoneGroups')))
            except Exception as e:
                self.__clean_dmesg_and_backtrace()
                self.stop_suite(
                    "Caught Exception while ZoneGroupTopology.GetZoneGroupState() from {}: <{}>".format(d, e))
            else:
                if missing and retry:
                    msg = "Missing devices {} in ZoneGroupTopology of {} but will retry...".format(missing, d)
                    self.logger.info(msg)
                    retry -= 1
                    time.sleep(retry_wait_secs)
                    continue
                elif missing:
                    self.__forced_exit_and_disable_node(
                        "Missing devices {} in ZoneGroupTopology of {}.".format(missing, d))
                i += 1

        try:
            diag_id = self.my_devices[0].ZoneGroupTopology.SubmitDiagnostics(
                include_controller=False)
            self.logger.debug("Submitted DIAG from <{}>. Diag ID: {}"
                              .format(self.my_devices[0], diag_id))
        except Exception as e:
            self.__clean_dmesg_and_backtrace()
            self.stop_suite("Caught Exception while trying to submit diagnostics "
                            "from {}: <{}>".format(self.my_devices[0], e))
        else:
            for zp in self.my_devices:
                if zp.cli.has_backtrace():
                    self.stop_suite("Diag submission (id: {}) didn't clear backtrace for {}".format(diag_id, zp))
        finally:
            self.__clean_dmesg_and_backtrace()

    def test_zombie(self, zp, thr_name, signal):
        """
        Acquires the associated process identifier of
        the thread name and sends the desired signal.
        Waits for anacapad to restart and confirms the
        boot sequence has increased.

        :param obj zp:`sonos.client.internal.SonosZoneComponent`
        :param str thr_name: Anacapa thread name
        :param str signal: Unix signal to send to anacapa thread
        """
        if zp.modelNumber in SUPPORTS_BREAKPAD:
            self.skip("Skipping test on model {} until mini dump support is added"
                      " to this test".format(zp.modelNumber))
        proc_dict = zp.diag.anacapa_thread_to_pid_map()

        if thr_name not in proc_dict:
            self.__forced_exit_and_disable_node(
                "Thread name {} not found in process list/threadinfo for {}.".format(
                    thr_name, zp))
        else:
            self.logger.info("{} {}".format(thr_name, proc_dict[thr_name]))

            # all of TERM_SIGNALS are fatal, so we are waiting for the startup byebye
            # SWPBL-54687: ZP90PPC, marginally failed when waiting for 5 secs, hence changed to 10 secs
            result = zp.terminate_thread_and_wait_for_anacapad(
                signal=signal,
                pid=proc_dict[thr_name],
                wait=False, byebye_timeout_secs=60, alive_timeout_secs=10)
            if not result:
                self.__forced_exit_and_disable_node("anacapad ignores signal for ZP {}.".format(zp))
            else:
                if zp.modelNumber not in ZombieProcesses.NO_BACKTRACE_SUPPORT:
                    if not zp.cli.has_backtrace():
                        self.__forced_exit_and_disable_node(
                            "anacapad didn't create backtrace file for ZP {}.".format(zp))
                    else:
                        if not zp.cli.has_backtrace(backtrace=TMP_BACKTRACE_FILE):
                            self.__forced_exit_and_disable_node(
                                "anacapad didn't create temp backtrace file for ZP {}.".format(zp))
                else:
                    if zp.cli.has_backtrace():
                        self.__forced_exit_and_disable_node(
                            "anacapad shouldn't create backtrace file for ZP {}.".format(zp))
