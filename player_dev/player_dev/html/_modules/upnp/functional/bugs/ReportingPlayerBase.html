[docs]    def setup_trackplay_monitor_diag(self, zp):
        """
        Prepare the udp logger and the track playtime detection 
        callback.
        """
        self.logger.debug("setting up database UDPLogger")
        self.clear_detected_flags()
        zp.diag.start_udp_log()
        zp.diag.set_diag_level(*self.CHSRC_DIAG_LEVEL)
        zp.diag.set_diag_level(*self.CHSNK_DIAG_LEVEL)
        zp.diag.set_diag_level(*self.SONOSCP_DIAG_LEVEL)
        zp.diag.set_diag_level(*self.TRACK_PLAY_MONITOR_DIAG_LEVEL)
        self.track_playtime_monitor_cb = lambda msg: self.detected_track_playtime_end(msg)
        '''
        Disable crossfade mode because setting the next track as playback is ending
        with crossfade on sometimes catches the start of the next track which
        then skips the intended next track.
        '''
        zp.AVTransport.SetCrossfadeMode(False)
        self.tolerance = 0

[docs]    def detected_track_playtime_end(self, msg):
        """
        endPlayingTrack message detected. 
        set flag then parse expected play time and 
        Sonos calculated playtime.
        """
        self.logger.debug("caught endPlayingTrack message on DUT (msg={})".format(msg))
        self.track_playtime_ended = True
        # Sometimes there is a [date time] value in [] which are regex special characters.
        if ']' in msg:
            msg_parts = msg.split(']')
            if len(msg_parts):
                msg = msg_parts[1]
        self.expected_time = self.get_expected_playtime(msg)
        self.sonos_time = self.get_sonos_playtime(msg)

        

[docs]    def compare_playtime_within_tolerance(self, expected_playtime, reported_playtime, tolerance):
        '''
        If the reported playtime is greater than the expected playtime then 
        the difference between them should be within the specified tolerance value. 
        This
        Arguments: 
            expected_playtime - Expected play time in seconds.
            reported_playtime - Reported play time in seconds
            tolerance - An allowable difference in seconds
        Return: True if within tolerance else false.
        '''
        if (expected_playtime >= reported_playtime):
            return True

        if (reported_playtime >= expected_playtime):
            if tolerance >= (reported_playtime-expected_playtime):
                return True
        
        self.logger.debug("Actual track playtime was not within the calculated tolerance of the expected playtime. expected time:{}; actual time:{}, tolerance:{}"
                      .format(expected_playtime,reported_playtime,tolerance))
        return False

        

[docs]    def get_sonos_playtime(self, msg):
        '''
        Parse Sonos reported playtime from trackplaymonitor module
        via udp logger.
        '''
        match_groups = re.match(r'.*PlayTime: (\d*).*',msg) 
        if match_groups:
            playtime = match_groups.group(1)
            return int(playtime)
        else:
            return -1

        

[docs]    def get_expected_playtime(self, msg):
        '''
        Parse Track Metadata reported playtime from trackplaymonitor module
        via udp logger.
        '''
        match_groups = re.match(r'.*,(\d*).*',msg)
        if match_groups:
            expected_playtime = re.sub('\);','', match_groups.group(1))
            return int(expected_playtime)
        else:
            return -1

        

[docs]    def calculate_track_delta_tolerance(self, expected_track_time):
        """
        Calculate the delta tolerance time for the expected track length
        Input:
            expected_track_time number of seconds for expected track time.
        Return:
            tolerance: Integer number of seconds
        """
        tolerance = (int(expected_track_time) * self.TRACK_PLAYTIME_DELTA_FACTOR) + 1
        self.logger.debug("calculate tolerance {}; expected_track_time {} ".format(tolerance, expected_track_time))
        return tolerance

        '''
        Return string time as timedelta object
        '''
        x = time.strptime(strtime.split(',')[0],'%H:%M:%S')
        dt_delta = datetime.timedelta(hours=x.tm_hour,minutes=x.tm_min,seconds=x.tm_sec)
        return dt_delta
