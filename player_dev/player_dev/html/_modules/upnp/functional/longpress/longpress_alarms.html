        """
        Yields a player under test, a player to fire an alarm and
        a list of players to cast audio source and whether to create
        an alarm using the buzzer URI or a programmed URI.
        """
        for player in self.generators.generate_playback_device_per_platform():
            sources = self.test_players[:]
            sources.remove(player)
            # Shuffle remaining list of players and create
            # player sources based on run time configuration
            random.shuffle(sources)
            alarm_zone = sources[0]
            sources.remove(alarm_zone)
            for is_buzzer_alarm in (False, True):
                yield (
                    (player, alarm_zone, sources[:self.num_audio_sources]),
                    is_buzzer_alarm
                    )

[docs]    def setup_sources(self, test_zones):
        """
        Start persistent log on all test zones.
        Assign play URI to player under test and sources.
        Start playback on sources, verifying on player and potential
        alarm source.
        Start playback on player, verifying on sources.
        Return a tuple of the last played rotation list and a
        dictionary of player to URI mapping.
        :param tuple test_zones: Player under test, alarm zone and
        audio sources.
        """
        player, alarm_zone, sources = test_zones
        players = [player] + sources
        [self.start_persistent_log(zp) for zp in self.zones_from_generator(test_zones)]
        player_to_uri = self.assign_uri_to_players(players)
        rotation = [zp for zp in sources
                    if self.start_playback(
                            zp,
                            [player],
                            uri=player_to_uri[zp])]
        ret = self.start_playback(
            player, sources, uri=player_to_uri[player])
        self.verifyTrueOrStop(
            ret, "Expect player <{}> to be playing".format(player))
        return rotation, player_to_uri[player]



[docs]    def create_alarm(self, alarm_zone, is_buzzer_alarm):
        """
        Creates an alarm on the player to fire in
        SECONDS_IN_FUTURE_TO_RUN_ALARM
        Returns a tuple of alarm ID and alarm data
        :param :class:`sonos.client.internal.SonosZoneComponent` alarm_zone:
        :param :obj:`bool` is_buzzer_alarm: Is this the buzzer alarm ?
        :return: Return a tuple of :obj:`str` alarm ID
        and :class:~sonos.common.XMLAccessor alarm data
        :rtype :obj:`tuple`
        """
        alarm = self.alarm_manager.create_default_alarm()
        alarm.volume = 5
        if not is_buzzer_alarm:
            alarm.program_uri = self.assign_uri_to_players([alarm_zone])[alarm_zone]
        alarm.udn = alarm_zone.raw_udn
        alarm.run_in_seconds(SECONDS_IN_FUTURE_TO_RUN_ALARM)
        self.logger.debug("Alarm ID: {}".format(alarm.alarm_id))
        self.verifyTrueOrStop(alarm.enabled,
            "Expect Alarm ID {} to be Enabled".format(alarm.alarm_id))
        return alarm.alarm_id, alarm



[docs]    def wait_for_alarm_to_fire(self, alarm_zone, alarm_id, zones_to_verify):
        """
        Confirm the alarm has fired and the last played event (alarm firing)
        diagnostic message on the zones which are to receive the event.
        Stops test when the alarm has not fired.
        :param :class:`sonos.client.internal.SonosZoneComponent` alarm_zone:
        :param str alarm_id:
        :param list :class:`sonos.client.internal.SonosZoneComponent` zones_to_verify:
        """
        last_played, serial_num_to_zp = self.register_watch(
            alarm_zone,
            zones_to_verify,
            LAST_PLAYED_REGEX_MSGS,
            self.cb_confirm_last_played)
        try:
            wait_until_true(
                lambda: alarm_zone.AVTransport.is_alarm_running(),
                timeout_seconds=SECONDS_IN_FUTURE_TO_RUN_ALARM,
                iteration_delay=1.0,
                start_delay=START_DELAY)
        except TimeoutError:
            self.stop(
                "Stop,  expected alarm {} is not running".format(alarm_id))
        try:
            wait_until_true(lambda: not last_played,
                            iteration_delay=1.0,
                            timeout_seconds=LAST_PLAYED_TIMEOUT_SEC)
        except TimeoutError:
            for serial_num in last_played.keys():
                self.logger.error("Player <{}> did not receive expected event"
                                  "of start playback on <{}>".format(
                                      serial_num_to_zp[serial_num], alarm_zone))
            self.stop("Stopping due to failure to receive expected start play event")
        self.unregister_watch(zones_to_verify, LAST_PLAYED_REGEX_MSGS)
        self.logger.info(
            "Alarm ID {} has fired on <{}>".format(alarm_id, alarm_zone))



[docs]    def pause_programmed_alarm(self, test_zones, alarm_data):
        """
        Confirms programmed alarm is paused and resumes on
        player under test via a long press.
        :param tuple :class:`sonos.client.internal.SonosZoneComponent` test_zones:
        :param :class:`sonos.client.alarms.Alarm` alarm_data:
        """
        player, alarm_zone, sources = test_zones
        self.logger.info(
            "Pausing alarm source <{}>".format(alarm_zone))
        self.pause_alarm_using_button_press(alarm_zone, [player] + sources)
        self.long_press_action(player, LongPressAction.STEAL_AVT)
        self.verifyEqualOrStop(
            alarm_zone.raw_udn,
            self.udn,
            "Expect <{}> to 'steal' paused/stopped content from <{}>".format(
                player, alarm_zone))
        self.verifyTrueOrFailCase(
            self.is_playing(player),
            "Expect <{}> to resume paused content".format(player))
        self.verifyEqualOrStop(
            u'x-rincon-queue:{}#0'.format(player.raw_udn),
            player.AVTransport.get_avtransport_uri(),
            "Player <{}> to match expected AVTransport URI".format(player))



[docs]    @combinatorial('generate_alarm_source_and_audio_source')
    def test_long_press_alarm(self, test_zones, is_buzzer_alarm):
        """
        Confirm long press functionality with a fired alarm.
        1. Start playback on sources and player under test (PUT)
        2. Verify long press on PUT
        3. Create and start alarm on specified alarm zone
        4. When alarm fires, verify long press on PUT plays the
        alarm.
        5. Pause all non alarm audio sources
        6. Stop/Pause playing alarm
        7a. Confirm buzzer alarm stopped and AVTransport cleared
        7b. Confirm programmed alarm paused and resumed on PUT
        on long press.
        :param tuple :class:`sonos.client.internal.SonosZoneComponent` test_zones:
        :param bool is_buzzer_alarm: Create buzzer alarm ?
        """
        player, alarm_zone, sources = test_zones
        self.skip_or_continue_with_test([player])
        rotation, player_uri = self.setup_sources(test_zones)
        self.verify_long_press(
            player, rotation[::-1], player, player_uri, invoke_play=False)
        self.resume_playback(player, sources)
        alarm_id, alarm_data = self.create_alarm(alarm_zone, is_buzzer_alarm)
        self.wait_for_alarm_to_fire(alarm_zone, alarm_id, [player] + sources)
        # Update the rotation to include the alarm zone as last played
        rotation.append(alarm_zone)
        self.verify_long_press(
            player, rotation[::-1], player, player_uri, invoke_play=False)
        # Pause all audio content except alarm zone
        self.logger.info(
            "Pausing/Stopping all audio sources except alarm")
        for source in sources:
            # Verify action on all test players
            verify_list = sources[:] + [player] + [alarm_zone]
            verify_list.remove(source)
            self.logger.info("Pausing/Stopping <{}>".format(source))
            self.pause_playback(source, verify_list)
        if is_buzzer_alarm:
            self.logger.info(
                "Stopping buzzer alarm source <{}>".format(alarm_zone))
            self.pause_buzzer_using_button_press(alarm_zone, [player] + sources)
        else:
            self.pause_programmed_alarm(test_zones, alarm_data)
