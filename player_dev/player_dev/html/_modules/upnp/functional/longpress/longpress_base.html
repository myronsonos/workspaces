[docs]    def extract_udn_from_msg(self, msg, regexs):
        """
        Return the first match on a search for the UDN in the
        message.
        :param str msg: The UDP message
        :param tuple regexs: A tuple of regular expressions
        to search
        :return: The UDN from the message
        :rtype :obj:`str`
        """
        for regex in regexs:
            match = re.search(regex, msg)
            if match:
                return match.group(1)
        return None



[docs]    def cb_confirm_generic(self, msg, zp, sn_to_udn_dict, regexs):
        """
        Removes the zone player from the dictionary when the player
        reports the expected UDN transition.
        :param str msg: The UDP message from the player
        :param :class:`sonos.client.internal.SonosZoneComponent` zp:
        :param dict sn_to_udn_dict: The dictionary of serials' to
        expected UDN
        :param tuple regexs: A tuple consisting of regular expressions
        to search for the matching UDN
        """
        self.logger.info("Report from {}".format(zp.serialNumber))
        actual_udn = self.extract_udn_from_msg(msg, regexs)
        expected_udn = sn_to_udn_dict[zp.serialNumber]
        if actual_udn != expected_udn:
            self.logger.debug(
                "Unexpected last played event "
                "{} from <{}> received {} expected {}".format(
                    msg, zp, actual_udn, expected_udn))
        else:
            _ = sn_to_udn_dict.pop(zp.serialNumber)

[docs]    def long_press(self, zp, watch, callback, timeout_sec=5):
        """
        Spoofs a button long press event using the cli interface
        and waits for the event to be posted by the callback
        which has been registered to watch for the given regular
        expression
        :param :class:`sonos.client.internal.SonosZoneComponent` zp:
        The zone player the spoofed long press will be performed
        :param str watch: The regular expression to watch in UDP log
        :param callback: The callback method when the regular expression
        is matched
        :param int timeout_sec: Maximum seconds to wait before giving
        up on the long press action.
        """
        event = self.longpress_event
        event.clear()
        self.udn = ""
        zp.diag.watch_udp_log_for_string(
            watch,
            lambda msg, (zp, event): callback(msg, zp, event), (zp, event))
        zp.cli.spoof_long_press()
        try:
            wait_until_true(
                lambda: self.longpress_event.isSet(),
                timeout_seconds=timeout_sec)
        except TimeoutError:
            self.stop("Stopping due to not detecting expected long press on <{}>".format(zp))
        finally:
            zp.diag.stop_watching_udp_log_for_string(watch)
            self.logger.info("<{}> cycling to {}".format(zp, self.udn))

[docs]    def pause_playback(self, zp, zones):
        """
        Pauses playback on the zone player and confirms the list of specified
        zones have noticed the last paused event.
        :param :class:`sonos.client.internal.SonosZoneComponent` zp:
        :param list :class:`sonos.client.internal.SonosZoneComponent` zones:
        :return: Has the zone player paused and the event noticed on all
        specified zones ?
        :rtype: :obj:`bool`
        """
        last_paused, serial_num_to_zp = self.register_watch(
            zp,
            zones,
            LAST_PAUSED_STOPPED_REGEX_MSGS,
            self.cb_confirm_last_paused)
        self.logger.debug(
            "Last paused dictionary {}".format(last_paused))
        try:
            zp.AVTransport.Pause_and_wait(timeout_seconds=15)
        except TimeoutError:
            self.stop("Stopping due to failure to pause playback on <{}>".format(zp))
        try:
            wait_until_true(lambda: not last_paused, timeout_seconds=10)
        except TimeoutError:
            for serial_num in last_paused.keys():
                self.logger.error("Player <{}> did not receive expected event"
                                  "of pause playback on <{}>".format(
                                      serial_num_to_zp[serial_num], zp))
            self.stop("Stopping due to failure to receive expected pause event")
        self.unregister_watch(zones, LAST_PAUSED_STOPPED_REGEX_MSGS)
        return True

[docs]    def register_watch(self, zp, zones, regexs, callback):
        """
        Sets up a call back when a player recognizes a last played event
        Returns a dictionary of the expected zones (serial number as key)
        which are expected to receive the last played event.
        :param :class:`sonos.client.internal.SonosZoneComponent` zp: The
        zone player which is expected to generate the event
        :param list :class:`sonos.client.internal.SonosZoneComponent` zones:
        The zones which are expected to receive the event
        :param tuple regexs: A tuple of regular expressions to set up UDP
        watch
        :return: A tuple of dictionaries.
        A dictionary of zones' serial number with the expected zone
        A dictionary of zones' serial number with the corresponding zone object
        raw UDN
        :rtype: :obj:`tuple`
        """
        # a dictionary of zone keys with expected UDN of player
        # playback commences
        last_played = {}
        [last_played.update({zone.serialNumber: zp.raw_udn}) for zone in zones]
        serial_num_to_zp = {}
        [serial_num_to_zp.update({zone.serialNumber: zone}) for zone in zones]
        self.logger.debug(
            "Last played dictionary {}".format(last_played))
        for zp in zones:
            for regex in regexs:
                zp.diag.watch_udp_log_for_string(
                    regex,
                    lambda msg, (zp, last_played): callback(msg, zp, last_played), (zp, last_played))
        return last_played, serial_num_to_zp



[docs]    def start_playback(self, zp, zones, uri=''):
        """
        Starts playback on the zone and confirms the zones
        have received the event the zone is playing.
        :param :class:`sonos.client.internal.SonosZoneComponent` zp:
        :param list :class:`sonos.client.internal.SonosZoneComponent` zones:
        :param str uri: Set the AVT URI when non-empty
        :return: True on success else stop.
        :rtype: :obj:`bool`
        """
        if uri:
            self.logger.debug("Setting URI {} on <{}>".format(uri, zp))
            zp.AVTransport.SetAVTransportURI(
                uri, zp.AVTransport.generate_didl_lite())
            zp.RenderingControl.SetVolume(5)
        last_played, serial_num_to_zp = self.register_watch(
            zp, zones, LAST_PLAYED_REGEX_MSGS, self.cb_confirm_last_played)
        self.logger.debug("Expecting to receive PLAYING/RESUME "
                          "event on {}".format(last_played.keys()))
        play_time = time.time()
        try:
            zp.AVTransport.Play_and_wait(timeout_seconds=15)
        except TimeoutError:
            self.stop(
                "Stopping due to failure to start playback on <{}>".format(zp))
        try:
            # Increased timeout from 10 seconds to over a minute.
            # Some players take close to a minute on some test runners.
            wait_until_true(lambda: not last_played,
                            timeout_seconds=LAST_PLAYED_TIMEOUT_SEC)
        except TimeoutError:
            for serial_num in last_played.keys():
                self.logger.error("Player <{}> did not receive expected event"
                                  "of start playback on <{}>".format(
                                      serial_num_to_zp[serial_num], zp))
            self.stop("Stopping due to failure to receive expected start play event")
        self.unregister_watch(zones, LAST_PLAYED_REGEX_MSGS)
        self.logger.debug(
            "All players received the "
            "PLAYING/RESUME event in {} sec".format(time.time() - play_time))
        return True

    resume_playback = start_playback

        :return: Does the zone player have the desired volume ?
        :rtype: :obj:`bool`
        """
        self.logger.debug("<{}> applying desired volume {} (set {})".format(
            zp, desired_volume, set_volume))
        if set_volume:
            zp.RenderingControl.SetVolume(desired_volume)
        try:
            wait_until_true(
                lambda: zp.RenderingControl.GetVolume() == desired_volume,
                start_delay=start_delay,
                timeout_seconds=15,
                iteration_delay=1)
        except TimeoutError:
            self.logger.error("<{}> failed to apply desired volume {} - JIRA SWPBL-92248".format(
                zp, desired_volume))
            return False
        else:
            self.logger.debug("<{}> applied desired volume {}".format(
                zp, desired_volume))
            return True

[docs]    def verify_long_press(self, test_player, longpress_rotation,
                          master, exp_uri, invoke_play=True):
        """
        Perform long press on the test player cycling through the expected
        playable content.
        The master is the AVTransport controlling player, for HT it's the
        master, for stand alone it's the test player.
        :param :class:`sonos.client.internal.SonosZoneComponent` test_player:
        :param list longpress_rotation:
        :class:`sonos.client.internal.SonosZoneComponent`
        :param :class:`sonos.client.internal.SonosZoneComponent` master:
        :param str exp_uri: Expected URI when test player exhausts rotation
        :param bool invoke_play: Should an AVTransport Play be invoked when
        player exhausts rotation
        """
        self.logger.debug(
            "Long press player under test <{}>".format(test_player))
        self.logger.debug(
            "Long press master <{}>".format(master))
        self.logger.debug(
            "Long press rotation list {}".format(longpress_rotation))
        self.logger.debug(
            "Long press expected URI when stand alone {}".format(exp_uri))
        for zp in longpress_rotation:
            if zp.serialNumber == test_player.serialNumber:
                self.logger.info("Skipping self <{}> in rotation".format(zp))
                continue
            # If we're playing content AND the test player is a legacy
            # device, do fast ramp down to volume 0 using button press.
            # A spoofed fast ramp down to volume 0 increases the code
            # coverage on the player under test
            if self.is_playing(master):
                self.verify_continuous_playing_state(master)
                if test_player.modelNumber not in CAP_TOUCH_DEVICES:
                    self.logger.debug(
                        "Issuing fast ramp down on <{}>".format(test_player))
                    self.wait_for_ramp_down_volume_zero_using_button(test_player)
                else:
                    self.wait_for_volume_change(master, 0, set_volume=True)
            else:
                self.wait_for_volume_change(master, 0, set_volume=True)
            self.logger.info("Long press to switch to {}".format(zp.raw_udn))
            self.long_press_action(test_player, LongPressAction.CYCLE)
            time.sleep(2)
            self.verifyEqualOrStop(
                zp.raw_udn,
                self.udn,
                "Expect player under test to cycle to next playing content")
            self.verifyTrueOrFailCase(
                self.is_playing(master),
                "Player under test <{}> playing".format(test_player))
            self.wait_for_volume_change(
                test_player, EXPECTED_VOLUME_AFTER_UNMUTE)
            self.wait_for_volume_change(
                test_player, 5, set_volume=True)
            self.verifyTrueOrFailCase(
                self.wait_for_uri_change(master, zp.raw_udn),
                "Expect <{}> CurrentURI to point to {}".format(
                    master, zp.raw_udn))
        # When the long press rotation is exhausted a subsequent
        # long press renders the player stand alone
        time.sleep(5)
        self.logger.info("Long press to become stand alone")
        self.long_press_action(test_player, LongPressAction.END_OF_LIST)
        # On stand alone players that were playing at the time
        # long press commenced, the current URI.
        self.verifyTrueOrFailCase(
            self.wait_for_uri_change(master, exp_uri),
            "Expect player to standalone {}".format(exp_uri))
        # Confirm an attempt to play will generate
        # UPNP_ERROR_701_TRANSITION_UNAVAILABLE
        # A successful Play modifies the rotation list!!!
        if invoke_play:
            try:
                master.AVTransport.Play()
            except UPnPError as e:
                self.verifyEqualOrFailCase(
                    UPNP_ERROR_701_TRANSITION_UNAVAILABLE,
                    e.upnp_error,
                    "Expect UPnP error on play attempt")
            else:
                self.logger.error("Unexpected play on <{}>".format(master))
                self.logger.error("Expected URI '{}'".format(exp_uri))
                self.logger.error("Actual URI '{}'".format(
                    master.AVTransport.get_avtransport_uri()))
                self.stop("Stopping due to unexpected play")
        self.logger.debug("Long press test ended")



[docs]    def verify_continuous_playing_state(
            self,
            zp,
            continuous_playing_sec=CONTINUOUS_PLAYING_SEC,
            transition_to_playing_sec=TRANSITION_TO_PLAYING_SEC):
        """
        Confirm the play state remains PLAYING on zone player during the
        required continuous playing interval. Fails test when zone player
        transitions out of the PLAYING_STATE during the sampling period.
        :param :class:`sonos.client.internal.SonosZoneComponent` zp:
        The zone player under test
        :param float continuous_playing_sec: The number of continuous seconds
        to remain in PLAYING_STATE
        :param float transition_to_playing_sec: The number of seconds before
        sampling the state
        """
        avts = AVTransportStateTransitionTimer(zp.AVTransport)
        avts.start(continuous_playing_sec,
                   transition_to_playing_sec,
                   user_defined_start_states=[zp.AVTransport.PLAYING_STATE])
        self.verifyTrueOrFailCase(
            not avts.state_changed,
            "Expect AVT state to remain playing for {} seconds of '{}'".format(
                continuous_playing_sec, zp.AVTransport.get_avtransport_uri()))

        """
        Uses a spoofed button press to acknowledge a buzzer alarm.
        Confirms the specified zones have received the stop event.
        Returns True when all zones have received the stop event
        else the test stops.
        :param :class:`sonos.client.internal.SonosZoneComponent` zp:
        :param list :class:`sonos.client.internal.SonosZoneComponent` zones:

            self.cb_confirm_last_clear_avt)
        self.pause_with_button_press_or_upnp(zp)
        try:
            wait_until_true(lambda: not last_stopped, timeout_seconds=10)
        except TimeoutError:
            for serial_num in last_stopped.keys():
                self.logger.error("Player <{}> did not receive expected stop event"
                                  "on buzzer alarm <{}>".format(
                                      serial_num_to_zp[serial_num], zp))
            self.logger.info(
                "DiagnosticID: {} due to not receiving expected stop event".format(
                    zp.ZoneGroupTopology.SubmitDiagnostics(include_controller=False)))
            self.stop("Stopping due to failure to receive expected stop event")
        self.unregister_watch(zones, LAST_CLEAR_AVT_REGEX_MSGS)
        return True
