[docs]    def generate_test_vector(self):
        """
        Generates a test vector containing the DUT, Mute state, LED state,
        sequence of LED bitmasks to test for, the voice tap file, and a flag
        indicating whether or not near field should be tested.

        :yields: a :obj:`tuple`
        """
        seen_models = []
        for zp in self.voice_players:
            if zp.modelNumber not in seen_models:
                seen_models.append(zp.modelNumber)
                for muted, led_on, led_pattern, tap, near_field in TEST_VECTORS:
                    yield (zp, muted, led_on, led_pattern, tap, near_field)



[docs]    def setUpTest(self, zp, muted, led_on, led_pattern, tap, near_field):
        """
        Sets up the test by clearing anacapa.trace, preparing a VoiceOverlay
        object, preparing a log inspector, and setting diag levels.

        :param zp: The DUT
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`
        :param muted: Whether or not the DUT should be muted to start the test
        :type muted: :obj:`bool`
        :param led_on: Whether or not the DUT should have the status LED on
        :type led_on: :obj:`bool`
        :param led_pattern: The pattern of LED bit masks to watch for
        :type led_pattern: :obj:`list` of :obj:`str`
        :param tap: The voice tap file name
        :type tap: :obj:`str`
        :param near_field: Whether or not to test near field
        :type near_field: :obj:`bool`
        """
        self.logger.info("Setup Test")
        self.logger.info("Test Vector: ZP: {} Mute: {} LED: {} Patter: {}"
                         .format(zp.serialNumber, 'ON' if muted else 'OFF',
                                 'ON' if led_on else 'OFF', led_pattern))
        self.voice_overlay = VoiceOverlay(zp)
        self.log_inspector = AnacapaTraceLogInspector(zp)
        self.set_diagcs()
        self.veh = self.voice_event_handler[zp.raw_udn]



[docs]    def tearDownTest(self, zp, muted, led_on, led_pattern, tap, near_field):
        """
        Tears down each test by deleting any voice created alarms, checking for
        a backtrace, removing any voice overlay files, and reseting the mute
        and LED states.

        :param zp: The DUT
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`
        :param muted: Whether or not the DUT should be muted to start the test
        :type muted: :obj:`bool`
        :param led_on: Whether or not the DUT should have the status LED on
        :type led_on: :obj:`bool`
        :param led_pattern: The pattern of LED bit masks to watch for
        :type led_pattern: :obj:`list` of :obj:`str`
        :param tap: The voice tap file name
        :type tap: :obj:`str`
        :param near_field: Whether or not to test near field
        :type near_field: :obj:`bool`
        """
        self.veh.reset_states()
        zp.diag.post_diagmsg("********TearDownTest********")
        self.logger.info("Tear down test")
        self.logger.info("Test Vector: ZP: {} Mute: {} LED: {} Pattern: {}"
                         .format(zp.serialNumber, 'ON' if muted else 'OFF',
                                 'ON' if led_on else 'OFF', led_pattern))
        if tap == MULTITURN_TAP:
            url = self.get_voicetap_url(DELETE_ALARM_TAP)
            self.voice_overlay.load_and_start_overlay(url)
            try:
                self.veh.wait_for_event(self.veh.AVS_DIRECTIVE_RECD)
            except TimeoutError as e:
                zp.cli.get_anacapa_trace()
                raise e
        self.check_for_backtrace(zp)
        self.voice_overlay.remove_overlay_file()
        zp.RenderingControl.SetMute(False)
        zp.DeviceProperties.SetLEDState(True)
        zp.diag.post_diagmsg("********TearDownTest: Exit********")

[docs]    def check_for_backtrace(self, zp):
        if zp.cli.has_backtrace():
            """
            If a backtrace is found, exit the test so as to leave the device
            in the failed state and not dirty the log
            """
            self.logger.critical("Found backtrace on {} exiting".format(zp))
            quit()



[docs]    def wait_for_led_apply(self, bitmasks):
        """
        Waits for anacapa.trace to state that the given LED bitmask has been
        applied. If the wait times out, this method will also mark the test case
        as failed.

        :param bitmask: The LED bitmask to wait for
        :type bitmask: :obj:`str`
        """
        self.logger.info("LED masks to wait for: {}".format(bitmasks))
        APPLY_LED = '.*apply led mode mode:{}.*'
        for mask in bitmasks:
            self.logger.info("Waiting for LED mask: {}".format(mask))
            self.verifyTrueOrFailCase(
                self.log_inspector.wait_for_next(APPLY_LED.format(mask), 30,
                                                 delay=10),
                "Should find {} LED bitmask".format(mask))

    def test_leds(self, zp, muted, led_on, led_pattern, tap, near_field):
        """
        Tests the voice related LED states by playing a voice tap and checking
        anacapa.trace for the correct LED bitmask settings. If the near_field
        parameter is set to True, it will also test near field but only does
        so with no voice command given. The reason for only testing silence
        with near field is that voice tap playback is triggered by unmuting
        the microphone. As a result, attempting to trigger near field with
        a voice tap would result in the same behavior as testing far field
        with a voice tap.

        :param zp: The DUT
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`
        :param muted: Whether or not the DUT should be muted to start the test
        :type muted: :obj:`bool`
        :param led_on: Whether or not the DUT should have the status LED on
        :type led_on: :obj:`bool`
        :param led_pattern: The pattern of LED bit masks to watch for
        :type led_pattern: :obj:`list` of :obj:`str`
        :param tap: The voice tap file name
        :type tap: :obj:`str`
        :param near_field: Whether or not to test near field
        :type near_field: :obj:`bool`
        """
        self.veh.reset_states()
        zp.cli.clear_anacapa_trace()
        zp.diag.post_diagmsg("*****leds_zp Start Test: Mute {} LED {} Tap {}*****"
                             .format(muted, 'On' if led_on else 'Off', tap))
        zp.diag.post_diagmsg("{}muting".format('' if muted else 'Un'))
        self.verifyTrueOrStop(zp.RenderingControl.SetMute(muted), "Mute should "
                              "be {} to start the test".format(muted))
        zp.diag.post_diagmsg("Settting LED to {}".format('On' if led_on else 'Off'))
        self.verifyTrueOrStop(zp.DeviceProperties.SetLEDState(led_on), "LED "
                              "should be {} to start the test".format(led_on))
        url = self.get_voicetap_url(tap)
        zp.diag.post_diagmsg("*******Starting Overlay*********")
        self.voice_overlay.load_and_start_overlay(url)
        if tap == MULTITURN_TAP:
            try:
                wait_until_true(lambda: not self.veh.is_ext_audio_src_playing(zp.raw_udn))
            except TimeoutError as e:
                zp.cli.get_anacapa_trace()
                raise e
        self.wait_for_led_apply(led_pattern)
        self.logger.info("Removing voice file")
        self.voice_overlay.remove_overlay_file()
        if near_field:
            self.logger.info("Testing near field")
            zp.diag.post_diagmsg("*****leds_zp Testing near field*****")
            zp.cli.touch_cap_zone('m', hold_time=1000)
            self.wait_for_led_apply(led_pattern)
