[docs]class LEDButtonState(object):
    """
    class to hold the LED and Lock Button states
    """
    # default
    def __init__(self, button_lock_state=False, led_on_state=True):
        """
        initialize with default states
        :param bool button_lock_state: button lock state
        :type button_lock_state: :obj:`bool`
        :param bool led_on_state: LED on state
        :type led_on_state: :obj:`bool`
        """
        self.button_lock_state = button_lock_state
        self.LED_on_state = led_on_state


[docs]    def validate_zp_states_match(self, zp):
        """
        method to validate that the zone player's LED and button lock state
        match that of the member states
        :param zp: the zone player to evaluate
        :type zp: :obj:`sonos.client.internal.SonosZoneComponent`
        :return: True if match, false otherwise
        """
        led_state = zp.DeviceProperties.GetLEDState()
        button_state = zp.DeviceProperties.GetButtonLockState()
        return (led_state == self.LED_on_state) and \
               (button_state == self.button_lock_state)

        """
        toggle the LED on state of the zone player
        :param zp: the zone player whose LED on state to toggle
        :type zp: :obj:`sonos.client.internal.SonosZoneComponent`
        """
        self.logger.info("toggling LED state: {}".format(zp))
        zp.DeviceProperties.SetLEDState(not zp.DeviceProperties.GetLEDState())

[docs]    def set_zp_to_state(self, zp, led_button_state):
        """
        set the given zone player to the given LED on and button lock states
        :param zp: the zone player
        :type zp: :obj:`sonos.client.internal.SonosZoneComponent`
        :param led_button_state: the LED on and button lock state
        :type led_button_state: :obj:`bool`
        """
        self.logger.info("setting zp to state: {} --> {}".format(
            zp, led_button_state))
        zp.DeviceProperties.SetLEDState(led_button_state.LED_on_state)
        zp.DeviceProperties.SetButtonLockState(
            led_button_state.button_lock_state)



[docs]    def validate_state(self, zp_state_note):
        """
        validate that the given zone player's LED on and button lock states
        match the given state
        :param list of tuples zp_state_note: a list of zone player, state,
        and note
        :type zp_state_note: :obj:`tuple`
        """
        for zp, state, note in zp_state_note:
            valid = state.validate_zp_states_match(zp)
            self.logger.info(
                "Validating state of {}. Should be {}. Passed: {}".format(
                    zp, state, valid
                ))
            self.verifyTrueOrFailCase(valid, note)



[docs]    def cleanup_and_initialize(self, zp_state_note):
        """
        cleanup the bonded configuration of the zp and set the button and
        LED states to the given state. also validate that it worked
        :param list of tuples zp_state_note: a list of zone player, state,
        and note
        :type zp_state_note: :obj:`tuple`
        """
        # track the last zp in the list and use it to gauge when the players
        #  have all come back online
        zps = []
        for zp, state, note in zp_state_note:
            # only wait for ht satellite capable devices to come back online
            # reasoning is that the satellites are the only ones whose services
            # go down during this time, and also because the masters will only
            # be advertising at a rate of once per 900s at worst.
            # also only wait for players that were previously bonded.
            # so only non-ht-master devices that were previously bonded.
            if not zp.is_ht_master_capable() and zp.is_bonded():
                zps.append(zp)
            self.logger.info("Cleaning zp @ {}".format(zp))
            zp.DeviceProperties.cleanup_bonded_configuration()
            self.set_zp_to_state(zp, state)
            self.validate_state([(zp, state, note)])
        self.logger.info("Waiting for zps to come back alive")
        upnp_monitor.threaded_wait_for_upnp_alive(zps)

    def test_stereo_pair_and_sub(self, left, right):
        """
        test that the stereo pairs revert to their initial states after
        un-bonding. stereo pair + bonded sub
        :param left: the left stereo pair device
        :type left: :obj:`sonos.client.internal.SonosZoneComponent`
        :param right: the right player
        :type right: :obj:`sonos.client.internal.SonosZoneComponent`
        """
        self.logger.info("----------------Starting Stereo Pair "
                         "Test----------------")
        # cleanup & initialization
        initialization_zones_states_notes = [
            (left, STEREO_PAIR_TEST_STATES[INITIAL][
                LEFT], "initialize left state"),
            (right, STEREO_PAIR_TEST_STATES[
                INITIAL][RIGHT], "initialize right state"),
            (self.sub, STEREO_PAIR_TEST_STATES[INITIAL][
                SUB], "initialize sub state")
        ]
        self.cleanup_and_initialize(initialization_zones_states_notes)

        # pair
        self.logger.info("Pairing")
        # pair
        left.DeviceProperties.add_stereo_bonded_zone_sub_to_this_device(
            right, self.sub)
        self.logger.info("Waiting for sub to come back alive")
        self.sub.wait_for_upnp_alive()

        # 2. post-pair - validate states
        pairing_zones_states_notes = [
            (left, STEREO_PAIR_TEST_STATES[INITIAL][LEFT],
             "left pair state expected"),
            (right, STEREO_PAIR_TEST_STATES[INITIAL][LEFT],
             "right pair state expected"),
            (self.sub, STEREO_PAIR_TEST_STATES[INITIAL][LEFT],
             "sub pair state expected")
        ]
        self.validate_state(pairing_zones_states_notes)

        # change states
        self.logger.info("Toggling state")
        self.toggle_states(left)

        # 3. post-toggle - validate states
        toggling_zones_states_notes = [
            (left, STEREO_PAIR_TEST_STATES[TOGGLE][LEFT],
             "left toggle expected"),
            (right, STEREO_PAIR_TEST_STATES[TOGGLE][LEFT],
             "right toggle expected"),
            (self.sub, STEREO_PAIR_TEST_STATES[TOGGLE][LEFT],
             "sub toggle expected")
        ]
        self.validate_state(toggling_zones_states_notes)

        # unpair & validation
        left.DeviceProperties.cleanup_bonded_configuration()
        self.logger.info("Waiting for sub to come back alive")
        self.sub.wait_for_upnp_alive()
        unpair_zones_states_notes = [
            (left, STEREO_PAIR_TEST_STATES[INITIAL][
                LEFT], "un-pair left state"),
            (right, STEREO_PAIR_TEST_STATES[
                INITIAL][RIGHT], "un-pair right state"),
            (self.sub, STEREO_PAIR_TEST_STATES[
                INITIAL][SUB], "un-pair sub state")
        ]
        self.validate_state(unpair_zones_states_notes)

    def test_ht_50_and_51(self, master, left, right):
        """
        test that the sub and satellites revert to their initial states after
        un-bonding from a HT master
        :param master: the home theater master zone player
        :type master: :obj:`sonos.client.internal.SonosZoneComponent`
        :param left: the left player
        :type left: :obj:`sonos.client.internal.SonosZoneComponent`
        :param right: the right player
        :type right: :obj:`sonos.client.internal.SonosZoneComponent`
        """
        # check for ZP120 case (one satellite -> left == right)
        is_zp120_case = (left == right)

        # announce
        self.logger.info("----------------Starting 5.1 Test----------------")

        # cleanup & initialization
        self.logger.info("Initializing zone players")
        initialization_zones_states_notes = [
            (master, HT_51_TEST_STATES[INITIAL][
                MASTER], "master initial state expected"),
            (self.sub, HT_51_TEST_STATES[INITIAL][SUB],
             "sub initial state expected"),
            (left, HT_51_TEST_STATES[INITIAL][LEFT],
             "left initial state expected"),
            (right, HT_51_TEST_STATES[INITIAL][RIGHT],
             "right initial state expected")
        ]

        # ignore the right if it's a zp 120 case
        if is_zp120_case:
            initialization_zones_states_notes.pop(-1)

        self.cleanup_and_initialize(initialization_zones_states_notes)

        # pair
        master.DeviceProperties.create_50_config(left, right)
        self.logger.info("Waiting for satellites to come back alive")
        upnp_monitor.threaded_wait_for_upnp_alive([left, right])
        self.logger.info("Created 5.0 configuration")
        pairing_zones_states_notes = [
            (master, HT_51_TEST_STATES[INITIAL][MASTER],
             "master 5.0 state expected"),
            (left, HT_51_TEST_STATES[INITIAL][MASTER],
             "left 5.0 state expected"),
            (right, HT_51_TEST_STATES[INITIAL][MASTER],
             "right 5.0 state expected")
        ]
        self.validate_state(pairing_zones_states_notes)

        # add sub
        hsub = HTPositions("SW,SW")
        master.DeviceProperties.add_ht_satellite(self.sub, hsub)
        self.logger.info("Created 5.1 configuration")
        self.logger.info("Waiting for sub to come back alive")
        self.sub.wait_for_upnp_alive()
        sub_pairing_zones_states_notes = [
            (master, HT_51_TEST_STATES[INITIAL][MASTER],
             "master 5.1 state expected"),
            (left, HT_51_TEST_STATES[INITIAL][MASTER],
             "left 5.1 state expected"),
            (right, HT_51_TEST_STATES[INITIAL][MASTER],
             "right 5.1 state expected"),
            (self.sub, HT_51_TEST_STATES[INITIAL][MASTER],
             "sub 5.1 state expected")
        ]
        self.validate_state(sub_pairing_zones_states_notes)

        # change states
        self.toggle_states(master)
        self.logger.info("Toggled states")

        # validate toggle state
        self.logger.info("Verifying toggle states")
        toggle_zones_states_notes = [
            (master, HT_51_TEST_STATES[FINAL][MASTER],
             "master 5.1 toggle state expected"),
            (left, HT_51_TEST_STATES[FINAL][MASTER],
             "left 5.1 toggle state expected"),
            (right, HT_51_TEST_STATES[FINAL][MASTER],
             "right 5.1 toggle state expected"),
            (self.sub, HT_51_TEST_STATES[FINAL][MASTER],
             "sub 5.1 toggle state expected")
        ]
        self.validate_state(toggle_zones_states_notes)

        # cleanup satellites validate
        self.logger.info("Un-bonding 5.1 configuration")
        cleanup_zones_states_notes = [
            (master, HT_51_TEST_STATES[FINAL][MASTER],
             "Master 5.1 un-pair state expected"),
            (self.sub, HT_51_TEST_STATES[INITIAL][
                SUB], "sub 5.1 un-pair state expected"),
            (left, HT_51_TEST_STATES[INITIAL][LEFT],
             "left  5.1 un-pair state expected"),
            (right, HT_51_TEST_STATES[INITIAL][RIGHT],
             "right  5.1 un-pair state expected")
        ]
        # ignore the right if it's a zp 120 case
        if is_zp120_case:
            cleanup_zones_states_notes.pop(-1)

        master.DeviceProperties.cleanup_bonded_configuration()
        self.logger.info("Waiting for satellites to come back alive")
        upnp_monitor.threaded_wait_for_upnp_alive([self.sub, left, right])
        self.validate_state(cleanup_zones_states_notes)
