    Grab a Sub and bond as satellite to Playbar/S1.  Then playback to Playbar/S1.  Disable "Sub" and then inspect log to
    look for <diag:leds_zp,2> apply led mode mode:0xXXXXXX bitmask.

    .. warning:: Original bug wes intermittent in reproducibility.  When it does reproduce, the time to failure varies

    .. note:: Due to SWPBL-55824, this test was born to check led bitmask that was being logged to determine expected.

    MUSIC_SOURCE = ['radio']

[docs]    def setUpTest(self, master, sub):
        # As a sub, how do I unbond myself when I am bonded via UPNP and nto through the HT Master or GC?
        self.logger.info("Performing cleanup of {}".format((master, sub)))
        [cleanup(zp) for zp in (master, sub)]
        self.logger.info("Adding files as URIs to {}'s queue: {}".format(zp,
                                                                         LEDBitmaskTests.WEBSERVER_TEST_FILES))
        uris = add_files_as_uri_to_queue(master,
                                         LEDBitmaskTests.WEBSERVER_TEST_FILES.values(),
                                         "http://{}".format(self.__websrv.host()))
        self.logger.info("URIs: {}".format(uris))

        for zp in (master, sub):
            self.__results[zp.device.get_uuid()]["MONITORED_ERROR_STRINGS"] = defaultdict(dict)
            for expr in LEDBitmaskTests.MONITORED_ERROR_STRINGS:
                self.__results[zp.device.get_uuid()]["MONITORED_ERROR_STRINGS"][expr] = 0
            self.__results[zp.device.get_uuid()][LEDBitmaskTests.LED_MODE_LOG_PATTERN] = deque()
            # Default when bonded as satellite sub
            self.__results[zp.device.get_uuid()][LEDBitmaskTests.EQ_SUB_ENABLE_LOG_PATTERN] = True
            [zp.diag.set_diag_level(k, v) for k, v in LEDBitmaskTests.DIAG_LEVELS.iteritems()]
            self._setup_udp_logger(zp)

    def _setup_udp_logger(self, zp):
        """
        Setup udp logger and pass in callbacks for strings we are looking for

        :param zp: sonos zone player for which to setup udp logger
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`

        :return: None
        """
        self.logger.info("Setting up udp logger on : {}".format(zp.modelNumber))
        try:
            zp.diag.start_udp_log()
        except SocketError as e:
            self.logger.critical("UDPLogger port error: {}".format(e))
            self.stop_suite()
        [zp.diag.watch_udp_log_for_string(error_string,
                                          lambda msg: self.detected_error(msg, zp))
         for error_string in LEDBitmaskTests.MONITORED_ERROR_STRINGS]
        zp.diag.watch_udp_log_for_string(LEDBitmaskTests.LED_MODE_LOG_PATTERN,
                                         lambda msg: self.update_zp_led_state(msg, zp))
        zp.diag.watch_udp_log_for_string(LEDBitmaskTests.SATELLITE_AUDIO_DISABLED_LOG_PATTERN,
                                         lambda msg: self.update_zp_audio_disabled_state(msg, zp))
        zp.diag.watch_udp_log_for_string(LEDBitmaskTests.EQ_SUB_ENABLE_LOG_PATTERN,
                                         lambda msg: self.update_zp_audio_disabled_state(msg, zp))
        zp.diag.watch_udp_log_for_string(LEDBitmaskTests.SHUTTING_DOWN_LOG_PATTERN,
                                         lambda msg: self.detected_shutdown(msg, zp))


[docs]    def detected_error(self, msg, zp):
        """
        Process the MONITORED_ERROR_STRINGS as counters.

        :param msg: UDP log msg
        :type zp: :obj:`str`

        :param zp: sonos zone player for which msg was detected
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`

        :return: None
        """
        # self._collect_anacapa_and_threadinfo_from(zp)
        for expr in filter(lambda x: x in msg, LEDBitmaskTests.MONITORED_ERROR_STRINGS):
            self.__results[zp.device.get_uuid()]["MONITORED_ERROR_STRINGS"][expr] += 1



[docs]    def detected_shutdown(self, _, zp):
        """
        Shutdown has been detected by a zp...we need to restart the UDP logger.

        :param zp: sonos zone player for which msg was detected
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`

        :return: None
        """
        # Workaround for https://jira.sonos.com/browse/SWPBL-55060
        time.sleep(3)
        [zp.diag.set_diag_level(k, v) for k, v in LEDBitmaskTests.DIAG_LEVELS.iteritems()]
        self._setup_udp_logger(zp)



[docs]    def update_zp_led_state(self, msg, zp):
        """
        Updates ZP's LED bitmask from UDP log lines.

        :param msg: UDP log msg
        :type zp: :obj:`str`

        :param zp: sonos zone player for which msg was detected
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`

        :return: None
        """
        prog = re.compile(LEDBitmaskTests.LED_MODE_LOG_PATTERN, flags=re.DOTALL)
        self.__results[zp.device.get_uuid()][LEDBitmaskTests.LED_MODE_LOG_PATTERN].append(
            SonosZoneComponent.LEDBitmask(prog.match(msg).groups()[0]))
        self.logger.info("{} LED bitmask: {}".format(zp, self.__results[zp.device.get_uuid()][LEDBitmaskTests.LED_MODE_LOG_PATTERN]))



[docs]    def update_zp_audio_disabled_state(self, msg, zp):
        """
        Updates ZP's "Audio Enabled" state from UDP log lines.

        :param msg: UDP log msg
        :type zp: :obj:`str`

        :param zp: sonos zone player for which msg was detected
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`

        :return: None
        """
        prog = re.compile(LEDBitmaskTests.EQ_SUB_ENABLE_LOG_PATTERN, flags=re.DOTALL)
        if prog.match(msg) is not None:
            self.__results[zp.device.get_uuid()][LEDBitmaskTests.EQ_SUB_ENABLE_LOG_PATTERN] = True if prog.match(msg).groups()[
                                                                                        0] == "1" else False
        else:
            prog = re.compile(LEDBitmaskTests.SATELLITE_AUDIO_DISABLED_LOG_PATTERN, flags=re.DOTALL)
            self.__results[zp.device.get_uuid()][LEDBitmaskTests.EQ_SUB_ENABLE_LOG_PATTERN] = True if prog.match(msg).groups()[
                                                                                        0] == "0" else False
        self.logger.info("{} Sub Enabled: {}".format(zp, self.__results[zp.device.get_uuid()][LEDBitmaskTests.EQ_SUB_ENABLE_LOG_PATTERN]))

    def test_led(self, master, sub):
        """
        Check ZP's LED bitmask from ungrouped, grouped playing, grouped playing (Sub disabled) to ungroup again.
        """
        self.logger.info("Master: {}".format(master))
        self.logger.info("Subwoofer: {}".format(sub))

        if master.is_ht_master_capable():
            master.DeviceProperties.add_ht_satellite(sub, HTPositions.SW)
        else:
            master.DeviceProperties.add_bonded_sub_to_this_device(sub)

        # Intentional toggle to force logging
        master.RenderingControl.set_sub_enable(True)
        self.logger.info("Master {} starting {} playback.".format(master, 'queue'))
        start_playback_with_source_selector(master, ('queue', 1))
        self.logger.info("Master {} started playback.".format(master))

        master.AVTransport.SetPlayMode("REPEAT_ONE")
        time.sleep(LEDBitmaskTests.R_AUDIOSTAT_POLL_INTERVAL_SECS)
        # What I realy want is sub.AVTransport.is_playing() and sub.RenderingControl.GetEQ('SubEnable')
        wait_until_true(lambda:
                        sub.is_bonded() and master.AVTransport.is_playing() and int(master.RenderingControl.GetEQ(
                            'SubEnable')) and len(
                            self.__results[sub.device.get_uuid()][LEDBitmaskTests.LED_MODE_LOG_PATTERN]),
                        iteration_delay=1, timeout_seconds=5,
                        reason='Expected Sub to be in Playing, Bonded and SubEnable state.')
        last_actual_led_bitmask = self.__results[sub.device.get_uuid()][LEDBitmaskTests.LED_MODE_LOG_PATTERN].pop()

        self.verifyTrueOrFailCase(last_actual_led_bitmask.is_hhid_set(),
                                  "Expected {} in bitmask".format(hex(SonosZoneComponent.LEDBitmask.R_LED_HHID)))
        self.verifyTrueOrFailCase(not last_actual_led_bitmask.is_led_state_off(),
                                  "Unexpected {} in bitmask".format(hex(SonosZoneComponent.LEDBitmask.R_LED_STATE_OFF)))
        self.verifyTrueOrFailCase(last_actual_led_bitmask.is_playing_set(),
                                  "Expected {} in bitmask".format(hex(SonosZoneComponent.LEDBitmask.R_LED_PLAYING)))
        self.verifyTrueOrFailCase(not last_actual_led_bitmask.is_audio_off_set(),
                                  "Unexpected {} in bitmask".format(hex(SonosZoneComponent.LEDBitmask.R_LED_AUDIO_OFF)))

        # Clear sub's historical LED states
        self.__results[sub.device.get_uuid()][LEDBitmaskTests.LED_MODE_LOG_PATTERN].clear()

        master.RenderingControl.set_sub_enable(False)
        time.sleep(LEDBitmaskTests.WAIT_FOR_FAILURE)  # "apply led" takes some time to update from last action.
        # What I realy want is sub.AVTransport.is_playing() and sub.RenderingControl.GetEQ('SubEnable')
        wait_until_true(lambda:
                        sub.is_bonded() and master.AVTransport.is_playing() and not int(master.RenderingControl.GetEQ(
                            'SubEnable')) and len(
                            self.__results[sub.device.get_uuid()][LEDBitmaskTests.LED_MODE_LOG_PATTERN]),
                        iteration_delay=1, timeout_seconds=10,
                        reason='Expected Sub to be in Playing, Bonded and Not SubEnable state.')
        self.verifyTrueOrFailCase(self.__results[sub.device.get_uuid()][LEDBitmaskTests.EQ_SUB_ENABLE_LOG_PATTERN] == False,
                                  "Sub is disabled as expected.")
        last_actual_led_bitmask = self.__results[sub.device.get_uuid()][LEDBitmaskTests.LED_MODE_LOG_PATTERN].pop()

        self.verifyTrueOrFailCase(last_actual_led_bitmask.is_hhid_set(),
                                  "Expected {} in bitmask".format(hex(SonosZoneComponent.LEDBitmask.R_LED_HHID)))
        self.verifyTrueOrFailCase(not last_actual_led_bitmask.is_led_state_off(),
                                  "Unexpected {} in bitmask".format(hex(SonosZoneComponent.LEDBitmask.R_LED_STATE_OFF)))
        self.verifyTrueOrFailCase(last_actual_led_bitmask.is_playing_set(),
                                  "Expected {} in bitmask".format(hex(SonosZoneComponent.LEDBitmask.R_LED_PLAYING)))
        self.verifyTrueOrFailCase(last_actual_led_bitmask.is_audio_off_set(),
                                  "Expected {} in bitmask".format(hex(SonosZoneComponent.LEDBitmask.R_LED_AUDIO_OFF)))

        # Clear sub's historical LED states
        self.__results[sub.device.get_uuid()][LEDBitmaskTests.LED_MODE_LOG_PATTERN].clear()

        master.DeviceProperties.cleanup_bonded_configuration()
        time.sleep(LEDBitmaskTests.R_AUDIOSTAT_POLL_INTERVAL_SECS)

        last_actual_led_bitmask = self.__results[sub.device.get_uuid()][LEDBitmaskTests.LED_MODE_LOG_PATTERN].pop()

        self.verifyTrueOrFailCase(last_actual_led_bitmask.is_hhid_set(),
                                  "Expected {} in bitmask".format(hex(SonosZoneComponent.LEDBitmask.R_LED_HHID)))
        self.verifyTrueOrFailCase(not last_actual_led_bitmask.is_led_state_off(),
                                  "Unexpected {} in bitmask".format(hex(SonosZoneComponent.LEDBitmask.R_LED_STATE_OFF)))
        self.verifyTrueOrFailCase(not last_actual_led_bitmask.is_playing_set(),
                                  "Unexpected {} in bitmask".format(hex(SonosZoneComponent.LEDBitmask.R_LED_PLAYING)))
        self.verifyTrueOrFailCase(not last_actual_led_bitmask.is_audio_off_set(),
                                  "Unexpected {} in bitmask".format(hex(SonosZoneComponent.LEDBitmask.R_LED_AUDIO_OFF)))

        [self.verifyTrueOrFailCase(sum(self.__results[zp.device.get_uuid()]["MONITORED_ERROR_STRINGS"].values()) == 0,
                                   "Detected failure message in {} anacapa.trace: {}".format(sub, self.__results[zp.device.get_uuid()][
                                       "MONITORED_ERROR_STRINGS"])) for zp in (master, sub)]
