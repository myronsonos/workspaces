[docs]    def test_enable_secure_mode(self):
        """
        Test that the secure mode setting can be set.
        """
        original_mode = self.zp.SystemProperties.get_secure_mode()
        if original_mode:
            new_mode = 0
        else:
            new_mode = 1
        try:
            self.zp.SystemProperties.set_secure_mode(new_mode)
            self.verifyEqualOrFailCase(
                new_mode, self.zp.SystemProperties.get_secure_mode(),
                "Should have changed secure mode")
        finally:
            self.zp.SystemProperties.set_secure_mode(original_mode)

[docs]    def test_set_led_state_secured(self):
        """
        Test that LED state change is secured when secure mode is set
        """
        original_state = self.zp.DeviceProperties.GetLEDState()
        original_mode = self.zp.SystemProperties.get_secure_mode()
        try:
            new_state = not original_state
            self.zp.SystemProperties.set_secure_mode(0)
            self.zp.DeviceProperties.SetLEDState(new_state)
            self.verifyEqualOrFailCase(self.zp.DeviceProperties.GetLEDState(),
                                       new_state,
                                       "Should have changed LED state")

            self.zp.SystemProperties.set_secure_mode(1)
            try:
                with self.verifyRaisesOrFailCase(
                    UPnPError, "Check that LED state change not allowed") as cm:
                    self.zp.DeviceProperties.SetLEDState(original_state)
            except WorkflowAssertionError:
                self.verifyEqualOrFailCase(cm.exception.upnp_error,
                                           UPNP_ERROR_606_UNAUTHORIZED,
                                           "Should have been unauthorized")
            self.verifyEqualOrFailCase(self.zp.DeviceProperties.GetLEDState(),
                                       new_state,
                                       "Should not have changed LED state")
        finally:
            self.zp.SystemProperties.set_secure_mode(None)
            self.zp.DeviceProperties.SetLEDState(original_state)
            if original_mode is not None:
                self.zp.SystemProperties.set_secure_mode(original_mode)



[docs]    def test_rename_room_secured(self):
        """
        Test that room renaming is secured when secure mode is set
        """
        original_name = self.zp.DeviceProperties.get_zone_name()
        original_mode = self.zp.SystemProperties.get_secure_mode()
        try:
            base_name = original_name[:54]
            first_rename = base_name + "renamed-1"
            self.zp.SystemProperties.set_secure_mode(0)
            self.zp.DeviceProperties.set_zone_name(first_rename)
            self.verifyEqualOrFailCase(self.zp.DeviceProperties.get_zone_name(),
                                       first_rename,
                                       "Should have changed zone name")

            self.zp.SystemProperties.set_secure_mode(1)
            second_rename = base_name + "renamed-2"
            try:
                with self.verifyRaisesOrFailCase(
                        UPnPError, "Check that room rename not allowed") as cm:
                    self.zp.DeviceProperties.set_zone_name(second_rename)
            except WorkflowAssertionError:
                self.verifyEqualOrFailCase(cm.exception.upnp_error,
                                           UPNP_ERROR_606_UNAUTHORIZED,
                                           "Should have been unauthorized")
            self.verifyEqualOrFailCase(self.zp.DeviceProperties.get_zone_name(),
                                       first_rename,
                                       "Should not have renamed room")
        finally:
            self.zp.SystemProperties.set_secure_mode(None)
            self.zp.DeviceProperties.set_zone_name(original_name)
            if original_mode is not None:
                self.zp.SystemProperties.set_secure_mode(original_mode)


    def _set_grants(self):
        hhid = self.zp.hhid
        add_grant("otto", hhid, "owner", 120)
        add_grant(DEFAULT_ANON_TOKEN, hhid, "anon", 120)

        with self.zp.SystemProperties.enable_secure_mode():
            with self.zp.auth_header("otto", 1):
                self._set_grants()
                original_state = self.zp.DeviceProperties.GetLEDState()
                new_state = not original_state
                self.zp.DeviceProperties.SetLEDState(new_state)
                self.verifyEqualOrFailCase(
                    self.zp.DeviceProperties.GetLEDState(), new_state,
                    "Should have changed LED state from original")
                self.zp.DeviceProperties.SetLEDState(original_state)
                self.verifyEqualOrFailCase(
                    self.zp.DeviceProperties.GetLEDState(), original_state,
                    "Should have changed LED state back to original")

    def test_rtp_ttl(self):
        """
        Test that the correct ttl is sent back when no TTL request happens and
        that the correct TTL is honored.
        """
        hhid, ttr, rtp = self._set_grants()

        original_led_state = self.zp.DeviceProperties.GetLEDState()
        self.logger.debug("Emptying requests from role-to-permissions")
        rtp.clear_requests()
        with self.zp.SystemProperties.enable_secure_mode():
            with self.zp.auth_header("otto", None):
                self.logger.info("First LED state {} set at {}".format(
                    time.ctime(), not original_led_state))
                self.zp.DeviceProperties.SetLEDState(not original_led_state)
                requests = [request for request in rtp.requests()]
                self.verifyListEqualOrFailCase(
                    [(["owner"], None, ['lan'])], requests,
                    "Should have one request from owner")

                self.logger.info("Waiting 5 seconds to attempt second request "
                                 "to ensure 20s TTL for cache is honored.")
                time.sleep(5)
                self.logger.info("Second LED state {} set at {}".format(
                    time.ctime(), original_led_state))
                self.zp.DeviceProperties.SetLEDState(original_led_state)
                requests = [request for request in rtp.requests()]
                self.verifyListEqualOrFailCase(
                    [], requests,
                    "Should have no requests from owner because of cache")

                self.logger.info("Waiting 25 seconds to attempt third request "
                                 "to ensure 20s TTL for cache is honored.")
                time.sleep(25)
                self.logger.info("Third LED state {} set at {}".format(
                    time.ctime(), not original_led_state))
                self.zp.DeviceProperties.SetLEDState(not original_led_state)
                requests = [request for request in rtp.requests()]
                self.verifyListEqualOrFailCase(
                    [(["owner"], None, ['lan'])], requests,
                    "Should have one request from owner after cache expiration")
