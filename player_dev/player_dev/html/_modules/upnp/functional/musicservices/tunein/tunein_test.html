    def __init__(self):
        super(TuneInTest, self).__init__(

    def _parse_args(self):
        parser = argparse.ArgumentParser()
        parser.add_argument(
            "--num_trending_stations", type=int,
            default=NUM_TRENDING_STATIONS,
            help="Number of trending stations to play - "
            "default {} (0 - all)".format(
                NUM_TRENDING_STATIONS))
        parser.add_argument(
            "--num_episodes", type=int,
            default=NUM_EPISODES,
            help="Number of shows to play - default {} (0 - all)".format(
                NUM_EPISODES))
        args, unknown_args = parser.parse_known_args()
        sys.argv[1:] = unknown_args
        self.num_trending_stations = args.num_trending_stations
        self.num_episodes = args.num_episodes

[docs]    def generate_tunein_trending(self):
        """
        Generates N TuneIn radio stations which are in the "Trending"
        category based on the command line option.
        If num_trending_stations == 0 then play all Trending stations
        else play the num_trending_stations.
        """
        to_play = self.audio_casts_to_play(
            self.num_trending_stations, self.trending_stations)
        self.logger.info(
            "Selected to play {} random trending stations".format(to_play))
        shuffled_stations = self.trending_stations[:]
        random.shuffle(shuffled_stations)
        for zp in self.test_zones:
            for station_metadata in shuffled_stations[:to_play]:
                yield (zp, station_metadata)

            self.logger.info(
                "Detected SSL failure, capturing player DNS cache")

        zp.diag.stop_watching_udp_log_for_string(CERT_FAIL_MSG)



[docs]    def play_tunein_stream(self, zp, station_metadata, secure=False):
        """
        Play the TuneIn stream given the station metadata
        """
        uri, uri_metadata = zp.AVTransport.create_tunein_uri_metadata(
            station_metadata)
        zp.AVTransport.SetAVTransportURI(uri, uri_metadata)
        zp.AVTransport.Play_and_wait()
        self.logger.info(
            "Now playing radio stream '{}' "
            "decoder type '{}' framer name '{}' "
            "framer URI '{}'".format(
                station_metadata.title.encode('utf-8'),
                self.decoder_type,
                self.framer_name,
                self.framer_uri))
        pos_info = zp.AVTransport.GetPositionInfo()
        self.logger.info("TrackURI: {}".format(pos_info['TrackURI']))
        if secure:
            self.verifyTrueOrWarn(
                'https' in pos_info['TrackURI'],
                "Expect secure playback on '{}'".format(
                    station_metadata.title.encode('utf-8')))
        avts = AVTransportStateTransitionTimer(zp.AVTransport)
        avts.start(CONTINUOUS_PLAYING_SEC,
                   TRANSITION_TO_PLAYING_SEC,
                   user_defined_start_states=[zp.AVTransport.PLAYING_STATE])
        self.verifyTrueOrFailCase(
            not avts.state_changed,
            "Played continuously {} seconds of radio stream '{}'".format(
                CONTINUOUS_PLAYING_SEC, station_metadata.title.encode('utf-8')))
