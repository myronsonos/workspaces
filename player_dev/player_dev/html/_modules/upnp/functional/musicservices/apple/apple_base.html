[docs]    def get_apple_collection(self, title):
        """
        Returns one of the top level collections based in title
        from the root of Apple Music
        :param str title: The title to search
        :return: List of mediaCollection (Sonos.wsdl)
        :rtype :obj:`list`
        """
        for item in self.apple_root:
            if item.title == title:
                res, warn = self.smapi_client.getMetadata(
                    item.id, 0, 100, self.smapi_client)
                return res and res.Items
        return []



[docs]    def update_apple_media_collection_list(self, root, item_type, playable):
        """
        Fetches the metadata for the specified root and goes through
        each mediaCollection and updates the playable list for items
        with item_type matches.
        :param obj root: mediaCollection object per Sonos.wsdl
        :param str item_type: The type of objection to filter
        :param list playable: A list
        """
        radios, warn = self.smapi_client.getMetadata(
            root.id, 0, 100, self.smapi_client)
        for radio in radios.Items:
            if hasattr(radio, 'canPlay') or radio.itemType == item_type:
                self.logger.info(
                    "Appending '{}'".format(radio.title.encode('utf-8')))
                playable.append(radio)
            else:
                self.logger.info("Skipping '{}' itemType={}".format(
                    radio.title.encode('utf-8'), radio.itemType))



[docs]    def get_apple_radio_stations(self):
        """
        Returns all playable radio stations from Radio collections
        :return: List of mediaCollection (Sonos.wsdl)
        :rtype :obj:`list`
        """
        radio_collections = self.get_apple_collection(u'Radio')
        playable = list()
        for radio in radio_collections:
            if hasattr(radio, 'canPlay') or radio.itemType == APPLE_RADIO_STREAM:
                playable.append(radio)
            else:
                self.update_apple_media_collection_list(
                    radio, APPLE_RADIO_STREAM, playable)
        return playable



[docs]    def get_playables(self, results):
        """
        Returns a list of playable container/track from results
        :param list results: List of mediaCollection from Sonos.wsdl
        :return: A list of playable container/track
        :rtype :obj:`list` mediaCollection
        """
        playable = list()
        for result in results:
            if ((hasattr(result, 'itemType') and result.itemType == u'track') or
                (hasattr(result, 'canPlay') and result.canPlay) or
                (hasattr(result, 'trackMetadata') and result.trackMetadata.canPlay)):
                playable.append(result)
        if playable:
            return playable
        for result in results:
            res, warn = self.smapi_client.getMetadata(
                result.id, 0, 100, self.smapi_client)
            if res and res.Items:
                return self.get_playables(res.Items)
        return []



[docs]    def create_apple_stream_uri_metadata(self, stream_id, title):
        """
        Returns a tuple of URI and metadata
        :param str stream_id: Apple stream identifier
        :param str title: Apple stream title
        :return: Tuple of Apple URI and metadata
        :rtype: :obj:`tuple`
        """
        uri_metadata = APPLE_RADIO_METADATA_TEMPLATE.format(
            title.encode('ascii', 'ignore'),
            self.svc_acct.service_id,
            self.svc_acct.username)
        stream_type = stream_id[:stream_id.index(':')]
        uri = APPLE_URI_TEMPLATE.format(
            stream_type,
            urllib.quote(stream_id, URL_ENCODE_EXCLUDE_CHARS),
            self.smapi_id, self.svc_acct.serial_number)
        return uri, uri_metadata

    dispatch_uri['stream'] = [create_apple_stream_uri_metadata, None]
    dispatch_uri['program'] = [create_apple_stream_uri_metadata, None]

[docs]    def set_apple_uri_metadata(self, zp, content):
        """
        Based on content itemType, sets the uri/metadata
        and AVTransport URI.
        :param :class:`sonos.client.internal.SonosZoneComponent` zp:
        :param :obj: content: mediaCollection from Sonos.wsdl
        """
        self.dispatch_uri['track'][1] = zp.AVTransport.add_to_queue
        self.dispatch_uri['playlist'][1] = zp.AVTransport.add_to_queue
        self.dispatch_uri['album'][1] = zp.AVTransport.add_to_queue
        self.dispatch_uri['stream'][1] = zp.AVTransport.SetAVTransportURI
        self.dispatch_uri['program'][1] = zp.AVTransport.SetAVTransportURI
        uri_metadata_func, set_avt_func = self.dispatch_uri[content.itemType]
        uri, metadata = uri_metadata_func(self, content.id, content.title)
        set_avt_func(uri, metadata)
