[docs]class ThreadInfo(Thread):
    """
    Collect thread information from the zone component
    until we're told to stop.
    """
    def __init__(self, zp, sample_frequency=1.0):
        Thread.__init__(self, name=self.__class__.__name__)
        self.daemon = True
        self.zp = zp
        self.sample_frequency = sample_frequency
        self.filename = 'thread_info_{}_{}_{}.txt'.format(
            self.zp.modelNumber,
            self.zp.serialNumber,
            datetime.now().strftime("%H_%M_%S"))
        self.stopping = Event()


[docs]    def run(self):
        """
        Sample the thread information based on the frequency
        until we're told to stop.
        """
        with open(self.filename, 'w+') as fh:
            while not self.stopping.isSet():
                thread_info = self.zp.diag.get_thread_info()
                fh.write(
                    'Snapshot at {}\n'.format(datetime.now().ctime()))
                fh.write(thread_info)
                self.stopping.wait(self.sample_frequency)



[docs]    def stop(self):
        """
        Signal to stop sampling and terminate
        """
        self.stopping.set()
        Thread.join(self)




[docs]class SpotifyLongTrackDurationSeek(SpotifyBase):
    """
    Confirm player switches to next track within the expected
    time frame after a seek to end of track is performed.
    Prior NTS libraries had the behavior of taking a very
    long time (minutes) on a track change after a seek was
    performed near the end of a long duration track.
    """
    def __init__(self):
        super(SpotifyLongTrackDurationSeek, self).__init__(
            logging_to_file=True, parameterized_setup=True)
        self.mod_log_levels['spot'] = 5
        self.mod_log_levels['avt_impl'] = 2
        self.mod_log_levels['upnp'] = 2
        self.iterations = DEFAULT_ITERATIONS
        self.use_thread_info = USE_THREAD_INFO

    def _parse_args(self):
        parser = argparse.ArgumentParser()
        parser.add_argument(
            "--iterations", required=False, type=int,
            default=DEFAULT_ITERATIONS,
            help="Iterations of playback in repeat mode, default {}".format(
                DEFAULT_ITERATIONS))
        args, unknown_args = parser.parse_known_args()
        sys.argv[1:] = unknown_args
        self.iterations = args.iterations

[docs]    def start_playback(self, zp):
        zp.AVTransport.use_queue()
        zp.AVTransport.clear_queue_and_wait()
        num_tracks = self.add_to_queue(zp, CHOPIN_PIANO_CONCERTO_NO_1)
        self.verifyEqualOrStop(7, num_tracks, "Expect to add tracks")
        num_tracks = self.add_to_queue(zp, LONG_DURATION_SINGLES)
        self.verifyEqualOrStop(10, num_tracks, "Expect to add tracks")
        zp.AVTransport.Seek('TRACK_NR', 1)
        zp.AVTransport.SetPlayMode("NORMAL")
        zp.RenderingControl.SetVolume(5)
        zp.AVTransport.Play_and_wait()
        return num_tracks



[docs]    def wait_on_operation_with_thread_info(self, zp, func,
                                           exception_type, *args, **kwargs):
        start = time.time()
        thr = ThreadInfo(zp)
        thr.start()
        err = None
        try:
            func(*args, **kwargs)
        except exception_type as err:
            self.logger.error("Caught expected {}".format(err))
        finally:
            thr.stop()
            self.logger.debug(
                "Waited {} seconds for seek".format(
                    time.time() - start))
            if err:
                self.stop("Stopping due to operation timing out")



[docs]    @combinatorial('generate_zones')
    def test_seek_on_long_duration_tracks(self, zp):
        """
        The test album contains a few long duration tracks.
        Play the album in its entirety and confirm zone player
        switches to the next track within the expected time frame
        after a seek to end of track operation has been performed.
        """
        num_tracks = self.start_playback(zp)
        zp.AVTransport.SetPlayMode("REPEAT_ALL")
        pos_info = zp.AVTransport.GetPositionInfo()
        actual_track = int(pos_info['Track'])
        tracks_to_play = self.iterations * num_tracks
        while tracks_to_play > 0:
            self.logger.info(
                "Playing track={} duration={} title='{}'".format(
                    pos_info['Track'],
                    self.get_seconds_from_pos_info(pos_info),
                    self.get_track_title(pos_info)))
            self.verifyFalseOrFailCase(
                zp.AVTransport.wait_for_avt_state(
                    zp.AVTransport.STOPPED_STATE,
                    timeout_seconds=10),
                "Expect continuous track playback")
            if self.use_thread_info:
                self.wait_on_operation_with_thread_info(
                    zp,
                    zp.AVTransport.seek_to_position_before_track_end,
                    UPnPTimeout,
                    1)
            else:
                zp.AVTransport.seek_to_position_before_track_end(1)
            track_change_timeout = (SECONDS_BEFORE_TRACK_END
                                    + TRACK_CHANGE_TIMEOUT_JITTER)
            res = zp.AVTransport.wait_until_any_track_change(
                timeout=track_change_timeout, delay=1.0)
            self.verifyTrueOrFailCase(
                res, "Track changed within expected timeout")
            pos_info = zp.AVTransport.GetPositionInfo()
            if not actual_track % num_tracks:
                expected_track = 1
            else:
                expected_track = actual_track + 1
            actual_track = int(pos_info['Track'])
            self.verifyEqualOrStop(
                expected_track,
                actual_track,
                "Playing expected track")
            tracks_to_play = tracks_to_play - 1
