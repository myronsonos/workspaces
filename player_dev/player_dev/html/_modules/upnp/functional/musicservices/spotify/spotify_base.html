[docs]def create_per_model_asyncio_buffer_size_dict(buffer_size):
    """
    Creates a dictionary that maps a model number (S1, S9, etc.)
    to a buffer size for the asynchronous buffered stream I/O
    SH4 platform does not support asynchronous buffered stream I/O
    :param int buffer_size: Asynchronous buffer size
    """
    per_model_asyncio_buffer_size_dict = {}
    if buffer_size is None:
        return per_model_asyncio_buffer_size_dict
    for model in PLAYBACK_DEVICES:
        if model in BUFFERED_ASYNCIO_NOT_AVAILABLE:
            continue
        per_model_asyncio_buffer_size_dict.update({model: buffer_size})
    return per_model_asyncio_buffer_size_dict




[docs]def per_model_asyncio_buffer_size_to_dict_update(
        per_model_asyncio_buffer_size_dict,
        per_model_asyncio_buffer_size_list):
    """
    Converts a list comprised of 'model number: buffer size' into a dictionary
    with model number as key
    :param dict per_model_asyncio_buffer_size_dict: Dictionary to update
    :param list per_model_asyncio_buffer_size_list: List of model:<buffer size>
    :return: A dictionary of mapping model(s) to buffer size(s)
    :rtype: `dict`
    """
    for model_buffer_size in per_model_asyncio_buffer_size_list:
        model, buffer_size = tuple(model_buffer_size.split(":"))
        if model in BUFFERED_ASYNCIO_NOT_AVAILABLE:
            continue
        per_model_asyncio_buffer_size_dict.update({model: int(buffer_size)})

    def _parse_base_args(self):
        """
        Parse base class command line switches
        Example of using the base command line switches for asyncronous
        buffered stream I/O buffer size control
        1. --generic_asyncio_buffer_size=1048576
        Sets the buffer size using the test point on zone players which
        are part of the test
        2. --per_model_asyncio_buffer_size S1:524288 S9:2097152
        Sets the buffer size on S1 models to 524288 and S9 models to 2097152
        The remaining zone players in the test use the default built-in
        platform buffer size
        3. --generic_asyncio_buffer_size=1048576 --per_model_asyncio_buffer_size S1:524288 S9:2097152
        Sets the buffer size to 1048576 on all zone players in test except
        S1 models to 524288 and S9 models to 2097152
        """
        parser = argparse.ArgumentParser()
        parser.add_argument(
            "--generic_asyncio_buffer_size", required=False, type=int,
            default=None,
            help="Size of asynchronous buffer on HH - overrides built-in")
        parser.add_argument(
            "--per_model_asyncio_buffer_size", required=False, nargs='*',
            default=[],
            help="Per model asyncio buffer size - overrides built-in")
        args, unknown_args = parser.parse_known_args()
        sys.argv[1:] = unknown_args
        self.generic_asyncio_buffer_size = args.generic_asyncio_buffer_size
        self.per_model_asyncio_buffer_size_dict = create_per_model_asyncio_buffer_size_dict(args.generic_asyncio_buffer_size)
        per_model_asyncio_buffer_size_to_dict_update(self.per_model_asyncio_buffer_size_dict,
                                                     args.per_model_asyncio_buffer_size)

    def __init__(self, logging_to_file=False, parameterized_setup=False):
        super(SpotifyBase, self).__init__(
            parameterized_setup=parameterized_setup)

        self.websrv = None
        self.spotify_smapi_id = 0

[docs]    def player_update_asyncio_buffer_size(self, zp):
        """
        Uses test point to set the buffer size on the
        zone player's asynchronous buffered stream.
        Prevent setting the buffer size on a ZP90 SH4
        since the asynchronous buffered stream implementation
        is not supported in the SH4 platforms.
        :param :class:`~sonos.client.SonosZoneComponent` zp:
        """
        if zp.is_zp90() and zp.is_zp90_sh4():
            return
        if zp.modelNumber in self.per_model_asyncio_buffer_size_dict:
            model_buffer_size = self.per_model_asyncio_buffer_size_dict[zp.modelNumber]
            self.logger.debug("Setting asyncio buffer size {} "
                              "on <{}>".format(model_buffer_size, zp))
            assert zp.diag.set_asyncio_buffer_size(model_buffer_size), \
                "Failed to set asyncio buffer size on player"


    def _setUpTest(self, zp, error_strings=PLAYBACK_ERRORS):
        zp.diag.post_diagmsg(
            'AUTOMATION INFO: Starting {} at {}'.format(
                self.get_test_name(), datetime.now()))
        if self.logging_to_file:
            zp_log = "{}_{}_{}_playback.log".format(
                zp.modelNumber,
                datetime.now().strftime("%H_%M_%S"),
                self.get_test_name())
            zp.diag.start_udp_log_to_file(zp_log)
        else:
            zp.diag.start_udp_log()
        [zp.diag.set_diag_level(mod, self.mod_log_levels[mod])
         for mod in self.mod_log_levels]
        [zp.diag.watch_udp_log_for_string(
                error_string, lambda msg: self.fail(msg))
         for error_string in error_strings]
        self._cleanup(zp)
        self.player_update_asyncio_buffer_size(zp)

    def _tearDownTest(self, zp):
        zp.diag.post_diagmsg(
            'AUTOMATION INFO: Ending {} at {}'.format(
                self.get_test_name(), datetime.now()))
        [zp.diag.set_diag_level(mod, 0) for mod in self.mod_log_levels]
        if self.logging_to_file:
            zp.diag.stop_udp_log_to_file()
        else:
            zp.diag.stop_udp_log()
        self._cleanup(zp)

[docs]    def create_containers(self, mediaCollection):
        containers = []
        for index in range(mediaCollection.count):
            if (mediaCollection.Items[index].canPlay and
                mediaCollection.Items[index].itemType == u"playlist"):
                containers.append(
                    ContainerFormat(
                        self.form_uri_from_id(mediaCollection.Items[index].id),
                        mediaCollection.Items[index].title.encode('utf-8')))
        [self.logger.debug('ContainerFormat("{}", "{}")'.format(
                    container.uri, container.title))
         for container in containers]
        return containers

    def _add_tracks(self, zp, tracks):
        track_uris = list()
        track_md = list()
        for track in tracks:
            track_uris.append(
                track.track_uri.format(
                    self.spotify_smapi_id, self.svc_acct.serial_number))
            track_md.append(
                track.track_md.format(
                    self.svc_acct.service_id, self.svc_acct.username))

        queued_data = zp.AVTransport.AddMultipleURIsToQueue(
            num_uris=len(tracks),
            uris=" ".join(track_uris),
            metadata=" ".join(track_md),
            container_uri="",
            container_metadata="",
            track_num=0,
            as_next=True)
        return int(queued_data['NewQueueLength'])

    def _add_container(self, zp, container):
        metadata = self.zp.AVTransport.generate_didl_lite(
            title=escape(container.title),
            svc_id=self.svc_acct.service_id,
            acct_id=self.svc_acct.username)
        return zp.AVTransport.add_to_queue(container.uri, metadata)

[docs]    def should_sleep_or_not_before_track_end(self, duration, played,
                            min_allowance=MIN_SECONDS_BEFORE_TRACK_END,
                            max_allowance=MAX_SECONDS_BEFORE_TRACK_END):
        """
        Sleep depending on the track duration, played
        time and some slack (randomized) from end of track.
        The slack is based on a desired minimum/maximum
        seconds from end of track to be used as input
        to get a random value.
        :param int duration: Track duration in seconds
        :param int played: Played time in seconds
        :param float min_allowance: Minimum seconds for
        input to random
        :param floag max_allowance: Maximum seconds for
        input to random
        """
        track_end_allowance = random.uniform(min_allowance, max_allowance)
        sleep_duration = duration - played - track_end_allowance
        if sleep_duration > track_end_allowance:
            self.logger.debug(
                "Sleep {} seconds "
                "TrackDuration={} "
                "RelTime={} "
                "track_end_allowance={}".format(
                    sleep_duration, duration, played, track_end_allowance))
            time.sleep(sleep_duration)



[docs]    def add_to_queue(self, zp, play_format, repeat=1):
        """
        Adds the container or the track(s) to queue
        :param :class:`sonos.client.internal.SonosZoneComponent` zp:
        :param play_format: A single ContainerFormat, or SpotifyTrackFormat
        or a list/tuple of SpotifyTrackFormat
        :param int repeat: Number of repeats for single track
        SpotifyTrackFormat
        """
        if hasattr(play_format, 'track_uri') and hasattr(play_format, 'track_md'):
            # single track SpotifyTrackFormat
            return self._add_tracks(zp, [play_format for x in range(repeat)])
        elif hasattr(play_format[0], 'track_uri') and hasattr(play_format[0], 'track_md'):
            # multiple track SpotifyTrackFormat
            return self._add_tracks(zp, play_format)
        return self._add_container(zp, play_format)

[docs]    def add_mixed_tracks_to_queue(self, zp, spotify_tracks,
                                  local_http_uris, shuffle=False):
        """
        Takes Spotify tracks and local tracks and populates
        the play queue. Optionally randomizes the position of
        the local tracks in the play queue.
        :param :class:`sonos.client.internal.SonosZoneComponent` zp:
        :param list spotify_tracks: Spotify tracks to add
        :param list local_http_uris: Local tracks' HTTP URIs
        :param bool shuffle: Shuffle local HTTP URIs before adding
        to queue ?
        :return: Number of tracks queued
        :rtype: :obj:`int`
        """
        http_uris = local_http_uris[:]
        if shuffle:
            random.shuffle(http_uris)
        mixed_tracks = zip(spotify_tracks, http_uris)
        for spot_track, local_track in mixed_tracks:
            self.add_to_queue(zp, spot_track)
            zp.AVTransport.add_to_queue(local_track)
        return zp.AVTransport.get_number_of_tracks_in_queue()
