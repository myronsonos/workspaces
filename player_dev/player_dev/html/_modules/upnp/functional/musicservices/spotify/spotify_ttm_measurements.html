[docs]def plot_data(xlabel, ylabel, title, data, filename):
    """
    Plots a two-dimensional graph from data derived
    from a dictionary that has the following format:
    {
    'key0': [value1, value2, ... valueN],
    'key1': [value1, value2, ... valueN],
    ...
    'keyN': [value1, value2, ... valueN]
    }
    """
    ind = np.arange(len(data[data.keys()[0]]))
    width = 0.05
    # Approximate figure size width/heigth based on available data
    figSize = (12, 6)
    fig, ax = plt.subplots(figsize=figSize)
    rects_list = []
    color_list = []
    for arch in data:
        values = data[arch]
        rects = ax.plot(values, color=COLOR_LIST[arch], linewidth=4)
        rects_list.append(rects[0])
        color_list.append(arch)
    ax.set_xlabel(xlabel)
    ax.set_ylabel(ylabel)
    ax.set_title(title)
    ax.set_xticks(ind)
    ax.legend(tuple(rects_list),
              tuple(color_list),
              bbox_to_anchor=(1.1, .75),
              loc='right')
    ax.grid(which='both', axis='both')
    plt.savefig(filename)

        zp.AVTransport.use_queue()
        zp.AVTransport.clear_queue_and_wait()
        num_tracks = self.add_to_queue(zp, CHOPIN_PIANO_CONCERTO_NO_1)
        self.verifyGreaterOrStop(num_tracks, 0, "Expect to add tracks")
        zp.AVTransport.Seek('TRACK_NR', 1)
        zp.AVTransport.SetPlayMode("REPEAT_ALL")
        zp.RenderingControl.SetVolume(5)
        if play:
            zp.AVTransport.Play_and_wait()
        return num_tracks

        """
        Callback when the player has buffered the first byte of the

            self.msg.append(diff)

[docs]    def get_expected_actual_track(self, zp, actual_track,
                                  num_tracks, direction):
        """
        Returns a tuple of expected, actual track based on the current
        track, number of tracks in playlist and direction of AVT action
        """
        expected = 0
        if direction == PREVIOUS:
            if actual_track == 1:
                expected = num_tracks
            else:
                expected = actual_track - 1
        elif direction == NEXT:
            if actual_track == num_tracks:
                expected = 1
            else:
                expected = actual_track + 1
        else:
            raise ValueError("Unexpected direction - {}".format(direction))
        return (expected, zp.AVTransport.get_current_track())


    def _setup_play(self, zp, measuring):
        """
        Common code to setup the play q and start playback.
        Initializes the measurement list to empty.
        Starts playback of the playlist
        Returns tuple of number of tracks in queue and the actual
        track number the player is playing.
        """
        self.msg = list()
        self.measuring = measuring
        num_tracks = self.start_playback(zp)
        actual_track = zp.AVTransport.get_current_track()
        return (num_tracks, actual_track)


[docs]    def slow_seek_common(self, zp, delta, msg):
        """
        Common code slow seeks +/- when we're pre-fetching next track.
        """
        num_tracks, actual_track = self._setup_play(zp, False)
        time.sleep(10)
        # Position to 30 seconds before track end
        for x in range(self.samples):
            self.loading_next_track.clear()
            zp.AVTransport.seek_to_position_before_track_end(30)
            self.loading_next_track.wait(30)
            self.prebuffering.clear()
            self.wait_for_prebuffering(self.prebuffering)
            self.logger.info("Pre-buffered next track")
            self.measuring = True
            self.prebuffering.clear()
            self.zp_hit_next = time.time()
            zp.AVTransport.Seek('TIME_DELTA', delta)
            self.wait_for_prebuffering(self.prebuffering)
            self.verifyTrueOrFailCase(
                zp.AVTransport.wait_for_avt_state(
                    zp.AVTransport.PLAYING_STATE),
                "Expect playing after {}".format(msg))
            self.measuring = False
            time.sleep(5)
        self.calc_delay(zp, self.msg, msg)

[docs]    def next_previous_common(self, zp, direction, msg):
        """
        Common code for next/previous TTM measurements
        Measures TTM of skipping to previous/next track when we're playing
        current track.
        1. Play track.
        2. Sleep a couple of seconds.
        3. AVT operation Previous or Next (starts measurement timer)
        4. Stop measurement timer when the first-byte message is received
        5. Confirm previous/next track is playing
        6. Repeat 1-6 until X samples have been collected
        """
        if direction == NEXT:
            AVT_method = zp.AVTransport.Next
        elif direction == PREVIOUS:
            AVT_method = zp.AVTransport.Previous
        else:
            raise ValueError("Unsupported {}".format(direction))
        num_tracks, actual_track = self._setup_play(zp, False)
        tracks_to_play = self.samples
        self.measuring = True
        while tracks_to_play > 0:
            time.sleep(10)
            self.zp_hit_next = time.time()
            AVT_method()
            wait_until_true(lambda: zp.AVTransport.is_transitioning(),
                            timeout_seconds=TIMEOUT_FOR_STATE_CHANGE,
                            reason="failed to enter TRANSITIONING STATE")
            wait_until_true(lambda: zp.AVTransport.is_playing(),
                            timeout_seconds=TIMEOUT_FOR_STATE_CHANGE,
                            reason="failed to enter PLAYING STATE")
            expected_track, actual_track = self.get_expected_actual_track(
                zp, actual_track, num_tracks, direction)
            self.verifyEqualOrStop(
                expected_track,
                actual_track,
                "Playing expected track")
            tracks_to_play = tracks_to_play - 1
        # We start meausuring after first track has already been playing
        self.calc_delay(zp, self.msg, msg)

    def test_play_random_track(self, zp, ignore):
        """
        Measures TTM when selecting random track from playlist
        Play first track from playlist.
        Sleep for a bit.
        1. Select a random track other than the one playing.
        2. Point queue position to new track
        3. Start measurement timer
        4. Play new track
        5. Stop measurement timer
        6. Sleep for 10 seconds
        7. Repeat 1 until X samples have been taken
        """
        num_tracks, curr_track = self._setup_play(zp, False)
        tracks_to_play = self.samples
        self.logger.info("Playing initial track {}".format(curr_track))
        next_track = curr_track
        self.measuring = True
        time.sleep(10)
        while tracks_to_play > 0:
            # Pick a track other than the one playing
            while next_track == curr_track:
                next_track = randint(1, num_tracks)
