[docs]    def add_test_tracks(self, zp, container, min_tracks,
                        tracks_to_remove_tuple):
        """
        Given an album container, it adds the container to the
        default play queue and removes the tracks specified in the
        supplied tracks_to_remove_tuple.
        Returns the number of tracks in queue after all unwanted tracks
        have been removed.
        """
        num_tracks = self.add_to_queue(zp, container)
        self.verifyGreaterOrStop(
            num_tracks, min_tracks, "Expect to add tracks")
        for starting_track, tracks_to_remove in tracks_to_remove_tuple:
            self.verifyTrueOrStop(
                zp.AVTransport.RemoveTrackRangeFromQueue(
                    starting_track, tracks_to_remove),
                "Removed {} track(s) from queue starting track {}".format(
                    tracks_to_remove, starting_track))
            num_tracks = num_tracks - tracks_to_remove
        self.verifyEqualOrStop(
            num_tracks,
            zp.AVTransport.get_number_of_tracks_in_queue(),
            "Tracks in queue")
        return num_tracks

[docs]    def start_playback(self, zp, repeat=False, track=1):
        """
        Starts playback of the test album, trimmed to
        contain tracks suitable for the short or normal length
        tracks. For purposes of this test a short track is from
        4 to 10 seconds and a normal track is 45 to 75 seconds.
        """
        zp.AVTransport.use_queue()
        if repeat:
            zp.AVTransport.SetPlayMode("REPEAT_ALL")
        zp.AVTransport.clear_queue_and_wait()
        if self.normal_tracks:
            num_tracks = self.add_normal_length_tracks(zp)
        else:
            num_tracks = self.add_short_length_tracks(zp)
        zp.AVTransport.Seek('TRACK_NR', track)
        zp.RenderingControl.SetVolume(5)
        zp.AVTransport.Play_and_wait()
        return num_tracks



[docs]    def perform_next_based_on_probability(self, zp, track, duration,
                                          iterations,
                                          probability=DEFAULT_NEXT_PROBABILITY):
        """
        Based on probability, invoke Next() on currently playing track or wait
        for next track change.
        Return the number of remaining Next() operations for the test.
        We're being conservative on hitting Next() so that the Spotify backend
        does not rate limit the service.
        :param :class:`~sonos.client.SonosZoneComponent` zp:
        :param int track: Track number
        :param int duration: Track duration in seconds
        :param int iterations: The number of remaining next operations
        :param float probability: The likelihood of invoking the operation
        :return: The remaining number of next iterations
        :rtype: :obj:`int`
        """
        if random.random() < probability:
            iterations = iterations - 1
            self.logger.info(
                "Next on track {}"
                " - next ops to do {}".format(
                    track, iterations))
            zp.AVTransport.Next()
            wait_until_true(lambda: zp.AVTransport.is_transitioning(),
                            timeout_seconds=3,
                            reason="failed to enter TRANSITIONING STATE")
            wait_until_true(lambda: zp.AVTransport.is_playing(),
                            timeout_seconds=6,
                            reason="failed to enter PLAYING STATE")
        else:
            zp.AVTransport.wait_until_any_track_change(
                timeout=duration, delay=0.25)
        return iterations

    def test_play_and_seek_to_beginning(self, zp):
        """
        Confirms player plays current track when a seek to the start
        is invoked prior to end of track play. Addresses SWPBL-62066.
        The use of whether to invoke seek is probabilistic since it
        reduces the chance the Spotify rate limiter will kick-in.
        Will invoke DEFAULT_SEEK_TO_BEGINNING_ITERATIONS_SHORT
        1. Play track N (very short track ~4 seconds)
        2. Sleep until near the end of track N
        3. Scrub to beginning on track N (based on chance)
        4. Confirm playing and track N
        5. Repeat until iterations exhausted
        """
        if self.normal_tracks:
            track_change_sampling = 1.0
            start_track = 1
        else:
            track_change_sampling = 0.25
            start_track = 4
        iterations = self.seek_to_begin_iterations
        num_tracks = self.start_playback(zp, repeat=True, track=start_track)
        curr_track = zp.AVTransport.get_current_track()
        self.logger.info(
            "Test will perform {} Seek to begin"
            " operations on random tracks".format(iterations))
        while iterations > 0:
            pos_info = zp.AVTransport.GetPositionInfo()
            curr_track = int(pos_info['Track'])
            duration = self.get_seconds_from_pos_info(pos_info)
            track_change_timeout = duration + TRACK_CHANGE_TIMEOUT_SLACK
            played = self.get_seconds_from_pos_info(pos_info, name='RelTime')
            self.should_sleep_or_not_before_track_end(
                duration, played,
                min_allowance=self.min_allowance_before_track_end,
                max_allowance=self.max_allowance_before_track_end)
            if random.random() < self.seek_probability:
                iterations = iterations - 1
                self.logger.info(
                    "Seek to pos 0 on track {}"
                    " - seek ops to do {}".format(curr_track, iterations))
                zp.AVTransport.Seek('REL_TIME', '00:00:00')
                wait_until_true(lambda: zp.AVTransport.is_transitioning(),
                                timeout_seconds=3,
                                reason="failed to enter TRANSITIONING STATE")
                wait_until_true(lambda: zp.AVTransport.is_playing(),
                                timeout_seconds=6,
                                reason="failed to enter PLAYING STATE")
                actual_track = zp.AVTransport.get_current_track()
                if curr_track != actual_track:
                    self.collect_perf_counters(zp)
                    msg = "Stopping - Wrong track detected after seek pos 0"
                    " - expected {} actual {}".format(
                        curr_track, actual_track)
                    zp.diag.post_diagmsg('AUTOMATION FAIL: {} at {}'.format(
                        msg, datetime.now()))
                    self.stop(msg)
                self.verifyEqualOrFailCase(
                    curr_track,
                    actual_track,
                    "Expect track [{}/{}]".format(curr_track, num_tracks))
            else:
                self.logger.info(
                    "Waiting {} seconds for track change".format(duration))
                zp.AVTransport.wait_until_any_track_change(
                    timeout=track_change_timeout, delay=track_change_sampling)
            self._check_dmesg_for_call_trace(zp)

    def test_next_close_to_end(self, zp):
        """
        Confirms player switches to next track when a Next operation
        is invoked prior to end of track play.
        The test invokes next on a track based on probability to avoid
        Spotify rate limiting.
        Will invoke DEFAULT_NEXT_ITERATIONS
        1. Play track N
        2. Sleep until there's a second remaining in track N
        3. Next (based on chance)
        4. Confirm playing and track N + 1
        5. Repeat until time exhausted
        Errors seen without fix, either ERROR_LOST_CONNECTION,
        or ERROR_NO_RESOURCE and step (4) player on track N + 2.
        """
        num_tracks = self.start_playback(zp, repeat=True)
        curr_track = zp.AVTransport.get_current_track()
        iterations = self.next_iterations
        self.logger.info(
            "Test will perform {} Next"
            " operations on random tracks".format(iterations))
        while iterations > 0:
            pos_info = zp.AVTransport.GetPositionInfo()
            curr_track = int(pos_info['Track'])
            duration = self.get_seconds_from_pos_info(pos_info)
            played = self.get_seconds_from_pos_info(pos_info, name='RelTime')
            if not curr_track % num_tracks:
                expected_track = 1
            else:
                expected_track = curr_track + 1
            self.should_sleep_or_not_before_track_end(
                duration, played,
                min_allowance=self.min_allowance_before_track_end,
                max_allowance=self.max_allowance_before_track_end)
            iterations = self.perform_next_based_on_probability(
                zp, curr_track, duration, iterations,
                probability=self.next_probability)
            actual_track = zp.AVTransport.get_current_track()
            if expected_track != actual_track:
                self.collect_perf_counters(zp)
                msg = (
                    "Unexpected track after next - expected {} actual {}".
                        format(expected_track, actual_track))
                zp.diag.post_diagmsg('AUTOMATION FAIL: {} at {}'.format(
                    msg, datetime.now()))
                self.warn(msg)
            self.verifyEqualOrFailCase(
                expected_track,
                actual_track,
                "Expect track [{}/{}]".format(expected_track, num_tracks))
            self._check_dmesg_for_call_trace(zp)

    def test_pause_resume(self, zp):
        """
        1. Start playback of playlist
        2. After one second, pause and confirm track
        3. Sleep for DEFAULT_PAUSE_SECONDS, resume play confirm track
        4. Confirm track change
        5. Repeat 2-4 until time limit exhausted
        """
        num_tracks = self.start_playback(zp, repeat=True)
        curr_track = zp.AVTransport.get_current_track()
        end = time.time() + self.duration
        track_change_sampling = 1.0 if self.normal_tracks else 0.1
        self.logger.info(
            "Test will run for {} seconds"
            " pausing/resuming on tracks".format(self.duration))
        while time.time() < end:
            # Minimum track length 4 seconds for short duration
            # sleep early
            time.sleep(1)
            zp.AVTransport.Pause_and_wait()
            curr_track = self.verify_paused_resumed_track(
                zp,
                curr_track,
                self.warn,
                "Wrong track detected during pause"
                " - expected {} actual {}",
                self.verifyEqualOrWarn,
                "Paused track")
            time.sleep(DEFAULT_PAUSE_SECONDS)
            zp.AVTransport.Play_and_wait()
            curr_track = self.verify_paused_resumed_track(
                zp,
                curr_track,
                self.warn,
                "Wrong track detected after resume "
                "- expected {} actual {}",
                self.verifyEqualOrWarn,
                "Resumed track")
            if not curr_track % num_tracks:
                expected_track = 1
            else:
                expected_track = curr_track + 1
            duration = int(
                convert_time_to_timedelta(
                    zp.AVTransport.get_current_track_duration()).total_seconds())
            track_change_timeout = duration + TRACK_CHANGE_TIMEOUT_SLACK
            zp.AVTransport.wait_until_any_track_change(
                timeout=track_change_timeout, delay=track_change_sampling)
            curr_track = self.verify_paused_resumed_track(
                zp,
                expected_track,
                self.stop,
                "Wrong track change"
                " - expected {} actual {}",
                self.verifyEqualOrFailCase,
                "Playing track")
            self._check_dmesg_for_call_trace(zp)

    def test_seek_to_end(self, zp):
        """
        Note that this test will most likely generate a STREAM_LIMIT_ERROR
        due to Spotify's rate limiting policies on track play if the
        probability on seek operations is close to 1.00.
        Randomizing the periodicity of performing a seek operation may
        alleviate the STREAM_LIMIT_ERROR.
        Currently the default is to seek (on average) every other 3rd track.
        Note this generated an ERROR_NO_RESOURCE without the randomization
        on seek operations when the execution environment had a single player.
        Automation note: The following command line avoided the STREAM_LIMIT
        error yet it manifested in CORRUPT_FILE followed by LOST_CONNECTION.
        --iterations=10 --seek_probability=0.333
        approximately 2hours for 6 players
        1. Start playback of playlist
        2. Sleep for 1 seconds
        3. Seek to end of track depending on probability
        4. Wait for track change
        5. Confirm track change
        5. Repeat 2-5 until iterations exhausted
        """
        num_tracks = self.start_playback(zp, repeat=True)
        curr_track = zp.AVTransport.get_current_track()
        iterations = self.seek_to_end_iterations
        track_change_sampling = 1.0 if self.normal_tracks else 0.1
        self.logger.info(
            "Test will perform {} Seek to end operations"
            " on random tracks".format(iterations))
        while iterations > 0:
            duration = int(
                convert_time_to_timedelta(
                    zp.AVTransport.get_current_track_duration()).total_seconds())
            track_change_timeout = duration + TRACK_CHANGE_TIMEOUT_SLACK
            time.sleep(1)
            if random.random() < self.seek_probability:
                iterations = iterations - 1
                self.logger.info(
                    "Seek on track {}"
                    " - seek ops to do {}".format(curr_track, iterations))
                zp.AVTransport.seek_to_position_before_track_end(1)
            res = zp.AVTransport.wait_until_any_track_change(
                timeout=track_change_timeout, delay=track_change_sampling)
            if not res:
                self.collect_perf_counters(zp)
                msg = (
                    "Stopping - no track change within {} seconds on track {}".
                        format(track_change_timeout, curr_track))
                zp.diag.post_diagmsg('AUTOMATION FAIL: {} at {}'.format(
                    msg, datetime.now()))
                self.stop(msg)
            actual_track = zp.AVTransport.get_current_track()
            if not curr_track % num_tracks:
                expected_track = 1
            else:
                expected_track = curr_track + 1
            if expected_track != actual_track:
                self.collect_perf_counters(zp)
                msg = (
                    "Wrong track after track change - expected {} actual {}".
                        format(expected_track, actual_track))
                zp.diag.post_diagmsg('AUTOMATION FAIL: {} at {}'.format(
                    msg, datetime.now()))
                self.warn(msg)
            curr_track = actual_track
            self.verifyEqualOrFailCase(
                expected_track,
                actual_track,
                "Playing track {}".format(curr_track))
            self._check_dmesg_for_call_trace(zp)


    def _check_dmesg_for_call_trace(self, zp):
        if zp.cli.command("dmesg | grep 'Call trace'"):
            msg = 'Call trace found in dmesg'
            zp.diag.post_diagmsg('AUTOMATION FAIL: {} at {}'.format(
                msg, datetime.now()))
            self.stop_suite(
                "'Call trace' found in dmesg on <{}>".format(zp))
