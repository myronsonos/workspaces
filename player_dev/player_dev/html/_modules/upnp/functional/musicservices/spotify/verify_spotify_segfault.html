    """
    During the initial integration of NTS library, there were
    several tracks which caused the player to restart anacapad
    due to a segmentation fault in the NTS library.
    This script verifies the cuplrit tracks can be played
    without a fault.
    Verifies the following JIRA bugs.
    SWPBL-56579 - segfault while playing "Never Sleep Alone (feat. Tess Comrie)
    SWPBL-56681 - segfault while playing "Somewhere In Paradise (feat. Jeremih
    & R. Kelly"
    SWPBL-56277 - segfault while playing "The Angry Birds Movie"
    SWPBL-55781 - segfault while playing Spotify using NTS library

    Updated 11/7/2016 to play a mixed queue of Spotify tracks (several
    tracks which caused SEGFAULT) and local tracks comprised of all
    the supported framers SWPBL-65362.
    """
    def __init__(self):
        super(VerifySpotifySegfaults, self).__init__(
            logging_to_file=True, parameterized_setup=True)
        self.mod_log_levels['spot'] = 6
        self.mod_log_levels['zp'] = 4
        self.mod_log_levels['flac'] = 4

    def _parse_args(self):
        parser = argparse.ArgumentParser()
        parser.add_argument(
            "--flac", action='store_true',
            help="Test using Spotify and local FLAC")
        args, unknown_args = parser.parse_known_args()
        sys.argv[1:] = unknown_args
        self.flac = args.flac

[docs]    @combinatorial('generate_mixed_track_zones')
    def test_spotify_segfault_tracks(self, zp):
        """
        Play each track in container sequentially in container.
        1. Play 15 seconds and then
        2. Seek to 10 seconds before end
        3. Wait for track change
        4. Repeat 1-3 until queue exhausted
        """
        num_tracks = self.add_mixed_tracks_to_queue(
            zp, SPOTIFY_SEGFAULT_TRACKS, self.http_uris, shuffle=True)
        self.verifyGreaterOrStop(num_tracks, 0, "Expect to add tracks")
        curr_track = 1
        zp.AVTransport.Seek('TRACK_NR', curr_track)
        zp.RenderingControl.SetVolume(5)
        zp.AVTransport.Play_and_wait()
        if not zp.AVTransport.is_playing():
            self.stop(
                "Failed to play track [{}/{}] - exceeded timeout".format(
                    curr_track, num_tracks))
        while curr_track <= num_tracks:
            res = zp.AVTransport.wait_for_avt_state(
                zp.AVTransport.STOPPED_STATE,
                timeout_seconds=15)
            self.verifyFalseOrFailCase(res, "Expect continuous playback")
            pos_info = zp.AVTransport.GetPositionInfo()
            if curr_track != int(pos_info['Track']):
                self.logger.info("Detected wrong track")
                time.sleep(30)
            self.verifyEqualOrStop(
                curr_track,
                int(pos_info['Track']),
                "Playing expected track")
            if self.get_track_title(pos_info).endswith('.flac'):
                self.logger.warning(
                    "Track seek on {} is skipped due to SWPBL-65441".format(
                        self.get_track_title(pos_info)))
                try:
                    zp.AVTransport.Next()
                except UPnPError as e:
                    if e.upnp_error == UPNP_ERROR_711_ILLEGAL_SEEK:
                        pass
                curr_track = curr_track + 1
                continue
            self.logger.info(
                'About to seek track={} duration={} title="{}"'.format(
                    pos_info['Track'],
                    pos_info['TrackDuration'],
                    self.get_track_title(pos_info)))
            zp.AVTransport.seek_to_position_before_track_end(
                SECONDS_BEFORE_END)
            # If we're playing last track, wait for stop and queue position 1
            if curr_track == num_tracks:
                res = zp.AVTransport.wait_for_avt_state(
                    zp.AVTransport.STOPPED_STATE,
                    timeout_seconds=TRACK_CHANGE_TIMEOUT,
                    iteration_delay=1.0)
                self.verifyTrueOrFailCase(
                    res, "Played last track {}".format(curr_track))
                if 1 != zp.AVTransport.get_queue_position():
                    self.logger.info("Detected wrong track after playing last")
                    time.sleep(30)
                self.verifyEqualOrStop(
                    1,
                    zp.AVTransport.get_queue_position(),
                    "Track position after last track played")
                break
            next_track = curr_track + 1
            res = zp.AVTransport.wait_until_track_change(
                next_track, timeout=TRACK_CHANGE_TIMEOUT, delay=1.0)
            if not res:
                self.logger.error("Did not detect track change")
                time.sleep(30)
            self.verifyTrueOrStop(
                res,
                "Now playing track [{}/{}]".format(next_track, num_tracks))
            curr_track = next_track
