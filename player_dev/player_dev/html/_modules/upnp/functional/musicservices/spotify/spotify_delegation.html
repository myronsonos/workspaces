    """
    Confirms GC quick hand-off does not exceed a threshold SWPBL-60087
    Confirms track being played same as old GC SWPBL-60090
    During Spotify NTS integration, it was found that handing-off to a
    new GC took excessively long time (over 25 seconds) for the operation
    to complete (group size 3). In addition, the track being played by the
    new GC was not the track that had been playing on the old GC.
    """
    min_transition = 100.0
    max_transition = 0.0

    def _parse_args(self):
        parser = argparse.ArgumentParser()
        parser.add_argument(
            "--group_size", type=int,
            default=DEFAULT_GROUP_SIZE,
            help="Perform test with a group size of {} players".format(DEFAULT_GROUP_SIZE))
        args, unknown_args = parser.parse_known_args()
        sys.argv[1:] = unknown_args
        self.group_size = args.group_size

[docs]    def generate_grp_zones(self):
        """
        Using the unique playback devices and the desired
        group size, yield all combinations to test the quick GC hand-off.
        """
        test_zones = self.test_devices[0:self.group_size]
        shuffle(test_zones)
        for gc, gm_to_gc in combinations(test_zones, PAIR):
            # exclude gc and gm_to_gc in members' list
            gms = test_zones[:]
            gms.remove(gc)
            # SWPBL-113607: new GC will be first GM
            gm_to_gc = gms[0]
            yield (gc, gms, gm_to_gc)

    def _setup_udp_logger(self, zp):
        """
        Setup udp logger and pass in callbacks for strings we are looking for
        :param zp: sonos zone player for which to setup udp logger
        :type zp: :class:``~sonos.client.internal.SonosZoneComponent`
        """
        self.logger.info("Setting up udp logger on : <{}>".format(zp))
        try:
            zp_log = "{}_{}_playback.log".format(
                zp.modelNumber,
                datetime.now().strftime("%H_%M_%S"))
            zp.diag.start_udp_log_to_file(zp_log)
        except SocketError as e:
            self.logger.critical("UDPLogger port error: {}".format(e))
            self.stop_suite()
        zp.diag.watch_udp_log_for_string(
            PLAYING_MUSIC,
            lambda msg: self.detected_playing_music(zp))
        [zp.diag.watch_udp_log_for_string(
                error_string, lambda msg: self.cb_transport_error(msg, zp))
         for error_string in PLAYBACK_ERRORS]


[docs]    def start_playback(self, zp):
        """
        Start playback
        :param test_zone: Sonos zone component
        :type test zone: :`~sonos.client.SonosZoneComponent`
        """
        self.logger.info("Beginning playback on {}".format(zp))
        self._add_tracks_to_queue(zp)
        qpos = randint(1, self.num_tracks)
        zp.AVTransport.set_queue_position(qpos)
        zp.AVTransport.SetCrossfadeMode(False)
        zp.AVTransport.SetPlayMode("REPEAT_ALL")
        zp.RenderingControl.SetVolume(1)
        zp.AVTransport.Play_and_wait()


    def _add_tracks_to_queue(self, zp):
        """
        Add tracks to queue on this zp
        :param test_zone: Sonos zone component
        :type test zone: :`~sonos.client.SonosZoneComponent`
        """
        self.logger.info("Adding tracks to queue on <{}>".format(zp))
        zp.AVTransport.use_queue()
        metadata = zp.AVTransport.generate_didl_lite(
            title=ROCK_BALLADS.title,
            svc_id=self.svc_acct.service_id,
            acct_id=self.svc_acct.username)
        self.num_tracks = zp.AVTransport.add_to_queue(
            ROCK_BALLADS.uri, metadata)
        msg = "Have {} tracks in queue {} removing restricted"
        self.logger.info(msg.format(self.num_tracks, "before"))
        self.num_tracks = zp.AVTransport.remove_restricted_tracks()
        self.logger.info(msg.format(self.num_tracks, "after"))

    def _cleanup(self, zp):
        """
        Clear avt and queue, remove bonded config if it exists
        :param zp: Sonos zone component
        :type zp: :class:`~sonos.client.SonosZoneComponent`
        """
        self.logger.info("Performing cleanup on: <{}>".format(zp))
        zp.DeviceProperties.cleanup_bonded_configuration()
        zp.AVTransport.clear_play_state()
        zp.AVTransport.clear_queue_and_wait()

    def _collect_perf_counters(self, zp):
        """
        Collect performance counters from zp
        :param zp: Sonos zone component
        :type zp: :class:`~sonos.client.SonosZoneComponent`
        """
        PERF_URL = 'http://{}:1400/status/perf'
        perf_log = '{}_{}_perf_counters.xml'.format(
            zp.modelNumber, datetime.now().strftime('%H_%M_%S'))
        perf_data = zp.http.getURL(PERF_URL.format(zp.ip))
        with open(perf_log, 'w') as f:
            f.write(perf_data)
        self.logger.info("Perf counters saved to: {}".format(perf_log))

        """
        Callback on detecting music playback
        :param zp: Sonos zone component from which to submit
        :type zp: :class:`~sonos.client.SonosZoneComponent`
        """
        self.logger.info("Detected playing music on {}".format(zp))

    def _did_all_gms_play(self, gms):
        """
        Check if all gms played TV
        :param list gms: List of Sonos zone components as group members
        :return: Did all devices play tv ?
        :rtype: :obj:`bool`
        """
        self.logger.info(
            "Played music state values: {}".format(
                self.playing_music))
        return all([self.playing_music[zp] for zp in gms])

    def _verify_all_gms_playing_music(self, gms):
        """
        Wait until all gms are playing on timeout, stop the test
        :param list gms: List of Sonos zone components as group members
        """
        try:
            wait_until_true(lambda: self._did_all_gms_play(gms),
                            iteration_delay=1,
                            timeout_seconds=MUSIC_DETECT_TIMEOUT)
        except TimeoutError:
            self.error_msg = 'Did not play music on all players'
            for zp in self.playing_music:
                if not self.playing_music[zp]:
                    self.logger.critical(
                        "Did not detect music playing on <{}>".format(zp))
            self.stop(self.error_msg)
        else:
            self.logger.log(
                AssertionLevel.PASS,
                "Playing music on all gms: <{}>".format(gms))

    def _verify_all_zps_linked(self, gc, gms):
        """
        Verify members in gms are linked to gc.
        Stops test when a member zone player not grouped to gc.
        :param zp: Sonos zone component - Group coordinator
        :type zp: :class:`~sonos.client.SonosZoneComponent`
        :param list gms: Zone members
        """
        expected_uuids = [zp.raw_udn for zp in gms]
        actual_uuids = gc.ZoneGroupTopology.get_zone_player_uuids_in_group()
        missing_uuids = [uuid for uuid in expected_uuids
                         if uuid not in actual_uuids]
        if missing_uuids:
            self.logger.critical(
                "Current UUID's in group: {}"
                .format(gc.ZoneGroupTopology.get_zone_player_uuids_in_group()))
            self.logger.critical("Missing UUIDs: <{}>".format(missing_uuids))
            self.stop("Some members are not linked to <{}>".format(zp))
        else:
            self.logger.log(
                AssertionLevel.PASS,
                "All members linked to: <{}>".format(gc))

[docs]    def verify_grouping_quick_hand_off(self, gc, gms, gm_to_gc,
                                       gc_timeout=NEW_GC_TRANSITION_TIMEOUT):
        """
        Zones specified in gms are grouped to the initial gc.
        Zone gm_to_gc is grouped to existing group and the original gc
        becomes a standalone player.
        Confirms group coordinator quick hand-off.
        :param :class:`sonos.client.internal.SonosZoneComponent` gc:
        Initial group coordinator
        :param list gms: Members of group
        :param :class:`sonos.client.internal.SonosZoneComponent` gm_to_gc:
        Final group coordinator
        :param int gc_timeout: Maximum number of seconds to wait for
        quick hand-off.
        """
        self.logger.info("Linking zones to gc: <{}>".format(gc))
        self.logger.info("Members <{}>".format(gms))
        self.logger.info("Zone <{}> will acquire role as GC".format(gm_to_gc))
        for zp in gms:
            self.playing_music[zp] = False
        gc.AVTransport.link_zones(gms, timeout=120)
        self._verify_all_zps_linked(gc, gms)
        self.start_playback(gc)
        # verify all gms playing music
        self.logger.info("Checking if all GM players are playing music")
        self._verify_all_gms_playing_music(gms)
        original_track = gc.AVTransport.get_current_track()
        gm_uri = gm_to_gc.AVTransport.get_avtransport_uri()
        self.verifyEqualOrStop(
            'x-rincon:{}'.format(gc.raw_udn),
            gm_uri,
            'GM URI should contain GC raw UDN')
        time.sleep(15)
        self.logger.info("<{}> leaving group".format(gc))
        now = time.time()
        gc.AVTransport.BecomeCoordinatorOfStandaloneGroup(
            timeout_seconds=STANDALONE_TRANSITION_TIMEOUT)
        gc.AVTransport.SetAVTransportURI_and_wait("", "")
        transition_time = time.time() - now
        msg = "Expect new GC <{}> quick hand-off within {} seconds".format(
            gm_to_gc.friendly_name, gc_timeout)
        if not self.verifyLessOrFailCase(transition_time, gc_timeout, msg):
            for zp in [gc, gm_to_gc]:
                zp.diag.post_diagmsg(
                    'AUTOMATION FAIL: {} at {}'.format(msg, datetime.now()))
        self.min_transition = min(self.min_transition, transition_time)
        self.max_transition = max(self.max_transition, transition_time)
        res = gm_to_gc.AVTransport.wait_until_track_change(
            original_track, delay=1)
        playing_track = gm_to_gc.AVTransport.get_current_track()
        actual_uri = gm_to_gc.AVTransport.get_avtransport_uri()
        self.verifyEqualOrStop(
            'x-rincon-queue:{}#0'.format(gm_to_gc.raw_udn),
            actual_uri,
            'New GC should not be a GM')
        base_msg = (
            "New GC should play expected track {}, actual track {}".
            format(original_track, playing_track))
        self.logger.debug(base_msg)
        new_gc_msg = 'New GC = {}'.format(gm_to_gc)
        expected_uri_msg = 'Expected URI = x-rincon-queue:{}#0'.format(
            gm_to_gc.raw_udn)
        actual_uri_msg = 'Actual URI = {}'.format(actual_uri)
        if original_track != playing_track:
            for zp in [gc, gm_to_gc]:
                zp.diag.post_diagmsg(
                    'AUTOMATION FAIL: {} at {}'.format(base_msg, datetime.now()))
                for msg in [new_gc_msg, expected_uri_msg, actual_uri_msg]:
                    zp.diag.post_diagmsg(msg)
            for msg in [new_gc_msg, expected_uri_msg, actual_uri_msg]:
                self.logger.debug(msg)
            self.stop(base_msg)
        if not res and not self.did_zone_have_transport_error(gm_to_gc):
            self.logger.error(
                "Wrong track {} on new GC <{}> collecting perf".format(
                    playing_track, gm_to_gc.friendly_name))
            self._collect_perf_counters(gm_to_gc)
            msg = "New GC is playing wrong track {}".format(playing_track)
            for zp in [gc, gm_to_gc]:
                zp.diag.post_diagmsg(
                    'AUTOMATION FAIL: {} at {}'.format(msg, datetime.now()))
                for msg in [new_gc_msg, expected_uri_msg, actual_uri_msg]:
                    zp.diag.post_diagmsg(msg)
            for msg in [new_gc_msg, expected_uri_msg, actual_uri_msg]:
                self.logger.debug(msg)
            self.stop(msg)
        self._verify_all_zps_linked(gm_to_gc, gms)
        self.verifyTrueOrFailCase(
            all([gm.AVTransport.is_playing() for gm in gms]),
            "All members are playing")
