[docs]class SpotifyFastSlowPath(SpotifyBase):
    """
    Verifies players use fast/slow path on seek/pause/resume - SWPBL-60719
    Seek fast path is defined as when a user seeks to a position
    within a track and prior to the next track being played. There's a
    transitional time where the player (chsrc) is playing the next track
    but the framing for the current track is still pending. A seek and
    pause in this transitional time is a slow path since we're forcing the
    player to re-open the expiring track file.
    """
    def __init__(self):
        super(SpotifyFastSlowPath, self).__init__(
            logging_to_file=True, parameterized_setup=True)

[docs]    def wait_msg_event(self, event, timeout=5.0):
        """
        Waits for the event. Return the state of the event.
        :param threading.Event event:
        :param float timeout: Maximum seconds to wait
        :return: Is event set ?
        :rtype: :obj:`bool`
        """
        return event.wait(timeout)



[docs]    def verify_fast_seek(self, zp):
        """
        Test the player utilizes the fast seek path.
        Allow the player to play enough content on current track.
        Perform seek (forward/backward) UPnP operations.
        Confirm player uses expected code path.
        """
        zp.diag.post_diagmsg(
            'AUTOMATION INFO: Start verify_fast_seek at {}'.format(
                datetime.now()))
        time.sleep(5)
        msg = 'Seek fast path:'
        self.set_msg_event(zp, msg, self.event)
        for delta in ('+00:00:05', '-00:00:05',):
            zp.AVTransport.Seek('TIME_DELTA', delta)
            res = self.wait_msg_event(self.event)
            log_msg = "Expected {} on seek {}".format(msg, delta)
            if not self.verifyTrueOrFailCase(res, log_msg):
                zp.diag.post_diagmsg(
                    'AUTOMATION FAIL: {} at {}'.format(
                        log_msg, datetime.now()))
            self.verifyTrueOrFailCase(
                zp.AVTransport.wait_for_avt_state(
                    zp.AVTransport.PLAYING_STATE),
                "Expect playing after fast seek")
            self.event.clear()
        self.clear_msg_event(zp, msg)
        zp.diag.post_diagmsg(
            'AUTOMATION INFO: End verify_fast_seek at {}'.format(
                datetime.now()))



[docs]    def verify_fast_pause(self, zp):
        """
        Test the player utilizes the fast pause path.
        The testing is limited to players with limited frame-cache.
        Allow the player to play enough content on current track.
        Perform pause and resume.
        Confirm player uses expected code path.
        """
        zp.diag.post_diagmsg(
            'AUTOMATION INFO: Start verify_fast_pause at {}'.format(
                datetime.now()))
        time.sleep(5)
        msg = 'Resume from pause fast path'
        self.set_msg_event(zp, msg, self.event)
        zp.AVTransport.Pause()
        time.sleep(1)
        zp.AVTransport.Play_and_wait()
        res = self.wait_msg_event(self.event, timeout=19)
        log_msg = "Expected '{}' on resume".format(msg)
        if not self.verifyTrueOrFailCase(res, log_msg):

                log_msg, datetime.now()))
        self.verifyTrueOrFailCase(
            zp.AVTransport.wait_for_avt_state(

            "Expect playing after fast pause")
        self.clear_msg_event(zp, msg)
        zp.diag.post_diagmsg(
            'AUTOMATION INFO: End verify_fast_pause at {}'.format(
                datetime.now()))

[docs]    def verify_slow_pause(self, zp):
        zp.diag.post_diagmsg(
            'AUTOMATION INFO: Start verify_slow_pause at {}'.format(
                datetime.now()))
        zp.AVTransport.seek_to_position_before_track_end(20)
        msg = 'PLAYING QUEUEING QUEUE_URI'
        self.set_msg_event(zp, msg, self.event)
        res = self.wait_msg_event(self.event, timeout=30)
        if not res:
            self.fail("Failed track transition on slow pause")
            return
        msg = 'Resume from pause slow path'
        self.set_msg_event(zp, msg, self.event)
        self.logger.info("Pause to verify slow/pause path - 5 seconds pause")
        time.sleep(5)
        zp.AVTransport.Pause_and_wait()
        time.sleep(3)
        # zp.AVTransport.Seek('TIME_DELTA', '+00:00:01')
        zp.AVTransport.Play_and_wait()
        res = self.wait_msg_event(self.event, timeout=15)

[docs]    def start_playback(self, zp):
        container = LICENSE_TO_CHILL
        self.logger.info("{} playing container {}".format(zp, container))
        zp.AVTransport.use_queue()
        zp.AVTransport.clear_queue_and_wait()
        metadata = self.zp.AVTransport.generate_didl_lite(
            title=escape(container.title),
            svc_id=self.svc_acct.service_id,
            acct_id=self.svc_acct.username)
        num_tracks = zp.AVTransport.add_to_queue(container.uri, metadata)
        if not num_tracks > 0:
            self.stop("{} - failed to add tracks".format(zp))
        zp.RenderingControl.SetVolume(5)
        zp.AVTransport.SetPlayMode("REPEAT_ALL")
        zp.AVTransport.set_queue_position(randint(1, num_tracks))
        zp.AVTransport.Play_and_wait()
