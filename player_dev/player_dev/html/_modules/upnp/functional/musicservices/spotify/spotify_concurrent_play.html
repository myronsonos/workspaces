[docs]    def setUpFixture(self):
        super(SpotifyConcurrentPlay, self).setUpFixture()
        self.featured = self.get_featured_containers()
        if not self.featured:
            self.stop("Must have at least one featured playlist")
        self.unique_playback = [zp for zp in self.generators.generate_testbed_unique_playback_devices()]
        # Create a dictionary of zones mapping to a list of containers
        # that each zone will play.
        self.zp_to_container = {}
        dq = deque()
        for zp in self.unique_playback:
            self.zp_to_container[zp] = []
            dq.append(zp)
        # Distribute the playlist load among the available players.
        for container in self.featured:
            self.zp_to_container[dq[0]].append(container)
            dq.rotate()
        # Update the unique players based on available content
        # in the event there're more players than playlists.
        self.unique_playback = [zp for zp in self.zp_to_container
                                if self.zp_to_container[zp]]
        self._parse_args()
        self.track_change_timeout = self.seek_pos + 180
        self.logger.info(
            "Using {} seconds for seek position before track end".format(
                self.seek_pos))
        self.logger.info(
            "Using seek every {} track".format(self.seek_rate))
        self.logger.info(
            "Using {} seconds for track change timeout after seek".format(
                self.track_change_timeout))
        self.logger.info(
            "Using staggereded start {}".format(self.staggered_start))
        self.logger.info(
            "Using randomization track play/seek {}".format(self.randomize))
        self.logger.info(
            "Using {} between seeks on HH".format(self.min_seek_interval))
        self.logger.info(
            "Using Seek throttling on HH {}".format(self.seek_throttle))
        self.logger.info(
            "Using maximum playback duration {}".format(self.max_duration))



[docs]    def allow_seek(self, curr_track):
        """
        Returns True when conditions allows to seek on currently
        playing track based on the run time seek rate and the last
        invocation of seek on the HH.
        :param int curr_track: Track number playing
        :return: True when seeking is allowed else False
        :rtype: :obj:`bool`
        """
        allowed = False
        if not curr_track % self.seek_rate:
            if self.seek_throttle:
                with self.seek_lock:
                    now = time.time()
                    allowed = (now - self.last_seek_time) > self.min_seek_interval
                    if allowed:
                        self.last_seek_time = now
                    self.logger.debug(
                        "allowed seek {} now={} last={}".format(
                            allowed, now, self.last_seek_time))
            else:
                allowed = True
        return allowed



[docs]    def next_expected_track_number(self, pos, restricted, num):
        """
        Returns the next unrestricted track number that follows
        the specified position.
        :param int pos: current track number
        :param list restricted: unavailable track numbers
        :param int num: number of available tracks
        :return: Next playable track number
        :rtype: :obj:`int`
        """
        # Maximum number of attempts to find an unrestricted track is
        # the total number of tracks (playable and restricted)
        tries = num
        new = pos + 1
        while tries > 0:
            if new > num:
                new = 1
            if new not in restricted:
                self.logger.debug("next track number {}".format(new))
                return new
            tries = tries - 1
            new = new + 1
        self.logger.error("no playable tracks")
        return -1



[docs]    def wait_for_end_of_track(self, zp,
                              curr_track, restricted, num_tracks, timeout):

        """
        Waits for track change, verifies track position on change.
        Returns a tuple with a state and track position. The state
        of False indicates no more tracks to play while True there
        are more tracks.
        :param int curr_track: Current track playing
        :param list restricted: List of unplayable track numbers
        :param int num_tracks: Total tracks
        :param int timeout: Maximum time in seconds to wait for track
        change
        :return: Tuple (state, track)
        :rtype :obj:`tuple`
        """
        self.logger.debug(
            "{} waiting {} seconds for track change".format(zp, timeout))
        res = zp.AVTransport.wait_until_any_track_change(
            timeout=timeout, delay=1.0)
        self.verifyTrueOrFailCase(
            res,
            "{} switched to next track".format(zp))
        track_pos = zp.AVTransport.get_current_track()
        if curr_track == num_tracks or track_pos == 1:
            self.logger.info(
                "{} played last track {}".format(zp, curr_track))
            self.verifyEqualOrWarn(
                1,
                track_pos,
                "{} expected track position".format(zp))
            return (False, curr_track)
        curr_track = self.next_expected_track_number(
            curr_track, restricted, num_tracks)
        self.verifyEqualOrWarn(
            curr_track,
            track_pos,
            "{} playing expected track [{}/{}]".format(
                zp, curr_track, track_pos))
        return (True, curr_track)



[docs]    def play_container(self, zp, container, expiration):
        """
        Attempts to play all tracks in container.
        To expedite the test, plays the first 15 seconds of the
        track and then advances to 10 seconds before the end.
        Monitors for track changes and flags as failed test the
        unexpected track changes or any fatal playback errors.
        """
        self.logger.info("{} playing container {}".format(zp, container))
        zp.AVTransport.use_queue()
        zp.AVTransport.clear_queue_and_wait()
        metadata = self.zp.AVTransport.generate_didl_lite(
            title=escape(str(container.title)),
            svc_id=self.svc_acct.service_id,
            acct_id=self.svc_acct.username)
        num_tracks = zp.AVTransport.add_to_queue(container.uri, metadata)
        if not num_tracks > 0:
            self.stop("{} - failed to add tracks".format(zp))
        self.logger.info(
            "{} added {} tracks in playlist".format(zp, num_tracks))
        restricted = zp.AVTransport.get_restricted_track_numbers()
        curr_track = self.next_expected_track_number(0, restricted, num_tracks)
        self.verifyGreaterOrStop(
            curr_track,
            0,
            "First playable track number")
        zp.AVTransport.Seek('TRACK_NR', curr_track)
        zp.RenderingControl.SetVolume(5)
        self.logger.info(
            "{} playing {} tracks {} restricted {}".format(
                zp, num_tracks, len(restricted), restricted))
        zp.AVTransport.Play_and_wait()
        while curr_track <= num_tracks and time.time() < expiration:
            pos_info = zp.AVTransport.GetPositionInfo()
            track_pos = int(pos_info['Track'])
            duration = self.get_seconds_from_pos_info(pos_info)
            if duration < MINIMUM_TRACK_DURATION:
                self.logger.info(
                    "Short track '{}' with duration {} seconds".format(
                        self.get_track_title(pos_info), duration))
                (res, curr_track) = self.wait_for_end_of_track(
                    zp, curr_track, restricted, num_tracks, duration + 10)
                if res:
                    continue
                else:
                    break
            # Randomize the duration of initial play of the track
            timeout = MINIMUM_INITIAL_PLAYBACK
            if self.randomize:
                timeout = random.randint(-5, 5) + timeout
            self.logger.debug(
                "{} setting initial play duration {} seconds".format(
                    zp, timeout))
            res = zp.AVTransport.wait_for_avt_state(
                zp.AVTransport.STOPPED_STATE,
                timeout_seconds=timeout)
            self.verifyFalseOrFailCase(
                res,
                "{} unexpectedly failed playback [{}/{}]".format(
                    zp, curr_track, num_tracks))
            expected_track = curr_track == track_pos
            self.verifyEqualOrWarn(
                curr_track,
                track_pos,
                "{} expected track being played".format(zp))
            self.logger.info(
                "{} playing track={} duration={} title='{}'".format(
                    zp,
                    track_pos,
                    duration,
                    self.get_track_title(pos_info)))
            # If we're not playing the expected track, fail the test
            # and re-assign running track variable to current track to
            # continue test
            if not expected_track:
                if curr_track == num_tracks:
                    self.logger.info(
                        "{} played last track {}".format(
                            zp, curr_track))
                    self.verifyEqualOrWarn(
                        1,
                        track_pos,
                        "{} expected track position".format(zp))
                    break
                self.warn(
                    "{} unexpected switch to {} track from {}".format(
                        zp, track_pos, curr_track))
                curr_track = track_pos
                # Catch black listing service
                if curr_track == 1:
                    self.warn("{} unexpected track position 1".format(zp))
                    break
            # seekable tracks when rate allows
            if (self.allow_seek(curr_track) and self.seek_pos < duration):
                # randomize the seek position before track end if configured
                rand_seek_pos = self.seek_pos
                if self.randomize:
                    rand_seek_pos = random.randint(-5, 5) + rand_seek_pos
                self.logger.debug(
                    "{} invoking seek {} seconds BTE {} duration {}".format(
                        zp, rand_seek_pos, curr_track, duration))
                zp.AVTransport.seek_to_position_before_track_end(rand_seek_pos)
            (res, curr_track) = self.wait_for_end_of_track(
                zp, curr_track, restricted, num_tracks, duration)
            if not res:
                break



[docs]    def spotify_playlists(self, zp):
        """
        Play the assigned containers on the zone player.
        """
        # Stagger the startup to space out plays/seeks
        if self.staggered_start:
            wait_time = random.randint(
                1, self.seek_pos + MINIMUM_INITIAL_PLAYBACK)
            self.logger.info(
                "{} delaying start for {} seconds".format(
                    zp, wait_time))
            time.sleep(wait_time)
        self._setUpTest(zp)
        self.logger.info(
            "{} has {} containers to play".format(
                zp, len(self.zp_to_container[zp])))
        expiration = time.time() + self.max_duration
        for container in self.zp_to_container[zp]:
            self.play_container(zp, container, expiration)
        self._tearDownTest(zp)
