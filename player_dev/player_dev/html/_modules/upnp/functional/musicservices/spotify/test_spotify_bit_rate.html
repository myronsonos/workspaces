[docs]class SpotifyBitRate(SpotifyBase):
    """
    Bandwidth and delay tests to verify NTS bit rate adaptation
    The test is formulated to expect the NTS to downgrade
    to 160Kbit from 320Kbit tracks when the network degrades.
    When the network improves sufficiently to maintain playback
    on 320Kbit tracks, NTS will switch playing the higher rate.
    Verifies SWPBL-58303
    """
    def __init__(self):
        super(SpotifyBitRate, self).__init__(logging_to_file=True, parameterized_setup=True)

    def _get_fixture_path(self):
        """
        Return the absolute path this module is executing from.
        """
        return os.path.dirname(
            os.path.abspath(inspect.getfile(self.__class__)))

    def _load_test_config(self):
        """
        Consume and make available the traffic control file.
        """
        current_path = self._get_fixture_path()
        config_fn = os.path.join(current_path, TC_CONFIG)
        self.verifyTrueOrStop(
            os.path.exists(config_fn),
            "expect test config file '{}' to exist".format(config_fn))
        config = load_workflow_execution_configuration_from_json(config_fn)
        self.verifyIsNotNoneOrStop(
            config,
            "expect WorkflowExecutionConfiguration from '{}' "
            "to contain content".format(config_fn))
        self.execution_configuration.as_dict().update(config.as_dict())
        self.verifyTrueOrStop(
            hasattr(self.execution_configuration, "traffic_shaper"),
            "expect 'traffic_shaper' section to be specified "
            "in test configuration after loading")
        self.traffic_shaper = self.execution_configuration.traffic_shaper
        self.logger.debug(
            "Traffic shaper {} iface {}".format(
                self.traffic_shaper.host, self.traffic_shaper.interface))

[docs]    def play_spotify_container(self, zp, container):
        """
        """
        zp.AVTransport.use_queue()
        zp.AVTransport.clear_queue_and_wait()
        num_tracks = self.add_to_queue(zp, container)
        self.verifyGreaterOrStop(num_tracks, 0, "Expect to add tracks")
        zp.AVTransport.Seek('TRACK_NR', randint(1, num_tracks))
        zp.AVTransport.SetPlayMode("REPEAT_ALL")
        zp.RenderingControl.SetVolume(5)
        zp.AVTransport.Play_and_wait()
        res = zp.AVTransport.wait_for_avt_state(
            zp.AVTransport.STOPPED_STATE, timeout_seconds=3)
        self.verifyFalseOrStop(res, "Unexpectedly stopped")
        self.verifyEqualOrFailCase(
            320, self.current_rate, "Expect to play 320kbps track")



[docs]    def wait_until_track_rate_change(self, rate, timeout=300):
        try:
            wait_until_true(
                lambda: self.current_rate == rate,
                timeout_seconds=timeout,
                iteration_delay=5.0,
                reason="timed waiting for track rate change")
        except TimeoutError:
            return False
        else:
            return True



[docs]    def bit_rate_track_cb(self, msg):
        """
        Updates the current track play bit rate
        :param str msg: Player message
        """
        rate = re.search(
            ur"NTS: I Requesting Bytes: 0k =>(?:.*?), rate: (\d+)k", msg)
        if rate:
            self.current_rate = int(rate.group(1))



[docs]    def configure_mock_player_services(self):
        # Set up mock so we can use the feature config service
        bounce_anacapa = False
        self.mock_default_args = {'gateway': None,
                                  'stdout': sys.stdout,
                                  'stderr': sys.stderr}
        self.mock_player_cloud_services = MockPlayerCloudServicesControl(**self.mock_default_args)

        # Use feature config service to enable ABR and set timeout(minutes)
        feature_config = {"enableSpotifyAdaptiveBitrate": True, "timeoutSpotifyLowerBitrate": 1}
        self.mock_player_cloud_services.set_config_properties(feature_config)

        EXPECTED_OVERRIDES = {SERVICE_OVERRIDE: self.mock_player_cloud_services.SERVICE_OVERRIDE_URL,
                              SERVICE_OVERRIDE_MUTUAL_SSL: self.mock_player_cloud_services.SERVICE_OVERRIDE_URL_MUTUAL_SSL}

        # Override mutual SSL settings(*will restart anacapa) if the player needs it
        if not self.zp.SystemProperties.verifySysSettingStrings(EXPECTED_OVERRIDES):
            self.mock_player_cloud_services.set_player_connect_override_mutual_ssl(zp=self.zp, report=False)
            bounce_anacapa = True

        if bounce_anacapa:
            # Control whether to bounce anacapa instead of doing it through "set_player_connect_override_mutual_ssl"
            # In the small chance that the player has the ssl overrides but not the right cloudconfig settings
            self.zp.bounce_anacapa_and_wait()
            wait_until_true(predicate=lambda: self.zp.muse_rest.getGroupId() is not None,
                            timeout_seconds=10, iteration_delay=1,
                            reason="Waiting for <{}> to get a groupId".format(self.zp.friendly_name))

        # Verify cloudconfig.json has expected values
        expected_config = {"spotifyAdaptiveBitrate": {"enable": True, "timeoutMinutes": 1}}
        wait_until_true(lambda: self.zp.cli.has_feature_config() and
                                self.zp.cli.verify_feature_config(expected_config),
                        iteration_delay=1, timeout_seconds=60,
                        reason="cloudconfig file on <{}> has expected values <{}>?".format(self.zp.friendly_name,
                                                                                           feature_config))

        # Wait and verify cloud connected
        self.zp.diag.wait_for_cloud_connected(timeout_seconds=60)

        self.mock_player_cloud_services.clear_log()

    def test_spotify_bit_rate_delay(self, zp):
        """
        SWPBL-125470
        1. Start playback
        2. Confirm player using 320K track
        3. Increase delay
        4. Confirm player using 160K track
        5. Reduce delay
        6. Confirm player using 320K track after timeoutSpotifyLowerBitrate(set to 60 seconds in spotify_base)
        """
        self.skip("Not convinced the delay triggers the ABR mechanism reliably")

        self.play_spotify_container(zp, ROCK_BALLADS)

        # Start with low delay
        self.tc.enable_netem_delay(
            self.traffic_shaper.interface,
            zp,
            ARCH_TO_DELAY.get(zp.modelNumber, LOW_DELAY),
            port=NTS_PORTS)
        res = self.wait_until_track_rate_change(320)
        self.verifyTrueOrFailCase(
            res, "Expected to transition to 320kbps tracks")

        # Shape to high delay
        self.logger.debug(
            "Changing tc delay to {}ms on port {}".format(DEFAULT_HIGH_DELAY, NTS_PORTS))
        self.tc.change_netem_delay(
            self.traffic_shaper.interface, zp, DEFAULT_HIGH_DELAY)
        res = self.wait_until_track_rate_change(160)
        self.verifyTrueOrFailCase(
            res, "Expected to transition to 160kpbs tracks")

        # After timeoutSpotifyLowerBitrate and flush/skip should retry higher kbps track
        self.logger.debug(
            "Changing tc delay to {}ms on port {}".format(LOW_DELAY, NTS_PORTS))
        self.tc.change_netem_delay(
            self.traffic_shaper.interface, zp, LOW_DELAY)
        guaranteed_sleep(60)
        self.zp.AVTransport.Next()
        res = self.wait_until_track_rate_change(320)
        self.verifyTrueOrFailCase(
            res, "Expected to transition to 320kpbs tracks")

        res = zp.AVTransport.wait_for_avt_state(
                zp.AVTransport.STOPPED_STATE,
                timeout_seconds=15)
        self.verifyFalseOrFailCase(res, "Unexpect stop playing new track rate")

    def test_spotify_bit_rate_band(self, zp):
        """
        SWPBL-125470
        1. Start playback
        2. Confirm player using 320K track
        3. Reduce bandwidth
        4. Confirm player using 160K track
        5. Increase bandwidth
        6. Confirm player using 320K track after timeoutSpotifyLowerBitrate(set to 60 seconds in spotify_base)
        """
        self.play_spotify_container(zp, ROCK_BALLADS)

        # Start with 320kbps
        self.tc.enable_rate_limit(
            self.traffic_shaper.interface, zp, HIGH_BANDWIDTH, port=NTS_PORTS)
        res = self.wait_until_track_rate_change(320)
        self.verifyTrueOrFailCase(
            res, "Expected to transition to 320kbps tracks")

        # Shape down to 160kbps
        self.logger.debug(
            "Changing tc bandwidth to {}kbps on port {}".format(LOW_BANDWIDTH, NTS_PORTS))
        self.tc.change_rate_limit(
            self.traffic_shaper.interface, LOW_BANDWIDTH)
        res = self.wait_until_track_rate_change(160)
        self.verifyTrueOrFailCase(
            res, "Expected to transition to 160kbps tracks")

        # After timeoutSpotifyLowerBitrate and flush/skip should retry higher kbps track
        self.logger.debug(
            "Changing tc bandwidth to {}kbps on port {}".format(HIGH_BANDWIDTH, NTS_PORTS))
        self.tc.change_rate_limit(
            self.traffic_shaper.interface, HIGH_BANDWIDTH)
        guaranteed_sleep(60)
        self.zp.AVTransport.Next()
        res = self.wait_until_track_rate_change(320)
        self.verifyTrueOrFailCase(

            zp.AVTransport.STOPPED_STATE,

        self.verifyFalseOrFailCase(res, "Unexpect stop playing new track rate")
