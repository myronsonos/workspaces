    """
    Test autoplay functionality on player for linein content.
    Following tests are covered.
    1. Verify autoplay for Linein with following variations:
        a. Grouped zones linked or unlinked on autoplay.
        b. Autoplay enabled before or after line in content start
        c. Autoplay zone already playing music or idle when linein content
            starts.
        d. Autoplay on source or other then source zone player.
        e. Reboot of line in source, or autoplay zone player or both players.
            (SWPBL-63515)
    2. Verify when (for wired players) playing line-in on 3 or more target
        (line in content playback devices - aka 'remotes') we switch from wav to
        SBC codec.
    3. Verify when user hits pause, Linein playback stops. (SWPBL-65285)
    """

    AUTOPLAY_LINKED_CONFIGS = [LINKED_ON_AUTOPLAY, UNLINK_ON_AUTOPLAY]
    PLAYBACK_STATES = [IDLE, PLAYING]
    AUTOPLAY_STATES = [AUTOPLAY_ENABLED_AFTER_CONTENT_START,
                       AUTOPLAY_ENABLED_BEFORE_CONTENT_START]

        """
        Generate line in capable player, autoplay linked config, initial
        playback state of autoplay zone player and when autoplay is enabled
        (before linein source playback or after.
        """
        for zp in self.line_in_devices:
            for autoplay_linked_config in self.AUTOPLAY_LINKED_CONFIGS:
                for playback_state in self.PLAYBACK_STATES:
                    for when_autoplay_enabled in self.AUTOPLAY_STATES:
                        yield (zp,
                               autoplay_linked_config,
                               playback_state,
                               when_autoplay_enabled)

    def test_linein_autoplay_device_reboot(self,
                                           source_zp,
                                           is_autoplay_zp_source,
                                           rebooting_device):
        """
        Test linein autoplay resume after reboot.
        following variations:
        1. Is autoplay zp same as the linein source or different.
        2. Which player is rebooting: linein or autoplay player or both

        :param source_zp: Line in source zone player
        :type source_zp: :class:`~sonos.client.SonosZoneComponent`
        :param bool is_autoplay_zp_source: Is target autoplay player local or
            remote
        :param str rebooting_device: Rebooting device source or remote or both
        """
        self.logger.info("Line in device : {}".format(source_zp))
        if is_autoplay_zp_source:
            autoplay_zp = source_zp
        else:
            self.skip("Skipping test pending resolution of SWPBL-63515")
            autoplay_zp = [zp for zp in self.test_devices
                           if zp.is_playback_device() and
                           zp not in self.line_in_devices][0]
        self.logger.info("Playback device : {}".format(autoplay_zp))
        source_zp.DeviceProperties.SetAutoplayRoomUUID(
            autoplay_zp.raw_udn, source=AutoPlaySource.LINEIN.value)
        self.logger.info("Starting mplayer playback to line in source")
        with MplayerMp3LineOutPlayback(LINE_IN_TRACK) as mplayer_h:
            LineInStartHelper(
                mplayer_h, source_zp, autoplay_zp,
                logger=self.logger).verify_linein_start()
            self.verify_autoplay(source_zp=source_zp,
                                 autoplay_zp=autoplay_zp,
                                 source=AutoPlaySource.LINEIN)
            # reboot line_in_device
            self.logger.info("Now rebooting {}".format(rebooting_device))
            if rebooting_device == LINEIN_SOURCE:
                source_zp.reboot()
                # restore udp logger and diags after reboot
                self.restore_udplogger_diagcs([source_zp])
            elif rebooting_device == PLAYBACK_PLAYER:
                autoplay_zp.reboot()
                # restore udp logger and diags after reboot
                self.restore_udplogger_diagcs([autoplay_zp])
            elif rebooting_device == LINEIN_AND_PLAYBACK_PLAYER:
                source_zp.reboot()
                if not source_zp == autoplay_zp:
                    autoplay_zp.reboot()
                # restore udp logger and diags after reboot
                self.restore_udplogger_diagcs([source_zp, autoplay_zp])
            self.logger.info("{}: Now checking playback state"
                             .format(autoplay_zp))
            self.verify_autoplay(source_zp=source_zp,
                                 autoplay_zp=autoplay_zp,
                                 source=AutoPlaySource.LINEIN)



[docs]    def test_linein_autoplay_codec(self):
        """
        Verify codec type played on autoplay zones. For less than 3 remotes play
        wav. For 3 or more remotes switch to sbc. Also If remotes reduced. After
        pause and  resume, we switch to wav.
        """
        linein_zp = self.line_in_devices[0]
        self.logger.info("Line in device : {}".format(linein_zp))
        gms = [zp for zp in self.test_devices
               if zp.playback_device and
               zp not in self.line_in_devices]
        self.logger.info("GMs: {}".format(gms))
        # Skip test if there are not sufficient number of players for GMs since
        # we will not play SBC codec in this case
        self.verifyTrueOrSkip(len(gms) >= 3, "We need at least 3 GMs for test")
        linein_zp.DeviceProperties.SetAutoplayRoomUUID(
            linein_zp.raw_udn, source=AutoPlaySource.LINEIN.value)
        linein_zp.DeviceProperties.SetAutoplayLinkedZones(True, source='RCA')
        with MplayerMp3LineOutPlayback(LINE_IN_TRACK) as mplayer_h:
            LineInStartHelper(
                mplayer_h, linein_zp, linein_zp,
                logger=self.logger).verify_linein_start()
            self.verify_autoplay(source_zp=linein_zp,
                                 autoplay_zp=linein_zp,
                                 source=AutoPlaySource.LINEIN)
            self.verifyTrueOrFailCase(
                self.verify_devices_playing([linein_zp], codec=WAV),
                "Expect wav codec playback on linein zp")
            # start adding zones to group
            joined_zps = []
            for gm in gms:
                self.logger.info("Adding {} to group".format(gm))
                gm.AVTransport.join_group(linein_zp)
                joined_zps.append(gm)
                if len(joined_zps) < 3:
                    self.verifyTrueOrFailCase(
                        self.verify_devices_playing(joined_zps, codec=WAV),
                        "Expect for < 3 remotes, play wav codec")
                else:
                    self.verifyTrueOrFailCase(
                        self.verify_devices_playing(joined_zps+[linein_zp],
                                                    codec=SBC),
                        "Expect for >= 3 remotes, play SBC codec.")
                    break
            # remove all except 2 of the joined zps from group
            self.logger.info("Removing all except 2 gms from group")
            for zp in joined_zps[2:]:
                self.logger.info("Removing {} from group".format(zp))
                zp.AVTransport.leave_group()
            for zp in joined_zps[0:2]:
                self.playing_wav[zp] = False
            self.playing_wav[linein_zp] = False
            # Pause playback and restart. Expect codec to change to wav
            self.logger.info("Performing stop start of playback to cause "
                             "playback codec change")
            linein_zp.AVTransport.Pause_and_wait()
            linein_zp.AVTransport.Play_and_wait()
            self.verifyTrueOrFailCase(
                self.verify_devices_playing(joined_zps[0:2]+[linein_zp],
                                            codec=WAV),
                "Expect wav codec being played when number of remotes < 3")

    def test_linein_autostop(self, linein_zp, autoplay_location):
        """
        Verify when content stop, autostop kicks in 13 seconds
        :param linein_zp: Line in source
        :type linein_zp: :class:`~sonos.client.SonosZoneComponent`
        :param str autoplay_location: Autoplay location (on line in source
            itself or on other player or remote)
        """
        self.logger.info("Line in device : {}".format(linein_zp))
        # if device playing line in content is not source
        if autoplay_location == REMOTE:
            autoplay_zp = [zp for zp in self.test_devices
                           if zp.is_playback_device() and
                           zp not in self.line_in_devices][0]
        elif autoplay_location == LOCAL:
            autoplay_zp = linein_zp
        linein_zp.DeviceProperties.SetAutoplayRoomUUID(
            autoplay_zp.raw_udn, source=AutoPlaySource.LINEIN.value)
        with MplayerMp3LineOutPlayback(LINE_IN_TRACK) as mplayer_h:
            LineInStartHelper(
                mplayer_h, linein_zp, autoplay_zp,
                logger=self.logger).verify_linein_start()
            self.verify_autoplay(source_zp=linein_zp,
                                 autoplay_zp=autoplay_zp,
                                 source=AutoPlaySource.LINEIN)
        # before 13s silence verify autoplay_zp is still in playing state.
        self.verifyTrueOrFailCase(
            autoplay_zp.AVTransport.is_playing(),
            "{}: Before 13s silence time has elapsed, expect playing state"
            .format(autoplay_zp))
        # linein source has stopped.
        # after autostop time of 13 seconds, expect playback state on
        # autoplay_zp to be in paused state.
        success =  autoplay_zp.AVTransport.wait_for_avt_state(
                                        autoplay_zp.AVTransport.PAUSED_STATE,
                                        timeout_seconds=AUTOSTOP_TIME + 10)
        if success:
            self.logger.log(AssertionLevel.PASS,
                            "{}: In paused state as expected"
                            .format(autoplay_zp))
        else:
            self.fail(
                "{}: Not in paused state. Current state: {}. DIAG: {}"
                .format(autoplay_zp,
                        autoplay_zp.AVTransport.get_transport_state(),
                        autoplay_zp.ZoneGroupTopology.SubmitDiagnostics()))
        # allow sometime in paused state.
        time.sleep(5)

    def test_linein_user_stop(self, linein_zp, autoplay_location, stop_method):
        """
        Verify with autoplay on,
        1. When user hits play/pause via controller or
        button, playback stops until user triggers play via controller or
        buttons
        2. Playback does not resume until the line in silence time meets the
        auto stop interval and the line in source begins playing anew.
        :param linein_zp: Line in source
        :type linein_zp: :class:`~sonos.client.SonosZoneComponent`
        :param str autoplay_location: Autoplay location (on line in source
            itself or on other player or remote)
        :param str stop_method: How does user stop playback, via controller or
            play/pause button
        """
        self.logger.info("Line in device : {}".format(linein_zp))
        # if device playing line in content is not source
        if autoplay_location == REMOTE:
            autoplay_zp = [zp for zp in self.test_devices
                           if zp.is_playback_device() and
                           zp not in self.line_in_devices][0]
        elif autoplay_location == LOCAL:
            autoplay_zp = linein_zp
        linein_zp.DeviceProperties.SetAutoplayRoomUUID(
            autoplay_zp.raw_udn, source=AutoPlaySource.LINEIN.value)
        with MplayerMp3LineOutPlayback(LINE_IN_TRACK) as mplayer_h:
            LineInStartHelper(mplayer_h, linein_zp, autoplay_zp,
                              logger=self.logger).verify_linein_start()
            self.verify_autoplay(source_zp=linein_zp,
                                 autoplay_zp=autoplay_zp,
                                 source=AutoPlaySource.LINEIN)
            self.logger.info("Current Volume on autoplay zp: {}: {}".format(
                    autoplay_zp, autoplay_zp.RenderingControl.GetVolume()))
            if stop_method == CONTROLLER:
                # stop via upnp
                autoplay_zp.AVTransport.Pause_and_wait()
            elif stop_method == BUTTON:
                # stop via pressing pause/mute button
                autoplay_zp.cli.volume_mute()
            # allow driver/controller action to complete before checking state
            if not autoplay_zp.AVTransport.wait_for_avt_state(
                   autoplay_zp.AVTransport.PAUSED_STATE,
                   timeout_seconds=10):
                diag_id = autoplay_zp.diag.submit_direct_diag()
                if diag_id is not None:
                    self.stop("Autoplay device:{} not in paused state. DIAG "
                              "ID: {}".format(autoplay_zp, diag_id))
            # expect player to remain in stop stopped state
            self.verifyFalseOrFailCase(
                autoplay_zp.AVTransport.wait_for_avt_state(
                    autoplay_zp.AVTransport.PLAYING_STATE,
                    timeout_seconds=LINEIN_SILENCE_TIME+3),
                "{}: Expect not to be in playing state".format(autoplay_zp))
