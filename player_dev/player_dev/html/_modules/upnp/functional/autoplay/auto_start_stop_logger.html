[docs]    def get_files_from_nas(self, files_path, files_list=None):
        """
        Copy files from NAS
        :param str files_path: Path to download files from
        :param list files_list: The list of files to download
        """
        self.logger.info("Getting files from {} on NAS".format(files_path))
        with SonosAutomationSFTP() as sftp:
            sftp.chdir(files_path)
            if not files_list:
                files_list = sftp.listdir()
                self.logger.info(
                    "Available files on nas : {}".format(files_list))
            try:
                for a_file in files_list:
                    self.logger.info("Copying file: {}".format(a_file))
                    src = files_path + a_file
                    dst = self.resource_path + a_file
                    self.logger.debug("Source: {}".format(src))
                    self.logger.debug("Destination: {}".format(dst))
                    sftp.get(src, dst)
            except IOError:
                self.stop_suite("Required files not available")



[docs]    def play_track(self, file, playback_duration, encoding=None):
        """
        play the given track on the mplayer and pause after duration
        specified in playback arg
        :param str file: the path to the file to play
        :param int playback_duration: the amount of time to play the given
        track
        :param str encoding: to force an encoding format
        """
        args = [file]
        self.logger.info(
            "Playing back {} via MplayerAc3Playback".format(
                self.playback_file))

        if encoding:
            # if encoding has been specified, pass it to the
            # MplayerAc3Playback call. otherwise don't include a None
            # argument in call so as to allow default encoding.
            self.logger.info("using encoding {}".format(encoding))
            args.append(encoding)

        # begin playback
        with MplayerAc3Playback(*args) as m_player:
            start_time = time.time()

            while True:
                self.logger.info(
                    "Iterating through track, timestamp: {}".format(
                        m_player.player.time_pos))

                # sleep for 1 second so as to reduce the logging noise to 1s
                #  intervals
                guaranteed_sleep(1)
                elapsed = time.time() - start_time

                # pause if we've played the playback duration
                if elapsed >= playback_duration:

                    # sleep for the configured number of seconds
                    m_player.player.pause()
                    self.logger.info("going to sleep for pause")
                    return

[docs]    def set_up_routine(self):
        """
        set up routine for test
        """
        for zp in self.ht_ms:
            # cleanup bonded config
            zp.DeviceProperties.cleanup_bonded_configuration()

            # update the auto stop interval
            zp.cli.set_ht_autostop_threshold(self.auto_stop_int)

            # set diag levels:
            self.set_logging(zp)

            # reset udp logs
            zp.diag.start_udp_log(reset=True)

            zp.wait_for_upnp_alive()

            # setup auto play
            zp.DeviceProperties.SetAutoplayRoomUUID(zp.raw_udn)
            zp.DeviceProperties.SetAutoplayLinkedZones(True)

            # drop volume very low to prevent noises
            zp.RenderingControl.SetVolume(1)

            # set up callbacks
            self.set_up_udp_log_cbs(zp)
        self.reset_callback_maps()

        # generate HT satellites & do a cleanup
        for left, right in self.sat_pairs:
            [zp.DeviceProperties.cleanup_bonded_configuration() for zp in [
                left, right]]
        for sub in self.subs:
            sub.DeviceProperties.cleanup_bonded_configuration()

[docs]    def interval_is_expected(self, line):
        """
        check if the numeric value for the number of seconds of silence
        detected is 1 + the specified number of auto stop interval seconds
        :param line: the line detected by regex
        :type line: :obj:`str`
        :return: True/False
        :rtype: :obj:`bool`
        """
        regex = r"[0-9]* seconds of silence"
        matches = re.findall(regex, line)
        if len(matches) < 1:
            return False
        line = matches[0]
        if len(line) > 0:
            n_secs = line.split(' ')[0]
            if n_secs.isdigit():
                if int(n_secs) != self.auto_stop_int + 1:
                    return False
            else:
                self.logger.error(
                    "Line does not denote "
                    "number for n_secs: {} from {}".format(n_secs, line))
        return True

[docs]    def bond_all_ht_masters(self, ht_masters):
        """
        configure all HT masters to be in HT bonded configurations,
        be it either 3.1 or 5.0 configs.
        :param list ht_masters: a list of home theater masters to bond
        satellites to
        """
        # initialize counters
        n_sat_pairs_available = len(self.sat_pairs)
        n_subs_available = len(self.subs)

        # create the configs
        for zp in ht_masters:
            # make sure there are enough satellite pairs available first
            if n_sat_pairs_available > 0:
                left, right = self.sat_pairs[n_sat_pairs_available - 1]
                self.logger.info("Creating 5.0 config with {}, {}, {}".format(
                    zp, left, right
                ))
                zp.DeviceProperties.create_50_config(left, right)
                n_sat_pairs_available -= 1

            # if not enough satellite pairs available, use a sub satellite
            elif n_subs_available > 0:
                sub = self.subs[n_subs_available - 1]
                self.logger.info("Creating 3.1 config with {}")
                zp.DeviceProperties.create_31_config(sub)
                n_subs_available -= 1
            # we should never hit this
            else:
                self.stop("Somehow ran out of satellite pairs / subs")

    def test_autoplay_autostop_playback_sequentially(self, ht_master):
        """
        :param ht_master: the home theater master to test with
        :type ht_master: :obj`sonos.client.internal.SonosZoneComponent`
        test the SWPBL-91611 test case
        using the test method described in comments:

        - Set autostop threshold to 20 seconds by setting
        HTASilenceThresholdAutoStopSec: [20] in localsettings.txt on an HT
        capable device
        - Playback SPDIF on HT device to engage autoplay
        - Wait X seconds for autostop to engage (play silence to mitigate
        risk of sound card noise tickling noise detector).
        - Add sub to HT master to create 3.1 config (or any satellite config
        should work)
        - Wait at least another X seconds (where bug would normally kick in)
        - Playback SPDIF on HT device to engage autoplay

        There are 2 messages that indicate the bug:
        1. The re-beginning of of silence detection upon addition of a HT
        satellite to the already stopped HT master:
        <htap_autostop,2> beginning silence detection for auto stop @ 363256.5
        2. The arbitrarily numbered seconds of silence detected at autoplay:
        <htap_autostop,1> 41 seconds of silence detected. Engaging auto stop
        """
        self.ht_ms = [ht_master]
        self.set_up_routine()

        sn = ht_master.serialNumber

        """
        FIX FOR ENVIRONMENTAL NOISE DURING STARTUP OF ANACAPA
        if we've seen a beginning silence detection message prior to
        playback of the SPDIF source, it means that there was some noise
        on the line during the startup of the player, which caused
        autoplay to kick-in prematurely.
        In the event of this error, we skip playing content and checking
        for the autoplay message, as autoplay should have already occurred
        """
        if not self.beginning_silence_map[sn]:
            self.play_content()
            # make sure we got an auto play msg
            wait_until_true(
                lambda: self.start_autoplay_map[sn],
                reason="Waiting for autoplay to begin: {}\nAVT: {}\nToslink: {}"
                .format(
                    self.start_autoplay_map,
                    ht_master.AVTransport.get_avtransport_uri(),
                    ht_master.diag.get_toslink_stream_type()),
                timeout_seconds=AUTO_PLAY_WAIT_TIME
            )
        else:
            self.logger.info("NOTE: Auto play triggered prior to playback of "
                             "content!")

        # Begin silence
        # play silence for the autostop duration
        with MplayerAc3Playback(*[self.silence_file]) as m_player:
            self.logger.info("making sure we get a 'beginning silence' msg")
            # also make sure we got the "beginning silence detection message"
            wait_until_true(
                lambda: self.beginning_silence_map[sn],
                reason="Ensuring beginning silence is false:"
                       " {}".format(self.beginning_silence_map),
                timeout_seconds=BEGIN_SILENCE_WAIT_TIME + 10
            )

            """
            Due to instability during playback, we used to see
            failures here for one of two reasons:
                1. slight noise on the line caused the ZP under test to
                re-begin silence detection while we were waiting for
                auto-stop to kick-in. This would cause the auto-stop wait to
                timeout, failing the test. It happened about 30% of the time.
            Since there's nothing we can do to control noise that the
            soundcard may introduce on the line, we have a maximum
            number of attempts that we make to wait for auto-stop to kick-in
            without being interrupted by a new silence detection beginning.
                2. autostop would kick-in about 100-200ms later than we were
                expecting it to kick-in.
            Since we're not testing how long it takes for auto-stop to kick in,
            it seems reasonable to increase the maximum wait time by 5 seconds
            to account for this instability.
            """
            max_n_attempts = 5
            n_attempts = 0
            self.beginning_silence_map[sn] = False
            start_time = time.time()
            max_wait_time = self.auto_stop_int + AUTO_STOP_PROPAGATION_DELAY
            time_between_checks = 2

            while n_attempts < max_n_attempts and not self.autostop_map[sn]:
                guaranteed_sleep(time_between_checks)

                # if we detect silence again during this state, it means
                # that the silence detection was interrupted
                if self.beginning_silence_map[sn]:
                    self.beginning_silence_map[sn] = False
                    self.logger.info("restarting wait for autostop - "
                                     "saw another beginning silence detection")
                    start_time = time.time()
                    n_attempts += 1

                elapsed_time = time.time() - start_time
                if elapsed_time > max_wait_time:
                    self.fail("Timed out waiting for autostop to kick in.")

            if n_attempts == max_n_attempts:
                self.fail("Attempted to wait for autostop "
                          "to kick-in {} times. Failed last because of silence"
                          "detection kicking off multiple times."
                          "".format(max_n_attempts))

            # clear maps
            self.reset_callback_maps()

            # add satellite(s), and wait some time for message to show in log
            self.bond_all_ht_masters(self.ht_ms)
            wait_until_true(
                lambda: self.sat_added_map[sn],
                reason="Waiting for sat_added_map to occur: {}".format(
                    self.sat_added_map),
                timeout_seconds=ADD_SAT_WAIT_TIME
            )

            # wait a couple of seconds, then check to see if we hit a wrong
            # message from the satellite having been added
            guaranteed_sleep(BEGIN_SILENCE_WAIT_TIME)

            # make sure there's no "beginning silence detection" message upon
            # addition of satellite
            self.verifyTrueOrFailCase(
                not self.beginning_silence_map[sn],
                "Expected no beginning silence messages after adding satellite"
            )

            # wait another autostop thresh seconds
            guaranteed_sleep(self.auto_stop_int + AUTO_STOP_PROPAGATION_DELAY)

        # end silence

        # clear maps
        self.reset_callback_maps()

        # begin music
        # play content
        self.play_content()
        # end music

        # make sure we got an auto play msg
        wait_until_true(
            lambda: self.start_autoplay_map[sn],
            reason="Waiting for autoplay to begin: {}".format(
                self.start_autoplay_map),
            timeout_seconds=AUTO_PLAY_WAIT_TIME
        )

        # make sure there are no "seconds of silence detected" messages at
        # all immediately after the playback has finished
        guaranteed_sleep(BEGIN_SILENCE_WAIT_TIME)
        self.verifyTrueOrFailCase(
            not self.autostop_map[sn],
            "Expected auto stop not to kick in yet"
        )
