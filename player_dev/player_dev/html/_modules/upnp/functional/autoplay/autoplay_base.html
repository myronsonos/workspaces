[docs]    def setup_udp_logger(self, zp):
        """
        Setup udp logger and pass in callbacks for strings we are looking for
        :param zp: sonos zone player for which to setup udp logger
        :type zp: :class:``~sonos.client.internal.SonosZoneComponent`
        """
        self.logger.info("Setting up udp logger on : {}".format(zp.modelNumber))
        try:
            zp.diag.start_udp_log()
        except gaierror as e:
            self.logger.critical("UDPLogger port error: {}".format(e))
            self.stop_suite()
        zp.diag.watch_udp_log_for_string(
            ENTER_IDLE_STATE, lambda msg: self.cb_enter_idle_state(msg, zp))
        zp.diag.watch_udp_log_for_string(
            HT_AUTOSTOP, lambda msg: self.cb_autostop(msg, zp))
        zp.diag.watch_udp_log_for_string(
            STARTING_AUDIO_DECODER,
            lambda msg: self.cb_detected_playing(msg, zp))
        if zp.is_ht_master_capable():
            zp.diag.watch_udp_log_for_string(
                PLAYING_CHSNK, lambda msg: self.cb_detected_playing(msg, zp))
            zp.diag.watch_udp_log_for_string(
                PLAYING_SPDIF, lambda msg: self.cb_detected_playing(msg, zp))

[docs]    def get_tracks(self, tracks=None, path=None):
        """
        Copy tracks from local disc for web hosting
        :param list tracks: List of tracks to copy, default CPU tracks
        :param str path: Path to copy tracks from. Default CPU track location
        """
        self.logger.info("Getting {} from {}".format(tracks, path))
        if not os.path.exists(self.resource_path):
            os.mkdir(self.resource_path)
        with SonosAutomationSFTP() as sftp:
            sftp.chdir(path)
            self.logger.info("Available tracks on nas : {}".format(sftp.listdir()))
            try:
                for track in tracks:
                    self.logger.info("Copying test track: {}".format(track))
                    src = path + track
                    dst = self.resource_path + track
                    sftp.get(src, dst)
            except IOError:
                self.stop_suite("Required tracks not available")

[docs]    def cb_detected_playing(self, msg, zp):
        """
        Callback on detecting tv or linein (sbc) or music playback
        :param str msg: anacapa trace line containing detected string
        :param zp: Sonos zone component from which to submit
        :type zp: :class:`~sonos.client.SonosZoneComponent`
        """
        self.logger.info("Detected playing: {} on {}".format(msg, zp))
        if WAV in msg:
            self.playing_wav[zp] = True
        elif SBC in msg:
            self.playing_sbc[zp] = True
        elif MP3 in msg:
            self.playing_mp3[zp] = True
        # conditions below only occur for HT master devices
        if PLAYING_SPDIF in msg:
            self.playing_spdif = True
        elif PLAYING_CHSNK in msg:
            self.playing_chsnk = True

                f.write(trace)
                self.logger.info("Anacapa trace saved to: {}".format(trace_log))

[docs]    def verify_autoplay(self,
                        source_zp=None,
                        autoplay_zp=None,
                        gms=None,
                        source=None,
                        autoplay_linked_config=None):
        """
        Verify autoplay on autoplay zone player as well as group members
        depending on autoplay link config

        :param source_zp: linein or spdif source player
        :type source_zp: :class:`~sonos.client.SonosZoneComponent`
        :param autoplay_zp: Line in or spdif autoplay player
        :type autoplay_zp: :class:`~sonos.client.SonosZoneComponent`
        :param list gms: List of sonos zone component
        :param str source: Line in or spdif
        :param str autoplay_linked_config: AutoPlay Linked Zones config
        """
        self.logger.info("Verifying autoplay and linked state")
        self.logger.info("\nSource ZP: {}\nAutoplay ZP: {}\nSource: {}\nGMs: {}"
                         "\nAutoplayOnLinkedZones: {}"
                         .format(source_zp, autoplay_zp, source,
                                 gms, autoplay_linked_config))
        if source == AutoPlaySource.LINEIN:
            expected_uri = 'x-rincon-stream:{}'.format(source_zp.raw_udn)
        elif source == AutoPlaySource.SPDIF:
            expected_uri = ('x-sonos-htastream:{}:spdif'
                            .format(source_zp.raw_udn))
        # verify autoplay on autoplay zp.
        time.sleep(ALLOW_PLAYBACK_TIME)
        try:
            wait_until_true(
                lambda: (autoplay_zp.AVTransport.is_playing() and
                         expected_uri ==
                         autoplay_zp.AVTransport.get_avtransport_uri()),
                timeout_seconds=60)
            self.logger.info(
                "{}: Playing state: {}. Current URI: {}"
                .format(autoplay_zp,
                        autoplay_zp.AVTransport.is_playing(),
                        autoplay_zp.AVTransport.get_avtransport_uri()))
        except TimeoutError:
            self.logger.info("{}: Playing state: {}. Current URI: {}"
                             .format(autoplay_zp,
                                     autoplay_zp.AVTransport.is_playing(),
                                     autoplay_zp.AVTransport.
                                     get_avtransport_uri()))
            self.collect_anacapa_trace_from_devices([source_zp, autoplay_zp])
            self.fail("{}: Expect to be playing {} but device is "
                      "not in playing state".format(autoplay_zp, source))
        else:
            self.logger.log(AssertionLevel.PASS,
                            "{}: Playing expected source: {}"
                            .format(autoplay_zp, source))
            # verify autoplay on zps grouped to autoplay_zp if configured so
            # verify codec being played on autoplay zps.
            if gms:
                if autoplay_linked_config == UNLINK_ON_AUTOPLAY:
                    if self.are_all_zps_unlinked(autoplay_zp, gms):
                        self.logger.log(
                            AssertionLevel.PASS,
                            "GMs not playing Line in content.")
                    else:
                        self.fail("Expect no zps linked to autoplay zp: {}"
                                  .format(autoplay_zp))
                elif autoplay_linked_config == LINKED_ON_AUTOPLAY:
                    if source == AutoPlaySource.LINEIN:
                        # if number of remotes meaning non source player are
                        # 3 or more, we switch to SBC codec. Else we use WAV
                        # get number of gms to hit threshold to switch codec
                        if source_zp == autoplay_zp:
                            wav_threshold = 2
                        else:
                            wav_threshold = 1
                            # in this case, if number of wired gms are 3 of more
                            # we expect SBC codec to be used.
                        if len(gms) > wav_threshold:
                            codec = SBC
                        else:
                            codec = WAV
                    if source == AutoPlaySource.SPDIF:
                        codec = SBC
                    if self.are_all_zps_linked(autoplay_zp, gms):
                        self.logger.log(AssertionLevel.PASS,
                                        "All GMs linked to {}"
                                        .format(autoplay_zp))
                        if self.verify_devices_playing(gms, codec=codec):
                            self.logger.log(
                                AssertionLevel.PASS,
                                "All line devices playing {} content"
                                .format(source))
                        else:
                            self.fail("Some or all devices not playing {}"
                                      .format(source))
                    else:
                        self.fail("All zps not linked")



[docs]    def verify_devices_playing(self, devices, codec=None):
        """
        Verify devices are playing specified codec

        :param list devices: List of sonos zone component
        :param str codec: Codec
        :return: Did devices play specified codec ?
        :rtype: :obj:`bool`
        """
        self.logger.info("Verifying if devices playing {}".format(codec))
        self.logger.info("Devices: {}".format(devices))
        devices_playing = {}
        if codec == MP3:
            devices_playing = self.playing_mp3
        elif codec == SBC:
            devices_playing = self.playing_sbc
        elif codec == WAV:
            devices_playing = self.playing_wav
        try:
            wait_until_true(lambda: all([devices_playing[zp]
                                         for zp in devices]),
                            timeout_seconds=20)
        except TimeoutError:

            # print which devices are NOT playing for easier debugging
            self.logger.info("All devices not playing codec {}\n"
                             "devices_playing: {}\n"
                             "Devices playing SBC: {}\n"
                             "Devices playing WAV: {}\n"
                             "Devices playing MP3: "
                             "{}".format(codec, devices_playing,
                                         self.playing_sbc, self.playing_wav,
                                         self.playing_mp3))
            return False
        else:
            return True



[docs]    def get_num_zps_linked(self, gc, gms):
        """
        Get number of zps from gms list linked to gc

        :param gc: Sonos zone component that is group coordinator
        :type gc: :class:`~sonos.client.SonosZoneComponent`
        :param list gms: List of sonos zone components expected to be group
            members
        :return: Number of zones from gms linked to gc
        :rtype: :obj:`int`
        """
        num_zps_linked = 0
        for zp in gms:
            if gc.raw_udn in zp.AVTransport.get_avtransport_uri():
                num_zps_linked += 1
                self.logger.info("{}: in group with GC: {}".format(zp, gc))
            else:
                self.logger.info("{}: Not in group with GC: {}".format(zp, gc))
        if num_zps_linked == len(gms):
            self.logger.info("All GMs linked to GC")
        elif num_zps_linked == 0:
            self.logger.info("No ZPs linked to GC")
        return num_zps_linked

                when_autoplay_enabled,
                source_zp=None,
                autoplay_zp=None,
                mplayer=None,
                track=None,
                source=None,
                playback_state=False,
                gms=None,
                autoplay_linked_config=None,
                verify_ht_state=False):
        """
        Do autoplay test

        :param str when_autoplay_enabled: Autoplay enabled before or after
            content start
        :param source_zp: Sonos zone component that is source of linein/spdif
            content
        :type source_zp: :class:`~sonos.client.SonosZoneComponent`
        :param autoplay_zp: Sonos zone component that plays lines/spdif content
        :param autoplay_zp: Sonos zone component that is source of linein/spdif
            content
        :param mplayer: mplayer context manager to play linein/spdif content
        :type mplayer: :class:`~MplayerPlayback`
        :param str track: Track to play using mplayer
        :param str source: Source linein or spdif
        :param str playback_state: Initial playback state of autoplay zp
        :param list gms: List of sonos zone componets that are GMs
        :param str autoplay_linked_config: AutoplayLinkedZone config
        """
        self.logger.info("Performing autoplay test")
        if gms:
            self.logger.info("Group members: {}".format(gms))
            autoplay_zp.AVTransport.link_zones(gms)
        if autoplay_linked_config == LINKED_ON_AUTOPLAY:
            source_zp.DeviceProperties.SetAutoplayLinkedZones(
                True, source=source.value)
        elif autoplay_linked_config == UNLINK_ON_AUTOPLAY:
            source_zp.DeviceProperties.SetAutoplayLinkedZones(
                False, source=source.value)
        if playback_state == PLAYING:
            self.start_playback(autoplay_zp)
        if when_autoplay_enabled == AUTOPLAY_ENABLED_BEFORE_CONTENT_START:
            source_zp.DeviceProperties.SetAutoplayRoomUUID(
                autoplay_zp.raw_udn, source=source.value)
            self.verifyTrueOrStop(
                autoplay_zp.raw_udn == source_zp.DeviceProperties.GetAutoplayRoomUUID(source=source.value),
                "Expect autoplay to be set on {}".format(source_zp))
        if source == AutoPlaySource.LINEIN:
            self.logger.info("Starting mplayer playback to line in source")
        elif source == AutoPlaySource.SPDIF:
            self.logger.info("Starting mplayer ac3 playback to ht master")
        else:
            raise ValueError("Unexpected source {}".format(source))
        with mplayer(track) as mplayer_h:
            if when_autoplay_enabled == AUTOPLAY_ENABLED_AFTER_CONTENT_START:
                # play content for couple of seconds before setting autoplay
                # room
                time.sleep(2)
                source_zp.DeviceProperties.SetAutoplayRoomUUID(
                    autoplay_zp.raw_udn, source=source.value)
                self.verifyTrueOrStop(
                    autoplay_zp.raw_udn ==
                    source_zp.DeviceProperties.GetAutoplayRoomUUID(
                        source=source.value),
                    "Expect autoplay to be set on {}".format(source_zp))
            if source == AutoPlaySource.LINEIN:

[docs]    def verify_avts_not_changed(self, ht_master):
        """
        Verify no avt state change on ht master device ht_master to continue
        playing for atleast SPDIF_PLAYBACK_TIME
        :param ht_master: Sonos zone component - New gc of group
        :type ht_master: :class:`~sonos.client.SonosZoneComponent`
        :return: avts_unchanged
        """
        self.logger.info("Verifying avt state has not changed on {}"
                         .format(ht_master))
        avts = AVTransportStateTransitionTimer(ht_master.AVTransport)
        avts_unchanged = True
        avts.start(SPDIF_PLAYBACK_TIME + 2, SPDIF_PLAYBACK_TIME - 2,
                   user_defined_start_states=["PLAYING", "TRANSITIONING"])
        self.logger.info("Elapsed time without change in avt state: {}"
                         .format(avts.elapsed))
        if not avts.expiry:
            avts_unchanged = False
            self.fail('{} did not stay in "Playing" state for at least {} secs.'
                      .format(ht_master, SPDIF_PLAYBACK_TIME - 2))
        if avts.state_changed:
            avts_unchanged = False
            self.fail("{}'s TransportState changed unexpectedly expected "
                      "after {} secs.".format(ht_master, avts.elapsed))
        return avts_unchanged
