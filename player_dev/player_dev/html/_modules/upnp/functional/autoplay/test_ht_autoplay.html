    """
    Test autoplay functionality on player for SPDIF content.
    Following tests are covered.
    1. Verify autoplay for SPDIF with following variations:
        a. Grouped zones linked or unlinked on autoplay.
        b. Autoplay enabled before or after spdif content start
        c. Autoplay zone already playing music or idle when  spdif content
            starts.
        d. Reboot of ht master
    2. Verify delegation on autoplay (this is already covered in delegation
        stress test, but also tested here for completeness)
    3. Test autostop
    """

    AUTOPLAY_LINKED_CONFIGS = [LINKED_ON_AUTOPLAY, UNLINK_ON_AUTOPLAY]
    PLAYBACK_STATES = [IDLE, PLAYING]
    AUTOPLAY_STATES = [AUTOPLAY_ENABLED_AFTER_CONTENT_START,
                       AUTOPLAY_ENABLED_BEFORE_CONTENT_START]

        """
        Generate ht_master device, autoplay linked config, initial
        playback state of autoplay zone player and when autoplay is enabled
        (before linein source playback or after.
        """
        for ht_master in self.ht_masters:
            for autoplay_linked_config in self.AUTOPLAY_LINKED_CONFIGS:
                for playback_state in self.PLAYBACK_STATES:
                    for when_autoplay_enabled in self.AUTOPLAY_STATES:
                        yield (ht_master,
                               autoplay_linked_config,
                               playback_state,
                               when_autoplay_enabled)

    def _verify_delegation_ok(self, ht_master, gms):
        """
        Verify if delegation was successful

        :param ht_master: Sonos zone component - New gc of group
        :type ht_master: :class:`~sonos.client.SonosZoneComponent`
        :param gms: List of expected group members
        :type gms: :list:List of sonos zone component
        :return: was delegation successful?
        :rtype: :obj:`bool`
        """
        self.logger.info("{}: Verifying delegation ok.".format(ht_master))
        expected_uri = 'x-sonos-htastream:{}:spdif'.format(ht_master.raw_udn)
        if not ht_master.AVTransport.wait_for_avtransport_uri(expected_uri,
                                                              timeout=20):
            self.logger.info("Delegation NOT OK. "
                             "Unexpected AVT URI on HT master ")
            return False
        self.logger.info("AVT URI on HT Master OK.")
        if not self.are_all_zps_linked(ht_master, gms):
            self.logger.info("Delegation NOT OK. All devices not linked")
            return False
        else:
            self.logger.info("Delegation OK")
            if self._verify_ht_master_playing(spdif=True):
                self.logger.info("HT master playing SPDIF OK")
                if self.verify_devices_playing(gms, codec=SBC):
                    self.logger.info("All gms playing SBC")
                else:
                    self.logger.info("Not All gms playing SBC")
            else:
                self.logger.info("HT master not playing SPDIF")
            return True

    def _verify_ht_master_playing(self, spdif=False, chsnk=False):
        """
        Verify if ht master device is playing spdif or chsnk

        :param bool spdif:
        :param bool chsnk:
        :return: Is ht_master playing the right source
        :rtype: :obj:`bool`
        """
        try:
            if spdif:
                self.logger.info("Verifying if ht master using spdif")
                wait_until_true(lambda: self.playing_spdif)
            elif chsnk:
                self.logger.info("Verifying if ht master is using chsnk")
                wait_until_true(lambda: self.playing_chsnk)
        except TimeoutError:
            return False
        else:
            return True

    def test_delegation_to_ht_master(self,
                                     ht_master,
                                     when_autoplay_enabled,
                                     playback_state):
        """
        Test delagation to ht_master when ht_master is a group member and
        has TV content to play. The original GC should delegate to ht_master and
        becomes GM of the new group whose GC is ht_master playing TV

        :param ht_master: ht master capable sonos zone component
        :type ht_master: :class:`~sonos.client.SonosZoneComponent`
        :param str when_autoplay_enabled: When autoplay on ht_master is enabled
            before or after spdif content is available
        :param str playback_state: HT master state when spdif content starts,
            playing music or idle.
        """
        for zp in [zp for zp in self.test_devices if zp.playback_device]:
            self.logger.info(
                "{}: {}".format(zp, zp.AVTransport.get_avtransport_uri()))
        if when_autoplay_enabled == AUTOPLAY_ENABLED_BEFORE_CONTENT_START:
            self.logger.info("Enabling Autoplay on {}".format(ht_master))
            ht_master.DeviceProperties.SetAutoplayRoomUUID(ht_master.raw_udn)
        ht_master.DeviceProperties.SetAutoplayLinkedZones(True)
        zps = [zp for zp in self.test_devices if not zp == ht_master and zp.playback_device]
        gc = zps[0]
        gms = [zp for zp in self.test_devices if not zp == gc]
        gc.AVTransport.link_zones(gms)
        if self.are_all_zps_linked(gc, gms):
            if playback_state == PLAYING:
                self.start_playback(gc)
                if self.verify_devices_playing(gms, codec=MP3):
                    self.logger.info("Devices playing MP3")
                else:
                    self.stop("GMS: {} not playing expected content")
        else:
            self.stop("GMs: {} not linked to GC: {}".format(gms, gc))
        self.logger.info("Starting playback on HDMI/SPDIF")
        with MplayerAc3Playback(TV_TRACK) as mplayer_tv_h:
            # verify we do not delegate yet
            if not self.are_all_zps_linked(gc, gms):
                self.fail("All {} not linked to {}".format(gms, gc))
            post_delegation_gc = ht_master
            post_delegation_gms = [zp for zp in self.test_devices
                                   if not zp == ht_master and zp.playback_device]
            if when_autoplay_enabled == AUTOPLAY_ENABLED_AFTER_CONTENT_START:
                self.logger.info("Enabling Autoplay on {}".format(ht_master))
                ht_master.DeviceProperties.SetAutoplayRoomUUID(
                        ht_master.raw_udn)
            TosLinkStreamHelper(mplayer_tv_h, logger=self.logger).verify_tos_link_stream_type(ht_master,
                                                                                              STREAM_TYPE,
                                                                                              10)
            # verify delegation
            self.verifyTrueOrFailCase(
                self._verify_delegation_ok(post_delegation_gc,
                                           post_delegation_gms),
                "Expect delegation to HT master: {}".format(ht_master))

            stream_type = ht_master.diag.get_toslink_stream_type()
            self.verifyTrueOrFailCase(stream_type == STREAM_TYPE,
                                      "{} Expect {} stream type but found {}"
                                      .format(ht_master, STREAM_TYPE,
                                              stream_type))
            self.verify_avts_not_changed(ht_master)

    def test_linein_autoplay_with_ht_master(self,
                                            ht_master,
                                            linein_zp,
                                            htmaster_playing_tv):
        """
        Verify HT master plays line in in following cases:
        1. HT master is playing TV, line in content starts (htmaster is set
        as linein autoplay room).
        2. HT master is playing linein, TV content starts (htmaster
        autoplay tv is set), ht master continues to play line in
        :param ht_master: Ht master capable device
        :type ht_master: :class:`~sonos.client.SonosZoneComponent`
        :param linein_zp: Sonos zone component
        :type linein_zp: :class:`~sonos.client.SonosZoneComponent`
        :param bool htmaster_playing_tv: Is ht master already playing tv
        """
        self.logger.info("HT Master: {} Linein Source: {}"
                         .format(ht_master, linein_zp))
        if htmaster_playing_tv:
            self.logger.info(
                "Testing case where ht master currently playing TV then "
                "line in content starts")
            self.logger.info("Starting mplayer playback: SPDIF")
            ht_master.DeviceProperties.SetAutoplayRoomUUID(ht_master.raw_udn)
            with MplayerAc3Playback(TV_TRACK) as mplayer_tv_h:
                TosLinkStreamHelper(mplayer_tv_h, logger=self.logger).verify_tos_link_stream_type(ht_master,
                                                                                                 STREAM_TYPE,
                                                                                                 10)
                self.verifyTrueOrStop(
                    self._verify_ht_master_playing(spdif=True),
                    "{}: Expect to play TV".format(ht_master))
                self.verifyTrueOrStop(
                    self.verify_avts_not_changed(ht_master),
                    "{}: Expect no avt state change"
                    .format(ht_master))
                self.logger.info("Starting mplayer playback: LineIN")
                linein_zp.DeviceProperties.SetAutoplayRoomUUID(
                    ht_master.raw_udn, source='RCA')
                with MplayerMp3LineOutPlayback(LINE_IN_TRACK) as mplayer_linein_h:
                    try:
                        LineInStartHelper(mplayer_linein_h, linein_zp,
                                          ht_master).verify_linein_start()
                    except MPlayerReloadTrackException:
                        if ht_master.is_hideout():
                            pass
                        else:
                            self.stop("Line in playback failed on {}."
                                      .format(ht_master))
                    if ht_master.is_hideout():
                        expected_uri = ('x-sonos-htastream:{}:spdif'
                                        .format(ht_master.raw_udn))
                        current_uri = ht_master.AVTransport.get_avtransport_uri()
                        self.verifyTrueOrFailCase(
                            expected_uri == current_uri,
                            "Expect hideout to continue playing TV")
                    else:
                        self.verify_autoplay(source_zp=linein_zp,
                                             autoplay_zp=ht_master,
                                             source=AutoPlaySource.LINEIN)

        else:
            self.logger.info(
                "Testing case where ht master currently playing linein then "
                "TV content starts")
            self.logger.info("Starting mplayer playback: LineIn")
            linein_zp.DeviceProperties.SetAutoplayRoomUUID(ht_master.raw_udn,
                                                           source='RCA')
            with MplayerMp3LineOutPlayback(LINE_IN_TRACK) as mplayer_linein_h:
                LineInStartHelper(mplayer_linein_h, linein_zp, ht_master).verify_linein_start()
                self.verify_autoplay(source_zp=linein_zp,
                                     autoplay_zp=ht_master,
                                     source=AutoPlaySource.LINEIN)
                self.logger.info("Starting mplayer playback: SPDIF")
                ht_master.DeviceProperties.SetAutoplayRoomUUID(

                with MplayerAc3Playback(TV_TRACK) as mplayer_tv_h:
                    TosLinkStreamHelper(mplayer_tv_h, logger=self.logger).verify_tos_link_stream_type(ht_master,
                                                                                                     STREAM_TYPE,
                                                                                                     10)
                    self.verifyTrueOrStop(
                        self._verify_ht_master_playing(spdif=True),
                        "{}: Expect to play TV".format(ht_master))
                    stream_type = ht_master.diag.get_toslink_stream_type()
                    self.verifyTrueOrFailCase(
                        stream_type == STREAM_TYPE,
                        "{}: Expected Stream type: {}. Actual stream type: {}"
                        .format(ht_master, STREAM_TYPE, stream_type))
                    self.verifyTrueOrFailCase(
                        self.verify_avts_not_changed(ht_master),
                        "{}: Expect no avt state change".format(ht_master))

    def test_ht_master_autostop(self, ht_master):
        """
        Verify we engage autostop after autostop time on htmaster devices
        after there is silence for HTA_HTA_SILENCE_THRESHOLD_AUTOSTOP_SEC secs.
        :param ht_master: Ht master capable device
        :type ht_master: :class:`~sonos.client.SonosZoneComponent`
        """
        self.logger.info("{}: HT master device.".format(ht_master))
        ht_master.cli.set_ht_autostop_threshold(threshold=HT_AUTOSTOP_THRESHOLD)
        ht_master.wait_for_upnp_alive()
        self.restore_udplogger_diagcs([ht_master])
        ht_master.DeviceProperties.SetAutoplayRoomUUID(ht_master.raw_udn)
        with MplayerAc3Playback(TV_TRACK) as mplayer_tv_h:
            TosLinkStreamHelper(mplayer_tv_h, logger=self.logger).verify_tos_link_stream_type(ht_master,
                                                                                             STREAM_TYPE,
                                                                                             10)
            self.verifyTrueOrStop(
                self._verify_ht_master_playing(spdif=True),
                "{}: Expect to play TV".format(ht_master))
        try:
            wait_until_true(lambda: self.autostopped, timeout_seconds=(
                                        HT_AUTOSTOP_THRESHOLD +
                                        HT_AUTOSTOP_STEADY_STATE_BUFFER_TIME +
                                        HT_AUTOSTOP_SILENCE_ACCUMULATION_TIME))
        except TimeoutError:
            self.fail("{}: Autostop failed.".format(ht_master))
        else:
            self.logger.log(AssertionLevel.PASS,
                            "{}: Autostop OK".format(ht_master))
            self.verifyTrueOrFailCase(
                ht_master.AVTransport.wait_for_avtransport_uri(""),
                "{}: Expect AVTtransport URI to be empty".format(ht_master))
        ht_master.cli.set_ht_autostop_threshold()
        ht_master.wait_for_upnp_alive()
        self.restore_udplogger_diagcs([ht_master])
