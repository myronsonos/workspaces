[docs]class RestartAnacapa(TrueplayBaseFixture):
    """
    Tests restarting anacapa after Trueplay tuning
    """
    TEST_TYPE = ["TRUEPLAY_WEEKLY"]

    def _test_restart_anacapa(self, zp):
        """
        Tests restarting anacapa after Trueplay tuning
        :param :class:`sonos.client.internal.SonosZoneComponent` zp:
        """
        if zp.diag.has_testpoint_support():
            zp.RenderingControl.save_rendering_control_settings()
        else:
            sleep(35)
        self.bounce_anacapa(zp)
        self.verify_trueplay_tuning(
            zp, True, True)
        # Sleep an additional 6 seconds to let the topology monitor finish
        #   up prior to requesting the bond to be revoked.
        sleep(10)
