# UDN_year-month-day_hour-minute-second (all zero padded)
COEFFICIENT_ID_FORMATTER = "{}_{}{}%Y-%m-%d_%H-%M-%S"



[docs]def get_coefficient_id(zp, version):
    """
    Returns a formatted coefficient id for a given ZP and version
    :param zp:
    :param version:
    :return: Coefficient id string
    """
    zp_uuid = zp.raw_udn.replace(SONOS_ZONE_UDN_PREFIX, '')
    coefficient_id_formatter = COEFFICIENT_ID_FORMATTER.format(
        zp_uuid,
        version,
        '' if not version else '_')
    return datetime.now().strftime(coefficient_id_formatter)

    def test_old_and_new_calibration_rotation_sats_sub(
            self, ht_device, left_rear, right_rear):
        """
        Verifies that a HT system with a valid calibration in one
        orientation and an invalid calibration in another, does not
        try to load the invalid calibration when set to that orientation.
        Also tests an uncalibrated orientation.
        :param ht_device:
        :param left_rear:
        :param right_rear:
        """
        self.device_list = [ht_device, left_rear, right_rear, self.sub]
        bonded_members = [left_rear, right_rear, self.sub]
        # Invalid id and orientation
        invalid_coefficient_id = get_coefficient_id(
            ht_device, INVALID_VERSIONS[0])
        invalid_orientation = ht_device.supported_orientations[0]
        # Valid id and orientation
        valid_coefficient_id = get_coefficient_id(ht_device, VALID_VERSIONS[0])
        valid_orientation = ht_device.supported_orientations[1]
        # No calibration orientation
        uncalibrated_orientation = ht_device.supported_orientations[2]
        # Set an old calibration
        self.set_orientation(ht_device, invalid_orientation)
        self.tune_ht_with_surrounds_and_sub(
            ht_device, left_rear, right_rear,
            coefficient_id=invalid_coefficient_id,
            verify_calibration=False,
            verify_dsp=False)
        # Set a new calibration
        self.set_orientation(ht_device, valid_orientation)
        self.tune_ht_with_surrounds_and_sub(
            ht_device, left_rear, right_rear,
            coefficient_id=valid_coefficient_id,
            do_room_bonding=False)
        # Verify old cal/orientation is invalid
        self.set_orientation(ht_device, invalid_orientation)
        self.verify_trueplay_invalid(ht_device, bonded_members=bonded_members)
        # Verify uncalibrated orientation is invalid
        self.set_orientation(ht_device, uncalibrated_orientation)
        self.verify_trueplay_invalid(ht_device, bonded_members=bonded_members)
        # Step through some sat orientations
        for _ in self._generate_orientation_matrix((left_rear, right_rear)):
            self.verify_trueplay_invalid(ht_device,
                                         bonded_members=bonded_members)
        # Verify new cal/orientation is valid
        self.set_orientation(ht_device, valid_orientation)
        self.verify_trueplay_on(ht_device, bonded_members=bonded_members)
        # Step through some sat orientations
        for _ in self._generate_orientation_matrix((left_rear, right_rear)):
            self.verify_trueplay_on(ht_device, bonded_members=bonded_members)
        # Verify old cal/orientation is invalid
        self.set_orientation(ht_device, invalid_orientation)
        self.verify_trueplay_invalid(ht_device, bonded_members=bonded_members)
        # Step through some sat orientations
        for _ in self._generate_orientation_matrix((left_rear, right_rear)):
            self.verify_trueplay_invalid(ht_device,
                                         bonded_members=bonded_members)
        # Verify new cal/orientation is valid
        self.set_orientation(ht_device, valid_orientation)
        self.verify_trueplay_on(ht_device, bonded_members=bonded_members)

    def _test_calibration_boot(self, ht_device, left_rear, right_rear,
                               version_list, versions_are_valid):
        """
        Verifies a set of calibrations are loaded or ignored on boot
        depending on the given version list.
        :param ht_device:
        :param left_rear:
        :param right_rear:
        :param version_list: List of versions to be tested
        :param versions_are_valid: True if the versions are new. False if
        the versions are old
        """
        if left_rear.modelNumber in HT_SOLO_REAR_SURROUND_DEVICES:
            sat_count = 2
        else:
            sat_count = 3
        bonded_members = [left_rear, right_rear, self.sub]
        need_bonding = True
        # Step through a list of valid ids
        for coefficient_id in self.generate_coefficient_ids(
                ht_device, version_list):
            # Tune and verify calibration
            self.tune_ht_with_surrounds_and_sub(
                ht_device, left_rear, right_rear,
                coefficient_id=coefficient_id,
                do_room_bonding=need_bonding,
                verify_calibration=False,
                verify_dsp=versions_are_valid)
            # Only bond zps during the first iteration
            if need_bonding:
                need_bonding = False
            # Reboot the component
            self.bounce_anacapa(ht_device)
            sleep(2)
            # Wait for sats to re-attach
            wait_until_true(
                lambda: len(ht_device.topology.satellites) == sat_count,
                iteration_delay=sat_count,
                reason="Timeout waiting for HT Master to see" +
                       " all sats come back.")
            if versions_are_valid:
                # Verify calibration is valid
                self.verify_trueplay_on(
                    ht_device, bonded_members=bonded_members)
            else:
                # Verify calibration is invalid
                self.verify_trueplay_invalid(
                    ht_device, bonded_members=bonded_members)
