[docs]class SetBassTreble(TrueplayBaseFixture):
    """
    Tests setting Bass & Treble before & after Trueplay tuning
    """
    TEST_TYPE = ["TRUEPLAY_WEEKLY"]

    def _set_bass_treble(self, zp, level):
        """
        Sets Bass & Treble before Trueplay tuning
        :param :class:`sonos.client.internal.SonosZoneComponent` zp:
        :param int level
        """
        zp.RenderingControl.SetBass(level)
        zp.RenderingControl.SetTreble(level)

    def _verify_bass_treble(self, zp, expected_level):
        """
        Verifies Bass & Treble levels
        :param :class:`sonos.client.internal.SonosZoneComponent` zp:
        :param int expected_level
        """
        self.verifyEqualOrFailCase(
            expected_level, zp.RenderingControl.GetBass(),
            'Bass should == {}'.format(expected_level))
        self.verifyEqualOrFailCase(
            expected_level, zp.RenderingControl.GetTreble(),
            'Treble should == {}'.format(expected_level))

    def _test_set_bass_treble_post_tune(self, zp):
        """
        Tests setting Bass & Treble on devices after tuning
        Trueplay Tuning
        :param :class:`sonos.client.internal.SonosZoneComponent` zp:
        """
        self._verify_bass_treble(zp, BASS_TREBLE_ZERO)
        self._set_bass_treble(zp, BASS_TREBLE_THREE)
        zp.RenderingControl.SetRoomCalibrationStatus(False)
        self._verify_bass_treble(zp, BASS_TREBLE_THREE)
