[docs]    def verify_all_tuning_states(self, zp, config, bonded_zp_list):
        """
        Verifies all states of tuning in bonded_config using
        zoneplayers in bonded_zp_list will restore TruePlay HT delays
        to previous state after a subsequent tuning.
        :param :class:`sonos.client.internal.SonosZoneComponent` zp:
        :param str bonded_config
        :param list bonded_zp_list
        """
        self._verify_fully_tuned_restoration(zp, config, bonded_zp_list)
        self._verify_untuned_restoration(zp, config)
        self._verify_part_tuned_restoration(zp, config)


    def _verify_fully_tuned_restoration(
            self, zp, bonded_config, bonded_zp_list):
        """
        Verifies a fully tuned system in bonded_config using zoneplayers
        in bonded_zp_list will restore TruePlay HT delays to previous
        state after a subsequent tuning.
        :param :class:`sonos.client.internal.SonosZoneComponent` zp:
        :param str bonded_config
        :param list bonded_zp_list
        """
        self.logger.info("Verifying fully-tuned delay restoration")
        self.disable_sonar_and_remove_calibration(zp)
        self.tune_room(root_zp=zp, bonded_zps=bonded_zp_list,
                       bonded_type=bonded_config, use_fast_success=False)
        delay_list = self.get_list_delays(zp)
        self._apply_spatial_and_verify(zp, bonded_config)
        restored_delay_list = self.get_list_delays(zp)
        self.verifyEqualOrFailCase(restored_delay_list, delay_list,
                                   "Restore prior Sonar"
                                   " Spatial Tuning.")

    def _verify_untuned_restoration(self, zp, bonded_config):
        """
        Verifies an untuned system in bonded_config using zoneplayers
        in bonded_zp_list will restore TruePlay HT delays to previous
        state after a subsequent tuning.
        :param :class:`sonos.client.internal.SonosZoneComponent` zp:
        :param str bonded_config
        :param list bonded_zp_list
        """
        self.logger.info("Verifying untuned delay restoration")
        self.disable_sonar_and_remove_calibration(zp)
        self._apply_spatial_and_verify(zp, bonded_config)

    def _verify_part_tuned_restoration(self, zp, bonded_config):
        """
        Verifies a partially-tuned system (with spatial tuning only) in
        bonded_config using zoneplayers in bonded_zp_list will restore
        TruePlay HT delays to previous state after a subsequent
        tuning.  :param
        :class:`sonos.client.internal.SonosZoneComponent` zp:
        :param str bonded_config
        :param list bonded_zp_list
        """
        self.logger.info("Verifying partially-tuned delay restoration")
        self.disable_sonar_and_remove_calibration(zp)
        self.enter_config_mode(zp, bonded_config)
        self.set_coefficients(zp, bonded_config,
                              calibration_mode=SPATIAL_MODE)
        self.enter_config_mode(zp, bonded_config,
                               calibration_mode=SPECTRAL_MODE)
        self._apply_spatial_and_verify(zp, bonded_config)

    def _apply_spatial_and_verify(self, zp, bonded_type):
        """
        Applies spatial tuning to HT zp with HT string bonded_type
        e.g. 'ht_surrounds_sub' or 'ht_standalone'
        :param :class:`sonos.client.internal.SonosZoneComponent` zp:
        :param str bonded_type
        """
        delay_list = self.get_list_delays(zp)
        self.enter_config_mode(zp, bonded_type,
                               calibration_mode=SPATIAL_MODE)
        # tune with spatial coefficients / mode
        file_name = ALTERNATE_SPATIAL_COEFFS_FILES[zp.modelNumber]
        file_path = path.join(
            path.dirname(path.realpath(__file__)), file_name)
        with open(file_path, 'r') as coefficient_file:
            coefficient_file.read().replace('\n', '')
        alt_coeffs = file_path

        self.set_coefficients(zp, bonded_type,
                              calibration_mode=SPATIAL_MODE,
                              coefficients=alt_coeffs)

        new_delay_list = self.get_list_delays(zp)
        self.verifyNotEqualOrFailCase(new_delay_list, delay_list,
                                      "Apply new Sonar"
                                      " Spatial Tuning.")
        # restore
        self.enter_config_mode(zp, bonded_type,
                               calibration_mode=SPECTRAL_MODE)
        self.exit_config_mode(zp, options=CALIBRATE_FAIL)
        restored_delay_list = self.get_list_delays(zp)
        self.verifyNotEqualOrFailCase(restored_delay_list, new_delay_list,
                                      "Do not keep current Sonar"
                                      " Spatial Tuning.")
