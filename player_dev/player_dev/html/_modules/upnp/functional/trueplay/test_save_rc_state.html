[docs]    def generate_zp_and_restore_vector(self):
        """
        Generator to return a test device and an EQ test vector.
        :return: (zp, EQStartVector)
        :rtype: :tuple:
        """
        for zp in self.generate_standalone_devices():
            for test_vector in self.restore_test_vectors:
                yield zp, test_vector


    def _add_start_vectors(self):
        """
        Define a set of EQ Vectors to initialize the DUT
        to a known state.
        """
        self.eq_start_vector = namedtuple('EQStartVector',
                                          ['volume', 'bass', 'treble', 'mute'])
        self.initial_eq_vector = self.eq_start_vector(INITIAL_VOLUME, 2, 2,
                                                      False)
        self.eq_test_vectors = [
            self.eq_start_vector(3, 3, 3, True),
            self.eq_start_vector(4, 4, 4, True),
            self.eq_start_vector(5, 5, 5, True),
            self.eq_start_vector(6, 6, 6, True),
            self.eq_start_vector(7, 7, 7, True)
        ]

    def _add_restore_vectors(self):
        """
        Define a set of expected restore vectors.

        Test vectors for single devices:
        If ExitConfigMode is invoked with CALIBRATE_SUCCESS then the completion
        tone should play.
        If ExitConfigMode is invoked with CALIBRATE_FAIL then the completion
        tone should not play.
        If the initial URI was the buzzer then the restored URI should be empty
        Otherwise the initial URI should be restored.
        The initial volume should always be restored.

        If no actual calibration took place and no sonarEQ.xml file was created
        calibration-available should be False.

        On CALIBRATE_FAIL calibration-enabled should be True if
        inital_calibration_enabled was True otherwise it should be False.
        """

        self.restore_vector = namedtuple('RestoreVector',
                                         ['calibration_success',
                                          'use_track',
                                          'initial_volume',
                                          'expected_restored_volume',
                                          'initial_calibration_enabled',
                                          'expected_calibration_available',
                                          'expected_calibration_enabled'])

        self.restore_test_vectors = [
            self.restore_vector(CALIBRATE_SUCCESS, True, INITIAL_VOLUME,
                                INITIAL_VOLUME, True, False, True),
            self.restore_vector(CALIBRATE_SUCCESS, True, INITIAL_VOLUME,
                                INITIAL_VOLUME, False, False, True),
            self.restore_vector(CALIBRATE_SUCCESS, False, INITIAL_VOLUME,
                                INITIAL_VOLUME, True, False, True),
            self.restore_vector(CALIBRATE_SUCCESS, False, INITIAL_VOLUME,
                                INITIAL_VOLUME, False, False, True),
            self.restore_vector(CALIBRATE_FAIL, True, INITIAL_VOLUME,
                                INITIAL_VOLUME, False, False, False),
            self.restore_vector(CALIBRATE_FAIL, False, INITIAL_VOLUME,
                                INITIAL_VOLUME, True, False, True)
        ]

    def _set_eq_vector(self, zp, eq_vector):
        """
        Sets the starting eq_vector on the device under test.

        :param zp: The device under test
        :type zone: :class:`~sonos.client.internal.SonosZoneComponent`
        :param eq_vector: Test vector
        :type named_tuple: EQStartVector
        """
        (volume, bass, treble, mute) = eq_vector
        zp.RenderingControl.SetBass(bass)
        zp.RenderingControl.SetTreble(treble)
        zp.RenderingControl.SetMute(mute)
        zp.RenderingControl.SetVolume(volume)
        wait_until_true(lambda: zp.RenderingControl.GetBass() == bass)
        wait_until_true(lambda: zp.RenderingControl.GetTreble() == treble)
        wait_until_true(lambda: zp.RenderingControl.GetMute() == mute)
        wait_until_true(lambda: zp.RenderingControl.GetVolume() == volume)

    def _get_eq_vector(self, zp):
        """
        Returns the current eq_vector of the device under test.

        :param zp: The device under test
        :type zone: :class:`~sonos.client.internal.SonosZoneComponent`
        :return eq_vector: Test vector
        :rtype named_tuple: EQStartVector
        """
        # If success then mute should be off
        current_vector = self.eq_start_vector(zp.RenderingControl.GetVolume(),
                                              zp.RenderingControl.GetBass(),
                                              zp.RenderingControl.GetTreble(),
                                              zp.RenderingControl.GetMute())
        return current_vector

    def test_save_rc_state(self, zp, test_vector):
        """
        Enter/Exit ConfigMode and verify that completion-tone plays but the
        buzzer is not restored at the end.

        :param zp: The device under test
        :type zone: :class:`~sonos.client.internal.SonosZoneComponent`
        :param test_vector: Test vector
        :type named_tuple: RestoreVector
        """
        self.device_list = [zp]
        (calibration_success, use_track, initial_volume,
            expected_restored_volume, initial_calibration_enabled,
            expected_calibration_available,
            expected_calibration_enabled) = test_vector

        """
        Prepare the DUT with the starting vector values and an initial
        URI - either a track or buzzer.
        """
        self._set_eq_vector(zp, self.initial_eq_vector)
        zp.RenderingControl.SetRoomCalibrationStatus(
            initial_calibration_enabled)

        if use_track:
            self.add_test_track_to_avt(zp)
        else:
            zp.AVTransport.use_buzzer()

        zp.AVTransport.Play_and_wait()
        zp.AVTransport.Stop_and_wait()
        initial_uri = zp.AVTransport.get_current_uri()
        if use_track:
            expected_uri = initial_uri
        else:
            # Sonar should never restore the buzzer uri on completion
            expected_uri = ""

        """
        Enter Config Mode multiple times with different EQ vectors.
        Only the first enter should save the EQ.
        """
        self.enter_config_mode(zp)
        for new_vector in self.eq_test_vectors:
            self._set_eq_vector(zp, new_vector)
            self.enter_config_mode(zp)

        """
        Exit Config Mode with the specified calibration success value.
        Collect resulting actual values to compare with expected values.
        """
        self.exit_config_mode(zp, options=calibration_success,
                              use_fast_success=False)
        zp.AVTransport.wait_for_avt_state(AVTransportService.PLAYING_STATE)
        completion_uri = zp.AVTransport.get_current_uri()
        completion_volume = zp.RenderingControl.GetVolume()
        zp.AVTransport.wait_for_avt_state(AVTransportService.STOPPED_STATE)

        if calibration_success == CALIBRATE_SUCCESS:
            """
            If Calibration Status is SUCCES then
            verify that completion tone played and that
            it played at the correct volume.
            """
            self.verifyEqualOrFailCase(SONAR_COMPLETION_TONE, completion_uri,
                                       "The Sonar Completion Tone should"
                                       " have played.")
            self.verifyEqualOrFailCase(
                SONAR_COMPLETION_VOLUME, completion_volume,
                "Sonar Completion tone should have played at the"
                " correct volume.")
        else:
            """
            Otherwise verify that completion tone did not play.
            """
            self.verifyNotEqualOrFailCase(
                SONAR_COMPLETION_TONE, completion_uri,
                "The Sonar Completion Tone should not have played.")

        """
        Verify that the correct uri and volume were restored upon completion.
        """
        wait_until_true(lambda:
                        zp.AVTransport.get_current_uri() == expected_uri)

        wait_until_true(lambda:
                        zp.RenderingControl.GetVolume() ==
                        expected_restored_volume)

        """
        Read the current EQ Values from the DUT for comparison.
        """
        current_dict = self._get_eq_vector(zp)._asdict()

        sonar_status = zp.RenderingControl.GetRoomCalibrationStatus()
        sonar_enabled = int(sonar_status["RoomCalibrationEnabled"])
        self.verifyEqualOrFailCase(expected_calibration_enabled,
                                   bool(sonar_enabled),
                                   "Sonar should be enabled on success.")

        """
        Verify appropriate EQ values are restored.
        Volume is always restored.
        Mute is off if success otherwise it is initial value.
        Bass and Treble are only restored if failed.
        """
        initial_dict = self.initial_eq_vector._asdict()
        if calibration_success == CALIBRATE_SUCCESS:
            self.verifyFalseOrFailCase(current_dict["mute"],
                                       "Mute should be off if success.")
        else:
            self.verifyEqualOrFailCase(initial_dict["bass"],
                                       current_dict["bass"],
                                       "Base should be restored on fail.")
            self.verifyEqualOrFailCase(initial_dict["treble"],
                                       current_dict["treble"],
                                       "Treble should be restored on fail.")
            self.verifyEqualOrFailCase(initial_dict["mute"],
                                       current_dict["mute"],
                                       "Mute should be restored on fail.")

        """
        Just make sure it gets here as comparison value is depends
        on whether an initial Trueplay configuration existed.
        Don't verify the value itself. Just verify there is a value returned.
        """
        sonar_calibration_available = int(
            sonar_status["RoomCalibrationAvailable"])
        self.logger.debug("RoomCalibrationAvailable should be returned: {}"
                          .format(bool(sonar_calibration_available)))
