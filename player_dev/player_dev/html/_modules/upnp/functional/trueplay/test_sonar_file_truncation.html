        """
        Make a L{StaticProducer} that will produce the body of this response.

        This method will also set the response code and Content-* headers.

        @param request: The L{Request} object.
        @param fileForReading: The file object containing the resource.
        @return: A L{StaticProducer}.  Calling C{.start()} on this will begin
            producing the response.
        """
        byteRange = request.getHeader('range')
        if byteRange is None:
            self._setContentHeaders(request)
            request.setResponseCode(http.OK)
            return NoRangeStaticProducerTruncate(request, fileForReading)
        try:
            parsedRanges = self._parseRangeHeader(byteRange)
        except ValueError:
            log.msg("Ignoring malformed Range header %r" % (byteRange,))
            self._setContentHeaders(request)
            request.setResponseCode(http.OK)
            return NoRangeStaticProducerTruncate(request, fileForReading)

[docs]class SonarFileTruncation(TrueplayBaseFixture):
    """
    Verifies truncation and missing directories for Trueplay
    testtones are handled correctly by the ZP
    """

    '''
    This test needs to control overriding the test server manually
    since we will be using two separate webservers for testing.
    Prevent the base class from managing the override url.
    Keep track of what server we are using so we only set the string
    once when needed.
    '''
    requires_test_server = False
    current_server = None

    """
    1. Verify the player does not enter Sonar config mode on failures:
    A. Download tone files.
    B. Persistent storage area removed whilst file download.
    2. Verify the player enters Sonar config mode upon successfully
    fetching the tone files.
    """
    TEST_TYPE = ["TRUEPLAY_WEEKLY"]


[docs]    def generate_zp_bonded_type_matrix(self):
        """
        Yield a zp and a bonded type (Standalone, pair, pair with sub)
        to be used when entering config mode. Additional ZPs are
        not necessary since this is just testing a master-only logic path
        """
        for zp in self.generate_standalone_devices():
            # HT devices use different test tones than normal AIO
            if zp.modelNumber in TRUEPLAY_HT_DEVICES_WITH_SPATIAL_STANDALONE:
                bonded_types = HT_BONDED_TYPES
            else:
                bonded_types = AIO_BONDED_TYPES
            for index, bonded_type in enumerate(bonded_types):
                # Limit to two files per ZP
                if index >= 2:
                    break
                yield zp, bonded_type



[docs]    def setUpFixture(self):
        # Add dataio to the list of diag levels to adjust
        self.diag_level_list.append(('dataio', 8))
        # Call super
        super(SonarFileTruncation, self).setUpFixture()
        '''
        Build up a list of all Trueplay tones and the directory to save
        them in to feed into get_files_from_sqatools()
        '''
        archive_files = {}
        for tone_files in generate_tone_files(BONDED_TYPES):
            for tone_file in tone_files.values():
                archive_files[ARCHIVE_FILES_DIR + tone_file] = path.join(
                    TRUEPLAY_VERSION, tone_file)
        # Copy of the download url from the testbed.json file
        self.base_url = self.execution_configuration.build.download_url
        # Version object of the version string from the testbed.json file
        self.target_version_num = Version(
            self.execution_configuration.build.version_number)
        # Prepare the test tone directory by downloading all testtone files
        self.logger.info("Downloading tone files: {}".format(
            archive_files.keys()))
        DownloadManager.get_files_from_sqatools(
            self.base_url, archive_files, version=self.target_version_num)
        self.monitor_diagc = BoolCallBackSetter()
        '''
        Create two webservers. One will host truncated files, the
        other will host delayed transmission files. Use two separate
        default ports. The webserver will pick an available port if
        one is in use already, but to make it easier to clarify that
        they are running on two separate ports, actually set them
        to run on two separate ports.
        '''
        # Create a file truncating webserver
        self.websrv = WebServer(
            start_reactor=False, file_resource=StaticFileHandlerTruncate)
        self.websrv.start(
            port=DEFAULT_WEB1_PORT,
            path=path.realpath('.'))
        # Create a slow responding webserver
        self.websrv_alt = WebServer(
            start_reactor=False, file_resource=StaticFileHandlerDelay)
        self.websrv_alt.start(
            port=DEFAULT_WEB2_PORT,
            path=path.realpath('.'))

    def _set_test_server_override(self, test_server_url, active_zp):
        """
        Internal helper method to track and set the current
        test server override used for fetching the test tones.
        :param test_server_url: ip:port of the current web server
        """
        if self.current_server != test_server_url:
            self.current_server = test_server_url
            self.set_test_server_override('http://' + test_server_url,
                                          active_zp=active_zp)

[docs]    def wait_on_diag(self, zp, watch_str, timeout_seconds=7):
        """
        :param :class:`sonos.client.internal.SonosZoneComponent` zp:
        :param str watch_str: UDP watch string.
        :param int timeout_seconds: Time in seconds to wait for diag.
        """
        try:
            wait_until_true(
                lambda: self.monitor_diagc.get_state(),
                iteration_delay=1,
                timeout_seconds=timeout_seconds)
        except TimeoutError:
            pass
        zp.diag.stop_watching_udp_log_for_string(watch_str)
        return self.monitor_diagc.get_state()



[docs]    def verify_config_mode_failure(self, zp, bonded_type,
                                   watch_str, timeout=5):
        """
        Verifies the failure state (820) when downloading tone files fails
        """
        exception_hit = False
        try:
            self.enter_config_mode(zp, bonded_type=bonded_type, attempts=0)
        except UPnPError as e:
            exception_hit = True
            self.verifyEqualOrFailCase(
                UPNP_ERROR_820_R_UPNP_DP_CONFIGMODE_SONARCAL_DOWNLOAD_FAILED,
                e.upnp_error,
                'Player should hit 820 entering config mode')
        self.verifyTrueOrFailCase(exception_hit,
                                  "Expect enter sonar config to fail")
        self.verifyTrueOrFailCase(
            self.wait_on_diag(zp, watch_str, timeout_seconds=timeout),
            "Expect the player to generate diag '{}'".format(watch_str))
        self.exit_config_mode(zp)

[docs]    @combinatorial('generate_zp_bonded_type_matrix')
    def test_sonar_file_missing_dir(self, zp, bonded_type):
        """
        When the complete.ogg file is being downloaded by the
        player and the persistent storage directory on the player
        /jffs/sonar-tone has been removed during the write, the
        player should fail the operation with a RDLURLST_ERROR_WRITE (-5)
        error constant declared in all/oc/zone/zoneplayer/fileio.h.
        :param :class:`sonos.client.internal.SonosZoneComponent` zp:
        :param str bonded_type: the type of bond to test (standalone, pair)
        """
        self.device_list = [zp]
        tone_files = format_tone_files(bonded_type)
        self._set_test_server_override(self.websrv_alt.host(), zp)
        self.setup_diagc(zp)
        self._remove_tone_directory(zp)
        '''
        Watch for a message indicating a ZP is downloading the completion
        tone. When it is caught, delete the sonar tones directory before
        the download completes (Thanks to a delay configured webserver)
        '''
        watch_str = FETCHING_STRING.format(self.current_server,
                                           TRUEPLAY_VERSION,
                                           tone_files[COMPLETE])
        zp.diag.watch_udp_log_for_string(
            watch_str,
            lambda msg: Thread(
                target=lambda: self._remove_tone_directory(zp)).start())
        self.logger.info("Tone file '{}'".format(tone_files[COMPLETE]))

        watch_str = FETCH_FAIL_STRING.format(self.current_server,
                                             TRUEPLAY_VERSION,
                                             tone_files[COMPLETE],
                                             '-5')
        self.set_diag_watch(zp, watch_str)
        self.verify_config_mode_failure(zp, bonded_type, watch_str)
        self.reset_diagc(zp)

    def test_sonar_file_truncation(self, zp, bonded_type):
        """
        A tone file fetched from the server which has been truncated
        should generate a RDLURLST_ERROR_TRUNC (-4) error constant
        declared in all/oc/zone/zoneplayer/fileio.h
        :param :class:`sonos.client.internal.SonosZoneComponent` zp:
        :param str bonded_type: the type of bond to test (standalone, pair)
        """

        tone_files = format_tone_files(bonded_type)
        self._set_test_server_override(self.websrv.host(), zp)
        self.setup_diagc(zp)
        watch_str = FETCH_FAIL_STRING.format(self.current_server,
                                             TRUEPLAY_VERSION,
                                             tone_files[TESTTONE],
                                             '-4')
        self.set_diag_watch(zp, watch_str)
        self.verify_config_mode_failure(zp, bonded_type,
                                        watch_str, timeout=15)
        self.reset_diagc(zp)
