    def _test_multiple_calibration_orientation(self,
                                               master_zp,
                                               members,
                                               tuning_method):
        """
        After calibrating a room, when the orientation of one of the ZPs
        has changed, the calibration should move to an on, but unavailable
        state, implying user intent is on, but there is no tuning to enable.
        Once that orientation is tuned, It
        should become both on and available.
        All orientations should be recoverable and tunable.
        Verify by tuning all orientations, and then checking
        each orientation is tuned in a second pass.
        :param master_zp: main zp in the bonded room
        :param members: member zps in the bonded room (if any)
        :param tuning_method: The method that will be called to tune/bond
          the given room
        """
        devices = tuple([master_zp] + members)
        # Preserve if we have not yet tuned any devices
        zps_not_yet_tuned = True
        # Step through all zp orientations
        for _ in self._generate_orientation_matrix(devices,
                                                   use_limited_matrix=False):
            if not zps_not_yet_tuned:
                # Only test if tuning should be on & unavail after
                #  the first calibration
                self.logger.info("Orientation should have cal ON but"
                                 " unavailable for all devices.")
                for zp in devices:
                    self.verify_trueplay_tuning(zp, True, False,
                                                verify_sonar_eq_xml=False)
            # Tune new orientation so that Trueplay is on and available
            tuning_method(do_room_bonding=zps_not_yet_tuned)
            # Do not attempt bonding again in the next loop if we tuned/bonded
            #   at least one time
            if zps_not_yet_tuned:
                zps_not_yet_tuned = False

        # Step through all zp orientations again
        for _ in self._generate_orientation_matrix(devices,
                                                   use_limited_matrix=False):
            self.logger.info("All ZP orientations should"
                             " restore to a calibration")
            for zp in devices:
                # Verify the calibrated orientation is restored
                self.verify_trueplay_tuning(zp, True, True,
                                            verify_sonar_eq_xml=False)
            # verify that toggling off disables DSP blocks
            self.toggle_trueplay(master_zp, 'off', bonded_members=members)
            self.verify_dsp_bypass(master_zp,
                                   roomiir_bypass=True,
                                   should_add_track_to_avt=True,
                                   should_play_avt=True)
            # verify that toggling on enables DSP blocks
            self.toggle_trueplay(master_zp, 'on', bonded_members=members)
            self.verify_dsp_bypass(master_zp,
                                   roomiir_bypass=False,
                                   should_add_track_to_avt=True,
                                   should_play_avt=True)
