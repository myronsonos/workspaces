    def _verify_dsp_bypass_pre_and_post_tune(
            self, master_zp, right_member=None, include_sub=False):
        """
        Verifies the expected DSP bypass states when in an untuned
        state. Then tunes the room and verifies dsp bypass states.
        Then turns Trueplay off and verifies dsp bypass. Then unbonds
        and verifies uncalibrated bypass again.
        :param master_zp:
        :param right_member:
        :param include_sub:
        """
        # Define tuning method to be used
        if right_member and include_sub:
            tuning_method = partial(self.tune_stereo_pair_with_sub,
                                    master_zp, right_member)
            members = [right_member, self.sub]
        elif right_member:
            tuning_method = partial(self.tune_stereo_pair,
                                    master_zp, right_member)
            members = [right_member]
        elif include_sub:
            tuning_method = partial(self.tune_standalone_with_sub, master_zp)
            members = [self.sub]
        else:
            tuning_method = partial(self.tune_standalone, master_zp)
            members = []
        # Bond system, verify DSP bypass, then tune system
        # Bypass should be True for all 3 states
        tuning_method(use_fast_success=False,
                      room_bonded_callback=partial(self._verify_dsp_bypass,
                                                   master_zp,
                                                   bonded_members=members,
                                                   roomiir_bypass=True))
        # Wait for completion tone to end
        self.verify_playtime(master_zp, COMPLETION_TONE_DURATION)
        # Verify post tuned DSP bypass
        # Bypass False all around with TP On
        self._verify_dsp_bypass(master_zp,
                                bonded_members=members,
                                roomiir_bypass=False)
        # Turn off Trueplay
        self.toggle_trueplay(master_zp, 'off', bonded_members=members)
        # Bypass should be True all around with TP Off
        self._verify_dsp_bypass(master_zp,
                                bonded_members=members,
                                roomiir_bypass=True)
        # Turn Trueplay back on
        self.toggle_trueplay(master_zp, 'on', bonded_members=members)
        # Bypass should be False all around with TP On
        self._verify_dsp_bypass(master_zp,
                                bonded_members=members,
                                roomiir_bypass=False)
        device_list = [master_zp] + members

        # Unbond if needed
        [zp.DeviceProperties.cleanup_bonded_configuration()
         for zp in device_list if zp.is_bonded()]

        # Verify no devices have Trueplay available
        for zp in device_list:
            self.verify_trueplay_tuning(zp, False, False,
                                        verify_sonar_eq_xml=False)
            # Bypass should be True for master & members
            self._verify_dsp_bypass(master_zp,
                                    bonded_members=members,
                                    roomiir_bypass=True)

    def _verify_dsp_bypass_pre_and_post_tune_ht(
            self, ht_device, sats=None, include_sub=False):
        """
        Verifies the expected DSP bypass states when in an untuned state.
        Then tunes the Room and verifies the expected DSP bypass states
        post tuning. Then verifies full/ambient mode. Then verifies
        Trueplay off dsp bypass. Then unbonds and verifies Trueplay
        uncalibrated bypass status again.
        :param ht_device:
        :param sats: list of left and right sat to be bonded if any
        :param include_sub: True to test 51 or 31
        """
        if sats is None:
            sats = []
        # Define tuning method to be used
        if sats and include_sub:
            tuning_method = partial(self.tune_ht_with_surrounds_and_sub,
                                    ht_device, *sats)
            members = sats + [self.sub]
        elif sats:
            tuning_method = partial(self.tune_ht_with_surrounds,
                                    ht_device, *sats)
            members = sats
        elif include_sub:
            tuning_method = partial(self.tune_standalone_with_sub, ht_device)
            members = [self.sub]
        else:
            tuning_method = partial(self.tune_standalone, ht_device)
            members = []
        device_list = [ht_device] + members
        # Bond HT system, verify DSP bypass, then tune HT system
        # Bypass should be True for all 3 states
        tuning_method(use_fast_success=False,
                      room_bonded_callback=partial(self._verify_dsp_bypass,
                                                   ht_device,
                                                   bonded_members=members,
                                                   roomiir_bypass=True,
                                                   spatial_bypass=True,
                                                   satellite_bypass=True))
        # Wait for completion tone to end
        self.verify_playtime(ht_device, COMPLETION_TONE_DURATION)
        # Verify post tuned DSP bypass
        # Bypass False all around with TP On and in Ambient mode
        self._verify_dsp_bypass(ht_device,
                                bonded_members=members,
                                roomiir_bypass=False,
                                spatial_bypass=False,
                                satellite_bypass=False)
        # Satellite Bypass True with TP on and in Full mode
        ht_device.RenderingControl.set_surround_mode('full')
        self._verify_dsp_bypass(ht_device,
                                bonded_members=members,
                                roomiir_bypass=False,
                                spatial_bypass=False,
                                satellite_bypass=True)
        # Turn off Trueplay
        self.toggle_trueplay(ht_device, 'off', bonded_members=members)
        # Bypass should be True all around with TP Off
        self._verify_dsp_bypass(ht_device,
                                bonded_members=members,
                                roomiir_bypass=True,
                                spatial_bypass=True,
                                satellite_bypass=True)
        # Restore back to ambient
        ht_device.RenderingControl.set_surround_mode('ambient')
        # Turn Trueplay back on
        self.toggle_trueplay(ht_device, 'on', bonded_members=members)
        # Bypass should be False all around with TP On and in ambient
        self._verify_dsp_bypass(ht_device,
                                bonded_members=members,
                                roomiir_bypass=False,
                                spatial_bypass=False,
                                satellite_bypass=False)
        # Cleanup
        [zp.DeviceProperties.cleanup_bonded_configuration()
         for zp in device_list if zp.is_bonded()]

        if sats and not include_sub:
            # Verify HT device still has Trueplay On
            self.verify_trueplay_tuning(ht_device, True, True,
                                        verify_sonar_eq_xml=False)
            # Bypass should be False with TP On
            self._verify_dsp_bypass(ht_device,
                                    roomiir_bypass=False,
                                    spatial_bypass=False,
                                    satellite_bypass=False)
            for zp in sats:
                # Verify Sat is uncalibrated
                self.verify_trueplay_tuning(zp, False, False,
                                            verify_sonar_eq_xml=False)
                # Bypass should be True for now uncalibrated Sats
                self._verify_dsp_bypass(zp, roomiir_bypass=True)
        elif sats or include_sub:
            # Verify no devices have Trueplay available
            for zp in device_list:
                self.verify_trueplay_tuning(zp, False, False,
                                            verify_sonar_eq_xml=False)
            # Bypass should be True for HT master & members
            self._verify_dsp_bypass(ht_device,
                                    bonded_members=members,
                                    roomiir_bypass=True,
                                    spatial_bypass=True,
                                    satellite_bypass=True)
