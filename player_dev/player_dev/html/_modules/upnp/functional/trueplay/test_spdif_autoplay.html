[docs]    def set_spdif_signal_on(self, ht_device, verify=True, **_):
        """
        Starts autoplay on a given HT device. Should start
        almost immediately.
        :param ht_device:
        :param verify: Set to True to verify the SPDIF signal starts
        :param _: ignored kwargs
        """
        sleep(12)
        self.logger.info("Turning ON SPDIF signal for ZP {}".format(ht_device))
        ht_device.diag.testpoint.ht_spdif_override = 'signal'
        if verify:
            wait_until_true(
                lambda: ('x-sonos-htastream' in
                         ht_device.AVTransport.get_avtransport_uri()),
                iteration_delay=.5, timeout_seconds=16,
                reason="Timeout waiting for SPDIF to start.")



[docs]    def set_spdif_signal_off(self, ht_device, verify=True, **_):
        """
        Stops SPDIF from playing on the given HT device. Stops and clears
        within a couple seconds due to changed autostop threshold.
        :param ht_device:
        :param verify: Set to True to verify the SPDIF signal stops
        :param _: ignored kwargs
        """
        self.logger.info("Turning OFF SPDIF signal for ZP {}"
                         .format(ht_device))
        ht_device.diag.testpoint.ht_autostop_threshold = 1
        ht_device.diag.testpoint.ht_spdif_override = 'nosignal'
        if verify:
            wait_until_true(
                lambda: ('x-sonos-htastream' not in
                         ht_device.AVTransport.get_avtransport_uri()),
                iteration_delay=.5, timeout_seconds=15,
                reason="Timeout waiting for SPDIF to be stopped.")



[docs]    def verify_spdif_not_started(self, ht_device, mode, **_):
        """
        Turns on SPDIF and verifies that after 5s of
        source playback, the calibration tone remains in AVT, and
        autoplay has not triggered.
        :param ht_device:
        :param mode: spatial/spectral
        :param _: ignored kwargs
        """
        self.set_spdif_signal_on(ht_device, verify=False)
        '''
        Autoplay should trigger within a couple seconds of being
        toggled on. Wait 5 seconds to ensure SPDIF logic
        attempted to trigger autoplay.
        '''
        sleep(5)
        self.verifyEqualOrFailCase(
            ht_device.AVTransport.get_avtransport_uri(),
            SONAR_CALIBRATION_TONE,
            "During {}, calibration tone should remain in AVT"
            " if an SPDIF source starts.".format(mode))

    def test_SPDIF_restored_after_success(
            self, ht_device, left_rear, right_rear):
        """
        Sets a HT + Surr + Sub system into TV Playback mode. Then
        Trueplay tunes the system. When the completion tone is done
        playing, we expect the TV source to be restored and playing.
        :param ht_device:
        :param left_rear:
        :param right_rear:
        """
        self.device_list = [ht_device, left_rear, right_rear, self.sub]
        self.check_s11(ht_device)
        self.reset_anacapa_triggers(ht_device)
        # Tune the HT system, ensuring SPDIF is playing prior to
        #   entering config mode
        self.tune_ht_with_surrounds_and_sub(
            ht_device, left_rear, right_rear,
            room_bonded_callback=partial(self.set_spdif_signal_on, ht_device),
            use_fast_success=False)
        # Verify the completion tone plays for the full 16 seconds
        self.verify_playtime(ht_device, COMPLETION_TONE_DURATION)
        # After the completion tone duration has passed, SPDIF should be
        #   restored and playing.
        wait_until_true(
            lambda: ('x-sonos-htastream' in
                     ht_device.AVTransport.get_avtransport_uri()),
            iteration_delay=.5,
            timeout_seconds=6,
            info_threshold=6,
            reason="Timeout waiting for SPDIF to be restored.")
        # It should still be possible to autostop SPDIF after it was restored.
        self.set_spdif_signal_off(ht_device)

[docs]    @combinatorial('generate_standalone_ht_devices')
    def test_spdif_autoplay_capability_restores_after_midtune_trigger(
            self, ht_device):
        """
        Verifies that if SPDIF is triggered mid tune (Which is ignored
        and prevents Autoplay) can be successfully re-primed after
        the user manually selects TV, allowing Autoplay to function again.
        :param ht_device:
        """
        self.device_list = [ht_device]
        self.check_s11(ht_device)
        self.reset_anacapa_triggers(ht_device)
        # Tune HT system. Startup SPDIF mid-tune. It will be ignored.
        self.tune_standalone(ht_device,
                             spectral_coefficients_callback=
                             partial(self.set_spdif_signal_on,
                                     ht_device, verify=False),
                             use_fast_success=False)
        # We should be restored to an empty AVT when completion tone
        #   completes playback
        self.verify_playtime(ht_device, COMPLETION_TONE_DURATION)
        wait_until_true(
            lambda: ('' == ht_device.AVTransport.get_avtransport_uri()),
            iteration_delay=.5, timeout_seconds=20,
            reason="Timeout waiting for empty AVT.")
        # Turn on TV playback manually and verify playing
        self.logger.info("Manually enabling SPDIF on ZP {}".format(ht_device))
        ht_device.AVTransport.use_spdif()
        # Turn off SPDIF at the source, let it natually stop
        #   and prime autoplay
        self.set_spdif_signal_off(ht_device)
        # Turn SPDIF back on from the source. Autoplay should trigger.
        self.set_spdif_signal_on(ht_device)
