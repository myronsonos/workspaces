    """
    Tests the expected behaviors when attempting to Trueplay with
    missing players.
    """
    TEST_TYPE = ["TRUEPLAY"]

    def _topology_members(self, master):
        """
        Returns topology uuids for the room
        :param master:
        """
        if master.modelNumber in TRUEPLAY_HT_DEVICES:
            return master.topology.satellites
        else:
            return master.topology.members

    def _stop_member_player(self, master, missing_zp):
        """
        Stops a member and ensures it has left the masters topology
        :param master:
        :param missing_zp:
        """
        self.stop_anacapa(missing_zp)
        wait_until_true(
            lambda:
            missing_zp.raw_udn not in [member_node.UUID for member_node
                                       in self._topology_members(master)],
            iteration_delay=1,
            timeout_seconds=120,
            reason="Timeout waiting for member to leave master topology."
                   " Member: {}, Master {}".format(missing_zp, master))
        sleep(5)

    def _start_member_player(self, master, missing_zp):
        """
        Starts a member and ensures it has joined the master topology
        :param master:
        :param missing_zp:
        """
        self.start_anacapa(missing_zp)
        wait_until_true(
            lambda:
            missing_zp.raw_udn in [member_node.UUID for member_node
                                   in self._topology_members(master)],
            iteration_delay=1,
            timeout_seconds=120,
            reason="Timeout waiting for member to rejoin master."
                   " Member: {}, Master {}".format(missing_zp, master))
        sleep(20)

    def _test_toggle(self, master, missing_zp):
        """
        Tests toggling off / on of Trueplay while a member is
        missing from the room
        :param master:
        :param missing_zp:
        """
        # Shutdown member
        self._stop_member_player(master, missing_zp)
        # Test success of OFF toggle
        self.toggle_trueplay(master, 'off', check_available=False)
        # Test success of ON toggle
        self.toggle_trueplay(master, 'on', check_available=False)
        # Start the member back up
        self._start_member_player(master, missing_zp)

    def _test_failure_then_success_of_command(
            self, master, missing_zp, method, test_success=True):
        """
        Tests a given method after member 'missing_zp' has been taken offline.
        It will expect a given upnp error, then repeat the method
        with the member online, expecting no exception.
        :param master: The root ZP for the room
        :param missing_zp: ZP to be powered off
        :param method: Method to be tested
        :param test_success: Set to false to disable testing of success path
        """
        exception_caught = False
        # Shut down the target zp
        self._stop_member_player(master, missing_zp)
        try:
            # Execute our test method
            method()
        except UPnPError as e:
            exception_caught = True
            self.verifyTrueOrFailCase(
                e.upnp_error in [1034, 824, 820],
                "Upnp error should be 1034, 824, or 820 when"
                " failing to execute command {}"
                " with a missing ZP. Error was {}".format(method.func.__name__,
                                                          e.upnp_error))
        # Verify we received an exception
        if not exception_caught:
            self.fail("No exception tossed when trying to execute"
                      " method {}".format(method.func.__name__))
        # Start the target zp back up and wait for it to bond with the master
        self._start_member_player(master, missing_zp)
        # Execute our test method. Exceptions will be raised as failure
        if test_success:
            method()

    def test_missing_component_sub(self, zp):
        """
        Steps through failure then success of entering config mode.
        Then steps through failure and success of exiting config mode.
        Then calibrates the room, and steps through the failure and success
        of toggling Trueplay on and off
        :param zp:
        """
        self.device_list = [zp, self.sub]
        self.add_sub(*self.device_list)

        # Set calibration mode to spatial and bonded type to ht standalone sub
        # for devices which support spatial tuning in standalone
        # and standalone sub configs (S16 does NOT support spatial)
        if zp.modelNumber in TRUEPLAY_HT_DEVICES_WITH_SPATIAL_STANDALONE:
            calibration_mode = SPATIAL_MODE
            bonded_type = HT_STANDALONE_SUB
        else:
            calibration_mode = SPECTRAL_MODE
            bonded_type = STANDALONE_SUB
        # Test failure then success to go into Spectral with a missing sub
        self._test_failure_then_success_of_command(
            zp,
            self.sub,
            partial(self.enter_config_mode,
                    zp,
                    bonded_type=bonded_type,
                    calibration_mode=calibration_mode,
                    attempts=0))
        # Test failure to exit config mode
        self._test_failure_then_success_of_command(
            zp,
            self.sub,
            partial(self.exit_config_mode, zp),
            test_success=False)
        # Calibrate the room
        self.tune_standalone_with_sub(zp, do_room_bonding=False)
        # Test toggle
        self._test_toggle(zp, self.sub)

    def test_missing_component_ht(self, ht_device):
        """
        Steps through a calibration. During the calibration, will
        initially error for each step by taking a target member zp offline.
        The member will then be brought back online, and the step repeated
        successfully
        :param ht_device:
        """
        self.device_list = [ht_device] + self.a_satellite_pair + [self.sub]
        missing_zp = self.a_satellite_pair[0]
        self.add_surrounds_and_sub(*self.device_list)

        self.update_diag_levels(*self.a_satellite_pair)
        # Test failure then success to go into Spatial with a missing rear
        self._test_failure_then_success_of_command(
            ht_device,
            missing_zp,
            partial(self.enter_config_mode,
                    ht_device,
                    bonded_type=HT_SURROUNDS_SUB,
                    calibration_mode=SPATIAL_MODE,
                    attempts=0))
        # Test failure to go into Spectral with a missing sub
        self._test_failure_then_success_of_command(
            ht_device,
            self.sub,
            partial(self.enter_config_mode,
                    ht_device,
                    bonded_type=HT_SURROUNDS_SUB,
                    calibration_mode=SPECTRAL_MODE,
                    attempts=0),
            test_success=False)
        # Test failure to exit config mode
        self.enter_config_mode(ht_device,
                               bonded_type=HT_SURROUNDS_SUB,
                               calibration_mode=SPATIAL_MODE)

    def test_spectral_825_error(self, ht_device):
        """
        Tests the error 825 when attempting to enter spectral config mode
        after having been in spatial config mode.
        :param ht_device:
        """
        self.device_list = [ht_device]
        exception_caught = False
        try:
            # Enter Spectral config mode prior to Spatial config mode
            self.enter_config_mode(ht_device,
                                   calibration_mode=SPECTRAL_MODE,
                                   attempts=0)
        except UPnPError as e:
            exception_caught = True
            self.verifyEqualOrFailCase(
                825, e.upnp_error,
                "Upnp error should be 825 when entering Spectral"
                " config mode prior to Spatial. {}".format(ht_device))
        if not exception_caught:
            self.fail("No exception tossed when trying to enter"
                      " SPECTRAL mode prior to entering SPATIAL"
                      " mode for ZP {}".format(ht_device))
