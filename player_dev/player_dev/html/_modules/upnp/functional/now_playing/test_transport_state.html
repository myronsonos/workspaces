[docs]    def delta_seek_test(self, zp, time_delta, expectedFn):
        reltime = zp.AVTransport.GetPositionInfo()['RelTime']
        reltime_seconds = reltime_str_to_seconds(reltime)
        desired_track_position = reltime_seconds + time_delta

        delta_dir = '+' if time_delta >= 0 else '-'
        delta_seek_time_str = "{}00:00:{}".format(delta_dir, abs(time_delta))
        zp.AVTransport.Seek('TIME_DELTA', delta_seek_time_str)

        # give it up-to 5 seconds for the player to report a new position
        # we just care that it changed, the real test is done after we know the
        # the player reported a change in position
        wait_until_true(lambda: self.playpos_changed_by(zp, reltime_seconds, 2),
                        iteration_delay=1, timeout_seconds=5)

        reltime = zp.AVTransport.GetPositionInfo()['RelTime']
        reltime_seconds = reltime_str_to_seconds(reltime)

        expectedFn(reltime_seconds, desired_track_position,
                   "TIME_DELTA:{}: Track should seek to new time".format(time_delta))

[docs]    def test_combinatorial_transportstate_previous(self, zone_generator):
        zone4 = zone_generator
        zone4.AVTransport.use_queue()
        self.logger.debug("test_combinatorial_transportstate_previous, device: %s" % zone4)
        cleared = self.clear_queue(zone4)
        self.verifyTrueOrFailCase(cleared, "Queue should be empty")
        self.add_tracks_to_the_queue(zone4)

        zone4.AVTransport.Play_and_wait()
        current_transport_state = zone4.AVTransport.get_transport_state()
        self.verifyTrueOrFailCase(current_transport_state == "PLAYING" or current_transport_state == "TRANSITIONING",
                                  "Track should be playing")

        pos = zone4.AVTransport.get_queue_position()
        self.logger.debug("Start Position: %s; title: %s" % (pos, zone4.AVTransport.get_current_track_title()))
        zone4.AVTransport.Next()
        wait_until_true(lambda: zone4.AVTransport.get_queue_position() != pos,
                        iteration_delay=1)

        pos = zone4.AVTransport.get_queue_position()
        tracktitle_before_transportstate_next = zone4.AVTransport.get_current_track_title()
        self.logger.debug("Position after Next: %s; title: %s" % (pos, tracktitle_before_transportstate_next))

        zone4.AVTransport.Next()
        wait_until_true(lambda: zone4.AVTransport.get_queue_position() != pos,
                        iteration_delay=1)
        pos = zone4.AVTransport.get_queue_position()

        # Let track meta data settle after position change
        # time.sleep(1)
        wait_until_true(lambda: zone4.AVTransport.get_current_track_title() != tracktitle_before_transportstate_next)
        tracktitle_after_transportstate_next = zone4.AVTransport.get_current_track_title()
        self.logger.debug("Position after 2nd Next: %s; title: %s" % (pos, tracktitle_after_transportstate_next))

        zone4.AVTransport.Previous()
        wait_until_true(lambda: zone4.AVTransport.get_queue_position() != pos,
                        iteration_delay=1)
        pos = zone4.AVTransport.get_queue_position()

        # Let track meta data settle after position change
        # time.sleep(1)
        wait_until_true(lambda: zone4.AVTransport.get_current_track_title() == tracktitle_before_transportstate_next)
        tracktitle_after_transportstate_previous = zone4.AVTransport.get_current_track_title()
        self.logger.debug("Position after Previous: %s; title: %s" % (pos, tracktitle_after_transportstate_previous))

        self.verifyNotEqualOrFailCase(tracktitle_before_transportstate_next, tracktitle_after_transportstate_next, "Track titles should be different")
        self.verifyEqualOrFailCase(tracktitle_before_transportstate_next, tracktitle_after_transportstate_previous, "Both the track titles should be same")

        clear_queue = self.clear_queue(zone4)
        self.verifyTrueOrWarn(clear_queue, "Did clear Queue Successfully? If not warn test.")
