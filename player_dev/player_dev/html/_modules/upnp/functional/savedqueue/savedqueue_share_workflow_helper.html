def pause_anacapa(zone):
    """
    Anacapa pausing context manager.  Calls to stop and start anacapa on the
    zone component and waits for it to leave and join the subnet.  Ensures
    anacapa is restarted in the event of a failure within.

    Arguments:
        :param SonosZoneComponent zone: Zone p

    Returns:
        :return:
    """
    try:
        stop_anacapa(zone)
        yield
    finally:
        start_anacapa(zone)

[docs]    def use_queue(self, zone):
        """
        Point this zone to it's own queue, and wait for GetMediaInfo->CurrentURI to indicate that
        the AVTransportURI has been set appropriately

        Arguments:
        zone -- SonosZoneComponent


        """
        queue_udn = 'x-rincon-queue:%s#0' % zone.raw_udn
        zone.AVTransport.SetAVTransportURI(queue_udn, '')
        wait_until_true(lambda: zone.AVTransport.GetMediaInfo().get('CurrentURI') == queue_udn, timeout_seconds=60)

[docs]    def is_trackqueue_file_size_below_2mb(self, zone):
        """
        Is '/jffs/settings/trackqueue.rsq' on this device less than 2 MB?

        Arguments:
        zone -- SonosZoneComponent

        Returns:
        boolean -- trackqueue.rsq is < 2MB
        None -- no trackqueue.rsq file was found

        """
        file_size = zone.cli.get_file_size(SAVED_QUEUE_FILE)
        if file_size is not None:
            self.logger.debug("%s is %d and should be < %d." % (SAVED_QUEUE_FILE, file_size, MAX_QUEUE_FILE_SIZE))
            return file_size < MAX_QUEUE_FILE_SIZE
        else:
            return None

        """
        Replace the zone queue with the contents of this share

        Arguments:
        zone -- SonosZoneComponent
        share -- string, path to share which will replace current queue

        self.logger.debug("replacing queue on zone %s with AVTransportURI %s" % (zone.raw_udn, share))

        share_uri = PLAY_SHARE_URI_TEMPLATE % (zone.raw_udn, share)
        zone.AVTransport.replace_queue(share_uri)

[docs]    def do_link_zones(self, zone, members):
        """
        Link a list of members to this zone

        Arguments:
        zone -- SonosZoneComponent (GC)
        members -- list of SonosZoneComponents (GMs)

        """
        num_members = len(members)
        self.logger.info("linking %d members to %s" % (num_members, zone.raw_udn))
        self.logger.debug("linking members %s  to %s" % (members, zone.raw_udn))
        self.reset_group_state(zone)
        zone_group_id = zone.ZoneGroupTopology.get_zone_group_id()
        for mem in members:
            self.reset_group_state(mem)
            wait_until_true(lambda: mem.ZoneGroupTopology.get_zone_group_id() != zone_group_id)
        zone.AVTransport.link_zones(members)
        wait_until_true(lambda: [ mem.GroupManagement.is_group_coordinator_local() == False for mem in members ])
        wait_until_true(lambda: zone.ZoneGroupTopology.get_num_linked_zones() == len(members))

    def _process_share_config_data(self, config):
        """
        Extract necessary share information from WorkflowExecutionConfiguration

        This method expects a JSON dictionary with one (or more) shares defined
        in the following manner:
            [
            "share_key": {
                "description": <string>,
                "host": <string>,
                "key": <string>,
                "location: <string>,
                "password": <string>,
                "share_path": <string>,
                "username": <string>
            }
            ]

        Returns:
        dictionary -- map of share keys to tuple containing path and credential strings

        Ex:
           response = {'small_share':('//sb-sqa-nas2/Automation/queue_tests/small_share', ('','')),
                       'large_share':('//sb-sqa-nas2/Automation/queue_tests/large_share', ('',''))}

        """
        music_shares = {}
        if hasattr(config.testbed, 'music_shares'):
            for share in config.testbed.music_shares:
                key = share.key
                path = share.host + '/' + share.share_path
                credentials = (share.username, share.password)
                music_shares[key] = (path, credentials)
        return music_shares

[docs]    def verify_queue_data_equality(self, zone, expected_queue):
        """
        Verify that this zone queue contents matches the expected queue contents

        Arguments:
        zone -- SonosZoneComponent
        expected_queue -- XMLAccessor, queue to compare zone queue to

        """
        logging.debug("verifying that expected and test queue from %s is equal" % zone.raw_udn)
        test_queue = zone.ContentDirectory.get_queue()
        logging.debug("expected queue has %d elements, test queue has %d" % (len(expected_queue),
                                                                             len(test_queue)))
        expected = []
        actual = []
        for ele in expected_queue:
            expected.append([ele.id, ele.title.value])

        for ele1 in test_queue:
            actual.append([ele1.id, ele1.title.value])

        self.maxDiff = None
        self.verifySequenceEqualOrFailCase(expected, actual,
                "verified that expected and test queues from %s are equal" % (zone.raw_udn), max_diff=None)
