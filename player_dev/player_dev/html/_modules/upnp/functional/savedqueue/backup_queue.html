    def test_backupQueue_upnp_action(self, zp):
        '''
        This is the Good path test for the BackupQueue action.
        '''
        self.logger.info("TCM: P-7605 - Test BackupQueue UPnP Invocation - Good Path")
        self.logger.debug("Testing AVTransport.BackupQueue() on %s" % zp.raw_udn)
        zp.AVTransport.BecomeCoordinatorOfStandaloneGroup()
        wait_until_true(lambda: zp.GroupManagement.is_group_coordinator_local() == True)
        self.use_queue(zp)
        self.helpers.queue.clear_tracks_from_queue(zp)
        self.helpers.queue.add_test_tracks_to_queue(zp, (zp.AVTransport.BUZZER_URI))
        zp.AVTransport.BackupQueue()
        wait_until_true(lambda: self.does_trackqueue_file_exist(zp), iteration_delay=1)
        self.verify_trackqueue_file_exists(zp, True)
        self.remove_trackqueue_file(zp)



[docs]    @combinatorial(playback_zone_generator)
    def test_backupQueue_not_gc(self, mem):
        '''
        This is the error path test for the BackupQueue action when zone is not gc.
        '''
        self.logger.info("TCM: P-7605 - Test BackupQueue UPnP Invocation - fail when not gc")
        others = filter(lambda x: x != mem, self.playback_zones)
        gc = others[0]
        self.logger.debug("Testing AVTransport.BackupQueue() on non-gc %s" % mem.raw_udn)
        gc.AVTransport.BecomeCoordinatorOfStandaloneGroup()
        wait_until_true(lambda: gc.GroupManagement.is_group_coordinator_local() == True)
        self.use_queue(gc)
        self.helpers.queue.clear_tracks_from_queue(gc)
        self.helpers.queue.add_test_tracks_to_queue(gc, (gc.AVTransport.BUZZER_URI))
        gc.AVTransport.link_zones([mem])
        have_expected_failure = False
        try:
            mem.AVTransport.BackupQueue()
        except UPnPError, e:
            if e.upnp_error == sonos.client.zone_player.UPNP_ERROR_800_NOT_GROUP_COORDINATOR:
                have_expected_failure = True
        self.verifyTrueOrFailCase(have_expected_failure, "BackupQueue should throw a UPNP_ERROR_800_NOT_GROUP_COORDINATOR exception if zone is not the group coordinator.")
