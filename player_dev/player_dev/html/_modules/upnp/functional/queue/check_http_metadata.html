        """
        Verify that using SetAVTransportURI and SetNextAVTransportURI
        with complete and incomplete DIDL-Lite events the proper metadata
        """
        zone = generate_testbed_unique_playback_devices
        if len(self.track_source.tracks) <= 2:
            self.stop("Needs 3 or more tracks {} to work with".
                      format(len(self.track_source.tracks)))

        zone.AVTransport.SetPlayMode(zone.AVTransport.PLAYMODE_NORMAL)

        # Build the tracks we'll be using
        genDL = zone.AVTransport.generate_didl_lite

        tk1 = self.track_source.tracks[0].media_url
        md1 = genDL(title='Test Title 1',
                    artist='Test Artist 1',
                    upnp_class="object.item.audioItem.musicTrack")

        tk2 = self.track_source.tracks[1].media_url
        md2 = genDL(title='Test Title 2 ' + 'x'*1000 + ' !',
                    artist='', # blank artist
                    album='Long Album 2 ' + 'z'*1000 + ' $',
                    aa_uri=self.track_source.tracks[1].artwork_url,
                    upnp_class="object.item.audioItem.musicTrack")
        self.logger.debug(md2)

        tk3 = self.track_source.tracks[2].media_url
        md3 = genDL(title='Test Title 3 ' + 'X'*1000 + ' !',
                    artist='Some Artist 3 ' + 'Y'*1000 + ' @',
                    album='Long Album 3 ' + 'Z'*1000 + ' $',
                    aa_uri=self.track_source.tracks[1].artwork_url,
                    uri=tk3,
                    duration='0:12:34',
                    upnp_class="object.item.audioItem.musicTrack")

        def helper_event(variables, track, metadata):
            zone.AVTransport.wait_for_state_variables(variables,
                                                      lambda uri, md: (uri, md) == (track, metadata),
                                                      iteration_delay=0.5, timeout_seconds=10)

        # Set the current track -- (tk1/md1) incomplete data
        zone.AVTransport.SetAVTransportURI(tk1, md1, instance_id=self.queue_id)
        helper_event(['AVTransportURI', 'AVTransportURIMetaData'], tk1, md1)
        helper_event(['EnqueuedTransportURI', 'EnqueuedTransportURIMetaData'], tk1, md1)
        zone.AVTransport.wait_for_state_variables('CurrentTrackURI', lambda uri: uri == tk1,
                                                  iteration_delay=0.5, timeout_seconds=20)
        # incomplete data results in ZP-generated DIDL-Lite
        self.verifyNotEqualOrFailCase(md1, zone.AVTransport.get_state_variable('CurrentTrackMetaData'),
                                      "metadata should be ZP-generated")

        # Set the next track -- (tk2/md2) complete data
        zone.AVTransport.SetNextAVTransportURI(tk2, md2, instance_id=self.queue_id)
        helper_event(['NextAVTransportURI', 'NextAVTransportURIMetaData'], tk2, md2)
        #Commented out due to https://jira.sonos.com/browse/SWPBL-84817
        #helper_event(['NextTrackURI', 'NextTrackMetaData'], tk2, md2)

        # Move forward a track, so current will be (tk2/md2)
        zone.AVTransport.Next(instance_id=self.queue_id)
        helper_event(['AVTransportURI', 'AVTransportURIMetaData'], tk2, md2)
        helper_event(['NextAVTransportURI', 'NextAVTransportURIMetaData'], '', '')
        helper_event(['EnqueuedTransportURI', 'EnqueuedTransportURIMetaData'], tk2, md2)
        helper_event(['CurrentTrackURI', 'CurrentTrackMetaData'], tk2, md2)
        #Commented out due to https://jira.sonos.com/browse/SWPBL-84817
        #helper_event(['NextTrackURI', 'NextTrackMetaData'], '', '')

        # Set the next track -- (tk3/md3) copmlete+res data
        zone.AVTransport.SetNextAVTransportURI(tk3, md3, instance_id=self.queue_id)
        helper_event(['NextAVTransportURI', 'NextAVTransportURIMetaData'], tk3, md3)
        #Commented out due to https://jira.sonos.com/browse/SWPBL-84817
        #helper_event(['NextTrackURI', 'NextTrackMetaData'], tk3, md3)

        # Move forward a track, so current will be (tk3/md3) and no next
        zone.AVTransport.Next(instance_id=self.queue_id)
        helper_event(['AVTransportURI', 'AVTransportURIMetaData'], tk3, md3)
        helper_event(['NextAVTransportURI', 'NextAVTransportURIMetaData'], '', '')
        helper_event(['EnqueuedTransportURI', 'EnqueuedTransportURIMetaData'], tk3, md3)
        helper_event(['CurrentTrackURI', 'CurrentTrackMetaData'], tk3, md3)
        helper_event(['NextTrackURI', 'NextTrackMetaData'], '', '')
        # Since we have res@duration, CurrentTrackDuration should be filled
        zone.AVTransport.wait_for_state_variables('CurrentTrackDuration',
                                                  lambda dur: dur == '0:12:34',
                                                  iteration_delay=0.5, timeout_seconds=20)



[docs]    def test_combinatorial_queue(self, generate_testbed_unique_playback_devices):
        """
        Verify that using the Queue (public) with complete and incomplete DIDL-Lite
        events the proper metadata
        """
        zone = generate_testbed_unique_playback_devices

        zone.AVTransport.use_queue(queue_id=self.queue_id)
        zone.AVTransport.SetPlayMode(zone.AVTransport.PLAYMODE_NORMAL)
        zone.AVTransport.clear_queue_and_wait(instance_id=self.queue_id)

        # Build the tracks we'll be using
        genDL = zone.AVTransport.generate_didl_lite

        tk1 = self.track_source.tracks[0].media_url
        md1 = genDL(title='Test Title 1',
                    artist='Test Artist 1',
                    upnp_class="object.item.audioItem.musicTrack")

        tk2 = self.track_source.tracks[1].media_url
        md2 = genDL(title='Test Title 2 ' + 'x'*1000 + ' !',
                    artist='', # blank artist
                    album='Long Album 2 ' + 'z'*1000 + ' $',
                    aa_uri=self.track_source.tracks[1].artwork_url,
                    upnp_class="object.item.audioItem.musicTrack")

        tk3 = self.track_source.tracks[2].media_url
        md3 = genDL(title='Test Title 3 ' + 'X'*1000 + ' !',
                    artist='Some Artist 3 ' + 'Y'*1000 + ' @',
                    album='Long Album 3 ' + 'Z'*1000 + ' $',
                    aa_uri=self.track_source.tracks[1].artwork_url,
                    uri=tk3,
                    duration='0:12:34',
                    item_id='trk3',
                    upnp_class="object.item.audioItem.musicTrack")

        # Add the 3 tracks to the queue
        zone.AVTransport.AddURIToQueue(uri=tk1,
            metadata=md1, track_num=0, as_next=False,
            instance_id=self.queue_id)

        zone.AVTransport.AddURIToQueue(uri=tk2, 
            metadata=md2, track_num=0, as_next=False,
            instance_id=self.queue_id)

        response = zone.AVTransport.AddURIToQueue(uri=tk3,
            metadata=md3, track_num=0, as_next=False,
            instance_id=self.queue_id)
        
        self.verifyEqualOrFailCase(3, int(response['NewQueueLength']),
                                   "Should have added 3 tracks to queue")

        # Helper routine to save some code-space
        def helper_event(variables, track, metadata):
            zone.AVTransport.wait_for_state_variables(variables,
                                                      lambda uri, md: (uri, md) == (track, metadata),
                                                      iteration_delay=0.5, timeout_seconds=20)

        # Should be on the first track -- (tk1/md1) with incomplete data
        helper_event(['EnqueuedTransportURI', 'EnqueuedTransportURIMetaData'], tk1, md1)
        zone.AVTransport.wait_for_state_variables('CurrentTrackURI', lambda uri: uri == tk1,
                                                  iteration_delay=0.5, timeout_seconds=20)
        # incomplete data results in ZP-generated DIDL-Lite
        self.verifyNotEqualOrFailCase(md1, zone.AVTransport.get_state_variable('CurrentTrackMetaData'),
                                      "metadata should be ZP-generated")

        result = zone.AVTransport.GetPositionInfo(instance_id=self.queue_id)
        self.verifyEqualOrFailCase(tk1, result['TrackURI'], "entry-1 URI should match")
        self.verifyNotEqualOrFailCase(md1, result['TrackMetaData'], "entry-1 MD should NOT match")
        self.verifyNotEqualOrFailCase('0:00:00', result['TrackDuration'], "entry-1 should have non-zero duration")

        # Move to next track -- (tk2/md2) with complete data
        #helper_event(['NextTrackURI', 'NextTrackMetaData'], tk2, md2)
        zone.AVTransport.Next(instance_id=self.queue_id)
        helper_event(['EnqueuedTransportURI', 'EnqueuedTransportURIMetaData'], tk2, md2)
        helper_event(['CurrentTrackURI', 'CurrentTrackMetaData'], tk2, md2)

        result = zone.AVTransport.GetPositionInfo(instance_id=self.queue_id)
        self.verifyEqualOrFailCase(tk2, result['TrackURI'], "entry-2 URI should match")
        self.verifyEqualOrFailCase(md2, result['TrackMetaData'], "entry-2 MD should match")
        self.verifyEqualOrFailCase('0:00:00', result['TrackDuration'], "entry-2 should have zero duration")

        # Move to next track -- (tk3/md3) with complete+res data
        helper_event(['NextTrackURI', 'NextTrackMetaData'], tk3, md3)
        zone.AVTransport.Next(instance_id=self.queue_id)
        helper_event(['EnqueuedTransportURI', 'EnqueuedTransportURIMetaData'], tk3, md3)
        helper_event(['CurrentTrackURI', 'CurrentTrackMetaData'], tk3, md3)
        # Since we have res@duration, CurrentTrackDuration should be filled
        zone.AVTransport.wait_for_state_variables('CurrentTrackDuration',
                                                  lambda dur: dur == '0:12:34',
                                                  iteration_delay=0.5, timeout_seconds=20)

        result = zone.AVTransport.GetPositionInfo(instance_id=self.queue_id)
        self.verifyEqualOrFailCase(tk3, result['TrackURI'], "entry-3 URI should match")
        self.verifyEqualOrFailCase(md3, result['TrackMetaData'], "entry-3 MD should match")
        self.verifyEqualOrFailCase('0:12:34', result['TrackDuration'], "entry-3 duration should match")

        # No more tracks
        helper_event(['NextTrackURI', 'NextTrackMetaData'], '', '')

        # Check the results from a Browse()
        result = zone.ContentDirectory.Browse(object_id="Q:{}".format(self.queue_id),
                                              browse_flag="BrowseDirectChildren",
                                              filter="",
                                              starting_index=0,
                                              count=3,
                                              convert_xml=False)

        # NOTE: Result may be 'unicode', so convert to UTF-8 so ElementTree can do work
        root = ET.fromstring(result['Result'].encode('utf-8'))
        md_items = root.findall('{urn:schemas-upnp-org:metadata-1-0/DIDL-Lite/}item')
        self.verifyEqualOrStop(3, len(md_items), "Browse should return expected items")

        # trk1 -- filename
        self.verifyEqualOrFailCase('Q:{}/1'.format(self.queue_id),
            md_items[0].get('id'), "Browsed trk1 should have correct item ID")
        item = md_items[0].find("{http://purl.org/dc/elements/1.1/}title")
        self.verifyIsNotNoneOrStop(item, "Browsed trk1 should have title")
        self.verifyNotEqualOrFailCase('Test Title 1', item.text, "Browsed trk1 should not match metadata title")

        # trk2 -- full
        self.verifyEqualOrFailCase('Q:{}/2'.format(self.queue_id),
            md_items[1].get('id'), "Browsed trk3 should have correct item ID")

        # trk3 -- full w/ origID
        print ET.tostring(md_items[2])
        self.verifyEqualOrFailCase('Q:{}/3'.format(self.queue_id),
            md_items[2].get('id'), "Browsed trk3 should have correct item ID")
        self.verifyEqualOrFailCase('trk3',
            md_items[2].get('{urn:schemas-rinconnetworks-com:metadata-1-0/}origID'),
            "Browsed trk3 should have original item ID")
        
        item = md_items[2].find("{http://purl.org/dc/elements/1.1/}title")
        self.verifyIsNotNoneOrStop(item, "Browsed trk3 should have title")
        self.verifyEqualOrFailCase('Test Title 3 ' + 'X'*1000 + ' !', item.text, "Browsed trk3 should match metadata title")

        # do a more in-depth check of the XML for trk3
        tk3md = ET.fromstring(md3).find('{urn:schemas-upnp-org:metadata-1-0/DIDL-Lite/}item')
        for root_item, chk_item in zip(md_items[2].getchildren(), tk3md.getchildren()):
            self.verifyEqualOrFailCase(ET.tostring(chk_item), ET.tostring(root_item),
                                       "Browsed trk3 metadata should match")

 


[docs]    def test_combinatorial_queue_playlist(self, generate_testbed_unique_playback_devices):
        """
        Verify that enqueueing a playlist (m3u) does NOT event the enqueued DIDL-Lite
        """
        zone = generate_testbed_unique_playback_devices

        zone.AVTransport.use_queue(queue_id=self.queue_id)
        zone.AVTransport.SetPlayMode(zone.AVTransport.PLAYMODE_NORMAL)
        zone.AVTransport.clear_queue_and_wait(instance_id=self.queue_id)

        # The playlist to add with 'complete' metadata
        m3u = self.track_source.playlist
        md = zone.AVTransport.generate_didl_lite(title='A bunch of tracks',
                                                 artist='Various',
                                                 album='Eclectic Mix',
                                                 duration=6000) # duration, sure, why not?

        # Add to queue
        response = zone.AVTransport.AddURIToQueue(uri=m3u,
                                                  metadata=md,
                                                  track_num=0,
                                                  as_next=False,
                                                  instance_id=self.queue_id)

        self.verifyEqualOrFailCase(self.track_source.playlist_len, int(response['NewQueueLength']),
                                   "Should have added all playlist tracks")

        # Should be on first track
        tk = self.track_source.tracks[0].media_url
        zone.AVTransport.wait_for_state_variables('CurrentTrackURI', lambda uri: uri == tk,
                                                  iteration_delay=0.5, timeout_seconds=20)
        zone.AVTransport.wait_for_state_variables(['EnqueuedTransportURI', 'EnqueuedTransportURIMetaData'],
                                                  lambda uri, metadata: uri == m3u and metadata == md,
                                                  iteration_delay=0.5, timeout_seconds=20)
        self.verifyNotEqualOrFailCase(md, zone.AVTransport.get_state_variable('CurrentTrackMetaData'),
                                      "track metadata should be ZP-generated")

        result = zone.AVTransport.GetPositionInfo(instance_id=self.queue_id)
        self.verifyEqualOrFailCase(tk, result['TrackURI'], "GPI URI should match")
        self.verifyNotEqualOrFailCase(md, result['TrackMetaData'], "GPI MD should not match enqueued")
