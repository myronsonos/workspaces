[docs]    def test_combinatorial_desired_first_track_number(self, generate_testbed_unique_playback_devices):
        """
        Verify that setting the DesiredFirstTrackNumber when calling AddURIToQueue causes
        the track to be in the specified position
        """
        zone = generate_testbed_unique_playback_devices
        if len(self.track_source.tracks) <= 3:
            self.stop("Needs 4 or more tracks ({}) to work with".
                      format(len(self.track_source.tracks)))

        zone.AVTransport.use_queue(queue_id=self.queue_id)
        zone.AVTransport.SetPlayMode(zone.AVTransport.PLAYMODE_NORMAL)
        zone.AVTransport.clear_queue_and_wait(instance_id=self.queue_id)

        # Start with many tracks of known quantity

        track_list = []
        while len(track_list) < 10:
            track_list.append(self.track_source.tracks[0])

        self.add_http_tracks_to_queue(zone, track_list)

        # Append to end

        self.add_http_tracks_to_queue(zone, [self.track_source.tracks[1]])
        track_list.append(self.track_source.tracks[1])

        # Insert at beginning

        self.add_http_tracks_to_queue(zone, [self.track_source.tracks[1]], start_position=0+1)
        track_list.insert(0, self.track_source.tracks[1])

        # Insert at end

        self.add_http_tracks_to_queue(zone, [self.track_source.tracks[2]], start_position=len(track_list)+1)
        track_list.append(self.track_source.tracks[2])

        # Insert next to end

        self.add_http_tracks_to_queue(zone, [self.track_source.tracks[3]], start_position=len(track_list))
        track_list.insert(len(track_list)-1, self.track_source.tracks[3])

        # Insert next to beginning

        self.add_http_tracks_to_queue(zone, [self.track_source.tracks[3]], start_position=2)
        track_list.insert(1, self.track_source.tracks[3])

        # Validate

        self.verifyEqualOrFailCase(15, zone.AVTransport.get_number_of_tracks_in_queue(instance_id=self.queue_id),
                                   "Should have expected number of tracks")
        self.verify_tracks_in_queue(zone, track_list, 1)

        zone.AVTransport.clear_queue_and_wait(instance_id=self.queue_id)

[docs]    def test_combinatorial_add_with_bad_updateid(self, generate_testbed_unique_playback_devices):
        """
        Verify that an unexpected Queue UpdateID fails correctly
        """
        zone = generate_testbed_unique_playback_devices

        track_uris = []
        track_mds = []

        for track in self.track_source.tracks:
            md = zone.AVTransport.generate_didl_lite(title=track.title,
                                                     svc_id=None,
                                                     acct_id=None,
                                                     upnp_class="object.item.audioItem.musicTrack",
                                                     artist=track.artist,
                                                     album=track.album,
                                                     aa_uri=track.artwork_url)

            track_uris.append(track.media_url)
            track_mds.append(md)

        # AVT doesn't event the UpdateID?!  Get it from ContentDirectory
        # There's no getter, so use the cached value...
        # (variables are comma delimited, comma delimited)
        updateid = zone.ContentDirectory.get_queue_update_id(queue_id=self.queue_id)
        self.verifyIsNotNoneOrStop(updateid, "Need the current updateID")

        # now make the updateID wildly different
        updateid *= 2

        try:
            zone.AVTransport.AddMultipleURIsToQueue(num_uris=len(self.track_source.tracks),
                                                    uris=' '.join(track_uris),
                                                    metadata=' '.join(track_mds),
                                                    container_uri="",
                                                    container_metadata="",
                                                    track_num=0,
                                                    as_next=False,
                                                    update_id=updateid,
                                                    instance_id=self.queue_id)
        except sonos.exception.UPnPError as e:
            self.verifyEqualOrFailCase(
                sonos.client.zone_player.UPNP_ERROR_1028_STALE_OBJECT,
                e.upnp_error,
                "AddMultiple with bad updateid should fail")
        else:
            self.fail("AddMultiple with bad updateid should raise an exception")

[docs]    def test_combinatorial_shuffle_add(self, generate_testbed_unique_playback_devices):
        """
        Verify that appending/inserting to a shuffled playlist works as expected
        """
        zone = generate_testbed_unique_playback_devices
        if len(self.track_source.tracks) <= 3:
            self.stop("Needs 4 or more tracks ({}) to work with".
                      format(len(self.track_source.tracks)))

        zone.AVTransport.SetPlayMode(zone.AVTransport.PLAYMODE_SHUFFLE_NO_REPEAT)
        zone.AVTransport.use_queue(queue_id=self.queue_id)
        zone.AVTransport.clear_queue_and_wait(instance_id=self.queue_id)

        VARS = ['CurrentTrackURI', 'CurrentTrackMetaData', 'NextTrackURI', 'NextTrackMetaData']

        # Populate with 10 tracks of known values
        track_list = []
        while len(track_list) < 10:
            track_list.append(random.choice(self.track_source.tracks[0:1]))

        self.add_http_tracks_to_queue(zone, track_list, legacy=True, verify=False)

        # Seek to known position and get initial track metadata (so we can compare later)

        zone.AVTransport.set_queue_position(5, instance_id=self.queue_id)
        zone.AVTransport.wait_for_state_variables('CurrentTrack', lambda ct: ct == 5,
                                                  iteration_delay=0.5, timeout_seconds=45)


        initial_data = {}
        get_avt_metadata(zone, VARS, initial_data)

        # Append a track, it should not end up as the "next track"
        # (unless the next track is empty...)

        self.add_http_tracks_to_queue(zone, self.track_source.tracks[2], verify=False)

        track_list.append(self.track_source.tracks[2])

        # it's hard to test for next track not changing... so just wait
        time.sleep(1)

        after_append_data = {}
        get_avt_metadata(zone, VARS, after_append_data)

        self.verifyEqualOrFailCase(initial_data['CurrentTrackURI'], after_append_data['CurrentTrackURI'],
                                   "Appending to a shuffled queue should leave current playback track alone")
        self.verifyEqualOrFailCase(initial_data['CurrentTrackMetaData'], after_append_data['CurrentTrackMetaData'],
                                   "Appending to a shuffled queue should leave current playback track metadata alone")

        # Insert a track, but flag "EnqueueAsNext"
        self.add_http_tracks_to_queue(zone, self.track_source.tracks[3],
                                      start_position=1, as_next=True, verify=False)

        track_list.insert(0, self.track_source.tracks[3])

        # wait until the next track changes
        try:
            zone.AVTransport.wait_for_state_variables('NextTrackURI', lambda uri: uri != after_append_data['NextTrackURI'],
                                                      iteration_delay=0.5, timeout_seconds=45)
        except sonos.exception.TimeoutError:
            pass

        after_enqueue_data = {}
        get_avt_metadata(zone, VARS, after_enqueue_data)

        self.verifyEqualOrFailCase(after_append_data['CurrentTrackURI'], after_enqueue_data['CurrentTrackURI'],
                                   "Insert next to a shuffled queue should leave the curent playback track alone")
        self.verifyEqualOrFailCase(after_append_data['CurrentTrackMetaData'], after_enqueue_data['CurrentTrackMetaData'],
                                   "Insert next to a shuffled queue should leave the current playback track metadata alone")

        zone.AVTransport.clear_queue_and_wait(instance_id=self.queue_id)
