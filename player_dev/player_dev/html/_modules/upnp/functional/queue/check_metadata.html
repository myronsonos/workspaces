    def _verify_browse(self, zone, expected_track_list):
        """
        Verify the tracks in the queue via a Browse operation.
        """

        failed_checks = 0

        # since Browse may return less than the total results, we may need more calls
        start_position = 1
        tracks_checked = 0
        while tracks_checked < len(expected_track_list):
            result = zone.ContentDirectory.Browse(object_id="Q:0",
                                                  browse_flag="BrowseDirectChildren",
                                                  filter="",
                                                  starting_index=start_position - 1,  # 0-based!
                                                  count=len(expected_track_list) - tracks_checked,
                                                  convert_xml=False)

            # NOTE: Result may be 'unicode', so convert to UTF-8 so ElementTree can do work
            root = ET.fromstring(result['Result'].encode('utf-8'))
            md_items = root.findall('{urn:schemas-upnp-org:metadata-1-0/DIDL-Lite/}item')
            if len(md_items) == 0:
                self.logger.debug("no results?! pos=%d, tc=%d, exp=%d" % 
                                  (start_position, tracks_checked, len(expected_track_list)))
                failed_checks += 1
                break

            # NOTE: in these checks, we "lowercase" string comparisons, as they come out of the string table...
            # which can cause case sensitivity oddness.  (string table does not care about case, oddly)
            for item in md_items:
                track = expected_track_list[tracks_checked]

                # log the expected and actual information as DEBUG, so it's always available, but doesn't clutter the logs
                self.logger.debug("%d/%d: track: %s -- item: %s" % (tracks_checked, start_position, track, ET.tostring(item)))

                # NOTE: the strings that come out of ET are 'unicode' (not UTF-8), so no
                # string encode/decode is needed here

                if track['uri'] is not None:
                    res = item.find("{urn:schemas-upnp-org:metadata-1-0/DIDL-Lite/}res")
                    uri = zone.AVTransport.remove_account_sn_from_uri(res.text)
                    if res is None:
                        self.logger.debug("Browsed item should contain res element")
                        failed_checks += 1
                    elif uri != track['uri']:
                        self.logger.debug("Browsed item should match media URL: {} != {}".format(uri, track['uri']))
                        failed_checks += 1

                if track['duration'] is not None:
                    res = item.find("{urn:schemas-upnp-org:metadata-1-0/DIDL-Lite/}res")
                    if res is None or res.get('duration') is None:
                        self.logger.debug("Browsed item should contain duration attribute")
                        failed_checks += 1
                    elif res.get('duration') != track['duration']:
                        self.logger.debug("Browsed item should match duration")
                        failed_checks += 1

                if track['aa_uri'] is not None:
                    albumArtURI = item.find("{urn:schemas-upnp-org:metadata-1-0/upnp/}albumArtURI")
                    if albumArtURI is None:
                        self.logger.debug("Browsed item should contain album art element")
                        failed_checks += 1
                    elif not albumArtURI.text.upper().startswith(track['aa_uri'].upper()):
                        self.logger.debug("Browsed item should have expected album art URL")
                        failed_checks += 1

                if track['title'] is not None:
                    title = item.find("{http://purl.org/dc/elements/1.1/}title")
                    if title is None:
                        self.logger.debug("Browsed item should contain title element")
                        failed_checks += 1
                    elif title.text.lower() != track['title'].lower():
                        self.logger.debug("Browsed item should match title")
                        failed_checks += 1

                if track['album'] is not None:
                    album = item.find("{urn:schemas-upnp-org:metadata-1-0/upnp/}album")
                    if album is None:
                        self.logger.debug("Browsed item should contain album element")
                        failed_checks += 1
                    elif album.text != track['album']:
                        self.logger.debug("Browsed item should match album")
                        failed_checks += 1

                if track['artist'] is not None and track['has_browse_artist']:
                    creator = item.find("{http://purl.org/dc/elements/1.1/}creator")
                    if creator is None:
                        self.logger.debug("Browsed item should contain creator element")
                        failed_checks += 1
                    elif creator.text != track['artist']:
                        self.logger.debug("Browsed item should match artist")
                        failed_checks += 1

                if track['track'] is not None:
                    originalTrackNumber = item.find("{urn:schemas-upnp-org:metadata-1-0/upnp/}originalTrackNumber")
                    if originalTrackNumber is None:
                        self.logger.debug("Browsed item should contain originalTrackNumber element")
                        failed_checks += 1
                    elif originalTrackNumber.text != str(track['track']):
                        self.logger.debug("Browsed item should match track number")
                        failed_checks += 1

                if track['class'] is not None:
                    upnpClass = item.find("{urn:schemas-upnp-org:metadata-1-0/upnp/}class")
                    if upnpClass is None:
                        self.logger.debug("Browsed item should contain class element")
                        failed_checks += 1
                    elif upnpClass.text != track['class']:
                        self.logger.debug("Browsed item should match class")
                        failed_checks += 1

                tracks_checked += 1
                start_position += 1

        if failed_checks > 0:
            self.fail("Browsed Queue should match expected tracks (see debug log)")
            return False
        else:
            return True

    def _verify_traversal(self, zone, expected_track_list):
        """
        Verify the tracks in the Queue by seeking to each track.
        Check the CurrentTrack, EnqueuedTransport, and NextTrack information.
        """
        failed_checks = 0

        tracks_checked = 0
        while tracks_checked < len(expected_track_list):
            track = expected_track_list[tracks_checked]

            zone.AVTransport.set_queue_position(tracks_checked + 1)
            zone.AVTransport.wait_for_state_variables('CurrentTrack', lambda ct: ct == tracks_checked + 1,
                                                      iteration_delay=0.5, timeout_seconds=15)
            # NOTE: in some cases, it may take a bit to seek (network latency in requesting metadata to a service?)

            # When CurrentTrack is evented, the current/next URI should be ready, but be pedantic
            try:
                zone.AVTransport.wait_for_state_variables('CurrentTrackURI', lambda uri: uri == track['uri'])
                if tracks_checked + 1 < len(expected_track_list):
                    nexturi = expected_track_list[tracks_checked + 1]['uri']
                    zone.AVTransport.wait_for_state_variables('NextTrackURI', lambda uri: uri == nexturi)
            except sonos.exception.TimeoutError:
                pass  # don't care about timeout, as we'll be validating the variables.

            # log the expected and actual information as DEBUG, so it's always available, but doesn't clutter the logs
            self.logger.debug("%d: track: %s" % (tracks_checked + 1, track))
            self.logger.debug("    current: %s  --  %s" % (zone.AVTransport.get_state_variable('CurrentTrackURI'),
                                                           zone.AVTransport.get_state_variable('CurrentTrackMetaData')))
            self.logger.debug("    enqueued: %s -- %s" % (zone.AVTransport.get_state_variable('EnqueuedTransportURI'),
                                                          zone.AVTransport.get_state_variable('EnqueuedTransportURIMetaData')))
            self.logger.debug("    next: %s" % (zone.AVTransport.get_state_variable('NextTrackURI')))

            uri = zone.AVTransport.remove_account_sn_from_uri(zone.AVTransport.get_state_variable('CurrentTrackURI'))
            if track['uri'] != uri:
                self.logger.debug("Current track should match expected URI")
                failed_checks += 1

            if track['eq_uri'] != zone.AVTransport.get_state_variable('EnqueuedTransportURI'):
                self.logger.debug("Enqueued URI should match expected URI")
                failed_checks += 1

            # NOTE: Coherence returns UTF-8 encoded strings, so convert our 'unicode' input to UTF-8
            if track['eq_md'].encode('utf-8') != zone.AVTransport.get_state_variable('EnqueuedTransportURIMetaData'):
                self.logger.debug("Enqueued MD should match expected MD")
                failed_checks += 1

            if tracks_checked + 1 < len(expected_track_list):
                if expected_track_list[tracks_checked + 1]['uri'] != zone.AVTransport.get_state_variable('NextTrackURI'):
                    self.logger.debug("Next track should match expected URI")
                    failed_checks += 1
            else:
                if '' != zone.AVTransport.get_state_variable('NextTrackURI'):
                    self.logger.debug("Next track URI should be empty")
                    failed_checks += 1

            # NOTE: Coherence gives us UTF-8, so don't need to convert anything here
            root = ET.fromstring(zone.AVTransport.get_state_variable('CurrentTrackMetaData'))
            md_items = root.findall('{urn:schemas-upnp-org:metadata-1-0/DIDL-Lite/}item')
            if len(md_items) == 0:
                self.logger.debug("no CurrentTrackMetaData result?")
                failed_checks += 1
            else:
                item = md_items[0]

                # NOTE: the strings that come out of ET are 'unicode' (not UTF-8), so no
                # string encode/decode is needed here

                if track['uri'] is not None:
                    res = item.find("{urn:schemas-upnp-org:metadata-1-0/DIDL-Lite/}res")
                    uri = zone.AVTransport.remove_account_sn_from_uri(res.text)
                    if res is None:
                        self.logger.debug("Metadata item should contain res element")
                        failed_checks += 1
                    elif uri != track['uri']:
                        self.logger.debug("Metadata item should match media URL")
                        failed_checks += 1

                if track['duration'] is not None:
                    res = item.find("{urn:schemas-upnp-org:metadata-1-0/DIDL-Lite/}res")
                    if res is None or res.get('duration') is None:
                        self.logger.debug("Browsed item should contain duration attribute")
                        failed_checks += 1
                    elif res.get('duration') != track['duration']:
                        self.logger.debug("Browsed item should match duration")
                        failed_checks += 1

                if track['aa_uri'] is not None and track['has_aa']:
                    albumArtURI = item.find("{urn:schemas-upnp-org:metadata-1-0/upnp/}albumArtURI")
                    if albumArtURI is None:
                        self.logger.debug("Metadata item should contain album art element")
                        failed_checks += 1
                    elif not albumArtURI.text.upper().startswith(track['aa_uri'].upper()):
                        self.logger.debug("Metadata item should have expected album art URL")
                        failed_checks += 1

                if track['title'] is not None:
                    title = item.find("{http://purl.org/dc/elements/1.1/}title")
                    if title is None:
                        self.logger.debug("Metadata item should contain title element")
                        failed_checks += 1
                    elif title.text != track['title']:
                        self.logger.debug("Metadata item should match title")
                        failed_checks += 1

                if track['album'] is not None:
                    album = item.find("{urn:schemas-upnp-org:metadata-1-0/upnp/}album")
                    if album is None:
                        self.logger.debug("Metadata item should contain album element")
                        failed_checks += 1
                    elif album.text != track['album']:
                        self.logger.debug("Metadata item should match album")
                        failed_checks += 1

                if track['artist'] is not None and track['has_browse_artist']:
                    creator = item.find("{http://purl.org/dc/elements/1.1/}creator")
                    if creator is None:
                        self.logger.debug("Metadata item should contain creator element")
                        failed_checks += 1
                    elif creator.text != track['artist']:
                        self.logger.debug("Metadata item should match artist")
                        failed_checks += 1

                if track['track'] is not None:
                    originalTrackNumber = item.find("{urn:schemas-upnp-org:metadata-1-0/upnp/}originalTrackNumber")
                    if originalTrackNumber is None:
                        self.logger.debug("Metadata item should contain originalTrackNumber element")
                        failed_checks += 1
                    elif originalTrackNumber.text != str(track['track']):
                        self.logger.debug("Metadata item should match track number")
                        failed_checks += 1

                if track['class'] is not None:
                    upnpClass = item.find("{urn:schemas-upnp-org:metadata-1-0/upnp/}class")
                    if upnpClass is None:
                        self.logger.debug("Metadata item should contain class element")
                        failed_checks += 1
                    elif upnpClass.text != track['class']:
                        self.logger.debug("Metadata item should match class")
                        failed_checks += 1

            tracks_checked += 1

        if failed_checks > 0:
            self.fail("Traversed Queue should match expected tracks (see debug log)")
            return False
        else:
            return True
