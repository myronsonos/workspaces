    def _get_first_track_position(self, start_position, initial_queue_count):
        """
        Get the 'first track' for verifying the queue contents after
        adding to queue.

        :param start_position: Starting position for insertion into a queue
        :type start_position: :obj:`int`
        :param initial_queue_count: Number of tracks in the queue
        :type initial_queue_count: :obj:`int`

        :return: First track to check when verifying queue contents
        :rtype: :obj:`int`
        """
        first_track_should_be = initial_queue_count
        if start_position == 0 and first_track_should_be:
            first_track_should_be += 1  # there are tracks, so at the end of whats there
        if start_position > 0 and start_position <= initial_queue_count + 1:
            first_track_should_be = start_position  # insert into whats there
        if first_track_should_be == 0:
            first_track_should_be = 1  # no tracks at all, beginning

        return first_track_should_be


[docs]    def verify_tracks_in_queue(self, zp, expected_track_list, start_position=1):
        """
        Helper method for validating tracks in queue, 'start_position'
        parameter is 1-based.

        :param zp: SonosZoneComponent
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`
        :param expected_track_list: Expected list of tracks, in order
        :type expected_track_list: :obj:`list`
        :param start_position: Verify from this position in the queue/track list to the end
        :type start_position: :obj:`int`
        """
        # since Browse may return less than the total results, we may need more calls
        tracks_checked = 0
        while tracks_checked < len(expected_track_list):
            result = zp.ContentDirectory.Browse(object_id="Q:{}".format(self.queue_id),
                                                browse_flag="BrowseDirectChildren",
                                                filter="",
                                                starting_index=start_position - 1, # 0-based!
                                                count=len(expected_track_list) - tracks_checked,
                                                convert_xml=False)

            md_items = DIDLLite.DIDLElement.fromString(result['Result']).getItems()
            self.verifyTrueOrStop(len(md_items) != 0, "no results?! pos={}, tc={}, exp={}".
                 format(start_position, tracks_checked, len(expected_track_list)))

            for item in md_items:
                track = expected_track_list[tracks_checked]
                self.logger.debug("{}/{}: track: {} -- item: {}".format(
                    tracks_checked, start_position, track, item.toString()))

                # NOTE:
                #   Prior to ~2013-01-09, a ZP Browse() will return the data in ExtraMd, which is NOT
                #   what is sent as EnqueuedMD.  This means that the 'title' returned will likely be the filename
                #   and the 'albumArtURI' will be ZP-proxied (/getaa?u=...)
                #
                #   With the Browse() changes to support using the EnqueuedMD, the data returned should be the
                #   same as provided (except for item@id and item@parentID, which get replaced; the originals are
                #   stored in item@origID and item@origParentID).

                self.verifyTrueOrFailCase(hasattr(item, "title") and
                    item.title == track.title,
                    "Browsed complete md should have matching title value")
                self.verifyTrueOrFailCase(hasattr(item, "album") and
                    item.album == track.album,
                    "Browsed complete md should have matching album value")
                self.verifyTrueOrFailCase(hasattr(item, "creator") and
                    item.creator == track.artist.decode('utf-8'),
                    "Browsed complete md should have matching artist value")

                tracks_checked += 1
                start_position += 1



[docs]    def generate_generic_md(self, zp, http_track):
        """
        Given a zp and a :class:`~sonos.websrv.TrackMetadataTuple`, generate
        generic metadata.

        :param zp: SZC to use to generate metadata
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`
        :param http_track: TrackMetadataTuple with track information
        :type http_track: :class:`~sonos.websrv.TrackMetadataTuple`

        :return: metadata string
        :rtype: :obj:`string`
        """
        return zp.AVTransport.generate_didl_lite(title=http_track.title,
               svc_id=None, acct_id=None,
               upnp_class="object.item.audioItem.musicTrack",
               artist=http_track.artist, album=http_track.album,
               aa_uri=http_track.artwork_url)



[docs]    def add_http_tracks_to_queue(self,
                                 zp,
                                 http_tracks,
                                 start_position=0,
                                 generate_md=True,
                                 as_next=False,
                                 legacy=False,
                                 verify=True):
        """
        Adds one or more :class:`~sonos.websrv.TrackMetadataTuple` objects
        to the queue. Generates generic metadata for each track if the
        keyword argument is specified.

        :param zp: SZC to add http track to queue
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`
        :param http_tracks: a single track metadata tuple instance, or a list
        :type http_tracks: :class:`~sonos.websrv.TrackMetadataTuple` or :obj:`list`
        :param generate_md: Flag specifying whether to generate metadata
        :type generate_md: :obj:`bool`
        :param start_position: track_num argument for add_to_queue
        :type start_position: :obj:`int`
        :param as_next: as_next argument for add_to_queue
        :type as_next: :obj:`bool`
        :param legacy: Never use AddMultipleURIs if True
        :type legacy: :obj:`bool`
        :param verify: Verify tracks are in expected position after adding
        :type verify: :obj:`bool`

        :return: Result of zp.AVTransport.add_to_queue (number of tracks added)
        :rtype: :obj:`int`
        """

        initial_queue_count = zp.AVTransport.get_number_of_tracks_in_queue(instance_id=self.queue_id)
        first_track = self._get_first_track_position(start_position, initial_queue_count)

        if (isinstance(http_tracks, (list, tuple)) and
            not isinstance(http_tracks, TrackMetadataTuple)):
            if generate_md:
                md = [self.generate_generic_md(zp, t) for t in http_tracks]
            uris = [t.media_url for t in http_tracks]
        else:
            if generate_md:
                md = self.generate_generic_md(zp, http_tracks)
            uris = http_tracks.media_url

        if generate_md:
            num_tracks_added = zp.AVTransport.add_to_queue(uri=uris,
                                        metadata=md,
                                        track_num=start_position,
                                        instance_id=self.queue_id,
                                        as_next=as_next,
                                        legacy=legacy)
        else:
            num_tracks_added = zp.AVTransport.add_to_queue(uri=uris,
                                        track_num=start_position,
                                        instance_id=self.queue_id,
                                        as_next=as_next,
                                        legacy=legacy)

        total_tracks_expected = initial_queue_count + num_tracks_added

        self.verifyEqualOrFailCase(total_tracks_expected, zp.AVTransport.
                                   get_number_of_tracks_in_queue(instance_id=self.queue_id),
                                   "Number of tracks in Sonos Queue should match number added.")

        if verify:
            self.verify_tracks_in_queue(zp,
                [http_tracks] if isinstance(http_tracks, TrackMetadataTuple) else http_tracks,
                start_position=first_track)

        return num_tracks_added



[docs]    def add_http_tracks_to_queue_batched(self,
                                         zp,
                                         track_list,
                                         batch_count,
                                         start_position=0,
                                         update_id=None,
                                         as_next=False,
                                         verify=True):
        """
        Adds one or more :class:`~sonos.websrv.TrackMetadataTuple` objects
        to the queue (quick adding batched). Generates metadata for each
        track.

        :param zp: SZC to add http track to queue
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`
        :param track_list: a single track metadata tuple instance, or a list
        :type track_list: :class:`~sonos.websrv.TrackMetadataTuple` or :obj:`list`
        :param batch_count: Number of tracks to include in each batch.
        :type batch_count: :obj:`int`
        :param start_position: track_num argument for add_to_queue
        :type start_position: :obj:`int`
        :param update_id: update id to use, if None, will use 0 for first update
        ID, then the NewUpdateID returned by the AddMultipleURIs call for each
        successive update id. Otherwise, it will use the value of update_id
        for each call.
        :type update_id: :obj:`int` or None
        :param as_next: as_next argument for add_to_queue
        :type as_next: :obj:`bool`
        :param verify: Verify tracks are in expected position after adding
        :type verify: :obj:`bool`

        :return: Result of zp.AVTransport.add_to_queue (number of tracks added)
        :rtype: :obj:`int`
        """
        initial_queue_count = zp.AVTransport.get_number_of_tracks_in_queue()
        first_track = self._get_first_track_position(start_position, initial_queue_count)

        if batch_count > 16:
            self.logger.warn("batch count {} is too large, forcing to 16".
                             format(batch_count))
            batch_count = 16

        self.logger.info("Started adding {} tracks in batches of {}"
                         " to Sonos Queue...".format(len(track_list),
                         batch_count))

        num_tracks_added = 0
        while num_tracks_added < len(track_list):
            subset_count = 0
            batched_uris = []
            batched_metadata = []

            while subset_count < batch_count:
                if num_tracks_added + subset_count >= len(track_list):
                    break
                track = track_list[num_tracks_added + subset_count]

                # generate valid DIDL metadata
                md = self.generate_generic_md(zp, track)

                # make sure this track fits in the batching
                if len(' '.join(batched_uris + [track.media_url])) > BATCHING_BYTE_LIMIT:
                    self.logger.warn("could only fit {} tracks into this "
                                     " batch because of URI".
                                     format(subset_count))
                    break
                if len(' '.join(batched_metadata + [md])) > BATCHING_BYTE_LIMIT:
                    self.logger.warn("could only fit {} tracks into this "
                                     "batch because of metadata".
                                     format(subset_count))
                    break

                batched_uris.append(track.media_url)
                batched_metadata.append(md)
                subset_count += 1

            if subset_count == 0:
                self.logger.warn("subset count is 0!?")
                break

            temp_updateid = update_id
            if update_id is None:
                temp_updateid = 0

            # We have to wrap AddMultipleURIsToQueue instead of using add_to_queue.
            # add_to_queue currently does not give us the NewUpdateID and there is
            # no reliable way to retrieve it. Usually it is just the current update ID
            # + 1, but this is not always the case.
            response = zp.AVTransport.AddMultipleURIsToQueue(num_uris=subset_count,
                                                             uris=' '.join(batched_uris),
                                                             metadata=' '.join(batched_metadata),
                                                             container_uri="",
                                                             container_metadata="",
                                                             track_num=start_position,
                                                             as_next=as_next,
                                                             update_id=temp_updateid,
                                                             instance_id=self.queue_id)

            if update_id is not None:
                update_id = response['NewUpdateID']

            num_tracks_added += subset_count

            if start_position:
                start_position += subset_count

        total_tracks_expected = initial_queue_count + num_tracks_added

        self.verifyEqualOrFailCase(total_tracks_expected, zp.AVTransport.
                                   get_number_of_tracks_in_queue(instance_id=self.queue_id),
                                   "Number of tracks in Sonos Queue should match number added.")

        if verify:
            self.verify_tracks_in_queue(zp, track_list, start_position=first_track)

        return num_tracks_added
