[docs]    def test_combinatorial_remove_track_from_queue(self, generate_testbed_unique_playback_devices):
        """
        Verify that removing a single track from the beginning, middle and end of the queue
        results in the expected tracks remaining in the list.
        """
        zone = generate_testbed_unique_playback_devices
        if len(self.track_source.tracks) <= 3:
            self.stop("Needs 4 or more tracks ({}) to work with".
                      format(len(self.track_source.tracks)))

        zone.AVTransport.use_queue(queue_id=self.queue_id)
        zone.AVTransport.SetPlayMode(zone.AVTransport.PLAYMODE_NORMAL)
        zone.AVTransport.clear_queue_and_wait(instance_id=self.queue_id)

        track_list = []
        while len(track_list) < 5:
            track_list.append(random.choice(self.track_source.tracks[0:2]))

        self.add_http_tracks_to_queue(zone, track_list)

        # remove from the end -- difference between Q:X:5 and using the
        # instance_id argument to specify queue?
        zone.AVTransport.RemoveTrackFromQueue("Q:{}:5".format(self.queue_id))

        # remove from the middle
        zone.AVTransport.RemoveTrackFromQueue("Q:{}:3".format(self.queue_id))

        # remove from the beginning
        zone.AVTransport.RemoveTrackFromQueue("Q:{}:1".format(self.queue_id))

        expectedinqueue = track_list[1], track_list[3]

        self.verifyEqualOrFailCase(2, zone.AVTransport.
            get_number_of_tracks_in_queue(instance_id=self.queue_id),
            "Queue should only contain 2 items")

        self.verify_tracks_in_queue(zone, expectedinqueue)

[docs]    def test_combinatorial_remove_track_from_queue_same_update_id(self, generate_testbed_unique_playback_devices):
        """
        Verify multiple remove track from queue requests using the same update id causes the second request to fail
        Expect: UPnP Error 1028, overall test should pass
        """

        zone = generate_testbed_unique_playback_devices
        if len(self.track_source.tracks) <= 3:
            self.stop("Needs 4 or more tracks ({}) to work with".
                      format(len(self.track_source.tracks)))

        zone.AVTransport.use_queue(queue_id=self.queue_id)
        zone.AVTransport.SetPlayMode(zone.AVTransport.PLAYMODE_NORMAL)
        zone.AVTransport.clear_queue_and_wait(instance_id=self.queue_id)

        track_list = []
        while len(track_list) < 10:
            track_list.append(random.choice(self.track_source.tracks[0:4]))

        self.add_http_tracks_to_queue(zone, track_list)

        wait_until_true(lambda: zone.AVTransport.
            get_number_of_tracks_in_queue(instance_id=self.queue_id) == 10)
        time.sleep(1)
        updateid = zone.ContentDirectory.Browse("Q:{}".format(self.queue_id),
                                                "BrowseMetadata", "", 0, 0).get("UpdateID")
        # attempt to process 2 back to back queue removals using the same updateid. The second
        # removal should fail with a upnp 1028 error
        zone.AVTransport.RemoveTrackFromQueue("Q:{}:5".format(self.queue_id), 0, updateid)
        try:
            zone.AVTransport.RemoveTrackFromQueue("Q:{}:5".format(self.queue_id), 0, updateid)
        except sonos.exception.UPnPError, e:
            self.logger.info("Second RemoveTrackFromQueue returned: {}".format(e))

        self.verifyEqualOrFailCase(9, zone.AVTransport.
                get_number_of_tracks_in_queue(instance_id=self.queue_id),
                "Queue should contain 9 items, only one deletion should have succeeded")

            get_number_of_tracks_in_queue(instance_id=self.queue_id) == 10)
