Tests to cover the 'pause at end' queue policy. This causes the player to remain at the end of the last track in the

    def _set_up_end_of_queue(self, zone):
        zone.AVTransport.use_queue(self.queue_ids[zone.ip])
        helpers.generate_metadata_and_add_tracks_to_queue(zone, self.queue_ids[zone.ip], self.track_list)
        zone.AVTransport.set_queue_position(5)
        zone.AVTransport.Seek('REL_TIME', "0:00:09")
        wait_until_true(lambda: 'Play' in zone.AVTransport.get_current_transport_actions(),
                        timeout_seconds=5)
        zone.AVTransport.Play()
        try:
            wait_until_true(lambda: zone.AVTransport.is_paused(),
                            timeout_seconds=5)
        except sonos.exception.TimeoutError:
            self.fail("END OF QUEUE SETUP FAIL: Player did not enter paused state as expected")


[docs]    @combinatorial('generate_testbed_unique_playback_devices')
    def test_last_track_deleted_while_playing_second_to_last_track(self, zone):
        """
        While playing the second to last track in queue verify the player pauses at the end of playback once
        the last track in the queue is removed. Verifies no regressions for PTS-2119
        :param zone: Unique Sonos zone component
        :type zone: :class:`-sonos.client.internal.SonosZoneComponent`
        """
        helpers.generate_metadata_and_add_tracks_to_queue(zone, self.queue_ids[zone.ip], self.track_list)
        zone.AVTransport.use_queue(queue_id=self.queue_ids[zone.ip])
        zone.AVTransport.Play_and_wait()
        zone.AVTransport.Seek("TRACK_NR", 4)
        zone.AVTransport.Seek("REL_TIME", "0:02:50")
        zone.Queue.RemoveTrackRange(queueid=self.queue_ids[zone.ip],
                                    updateid=0,
                                    startingindex=5,
                                    numberoftracks=1)
        try:
            wait_until_true(zone.AVTransport.is_paused, timeout_seconds=10)
            self.verifyEqualOrFailCase(4, zone.AVTransport.get_queue_position(),
                                       "Player should be at queue position 4")
            self.verifyEqualOrFailCase("0:02:56", str(zone.AVTransport.GetPositionInfo()["RelTime"]),
                                       "Playhead should be at the end of the track")
        except sonos.exception.TimeoutError:
            self.fail("Player did not enter the paused state when expected")



[docs]    @combinatorial('generate_testbed_unique_playback_devices')
    def test_delegation_when_paused_at_end(self, zone):
        """
        Verify you can delegate the currently paused item to a new GC and then resume playback on that new
        player. Regression test for PTS-2117
        :param zone: Unique Sonos zone component
        :type zone: :class:`-sonos.client.internal.SonosZoneComponent`
        """
        self._set_up_end_of_queue(zone)
        new_gc = filter(lambda zp: (zp.is_playback_device()
                                    and not zp.AVTransport.is_group_member())
                        and (zp.raw_udn != zone.raw_udn),
                       self.my_devices)[0]

        new_gc.AVTransport.join_group(zone)
        zone.AVTransport.BecomeCoordinatorOfStandaloneGroup()
        self.verifyTrueOrFailCase(new_gc.AVTransport.is_paused,
                                  "New GC should remain in the paused state")
        self.verifyEqualOrFailCase(5, new_gc.AVTransport.get_queue_position(),
                                   "New GC should be at the same queue position")
        self.verifyEqualOrFailCase("0:00:09", str(new_gc.AVTransport.GetPositionInfo()["RelTime"]),
                                   "Playhead should still be at the end of the track")
        try:
            new_gc.AVTransport.Play_and_wait()
            self.verifyEqualOrFailCase(1, new_gc.AVTransport.get_queue_position(),
                                       "New GC should be playing at queue position 1")
        except sonos.exception.TimeoutError:
            self.fail("New GC did not enter playing state")

    def test_private_queue_without_pauseAtEnd_policy(self, zone):
        """
        Verify the player behavior when sending a Seek(TRACK_NR) command after adding a new track to the end
        of the queue.
        'Z' added to name to ensure this test runs last in the suite since it will override the
        private queue used in all other tests.
        :param zone: Unique Sonos zone component
        :type zone: :class:`-sonos.client.internal.SonosZoneComponent`
        """
        zone = filter(lambda zp: zp.is_playback_device() and not zp.AVTransport.is_group_member(), self.my_devices)[0]
        new_queue = zone.Queue.CreateQueue(queueownerid=zone.ip, queueownercontext=zone.modelName)
        helpers.generate_metadata_and_add_tracks_to_queue(zone, new_queue, self.track_list)
        zone.AVTransport.use_queue(new_queue)
        zone.AVTransport.set_queue_position(5)
        zone.AVTransport.Seek('REL_TIME', "0:02:49")
        zone.AVTransport.Play()
        try:
            wait_until_true(lambda: zone.AVTransport.is_stopped())
        except sonos.exception.TimeoutError:
            self.fail("player did not enter the stopped state as expected")

        try:
            wait_until_true(lambda: zone.AVTransport.get_queue_position() == 1,
                            timeout_seconds=2)
        except sonos.exception.TimeoutError:
            self.fail("Playhead did not return to the first track in the queue as expected")
