[docs]def wait_until_true_or_continue(predicate):
    """
    Calls :func:`~sonos.services.common.wait_until_true` and ignores the
    :exc:`~sonos.exception.TimeoutException` is raises so assertions can be
    performed on failures

    :param predicate: the predicate to call when waiting until true
    :type predicate: :class:`method <types.MethodType>` or
        :class:`function <types.FunctionType>`
    """
    try:
        wait_until_true(predicate)
    except TimeoutError:
        pass

        # Override super
        for zp in self.playback_zones:

                        "AVTransportURI to be reset on {}".format(
                            zp.modelNumber))
            if zp.AVTransport.get_number_of_tracks_in_queue(
                    instance_id=self.queue_id) > 0:
                self._cleanup_zone(zp)
            if zp.AVTransport.get_current_play_mode() != 'NORMAL':
                zp.AVTransport.SetPlayMode('NORMAL')

    def _create_test_playlist(self, zp):
        if not getattr(self, 'playlist_id', None):
            [zp.AVTransport.add_to_queue(uri, instance_id=self.queue_id) for uri
                in self.selected_tracks]
            wait_until_true(
                lambda: zp.AVTransport.get_number_of_tracks_in_queue(
                    instance_id=self.queue_id) == len(self.selected_tracks),
                reason="Timed out attempting to create a playlist on "
                    "<{}>".format(zp))
            self.playlist_id = zp.AVTransport.SaveQueue('TEST')
            self.playlist_md = str(zp.ContentDirectory.Browse(self.playlist_id,
                'BrowseMetadata', 0, 0, 0).get('Result'))
            self._cleanup_zone(zp)
        return (self.playlist_id, self.playlist_md)

    # TODO: this is duplicated in a bunch of other places
    def _cleanup_zone(self, zp):
        try:
            zp.AVTransport.RemoveAllTracksFromQueue(instance_id=self.queue_id)
            wait_until_true(
                lambda: zp.AVTransport.get_number_of_tracks_in_queue(
                    instance_id=self.queue_id) == 0,
                reason="Timed out attempting to remove all tracks from "
                    "<{}>".format(zp))
        except UPnPError:
            self.logger.error("caught UPnP error {} calling "
                "AVTransport.RemoveAllTracksFromQueue on <{}>".format(
                    zp.last_upnp_error_code, zp))

        """
        Verify that a track can be added to the device queue
        """
        self.short_description = "test adding track to device queue"
        for zp in self.playback_zones:

                zp.AVTransport.get_number_of_tracks_in_queue(
                    instance_id=self.queue_id),
                "Verify track added to queue for <{}>".format(zp))
            self._cleanup_zone(zp)

[docs]    def test_replace_queue(self):
        """
        Test basic queue replace functionality.
        """
        self.short_description = ("test replacing contents of an existing "
            "device queue")
        for zp in self.playback_zones:
            # add tracks to queue
            [zp.AVTransport.add_to_queue(uri, instance_id=self.queue_id) for
                uri in self.selected_tracks]
            wait_until_true_or_continue(
                lambda: zp.AVTransport.get_number_of_tracks_in_queue(
                    instance_id=self.queue_id) == len(self.selected_tracks))
            self.verifyEqualOrStop(len(self.selected_tracks),
                zp.AVTransport.get_number_of_tracks_in_queue(
                    instance_id=self.queue_id),
                "Verify add initial test tracks to queue on <{}>".format(zp))
            # remove all tracks
            zp.AVTransport.RemoveAllTracksFromQueue(instance_id=self.queue_id)
            wait_until_true_or_continue(
                lambda: zp.AVTransport.get_number_of_tracks_in_queue(
                    instance_id=self.queue_id) == 0)
            self.verifyEqualOrStop(0,
                zp.AVTransport.get_number_of_tracks_in_queue(
                    instance_id=self.queue_id),
                "Verify remove test tracks from queue on <{}>".format(zp))
            # add tracks to queue
            zp.AVTransport.add_to_queue(self.track_source.tracks[0].media_url, instance_id=self.queue_id)
            wait_until_true_or_continue(
                lambda: zp.AVTransport.get_number_of_tracks_in_queue(
                    instance_id=self.queue_id) == 1)
            self.verifyEqualOrFailCase(1,
                zp.AVTransport.get_number_of_tracks_in_queue(
                    instance_id=self.queue_id),
                "Verify add test track to queue after removing initial "
                "tracks on <{}>".format(zp))
            self._cleanup_zone(zp)



[docs]    def test_reorder_tracks(self):
        """
        Test basic queue re-ordering
        """
        self.short_description = "test reordering tracks in a device queue"
        for zp in self.playback_zones:
            [zp.AVTransport.add_to_queue(uri) for uri in self.selected_tracks]
            num_tracks = len(self.selected_tracks)
            self.verifyEqualOrStop(num_tracks,
                zp.AVTransport.get_number_of_tracks_in_queue(
                    instance_id=self.queue_id),
                "Verify expected number of test tracks on <{}>".format(zp))
            zp.AVTransport.use_queue(queue_id=self.queue_id)
            # wait_until_true(lambda: zp.raw_udn in zp.AVTransport.get_avtransport_uri())
            wait_until_true(
                lambda: zp.raw_udn in zp.AVTransport.GetMediaInfo().get(
                    'CurrentURI'))
            # move last track to first
            test_track = zp.ContentDirectory.get_queue(
                instance_id=self.queue_id)[num_tracks - 1].res.value
            zp.AVTransport.ReorderTracksInQueue(num_tracks, 1, 1,
                instance_id=self.queue_id)
            moved_track = zp.ContentDirectory.get_queue(
                instance_id=self.queue_id)[0].res.value
            self.verifyEqualOrFailCase(test_track,
                moved_track,
                "Verify last track moves to first position")
            # move last track to middle
            test_track = zp.ContentDirectory.get_queue(
                instance_id=self.queue_id)[num_tracks - 1].res.value
            zp.AVTransport.ReorderTracksInQueue(num_tracks, 1, 2,
                instance_id=self.queue_id)
            moved_track = zp.ContentDirectory.get_queue(
                instance_id=self.queue_id)[1].res.value
            self.verifyEqualOrFailCase(test_track,
                moved_track,
                "Verify last track moves to middle position")
            # move first track to last
            test_track = zp.ContentDirectory.get_queue(
                instance_id=self.queue_id)[0].res.value
            zp.AVTransport.ReorderTracksInQueue(1, 1, num_tracks + 1,
                                                instance_id=self.queue_id)
            moved_track = zp.ContentDirectory.get_queue(
                instance_id=self.queue_id)[num_tracks - 1].res.value
            self.verifyEqualOrFailCase(test_track,
                moved_track,
                "Verify first track moves to last position")
            # cleanup
            self._cleanup_zone(zp)

        """
        Test that we can remove a track from a device queue
        """
        self.short_description = ("test removing a single track from device "
            "queue")
        for zp in self.playback_zones:

            # remove track
            zp.AVTransport.RemoveTrackFromQueue("Q:{}/{}".format(self.queue_id,
                zp.AVTransport.get_number_of_tracks_in_queue(
                    instance_id=self.queue_id)))
            wait_until_true_or_continue(
                lambda: zp.AVTransport.get_number_of_tracks_in_queue(
                    instance_id=self.queue_id) == 0)
            self.verifyEqualOrFailCase(0,
                zp.AVTransport.get_number_of_tracks_in_queue(
                    instance_id=self.queue_id),
                "Verify RemoveTrackFromQueue causes track to be removed from "
                "queue on <{}>".format(zp))
            self._cleanup_zone(zp)

        """
        test basic functionality of AVTransport.RemoveTrackRangeFromQueue
        """
        total_tracks = 10
        remove_tracks = total_tracks / 2
        self.short_description = "Test removing track ranges from device queue"
        for zp in self.playback_zones:

                    instance_id=self.queue_id) == total_tracks)
            self.verifyEqualOrStop(total_tracks,
                zp.AVTransport.get_number_of_tracks_in_queue(
                    instance_id=self.queue_id),
                "Verify tracks were added to queue on <{}>".format(zp))
            # remove half and verify
            zp.AVTransport.RemoveTrackRangeFromQueue(remove_tracks,
                remove_tracks, instance_id=self.queue_id)
            wait_until_true_or_continue(
                lambda: zp.AVTransport.get_number_of_tracks_in_queue(
                    instance_id=self.queue_id) == remove_tracks)
            self.verifyEqualOrFailCase(remove_tracks,
                zp.AVTransport.get_number_of_tracks_in_queue(
                    instance_id=self.queue_id),
                "Verify removal of multiple tracks to queue on <{}>".format(zp))
            self._cleanup_zone(zp)
