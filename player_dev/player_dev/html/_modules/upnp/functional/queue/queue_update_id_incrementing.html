[docs]    def test_combinatorial_add_to_queue_update_id_change(self, generate_testbed_unique_playback_devices):
        """
        Verify the add to queue actions cause the QueueUpdateID to increment.
        """
        
        zone = generate_testbed_unique_playback_devices
        if len(self.track_source.tracks) <= 3:
            self.stop("Needs 4 or more tracks {} to work with".
                      format(len(self.track_source.tracks)))

        track_list = []
        while len(track_list) < 15:
            track_list.append(random.choice(self.track_source.tracks[0:3]))

        zone.AVTransport.use_queue(queue_id=self.queue_id)
        zone.AVTransport.SetPlayMode(zone.AVTransport.PLAYMODE_NORMAL)
        zone.AVTransport.clear_queue_and_wait(instance_id=self.queue_id)
        
        originupdateid = zone.ContentDirectory.Browse("Q:{}".
            format(self.queue_id), "BrowseMetadata", "", 0, 0)["UpdateID"]

        self.add_http_tracks_to_queue(zone, track_list)

        updateid2 = zone.ContentDirectory.Browse("Q:{}".
            format(self.queue_id), "BrowseMetadata", "", 0, 0)["UpdateID"]

        self.verifyNotEqualOrFailCase(originupdateid, updateid2, 
            "AddURIToQueue should cause the ContainerUpdateIds value to increment")

        self.add_http_tracks_to_queue_batched(zone, track_list, 15)
        updateid3 = zone.ContentDirectory.Browse("Q:{}".
            format(self.queue_id), "BrowseMetadata", "", 0, 0)["UpdateID"]

        self.verifyNotEqualOrFailCase(updateid2, updateid3, 
            "AddMultipleURIsToQueue should cause the ContainerUpdateIds value to increment")

[docs]    def test_combinatorial_bad_update_id_passed_in(self, generate_testbed_unique_playback_devices):
        """
        Verify passing in a stale QueueUpdateId causes the action to fail for each UPnP queue action that knows about updateid
        Tests: AddMultipleURIsToQueue, RemoveTrackFromQueue, RemoveTrackRangeFromQueue, ReorderTracksInQueue
        DOES NOT TEST: AddURIToQueue, RemoveAllTracksFromQueue
        """
        
        zone = generate_testbed_unique_playback_devices
        if len(self.track_source.tracks) <= 3:
            self.stop("Needs 4 or more tracks {} to work with".
                      format(len(self.track_source.tracks)))
            
        zone.AVTransport.use_queue(queue_id=self.queue_id)
        zone.AVTransport.SetPlayMode(zone.AVTransport.PLAYMODE_NORMAL)
        zone.AVTransport.clear_queue_and_wait(instance_id=self.queue_id)
        
        track_list = []
        while len(track_list) < 15:
            track_list.append(random.choice(self.track_source.tracks[0:3]))

        # Get the correct updateid
        originupdateid = zone.ContentDirectory.Browse("Q:{}".
            format(self.queue_id), "BrowseMetadata", "", 0, 0)["UpdateID"]
        
        self.add_http_tracks_to_queue(zone, track_list)
        
        # Make the updateid wildly different
        updateid2 = originupdateid * 2
        try:
            zone.AVTransport.RemoveTrackFromQueue("Q:{}:4".
                format(self.queue_id), 0, updateid2)
        except sonos.exception.UPnPError as e:
            self.verifyEqualOrFailCase(
                sonos.client.zone_player.UPNP_ERROR_1028_STALE_OBJECT,
                e.upnp_error,
                "RemoveTrackFromQueue with bad updateid should return 1028(stale object) error")
        else:
            self.fail("RemoveTrackFromQueue with bad update id should "
                      "raise an exception!")

        # Skew updateid again
        updateid3 = updateid2*2
        
        try:
            zone.AVTransport.RemoveTrackRangeFromQueue(5, 4, self.queue_id, updateid3)
        except sonos.exception.UPnPError as e:
            self.verifyEqualOrFailCase(
                sonos.client.zone_player.UPNP_ERROR_1028_STALE_OBJECT,
                e.upnp_error,
                "RemoveTrackRangeFromQueue with bad updateid should return 1028(stale object) error")
        else:
            self.fail("RemoveTrackRangeFromQueue with bad update id should "
                      "raise an exception!")
        
        # Skew updateid again
        updateid4 = updateid3*2

        try:
            zone.AVTransport.ReorderTracksInQueue(6, 5, 12, self.queue_id, updateid3)
        except sonos.exception.UPnPError as e:
            self.verifyEqualOrFailCase(1028, e.upnp_error,
                "ReorderTracksInQueue with bad updateid should return 1028(stale object) error")
        else:
            self.fail("ReorderTracksInQueue with bad update id should "
                      "raise an exception!")
        
        # Skew updateid again
        updateid5 = updateid4*2
        
        # Process some batch metadata to use AddMultipleURIsToQueue
        # taken from helpers._add_http_tracks_to_queue_batched()
        BATCHING_BYTE_LIMIT = 10 * 1024
        num_tracks_added = 0
        while num_tracks_added < len(track_list):
            subset_count = 0
            batched_uris = []
            batched_metadata = []

            while subset_count < 15:
                if num_tracks_added + subset_count >= len(track_list):
                    break
                track = track_list[num_tracks_added + subset_count]

                # generate valid DIDL metadata
                md = zone.AVTransport.generate_didl_lite(title=track.title,
                                                         svc_id=None,
                                                         acct_id=None,
                                                         upnp_class="object.item.audioItem.musicTrack",
                                                         artist=track.artist,
                                                         album=track.album,
                                                         aa_uri=track.artwork_url)

                # make sure this track fits in the batching
                if len(' '.join(batched_uris + [track.media_url])) > BATCHING_BYTE_LIMIT:
                    self.logger.warn("could only fit {} tracks into this batch because of URI".format(subset_count))
                    break
                if len(' '.join(batched_metadata + [md])) > BATCHING_BYTE_LIMIT:
                    self.logger.warn("could only fit {} tracks into this batch because of metadata".format(subset_count))
                    break

                batched_uris.append(track.media_url)
                batched_metadata.append(md)
                subset_count += 1

            if subset_count == 0:
                self.logger.warn("subset count is 0!?")
                break
            else:
                break
            
        try:
            zone.AVTransport.AddMultipleURIsToQueue(num_uris=subset_count,
                                                    uris=' '.join(batched_uris),
                                                    metadata=' '.join(batched_metadata),
                                                    container_uri="",
                                                    container_metadata="",
                                                    track_num=2,
                                                    as_next=False,
                                                    update_id=updateid5,
                                                    instance_id=self.queue_id)
        except sonos.exception.UPnPError as e:
            self.verifyEqualOrFailCase(
                sonos.client.zone_player.UPNP_ERROR_1028_STALE_OBJECT,
                e.upnp_error,
                "AddMultipleURIsToQueue with bad updateid should return 1028(stale object) error")
        else:
            self.fail("AddMultipleURIsToQueue with bad update id should "
                      "raise an exception!")
