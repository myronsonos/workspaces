[docs]    def load_img(self):
        """

        server and attaches the new image.

        self.srv = upnp.functional.indexing.sls.SonosLibraryServer(
            self.cfg, self.build)
        self.srv.set_paramiko_log_level(logging.ERROR)
        return self.srv.setup_img()



[docs]    def get_directory_content(self, path):
        """
        Given a URL, it returns the HTTP response and a dictionary
        of elements containing the file name(s) and accompanying size(s).

        :param str path: URL of directory
        :return: a tuple of dictionary representing the file name(s)
            with respective size(s) in the specified URL and the HTTP
            response
        :rtype: :obj:`tuple`
        """
        response = self.srv.send_http_request_with_auth(path)
        self.verifyEqualOrFailCase(HTTPStatusCodes.ok,
                                   response.status_code,
                                   "Expected HTTP server response")
        files = XMLAccessor(response.content.replace('\n', '')).get('file')
        # Create a dictionary of files with respective sizes
        # File name: File size
        d = {}
        for f in files:
            d[f.value] = f.size
        return (d, response)

[docs]    def test_album_art_positive(self):
        """
        Access the album art directory and confirm the
        EXPECTED_ALBUM_JPG_FILE can be retrieved.
        Verify EXPECTED_ALBUM_JPG_FILE length on retrieval.
        """
        self.logger.info("Test album art JPG")
        (d, response) = self.get_directory_content(self.ALBUM_ART_DIR)
        exp_size = d[self.EXPECTED_ALBUM_JPG_FILE]
        response = self.srv.send_http_request_with_auth(
            self.ALBUM_ART_DIR +
            '/' +
            self.EXPECTED_ALBUM_JPG_FILE)
        self.verifyEqualOrFailCase(HTTPStatusCodes.ok,
                                   response.status_code,
                                   "Expected HTTP server response")
        self.verifyEqualOrFailCase(exp_size,
                                   response.headers['content-length'],
                                   "Expected size {}".format(
                self.EXPECTED_ALBUM_JPG_FILE))

        """
        Access a directory containing Chinese title tracks and confirm
        the server responds with UTF-8 encoding.
        """
        self.logger.info("Test UTF-8 encoding")
        (d, response) = self.get_directory_content(self.CHINESE_DIR)
        titles = list(d)
        res = all([isinstance(title, unicode) for title in titles])
        self.verifyTrueOrFailCase(res,
                                  "Expected all titles with UTF-8 encoding")



[docs]    def test_65k_dir(self):
        """
        Access a directory containing 65K files.
        Ensure "chunked" HTTP transfer-encoding.
        Verify 65K files returned.
        """
        self.logger.info("Test 65,000 file content directory")
        (d, response) = self.get_directory_content(self.MANY_FILES_DIR)
        self.verifyEqualOrFailCase(HTTPStatusCodes.ok,
                                   response.status_code,
                                   "Expected HTTP server response")
        self.verifyEqualOrFailCase('chunked',
                                   response.headers['transfer-encoding'],
                                   "Expected chunked encoding")
        self.verifyEqualOrFailCase(self.NUM_FILES_IN_65K_DIR,
                                   len(d),
                                   "Expected {} files".format(
                self.NUM_FILES_IN_65K_DIR))
