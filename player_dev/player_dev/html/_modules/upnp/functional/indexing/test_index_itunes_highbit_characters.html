[docs]    def test_highbit_characters_itunes_playback(self):
        """
        Verify tracks in Itunes playlists with highbit
        characters can still perform playback on
        new MAC file sharing.

        :param str platform: OS platform to index from

        """
        uri = self.zp.ContentDirectory.get_itunes_playlist_uri(PLAYLIST_NAME)
        try:
            self.zp.AVTransport.add_to_queue(uri)
        except UPnPError as e:
            self.fail('Caught <{}>'.format(e))
        else:
            self.zp.AVTransport.use_queue()
            self.zp.AVTransport.Play_and_wait()
            self.zp.AVTransport.Stop_and_wait()
