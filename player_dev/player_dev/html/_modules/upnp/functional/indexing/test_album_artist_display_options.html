    def _refresh_share_index_and_verify_expected_artists(self,
                                                         option,
                                                         object_id,
                                                         expected_artists):
        """
        Refresh the share Index with an AADO (Album artists display option)
        and verify that a browse will return the expected number of
        expected artists in the share.

        :param str option: album artists display option
        :param str object_id: ID of objects to browse in share
        :param expected_artists: the number of expected artists returned from
            the browse
        :type expected_artists: :obj:`unicode`

        """
        res = self.zp.ContentDirectory.RefreshShareIndex(option)
        self.verifyTrueOrStop(res, "Expect to re-index the share with the ADDO {}".format(option))
        res = self.zp.ContentDirectory.GetAlbumArtistDisplayOption()
        self.verifyEqualOrStop(unicode(option),
                                   res, '{} should be {}'
                                   .format(res, unicode(option)))
        album_artists = self.zp.ContentDirectory.Browse(object_id,
                                                        "BrowseDirectChildren",
                                                        0, 0, 0)
        total_matches = album_artists['TotalMatches']
        self.verifyEqualOrFailCase(expected_artists,
                                   total_matches,
                                   '{} should be {}'
                                   .format(expected_artists, total_matches))
