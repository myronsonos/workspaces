    def test_file_permissions(self, remfile, path, perm, exp_code):
        """
        Confirms the responses from the HTTP server on files which
        lack permissions.
        :param str remfile: remote file on the HTTP server
        :param str path: path to the remote file
        :param str perm: permssions to set on remote file
        :param int exp_code: expected code to verify test
        """
        sin, sout, serr = self.srv.ssh.exec_command('chmod {} {}'.format(
                perm, remfile))
        status = sout.channel.recv_exit_status()
        self.verifyEqualOrFailCase(0,
                                   status,
                                   "Set permission {} on {}".format(
                perm, remfile))
        response = self.srv.send_http_request_with_auth(path)
        self.verifyEqualOrFailCase(exp_code,
                                   response.status_code,
                                   "Expected HTTP server response")

    def test_partial_content(self, track_name, range_headers, exp_length, xpcode, exp_str_one, exp_str_two):
        """
        Confirms the partial content response from the server.
        Note that the expected strings can not be checked with
        a strict match from the text response because the data
        (expected strings) are within the partial content meta
        data.
        :param str track_name: name of track used for testing
        :param str range_headers: byte range
        :param int exp_length: the expected length
        :param int xpcode: expected return code
        :param str exp_str_one: expected first string return
        :param str exp_str_two: expected second string return
        """
        headers = {'Range': range_headers}
        resp = self.srv.send_http_request_with_auth(track_name, headers=headers)
        self.verifyEqualOrFailCase(xpcode,
                                   resp.status_code,
                                   "Response code should match")
        self.verifyEqualOrFailCase(exp_length,
                                   len(resp.content),
                                   "Expected partial length should match")
        self.verifyInOrFailCase(exp_str_one,
                                resp.text,
                                "Expected {} in response".format(exp_str_one))
        self.verifyInOrFailCase(exp_str_two,
                                resp.text,
                                "Expected {} in response".format(exp_str_two))

    def test_range(self, rangestr, xplen, xpcode):
        """
        Tests multiple range requests on files.
        :param str rangestr: byte range string
        :param int xplen: expected length
        :param int xpcode: expected return code
        """
        resp = self.srv.send_http_request_with_auth('')
        files = XMLAccessor(resp.text.replace('\n', '')).get('file')
        for f in files:
            furl = self.srv.get_share_path(0) + '/' + f.value
            self.logger.debug("Using URL: '{}'".format(furl))
            if self.OUT_OF_RANGE in rangestr:
                # replace with actual file size + 1 to exceed range length
                range_str = "bytes={}-".format(
                    int(f.size) + 1)
                self.logger.debug(
                    "Out-of-range test override {} with {}".format(
                        self.OUT_OF_RANGE, range_str))
            elif self.FINAL_4_BYTES in rangestr:
                range_str = "bytes={}-".format(
                    int(f.size) - 4)
            elif self.INVALID_BYTE_RANGE_SPEC in rangestr:
                range_str = "bytes={}-{}".format(
                    int(f.size), int(f.size) - 1)
            elif self.R2_EXCEEDS_FILE_SIZE in rangestr:
                range_str = "bytes=0-{}".format(
                    int(f.size) + 1)
            else:
                range_str = rangestr
            headers = {'Range': range_str}
            resp = self.srv.send_http_request_with_auth(f.value, headers=headers)
            self.verifyEqualOrFailCase(xpcode,
                                       resp.status_code,
                                       "Response code should match")
            # On self.FULL_FILE test, we're expecting the size of the file
            exp_len = int(f.size) if xplen == -1 else xplen
            self.verifyEqualOrFailCase(exp_len,
                                       int(resp.headers['Content-length']),
                                       "Response length should match")

    def test_query(self, query_mode):
        """
        Tests query requests such as after or before.
        The expectation of querying 'after' with a file name in question
        would be for the server to return a list of files other than file
        in question except when querying the last file in the directory
        in which case the server will not return any files.
        The server does not support 'before' it returns all the files in
        the directory of the queried file for all queried files.
        :param str query_mode: query type
        """
        # url = self.srv.server_url + 'Query Test/'
        response = self.srv.send_http_request_with_auth('')
        files = XMLAccessor(response.content.replace('\n', '')).get('file')
        files = [f.value for f in files]
        last = files[-1]
        for f in files:
            # NOTE: the reason we're not using the requests.get() call
            # with the params keyword is that the server intentionally
            # supports urlenconding and not the "form" encoding.
            # params = {query_mode: f}
            # query_response = self.srv.send_http_request_with_auth(url, params=params)
            # query_response = self.srv.send_http_request_with_auth('?{}={}'.format(
            #        query_mode, f), query=True)
            query_response = self.srv.send_http_request_with_auth(
                f, query='?{}='.format(query_mode))
            self.logger.debug("{}".format(query_response.content))
            qfiles = XMLAccessor(
                query_response.content.replace('\n', '')).get('file')
            if query_mode == 'after':
                if qfiles:
                    # The response should not contain the file under test
                    self.verifyTrueOrFailCase(
                        (f not in [x.value for x in qfiles]),
                        "'{}' should not be in '{}' response".format(
                            f, query_mode))
                else:
                    # We're not expecting any files on last file queried
                    self.verifyTrueOrFailCase(
                        f == last,
                        "Last should be empty")
            else:
                # The response should be the same as the number of
                # files present in the directory of the queried file.
                self.verifyEqualOrFailCase(
                    len(qfiles),
                    len(files),
                    "'{}' '{}' response".format(f, query_mode))
