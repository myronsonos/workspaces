[docs]    def process_share_config_data(self, share_list='basic_shares'):
        """
        Extract necessary share information from WorkflowExecutionConfiguration

        This method expects a JSON dictionary with one (or more) shares defined
        in the following manner:

        .. code-block:: json
           :linenos:

           [
           "share_key": {
               "description": <string>,
               "host": <string>,
               "key": <string>,
               "location: <string>,
               "password": <string>,
               "share_path": <string>,
               "username": <string>
               }
           ]

        :param list share_list: list of shares
        :return: map of share keys to tuple containing path and credential
            strings
        :rtype: :obj:`dict`

        Ex:

        .. code-block:: python

           response = {'small_share':('//sb-sqa-nas2/Automation/queue_tests/small_share', ('','')),
                       'large_share':('//sb-sqa-nas2/Automation/queue_tests/large_share', ('',''))}

        """
        music_shares = {}
        if hasattr(self.execution_configuration, share_list):

            for share in getattr(self.execution_configuration, share_list):
                key = share.key
                path = "//{}/{}".format(self.platform, share.share_path)
                username = share.username
                password = share.password
                music_shares[key] = (path, username, password)
        self.verifyIsNotNoneOrStop(music_shares,
            '{} should not be None'.format(music_shares))
        return music_shares

[docs]    def search_for_tracks_test(self, zp, search_strings, expecting_results):
        """
        Search for tracks on this device and verify they exist.

        Searching for an empty string will return all files indexed on a share.
        This is used for verifying a basic share and a share with unsupported
        file types.

        :param zp: SonosZoneComponent to add share to
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`
        :param search_strings: list of strings to search for
        :type search_strings: :obj:`list` of :obj:`str`
        :param bool expecting_results: should be True if search results are
            expected

        """
        for search_string in search_strings:
            browse_data = zp.ContentDirectory.search_for_tracks_by(
                search_string)
            if expecting_results:
                self.verifyIsNotNoneOrFailCase(browse_data,
                    'Browsed data should not be None')
            else:
                self.verifyIsNoneOrFailCase(browse_data,
                    'Browsed data should be None')

[docs]    def get_sls_cfg_and_start_server(self, host):
        """
        Get the correct configuration for the SonosLibraryServer
        based on the host to use and start the server with a
        specified xml list of shares.

        :param str host: hostname of the machine to start the
           SonosLibraryServer on
        :param str plist: list of shares to host in xml format

        """
        if self.args[0].sls:
            plist = self.get_sls_plist(self.__class__.__name__)
            sls_cfg = WorkflowExecutionConfigurationFactory.create_from_file(
                ABS_SONOS_LIB_SERVER_CONFIG, use_automation_server_config=True)
            for cfg in sls_cfg.sls_access:
                if cfg.server == host:
                    self.srv = SonosLibraryServer(cfg,
                                            self.execution_configuration.build)
                    self.srv.setup_img()
                    self.srv.install_test_share(plist)
                    self.srv.start()

[docs]    def get_element_value_from_track_item_xml(self, zp, element='title'):
        """
        Iterates through a :class:`~sonos.client.internal.SonosZoneComponent`'s
        ContentDirectory tracks and returns a list of their attribute values

        :param zp: a zone player
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`
        :param attr: a track attribute to get
        :type attr: :obj:`str`
        :returns: a list of attributes from all tracks
        :rtype: :obj:`list`
        """
        return [getattr(track_item, element).value for track_item in
            zp.ContentDirectory.search('TRACKS', ' ')[0]]



[docs]    def verify_indexed_tracks(self, zp, expected_total_tracks=None, play_tracks=True):
        """
        Checks to make sure the expected number of tracks has been indexed and
        attempts to playback each indexed track to confirm indexing worked
        properly

        :param zp: the target zone player to perform verifications against
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`
        :param expected_total_tracks: the number of tracks we expected to be
            indexed
        :type expected_total_tracks: :obj:`int`
        :param play_tracks: Should player attempt to play the tracks ?
        :type play_tracks: :obj:`bool`
        """
        track_uris = self.get_element_value_from_track_item_xml(zp, 'res')
        if expected_total_tracks is not None:
            self.verifyEqualOrStop(expected_total_tracks,
                len(track_uris),
                "Verify expected number of tracks")
        if play_tracks:
            zp.AVTransport.use_queue()
            for uri in track_uris:
                self.logger.info(
                    "{}: Playing track - {}".format(zp.modelNumber, uri))
                zp.AVTransport.add_to_queue(uri)
                zp.AVTransport.Play_and_wait()
                zp.AVTransport.Stop_and_wait()
                zp.AVTransport.clear_queue_and_wait()

    def _find_aa_test_track_uri(self, zp, track_ext, aa_ext):
        """
        Private helper method used to find an album art test track in 
        the music library, given the required track and album art 
        extensions.

        :param zp: a zone player
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`
        :param track_ext: Extension of track to search for
        :type track_uri: :obj:`str`
        :param aa_ext: Extension of album art to search for
        :type aa_ext: :obj:`str`

        """
        search_term = '{}-{}'.format(track_ext, aa_ext)
        track_data = zp.ContentDirectory.search_filenames(search_term)
        return track_data[0].res.value if track_data else None


[docs]    def perform_aa_test(self, zp, track_uri):
        """
        Verify that the album art is accessible for this track

        :param zp: a zone player
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`
        :param track_uri: URI of playback track
        :type track_uri: :obj:`str`

        """
        zp.AVTransport.add_to_queue(track_uri, "")
        zp.AVTransport.Play_and_wait()
        aa_uri = zp.AVTransport.get_current_track_album_art_uri()
        self.verifyIsNotNoneOrFailCase(aa_uri, 
            "Expected album art URI for track <{}>".format(track_uri))
        zp.AVTransport.Stop_and_wait()
        if aa_uri:
            data = zp.http.get_url(aa_uri)
            self.verifyTrueOrFailCase(len(data) > 0, 
                "Expected some data back from album "
                "art URI <{}> (track: <{}>)".format(aa_uri, track_uri))
