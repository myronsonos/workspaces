        """
        Given an update URL, the MDCR disk image file is selected and

[docs]    def push_img(self):
        """
        Sends the disk image file from the local machine to the remote server.

        :return: Name of the local disk image file.
        :raises: ValueError if size of local and remote files not the same.
        :rtype: :obj:`str`

        """
        transport = paramiko.Transport((self.server, DEFAULT_SSH_PORT))
        transport.connect(username=self.user, password=self.password)
        sftp = paramiko.SFTPClient.from_transport(transport)
        remfile = sftp.put(self.temp_dmg_file.name, '/tmp/{}'.format(
                self.mdcr_img))
        sftp.close()
        transport.close()
        if remfile.st_size != os.stat(self.temp_dmg_file.name).st_size:
            raise ValueError("Remote size {} != local size {}".format(
                    remfile.st_size, os.stat(self.temp_dmg_file.name).st_size))
        return '/tmp/{}'.format(self.mdcr_img)

[docs]    def set_paramiko_log_level(self, level):
        logging.getLogger("paramiko").setLevel(level)


    def _connect(self):
        """
        Initiate SSH connection to server, setting the minimum log level.
        Return SSHClient object class.

        :return: SSHClient class
        :rtype: :obj:class:`paramiko.SSHClient`
        """
        self.set_paramiko_log_level(logging.ERROR)
        ssh = paramiko.SSHClient()
        ssh.set_missing_host_key_policy(paramiko.AutoAddPolicy())
        ssh.connect(self.server,
                    username=self.user,
                    password=self.password)
        return ssh

    def _setup_sftp(self):
        """
        Initiate SFTP session to server.

        :return: SFTPClient class
        :rtype: :obj:class:`paramiko.SFTPClient`

        """
        self.transport = paramiko.Transport((self.server, DEFAULT_SSH_PORT))
        self.transport.connect(username=self.user, password=self.password)
        return paramiko.SFTPClient.from_transport(self.transport)

    def _get_credentials(self):
        """
        Decrypts and stores the credentials for accessing the shares.
        Installs the decryption binary on remote system, executes the
        command to decrypt the credentials, parses the output stores
        the credentials and removes the remote decrypt binary.
        """
        self.cred_user = None
        self.cred_pass = None
        tmpFile = tempfile.NamedTemporaryFile()
        DownloadManager.download_file(self.DECRYPT_BINARY_URL, tmpFile.name)
        self.sftp.put(tmpFile.name,
                      self.REMOTE_DECRYPT_EXEC_NAME,
                      confirm=True)
        status = self.sftp.chmod(self.REMOTE_DECRYPT_EXEC_NAME, int('777', 8))
        sin, sout, serr = self.ssh.exec_command(self.REMOTE_DECRYPT_EXEC_NAME)
        status = sout.channel.recv_exit_status()
        if status != 0:
            self.logger.error(
                "No credentials - stderr output '{}'".format(serr))
        cred_search = re.search('name=(.*)\n.*pass=(.*)\n',
                                serr.read(),
                                re.MULTILINE)
        if cred_search:
            self.cred_user = cred_search.group(1)
            self.cred_pass = cred_search.group(2)
        else:
            self.logger.error(
                "Couldn't find credentials - stderr output '{}'".format(serr))
        self.sftp.remove(self.REMOTE_DECRYPT_EXEC_NAME)

        """
        Updates the required .plist file on the server.
        The .plist contains arguments required to be passed to the
        launchctl deamon such as path of the Sonos Library server
        executable.

        :raises: ValueError if size of local and remote files not the same.

        """
        tmp_file = tempfile.NamedTemporaryFile()
        tmp_file.write(self.PLIST)
        tmp_file.flush()
        tmp_file.seek(0)
        remfile = self.sftp.put(tmp_file.name, self.PLIST_FILE)
        if remfile.st_size != os.stat(tmp_file.name).st_size:
            raise ValueError("Remote size {} != local size {}".format(
                    remfile.st_size, os.stat(tmp_file.name).st_size))

        """
        Attach the disk file image on the local file system.

        Returns boolean True if the image has been attached or False on error

        :return: Has the image been attached ?
        :rtype: :obj:`bool`
        """
        sin, sout, serr = self.ssh.exec_command('mkdir -p {}'.format(
                self.SERVER_MOUNT_DIR))
        status = sout.channel.recv_exit_status()
        if status != 0:
            self.logger.error("attach_img: couldn't create {} : {}".format(
                    self.SERVER_MOUNT_DIR, serr.read()))
            return False
        sin, sout, serr = self.ssh.exec_command(self.ATTACH_CMD.format(
                self.SERVER_MOUNT_DIR, self.upd_img))
        status = sout.channel.recv_exit_status()
        if status != 0:
            self.logger.error("attach_img: {} {}".format(
                    serr.read(), self.upd_img))
        return self.is_img_attached()

        """
        Detach the disk file image from the local file system.

        Returns boolean True if the image has been detached or False on error

        :return: Has the image been detached ?
        :rtype: :obj:`bool`
        """
        detached = False
        sin, sout, serr = self.ssh.exec_command(self.DETACH_CMD)
        if sout.channel.recv_exit_status() == 0:
            out = sout.read()
            if 'unmounted' in out and 'ejected' in out:
                detached = True
        elif 'No such file or directory' in serr.read():
            detached = True
        return detached

[docs]    def stop_server(self):
        """
        Stop the Sonos library server.

        Returns boolean True if the Sonos library server has
        stopped else False.

        :return: Has the Sonos library server stopped ?
        :rtype: :obj:`bool`
        """
        pid = self.is_server_running()
        if -1 == pid:
            return True
        self.logger.info("Stopping server with pid {}".format(pid))
        stopped = False
        sin, sout, serr = self.ssh.exec_command(self.SERVER_UNLOAD_CMD)
        if sout.channel.recv_exit_status() == 0:
            stopped = True
            self.logger.info("server with pid {} stopped".format(pid))
            self.server_pid = -1
        else:
            self.logger.error("Couldn't stop server: standard error contents")
            self.logger.error("{}".format(serr.read()))
        return stopped

        """

        Returns boolean True if the Sonos library server started else False.

        :return: Has the Sonos library server started ?
        :rtype: :obj:`bool`
        """
        if not self.stop_server():
            return False
        started = False

        if sout.channel.recv_exit_status() == 0:
            started = True
            self.logger.info("server running with pid {}".format(
                    self.is_server_running()))
        else:
            self.logger.error("Couldn't start server: standard error contents")
            self.logger.error("{}".format(serr.read()))
        return started



[docs]    def is_server_running(self):
        """
        Checks if the Sonos library server is running.

        Returns the numeric process identifier of the Sonos library server
        or zero if the server is not running.

        :return: Process identifier number of Sonos library server
        :rtype: :obj:`int`

        """
        self.server_pid = -1
        sin, sout, serr = self.ssh.exec_command(self.SERVER_LIST_CMD)
        out = sout.read()
        m = re.search(ur'^(\S+)\s+(\S+)\s+com.sonos.SonosLibraryServer',
                      out,
                      re.MULTILINE)
        if m:
            if m.group(1).isdigit():
                self.server_pid = int(m.group(1))
                self.logger.debug("Detected server running with pid {}".format(
                        self.server_pid))
            else:
                self.logger.warn("Detected renmant server")
                # return anything but -1
                return 0
        return self.server_pid

        """
        It parses the self.SONOS_LIB_SERVER_SHARE from the remote machine
        where the Sonos Libarary server is active looking for the share
        name the server is serving the HTTP requests.

        Parses the SONOS_LIB_SERVER_SHARE and creates a list
        of tuples (share name, absolute path). Returns an empy list
        if there're no shares on the file.

        TODO Per Eric B. It might be better to just fetch shares from XML
        TODO everytime we want to get this list. That way we are always getting
        TODO the latest share information, as I imagine we have tests
        TODO that modify shares to validate the server is working properly.
        TODO This can easily be done by converting get_server_shares into
        TODO a property getter.
        TODO self.server_shares[]
        TODO @property
        TODO def server_shares(self):
        TODO     ...

        :return: list of tuples comprised of share name and absolute path
        :rtype: :obj:`list`
        """
        self.server_shares = []
        remfile = self.sftp.open(self.SONOS_LIB_SERVER_SHARE, 'r')
        text = remfile.read()
        data = XMLAccessor(text.replace(u'\n', '').replace(u'\t', ''))
        if not hasattr(data, 'dict'):
            self.logger.warn("Did not find a share in {}".format(
                    self.SONOS_LIB_SERVER_SHARE))
            return self.server_shares
        items = data.get('dict')
        shares = []
        paths = []
        for item in items:
            if item.value:
                shares.append(item.value)
            elif (hasattr(item, 'string')):
                paths.append(item.get('string').value)
        if len(shares) > 0 and len(shares) == len(paths):
            self.server_shares = [(shares[item], paths[item])
                                  for item in range(len(shares))]
        self.logger.debug("Server server_shares {}".format(self.server_shares))
        return self.server_shares

[docs]    def send_http_request_with_auth(self, path, headers=None, timeout=None, query=None):
        """
        Access to Sonos Library Server HTTP pages using credentials.
        The provided path in addition to the root path the server is providing
        (http://<server:3445/Share where Share is the name of the key in the PLIST)
        along with the credentials (name/password) are encoded to generate a hash
        which is used in the Authentication key word of the requests headers.
        It returns the response from the Sonos Library Server.

        :param str path: partial URL
        :param dict headers: Dictionary of key:value pairs for headers in requests
        :param int timeout: Timeout in seconds
        :param str query: Query ?before= or ?after= not URL encoded
        :return: requests.Response object
        :rtype: :obj:`~requests.Response`
        """
        # Initialize the required headers dictionary if it's not provided by user
        if not headers:
            headers = {}
        # If query mode, the query is not url encoded that is
        # ?after=   or  ?before=
        # the ? and = characters are not to be url encoded - ugly!!!!
        slash = '/'
        if query:
            root = '/' + self.server_shares[0][0]
            updated_path = root + query + path
            quoted_root = urllib.quote(root, self.URL_ENCODE_EXCLUDE_CHARS)
            quoted_path = urllib.quote(path, self.URL_ENCODE_EXCLUDE_CHARS)
            quoted_updated_path = quoted_root + query + quoted_path
        else:
            updated_path = '/' + self.server_shares[0][0] + slash + path
            quoted_updated_path = urllib.quote(updated_path,
                                               self.URL_ENCODE_EXCLUDE_CHARS)
        self.logger.debug("Using path '{}' quoted path '{}'".format(
                updated_path, quoted_updated_path))
        s = HMAC.new(self.SHARED_KEY, None, SHA256)
        s.update(quoted_updated_path)
        s.update(self.cred_user)
        s.update(self.cred_pass)
        genhash = s.hexdigest().upper()
        headers['Authorization'] = 'hash ' + genhash
        full_path = self.server_url + quoted_updated_path
        return requests.get(full_path, headers=headers, timeout=timeout)


    def _wait_for_ready(self, path, retries=5):
        """
        Attempts to access the specified HTTP page returning True
        if the server responds. In the event the TCP port has not
        been opened for HTTP requests (Connection refused), it
        waits one second and retries to access the page again.
        Returns False when the retries have been exhausted without
        a response from the server.

        :param str path: server URL page
        :param int retries: number of attempts to access URL page
        :return: is the server ready ?
        :rtype: :obj:`bool`

        """
        refused = True
        url = "http://{}:{}/{}".format(self.server,
                                       self.SONOS_LIB_SERVER_PORT,
                                       path)
        attempts = 0
        while refused and attempts < retries:
            try:
                requests.get(url, timeout=10)
            except requests.exceptions.ConnectionError:
                self.logger.debug("{} connection refused".format(url))
                refused = True
                time.sleep(1)
                attempts = 1 + attempts
            else:
                refused = False
                self.logger.debug("server responded on {} on {} attempt".format(
                        url, attempts+1))
        return not refused


[docs]    def is_server_ready(self):
        """
        Returns True if the HTTP server responds to HTTP requests else False.
        If the server is ready, the variable self.server_url is updated to
        the full URL of the server's shared path.
        The request uses the first share, path tuple in the list returned
        from parsing the the .plist file.

        :return: is the server ready ?
        :rtype: :obj:`bool`
        """
        ret = False
        shares = self.get_server_shares()
        if shares:
            (share, path) = shares[0]
            ret = self._wait_for_ready(share)
            if ret:
                self.server_url = "http://{}:{}".format(
                    self.server,
                    self.SONOS_LIB_SERVER_PORT)
        return ret
