    def __usr_var_changed(self, var):
        """
        Executes in UDP Logger thread.
        :param var: Service variables that changed.
        :return: None
        """
        if isinstance(var.service.device, RootDevice):
            root_device = var.service.device
        elif isinstance(var.service.device.parent, RootDevice):
            root_device = var.service.device.parent
        else:
            raise NotImplementedError("RootDevice not detected in {}".format(var))
        self.log_debug(
            "USER Callback: {}'s \"{}\" changed from \"{}\" to \"{}\"".format(root_device, var.name, var.old_value,
                                                                              var.value))

        """
        Call back for UDP logging matched expressions.
        Executes in UDP Logger thread.

        :param zp: sonos zone player for which msg was detected
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`

        :param counters: Dictionary of the matched expression counters for zp

        if counters.values():

                                   {"module": m, "expr": expr,
                                    "counter": counters[expr],
                                    "comment": msg})

        """
        Signal handler when a FATAL error is encountere from matching UDP logging.

        :type signum: :obj: `int`

    def test_ht_playback_source_transition(self, master, satellites):
        """
        Setup HT and then play TV then local music and back to TV.

        Note that when in HT, satellites are invisible and we cannot
        use AVTransport to check states.
        """
        self.log_info("Master: {}".format(master))
        self.log_info("Satellites: {}".format(satellites))

        assert master.is_bonded(), "{} is expected to be bonded.".format(master)
        for s in satellites:
            assert s.is_satellite(), "{} is  expected to be satellite.".format(s)

        min_duration_before_transition = HTPlaybackSourceTransitionTests.THEORETICAL_CONTINUOUS_PLAYING_TIME \
                                         - HTPlaybackSourceTransitionTests.EXPECTED_CONTINUOUS_PLAYING_THRESHOLD
        max_wait_for_transition = HTPlaybackSourceTransitionTests.THEORETICAL_CONTINUOUS_PLAYING_TIME \
                                  + HTPlaybackSourceTransitionTests.EXPECTED_CONTINUOUS_PLAYING_THRESHOLD

        min_duration_before_transition_short = HTPlaybackSourceTransitionTests.THEORETICAL_CONTINUOUS_PLAYING_TIME_SHORT \
                                               - HTPlaybackSourceTransitionTests.EXPECTED_CONTINUOUS_PLAYING_THRESHOLD
        max_wait_for_transition_short = HTPlaybackSourceTransitionTests.THEORETICAL_CONTINUOUS_PLAYING_TIME_SHORT \
                                        + HTPlaybackSourceTransitionTests.EXPECTED_CONTINUOUS_PLAYING_THRESHOLD

        avts = AVTransportStateTransitionTimer(master.AVTransport)

        self.log_info("Master {} starting {} playback.".format(master, 'queue'))
        elapsed = start_playback_and_assert_playback_duration(avts, master, ('queue', 1),
                                                              max_wait_for_transition,
                                                              min_duration_before_transition,
                                                              True,
                                                              False)
        self.log_info("Elapsed: {}".format(elapsed))

        tv_tracks = OrderedDict([(k, v) for k, v in HTPlaybackSourceTransitionTests.SPDIF_TEST_FILES.iteritems() if
                                 32000.0 < v['SampleRate'] <= 48000.0])
        for i, (k, v) in enumerate(tv_tracks.iteritems()):
            self.log_info("Start TOSLINK signal into HT Master {}: {}".format(master, (k, v)))
            with MplayerAc3Playback(k, 'hwac3' if v['StreamInfo'].startswith('Dolby Digital') else 'pcm'):
                """
                cat /proc/driver/fpga/reg/23 gives us input SPDIF sample rates
                0x8x is 32000
                0x6x is 44100
                0x5x is 48000
                0x3x is 88200
                0x2x is 96000
                """
                self.log_info("Master {} starting {} playback.".format(master, 'spdif'))
                elapsed = start_playback_and_assert_playback_duration(avts,
                                                                      master,
                                                                      ('spdif', v),
                                                                      max_wait_for_transition,
                                                                      min_duration_before_transition,
                                                                      True,
                                                                      False,
                                                                      i > 0)
                self.log_info("Elapsed: {}".format(elapsed))

                # If last track track, we play Music
                if i == len(list(tv_tracks)) - 1:
                    self.log_info("Master {} starting {} playback.".format(master, 'queue'))
                    elapsed = start_playback_and_assert_playback_duration(avts,
                                                                          master,
                                                                          ('queue', 1),
                                                                          max_wait_for_transition_short,
                                                                          min_duration_before_transition_short,
                                                                          True,
                                                                          False)
                    self.log_info("Elapsed: {}".format(elapsed))
                self.log_info("Stop TOSLINK signal into Master {}: {}".format(master, (k, v)))

        self.log_info("Master {} starting {} playback.".format(master, 'queue'))
        elapsed = start_playback_and_assert_playback_duration(avts,
                                                              master,
                                                              ('queue', 1),
                                                              max_wait_for_transition,
                                                              min_duration_before_transition,
                                                              True,
                                                              False,
                                                              True)
        self.log_info("Elapsed: {}".format(elapsed))

        self.log_info("Master {} stopping playback.".format(master))
        master.AVTransport.Stop_and_wait()
