    def _setup_client(self):
        client = smapiclient.SMAPIClient(svcName="node-js SMAPI server", wsdl=WSDL, insecureURI=smapi_ctl.SMAPI_SERVER_SECURE_URI, secureURI=smapi_ctl.SMAPI_SERVER_SECURE_URI, flags=CAPABILITY_FLAGS, authModel='Anonymous')

        client.login()
        return client

    def _play_and_seek_podcast(self, zp, track_info, play_time, seek_position, pause_playback=False, before_end=False):
        '''
        Initiate playback of a podcast, and allow playback to continue for a given 'play_time', or seek to a certain position
        :param zp: The Player being used for playback
        :param track_info: The metadata of the track/podcast to be printed
        :param play_time: How long playback of the track should be allowed to go on for
        :param seek_position:
        :param pause_playback: (Boolean) Indicates whether playback should be paused after 'play_time' has elapsed
        :param before_end: (Boolean) Indicates whether 'seek_position' is a relative time before the track's  duration
        :return: The position of the track being played on the ZP in seconds
        '''
        skip_value = "{0}00:00:{1}".format('+', abs(seek_position)) #time.strftime('%H:%M:%S', time.gmtime(seek_position))
        self.verifyTrueOrFailCase(zp.AVTransport.add_to_queue(track_info[0], track_info[1], track_num=1) == 1, "Add track to queue")
        zp.AVTransport.use_queue()
        if before_end:
            zp.AVTransport.seek_to_position_before_track_end(seek_position)
        else:
            zp.AVTransport.Seek_and_wait('REL_TIME', skip_value)
            wait_until_true(lambda: reltime_str_to_seconds(zp.AVTransport.GetPositionInfo()['RelTime']) == seek_position)

        if not pause_playback:
            if not zp.AVTransport.is_playing():
                zp.AVTransport.Play_and_wait()
            wait_until_true(lambda: zp.AVTransport.is_playing(), iteration_delay=1, start_delay= play_time)
            zp.AVTransport.Pause_and_wait()
        else:
            zp.AVTransport.Play_and_wait()
            wait_until_true(lambda: zp.AVTransport.is_playing(), iteration_delay=1, start_delay= 5)
            zp.AVTransport.Pause_and_wait()
        self.verifyTrueOrFailCase(zp.AVTransport.add_to_queue(TRACK_NODE[0], TRACK_NODE[1], track_num=2) == 1, "Add track to queue")
        zp.AVTransport.set_queue_position(2)
        zp.AVTransport.use_queue()
        if not zp.AVTransport.is_playing():
            zp.AVTransport.Play_and_wait()
        wait_until_true(lambda: zp.AVTransport.is_playing(), iteration_delay=1, start_delay=5)
        zp.AVTransport.Pause_and_wait()
        zp.AVTransport.set_queue_position(1)
        zp.AVTransport.use_queue()
        if not zp.AVTransport.is_playing():
            zp.AVTransport.Play_and_wait()
        wait_until_true(lambda: zp.AVTransport.is_playing(), iteration_delay=1, start_delay=5)
        return reltime_str_to_seconds(zp.AVTransport.GetPositionInfo()['RelTime'])

    def _add_track_multiple_times_to_queue(self, zp, uri, metadata, num_times_to_add):
        starting_queue_len = zp.AVTransport.get_number_of_tracks_in_queue()
        [zp.AVTransport.add_to_queue(uri = uri, metadata = metadata, track_num=starting_queue_len+1) for n in range(num_times_to_add)]
