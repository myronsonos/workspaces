    """
    Tests the capture and replay of SPDIF audiotaps on all HT capable devices
    in the testbed.

    SPDIF signal sources are listed in SPDIF_TESTSIG_FILES

    1. Capture/download methods include:

    - CLI Capture method from a testrunner via curl/wget:

        wget  http://<HT ZP>:1400/audio_tap?tap=spdif -O cli_audiotap.spdif

    - URL Capture method:

        http://<HT ZP>:1400/snapshotspdiftap <- capture of previous 90s input
        http://<HT ZP>:1400/downloadspdiftap <- download capture as custom zip

    - URL / BASE64 method:

        http://<HT ZP>:1400/snapshotspdiftap <- capture of previous 90s input
        http://<HT ZP>:1400/spdiftap <- download capture as custom zip via XML

    - Diag method:

        UPnP or web URL can be used to submit a diag. Audiotap is embedded
        in the XML but is unreadable through regular diagnostics viewer,
        so you need to view page source to find the <SPDIFTap> element

    2. Playback of a captured audiotap can be achieved by loading the tap
    into /jffs on the ZP (must be same model, and ensure the file name is
    audiotap.spdif) supply a "reference SPDIF clock" input which matches the
    sample rate of the captured tap, then finally toggle anacapa to commence
    playback. The test checks stream type and sample rate match the original
    test signal played in for the capture

    def _cleanup(self, zp):
        """
        Clear avt and queue, remove bonded config if it exists
        Clear autoplay URI

        :param zp: HT master capable ZP

        """
        self.logger.info('Performing cleanup')
        zp.DeviceProperties.cleanup_bonded_configuration()
        zp.AVTransport.clear_play_state(timeout_standalone_seconds=30)
        zp.AVTransport.clear_queue_and_wait()
        zp.DeviceProperties.SetAutoplayRoomUUID('')

    def _setup_udp_logger(self, zp):
        """
        Setup udp logger for zp
        If zp is an HT capable device:
            1. Set diag levels for debug message detection
            2. Pass in callbacks for strings we are looking for
            3. Set/reset any callback flags as required

        :param zp: HT master capable ZP

        """
        self.logger.info('Setting up udp logger on : {}'.format(zp))
        zp.diag.start_udp_log()
        zp.diag.watch_udp_log_for_string(
            self.AUDIO_TAP_START_MSG,
            lambda msg: self._cb_detected_audiotap_start(msg, zp))
        zp.wait_for_upnp_alive()
        self.audiotap_playback_triggered = False

    def _cb_detected_audiotap_start(self, msg, zp):
        """
        Set callback action when 'msg' string is detected in UDP log
        Only activate alert flag if playback is in progress

        :param msg: string to be detected
        :param zp: zone player

        """
        self.logger.debug('Detected {} on {}'.format(msg, zp))
        self.audiotap_playback_triggered = True

    def _setup_webserver(self, resource_path, port=WEBSERVER_PORT):
        """
        Setup local web server

        :param resource_path: location of files to be served
        :param port: port number for http access
        :return: webserver instance

        """
        websrv = WebServer(start_reactor=False)
        try:
            websrv.start(port, resource_path)
        except TimeoutError:
            self.stop_suite('Could not start web server')
        else:
            self.logger.info('Started web server')
            return websrv

    def _spdif_play_test_source_and_capture_tap(self,
                                                ht_master,
                                                input_signal,
                                                capture_method=None):
        """
        Plays back a test signal, confirm signal is being received as expected
        Captures SPDIF audiotap using specified method (cli / url / base64
        url / diag)

        :param ht_master: HT master capable ZP
        :type ht_master: ZP
        :param strFileURL: test file URI on test runner
        :type strFileURL: utf-8 string
        :param dictTrackInfo: metadata accompanying the test file
        :type dictTrackInfo: dict
        :param capture_method: specified capture method
        :type capture_method: utf-8 string
        :return: URI (full system path, including filename) of captured tap
        :rtype: utf-8 string

        """
        strTestInputFileURL, dictTrackInfo = input_signal
        with MplayerAc3Playback(strTestInputFileURL, 'hwac3' if dictTrackInfo[
                'StreamInfo'].startswith('Dolby Digital') else 'pcm'):
            wait_until_true(
                lambda: (ht_master.diag.get_toslink_stream_type() ==
                         dictTrackInfo['StreamInfo']) and (
                        ht_master.diag.get_toslink_sample_rate() ==
                         dictTrackInfo['SampleRate']),
                iteration_delay=self.WAIT_POLL_FREQUENCY,
                timeout_seconds=self.WAIT_TIMEOUT,
                reason=('Error while playing back test SPDIF signal: '
                        'Expected SPDIF Stream Type: {} with Sample '
                        'Rate: {}, Actual SPDIF Stream Type: {} with '
                        'Sample Rate: {}'.
                        format(dictTrackInfo['StreamInfo'],
                               dictTrackInfo['SampleRate'],
                               ht_master.diag.get_toslink_stream_type(),
                               ht_master.diag.get_toslink_sample_rate())
                )
            )
            self.logger.info('Playing back test SPDIF signal {}: Expected '
                             'SPDIF Stream Type: {} with Sample '
                             'Rate: {}, Actual SPDIF Stream Type: {} with '
                             'Sample Rate: {}'.format(
                                strTestInputFileURL,
                                dictTrackInfo['StreamInfo'],
                                dictTrackInfo['SampleRate'],
                                ht_master.diag.get_toslink_stream_type(),
                                ht_master.diag.get_toslink_sample_rate()))
            if capture_method == 'cli':
                strTapFileURL = self.utils.get_tap_from_testrunner(
                                    ht_master,
                                    self.SPDIF_AUDIOTAP_TYPE,
                                    self.SPDIF_TAP_CAPTURE_DURATION,
                                    os.path.join(self.websrv.http.file_path,
                                                 self.SPDIF_TAP_FILENAME))
            elif capture_method == 'url':
                self.utils.capture_spdif_tap_with_url(
                                        ht_master,
                                        self.URL_SPDIF_TAP_CAPTURE_DURATION)
                strTapFileURL = self.utils.download_spdif_tap_from_url(
                                                    ht_master,
                                                    self.SPDIF_TAP_FILE_URI,
                                                    is_base64=False)
            elif capture_method == 'urlbase64':
                self.utils.capture_spdif_tap_with_url(
                                        ht_master,
                                        self.URL_SPDIF_TAP_CAPTURE_DURATION)
                strTapFileURL = self.utils.download_spdif_tap_from_url(
                                                    ht_master,
                                                    self.SPDIF_TAP_FILE_URI,
                                                    is_base64=True)
            elif capture_method == 'diag':
                diag_id = self.utils.capture_spdif_tap_with_diag(
                                        ht_master,
                                        self.URL_SPDIF_TAP_CAPTURE_DURATION)
                strTapFileURL = self.utils.download_spdif_tap_from_diag(
                                                    ht_master,
                                                    self.SPDIF_TAP_FILE_URI,
                                                    diag_id)
            else:
                raise NotImplementedError('Unknown SPDIF tap capture method '
                                          'requested by test')
            assert not ht_master.cli.does_file_exist('/jffs/{}'.format(
                self.SPDIF_TAP_FILENAME)), 'Audiotap data was not deleted ' \
                                           'from /jffs when using capture ' \
                                           'method:{}'.format(capture_method)
            return strTapFileURL

    def _get_SPDIF_clock_track_for_samplerate(self, samplerate):
        """
        Finds a clock source file corresponding with samplerate
        Confirms file exists
        Returns the file location and file metadata (stream type,
        samplerate, channel count, duration)

        :param samplerate: requested samplerate
        :type samplerate: float
        :return: file location of matching clock file, or None
        :rtype: utf-8 string

        """
        for clockfilename, clockfileinfo in \
                self.SPDIF_CLOCK_FILES.iteritems():
            if clockfileinfo['SampleRate'] == samplerate:
                assert os.path.isfile(clockfilename), \
                    'failed to find SPDIF clock track {}'.format(clockfilename)
                return [(clockfilename, clockfileinfo)]
        return None

    def _replay_validate_SPDIF_tap(self,
                                   ht_master,
                                   input_signal_info,
                                   strTestAudiotapFileURL):
        """
        Plays back the captured SPDIF audiotap file on HT ZP
        Requires a reference SPDIF input as clock during audiotap replay
        Asserts signal type of clock is correct before replaying tap
        Asserts replay tap stream type and sample rate match original input
        Cleans up audiotap files after use

        :param ht_master: HT master capable ZP
        :type ht_master: ZP
        :param input_signal_info: metadata plus filename of the original input
        :type input_signal_info: tuple
        :param strTestAudiotapFileURL: captured audiotap file URI
        :type strTestAudiotapFileURL: utf-8 string

        """
        strTestInputFileURL, dictTrackInfo = input_signal_info
        [(spdif_clock_track, spdif_clock_track_meta)] = \
            self._get_SPDIF_clock_track_for_samplerate(
                dictTrackInfo['SampleRate'])
        assert spdif_clock_track is not None, 'failed to find SPDIF clock ' \
                                              'track at samplerate: ' \
                                              '{}'.format(
                                                dictTrackInfo['SampleRate'])
        with MplayerAc3Playback(spdif_clock_track, 'hwac3' if
                                spdif_clock_track_meta['StreamInfo'].
                                startswith('Dolby Digital')
                                else 'pcm'):
            # check clock signal is being received as expected
            wait_until_true(
                lambda: ht_master.diag.get_toslink_stream_type() ==
                        spdif_clock_track_meta['StreamInfo'] and
                        ht_master.diag.get_toslink_sample_rate() ==
                        spdif_clock_track_meta['SampleRate'],
                iteration_delay=1.0,
                timeout_seconds=10.0,
                reason='Preparing to playback captured audiotap: '
                       'Error while playing clocking signal - the expected '
                       'SPDIF input Stream Type: {} with Sample Rate: {} '
                       'was not found. Actual SPDIF Stream Type: {} '
                       'with Sample Rate: {}'.format(
                            spdif_clock_track_meta['StreamInfo'],
                            spdif_clock_track_meta['SampleRate'],
                            ht_master.diag.get_toslink_stream_type(),
                            ht_master.diag.get_toslink_sample_rate())
            )
            # copy SPDIF audiotap AFTER checking the input clock file,
            # otherwise audiotap will start to play and our check will
            # be invalid:
            ht_master.cli.command('wget http://{}/{} -O /jffs/{}'.format(
                self.websrv.host(),
                self.SPDIF_TAP_FILENAME,
                self.SPDIF_TAP_FILENAME))
            # start local audiotap playback:
            ht_master.bounce_anacapa_and_wait()
            self._setup_udp_logger(ht_master)
            self.logger.info('Audiotap replay: '
                             'Waiting for SPDIF tap playback start trigger '
                             'message...')
            wait_until_true(
                lambda: self.audiotap_playback_triggered is True,
                iteration_delay=1.0,
                timeout_seconds=self.URL_SPDIF_TAP_CAPTURE_DURATION,
                reason='Audiotap replay: Expected SPDIF tap playback start '
                       'trigger message...'
            )
            # Validate our captured audiotap is playing back correctly:
            wait_until_true(
                lambda: ht_master.diag.get_toslink_stream_type() ==
                        dictTrackInfo['StreamInfo'] and
                        ht_master.diag.get_toslink_sample_rate() ==
                        dictTrackInfo['SampleRate'],
                iteration_delay=1.0,
                timeout_seconds=self.SPDIF_TAP_CAPTURE_DURATION,
                reason='Audiotap replay of {}:'
                       'Expected SPDIF tap input Stream Type: {} '
                       'with Sample Rate: {},'
                       'Actual SPDIF tap Stream Type: {} with Sample '
                       'Rate: {}'.format(
                    strTestAudiotapFileURL,
                    dictTrackInfo['StreamInfo'],
                    dictTrackInfo['SampleRate'],
                    ht_master.diag.get_toslink_stream_type(),
                    ht_master.diag.get_toslink_sample_rate())
            )
            self.logger.info('Audiotap replay of {}: '
                             'Expected SPDIF Stream Type: {} with Sample '
                             'Rate: {}, Actual SPDIF Stream Type: {} with '
                             'Sample Rate: {}'.format(
                                    strTestAudiotapFileURL,
                                    dictTrackInfo['StreamInfo'],
                                    dictTrackInfo['SampleRate'],
                                    ht_master.diag.get_toslink_stream_type(),
                                    ht_master.diag.get_toslink_sample_rate()))
        # Dispose of loaded audiotap file and stop playback on the zp:
        ht_master.cli.remove_file('/jffs/{}'.format(self.SPDIF_TAP_FILENAME))
        ht_master.bounce_anacapa_and_wait()


[docs]    def generate_unique_ht_master_devices(self):
        """
        Generator to yield an HT Master capable ZP from the testbed, with each
        input test SPDIF signal (test signal returned as tuple of file and
        file info)

        """
        spdif_test_files_dict = OrderedDict(
            [
                (strFileURL, dictTrackInfo) for strFileURL, dictTrackInfo in
                self.SPDIF_TESTSIG_FILES.iteritems()
            ]
        )
        for ht_master in \
                self.generators.generate_testbed_ht_master_capable_devices():
                    for input_SPDIF_signal, (strFileURL, dictTrackInfo) in \
                            enumerate(spdif_test_files_dict.iteritems()):
                                yield (ht_master, (strFileURL, dictTrackInfo))
