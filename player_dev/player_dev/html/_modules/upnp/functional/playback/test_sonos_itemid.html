    def _clear_queue(self):

            if zp.AVTransport.get_number_of_tracks_in_queue(): zp.AVTransport.clear_queue_and_wait()
            if zp.AVTransport.get_avtransport_uri != " ": zp.AVTransport.clear_play_state()

    def _get_item_flags(self, item_string, playlist=False):
        if (playlist):
            try:
                found = re.search('flags%3d(\d+)', item_string).group(1)
            except AttributeError:
                found = 'No flags found'
        else:
            try:
                found = re.search('flags=(\d+)', item_string).group(1)
            except AttributeError:
                found = 'No flags found'
        return found

    def _get_playlist_explicit_bit(self, item_string):
        return item_string[-1]

    def _get_legacy_flags(self, flag):
        return hex(int(flag) & RSONOSCP_LEGACY_FLAGS_MASK)

    def _get_smapi_playback_policies(self, flags):
        track_flag = int(flags)
        playback_policies = ''
        if (track_flag & RSONOSCP_FLAG_AUTHREQUIRED):
            playback_policies += 'RSONOSCP_FLAG_AUTHREQUIRED '
        if (track_flag & RSONOSCP_FLAG_CANSCROLL):
            playback_policies += 'RSONOSCP_FLAG_CANSCROLL '
        if (track_flag & RSONOSCP_FLAG_CANENUMERATE):
            playback_policies += 'RSONOSCP_FLAG_CANENUMERATE '
        if (track_flag & RSONOSCP_FLAG_CANPLAY):
            playback_policies += 'RSONOSCP_FLAG_CANPLAY '
        if (track_flag & RSONOSCP_FLAG_RESTRICTED):
            playback_policies += 'RSONOSCP_FLAG_RESTRICTED '
        if (track_flag & RSONOSCP_FLAG_CANADD):
            playback_policies += 'RSONOSCP_FLAG_CANADD, '
        if (track_flag & RSONOSCP_FLAG_CANCACHE):
            playback_policies += 'RSONOSCP_FLAG_CANCACHE, '
        if (track_flag & RSONOSCP_FLAG_NOSKIP):
            playback_policies += 'RSONOSCP_FLAG_NOSKIP, '
        if (track_flag & RSONOSCP_FLAG_OOB_METADATA):
            playback_policies += 'RSONOSCP_FLAG_OOB_METADATA, '
        if (track_flag & RSONOSCP_FLAG_WRITEABLE):
            playback_policies += 'RSONOSCP_FLAG_WRITEABLE, '
        if (track_flag & RSONOSCP_FLAG_USERCONTENT):
            playback_policies += 'RSONOSCP_FLAG_USERCONTENT, '
        if (track_flag & RSONOSCP_FLAG_CANRENAME):
            playback_policies += 'RSONOSCP_FLAG_CANRENAME, '
        if (track_flag & RSONOSCP_FLAG_ISEPHEMERAL):
            playback_policies += 'RSONOSCP_FLAG_ISEPHEMERAL, '
        if (track_flag & RSONOSCP_FLAG_CANREMOVE):
            playback_policies += 'RSONOSCP_FLAG_CANREMOVE, '
        if (track_flag & RSONOSCP_FLAG_CANRESUME):
            playback_policies += 'RSONOSCP_FLAG_CANRESUME, '
        if (track_flag & RSONOSCP_FLAG_NOSEEK):
            playback_policies += 'RSONOSCP_FLAG_NOSEEK, '
        if (track_flag & RSONOSCP_FLAG_ISEXPLICIT):
            playback_policies += 'RSONOSCP_FLAG_ISEXPLICIT, '
        if (len(playback_policies) > 1):
            return playback_policies[:-2]
        else:
            return 'Invalid Playback Policies'

    def test_flags_during_playback(self, zp):
        """
        Function: Play an explicit track
        Expected Outcome: Playback is successfully, the the track's metadata contains the expected flags (and legacy flags for backward compatibility)
        """
        zp.AVTransport.add_to_queue(EXPLICIT_TRACK[0], EXPLICIT_TRACK[1])
        zp.AVTransport.use_queue()
        try:
            zp.AVTransport.Play_and_wait()
        except sonos.exception.UPnPError:
            if (zp.last_upnp_error_code == 701):
                self.logger.warn("Caught UPNP ERROR. Verify that Content Filtering is disabled.")
            else:
                self.logger.warn("Caught UPNP ERROR: {0}", zp.last_upnp_error_code)
        self.verifyTrueOrFailCase(zp.AVTransport.get_transport_state() == 'PLAYING', 'Playback of explicit track initiated')
        explicit_track_flags = self._get_item_flags(str(zp.AVTransport.get_current_track_metadata()))
        explicit_track_legacy = self._get_legacy_flags(explicit_track_flags)
        self.verifyTrueOrWarn(explicit_track_flags == EXPECTED_EXPLICIT_FLAG,
                                  'Expected flags returned for explicit track. Expected: {0}, Returned: {1}'.format(EXPECTED_EXPLICIT_FLAG, explicit_track_flags))
        self.verifyTrueOrFailCase(explicit_track_legacy == TRACK_LEGACY_FLAGS,
                                  'Expected legacy flags returned for explicit track. Expected: {0}, Returned: {1}'.format(TRACK_LEGACY_FLAGS, explicit_track_legacy))
        self.logger.info("Explicit track flags: " + self._get_smapi_playback_policies(flags=explicit_track_flags))
        self._clear_queue()
        zp.AVTransport.add_to_queue(CLEAN_TRACK[0], CLEAN_TRACK[1])
        zp.AVTransport.use_queue()
        zp.AVTransport.Play_and_wait()
        self.verifyTrueOrFailCase(zp.AVTransport.get_transport_state() == 'PLAYING', 'Playback of explicit track initiated')
        clean_track_flags = self._get_item_flags(str(zp.AVTransport.get_current_track_metadata()))
        clean_track_legacy = self._get_legacy_flags(clean_track_flags)
        self.verifyTrueOrWarn(clean_track_flags == EXPECTED_CLEAN_FLAG,
                                  'Expected flags returned for clean track. Expected: {0}, Returned: {1}'.format(EXPECTED_CLEAN_FLAG, clean_track_flags))
        self.verifyTrueOrFailCase(clean_track_legacy == TRACK_LEGACY_FLAGS,
                                  'Expected legacy flags returned for clean track. Expected: {0}, Returned: {1}'.format(TRACK_LEGACY_FLAGS, clean_track_legacy))
        self.logger.info("Clean track flags: " + self._get_smapi_playback_policies(flags=clean_track_flags))

    def test_explicit_sonos_playlist(self, zp):
        """
        Function: Create a new Sonos Playlist that contains an explicit track
        Expected Outcome: The Sonos Playlist is successfully created, the track's resMD contains the expected flags
        """
        zp.AVTransport.add_to_queue(EXPLICIT_TRACK[0], EXPLICIT_TRACK[1], track_num=1)
        zp.AVTransport.add_to_queue(CLEAN_TRACK[0], CLEAN_TRACK[1], track_num=2)
        playlistId = zp.AVTransport.SaveQueue('Explicit Track')
        self.verifyTrueOrFailCase(zp.ContentDirectory.does_sonos_playlist_exist(playlistId), 'Playlist created')
        favorite = zp.ContentDirectory.get_sonos_playlist(playlistId)
        explicit_track_flags = self._get_item_flags(str(favorite[0].element.attrib['id']), playlist=True)
        explicit_track_explicit_bit = self._get_playlist_explicit_bit(str(favorite[0].element.attrib['id']))
        explicit_track_legacy = self._get_legacy_flags(explicit_track_flags)
        clean_track_flags = self._get_item_flags(str(favorite[1].element.attrib['id']), playlist=True)
        clean_track_explicit_bit = self._get_playlist_explicit_bit(str(favorite[1].element.attrib['id']))
        print (str(favorite[0].element.attrib['id']))
        print (str(favorite[1].element.attrib['id']))
        self.verifyTrueOrFailCase(explicit_track_legacy == TRACK_LEGACY_FLAGS,
                                  'Expected legacy flags returned for explicit track. Expected: {0}, Returned: {1}'.format(TRACK_LEGACY_FLAGS, explicit_track_legacy))
        self.verifyTrueOrWarn(explicit_track_flags == EXPECTED_EXPLICIT_FLAG,
                          'Expected flags returned for explicit track. Expected: {0}, Returned: {1}'.format(EXPECTED_EXPLICIT_FLAG, explicit_track_flags))
        self.verifyTrueOrWarn(explicit_track_explicit_bit == '1',
                                  'Playlist track explicit bit returned. Expected: {0}, Returned: {1}'.format('1', explicit_track_explicit_bit))
        self.verifyTrueOrFailCase(clean_track_flags == EXPECTED_CLEAN_FLAG,
                                  'Expected flags returned for clean track. Expected: {0}, Returned: {1}'.format(EXPECTED_CLEAN_FLAG, clean_track_flags))
        self.verifyTrueOrFailCase(clean_track_explicit_bit == '0',
                                  'Playlist track clean bit returned. Expected: {0}, Returned: {1}'.format('0', clean_track_explicit_bit))
        self._clear_queue()
        zp.AVTransport.use_sonos_playlist(playlistId)
        zp.AVTransport.use_queue()
        zp.AVTransport.Play_and_wait()
        self.verifyTrueOrFailCase(zp.AVTransport.get_transport_state() == 'PLAYING', 'Playback of explicit track initiated')
        explicit_track_flags = self._get_item_flags(str(zp.AVTransport.get_current_track_metadata()))
        explicit_track_legacy = self._get_legacy_flags(explicit_track_flags)
        self.verifyTrueOrWarn(explicit_track_flags == EXPECTED_EXPLICIT_FLAG,
                                  'Expected flags returned (Playback). Expected: {0}, Returned: {1}'.format(EXPECTED_EXPLICIT_FLAG, explicit_track_flags))
        self.verifyTrueOrFailCase(explicit_track_legacy == TRACK_LEGACY_FLAGS,
                                  'Expected legacy flags returned for explicit track. Expected: {0}, Returned: {1}'.format(TRACK_LEGACY_FLAGS, explicit_track_legacy))
        zp.ContentDirectory.remove_sonos_playlist(playlistId)
