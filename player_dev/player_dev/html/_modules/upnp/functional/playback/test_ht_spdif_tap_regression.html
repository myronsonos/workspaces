    # """
    # Performs on-device playback of regression tap library and outputs
    # captured dspout taps on each ZP, plus anacapa.trace for each ZP.
    # Audiotap regression library is located on Perforce at:
    # //audio-engineering/test/audiotap/
    #
    # However, this script expects the files to be mirrored on the Automation
    # NAS drive (this is a manual process that could be incorporated in a
    # future revision)
    #
    # Each audiotap file is located in either an S9 (Playbar) or S11 (Playbase)
    # folder, and recursively inside a folder labeled with the JIRA ticket.
    # In addition, a .json file must be present for each audiotap which
    # describes the samplerate, JIRA ID and further data which to date is
    # not used. The intention is to use the .json metadata about each file
    # to describe characteristics of the signal at various stages throughout
    # the duration of the capture. This will help us add future automation
    # around signal analysis for each audiotap test case.
    #
    # AC3 clock files are used to drive the SPDIF input on the target
    # ht_master ZP, which helps OSX users who would otherwise need to
    # manually change their audio interface sample rate settings.
    # The clock files are needed to start playback of the local audiotap on
    # a ZP, as described here:
    # https://confluence.sonos.com/pages/viewpage.action?pageId=5243181
    # """

    def _cleanup(self, zp, spdif_silence_threshold=10):
        """
        Clear avt and queue, remove bonded config if it exists
        :param zp: Sonos zone component
        """
        self.logger.info("Performing cleanup")
        zp.DeviceProperties.cleanup_bonded_configuration()
        zp.AVTransport.clear_play_state(timeout_standalone_seconds=30)
        time.sleep(spdif_silence_threshold)
        zp.AVTransport.clear_queue_and_wait()
        zp.DeviceProperties.SetAutoplayRoomUUID('')

    def _check_for_NAS_connection(self, files_src_path):
        """
        Checks to make sure NAS path is available for file access
        @param files_src_path: path to audiotap regression files
        @type files_src_path: string
        """
        assert os.path.exists(files_src_path), \
            "NAS mapping not established for {}".format(
                SHARE_PATH)

    def _scrape_tap_files_and_metadata(self, files, root):
        """
        Searches through specified location for audiotap file names,
        then retrieves associated metadata files, populates audiotap
        metadata for each audiotap found
        :param files: items in target folder / sub-folders
        :param root: parent directory containing files
        :return:
        """
        for item in files:
            # look for tap files:
            itemURL = os.path.join(root, item)
            itemURL_extension = os.path.splitext(item)[1]

            if item.startswith('.') or not os.path.isfile(itemURL) or not (
                    itemURL_extension in AUDIOTAP_FILE_EXTENSIONS):
                continue
            itemMetadataURL = os.path.join(AUDIOTAP_METADATA_PATH,
                                           item.replace(itemURL_extension,
                                                           ".json"))
            self.logger.info("Checking json file for {} "
                             "against {}".format(
                                itemURL,
                                itemMetadataURL))
            assert os.path.isfile(itemMetadataURL)
            itemMetadata = json.load(open(itemMetadataURL))
            jiraid = str(itemMetadata["jira"]["id"])
            model = str(itemMetadata["target"]["model"])
            samplerate = int(itemMetadata["metadata"]
                             ["samplerate"])
            # calculate file duration
            duration = math.ceil(
                os.path.getsize(itemURL) / (AUDIOTAP_BIT_DEPTH_IN_BYTES
                                            * AUDIOTAP_NUM_CHANNELS
                                            * samplerate))
            # add file details to dict
            self.DD_REGRESSION_TAP_TEST_FILES.update([
                    (item, {'Source': model,
                            'SampleRate': samplerate,
                            'JiraID': jiraid,
                            'Duration': duration})
                ]
            )

    def _setup_webserver(self, resource_path, port=WEBSERVER_PORT):
        """
        Setup local web server
        :param resource_path: location of files to be served
        :param port: port number for http access
        :return: webserver instance
        """
        websrv = WebServer(start_reactor=False)
        try:
            websrv.start(port, resource_path)
        except TimeoutError:
            self.stop_suite("Could not start web server")
        else:
            self.logger.info("Started web server")
            return websrv

    def _get_SPDIF_clock_track_for_samplerate(self, samplerate):
        """
        Finds the appropriate clock source file corresponding with samplerate
        Confirms file exists
        Returns the file location and metadata
        :param samplerate: requested samplerate
        :return:
        """
        for clockfilename, clockfileinfo in \
                self.SPDIF_CLOCK_FILES.iteritems():
            if clockfileinfo['SampleRate'] == samplerate:
                assert os.path.isfile(clockfilename), \
                    "failed to find SPDIF clock track {}".format(clockfilename)
                return [(clockfilename, clockfileinfo)]
        return None

    def _capture_output_tap_and_trace(self,
                                      tapfilename,
                                      tapfileinfo,
                                      ht_master):
        """
        Start dspout tap capture for each ZP
        Wait for a sufficient amount of time (audiotap duration)
        Stop dspout tap on each ZP
        Download, archive and delete trace files
        Add tap files to archive and tidy up
        :param tapfilename: filename of audiotap file to be played back
        :param tapfileinfo: metadata of audiotap file to be played back
        :param ht_master: zp (HT Master)
        :return:
        """
        # start audiotap capture subprocesses
        zp_proc = []
        audiotap_capture_file_zp_list = []

        self._start_tap_capture_process(ht_master,
                                        tapfileinfo,
                                        tapfilename,
                                        zp_proc,
                                        audiotap_capture_file_zp_list)
        # Wait for audiotap playback to complete
        self.sleep = time.sleep(tapfileinfo['Duration'])
        # kill audiotap capture process(es)
        self._kill_audiotap_capture_processes(zp_proc)
        # create an archive file
        archive_file = self._create_zip_file(ht_master, tapfileinfo, tapfilename)
        assert archive_file is not None, "Failed to create zip archive for " \
                                         "results capture for file: {" \
                                         "}".format(tapfilename)
        # download, archive and delete trace files
        self._archive_trace_file(ht_master,
                                 tapfileinfo,
                                 tapfilename,
                                 archive_file)
        # add tap files to archive and tidy up
        for file in audiotap_capture_file_zp_list:
            assert os.path.isfile(file), \
                "failed to find local version of tap file {}".format(file)
            archive_file.write(file)
            os.remove(file)

    def _start_tap_capture_process(self,
                                   zp,
                                   tapfileinfo,
                                   tapfilename,
                                   proc_array,
                                   audiotap_filename_array):
        """
        Start dspout tap capture for each ZP
        Add dspout tap capture file name to array for archive later
        :param zp: zp
        :param tapfilename: filename of audiotap file to be played back
        :param tapfileinfo: metadata of audiotap file to be played back
        :param proc_array: array for capture processes running across all zps
        :param audiotap_filename_array:  array of dspout audiotap filenames
        :return:
        """
        mac_address = zp.serialNumber
        model_number = zp.modelNumber
        version = zp.version
        audiotap_capture_file_zp = \
            self.AUDIOTAP_CAPTURE_FILE.format(
                version,
                tapfileinfo['JiraID'],
                os.path.splitext(tapfilename)[0],
                mac_address,
                model_number,
                self.config_iter)
        audiotap_capture_cmd_zp = self.AUDIOTAP_CAPTURE_CMD.format(
            zp.ip,
            audiotap_capture_file_zp
        )

        self.logger.info("Beginning audiotap capture on <{}>".format(zp))

        proc_array.append(
            subprocess.Popen(
                audiotap_capture_cmd_zp,
                shell=True,
                stdout=subprocess.PIPE,
                stderr=subprocess.PIPE,
                preexec_fn=os.setsid
            )
        )
        # add to list of tap files for zip archive processing
        audiotap_filename_array.append(audiotap_capture_file_zp)

    def _kill_audiotap_capture_processes(self, proc_array):
        """
        Terminate dspout audiotap capture processes
        :param proc_array: array for capture processes running across all zps
        :return:
        """
        for iter_proc in proc_array:
            os.killpg(iter_proc.pid, signal.SIGTERM)
            iter_proc.terminate()

    def _create_zip_file(self, ht_master, tapfileinfo, tapfilename):
        """
        Creates an archive instance for adding trace and dspout tap files
        :param ht_master: zp (HT Master)
        :param tapfilename: filename of audiotap file to be played back
        :param tapfileinfo: metadata of audiotap file to be played back
        :return:
        """
        return zipfile.ZipFile(
            self.ARCHIVE_FILE.format(
                ht_master.version,
                tapfileinfo['JiraID'],
                os.path.splitext(tapfilename)[0],
                self.config_iter),
            'w') or None

    def _archive_trace_file(self,
                            zp,
                            tapfileinfo,
                            tapfilename,
                            archive_file):
        """
        Add anacapa.trace file to archive instance, tidy up local folder after
        :param zp: zp
        :param tapfileinfo: metadata of audiotap file to be played back
        :param tapfilename: filename of audiotap file to be played back
        :param archive_file:
        :return:
        """
        mac_address = zp.serialNumber
        model_number = zp.modelNumber
        version = zp.version
        trace_log_file = '{}-{}-{}-' \
                         '{}-{}-Config_{}-anacapa.trace'.format(
            version,
            tapfileinfo['JiraID'],
            os.path.splitext(tapfilename)[0],
            mac_address,
            model_number,
            self.config_iter
        )
        trace = zp.cli.get_anacapa_trace()
        # write trace to file
        with open(trace_log_file, 'w') as f:
            f.write(trace)
            self.logger.info("Anacapa trace saved to: "
                             "{}".format(trace_log_file))
        # archive trace in zip file
        assert os.path.isfile(trace_log_file), \
            "{} trace file failed to create locally".format(
                trace_log_file)
        archive_file.write(trace_log_file,
                           compress_type=zipfile.ZIP_DEFLATED)
        # delete/tidy trace file from local drive
        os.remove(trace_log_file)

    def _toggle_anacapa(self, zp):
        """
        Toggles anacapa off then on
        Used to start or stop local audiotap file playback from jffs dir
        :param zp: Zone Player
        """
        zp.cli.stop_anacapa_nicely()
        zp.cli.start_anacapa()
        zp.wait_for_upnp_alive()

    def test_playback_dd_regression_taps(self, ht_master):
        """
        Increment config field for file naming purposes
        Enable SPDIF input for external clock input
        Set AVTransport to play TV
        Iterate through previously scraped list of audiotap files:
            Determine the clock frequency for the AC3 external file
            Stop Anacapa
            Upload the audiotap for this iteration
            Start up Anacapa
            Start playback of external clock reference signal (triggers
                audiotap playback to start)
            Start dspout tap capture for each ZP
            Wait for a sufficient amount of time (audiotap duration)
            Stop dspout tap on each ZP
            Download, archive and delete trace files
            Add tap files to archive and tidy up
        :param ht_master: HT master capable device
        :return:
        """
        ht_master.AVTransport.use_spdif()
        ht_master.AVTransport.Play_and_wait()
        self.config_iter += 1

        for tapfilename, tapfileinfo in \
                self.DD_REGRESSION_TAP_TEST_FILES.iteritems():

            if ht_master.modelNumber != tapfileinfo['Source']:
                continue

            [(spdif_clock_track, spdif_clock_track_meta)] = \
                self._get_SPDIF_clock_track_for_samplerate(tapfileinfo['SampleRate'])

            # make sure we are not already playing an audiotap which is
            # still loaded into memory, because
            # zp.diag.get_toslink_stream_type() would return whatever is in
            # the audiotap, not actual SPDIF input...
            ht_master.cli.command("rm -rf /jffs/audiotap.spdif")
            self._toggle_anacapa(ht_master)

            with MplayerAc3Playback(spdif_clock_track, 'hwac3'):
                wait_until_true(
                    lambda: ht_master.diag.get_toslink_stream_type() ==
                    spdif_clock_track_meta['StreamInfo'] and
                    ht_master.diag.get_toslink_sample_rate() ==
                    spdif_clock_track_meta['SampleRate'],
                    iteration_delay=1.0,
                    timeout_seconds=10.0,
                    reason="Expected SPDIF input Stream Type: {}, "
                           "Actual SPDIF Stream Type: {}".format(
                        spdif_clock_track_meta['StreamInfo'],
                                ht_master.diag.get_toslink_stream_type()))

                # copy SPDIF audiotap AFTER checking the input clock file,
                # otherwise audiotap will start to play and our check will
                # be faulty
                ht_master.cli.command("wget http://{}/{}/{}/{} "
                                      "-O /jffs/audiotap.spdif".format(
                                        self.websrv.host(),
                                        tapfileinfo["Source"],
                                        tapfileinfo["JiraID"],
                                        tapfilename))

                # start local audiotap playback:
                self._toggle_anacapa(ht_master)

                self._capture_output_tap_and_trace(tapfilename,
                                                   tapfileinfo,
                                                   ht_master)
