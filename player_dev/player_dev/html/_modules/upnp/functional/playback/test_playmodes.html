[docs]    def use_queue(self, zp):
        queue_udn = 'x-rincon-queue:%s#0' % zp.raw_udn
        zp.AVTransport.SetAVTransportURI(queue_udn, '')
        wait_until_true(lambda: zp.AVTransport.GetMediaInfo().get('CurrentURI') == queue_udn)


    # Clear Queue

[docs]    def clear_queue(self, zp):
        queue_cleared = False

        try:
            zp.AVTransport.clear_queue()
            wait_until_true(lambda: zp.ContentDirectory.is_queue_empty())
            queue_cleared = True
        except UPnPError:
            self.helpers.exceptions.log_upnp_error(zp)
        except TimeoutError:
            self.logger.error("caught TimeoutError waiting for AVTransport.clear_queue")

        self.verifyTrueOrStop(queue_cleared, "The queue should be cleared.")
        self.logger.debug("Cleared queue %s:" % queue_cleared)

[docs]    def add_tracks_to_the_queue(self, zp, queueLen=4):
        # use_queue() is required to have the queue active. Otherwise you cannot see what is in it!
        self.use_queue(zp)
        trackURI = "x-rincon-buzzer:0"
        for i in range(queueLen):
            self.logger.debug("URI for tracks added to queue %s:" % trackURI)
            queueInfo = zp.AVTransport.add_to_queue(trackURI)
            self.logger.debug("Updated queue info: %s" % queueInfo)

        Number_of_Tracks_In_Queue = zp.AVTransport.get_number_of_tracks_in_queue()
        self.verifyEqualOrStop(int(Number_of_Tracks_In_Queue), int(queueLen), "Number of tracks added to queue should equal queue length.")

        transport_state = zp.AVTransport.GetTransportInfo()

    def test_playmode_normal(self, zone1):
        self.logger.info("Testing on zone %s" % zone1)
        self.clear_queue(zone1)

        self.add_tracks_to_the_queue(zone1)

        zone1.AVTransport.SetPlayMode(zone1.AVTransport.PLAYMODE_NORMAL)
        self.verifyEqualOrFailCase(zone1.AVTransport.get_play_mode(), zone1.AVTransport.PLAYMODE_NORMAL, "Is playmode set to NORMAL? If not fail test case.")

        self.use_queue(zone1)
        zone1.AVTransport.Play()
        wait_until_true(lambda:self.is_playing(zone1))

        self.verifyTrueOrFailCase(self.is_playing(zone1), "Testcase failed as transport state is not set to playing.")


        for j in range(queueLen):
                pos = zone1.AVTransport.get_queue_position()
                self.logger.debug("Queue position %d" % (pos))
                self.logger.debug("Playing track %d" % (pos))

                if (j + 1) < queueLen:
                    self.logger.debug("Skipping to track %d" % (j + 2))
                    zone1.AVTransport.Next()
                    wait_until_true(lambda: zone1.AVTransport.get_queue_position() != pos)
                else:
                    self.logger.debug("Pausing track %d" % (j + 1))
                    zone1.AVTransport.Pause()
                    wait_until_true(lambda: self.is_paused(zone1))
                    self.verifyTrueOrFailCase(self.is_paused(zone1), "Should be paused")

            new_queue_pos = zone2.AVTransport.get_queue_position()
            self.verifyNotEqualOrFailCase(current_queue_pos, new_queue_pos, "queue position did not advance")

    def test_playmode_shuffle(self, zone3):

        self.verifyTrueOrSkip(hasattr(zone3, "ContentDirectory"), "Zone should contain ContentDirectory: %s " % zone3)

        self.clear_queue(zone3)
        self.add_tracks_to_the_queue(zone3)
        tracks = []

        self.logger.info("Testing on zone %s" % zone3)

        zone3.AVTransport.SetPlayMode(zone3.AVTransport.PLAYMODE_REPEAT_ALL)
        self.verifyEqualOrFailCase(zone3.AVTransport.get_play_mode(), zone3.AVTransport.PLAYMODE_REPEAT_ALL, "Is play mode set to REPEAT ALL? If not fail test case.")

        self.use_queue(zone3)

        zone3.AVTransport.Play()
        wait_until_true(lambda: self.is_playing(zone3))
        self.verifyTrueOrFailCase(self.is_playing(zone3), "Testcase failed as transport state is not set to playing.")

        for j in range(queueLen):
            self.logger.debug("loop indicator is %s" % j)
            track_position = zone3.AVTransport.get_queue_position()
            tracks.append(track_position)
            zone3.AVTransport.Next()
            wait_until_true(lambda: (zone3.AVTransport.get_queue_position() != track_position))
            self.logger.debug("Track position in queue %s:" % track_position)

        self.verifyTrueOrFailCase(len(tracks) == queueLen, "expected number of tracks played")
