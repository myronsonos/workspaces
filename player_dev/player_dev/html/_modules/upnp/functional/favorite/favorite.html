    def _constructXMLFromList(self, metadata_index, title_ext=''):
        begin_tag = '<DIDL-Lite xmlns:dc="http://purl.org/dc/elements/1.1/" xmlns:upnp="urn:schemas-upnp-org:metadata-1-0/upnp/" xmlns:r="urn:schemas-rinconnetworks-com:metadata-1-0/" xmlns="urn:schemas-upnp-org:metadata-1-0/DIDL-Lite/">  <item>'

        title_tag = '<dc:title>' + escape(self._favorite_data[metadata_index][0] + title_ext) + '</dc:title>'
        type_tag = '<r:type>' + self._favorite_data[metadata_index][1] + '</r:type>'
        playmode_tag = '<r:playmod>' + 'SHUFFLE' + '</r:playmod>'
        description_tag = '<r:description>' + self._favorite_data[metadata_index][2] + '</r:description>'

        res_tag = '<res protocolInfo="' + self._favorite_data[metadata_index][4] + '">' + escape(self._favorite_data[metadata_index][3] + title_ext) + '</res>'
        resMD_tag = '<r:resMD>' + escape(self._favorite_data[metadata_index][6]) + '</r:resMD>'

        end_tag = '<upnp:albumArtURI>' + escape(self._favorite_data[metadata_index][5]) + '</upnp:albumArtURI><upnp:class>object.itemobject.item.sonos-favorite</upnp:class></item></DIDL-Lite>'

        return begin_tag + title_tag + type_tag + playmode_tag + description_tag + res_tag + resMD_tag + end_tag

    def _constructXML (self, title, type, playmode, description, res, resMD):
        begin_tag = '<DIDL-Lite  xmlns:dc="http://purl.org/dc/elements/1.1/"  xmlns:upnp="urn:schemas-upnp-org:metadata-1-0/upnp/" xmlns:r="urn:schemas-rinconnetworks-com:metadata-1-0/" xmlns="urn:schemas-upnp-org:metadata-1-0/DIDL-Lite/">  <item>'
        title_tag = '<dc:title>' + title + '</dc:title>'
        type_tag = '<r:type>' + type + '</r:type>'
        playmode_tag = '<r:playmod>' + playmode + '</r:playmod>'
        description_tag = '<r:description>' + description + '</r:description>'
        res_tag = '<res>' + res + '</res>'
        resMD_tag = '<r:resMD>' + resMD + '</r:resMD>'

        end_tag = '<upnp:albumArtURI>x-file-cifs://stevesonosmbpro/Music/iTunes/iTunes%20Music%20Library.xml#AlbumArt</upnp:albumArtURI><upnp:class>object.itemobject.item.sonos-favorite</upnp:class></item></DIDL-Lite>'

        return begin_tag + title_tag + type_tag + playmode_tag + description_tag + res_tag + resMD_tag + end_tag
