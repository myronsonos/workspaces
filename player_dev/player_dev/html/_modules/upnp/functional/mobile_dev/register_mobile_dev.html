    def _bulk_add(self, zp, num_devices, name, udn):
        """
        Local helper function to perform a bulk add

        Arguments: Zoneplayer under test, number of mobile devices to add, name, udn

        Returns: None
        """
        for x in range(num_devices):
           device = {'name':"name%s%s" % (name, x), 'udn':"mobile%s%s" % (udn, x), 'ip_and_port':"4.4.4.%s:3401" % x}
           self.verifyEqualOrStop(0, self._register_mobile_device(zp, device),
                    "Registering Mobile Device mobile%s%s last_upnp_error_code should equal 0" % (udn, x))

        self.verifyEqualOrStop(True, self._do_settings_versions_match(zp),
                    "Version number sync for Mobile Device Settings %s %s" % (zp.modelNumber, zp.serialNumber))

    def _do_settings_versions_match(self, zp):
        """
        Verify the version numbers match on the mobiledevsettings.xml

        Arguments: Zoneplayer under test

        Returns: Boolean, do versions match?
        """
        versions_match = False
        try:
            wait_until_true(lambda: self.helpers.settings.are_sequence_elements_equal \
                                ([zp.settings.get_mobile_dev_settings_version() for zp in self.test_devices ]))
            versions_match = True
        except TimeoutError:
            self.logger.error("caught TimeoutError waiting for mobiledev.xml replicated settings versions to match")
        self.logger.debug("settings versions: %s" % [ (zp.serialNumber, zp.settings.get_mobile_dev_settings_version()) for zp in self.test_devices ])
        return versions_match

    def _log_diagnostic(self, ip, filename):
        try:
            url = urllib2.urlopen("http://%s:1400/support/review" % ip)
        except httplib.HTTPException, e:
            self.logger.info("HTTPException opening diagnostic review on %s" % (ip))
        except urllib2.HTTPError, e:
            self.logger.info("HTTPError = %s opening diagnostic review on %s" % (str(e.code), ip))
        except urllib2.URLError, e:
            self.logger.info("URLError = %s opening diagnostic review on %s" % (str(e.reason), ip))
        try:
            output = open(filename + ".xml", 'wb')
            output.write(url.read())
        except IOError:
            self.logger.info("Error writing to file: %s.xml" % filename)
        output.close()

    def _parse_mobiledevsettings(self, zp):
        """
        Use XML accsessor to parse attribute from mobiledevsettings.xml, then build list

        Arguments: Zoneplayer under test

        Returns: List of mobile devices
        """
        stored_devices = zp.settings.get_mobile_dev_settings()
        self.verifyLessOrStop(0, len(stored_devices), "mobiledevsettings.xml should not be empty on %s %s"
                % (zp.modelNumber, zp.serialNumber))
        stored_devices = stored_devices.__getattr__("Setting").get("Value")
        stored_devices = stored_devices.split(',')
        self.verifyLessOrStop(0, len(stored_devices), "Mobile Device List should not be empty %s %s"
                % (zp.modelNumber, zp.serialNumber))
        if stored_devices[len(stored_devices) - 1] == '':
            del stored_devices[len(stored_devices) - 1]
        return stored_devices

    def _register_mobile_device(self, zp, device):
        """
        Register a mobile device

        Arguments: Zoneplayer under test, Mobile Device information

        Returns: UPNP result code
        """
        try:
            zp.ZoneGroupTopology.RegisterMobileDevice(device['name'],
                                                      device['udn'],
                                                      device['ip_and_port'])
            # check if xml is well formed after adding mobile device. If it is
            # not, ElementTree ParseError is raised.
            zp.diag.get_topology()
        except sonos.exception.UPnPError:
            return zp.last_upnp_error_code
        except ParseError as e:
            self.logger.critical("XML Error: {}".format(e))
            self.fail("Topology xml mal-formed while adding mobile device: {}"
                      .format(device))
        else:
            return zp.last_upnp_error_code

    def _re_register_device(self, zp, device):
        """
        Re-register a Mobile Device to test the LRU algorithm.
        Parse the string in persistent storage, then call _register_mobile_device

        Arguments: Zoneplayer under test, mobile device information(udn;name;ip_port)

        Returns: UPNP result code
        """
        device = device.split(';')
        device = {'name':device[1], 'udn':device[0], 'ip_and_port':device[2]}
        self.verifyEqualOrStop(0, self._register_mobile_device(zp, device),
                "Registering Mobile Device %s last_upnp_error_code should equal 0" % (device['udn']))
        self.verifyEqualOrStop(True, self._do_settings_versions_match(zp),
                    "Version number sync for Mobile Device Settings %s %s" % (zp.modelNumber, zp.serialNumber))

        """
        Verify after a max length of 32 registered mobile devices, the list is rotated by aging
        out the oldest device. Also verify file replication

        Arguments: None

        Returns: None
        """
        MAX = 33
        for zp in self.test_devices:
            fail_count = 0
            name = "iphone"
            udn = "rotateudn%s" % zp.serialNumber
            self._bulk_add(zp, MAX, name, udn)

            stored_devices = self._parse_mobiledevsettings(zp)

            for item in range(len(stored_devices)):
                x = MAX - (item + 1)
                search_dev = "mobile%s%s;name%s%s;4.4.4.%s:3401" % (udn, x, name, x, x)
                if not self.verifyInOrFailCase(search_dev, stored_devices[item], "Device added to mobiledevsettings.xml %s %s"
                        % (zp.modelNumber, zp.serialNumber)):
                    fail_count += 1

            if not self.verifyEqualOrFailCase(32, len(stored_devices), "Max of 32 mobile devices in mobiledevsettings.xml %s %s"
                    % (zp.modelNumber, zp.serialNumber)):
                fail_count += 1
            if fail_count > 0:
                self._log_diagnostic(zp.ip, "test_list_rotate_%s" % zp.serialNumber)

[docs]    def test_lru(self):
        """
        LRU Algorithm is used when a device re-registers. If a mobile device registers and it's information is already in
        the list, the list is not modified except if the device moves into the last 25% of the entries (e.g. last 8 if the
        list size is 32).  In that case, the device is moved to the front of the list and the list is saved.

        Arguments: None

        Returns: None
        """
        MAX = 32
        for zp in self.test_devices:
            name = "iphone"
            udn = "lru%s" % zp.serialNumber
            self._bulk_add(zp, MAX, name, udn)
            stored_devices = self._parse_mobiledevsettings(zp)

            # Grab a device in the first 75th percentile
            first_75 = math.floor(len(stored_devices) - len(stored_devices) * .75)
            # Grab a device above the 25th percentile
            last_25 = math.ceil(len(stored_devices) - len(stored_devices) * .24)

            # Re-register device already in first 75% of entries, this should not change location in list
            self._re_register_device(zp, stored_devices[int(first_75) - 1])
            tmp_dev_list = self._parse_mobiledevsettings(zp)
            self.verifyEqualOrFailCase(stored_devices[int(first_75) - 1], tmp_dev_list[int(first_75) - 1],
                    "Dev %s of %s should not change location %s %s" % (first_75, len(stored_devices), zp.modelNumber, zp.serialNumber))

            # Re-register device already in last 25% of entries, this should move to first entry in list
            self._re_register_device(zp, stored_devices[int(last_25) - 1])
            tmp_dev_list = self._parse_mobiledevsettings(zp)
            self.verifyEqualOrFailCase(stored_devices[int(last_25) - 1], tmp_dev_list[0],
                    "Dev %s of %s should shift to location 0 %s %s" % (last_25, len(stored_devices), zp.modelNumber, zp.serialNumber))

        """
        Verify replication of mobiledevsettings.xml and the in memory data structures are populated correctly after
        adding 33 devices to a single zp.

        Arguments: None

        Returns: None
        """
        MAX = 33
        name = "iphone"
        udn = "rep%s" % self.test_devices[0].serialNumber
        self._bulk_add(self.test_devices[0], MAX, name, udn)

        stored_devices = self._parse_mobiledevsettings(self.test_devices[0])
        for zp in self.test_devices:
            # Grab in memory data structure
            media_servers = zp.diag.get_topology_media_servers()
            for dev in media_servers:
                # Filter out non mobile devices
                if "mobile" in dev.get("uuid"):
                    count = 0
                    # One instance of every in memory device and it must be backed to file (mobiledevsettings.xml)
                    for stored_dev in stored_devices:
                        if dev.get("uuid") + ';' in stored_dev:
                            count += 1

                    if not self.verifyEqualOrFailCase(1, count, "Verify single instance of device in memory backed by file %s %s"
                            % (zp.modelNumber, zp.serialNumber)):
                        self._log_diagnostic(zp.ip, "test_replication_%s" % zp.serialNumber)

[docs]    def test_update_udn(self):
        """
        Verify registration of an existing device with a new MobileDeviceName and MobileIPandPort updates the Name
        and IPandPortonly. Verify it does not create duplicate entries with the same MobileDeviceUDN but different
        Name or IPandPort.

        Arguments: None

        Returns: None
        """
        for zp in self.test_devices:
            for x in range(2):
                ip_port = "4.4.4.%s:3401" % x
                device = {'name':"updateudn%s" % x, 'udn':"updateudn%s" % zp.serialNumber
                    , 'ip_and_port':ip_port}
                self.verifyEqualOrStop(0, self._register_mobile_device(zp, device),
                        "Add Mobile Device %s %s" % (zp.modelNumber, zp.serialNumber))

            self.verifyEqualOrStop(True, self._do_settings_versions_match(zp),
                    "Version number sync for Mobile Device Settings %s %s" % (zp.modelNumber, zp.serialNumber))

            # Grab data from mobiledevsettings.xml
            stored_devices = self._parse_mobiledevsettings(zp)
            count = 0
            # Count instances of updated udn, should be one
            for mobiledev in stored_devices:
                if device['udn'] in mobiledev:
                    count += 1
                    dev = mobiledev

            self.verifyEqualOrFailCase(1, count, "Verify a UDN is updated and not duplicated % s %s"
                    % (zp.modelNumber, zp.serialNumber))
            self.verifyInOrFailCase("updateudn1", dev, "Verify MobileDevName string was updated %s %s"
                    % (zp.modelNumber, zp.serialNumber))
