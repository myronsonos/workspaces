    def _start_udp_log(self, zp):
        zp.diag.set_diag_level('sonoscp', 2)
        zp.diag.set_diag_level('upnp', 2)
        zp.diag.set_diag_level('trackplaymonitor', 5)
        zp.diag.start_udp_log()
        zp.diag.watch_udp_log_for_string(
            '<diag:sonoscp,2>', lambda msg: self._log_sonosscp_msg(msg))
        zp.diag.watch_udp_log_for_string(
            '<diag:upnp,', lambda msg: self._log_sonosscp_msg(msg))
        zp.diag.watch_udp_log_for_string(
            '<diag:trackplaymonitor', lambda msg: self._log_sonosscp_msg(msg))
        self._clear_udp_log()

    def _stop_udp_log(self, zp):
        zp.diag.set_diag_level('sonoscp', 1)
        zp.diag.set_diag_level('upnp', 1)
        zp.diag.set_diag_level('trackplaymonitor', 1)
        zp.diag.stop_udp_log()
        self._clear_udp_log()

    def _clear_udp_log(self):
        if len(self._sonosscp_msgs) > 0:
            del self._sonosscp_msgs[:]

    def _log_sonosscp_msg(self, msg):
        self._sonosscp_msgs.append(msg)

    def _remove_customsds(self):
        for customsd in self.execution_configuration.custom_service_descriptors:
            self.logger.debug("Removing {} customsd".format(customsd.name))
            self._first_zone.diag.set_customsd(customsd.sid)

    def _create_customsds(self):
        for customsd in self.execution_configuration.custom_service_descriptors:
            self.logger.debug("Adding {} customsd".format(customsd.name))
            self._first_zone.diag.set_customsd(
                customsd.sid,
                name=customsd.name,
                uri=customsd.uri,
                secure_uri=customsd.secure_uri,
                poll_interval=customsd.poll_interval,
                auth_type=customsd.auth_type,
                caps=['logging', 'playbackLogging'])

    def _add_smapi_accounts(self):
        for customsd in self.execution_configuration.custom_service_descriptors:
            self.logger.debug("Adding {} account".format(customsd.name))
            self._first_zone.SystemProperties.AddAccountX(
                customsd.account_type,
                customsd.username,
                customsd.password)
            if (not self._first_zone.SystemProperties.
                does_service_account_exist(customsd.account_type)):
                self.stop('Service account {} does not exist'.
                          format(customsd.name))

    def _remove_smapi_accounts(self):
        for customsd in self.execution_configuration.custom_service_descriptors:
            self.logger.debug("Removing {} account".format(customsd.name))
            try:
                self._first_zone.SystemProperties.RemoveAccount(
                    customsd.account_type, customsd.username,)
            except:
                self.logger.info('account {} was not found'.
                                 format(customsd.name))

    def _verify_streaming_track(self, zp, expected_final_play_time):
        self.logger.info("_verify_streaming_track")
        self._verify_interim_played_time(zp)
        is_stopped = lambda: zp.AVTransport.is_stopped()
        wait_until_true(is_stopped, timeout_seconds=expected_final_play_time)
        self._verify_final_play_time(zp, expected_final_play_time)

    def _verify_interim_played_time(self, zp):
        # Wait for entries in UDP log
        zp.AVTransport.Play_and_wait(timeout_seconds=10)
        sleep(10)
        interim_search_string = self.get_string_in_anacapa(
            'Report interim played time: uri:')
        interim_seconds = re.search('seconds:\s*(\d+)', interim_search_string)
        actual_interim_play_time = int(interim_seconds and
                                      interim_seconds.group(1))
        self.verifyIsNotNoneOrFailCase(actual_interim_play_time,
            'actual_interim_play_time should not be None')

    def _verify_final_play_time(self, zp, expected_final_play_time=None):
        # Wait for entries in UDP log
        sleep(10)
        final_search_string = self.get_string_in_anacapa(
            'Report final played time: uri:')
        final_seconds = re.search('secs:\s*(\d+)', final_search_string)
        actual_final_play_time = int(final_seconds and
                                     final_seconds.group(1))
        self.verifyIsNotNoneOrFailCase(actual_final_play_time,
            'actual_final_play_time should not be None')
        if expected_final_play_time:
            self.verifyEqualOrFailCase(
                actual_final_play_time, expected_final_play_time,
                'actual_final_play_time should '
                'equal expected_final_play_time')

                                                  timeout_seconds=10)

    def _load_config(self):
        module_dir = os.path.dirname(inspect.getfile(SMAPI_Reporting))
        cfg_file = os.path.join(module_dir, self.DEFAULT_CONFIG_FILE)
        cfg = WorkflowExecutionConfigurationFactory.create_from_file(cfg_file)
        self.execution_configuration.update(cfg)
        self.iheartradio_uri = (
            self.execution_configuration.iheartradio_station.uri)
        self.iheartradio_metadata = (
            self.execution_configuration.iheartradio_station.metadata)
        self.songza_uri = (

    def _set_avtransport_uri_and_verify(self, zp, uri, metadata):
        self._set_avtransport_uri_and_wait(zp, uri, metadata)
        self._verify_avtransport_uri(zp, uri)
        self._verify_avtransport_uri_metadata(zp, metadata)

    def _verify_avtransport_uri(self, zp, expected_uri):
        actual_uri = zp.AVTransport.get_avtransport_uri()
        self.verifyEqualOrStop(expected_uri, actual_uri,
                               'actual_uri should equal expected')
