    def __init__(self):
        """
        Initialize test - parse arguments
        """
        super(TOSLINKHDMIEventReporting, self).__init__(
            parameterized_setup=False)

        # parse command line arguments relative to test case
        parser = argparse.ArgumentParser(
            description="TOSLINK HDMI Event Reporting Automation")

        # get test JSON file
        parser.add_argument("--test_json", dest="test_json",
                            default=DEFAULT_JSON_TESTS,
                            help="path to JSON files containing test cases")

        # get the max number of playback iterations
        parser.add_argument("--max_n_playbacks", dest="max_n_playbacks",
                            default=MAX_N_ITERATIONS_PER_TEST, type=int,
                            help="max number of times to playback something "
                                 "before declaring any missing report items "
                                 "a test case failure.")

        args, unknown_args = parser.parse_known_args()

        # get max number of playbacks
        self.max_n_playbacks = args.max_n_playbacks
        if self.max_n_playbacks < 1:
            self.stop_suite("max_n_playbacks must be at least 1.")

        # set up directory structure
        local_path = os.path.dirname(os.path.abspath(__file__))
        self.resource_path = local_path + '/resource/'
        if not os.path.exists(self.resource_path):
            os.mkdir(self.resource_path)

        # get json file
        if args.test_json:
            self.test_json_directory = args.test_json
            self.read_in_test_JSONs()
            self.download_tracks()

        # test json is necessary for executing test
        if not self.test_jsons:
            raise Exception("Test JSON necessary for executing tests")

[docs]    def get_files_from_nas(self, files_path, files_list=None):
        """
        Copy files from NAS
        :param str files_path: Path to download files from
        :param list files_list: The list of files to download
        """
        self.logger.info("Getting files from {} on NAS".format(files_path))
        with SonosAutomationSFTP() as sftp:
            sftp.chdir(files_path)
            if not files_list:
                files_list = sftp.listdir()
                self.logger.info(
                    "Available files on nas : {}".format(files_list))
            try:
                for a_file in files_list:
                    self.logger.info("Copying file: {}".format(a_file))
                    src = files_path + a_file
                    dst = self.resource_path + a_file
                    self.logger.debug("Source: {}".format(src))
                    self.logger.debug("Destination: {}".format(dst))
                    sftp.get(src, dst)
                    return dst
            except IOError:
                self.stop_suite("Required files not available")

        """
        url = ('http://{}:1400/testpoint?name=mixer.fakeStreamError&error=-4'
               '&stream=0').format(zp.ip)
        r = requests.get(url)
        if r.status_code != 200:
            self.stop(("Error making stream error request to ZP @ IP {} - "
                      "HTTP status code: {}").format(zp.ip, r.status_code))



[docs]    def play_track(self, track, encoding=None):
        """
        play the given track on the mplayer
        :param str track: the filename of the track to play
        :param str encoding: to force an encoding format
        """
        args = [track]
        self.logger.info(
            "Playing back {} via MplayerAc3Playback".format(track))
        if encoding:
            # if encoding has been specified, pass it to the
            # MplayerAc3Playback call. otherwise don't include a None
            # argument in call so as to allow default encoding.
            self.logger.info("using encoding {}".format(encoding))
            args.append(encoding)
        with MplayerAc3Playback(*args) as mp:
            t_prev = 0
            while mp.player.time_pos > t_prev:
                self.logger.info(
                    "Iterating through track, timestamp: {}".format(
                        mp.player.time_pos))
                t_prev = mp.player.time_pos
                time.sleep(1)

[docs]    def compare_error_report_against_json(self, js, error_report):
        """
        method for comparing the expected error report JSON against the
        captured error report
        :param js: the JSON representing the expected error report values
        :param error_report: the actual error report captured from the UDP log
        :returns passed_test: boolean True/False for whether or not all
        criteria were passed for the given report
        """
        passed_test = True
        # go through each error key field
        for key in js:

            # the expected value of that error to have occurred
            expected_value = js[key]["value"]

            # make sure the key is in the error report captured
            if not self.verifyTrueOrWarn(
                key in error_report, "Expected key {} to be in error "
                                     "report.".format(key)):
                passed_test = False

            # flags are used to denote whether or not the error counts in the
            # report are expected to be
            # greater than or less than the count denoted in expected_value
            if "flag" in js[key]:
                flag = js[key]["flag"]
                # validate that captured error count is greater than count
                # given in expected

                if operator.__eq__(flag, "greaterThan"):
                    if not self.verifyTrueOrWarn(
                        error_report[key] > expected_value,
                        "Expected Value for key {} supposed to be "
                        "greater than {}. Is {}".format(
                            key, expected_value, error_report[key])
                    ):
                        passed_test = False

                # validate that captured error count is less than count given
                # in expected
                elif operator.__eq__(flag, "lessThan"):
                    if not self.verifyTrueOrWarn(
                        error_report[key] < expected_value,
                        "Expected Value for key {} supposed to be less"
                        " than {}. Is {}".format(
                            key, expected_value, error_report[key])
                    ):
                        passed_test = False

            # if no flag, do equality comparison
            else:
                if not self.verifyTrueOrWarn(
                    error_report[key] == expected_value,
                    "Expected Value for key {} supposed to be equal "
                    "to {}. Is {}".format(
                        key, expected_value, error_report[key])
                ):
                    passed_test = False

        return passed_test



[docs]    def get_error_report_from_usage_metrics(self, zp):
        """
        get the error report for the given ZP and return it in JSON format
        :param zp: the zone player whose error report we want to retrieve
        :return: JSON object representing error report
        """
        ht_events = self.umh.find_names_in_reports(
            zp.ip, ERROR_REPORT_SUBSTRING)
        self.verifyTrueOrStop(
            len(ht_events) > 0,
            "# HT Events from UsageMetrics > 0. Is {}".format(
                len(ht_events)))

        # error report initialization
        error_report = {}

        # take the most recent event only
        ht_events = [ht_events[-1]]

        # go through each event
        for event in ht_events:
            entries = event.entry
            for entry in entries:
                value = entry.value
                if usage_metrics_utils.is_float(value):
                    value = float(value)
                error_report[entry.key] = value

        return error_report



[docs]    def execute_logger_test(self, zp, track=None,
                            num_error_injections=0,
                            time_between_injections=0.0,
                            expected_error_report=None, encoding=None):
        """
        generic method for executing a test on the zp for the logger
        :param zp: the zone player to execute the test on (should be ht master)
        :param str track: the audio track file to playback through the device
        :param int num_error_injections: the number of stream error injections
        to inject
        :param float time_between_injections: the number of seconds between
        stream error injections
        :param json expected_error_report: the expected error report JSON to be
        used for further validations
        :param str encoding: a string to forward to mplayer to force a certain
        decoder for playback
        :returns passed_test: boolean True/False for whether or not all
        criteria was passed for the given test
        """
        passed_test = False
        # enter a HT session
        self.start_ht_session(zp=zp)

        # play track
        if track:
            self.play_track(track, encoding)
        else:
            raise Exception("Need track in order to proceed with test")

        # trigger stream error
        for i in range(num_error_injections):
            self.trigger_stream_error(zp)
            time.sleep(time_between_injections)

        # end HT session
        self.stop_ht_session(zp)

        # flush usage metrics
        self.verifyTrueOrFailCase(
            zp.diag.flush_usage_metrics_events(),
            "Flushing events succeeded")

        # get report from the UsageMetrics object
        self.wait_for_name_in_reports(zp.ip, ERROR_REPORT_SUBSTRING)
        error_report = self.get_error_report_from_usage_metrics(zp)

        # go through and do additional validations if expected error report
        # included
        if expected_error_report:
            passed_test = self.compare_error_report_against_json(
                js=expected_error_report, error_report=error_report)

        return passed_test

    def test_SPDIF_HDMI_logging(self, master, test_json):
        """
        primary test method - goes through the given JSON file and evaluates
        each test based on parameters.
        :param master: HT master
        :param test_json: the JSON containing the test case information
        """
        self.umh.metrics_event_clear(master.ip)
        self.umh.save_ip = master.ip
        # initialize diag levels where they need to be
        master.diag.set_diag_level(REPORTING_DIAG_LEVEL, 2)
        master.disable_speaker_audio()

        # cleanup bonded configuration
        master.DeviceProperties.cleanup_bonded_configuration()

        # TODO: stream error injection is currently broken. confirm when fix
        # SWPBL-88590 is implemented.

        # execute tests
        for test in test_json['tests']:

            # initialize values as None
            (notes, track, num_error_injections,
                time_between_injections, track_url, encoding,
                expected_error_report) = (None, None, None, None, None,
                                          None, None)

            # if the model number of the current device is not supported in
            # the test, skip it
            if 'supported_model_numbers' in test:
                if master.modelNumber not in test['supported_model_numbers']:
                    self.logger.info("Skipping test since model {} not "
                                     "supported".format(master.modelNumber))
                    continue

            # parse from JSON
            if 'track' in test:
                track = test['track']
            else:
                raise Exception("Need track in order to proceed with test")
            if 'encoding' in test:
                encoding = test['encoding']
            if 'num_stream_error_injections' in test:
                num_error_injections = test['num_stream_error_injections']
            if 'time_between_stream_error_injections' in test:
                time_between_injections = (
                    test['time_between_stream_error_injections'])
            if 'notes' in test:
                notes = test['notes']
                self.logger.info("-----------{}------------".format(notes))
            if 'expected_error_report' in test:
                expected_error_report = test['expected_error_report']

            # iterate through each test a certain number of times
            # until the expected behavior is achieved or until the max
            # number of iterations is exceeded
            passed_test = False
            n_iterations = 0
            while not passed_test and n_iterations < self.max_n_playbacks:

                # execute the test
                passed_test = self.execute_logger_test(
                    zp=master,
                    track=track, num_error_injections=num_error_injections,
                    time_between_injections=time_between_injections,
                    expected_error_report=expected_error_report,
                    encoding=encoding)

                # clear the events for next time around
                self.umh.metrics_event_clear(master.ip)

                n_iterations += 1

            self.verifyTrueOrFailCase(
                passed_test, "Passed test after {} "
                             "iterations.".format(n_iterations))
