    """
    Verifies the events submitted by the zone player when the player
    attempts to play back MP3 files which have an unsupported sampling rate.
    It uses a compilation of MP3s which exhibit the erred characteristics.
    """

    TEST_TYPE = ["TRANSPORT_ERRORS_EVENT_REPORTING"]
    SHARE_KEY = "mp3-unsupported-sample-rates"
    """
    The pattern to match on the diagnostic which will be used
    to compile the expected events from the player.
    """
    EVENT_KEY_PATTERN = ur'Transport error ERROR_UNSUPPORTED_FREQ for account type (\d+),(?:.*) share/server: (\S+), (?:.*), extra info: (\d+), (?:.*) framer: (\S+)'
    CHSRC_DIAG_LEVEL = ("chsrc", 2)
    """
    The diagnostic substring to look for when player fails to
    play the audio track.
    """
    CHSRC_TRANSPORT_ERROR_DIAGC = "Transport error ERROR_UNSUPPORTED_FREQ"
    """
    Maximum files that can be handled in a multi-track operation
    Reference all/oc/common/include/upnpcommon.h
    """
    RMAX_URIS_IN_LIST = 16
    """
    A list of test files to enqueue onto player's queue.
    The list is updated dynamically following indexing of the test share.
    """
    PLAYBACK_VECTOR_FILES = []

    def _cleanup(self, zp):
        """
        Sets the zone player's play state to a known state.
        :param obj zp:`sonos.client.internal.SonosZoneComponent`
        """
        zp.AVTransport.leave_group_and_wait()
        zp.AVTransport.clear_queue_and_wait()
        zp.AVTransport.clear_play_state()
        zp.AVTransport.SetPlayMode(zp.AVTransport.PLAYMODE_NORMAL)

    def _start_udp_log(self, zp):
        """
        Set up the required logging, clear the diagnostic
        message error count.
        :param obj zp:`sonos.client.internal.SonosZoneComponent`
        """
        self.mp3_transport_error_count = 0
        zp.diag.start_udp_log()
        zp.diag.set_diag_level(*self.CHSRC_DIAG_LEVEL)
        zp.diag.watch_udp_log_for_string(
            self.CHSRC_TRANSPORT_ERROR_DIAGC,
            lambda msg: self.watch_transport_error(msg))

    def _stop_udp_log(self, zp):
        """
        Stop the UDP logger.

        :param obj zp:`sonos.client.internal.SonosZoneComponent`
        """
        zp.diag.stop_udp_log()

    def _clear_event_keys(self):
        """
        Reset the dictionaries that hold information contained
        in the diagnostic when the player can't play the track.
        The keys in the dictionaries will be used to compile the
        expected events the player should report.
        """
        self.event_framer_type = {}
        self.event_server = {}
        self.event_extra_info = {}
        self.event_acct_type = {}


[docs]    def watch_transport_error(self, msg):
        """
        Call back on UDP logger to count the transport errors.
        Update the appropriate event dictionaries based on the
        diagnostic information.

        :param str msg: Diagnostic string which prompted the call back.
        """
        self.mp3_transport_error_count = self.mp3_transport_error_count + 1
        event_keys = re.search(self.EVENT_KEY_PATTERN, msg)
        if event_keys:
            self.event_acct_type[event_keys.group(1)] = True
            self.event_server[event_keys.group(2)] = True
            self.event_extra_info[event_keys.group(3)] = True
            self.event_framer_type[event_keys.group(4)] = True



[docs]    def wait_on_mp3_transport_error(self, count):
        """
        Waits for the count of diagnostic messages to be reached.

        :param int count: Number to expect.
        """
        try:
            wait_until_true(
                lambda: self.mp3_transport_error_count == count,
                iteration_delay=1,
                timeout_seconds=60)
        except TimeoutError:
            pass
        finally:
            return self.mp3_transport_error_count



[docs]    def play_queue(self, zp, count):
        """
        Plays the player queue and waits for the
        AV transport state to stop and the expected
        number of diagnostic messages.

        :param obj zp:`sonos.client.internal.SonosZoneComponent`
        :param int count: The number of tracks to play
        """
        zp.AVTransport.use_queue()
        zp.AVTransport.Play()
        expected_state = zp.AVTransport.STOPPED_STATE
        self.verifyTrueOrFailCase(
            zp.AVTransport.wait_for_avt_state(expected_state),
            "Expect to enter {} state".format(expected_state))
        self.verifyEqualOrFailCase(
            count,
            self.wait_on_mp3_transport_error(count),
            "Expect exclusion counts to equal")



[docs]    def add_track_list_to_queue(self, zp, selected_tracks):
        """
        Adds the selected tracks to the default player queue.

        :param obj zp:`sonos.client.internal.SonosZoneComponent`
        :param list selected_tracks: List of tracks
        """
        # string of URIs, delimeted by space, used by AddMultipleURIsToQueue
        selected_tracks_uris_string = str(' ').join(selected_tracks)
        # string of track metadata, delimeted by space
        # used by AddMultipleURIsToQueue
        selected_tracks_md = str(' ').join(
            [zp.AVTransport.generate_didl_lite()
             for x in range(len(selected_tracks))])
        queue_data = zp.AVTransport.AddMultipleURIsToQueue(
            num_uris=len(selected_tracks),
            uris=selected_tracks_uris_string,
            metadata=selected_tracks_md,
            container_uri="",
            container_metadata="",
            track_num=0,
            as_next=False)
        return queue_data



[docs]    def compile_expected_events(self):
        """
        Based on the accumulated information derived from the
        player diagnostics, compile the expected events that the
        player should report.
        :return: Tuple of expected events
        :rtype: :obj:`tuple`
        """
        # Minimum expected events from player - built dynamically
        # based on DIAGC from player.
        # status, framer type and share path
        # chsrc.cxx:7709
        # The server path ('srv') reported will be a subset of the URI
        # The extra info ('ex') is the sample rate
        exp_events = []
        exp_events.append(('chsrc:te', 'st', '9'))
        for key in self.event_framer_type.keys():
            exp_events.append(('chsrc:te', 'fr', key))
        for key in self.event_acct_type.keys():
            exp_events.append(('chsrc:te', 'act', key))
        for key in self.event_server.keys():
            exp_events.append(('chsrc:te', 'srv', key))
        for key in self.event_extra_info.keys():
            exp_events.append(('chsrc:te', 'ex', key))
        return tuple(exp_events)



[docs]    @combinatorial('generate_testbed_unique_playback_devices')
    def test_mp3_unsupported_sample_rates(self, zp):
        """
        The test starts with restarting the zone player so that
        the usage event metrics URL override will go into affect.
        It enqueues all the MP3 files from the test share and
        attempts to play the files, monitoring for the expected
        player diagnostic.
        After all the files have been played, it forces the player
        to send all the accumulated usage metrics events for
        analysis (player events buffered but not sent).
        The test compiles a list of expected events with their
        respective values and verifies whether those events
        have been received by the server.

        :param obj zp:`sonos.client.internal.SonosZoneComponent`
        """
        self._cleanup(zp)
        self._clear_event_keys()
        self.verifyTrueOrStop(
            self.wait_for_reporting_subsystems([zp]),
            "Zone reporting subsystem initialized")
        self._start_udp_log(zp)
        start_index = 0
        end_index = 0
        queue_data = None
        for track_index in range(0,
                                 len(self.PLAYBACK_VECTOR_FILES),
                                 self.RMAX_URIS_IN_LIST):
            start_index = track_index
            end_index = start_index + self.RMAX_URIS_IN_LIST
            self.logger.info("Adding track index {} to {}".format(
                    track_index, track_index + self.RMAX_URIS_IN_LIST))
            selected_tracks = self.PLAYBACK_VECTOR_FILES[start_index:end_index]
            queue_data = self.add_track_list_to_queue(zp, selected_tracks)

        self.verifyEqualOrFailCase(
            len(self.PLAYBACK_VECTOR_FILES),
            int(queue_data['NewQueueLength']),
            "Qeueued tracks")
        self.umh.metrics_event_clear(zp.ip)
        self.umh.save_ip = zp.ip
        self.play_queue(zp, len(self.PLAYBACK_VECTOR_FILES))
        self.flush_and_wait_for_events(zp)
        exp_events = self.compile_expected_events()
        self.verify_events_in_reports(exp_events, zp)
        self._stop_udp_log(zp)
        self._cleanup(zp)
