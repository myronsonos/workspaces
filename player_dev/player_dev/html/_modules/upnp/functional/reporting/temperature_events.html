    """
    Temperature based event reporting tests.
    Applicable to zone components that support temperature sensor.
    Performs temperature tests using the spoofing interface and
    verifies the zone component reports the expected values in
    the usage metrics report.
    Add local content playback during temperature reporting
    to make sure the player reports the events on released
    builds (SWPBL-87362).
    """
    TEST_TYPE = ["TEMPERATURE_EVENT_REPORTING"]
    """
    Models which support CPU sensor
    """
    CPU_TEMPERATURE_MODELS = (ENCORE, SOLBASE, S1, S3, S9, SUB, ELREY, ROYALE)
    """
    Models which support AMP sensor
    """
    AMP_TEMPERATURE_MODELS = (ENCORE, SOLBASE, S9, ELREY)
    """
    Define constants which will be reported in the usage metrics
    report when the condition is met.
    The CPU & AMP fault and warning status information from the
    platforms/common/ctl/audioctl.h file
    #define SYSTAT_AUDIO_FAULT_TEMPERATURE_STATUS		0x10000
    #define SYSTAT_AUDIO_WARN_TEMPERATURE_STATUS		0x20000
    #define SYSTAT_CPU_FAULT_TEMPERATURE_STATUS		0x40000
    #define SYSTAT_CPU_WARN_TEMPERATURE_STATUS		0x80000
    """
    AUDIO_FAULT_STATUS = "0x00010000"
    AUDIO_WARN_STATUS = "0x00020000"
    CPU_FAULT_STATUS = "0x00040000"
    CPU_WARN_STATUS = "0x00080000"
    AUDIO_STAT_POLL_INTERVAL = 5

    def __init__(self):
        super(TestTemperatureEventReporting, self).__init__(parameterized_setup=True,
                                                            require_audio=True)

    def _cleanup_play_state(self, zp):
        """
        Clears AVT state on player

        :param :class:`sonos.client.internal.SonosZoneComponent` zp:
        """
        zp.AVTransport.clear_play_state()
        zp.AVTransport.Stop_and_wait()
        zp.AVTransport.clear_queue_and_wait()

        zp.AVTransport.use_queue()

    def _setUpTest(self, zp):
        """
        Restart anacapad so that the override O_EVENTMETRICS_URL takes affect.
        :param :class:`sonos.client.internal.SonosZoneComponent` zp:
        """
        self._cleanup_play_state(zp)
        self.set_diag_level(zp)
        self.verifyTrueOrFailCase(
            zp.diag.flush_usage_metrics_events(),
            "Flushing events during test set up should succeed")
        self.umh.metrics_event_clear(zp.ip)
        self.start_queue_playback(zp)

    def _tearDownTest(self, zp):
        """
        Per DUT unique clean-up.
        Stop UDP logger, clear dmesg to appease programs which look
        for specific keywords on dmesg output.
        :param :class:`sonos.client.internal.SonosZoneComponent` zp:
        """
        self._cleanup_play_state(zp)
        self.umh.save_ip = None
        self.umh.metrics_event_clear(zp.ip)
        zp.diag.stop_udp_log()
        zp.cli.clear_dmesg()

[docs]    def spoof_temperature(self, zp, sensor, temperature, watch_str):
        """
        Wrapper to spoof either the AMP or CPU temperatures.
        The supplied watch string is the subset of the diagnostic
        string which will be generated when the spoofed event is
        digested by anacapad.

        :param :class:`sonos.client.internal.SonosZoneComponent` zp:
        :param str sensor: 'AMP or 'CPU'
        :param int temperature: Temperature in Celcius degrees.
        :param str watch_str: Diagnostic text to wait
        """
        self.clear_watch_str_list(zp)
        self.register_diag(zp, watch_str)
        if sensor == AMP_SENSOR:
            if temperature == 0:
                zp.cli.clear_spoofed_amp_temperature()
            else:
                zp.cli.spoof_amp_temperature(temperature)
        elif sensor == CPU_SENSOR:
            if temperature == 0:
                zp.cli.clear_spoofed_cpu_temperature()
            else:
                zp.cli.spoof_cpu_temperature(temperature)
        # Confirm spoofed temperature - wait for diag to be reported
        return self.wait_on_diag(timeout_seconds=90)



[docs]    def perform_warn_test(self, zp, warn_type, warn_temp, warn_incr):
        """
        Perform warning tests, either CPU or AMP based on the provided warning
        type.
        Spoof the temperatures and verify the expected values are reported
        in the usage metrics report.
        Confirm that a single warning is reported in the usage metrics
        when the warning state persists.
        The warn_incr is a boolean which directs the method to perform a
        subsequent test (with the warning temperature increased by 1 degree)
        following the initial warning temperature test.

        :param :class:`sonos.client.internal.SonosZoneComponent` zp:
        :param str warn_type: 'CPU' or 'AMP'
        :param int warn_temp: Warning temperature
        :param bool warn_incr: Perform test to increase the warning
        temperature ?
        """
        if warn_type == AMP_SENSOR:
            watch_str = AMP_TEMP_WARNING_STRING
            status_value = self.AUDIO_WARN_STATUS
        else:
            watch_str = CPU_TEMP_WARNING_STRING
            status_value = self.CPU_WARN_STATUS

        EXP_EVENTS = (
            ('temperature_volume', warn_type.lower(), str(warn_temp)),
            ('hw:st', 'status', status_value),)
        self.verifyTrueOrFailCase(
            self.spoof_temperature(zp, warn_type, warn_temp, watch_str),
            "Should have reported {} temperature warn spoof".format(warn_type))
        self.verify_events_in_reports(EXP_EVENTS, zp)

        self.umh.metrics_event_clear(zp.ip)

        if warn_incr:
            watch_str_incr = "{},{},".format(warn_type.lower(), warn_temp + 1)
            EXP_EVENTS = (
                ('temperature_volume', warn_type.lower(), str(warn_temp + 1)),)
            self.verifyTrueOrFailCase(
                self.spoof_temperature(zp,
                                       warn_type,
                                       warn_temp + 1,
                                       watch_str_incr),
                "Verified increase in {} warn temperature".format(warn_type))
            self.verify_events_in_reports(EXP_EVENTS, zp)

            # We're not expecting another warning event since only
            # the warning temperature has increased.
            event = ('hw:st', 'status', status_value)
            self.verifyFalseOrFailCase(
                self.umh.find_event_in_reports(zp.ip, event),
                "Expect not to find {} warning event".format(warn_type))
        watch_str = "Leaving hardware warning state"
        self.verifyTrueOrFailCase(
            self.spoof_temperature(zp, warn_type, 0, watch_str),
            "Should have exited {} temperature warn spoof".format(warn_type))



[docs]    def perform_fault_test(self, zp, fault_type, fault_temp):
        """
        Perform fault tests, either CPU or AMP based on the provided fault type.
        Spoof the temperatures and verify the expected values are reported
        in the usage metrics report.
        Confirm that a single fault is reported if the usage metrics
        when the fault state persists.

        :param :class:`sonos.client.internal.SonosZoneComponent` zp:
        :param str fault_type: 'CPU' or 'AMP'
        :param int fault_temp: Fault temperature in Celsius degrees.
        """
        self.logger.info("fault_type={} fault_temp={}".format(
                fault_type,
                fault_temp))

        if fault_type == AMP_SENSOR:
            watch_str = AMP_TEMP_FAULT_STRING
            status_value = self.AUDIO_FAULT_STATUS
        else:
            watch_str = CPU_TEMP_FAULT_STRING
            status_value = self.CPU_FAULT_STATUS

        self.verifyTrueOrFailCase(
            self.spoof_temperature(zp,
                                   fault_type,
                                   fault_temp,
                                   watch_str),
            "Should have reported {} temperature fault spoof".format(fault_type))

        EXP_EVENTS = (
            ('hw:st', 'status', status_value),
            ('temperature_volume', fault_type.lower(), str(fault_temp)),)
        self.verify_events_in_reports(EXP_EVENTS, zp)

        self.umh.metrics_event_clear(zp.ip)

        watch_str = "{},{},".format(fault_type.lower(), fault_temp + 1)
        self.verifyTrueOrFailCase(
            self.spoof_temperature(zp, fault_type, fault_temp + 1, watch_str),
            "Verified temperature increase in {} fault".format(fault_type))

        # Wait AUDIO_STAT_POLL_INTERVAL and then check for
        # the new temperature in the event report
        time.sleep(self.AUDIO_STAT_POLL_INTERVAL)

        EXP_EVENTS = (('temperature_volume',
                       fault_type.lower(),
                       str(fault_temp + 1)),)
        self.verify_events_in_reports(EXP_EVENTS, zp)

        # We're not expecting another fault event since only the
        # temperature has increased.
        event = ('hw:st', 'status', status_value)
        self.verifyFalseOrFailCase(
            self.umh.find_event_in_reports(zp.ip, event),
            "Expect not to find {} fault event".format(fault_type))

        watch_str = "Leaving hardware fault state"
        self.verifyTrueOrFailCase(
            self.spoof_temperature(zp, fault_type, 0, watch_str),
            "Should have exited {} temperature fault spoof".format(fault_type))



[docs]    @combinatorial('cpu_temperature_reporting_zp_generator')
    def test_cpu_temperature_reporting(self, zp):
        """
        On players which support the CPU temperature sensor
        conduct CPU spoofing tests and confirm the usage metrics report
        contains the expected values.
        :param :class:`sonos.client.internal.SonosZoneComponent` zp:
        """
        thresh = zp.cli.get_temperature_thresholds()
        self.perform_fault_test(zp, CPU_SENSOR, thresh.CPUFault)
        if thresh.CPUWarn > 0:
            # Protect against bumping up to the fault state
            bWarnInc = True if (thresh.CPUWarn + 1) < thresh.CPUFault else False
            self.perform_warn_test(zp, CPU_SENSOR, thresh.CPUWarn, bWarnInc)
