[docs]def is_ip(value):
    """
    Is the value an IP/MASK ?
    :param str value:
    """
    try:
        res = re.search("(\d+)(?:\.)(\d+)(?:\.)(\d+)(?:\.)(\d+)", value)
        return (
            (int(res.group(1)) > 0 and int(res.group(1)) < 256) and
            (int(res.group(2)) < 256) and
            (int(res.group(3)) < 256) and
            (int(res.group(4)) < 256)
            )
    except AttributeError:
        return False

[docs]def is_udn(value):
    """
    Is the MAC identifier referenced in the UDN numeric ?
    :param str value: String to validate
    """
    try:
        res = re.search("RINCON_(([0-9]|[A-F])){12}", value)
        return is_numeric(res.group(1), base=16)
    except AttributeError:
        return False

[docs]def validate_key_in_event(event, entry, verifier, logger=None, raise_exc=True):
    """
    Validates entry key in event using the provided function verifier
    Returns a tuple of status, message.
    :param `~sonos.services.common.XMLAccessor` event:
    :param `~sonos.services.common.XMLAccessor` entry:
    :param verifier: method to verify the entry
    :type verifier: :obj:`callable`
    :param `logging.Logger` logger:
    :param bool raise_exc:
    :return: A tuple of boolean status and message string
    :rtype: :obj:`tuple`
    """
    if logger:
        logger.debug("Validating {}".format(entry))
    if verifier:
        result = verifier(entry)
        if not result:
            err = "result={} for {} in {} event".format(result,
                                                        entry,
                                                        event.name)
            if raise_exc:
                raise ValueError("{}".format(err))
            if logger:
                logger.error("Key validation failed {}".format(err))
            return (False, err)
        else:
            return (True, "")
    else:
        warn = "Warning!!! no function verifier for {}={}".format(
            entry.key, entry.value)
        if logger:
            logger.warn(warn)
        return (True, warn)




[docs]def validate_keys_in_report(report, keys_dict, logger=None, raise_exc=True):
    """
    Uses expected keys in dict to validate the keys in the report.
    By default it raises an exception on validation of any key
    that is missing from the report.
    Returns a list of tuples for keys which are missing or have failed
    validation.
    :param `~sonos.services.common.XMLAccessor` event:
    :param dict keys_dict:
    :param `logging.Logger` logger:
    :param bool raise_exc:
    :return: A list of tuples
    :rtype: :obj:`list`
    """
    results = []
    for _key in keys_dict:
        try:
            entry = report.get(_key)
        except AttributeError as err:
            err = "Warning!!! - expected key {} missing".format(_key)
            if logger:
                logger.warn(err)
            if raise_exc:
                raise err
            results.append((False,  err))
            continue
        # We're expecting a single field, but the player may
        # report multiple instances of the key. It's a bug on player,
        # and the following is a defensive way to prevent an error
        # when validating the entry
        if isinstance(entry, list):
            if logger:
                logger.warn(
                    "Found multiple entries for {} key"
                    " using first value {}".format(_key,
                                                   [e.value for e in entry]))
            entry = entry[0]
        function = keys_dict[_key]
        if function:
            try:
                result = function(entry)
                if not result:
                    err = "Failed to validate '{}' key with '{}' value".format(
                                entry.key, entry.value)
                    if logger:
                        logger.fail(err)
                    if raise_exc:
                        raise ValueError(err)
                    results.append((False, err))
            except AttributeError as err:
                err = "AttributeError for key '{}' entry='{}'".format(
                    _key, entry)
                if logger:
                    logger.warn(err)
                if raise_exc:
                    raise AttributeError(err)
                results.append((False, err))
        else:
            err = "Warning!!! no function handler for {}={}".format(
                        entry.key, entry.value)
            if logger:
                logger.warn(err)
    return results

[docs]def validate_event(event, event_dict, logger=None, raise_exc=True):
    """
    Uses the keys from the received event to iterate and verify against
    the given dictionary of expected keys.
    If any key from the received event is not in the expected dictionary,
    the function returns a tuple of False and a message that lists
    the unexpected keys.
    Expands the <entry></entry> contained in an event to a list and
    iterates through each entry verifying the key.
    :param `~sonos.services.common.XMLAccessor` event:
    :param dict event_dict:
    :param `logging.Logger` logger:
    :param bool raise_exc:
    :return: A list of tuples
    :rtype: :obj:`list`
    """
    entries = []
    data = event.get('entry')
    if type(data) is list:
        entries = data
    else:
        entries.append(data)
    unexpected_keys = set(
        [entry.key for entry in entries]).difference(set(event_dict.keys()))
    if unexpected_keys:
        err = "Unexpected keys {} from event {}".format(unexpected_keys,
                                                        event.name)
        if raise_exc:
            raise ValueError("{}".format(err))
        return [(False, err)]
    results = []
    for entry in entries:
        if logger:
            logger.debug("Entry '{}'".format(entry))
        status, msg = validate_key_in_event(
            event,
            entry,
            event_dict[entry.key],
            logger=logger,
            raise_exc=raise_exc)
        if not status:
            results.append((status, msg))
    return results
