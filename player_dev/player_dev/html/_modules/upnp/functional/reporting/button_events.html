    def _setUpTest(self, zp):
        """
        Sets up test
        Restart anacapad so that the override O_EVENTMETRICS_URL takes affect.
        Configure the DUT as a stand alone device with nothing in the queue and
        the play mode set to normal
        Set the volume == 0.
        Set diags reporting=2, buttons=2, chsrc=2, zonereportmgr=5, usagemetrics=5

        :param zp: The zone player being set up
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`
        """
        zp.bounce_anacapa_and_wait()
        zp.AVTransport.leave_group_and_wait()
        zp.AVTransport.clear_queue_and_wait()
        zp.AVTransport.clear_play_state()
        zp.RenderingControl.SetVolume(0)
        zp.AVTransport.use_queue()
        zp.AVTransport.SetPlayMode(zp.AVTransport.PLAYMODE_NORMAL)
        self.diag_triggered = False
        self.set_diag_level(zp)
        zp.diag.watch_udp_log_for_string(
            self.REPORTING_INTERVAL_MSG,
            lambda msg: self.watch_diag(msg))
        self.umh.metrics_event_clear(zp.ip)
        self.umh.save_ip = zp.ip
        self.verifyTrueOrStop(
            self.wait_on_diag(timeout_seconds=60),
            "Should reported interval")
        self.verifyTrueOrFailCase(
            self.flush_and_wait_for_events(zp, timeout_seconds=30),
            "Expect to receive initial events")
        ret = self.umh.validate_event_time_in_reports(zp.ip)
        self.verifyTrueOrFailCase(ret, "All event times validated")
        self.umh.metrics_event_clear(zp.ip)

    def _tearDownTest(self, zp):
        """
        Per DUT unique clean-up

        :param zp: The zone player being set up
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`
        """
        self.umh.save_ip = None
        self.umh.metrics_event_clear(zp.ip)
        zp.RenderingControl.SetVolume(0)
        zp.AVTransport.clear_queue_and_wait()
        zp.AVTransport.clear_play_state()
        zp.diag.stop_udp_log()


[docs]    def min_ms_hold(self, repeats):
        """
        Return the minimum number of ms to hold button to
        honor the number of requested repeats.

        :param int repeats: Button repeats
        :return: The number of ms to hold a button, given the number of repeats
                requested
        :rtype: :obj:`int`
        """
        return (340 + ((repeats - 1) * 160))



[docs]    def expected_volume_from_repeats(self, base, repeats, steps, op=operator.add):
        """
        Return the expected volume from the requested button repeats.
        Accepts a base volume level and computes the ending volume
        level based on the number of step increment(s) and the number
        of repeat(s). Assumes a volume increase.

        :param int base: Base volume.
        :param int repeats: Button repeats.
        :param int steps: Volume increment per repeat.
        :param obj op: Operation to perform - defaults to increase
        :returns: Integer volume level
        :rtype: :obj:`int`
        """
        return op(base, (steps * repeats))



[docs]    def confirm_transport_state(self, function, message):
        """
        Waits until the passed in function returns True. Will time out after 10
        seconds of waiting. At this point, it will mark a failure with the
        reason set to the message passed in. Commonly used in test as a
        wrapper to AVTransport is_playing() and is_paused()

        :param function: A function that returns a boolen
        :type:
        :param message: The message to log when a failure is hit
        :type: obj:`str`
        """
        try:
            wait_until_true(
                lambda: function(),
                iteration_delay=1,
                timeout_seconds=10)
        except TimeoutError:
            self.fail(message)



[docs]    def wait_on_rendering_ctrl_volume(self, zp, expected):
        """
        Waits 10 secs or until the the ZP's volume reaches the
        expected value passed in.

        :param zp: The zone player being set up
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`
        :param expected: The volume level to wait for
        :type expected: :obj:`int`
        """
        try:
            wait_until_true(
                lambda: zp.RenderingControl.GetVolume() == expected,
                iteration_delay=1,
                timeout_seconds=10)
        except TimeoutError:
            pass
        finally:
            return zp.RenderingControl.GetVolume()



[docs]    def volume_ctrl_with_repeat(self, zp, repeat, incr, orig_volume_level,
                                vol_down_str, vol_up_str):
        """
        Spoofs a volume up change followed by a volume down. In both cases, it
        verifies that the volume has changed by the correct amount, based on
        the given number of button repeats expected.

        -- Note:
        Repeat is recognized if a button is held for at least 340ms
        @340ms adds PLAYBACK_VOLUME_INCR to the level
        every 160ms after adds PLAYBACK_VOLUME_INCR to the level
        A hold of 500ms will increase the volume by 2X PLAYBACK_VOLUME_INCR

        :param zp: The zone player being set up
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`
        :param repeat: The number of button repeats to generate
        :type repeat: :obj:`int`
        :param incr: Expected increase in volume per repeat
        :type incr: :obj:`int`
        :param orig_volume_level: The starting volume level
        :type orig_volume_level: :obj:`int`
        :param vol_down_str: The log string to watch for on volume decrease
        :type vol_down_str: :obj:`str`
        :param vol_up_str: The log string to watch for on volume increase
        :type vol_up_str: :obj:`str`
        """

        ms_wait = self.min_ms_hold(repeat)
        expected_volume = self.expected_volume_from_repeats(orig_volume_level, repeat, incr)
        self.spoof_buttons(zp, zp.cli.volume_up, vol_up_str, hold_time=ms_wait)
        self.verifyEqualOrFailCase(
            expected_volume,
            self.wait_on_rendering_ctrl_volume(zp, expected_volume),
            "Repeat increase expected volume level")

        # volume down with repeat
        self.spoof_buttons(zp, zp.cli.volume_down, vol_down_str, hold_time=ms_wait)
        self.verifyEqualOrFailCase(
            orig_volume_level,
            self.wait_on_rendering_ctrl_volume(zp, orig_volume_level),
            "Repeat decrease expected volume level")

    def test_button_reporting(self, zp):
        """
        Verifies that button events are correctly reported with usage metrics

        :param zp: The zone player being set up
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`
        """
        self.test_zones.append(zp)
        self._setUpTest(zp)
        # PLAY/PAUSE with nothing to play generates a PLAYPAUSE event
        self.spoof_buttons(zp, zp.cli.volume_mute, 'Error \(701\) invoking play/pause state')

        zp.AVTransport.add_to_queue([t.media_url for t in self.websrv.http_tracks[0:4]])
        zp.AVTransport.Play_and_wait()
        pb_volume_level = self.PLAYBACK_VOLUME_LEVEL
        zp.RenderingControl.SetVolume(pb_volume_level)

        self.spoof_buttons(zp, zp.cli.volume_mute, 'running pauseRoutine')
        self.confirm_transport_state(zp.AVTransport.is_paused,
                                     "Expected transport state to pause when "
                                     "spoofing mute")

        self.spoof_buttons(zp, zp.cli.volume_mute, 'BUTTON0 - RELEASED')
        self.confirm_transport_state(zp.AVTransport.is_playing,
                                     "Expected transport state to playing when "
                                     "spoofing unmute")

        self.spoof_buttons(zp, zp.cli.volume_up, 'BUTTON1 - RELEASED')
        expected_volume = pb_volume_level + self.VOLUME_INCR_DECR
        self.verifyEqualOrFailCase(
            expected_volume,
            self.wait_on_rendering_ctrl_volume(zp, expected_volume),
            "Expected volume level on spoofed volume UP")

        self.spoof_buttons(zp, zp.cli.volume_down, 'BUTTON2 - RELEASED')
        expected_volume = pb_volume_level
        self.verifyEqualOrFailCase(
            expected_volume,
            self.wait_on_rendering_ctrl_volume(zp, expected_volume),
            "Expected volume level on spoofed volume DOWN")

        # volume up with repeat
        # Repeat is recognized if held for at least 340ms
        # @340ms adds PLAYBACK_VOLUME_INCR to the level
        # every 160ms after adds PLAYBACK_VOLUME_INCR to the level
        # A hold of 500ms will increase the volume by 2X PLAYBACK_VOLUME_INCR
        self.volume_ctrl_with_repeat(
            zp, 2, self.REPEAT_INCR,
            pb_volume_level, 'BUTTON2 - RELEASED', 'BUTTON1 - RELEASED')

        self.volume_ctrl_with_repeat(
            zp, 6, self.REPEAT_INCR,
            zp.RenderingControl.GetVolume(), 'BUTTON2 - RELEASED', 'BUTTON1 - RELEASED')

        self.verifyTrueOrFailCase(
            self.flush_and_wait_for_events(zp, timeout_seconds=30),
            "Expect to receive events")
        self.verify_events_in_reports(self.EXP_EVENTS, zp)
        self._tearDownTest(zp)

    def test_capacitive_button_reporting(self, zp):
        """
        Verifies that cap touch events are correctly reported with usage metrics

        :param zp: The zone player being set up
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`
        """
        self.test_zones.append(zp)
        self._setUpTest(zp)
        # PLAY/PAUSE with nothing to play generates a PLAYPAUSE event
        self.spoof_buttons(zp, zp.cli.volume_mute, 'Error \(701\) invoking play/pause state')

        zp.AVTransport.add_to_queue([t.media_url for t in self.websrv.http_tracks[0:4]])
        zp.AVTransport.Play_and_wait()
        pb_volume_level = self.PLAYBACK_VOLUME_LEVEL
        zp.RenderingControl.SetVolume(pb_volume_level)

        self.spoof_buttons(zp, zp.cli.volume_mute, 'running pauseRoutine')
        self.confirm_transport_state(zp.AVTransport.is_paused,
            "Expected transport state to pause when spoofing mute")

        self.spoof_buttons(zp, zp.cli.volume_mute, 'CAPZONEB - RELEASED')
        self.confirm_transport_state(zp.AVTransport.is_playing,
            "Expected transport state to playing when spoofing unmute")

        if zp.cli.upnp_model_number in MODEL_NUMBER_TO_ORIENTATION and 'right' == zp.cli.get_device_orientation():
            vol_up_str = 'CAPZONEA - RELEASED'
            vol_down_str = 'CAPZONEC - RELEASED'
        else:
            vol_up_str = 'CAPZONEC - RELEASED'
            vol_down_str = 'CAPZONEA - RELEASED'

        self.spoof_buttons(zp, zp.cli.volume_up, vol_up_str)
        expected_volume = pb_volume_level + self.CAPACITIVE_LOW_VOLUME_INCR_DECR
        self.verifyEqualOrFailCase(
            expected_volume,
            self.wait_on_rendering_ctrl_volume(zp, expected_volume),
            "Expected volume level on spoofed volume UP")

        self.spoof_buttons(zp, zp.cli.volume_down, vol_down_str)
        expected_volume = pb_volume_level
        self.verifyEqualOrFailCase(
            expected_volume,
            self.wait_on_rendering_ctrl_volume(zp, expected_volume),
            "Expected volume level on spoofed volume DOWN")

        self.clear_watch_str_list(zp)
        self.verifyTrueOrFailCase(
            self.spoof_track(zp,
                             'next',
                             "Start streaming mp3 track \(2/4\) http://{}/music2.mp3".format(self.websrv.host())),
            "Expect to switch to next track")
        self.confirm_transport_state(zp.AVTransport.is_playing,
                                     "Playing track 2/4")

        self.verify_continuous_playing_state(zp)
        self.verifyTrueOrFailCase(
            self.spoof_track(zp,
                             'previous',
                             "Start streaming mp3 track \(1/4\) http://{}/music1.mp3".format(self.websrv.host())),
            "Expect to switch to previous track")
        self.confirm_transport_state(zp.AVTransport.is_playing,
                                     "Playing track 1/4")

        self.verify_continuous_playing_state(zp)
        self.volume_ctrl_with_repeat(
            zp, 2, self.CAPACITIVE_LOW_VOLUME_INCR_DECR,
            pb_volume_level, vol_down_str, vol_up_str)

        self.volume_ctrl_with_repeat(
            zp, 6, self.CAPACITIVE_LOW_VOLUME_INCR_DECR,
            pb_volume_level, vol_down_str, vol_up_str)

        # The step and repeat change when the playback volume exceeds 16.
        pb_volume_level = self.CAPACITIVE_PLAYBACK_VOLUME_LEVEL
        zp.RenderingControl.SetVolume(pb_volume_level)

        self.volume_ctrl_with_repeat(
            zp, 5, self.CAPACITIVE_VOLUME_INCR_DECR,

        self.verifyTrueOrFailCase(
            self.flush_and_wait_for_events(zp, timeout_seconds=30),

        self.verify_events_in_reports(self.CAPACITIVE_EXP_EVENTS, zp)
        self._tearDownTest(zp)
