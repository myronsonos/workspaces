[docs]    def setup_radio_stations(self, stations):
        """
        Populates a list comprised of the stations to be tested
        :param tuple stations: Stations to search
        """
        self.station_data = []
        for search_type, search_station in stations:
            search_results = self.smapi_client.search(
                search_type, search_station)
            if not search_results:
                self.fail("No hits for {}".format(search_station))
            else:
                self.logger.info(
                    "Found {} hits for {} station".format(
                        len(search_results), search_station))
                for result in search_results:
                    if search_station.lower() in result.title.lower():
                        self.station_data.append(result)
                        break
        return self.station_data

    def test_frame_cache_device_does_not_count_while_paused(self, zp):
        """
        Verify that frame cache capable devices do not report time that was
        spent in the paused state. A Play:1 will likly not be able to
        buffer more than 2 seconds of track data in the milliseconds it
        spends in the playing state. Verify that after 3 seconds of pause
        it doesn't report a fully buffered 10 second track.
        """

        self.setup_add_uri_to_queue_and_play(
            zp, self.websrv.http_tracks[4].media_url)
        zp.AVTransport.Pause_and_wait()
        time.sleep(3)
        self._flush_and_wait(zp)
        EXP_EVENTS = (('chsrc:framed', 'ms', 10000),)
        self.verify_relative_events_in_reports(EXP_EVENTS, zp.ip, operator.le)
        self.collect_perf_counters(zp)



[docs]    @combinatorial('local_unique_players')
    def test_reported_time_no_longer_than_frames_in_track(self, zp):
        """
        Verify devices with >=64mb of memory report a reasonable amount of buffered
        audio that does not include frames cached while paused. This test runs specifically
        against a Play:1 or an Encore.

        Test track for this test is 10sec and 10005 frames. See core/src/sonos/resources/Music5.mp3

        ZP80, ZP90SH4, and ZP100 report incorrect played milliseconds in this test. Decided to skip given players
        are EoL - SWPBL-62633
        """
        if zp.cli.is_sh4():
            self.skip("{} reports incorrect played milliseconds in this test - SWPBL-62633".format(zp.modelNumber))
        self.setup_add_uri_to_queue_and_play(
            zp, self.websrv.http_tracks[4].media_url)
        wait_until_true(lambda: zp.AVTransport.is_stopped(),
                        timeout_seconds=12,
                        iteration_delay=1)
        self._flush_and_wait(zp)
        EXP_EVENTS = (('chsrc:framed', 'type', 'mp3'),
                      ('chsrc:framed', 'ms', '10005'),)
        self.verify_events_in_reports(EXP_EVENTS, zp)



[docs]    def play_tunein_stream(self, zp, station_metadata):
        """
        Play the TuneIn stream given the station metadata
        :param :class:`sonos.client.internal.SonosZoneComponent` zp:
        :param :obj: station_metadata: mediaMetadata from Sonos.wsdl
        """
        self._set_up_test(zp)
        uri, uri_metadata = zp.AVTransport.create_tunein_uri_metadata(
            station_metadata)
        zp.AVTransport.SetAVTransportURI(uri, uri_metadata)
        zp.AVTransport.Play_and_wait()
        avts = AVTransportStateTransitionTimer(zp.AVTransport)
        avts.start(CONTINUOUS_PLAYING_SEC,
                   TRANSITION_TO_PLAYING_SEC,
                   user_defined_start_states=[zp.AVTransport.PLAYING_STATE])
        self.verifyTrueOrFailCase(
            not avts.state_changed,
            "Played continuously {} seconds of radio stream '{}'".format(
                CONTINUOUS_PLAYING_SEC, station_metadata.title.encode('utf-8')))
        zp.AVTransport.Stop_and_wait()
