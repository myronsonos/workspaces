    def _setUpTest(self, zp):
        """
        Restart anacapad so that the override O_EVENTMETRICS_URL takes affect.
        """
        zp.DeviceProperties.cleanup_bonded_configuration()
        zp.bounce_anacapa_and_wait()
        zp.AVTransport.clear_play_state()
        self.diag_triggered = False
        self.set_diag_level(zp)
        zp.diag.watch_udp_log_for_string(
            self.REPORTING_INTERVAL_MSG,
            lambda msg: self.watch_diag(msg))
        self.verifyTrueOrStop(
            self.wait_on_diag(timeout_seconds=60),
            "Should reported interval")
        self.verifyTrueOrFailCase(
            zp.diag.flush_usage_metrics_events(),
            "Flushing events succeeded")
        self.umh.metrics_event_clear(zp.ip)

    def _tearDownTest(self, zp):
        """
        Per DUT unique clean-up
        """
        self.umh.save_ip = None
        self.umh.metrics_event_clear(zp.ip)
        zp.diag.stop_udp_log()

[docs]    def test_s9_orientation_while_playing(self):
        """
        On playbar orientation changes while playing are ignored.
        Note that spoofing a new orientation using the
        cli.set_device_orientation() is avoided since the device won't
        change orientation and the method will attempt to confirm the
        change which will never happen.
        Initially set the device to its natural orientation by sensing detect.
        Go through the remaining orientation permutations and confirm the device
        ignores spoofed orientations while playing. Confirm no orientation events
        while in play mode.
        While in pause state confirm the device honors the last spoofed orientation.
        Confirm the last orientation event is received.
        On exit restore the device's orientation to sense detect.
        """
        self._setUpTest(self.s9)
        self.s9.cli.set_device_orientation('detect')
        natural = self.s9.cli.get_device_orientation()
        self.verifyTrueOrFailCase(
            self.s9.diag.flush_usage_metrics_events(),
            "Flushing events succeeded")
        self.s9.AVTransport.leave_group_and_wait()
        self.s9.AVTransport.clear_queue_and_wait()
        self.s9.AVTransport.clear_play_state()
        self.s9.RenderingControl.SetVolume(1)
        self.s9.AVTransport.use_queue()
        self.s9.AVTransport.SetPlayMode(self.s9.AVTransport.PLAYMODE_NORMAL)
        self.umh.metrics_event_clear(self.s9.ip)
        self.umh.save_ip = self.s9.ip
        self.s9.AVTransport.add_to_queue([t.media_url for t in self.websrv.http_tracks])
        ret = self.s9.AVTransport.Play_and_wait()
        self.verifyTrueOrFailCase(
            self.s9.AVTransport.is_playing(),
            "Expect playbar transport state to be playing")
        self.verifyFalseOrFailCase(
            self.s9.cli.is_orientation_sensing_enabled(),
            "Expecting orientation sensing to be disabled while playing")
        for orientation in self.S9_ORIENTATION_EVENTS.keys():
            if orientation != natural:
                spoofed = orientation
                self.logger.debug(
                    "Attempting to override orientation to {}".format(orientation))
                # Because the Playbar is playing, it will ignore orientation
                # changes hence we can't use the cli.set_device_orientation() call
                self.s9.cli.command("echo 'orientation-reporting={}' > /proc/driver/accel".format(orientation))
                self.verifyEqualOrFailCase(
                    "{} (override)".format(natural),
                    self.s9.cli.get_device_orientation(),
                    "Expect change in orientation to be ignored")
        orientation = spoofed
        # Not expecting to get any orientation event
        self.flush_and_wait_for_events(self.s9, timeout_seconds=30)
        event = ('orientation_change',
                 'new',
                 self.S9_ORIENTATION_EVENTS[orientation])
        unexpected_events = self.umh.find_event_in_reports(self.s9.ip, event)
        self.verifyFalseOrFailCase(
            unexpected_events,
            "Unexpected orientation event while playing {}".format(event))
        # Go into pause mode and confirm orientations are enabled
        self.s9.AVTransport.Pause_and_wait()
        self.verifyTrueOrFailCase(
            self.s9.cli.is_orientation_sensing_enabled(),
            "Expecting orientation sensing to be enabled while in pause")
        # The last spoofed attempt while playing will be exercised in pause mode.
        try:
            wait_until_true(
                lambda: self.s9.cli.get_device_orientation() == "{} (override)".format(orientation),
                iteration_delay=2,
                timeout_seconds=10)
        except TimeoutError:
            self.fail("Expected final orientation {}".format(orientation))
        else:
            self.verifyEqualOrFailCase(
                "{} (override)".format(orientation),
                self.s9.cli.get_device_orientation(),
                "Expected final orientation")

        self.verifyTrueOrFailCase(
            self.flush_and_wait_for_events(self.s9, timeout_seconds=15),
            "Received events after pause")
        event = ('orientation_change',
                 'new',
                 self.S9_ORIENTATION_EVENTS[orientation])
        ret = self.umh.find_event_in_reports(self.s9.ip, event)
        self.verifyTrueOrFailCase(
            ret,
            "Expect {} = {} in report event".format(
                event[1],
                event[2]))
        self.s9.AVTransport.clear_queue_and_wait()
        self.s9.AVTransport.clear_play_state()
        self.s9.cli.set_device_orientation('detect')
        self._tearDownTest(self.s9)

    def test_orientation_reporting(self, zp, orientation_dict):
        """
        """
        self._setUpTest(zp)
        self.umh.metrics_event_clear(zp.ip)
        self.umh.save_ip = zp.ip
        zp.cli.set_device_orientation('detect')
        natural = zp.cli.get_device_orientation()
        for orientation in orientation_dict.keys():
            if orientation != natural:
                self.orientation_test(zp, orientation, orientation_dict)
        zp.cli.set_device_orientation('detect')
        self.verifyEqualOrFailCase(
            natural,
            zp.cli.get_device_orientation(),
            "Restored orientation")
        self.verifyTrueOrFailCase(
            self.flush_and_wait_for_events(zp, timeout_seconds=15),
            "Events received")
        # Build the expected events dynamically
        EXP_EVENTS = []
        EXP_EVENTS.append(
            ('orientation_change',
             'old',
             orientation_dict[natural]))
        for orientation in orientation_dict.keys():
            if orientation != natural:
                EXP_EVENTS.append(
                    ('orientation_change',
                     'new',
                     orientation_dict[orientation]))
        self.verify_events_in_reports(EXP_EVENTS, zp)
        self._tearDownTest(zp)
