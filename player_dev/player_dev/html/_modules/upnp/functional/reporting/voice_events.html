        """
        Convert wav tap file to raw for given zp type
        :param zp: Sonos zone component for which to get voicetap url
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`
        :param str wav_path: Full path of wav file
        """
        self.logger.info("Converting wav to raw voicetap for model: {}".format(
            zp.modelNumber))
        raw_path = os.path.splitext(wav_path)[0]
        converter = WavToRaw(infile=wav_path,
                             outfile_name=raw_path + '.raw',
                             num_mics=zp.diag.get_num_mics(),
                             num_dacs=zp.diag.get_num_dacs(),
                             overwrite=True)
        converter.convert()



[docs]    def get_voicetap(self, resource_path):
        """
        Copy voicetap from NAS
        :param str resource_path: Location where voicetap is stored locally.
        """
        self.logger.info("Getting {} from {}"
                         .format(VOICETAP_WAV, VOICETAP_WAV_PATH))
        wav_path = ''
        if not os.path.exists(resource_path):
            os.mkdir(resource_path)
        with SonosAutomationSFTP() as sftp:
            sftp.chdir(VOICETAP_WAV_PATH)
            voicetaps = sftp.listdir()
            self.logger.info(
                "Available taps on nas : {}".format(voicetaps))
            try:
                for tap in voicetaps:
                    if tap == VOICETAP_WAV:
                        self.logger.info("Copying tap: {}".format(tap))
                        src = VOICETAP_WAV_PATH + tap
                        dst = resource_path + VOICETAP_WAV
                        sftp.get(src, dst)
                        wav_path = dst
            except IOError:
                self.stop_suite("Required tap not available")
            self.logger.debug("Wav path: {}".format(wav_path))
            return wav_path



[docs]    def get_voicetap_url(self, zp):
        """
        Get voicetap file url after hosting on web server
        :param zp: Sonos zone component for which to get voicetap url
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`
        :return str url: Voice tap file url
        """
        local_path = os.path.dirname(os.path.abspath(__file__))
        resource_path = local_path + '/resource/'
        wav_path = self.get_voicetap(resource_path)
        self.convert_wav_to_raw(zp, wav_path)
        websrv = WebServer()
        websrv.start(9000, resource_path)
        url = 'http://{}/{}'.format(websrv.host(), VOICETAP_RAW)
        return url

        """
        Wait for voicetap playback to complete.
        :return: True if diagnostic received else False
        :rtype: :obj:`bool`
        """
        self.diag_triggered = False
        watch_string = 'input overlay file eof'
        zp.diag.watch_udp_log_for_string(
            watch_string,
            lambda msg: self.watch_diag(msg))
        return self.wait_on_diag(20)

[docs]    def verify_all_voiceupload_entries_present(self, entries):
        """
        Verify all entries are present from expected voice entries.
        :param list entries: List of expect voice entries.
        """
        keys = []
        [keys.append(entry.key) for entry in entries]
        key_status = [key in keys for key in self.EXPECTED_VOICE_UPLOAD_KEYS]
        missing = [key for key, status
                   in zip(self.EXPECTED_VOICE_UPLOAD_KEYS, key_status)
                   if not status]
        if missing:
            self.logger.error(
                "Missing entry key in voice uploader report {}".format(missing))
        self.verifyTrueOrStop(all(key_status),
                              "Expect all known entry keys: {} to be present"
                              .format(self.EXPECTED_VOICE_UPLOAD_KEYS))



[docs]    def validate_voiceupload_entries(self, entries, zp):
        """
        Verify entry values for each entry in voice event
        :param list entries: List of voice entries found in event.
        """
        for entry in entries:
            fail = False
            if 'wwd' in entry.key or 'win' in entry.key:
                fail = not is_float(entry.value)
                msg = ("{}: Not expecting '{}' value other than float"
                       .format(entry.key, entry.value))
            elif entry.key == 'service':
                fail = entry.value not in ['avs1', 'avs2']
                msg = ("{}: Expecting '{}' value in ['avs1', 'avs2']"
                       .format(entry.key, entry.value))
            elif entry.key in ['opMs', 'wwDirection', 'samples']:
                fail = not is_numeric(entry.value)
                msg = ("{}: Not expecting '{}' value other than int type"
                       .format(entry.key, entry.value))
            elif entry.key in ['localSilence', 'timeout', 'rcvdResponse']:
                fail = entry.value not in ['false', 'true']
                msg = ("{}: Expecting '{}' value in ['false', 'true']"
                       .format(entry.key, entry.value))
            elif entry.key == 'success':
                # store success for comparison to serviceCode
                success = True if entry.value == 'true' else False
                fail = entry.value not in ['false', 'true']
                msg = ("{}: Expecting '{}' value in ['false', 'true']"
                       .format(entry.key, entry.value))
            # Make sure the event time is parseable in the expected format
            elif entry.key == 'doneTime':
                fail = not isinstance(datetime.datetime.strptime(
                    entry.value, EVENT_TIME_FORMAT), (datetime.datetime,))
                msg = ("{}: Expecting date in format {} entry.value {}"
                       .format(entry.key, EVENT_TIME_FORMAT, entry.value))
            elif entry.key == 'dialogId':
                fail = zp.serialNumber not in entry.value
                msg = "{}: Expecting DUT SN ({}) in entry.value {}".format(
                    entry.key, zp.serialNumber, entry.value)
            elif entry.key == 'eventType':
                fail = 'UtteranceUpload' not in entry.value
                msg = "{}: Expecting entry.value {}".format(
                    entry.key, entry.value)
            elif entry.key == 'serviceCode':
                fail = entry.value == success
                msg = ("{}: should be 0 if success key is True, otherwise "
                       "non-zero: entry.value {}".format(entry.key, entry.value))
            elif entry.key == 'wwEngine':
                fail = entry.value not in WW_ENGINE_STR_CHOICES
                msg = "{}: Expecting entry.value {}".format(
                    entry.key, entry.value)
            elif entry.key == 'wwSensitivity':
                fail = not is_numeric(entry.value)
                msg = ("{}: Not expecting '{}' value other than int type"
                       .format(entry.key, entry.value))
            elif entry.key == 'forced':
                fail = not is_boolean(entry.value)
                msg = ("{}: Not expecting '{}' value other than bool type"
                       .format(entry.key, entry.value))
            else:
                fail = True
                msg = "{} not found in expected voice event keys: {}".format(
                        entry.key, self.EXPECTED_VOICE_UPLOAD_KEYS)
            self.verifyTrueOrFailCase(not fail, msg)



[docs]    def verify_serial_number_in_report(self, zp, cfg_rpt):
        """
        Verify the report is from dut serial
        :param :class:`sonos.services.common.XMLAccessor`cfg_rpt:
            Config report accessor object.
        """
        # The format of the serial number in the report is
        # "aa-bb-cc-dd-ee-ff:X" format.
        # Modify the reported serial number in order
        # to compare with internal expected serial number
        self.logger.debug("{} \"{}\"".format(zp, cfg_rpt))
        temp = cfg_rpt['sn']
        sn = temp[:temp.index(':')].replace('-', '')
        self.verifyEqualOrStop(
            zp.serialNumber, sn, "Expect zone serial number in report")

            "Flushing events succeeded")
        self.wait_for_name_in_reports(zp.ip, VOICE_UPLOAD_EVENT_NAME)
        voice_events = self.umh.find_names_in_reports(
            zp.ip, VOICE_UPLOAD_EVENT_NAME)
        self.verifyGreaterOrStop(
            len(voice_events), 0,
            "Expect at least 1 voice upload event")
        for event in voice_events:
            entries = event.entry
            if entries[0].key == 'eventType' and entries[0].value == 'UtteranceUpload':
                self.verify_all_voiceupload_entries_present(entries)
                self.validate_voiceupload_entries(entries, zp)

    def test_voice_info_in_player_config_report(self, zp):
        """
        Verify number of voice accounts and microphone enabled
        information available in player config report.
        """
        self.voice_overlay = VoiceOverlay(zp)

        self.verifyTrueOrStop(
            self.wait_for_report(zp.ip),
            "Received player config report from {}".format(zp))
        try:
            report = self.umh.get_reports(zp.ip)[0]
            # convert the XMLAccessor object to dictionary
            cfg_rpt = report.player_config_report()
        except IndexError:
                self.fail("{} Expect a single player config report"
                          .format(zp))
        # Verifier
        self.verifyIsNotNoneOrFailCase(
            cfg_rpt, "Expect voice element in player config report")
        self.verify_serial_number_in_report(zp, cfg_rpt)
        voice_accounts = zp.voice_service.get_accounts()
        mic_enabled = 1 if zp.cli.is_mic_on() else 0
        self.verifyTrueOrFailCase(
            all([k in cfg_rpt['voice'] for k in PLAYER_CONFIG_VOICE_KEYS]),
            "Expect all keys to be present")
        self.verifyTrueOrFailCase(
            int(cfg_rpt['voice']['num']) == len(voice_accounts),
            "Expect num voice account to match")
        # Verify the names
        svcs = cfg_rpt['voice']['svcs']
        voice_account_names = [account.service for account in voice_accounts]
        for svc in svcs:
            self.verifyTrueOrFailCase(
                svc.name in voice_account_names,
                "Expect voice service '{}' in report".format(svc.name))
            ww_in_report = False
            for account in voice_accounts:
                # We're expecting either of these two engines
                # where the wake word in the player config report
                # is constructed.
                # Taken from voicewakefilemanager.cxx:VoiceWakeFileManager::formatWakeFile
                if svc.ww is not None:
                    result = re.search(ur"(\S+)\.(\d+)\.(\S+)_(\S+)", svc.ww)
                    if result:
                        ww_in_report = True
                    self.verifyTrueOrFailCase(ww_in_report, "The wakeword variant "
                                              "should be included in the config report")
        self.verifyTrueOrFailCase(
            int(cfg_rpt['voice']['ena']) == mic_enabled,
            "Expect mic info to match")
