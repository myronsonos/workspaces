        self.logger.info("Stop & wait")
        zp.AVTransport.Stop_and_wait()
        self.logger.info("Clear play state")
        # Bump the timeout - in some HH cases > 30 seconds
        zp.AVTransport.clear_play_state(timeout_standalone_seconds=40)
        self.logger.info("Clear queue & wait")
        zp.AVTransport.clear_queue_and_wait()
        self.logger.info("Set volume")
        zp.RenderingControl.SetVolume(0)
        self.logger.info("Use queue")
        # Bump the timeout - in some HH cases > 40 seconds
        zp.AVTransport.use_queue(timeout_seconds=60)
        self.logger.info("Set play mode")
        zp.AVTransport.SetPlayMode("NORMAL")
        self.logger.info("Remove all alarms")
        self.alarm_manager.delete_all_alarms()
        self.logger.info("Clean up complete")

[docs]    def is_alarm_start_ahead(self, t1, t2):
        """
        Returns True when the 24H clock t2 is "older"
        than the 24H clock t1
        """
        h1, m1, s1 = t1.split(":")
        h2, m2, s2 = t2.split(":")
        d1 = int(h1) * 3600 + int(m1) * 60 + int(s1)
        d2 = int(h2) * 3600 + int(m2) * 60 + int(s2)
        return (d1 < d2)

[docs]    def wait_for_alarm_to_fire(self, zp):
        """
        Waits for the current alarm to fire. In the event
        the alarm has been detected not to have fired, retrieve
        all the alarms from the player and issue a warning.
        """
        try:
            wait_until_true(
                lambda: zp.AVTransport.is_alarm_running(),
                timeout_seconds=15,
                iteration_delay=1,
                reason="Alarm has not fired")
        except TimeoutError:
            self.dump_zone_alarms(zp)
            self.logger.info(
                "device <{}> playing alarm {}".format(
                    zp, zp.AVTransport.is_playing()))
            self.warn("Yikes!!! alarm did not fire when it should have")



[docs]    def verifyAlarmFiredAndIsInCorrectState(self, zp, alarmId, sdate,
                                            starttime, stime, endEnabled):
        """
        Verifies expected alarm has fired.
        """
        self.logger.debug(
            "device <{}>; alarmId: {}; sdate: {}; starttime: {}; "
            "stime: {}; endEnabled: {}".format(
                zp, alarmId, sdate, starttime, stime, endEnabled))
        self.logger.info(
            "Setting test date and time to {}, {}".format(sdate, stime))
        starting_time_generation = zp.AlarmClock.TimeGeneration
        self.verifyTrueOrFailCase(
            zp.AlarmClock.SetTimeNow("{} {}".format(sdate, stime), TIMEZONE),
            "Expect SetTimeNow({} {}, {}) to succeed".format(
                sdate, stime, TIMEZONE))
        self.logger.debug("Set start alarm realtime: {}".format(time.time()))
        wait_until_true(
            lambda: zp.AlarmClock.TimeGeneration > starting_time_generation)
        currentTime = zp.AlarmClock.GetTimeNow()
        self.logger.debug("device <{}> time: {}".format(zp, currentTime))
        self.logger.debug("Check is running realtime: %s" % time.time())
        actual_utc_ymd = currentTime['CurrentUTCTime'].split(" ")[0]
        self.verifyEqualOrStop(sdate, actual_utc_ymd,
                               "Expect Year/Month/Day to match")
        self.wait_for_alarm_to_fire(zp)
        zp.GroupRenderingControl.SetGroupMute(True)
        alarmProperties = zp.AVTransport.GetRunningAlarmProperties()
        loggedStartTime = alarmProperties.get(
            "LoggedStartTime").split(' ')[1]
        self.verifyEqualOrFailCase(
            str(starttime),
            str(loggedStartTime),
            "Expected start time should match logged time")
        zp.AVTransport.Stop()
        wait_until_true(lambda: zp.AVTransport.is_stopped())
        wait_until_true(lambda: not zp.AVTransport.is_alarm_running())
        self.verifyEqualOrFailCase(
            endEnabled,
            self.is_alarm_enabled(zp, alarmId),
            "Is alarm enabled")



[docs]    def find_alarm_event(self, zp, exp_alarm):
        """
        Returns a dictionary which has values from the fired alarm.
        """
        alarm_events = self.umh.find_names_in_reports(zp.ip, 'zpAlarm')
        self.logger.debug(
            "Search for alarm {} with start {}".format(
                exp_alarm['id'], exp_alarm['start']))
        for alarm_event in alarm_events:
            entries = alarm_event.get('entry')
            d = {}
            [d.__setitem__(entry.key, entry.value)
             for entry in entries if entry.key in ALARM_KEYS]
            self.logger.debug(
                "Comparing event alarm {} with start {}".format(
                    d['id'], d['start']))
            if d['id'] == exp_alarm['id'] and d['start'] == exp_alarm['start']:
                return d
        return {}



[docs]    def verify_alarm_event(self, zp, exp_alarms):
        """
        Given a list of expected alarms, verify what the player has evented.
        :param list exp_alarms: Expected alarm list
        """
        for exp_alarm in exp_alarms:
            alarm_event = self.find_alarm_event(zp, exp_alarm)
            if alarm_event:
                self.logger.debug(
                    "Found alarmId {}".format(exp_alarm['id']))
                self.logger.debug(
                    "Values from report '{}'".format(alarm_event.values()))
                self.logger.debug(
                    "Expected values '{}'".format(exp_alarm.values()))
                for key in ALARM_KEYS:
                    self.verifyEqualOrFailCase(
                        alarm_event[key],
                        exp_alarm[key],
                        "Comparing '{}'".format(key))
            else:
                self.fail(
                    "Failed to find alarm {}".format(exp_alarm['id']))



[docs]    def verify_alarm_test(self, zp, data):
        """
        For given data, verifies alarm test and events with
        the corresponding information.
        1. Creates alarm
        2. Sets the date & time on player
        3. Confirms alarm has fired
        4. Confirms event information in usage report
        """
        (startTime, duration, recurrence,
         enabled, roomUUID, programURI,
         programMetaData, playMode, actionType, volume,
         hasLinkedZones, setCST, fallBack) = data
        alarmId = "0"
        roomUUID = zp.raw_udn
        alarmCount = self.alarm_manager.num_alarms
        self.verifyEqualOrFailCase(
            0,
            alarmCount,
            "Expect no alarms at the beginning of the test")
        try:
            alarmId = zp.AlarmClock.CreateAlarm(
                startTime, duration, recurrence,
                enabled, roomUUID, programURI,
                programMetaData, playMode, volume, hasLinkedZones)
            self.logger.debug("Created Alarm Id {}".format(alarmId))
        except UPnPError, e:
            self.stop("Unexpected exception when creating alarm. {}".format(e))
        self.verifyGreaterOrStop(int(alarmId), 0, "Expecting new alarm")
        self.verifyTrueOrStop(self.alarm_manager.should_refresh(), "Call 'should_refresh' to verify that the alarm version changed.")
        expectedAlarmCount = alarmCount + 1
        alarmCount = self.alarm_manager.num_alarms
        self.verifyEqualOrFailCase(
            expectedAlarmCount,
            alarmCount,
            "Creating one alarm should increase the number "
            "of alarms in the system by 1")
        # Compile a list of expected alarms
        exp_alarms = []
        if enabled:
            days = ['O']
            rec = recurrence.split('_')
            if len(rec) > 1:
                days = list(rec[1])
            for d in days:
                # ONCE is Monday 2001-1-1
                exp_alarm = {}
                fdate = dates.get(d)
                if self.is_alarm_start_ahead(startTime, setCST):
                    year, month, day = fdate.split("-")
                    next_day = (datetime.date(int(year), int(month), int(day))
                                + datetime.timedelta(hours=24))
                    next_date = "{}-{:02d}-{:02d}".format(
                        next_day.year, next_day.month, next_day.day)
                    exp_alarm['start'] = next_date + " " + startTime
                    self.logger.debug(
                        "Start time {} < set alarm firing on {}".format(
                            startTime, setCST, exp_alarm['start']))
                else:
                    exp_alarm['start'] = fdate + " " + startTime
                exp_alarm['vol'] = volume
                # SWPBL-120952 - players should not send URIs in event reports
                # The keyword 'uri' was removed and in its place the 'fb_buzz'
                # was introduced to signify if the player resorted to playing
                # the buzzer URI when the programmed URI failed.
                exp_alarm['fb_buzz'] = fallBack
                exp_alarm['id'] = alarmId
                exp_alarm['dur'] = None if not duration else duration
                exp_alarm['linked'] = '0' if not hasLinkedZones else '1'
                exp_alarm['eventType'] = actionType
                exp_alarm['mode'] = playMode
                exp_alarms.append(exp_alarm)
                self.verifyAlarmFiredAndIsInCorrectState(
                    zp, alarmId, fdate, startTime,
                    setCST, recurrence != 'ONCE')
        self.flush_event_report(zp)
        self.verifyTrueOrFailCase(
            self.wait_for_name_in_reports(zp.ip, 'zpAlarm'),
            "Received zpAlarm from <{}>".format(zp))
        self.verify_alarm_event(zp, exp_alarms)
