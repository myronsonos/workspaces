        """
        Using the unique playback devices and the desired
        group size, yield all combinations to test the quick GC hand-off.
        """
        players = self.test_players[:]
        random.shuffle(players)
        zps = players[:self.group_size]
        for group in itertools.permutations(zps, self.group_size):
            # The first element - zone player is the GC
            # The remaining elements - zone player members
            yield (GroupConfig((group[0], list(group[1:])),), )

[docs]    def verify_grouping_report_on_old_gc(self, group_cfg, zps_dict):
        """
        Verify zpGrouping report from old GC
        :param :class:`GroupConfig` group_cfg:
        :param dict zps_dict: Dictionary of zpGrouping events with
        zone player as key
        """
        gc = group_cfg.oldGC
        gms = group_cfg.gms
        gm_to_gc = group_cfg.newGC
        self._common_debug_banner(gc, ReportNames.ZP_GROUPING, zps_dict[gc])
        self.verifyGreaterEqualOrStop(len(zps_dict[gc]),
                                      1,
                                      "Minimum zpGrouping reports from old GC")
        event = zps_dict[gc][0]

        self.verifyEqualOrFailCase('DelegateGC',
                                   event['eventType'],
                                   "Expect DelagateGC event type")
        self.verifyEqualOrFailCase(0,
                                   int(event['result']),
                                   "Expect delegation to succeed")
        self.verifyEqualOrFailCase(gc.raw_udn,
                                   event['oldgc'],
                                   "Expect oldgc raw UDN to match")
        self.verifyEqualOrFailCase(gm_to_gc.raw_udn,
                                   event['newgc'],
                                   "Expect newgc raw UDN to match")
        # The group ID is a unique text generated by the GC
        self.verifyEqualOrFailCase(gc.raw_udn,
                                   event['gid'].split(":")[0],
                                   "Expect group id to match")
        self.verifyEqualOrFailCase(len(gms) + 1,
                                   int(event['members']),
                                   "Expect the size of the group to match")

[docs]    def verify_chsnk_report_on_new_gc(self, group_cfg, event_dict):
        """
        Verify chsnk framed report on new GC
        :param :class:`GroupConfig` group_cfg:
        :param dict event_dict: Dictionary of events per zone player
        """
        zps_dict = event_dict[ReportNames.CHSNK_FRAMED]
        gm_to_gc = group_cfg.newGC
        self._common_debug_banner(gm_to_gc, ReportNames.CHSNK_FRAMED, zps_dict[gm_to_gc])
        self.verifyGreaterEqualOrFailCase(len(zps_dict[gm_to_gc]),
                                          3,
                                          "Minimum chsnk_framed reports on new GC")
        # The first chsnk_framed report of the new GC reflects that it was a member
        event = zps_dict[gm_to_gc][0]
        self._verify_first_chsnk_framed_report(zps_dict[gm_to_gc][0],
                                               group_cfg,
                                               event_dict,
                                               'new GC')
        # Point to the second chsrc_framed report on new GC
        event = zps_dict[gm_to_gc][1]
        chsrc = event_dict[ReportNames.CHSRC_FRAMED][gm_to_gc][0]
        self._verify_common_chsnk_report_from_gc(gm_to_gc, event, chsrc, 'new GC')
        self.verifyEqualOrFailCase(gm_to_gc.raw_udn,
                                   event['audioSrcUUID'],
                                   "Expect new GC audio source UUID to match")

        # Verify chsnk report on line-in, it's the last report
        event = zps_dict[gm_to_gc][-1]
        try:
            self.verifyEqualOrFailCase('true',
                                       event['isGC'],
                                       "Expect new GC to report as GC")
        except KeyError:
            self.fail("Missing 'isGC' key in chsnk report from <{}>".format(gm_to_gc))
        self._verify_common_chsnk_line_in(event, 'new GC', 'sbc')

        """
        Verify chsnk report on old GC
        1. The framer type on chsnk should match the framer type

        zps_dict = event_dict[ReportNames.CHSNK_FRAMED]

        self.verifyGreaterEqualOrFailCase(len(zps_dict[gc]),
                                          2,
                                          "Minimum chsnk_framed reports on old GC")
        # Since we're playing two tracks on the old GC,
        # we're expecting two channel source and two channel sink reports
        for idx in range(2):
            event = zps_dict[gc][idx]
            chsrc = zps_dict_chsrc[gc][idx]
            self._verify_common_chsnk_report_from_gc(gc, event, chsrc, 'old GC')


    def _verify_first_chsnk_framed_report(self, event, group_cfg, event_dict, origin):
        """
        Verifies the first chsnk_framed report from a member which includes
        the new GC which was a member prior to delegation.
        Confirms the member reports as GM, the audio source maps to the old GC
        and the framer played reflects the old GC's framer.
        """
        gc = group_cfg.oldGC
        zps_dict_chsrc = event_dict[ReportNames.CHSRC_FRAMED]
        # The member's first report should align with the first GC's info
        try:
            self.verifyEqualOrFailCase('true',
                                       event['isGM'],
                                       "Expect {} to report as GM".format(origin))
        except KeyError:
            self.fail("Missing 'isGM' in chsnk_framed report from {}".format(origin))

        self.verifyEqualOrFailCase(gc.raw_udn,
                                   event['audioSrcUUID'],
                                   "Expect {} to report correct source".format(origin))
        # The framer type the member reports, is verified by cross referencing the
        # channel source report from the GC.
        # The GC's first channel source report is the reference
        # The channel sink reports a framer, but in some instances it's not
        # an exact match with what the channel source reported.
        # For example, channel source reports framer m4a-m4a while the channel
        # sink reports just m4a
        self.confirm_framer_type(event['fr'], zps_dict_chsrc[gc][0]['type'])

        """

[docs]    def play_and_pause_line_in_source(self, gc, line_in_zp):
        """
        Playing line-in on the GC should generate a ChangeTransport event
        The compressed mode assures that the chsnk framer reports the framer
        type as 'sbc'
        :param :class:`sonos.client.internal.SonosZoneComponent` gc:
        :param :class:`sonos.client.internal.SonosZoneComponent` line_in_zp:
        """
        for encoding in ("COMPRESSED", ):
            with MplayerMp3LineOutPlayback(LINE_IN_AUDIO_TRACK):
                gc.SystemProperties.set_line_in_encoding(encoding)
                gc.AVTransport.use_line_in(line_in_zp)
                gc.AVTransport.Play_and_wait()
                self.verify_continuous_playing_state(gc)
                gc.AVTransport.Pause_and_wait()


    def _build_expected_names_report(self, group_cfg):
        """
        Builds a dictionary of expected name reports based on role
        :param `GroupConfig` group_cfg:
        :return: A dictionary of expected name reports per zone
        :rtype: :obj:`dict`
        """
        expected_name_reports = {}
        expected_name_reports[ReportNames.ZP_GROUPING] = {}
        expected_name_reports[ReportNames.CHSNK_FRAMED] = {}
        expected_name_reports[ReportNames.CHSRC_FRAMED] = {}
        for zp in group_cfg.gms:
            expected_name_reports[ReportNames.ZP_GROUPING][zp] = 2
            expected_name_reports[ReportNames.CHSNK_FRAMED][zp] = 4
        expected_name_reports[ReportNames.ZP_GROUPING][group_cfg.oldGC] = 1
        expected_name_reports[ReportNames.CHSNK_FRAMED][group_cfg.oldGC] = 2
        expected_name_reports[ReportNames.CHSRC_FRAMED][group_cfg.oldGC] = 2
        expected_name_reports[ReportNames.ZP_GROUPING][group_cfg.newGC] = 2
        expected_name_reports[ReportNames.CHSNK_FRAMED][group_cfg.newGC] = 3
        expected_name_reports[ReportNames.CHSRC_FRAMED][group_cfg.newGC] = 1
        return expected_name_reports

    def test_grouping_event_reports(self, group_cfg):
        """
        1. Start playback on GC X
        2. Group players to GC
        3. Confirm playback on GC X
        3(a). Next track on GC X
        3(b). Confirm playback on GC X
        4. Delegate to GC Y
        5. Confirm playback on GC Y
        5(a). Next track on GC Y
        5(b). Confirm playback on GC Y
        6. Play line-in on GC Y
        7. Remove players from group
        8. Verify reports from all players
        :param :class:`GroupConfig` group_cfg:
        """
        gc = group_cfg.oldGC
        gms = group_cfg.gms
        gm_to_gc = group_cfg.newGC
        self.logger.info("Linking zones to GC: <{}>".format(gc))
        self.logger.info("Members <{}>".format(gms))
        self.logger.info("Player <{}> will acquire role as GC".format(gm_to_gc))
        self.logger.info("Starting playback on GC <{}>".format(gc))
        self.start_queue_playback(gc, track_to_play=1)
        self.logger.info("Grouping {} to GC <{}>".format(gms, gc))
        gc.AVTransport.link_zones(gms, timeout=120)
        self.logger.info("All players grouped ... waiting")
        self.confirm_playback_on_gc(gc)
        self.logger.info("<{}> leaving group".format(gc))
        gc.AVTransport.BecomeCoordinatorOfStandaloneGroup(
            timeout_seconds=STANDALONE_TRANSITION_TIMEOUT)
        gc.AVTransport.SetAVTransportURI_and_wait("", "")
        self.confirm_playback_on_gc(gm_to_gc)
        group_cfg.remove_gm(gm_to_gc)
        self.play_and_pause_line_in_source(gm_to_gc, self.line_in_players[0])
        # Members leave group, expect an AddMember and a RemoveMember report
        [gm.AVTransport.leave_group() for gm in gms]
        # Expect the zpGrouping event from all players
        zps_dict_group = {}
        zps_dict_chsnk = {}
        zps_dict_chsrc = {}
        event_dict = {ReportNames.ZP_GROUPING: zps_dict_group,
                      ReportNames.CHSNK_FRAMED: zps_dict_chsnk,
                      ReportNames.CHSRC_FRAMED: zps_dict_chsrc}
        expected_name_reports = self._build_expected_names_report(group_cfg)
        for zp in group_cfg.all_zps:
            self.flush_event_report(zp)
            zps_dict_group[zp] = self.harvest_name_reports(zp,
                                                           ReportNames.ZP_GROUPING,
                                                           expected_name_reports)
            zps_dict_chsnk[zp] = self.harvest_name_reports(zp,
                                                           ReportNames.CHSNK_FRAMED,
                                                           expected_name_reports)
            if zp in (gc, gm_to_gc, ):
                zps_dict_chsrc[zp] = self.harvest_name_reports(zp,
                                                               ReportNames.CHSRC_FRAMED,
                                                               expected_name_reports)

        for zp in (gc, gm_to_gc, ):
            self.logger.debug("GC <{}> chsrc reports".format(zp))
