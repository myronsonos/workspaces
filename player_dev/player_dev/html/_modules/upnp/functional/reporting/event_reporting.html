[docs]    def queue_audio_tracks(self, zp):
        """
        Queues local audio content on player

        :param :class:`sonos.client.internal.SonosZoneComponent` zp:
        :return: Number of tracks in play queue
        :rtype: :obj:`int`
        """
        track_uris = []
        track_md = []
        for track in self.audio_tracks:
            track_uris.append("http://{}/{}".format(self.audio_server.host(),
                                                    track))
            track_md.append(self.zp.AVTransport.generate_didl_lite())

        queued_data = zp.AVTransport.AddMultipleURIsToQueue(
            num_uris=len(self.audio_tracks),
            uris=" ".join(track_uris),
            metadata=" ".join(track_md),
            container_uri="",
            container_metadata="",
            track_num=0,
            as_next=True)
        qlen = int(queued_data['NewQueueLength'])
        assert qlen == len(self.audio_tracks), "Expect tracks in play queue"
        return qlen



[docs]    def start_queue_playback(self, zp, track_to_play='random'):
        """
        Queues tracks and starts playback in REPEAT mode

        :param :class:`sonos.client.internal.SonosZoneComponent` zp:
        :param int track_to_play: Either a track number or random string
        """
        qlen = self.queue_audio_tracks(zp)
        zp.AVTransport.use_queue()
        zp.AVTransport.SetPlayMode("REPEAT_ALL")
        zp.RenderingControl.SetVolume(5)
        # zp.AVTransport.Seek('TRACK_NR', random.randint(1, qlen))
        # TBF!!! chasing noderx rx buffer full on track 4 ?
        # All tests fail when starting with #3, #4
        # All tests passed when starting with #5, #6, #7, #1, #2
        # TBF!!! chasing <chsnk,0> Old/New packet mismatch
        #                <chsnk,0> seamless source change failed (remote)
        # when starting with #8
        #
        if isinstance(track_to_play, int):
            if track_to_play < 1 or track_to_play > qlen:
                raise ValueError("Can't play track {} - "
                                 " queue length {}".format(track_to_play,
                                                           qlen))
            zp.AVTransport.Seek('TRACK_NR', track_to_play)
        elif track_to_play == 'random':
            zp.AVTransport.Seek('TRACK_NR', random.randint(1, qlen))
        else:
            raise ValueError("Unsupported keyword {}".format(track_to_play))
        zp.AVTransport.Play_and_wait()

[docs]    def restore_tracking_mode(self):
        """
        Restores the UMTracking system properties variable to
        its original value. If the value did not exist, it removes
        the UMTracking variable from the system properties.
        """
        if not self.umtracking_val:
            self.logger.debug("Removing {} variable".format(UMTRACKING))
            self.zp.SystemProperties.Remove(UMTRACKING)
        else:
            self.logger.debug("Restoring {} variable to '{}'".format(
                UMTRACKING, self.umtracking_val))
            self.zp.SystemProperties.SetString(UMTRACKING,
                                               self.umtracking_val)


    def _instantiate_override(self):
        """
        Make the HH use the override setting - once on suite set up
        Save tracking variable and set tracking to report.
        """
        self.save_tracking_mode()
        self.zp.SystemProperties.SetString(UMTRACKING, u'Report')
        url = self.EVENTMETRICS_URL_OVERRIDE_VALUE.format(self.websrv.host())
        self.zp.SystemProperties.SetString(
            self.EVENTMETRICS_URL_OVERRIDE_NAME,
            url)
        self.temperature_report_override(self.zp)

    def _remove_override(self):
        """
        Remove the override from the HH - once on suite tear down
        Restores the UMTracking mode variable.
        """
        self.restore_tracking_mode()
        self.zp.SystemProperties.Remove(self.EVENTMETRICS_URL_OVERRIDE_NAME)
        self.zp.SystemProperties.Remove(self.TEMPERATURE_REPORT_URL_OVERRIDE_NAME)
        for zp in self.generators.generate_testbed_devices():
            if not zp.is_virtual():
                zp.bounce_anacapa_and_wait()

[docs]    def wait_for_reporting_subsystems(self, zones, timeout_seconds=90):
        """
        For a given list of zones, wait until all zones have initialized
        their respective reporting subsystems.
        Returns True when all zones have initialized, else False.
        :param list zones: List of zones to wait
        :param int timeout_seconds: Maximum time in seconds to wait for
        all zones to initialize
        :return: Have the zones initialized reporting ?
        :rtype: :obj:`bool`
        """
        results = run_concurrent_zone_player_processes(
            self.wait_for_reporting_subsystem, zones, timeout_seconds)
        failed_zones = [zp for zp, initialized in zip(zones, results)
                        if not initialized]
        if failed_zones:
            self.logger.error(
                "Zones failed to initialize reporting subsystem <{}>".format(
                    failed_zones))
        return all(results)

        """
        Tests whether the events are present in the
        usage metrics reports.

        :param tuple events: Events to be verified
        :param :class:`~sonos.client.internal.SonosZoneComponent` zp: The DUT
        """
        for event in events:
            try:
                wait_until_true(
                    lambda: self.umh.find_event_in_reports(zp.ip, event),
                    iteration_delay=15, timeout_seconds=180,
                    retry=lambda: self.flush_event_report(zp),
                    reason='Timedout waiting for event {}'.format(event))
            except TimeoutError as e:
                self.fail(e)
                return

[docs]    def flush_and_wait_for_events(self,
                                  zp,
                                  timeout_seconds=15,
                                  report_type=PlayerReports.USAGE_REPORT):
        """
        Captures the internal depth of events that were captured from
        the player, flushes the player's event buffer and waits for
        at least one event report of the specified type to be received.
        Defaults to wait for usage metrics report.
        Returns True if the report type has been received, False if the
        specified timeout has been exceeded.
        :param :class:`~sonos.client.internal.SonosZoneComponent` zp:
        :param int timeout_seconds: Maximum number of seconds to wait
        for player to send event report
        :param: class:`PlayerReports` report_type: Report
        :return: True if received event else False
        :rtype: :obj:`bool`
        """
        current = self.umh.metrics_event_depth(zp.ip)
        self.logger.debug(
            "Current internal metrics queue depth {} on <{}>".format(current,
                                                                     zp))
        self.verifyTrueOrFailCase(
            self.flush_event_report(zp),
            "Flushed reporting events on <{}>".format(zp))
        try:
            wait_until_true(
                lambda: (self.umh.metrics_event_depth(zp.ip) > current and
                         self.umh.has_report(zp.ip, report_type=report_type)),
                iteration_delay=1,
                timeout_seconds=timeout_seconds)
        except TimeoutError:
            return False
        else:
            return True

        """
        Waits for the specified name in the metrics report.
        :param str ip: Zone IP address
        :param str name: The name of the field in the metrics report
        :param int timeout_seconds: Maximum time in seconds to wait
        for report arrival.
        :param int count: The number of name reports expected to match or

        :return: True if the count of name report(s) received from specified
        zone IP matches or exceeeds the expected count.
        :rtype: :obj:`bool`
        """
        self.logger.debug("Will wait to accumulate {} "
                          "{} reports from {}".format(count,
                                                      name,
                                                      ip))
        try:
            wait_until_true(
                lambda: len(self.umh.find_names_in_reports(ip, name)) >= count,
                timeout_seconds=timeout_seconds,
                iteration_delay=5,
                reason="Gave up waiting on name {} "
                "from {} with count {}".format(name,
                                               ip,
                                               count))
            return True
        except TimeoutError:
            return False

[docs]    def verify_continuous_playing_state(
            self,
            zp,
            continuous_playing_sec=CONTINUOUS_PLAYING_SEC,
            transition_to_playing_sec=TRANSITION_TO_PLAYING_SEC):
        """
        Confirm the play state remains PLAYING on zone player during the
        required continuous playing interval. Fails test when zone player
        transitions out of the PLAYING_STATE during the sampling period.
        :param :class:`sonos.client.internal.SonosZoneComponent` zp:
        The zone player under test
        :param float continuous_playing_sec: The number of continuous seconds
        to remain in PLAYING_STATE
        :param float transition_to_playing_sec: The number of seconds before
        sampling the state
        """
        avts = AVTransportStateTransitionTimer(zp.AVTransport)
        avts.start(continuous_playing_sec,
                   transition_to_playing_sec,
                   user_defined_start_states=[zp.AVTransport.PLAYING_STATE])
        self.verifyTrueOrFailCase(
            not avts.state_changed,
            "Expect AVT state to remain playing for {} seconds of '{}'".format(
                continuous_playing_sec, zp.AVTransport.get_avtransport_uri()))
