"""
This test suite is hard-coded to use a premium Spotify account with a specifc album.
Please make sure that this type of account set up on the controller before running the suite.

QA Test Account Username 	sonosSpotify6
QA Test Account Password 	sonos6

Album: Dopamine
Artist: BORNS

This test suite covers the trackPlayReporting mechanism currently visible via the 'spot' module in the
anacapad.trace log file. This behavior being tested here is current as of 12/22/2015.
There does not seem tp exist any kind of documentation on the reporting requirements.
As far as I can tell, the player is reporting total seconds played after a track is over,
with the total time clamped to the track's total length. There doesn't seem top be any kind of requirement
on how fast we should report after a track, other then as fast as possible.
"""

[docs]    @skip('not ready for rebooting')
    def test_report_on_reboot(self):
        """
        Verify a player sends the setPlayedSeconds call when anacapa is shutdown cleanly. The report should contain
        the total play time of the track.
        """
        wait_until_true(lambda: upnp_playback.get_current_track_position(self.zp) == 5, timeout_seconds=15, iteration_delay=.5)
        self.zp.diag.start_udp_log()
        self.zp.diag.watch_udp_log_for_string('Played 2pA4ip3VIEVcIa3qE02oAX')
        self.zp.force_subnet_removal()
        self.zp.bounce_anacapa_and_wait()
        wait_until_true(lambda: self.zp.diag.get_search_results(query='Played 2pA4ip3VIEVcIa3qE02oAX') == 1, timeout_seconds=10)

    def test_scrub_forwards_within_track(self):
        """
        Verify the player accurately reports the amount of time listened when the user scrubs forwards
        through a track and doesn't listen to all of it.

        Expected: Player should report the total amount of time that was actually heard by the user.
        """
        wait_until_true(lambda: upnp_playback.get_current_track_position(self.zp) == 5, timeout_seconds=15, iteration_delay=.5)
        self.zp.AVTransport.Seek(unit='REL_TIME',
                                      target='0:00:15')
        wait_until_true(lambda: upnp_playback.get_current_track_position(self.zp) == 25, timeout_seconds=15, iteration_delay=.5)
        self.zp.diag.start_udp_log()
        self.zp.diag.watch_udp_log_for_string('Played 2pA4ip3VIEVcIa3qE02oAX for 13000 \(of 174000\)')
        self.zp.Queue.RemoveAllTracks(queueid=0, updateid=0)
        wait_until_true(lambda: self.zp.Queue.Browse(queue_id=0,
                                                          starting_index=0,
                                                          requested_count='')['NumberReturned'] == '0',
                        timeout_seconds=5)
        wait_until_true(lambda: self.zp.diag.get_search_results(query='Played 2pA4ip3VIEVcIa3qE02oAX for 13000 \(of 174000\)') == 1, timeout_seconds=10)

[docs]    def test_no_report_while_track_plays(self):
        """
        Verify the SMAPI setPlayedSeconds call is not sent to the server before the track has finished
        playing (setPlayedSeconds should NOT report based on buffered audio, only on audio listened to).
        """
        self.zp.diag.start_udp_log()
        self.zp.diag.watch_udp_log_for_string('Played 2pA4ip3VIEVcIa3qE02oAX for')
        wait_until_true(lambda: upnp_playback.get_current_track_number(self.zp) == 2, timeout_seconds=180, iteration_delay=.5)
        self.zp.diag.stop_udp_log()
        if self.zp.diag.get_search_results(query='Played 2pA4ip3VIEVcIa3qE02oAX for') == 1:
            self.fail('Player sent a track played report during track playback')
