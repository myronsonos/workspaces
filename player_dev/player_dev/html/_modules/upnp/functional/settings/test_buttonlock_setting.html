        """
        Enable the UDP Log output on a given list of ZPs
        :param zps: ZPs that need to event Anacapa.trace log messages over UDP
        """
        for zp in zps:
            try:
                self.logger.info("Starting UDP Logger on ZP {}".format(zp))
                zp.diag.start_udp_log()
            except Exception as e:
                self.logger.critical("UDPLogger START port error on ZP {}."
                                     " \nError: {}".format(zp, e))
                self.stop_suite("Failed to start UDP logger on a device.")

[docs]    def verify_led(self, zps, state, wait_for_toggle=False):
        """
        Verifies led setting is in a given state for a list of ZPs
        :param zps: List of ZPs to be tested
        :param state: True to verify LED ON state. False otherwise.
        :param wait_for_toggle: True to wait for LED to toggle to state
          False to immediately validate LED state.
        """
        # Capture state bool as an 'On' or 'Off' value
        state_str = bool_to_onoff(state)

        for zp in zps:
            if wait_for_toggle:
                wait_until_true(
                    lambda: zp.DeviceProperties.GetLEDState() == state,
                    iteration_delay=1,
                    reason="Timeout waiting for LED to toggle "
                           "to {} on ZP {}".format(state, zp))
            else:
                # Verify All ZPs synced to the new UPNP value
                self.verifyEqualOrFailCase(
                    zp.DeviceProperties.GetLEDState(), state,
                    "Button Lock State should be set to {} on ZP {}".format(state, zp))

            # Verify able to update UPNP Service state variable
            wait_until_true(
                lambda: (zp.DeviceProperties.get_state_variable("LEDState")
                         == state_str),
                reason='Timeout waiting for LEDState state variable to update. '
                       'LEDState should have '
                       'been {} for {}.'.format(state_str, zp))

            # Verify localsettings.txt entry for LEDState
            self.verifyEqualOrFailCase(
                zp.diag.get_localsettings_as_dict().get('LED'), state_str,
                "LED in LocalSettings should be set "
                "to {} on ZP {}".format(state_str, zp))

    def _test_bonded_room_buttonlock_led_sync(self, master, members, bonding_method):
        """
        Tests ButtonLock and LED on a bonded room. Verifies room synchronization
        when bonding and when losing / regaining players. Verifies button presses
        are or are not disabled depending on ButtonLock state.
        :param master: Room coordinator
        :param members: Member ZPs
        :param bonding_method: Method to call to bond players together
        """
        # Use a random True or False value to test initial button lock state
        buttonlock_led_state = choice([True, False])
        member_buttonlock_led_state = not buttonlock_led_state

        # Bond a room with buttonlock and LED on/off.
        #   The members will sync their buttonlock and LED state when done bonding.
        self.toggle_led(master, buttonlock_led_state)
        self.toggle_button_lock(master, buttonlock_led_state)
        for zp in members:
            self.toggle_led(zp, member_buttonlock_led_state)
            self.toggle_button_lock(zp, member_buttonlock_led_state)

        # Bond room
        bonding_method()
        self.update_diag_levels(master, *members)

        # Verify members and master have LED synced
        self.verify_led([master] + members, buttonlock_led_state,
                        wait_for_toggle=True)

        # Verify members and master have ButtonLock synced
        self.verify_button_lock([master] + members, buttonlock_led_state,
                                wait_for_toggle=True)

        # Test toggling member ZPs out of sync, then reboot the SUB / master.
        #   When the sub / master comes back online, all members should be synchronized.
        for zp in [self.sub, master]:
            # Toggle ButtonLock and LED out of sync again on member ZPS
            for member in members:
                self.toggle_led(member, member_buttonlock_led_state)
                self.toggle_button_lock(member, member_buttonlock_led_state)
            # Reboot the SUB and verify all bonded members have ButtonLock synced
            #   when it comes back online.
            zp.bounce_anacapa_and_wait()
            self.update_diag_levels(zp)
            self.verify_led([master] + members, buttonlock_led_state,
                            wait_for_toggle=True)
            self.verify_button_lock([master] + members, buttonlock_led_state,
                                    wait_for_toggle=True)

    def _test_button_lock(self, master, members=None):
        """
        Toggles ButtonLock on a ZP and optional members on and then
        attempts to use volume, mute, and play controls, expecting the controls to
        not function.
        """
        self.prep_room(master, members=members)
        # Test ButtonLock ON
        self._test_button_lock_in_state(True, master, members=members)
        # Test ButtonLock OFF
        self._test_button_lock_in_state(False, master, members=members)

    def _test_button_lock_in_state(self, state, master, members=None):
        """
        Tests button lock in an ON or OFF state, and verifies all expectations
        while in said state
        :param state: True to test ON state. False to test OFF state.
        :param master: Room coordinator
        :param members: Member ZPs
        """
        # Capture if we should expect button commands to work or not.
        #   If state is True (On) then we expect buttons to not work.
        should_change = not state
        # Toggle ButtonLock
        self.toggle_button_lock(master, state, members=members)
        # Verify volume, functionality
        self._test_volume(master, members=members,
                          direction='up', should_adjust=should_change)
        self._test_volume(master, members=members,
                          direction='down', should_adjust=should_change)
        # Verify Play/Pause functionality
        self._test_play_pause(master, members=members, should_pause=should_change)
        # Verify Button Press (Join) always works whether in or out of ButtonLock
        self._test_join_button_press(master, members=members, should_enable=True)
        # Verify Diag Mode always works whether in or out of ButtonLock
        self._test_diag_mode(master, members=members, should_enable=True)

    def _test_join_button_press(self, master, members=None, should_enable=True):
        """
        Tests the Join button press can('t) be enabled be enabled depending on the
        current ButtonLock status. We should always expect Join to work if button
        lock is on or off.
        :param master: Room coordinator
        :param members: Member ZPs
        :param should_enable: True if Join should work when pressed
        """
        netstart_string = "<buttons,0> signaling netstartd succeeded"

        all_zps = [master]
        if members is not None:
            all_zps.extend(members)

        for zp in all_zps:
            self.safe_associate(zp)

            # Start watching for button presses
            zp_log_inspector = AnacapaTraceLogInspector(zp)
            zp_log_inspector.mark_last_log_entry()

            # Perform the 'join' operation
            zp.button_press()
            zp.button_press()

            if should_enable:
                # Verify the button press took place
                self.verifyTrueOrFailCase(zp_log_inspector.wait_for_next(netstart_string,
                                            25, delay=5), "Expect to detect 'join' "
                                            "button press: {}".format(zp))
            else:
                # Verify button press is in a not detected state
                self.verifyFalseOrFailCase(zp_log_inspector.wait_for_next(netstart_string, 5),
                                           "Expect button press 'join' operation "
                                           "to not be detected: {}".format(zp))

    def _test_diag_mode(self, master, members=None, should_enable=True):
        """
        Tests that the Diag Mode button press can('t) be enabled depending
        on the current ButtonLock status.
        :param master: Room coordinator
        :param members: Member ZPs
        :param should_enable: True if diag mode is expected to function
        """
        enter_diag_mode_string = '<buttons,3> inline action \(.. \- LED_DIAG_MODE_START\)'
        exit_diag_mode_string = '<buttons,3> inline action \(.. \- LED_DIAG_MODE_STOP\)'

        all_zps = [master]
        if members is not None:
            all_zps.extend(members)

        for zp in all_zps:
            self.safe_associate(zp)

            # Start watching for diag mode
            zp_log_inspector = AnacapaTraceLogInspector(zp)
            zp_log_inspector.mark_last_log_entry()

            # Enter diag mode
            zp.cli.button_diag()

            if should_enable:
                # Verify diag mode entered.
                self.verifyTrueOrFailCase(zp_log_inspector.wait_for_next(enter_diag_mode_string,
                                            45, delay=5), "Expect to enter diag mode: "
                                            "{}".format(zp))

                # Let buttons module settle down
                sleep(1)

                zp_log_inspector.mark_last_log_entry()

                # Exit diag mode
                zp.button_press()

                # Verify diag mode exited
                self.verifyTrueOrFailCase(zp_log_inspector.wait_for_next(exit_diag_mode_string,
                                            20, delay=5), "Expect to exit diag mode: "
                                            "{}".format(zp))
                sleep(1)
            else:
                # Verify not in diag mode
                self.verifyFalseOrFailCase(zp_log_inspector.wait_for_next(enter_diag_mode_string, 5),
                                           "Expect diag mode to "
                                           "not start: {}".format(zp))
                zp_log_inspector.mark_last_log_entry()
                self.verifyFalseOrFailCase(zp_log_inspector.wait_for_next(exit_diag_mode_string, 5),
                                           "Expect diag mode to "
                                           "not stop: {}".format(zp))

    def _test_play_pause(self, master, members=None, should_pause=True):
        """
        Tests that play-pause control can(')t change depending on the current
        ButtonLock status.
        :param master: Room coordinator
        :param members: Member zps
        :param should_pause: True if pause is expected to function
        """
        all_zps = [master]
        if members is not None:
            all_zps.extend(members)

        # Test play/pause is enabled or disabled on all ZPs in the room.
        for zp in all_zps:
            # Skip any non-playback device member tests since they do not have a play/pause button
            # Also skip any playback devices that do not have play/pause control
            if zp.modelNumber in (NON_PLAYBACK_DEVICES + PLAYBACK_DEVICES_JOIN_BUTTON_NO_CAP_TOUCH):
                self.logger.info("No play/pause control on ZP {}, skipping regardless of ButtonLock status".format(zp))
                continue

            # Begin playback of track
            master.AVTransport.Play_and_wait()

            # Tap the play/pause button
            zp.cli.volume_mute()

            # Wait for adjustment to take place
            if should_pause:
                # Wait until the AVTransport is actually paused.
                wait_until_true(
                    lambda: master.AVTransport.is_paused(),
                    reason='Timeout waiting for Room to pause. {}'.format(master))
            else:
                # Wait for 3 seconds to let any unexpected events through
                sleep(3)

            # Stop playback of track
            master.AVTransport.Stop_and_wait()

    def _test_volume(self, master, members=None,
                     direction='up', should_adjust=True):
        """
        Tests that volume control can('t) change depending on the current
        ButtonLock status.
        :param master: Room coordinator
        :param members: Member zps
        :param direction: up or down volume
        :param should_adjust: True if the volume is expected to adjust.
          False if otherwise.
        """
        all_zps = [master]
        if members is not None:
            all_zps.extend(members)

        for dut_zp in all_zps:
            # Skip any non-playback device member tests since they do not have vol buttons;
            # Also skip any playback devices that do not have vol control
            if dut_zp.modelNumber in (NON_PLAYBACK_DEVICES + PLAYBACK_DEVICES_JOIN_BUTTON_NO_CAP_TOUCH):
                self.logger.info("No vol control on ZP {}, skipping regardless of ButtonLock status".format(dut_zp))
                continue

            # Set a base volume on the room
            master.RenderingControl.SetVolume(self.volume_start)

            if direction == 'up':
                vol_command = dut_zp.cli.volume_up
                adjusted_volume = self.volume_start + self.volume_threshold
                verifier = self.verifyGreaterEqualOrFailCase

                def volume_comparator(current_vol):
                    # Volume adjusted up
                    return current_vol >= adjusted_volume

            elif direction == 'down':
                vol_command = dut_zp.cli.volume_down
                adjusted_volume = self.volume_start - self.volume_threshold
                verifier = self.verifyLessEqualOrFailCase

                def volume_comparator(current_vol):
                    # Volume adjusted down
                    return current_vol <= adjusted_volume

            else:
                raise TypeError('Must be up or down')

            # Execute vol up or down command.
            vol_command()

            if should_adjust:
                # Wait for volume adjustment to event through the Room
                wait_until_true(
                    lambda: volume_comparator(master.RenderingControl.GetVolume()),
                    reason="Timeout waiting for volume change on ZP {}".format(master))
            else:
                # Sleep 3 seconds to let any unexpected events through
                sleep(3)

            for member in all_zps:
                if should_adjust:
                    # Verify volume adjustment
                    verifier(
                        member.RenderingControl.GetVolume(), adjusted_volume,
                        "Expect volume to be "
                        "{}={} on ZP {}".format('>' if direction == 'up' else '<',
                                                adjusted_volume, member))
                else:
                    # Verify volume did not adjust
                    self.verifyEqualOrFailCase(
                        member.RenderingControl.GetVolume(), self.volume_start,
                        "Expect volume to be unchanged "
                        "({}) on ZP {}".format(self.volume_start, member))
