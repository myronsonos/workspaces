[docs]class WaitOnUDPLogging(DatagramProtocol, object):
    """
    A helper class that keeps a dictionary (key being the IP address)
    of the device that sent the log message using UDP.
    """

    """
    Expected signature on datagram message from player.
    """
    DIAG_LOG_COOKIE = 0xbeeeeeef
    """
    Default UDP port the player will send messages to.
    """
    DEFAULT_UDP_PORT = 8915
    logger = logging.getLogger('SONOS.WaitOnUDPLogging')
    is_running = False

    def __init__(self, port=DEFAULT_UDP_PORT):
        """
        Initializes dictionary of IP addresses.
        :param int port: UDP port to listen.
        """
        self.port = port
        super(WaitOnUDPLogging, self).__init__()
        self.log_msg_recv = {}
        self.is_running = self.start()


[docs]    def datagramReceived(self, data, (host, port)):
        """
        Callback on received datagram.
        Sanity check on received log message from device.
        Marks as received the dictionary key on success.
        """
        log_statement = None
        log_cookie = struct.unpack_from("!L", data, 0)[0]
        log_version = struct.unpack_from("!H", data, 4)[0]
        log_sequence = struct.unpack_from("!H", data, 6)[0]
        if ((log_cookie == self.DIAG_LOG_COOKIE) and (log_version == 1)):
            log_statement = data[8:]
            self.logger.info("{}:{} {}".format(
                    host, log_sequence, log_statement))
            self.log_msg_recv[host] = True
        else:
            self.logger.error("invalid log msg recv'd: {}".format(data))



[docs]    def log_msg_seen(self, host, timeout=30):
        """
        Checks if a message has been received from the specified IP address
        waiting a maximum of timeout seconds before giving up.
        Returns True if a message has been received else False.

        :param str host: String IP address of device
        :param int timeout: Maximum seconds to wait before giving up
        on the device to send log messages.
        :return: Boolean
        :rtype: :obj:`bool`
        """
        try:
            wait_until_true(
                lambda: host in self.log_msg_recv,
                1.0,
                timeout)
        except sonos.exception.TimeoutError:
            self.logger.warn("No messages from {} in {} seconds".format(
                    host,
                    timeout))
            return False
        else:
            return True



[docs]    def start(self):
        """
        Posts a listener on the UDP port.
        """
        is_running = False
        if not self.is_running:
            try:
                self.logger.debug("Posting listen on port {}".format(
                    self.port))
                self.udp_handle = reactor.listenUDP(self.port, self)
                is_running = True
            except CannotListenError as e:
                self.logger.error(
                    "caught twisted.internet.error: {}".format(e))
        else:
            self.logger.info("WaitOnUDPLogging already started!")

        return is_running



[docs]    def stop(self):
        """
        Removes listen handler.
        """
        reactor.callLater(1, self.udp_handle.stopListening)
        self.is_running = False

    """
    Verifies persistent UDP logging on player devices based on
    the "LogToExternal" keyword in the localsettings.txt file.
    For now the two options on that keyword are:
    [serial:0]
    [udp:<IP address>:<UDP port>]
    Confirm serial console logging (when available on device).
    Confirm UDP logging.
    """
    TEST_UDP_PORT = 2014
    TEST_TYPE = ["SETTINGS"]

    def _start_udp(self):
        self.verifyTrueOrStop(reactor.running,
                              "Twisted reactor should be running")
        self.udp_handle = WaitOnUDPLogging(port=self.TEST_UDP_PORT)


[docs]    def setUpFixture(self):
        """
        Acquires the local interface IP address so that the test device
        will direct the UDP log messages to that address.
        NOTE: What we actually need is the IP address of the interface
        where the devices are discovered. The first non-virtual interface
        IP address may not necessarily be that.
        Activates the UDP monitor.
        """
        self.ip = Network().get_external_non_virtual_ips()[0]
        self._start_udp()


    def _setup_zp_test(self, zp):
        """
        Removes the LogToExternal keyword from the localsettings.txt file
        """
        zp.cli.command(REMOVE_LOG_TO_EXTERNAL)

    def _teardown_zp_test(self, zp, value):
        """
        Bring the device to a known state.

        Restores the contents of the localsettings file to the original
        and bounces anacapa. Failure to do that will cause the device
        to either send to the serial console or the UDP destination
        while the localsettings.txt does not reflect that.
        :param zp: Device object
        :type other_zp: :class:`~sonos.client.internal.SonosZoneComponent`
        :param str value: String capturing the value (if any) of the
        LogToExternal key word prior the commencing test.
        """
        self._setup_zp_test(zp)
        if value:
            var = 'LogToExternal: [{}]'.format(value)
            self.logger.debug(
                "Restoring localsettings with value='{}'".format(var))
            zp.cli.command(RESTORE_LOG_TO_EXTERNAL.format(
                    var,
                    LOCAL_SETTINGS_FILE))
        zp.bounce_anacapa_and_wait()

[docs]    @combinatorial('generate_testbed_unique_playback_devices')
    def dont_test_serial_now(self, zp):
        """
        NOTE!!! This test applicable to devices that have a serial
        console attached. In the event that all test bed devices
        have a serial console, this test can be enabled to run on
        that test bed.

        Removes existing (if any) LogToExternal keyword from file and
        saves the LogToExternal key word value from the localsettings.txt
        dictionary.

        Appends LogToExternal: [serial:0] to the localsettings.txt file.
        Bounces and waits for anacapa, retrieves the anacapa trace file
        and isolates the contents of that file from the last reboot.
        Searches for the diagnostic to determine if the
        persistent log outputs to the serial console.

        Before completing the test, the saved LogToExternal key word (if any)
        is restored to the original value prior to test.
        :param zp: Device object
        :type other_zp: :class:`~sonos.client.internal.SonosZoneComponent`
        """
        zp_localsettings_dict = zp.diag.get_localsettings_as_dict()
        log_to_external_value = zp_localsettings_dict.get(
            LOG_TO_EXTERNAL_KEYWORD)
        self.logger.debug("{}".format(zp_localsettings_dict))
        self._setup_zp_test(zp)
        self.logger.debug("test_serial: {}: {}".format(
                LOG_TO_EXTERNAL_KEYWORD,
                log_to_external_value))
        zp.cli.command(SET_LOCAL_SETTINGS_SERIAL)
        zp.bounce_anacapa_and_wait()
        anacapa_trace = zp.cli.command(GET_ANACAPA_TRACE)
        last_anacapa_trace = anacapa_trace[anacapa_trace.rfind('Version:'):]
        self.logger.debug("{}".format(last_anacapa_trace))
        self.verifyTrueOrFailCase(
            'Using serial console logger' in last_anacapa_trace,
            "Expect logger on serial console")
        self._teardown_zp_test(zp, log_to_external_value)



[docs]    @combinatorial('generate_testbed_unique_playback_devices')
    def test_udp(self, zp):
        """
        Removes existing (if any) LogToExternal keyword from file and
        saves the LogToExternal key word value from the localsettings.txt
        dictionary.

        Appends LogToExternal: [udp:<ip>:<port>] to the localsettings.txt
        file where the <ip> and <port> is the remote host address and
        destination port the device will forward the log messages as
        UDP datagrams.

        Bounces and waits for anacapa, retrieves the anacapa trace file
        and isolates the contents of that file from the last reboot.
        Searches for the diagnostic to determine if the device is
        forwarding the logging message to the specified value.

        Confirms via the WaitOnUDPLogging class the arrival of any logging
        messages from the test device.

        Before completing the test, the saved LogToExternal key word (if any)
        is restored to the original value prior to test.

        Applicable to zone player devices (not available on WD100 and bridges)

        :param zp: Device object
        :type other_zp: :class:`~sonos.client.internal.SonosZoneComponent`
        """
        zp_localsettings_dict = zp.diag.get_localsettings_as_dict()
        log_to_external_value = zp_localsettings_dict.get(
            LOG_TO_EXTERNAL_KEYWORD)
        self._setup_zp_test(zp)
        self.logger.debug("test_udp: {}: {}".format(
                LOG_TO_EXTERNAL_KEYWORD,
                log_to_external_value))
        zp.cli.command(SET_LOCAL_SETTINGS_UDP.format(
                       LOCAL_SETTINGS_FILE,
                       TEMP_SETTINGS_FILE,
                       self.ip,
                       self.TEST_UDP_PORT))
        zp.bounce_anacapa_and_wait()
        self.verifyTrueOrFailCase(
            self.udp_handle.log_msg_seen(zp.ip),
            "Expect to receive messages from device")
        self._teardown_zp_test(zp, log_to_external_value)
