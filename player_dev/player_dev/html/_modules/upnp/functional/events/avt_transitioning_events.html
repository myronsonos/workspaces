[docs]    def checkTransportState(self, zone, state, last_state=[]):
        zone_state = zone.AVTransport.get_state_variable("TransportState")
        if len(last_state) == 0 or last_state[0] != zone_state:
            self.logger.debug("TransportState is now %s" % (zone_state))
        if len(last_state) > 0:
            last_state[0] = zone_state
        return zone_state == state

    def test_events_on_play(self, zone, client, version):
        # Get the AVT service object and grab the current subscription ID before resubscribing
        avt_service = next((service for service in zone.services if service.id == "urn:upnp-org:serviceId:AVTransport"))
        before_avt_sub = avt_service.get_sid()

        # Update the AVTransport service subscription appropriately based on the FW version passed in
        subscribe_help.process_new_avt_subscription(zone, version)

        # Get the new AVT subscription ID and make sure it's incremented as expected, if not fail the test here
        after_avt_sub = avt_service.get_sid()
        self.verifyGreaterOrFailCase(after_avt_sub, before_avt_sub,
                                     "AVTransport subscription should have increased with explicit subscription")

        # Set AVT and wait until it succeeds, then start playing
        zone.AVTransport.SetAVTransportURI_and_wait(self.http_tracks[0]['slow_url'], DUMMY_DIDL, 10)
        current_transport_state = zone.AVTransport.get_state_variable_and_time("TransportState")

        self.logger.debug("Starting playback")
        zone.AVTransport.Play()
        self.logger.debug("Play UPnP action completed")

        # Verifications - based on client reported firmware version. We use the dict key passed in from the generator to
        # determine which verification to use.
        if client == "old_client":
            wait_until_true(lambda: zone.AVTransport.get_state_variable_and_time("TransportState") != current_transport_state,
                            iteration_delay=0.5,
                            timeout_seconds=5)
            transport_state = zone.AVTransport.get_state_variable("TransportState")
            self.verifyNotEqualOrFailCase("TRANSITIONING", transport_state,
                                          "Older clients should never receive a transitioning event")

        else:
            try:
                last_state_tracker = [None]
                wait_until_true(lambda: self.checkTransportState(zone, "TRANSITIONING", last_state_tracker),
                                timeout_seconds=5)
                transport_state = zone.AVTransport.get_state_variable("TransportState")
                self.verifyEqualOrFailCase("TRANSITIONING", transport_state,
                                          "Newer clients should receive a transitioning event")
            except sonos.exception.TimeoutError:
                # If we never see TRANSITIONING we really have nothing to verify here, skip the verification rather than pass
                # and dump the state variable to debug
                self.logger.debug("transport state == %s" % zone.AVTransport.get_state_variable("TransportState"))
                self.fail("Did not get a transitioning state")

                self.verifyEqualOrFailCase("TRANSITIONING", transport_state,
                                          "Newer clients should receive a transitioning event")

    def test_GetTransportInfo_while_transitioning(self, zone, client,
            version):
        """
        Verify that GetTransportInfo() reports "PLAYING" rather than "TRANSITIONING" to older Sonos controllers,
        but returns "TRANSITIONING" in all other cases.
        """
        # Update the AVTransport service subscription appropriately based on the FW version passed in
        subscribe_help.process_new_avt_subscription(zone, version)

        # prepare the User-Agent HTTP header value that we'll use when calling GetTransportInfo() below
        user_agent_value = ''
        if version:
            user_agent_value = "Linux UPnP/1.0 PlayToSonos/%s" % version

        # setup the queue for this test
        self.add_one_fast_and_three_slow_urls_to_the_queue(zone)

        zone.AVTransport.use_queue()
        # Wait until the track is actually playing, since the webserver delays responses for 2 seconds
        zone.AVTransport.Play_and_wait(timeout_seconds=10)

        # Here is the start of the actual test
        zone.AVTransport.Next()
        self.logger.debug("immediately after Next() == %s" % zone.AVTransport.get_state_variable("TransportState"))

        # immediately call GetTransportInfo() to get the current TransportState value
        with CustomUserAgentString(zone.AVTransport, user_agent_value):
            transport_info = zone.AVTransport.GetTransportInfo()
        transport_state = transport_info.get('CurrentTransportState')

        # Verifications - based on client reported firmware version. We use the dict key passed in from
        # the generator to determine which verification to use.
        if client == "old_client":
            self.verifyNotEqualOrFailCase("TRANSITIONING", transport_state,
                                          "Older clients should never see the transitioning state")
            self.verifyEqualOrFailCase("PLAYING", transport_state,
                                        "Older clients should always see the playing state")
        else:
            self.verifyEqualOrFailCase("TRANSITIONING", transport_state,
                                       "Newer clients should receive a transitioning event")
