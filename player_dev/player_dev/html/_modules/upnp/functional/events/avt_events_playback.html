"""
 UPnP Test:  AVTransport.AVTEventsPlayback

 Tests for the expected AVTransport events when there are changes to the channel state of a Sonos player.

 Scope: Verifies that when the Sonos player changes from STOPPED/PAUSED to PLAY and then back to
 STOPPED/PAUSED, that its AVTransport service will send a new LastChange event with the expected fields,
 including fields within "CurrentMetaData" and "NextMetaData". Current tests cover the local library MP3
 files, Pandora station, Internet Radio and Line-In sources. These tests check for the presence of metadata
 fields but not always its contents. Exceptions include expected formatting (like the current track
 duration having to be of the H+:MM:SS format).

    def _process_event_lastchange(self, xmldata):
        self.eventvars = {}
        tree = utils.parse_xml(xmldata, 'utf-8').getroot()
        namespace_uri, tag = tree.tag[1:].split("}", 1)
        instances = tree.findall('{%s}InstanceID' % namespace_uri)
        for instance in instances:
            for var in instance.getchildren():
                namespace_uri, tag = var.tag[1:].split("}", 1)
                self.eventvars[tag] = var.attrib['val']
                if tag == "CurrentTrackMetaData":
                    self.logger.info("CurrentTrackMetaData = '%s'" % self.eventvars[tag])

    def _check_evented_var(self, var, expected_value=None):
        self.verifyTrueOrFailCase(var in self.eventvars,
                            "'%s' should have been evented" % var)
        if expected_value:
            self.verifyEqualOrFailCase(expected_value, self.eventvars[var],
                                       "'%s' should be '%s'" % (var, expected_value))

    def _check_current_track_duration(self, expected_duration=None):
        self.verifyTrueOrFailCase('CurrentTrackDuration' in self.eventvars,
                                "'CurrentTrackDuration' should have been evented")
        duration = self.eventvars['CurrentTrackDuration']
        currentTrackDuration_ok = re.match("^\d{1,2}:\d\d:\d\d$", duration)
        self.verifyTrueOrFailCase(currentTrackDuration_ok,
                                  "'CurrentTrackDuration' should be of the expected format")
        if expected_duration:
            self.verifyEqualOrFailCase(expected_duration, duration,
                                       "'CurrentTrackDuration' should be '%s'" % expected_duration)

    def _check_track_metadata(self, tag,
                              expected_class=None,
                              require_album_art=False,
                              require_album=False,
                              require_radio_show_md=False,
                              require_album_artist=False,
                              require_original_track_number=False,
                              require_creator=False):
        self.verifyTrueOrFailCase(tag in self.eventvars, "'%s' should have been evented" % tag)
        didl_element = DIDLLite.DIDLElement.fromString(self.eventvars[tag])
        items = didl_element.getItems()
        self.verifyTrueOrFailCase(len(items) == 1,
                                  "'%s' should have one MusicTrack DIDL-Lite element" % tag)
        for item in items:
            self.logger.info("item: '%s'" % item)
            self.verifyTrueOrFailCase(hasattr(item, "title") and len(item.title) > 0,
                                      "'%s' contains track title" % tag)

            self.verifyTrueOrFailCase(hasattr(item, "upnp_class") and len(item.upnp_class) > 0,
                                          "'%s' contains UPnP class" % tag)
            if expected_class:
                self.verifyEqualOrFailCase(expected_class, item.upnp_class,
                                          "'%s's UPnP class should be %s" % (tag, expected_class))
            if require_album:
                self.verifyTrueOrFailCase(hasattr(item, "album") and len(item.album) > 0,
                                          "'%s' contains track album" % tag)
            if require_album_art:
                self.verifyTrueOrFailCase(hasattr(item, "albumArtURI") and len(item.albumArtURI) > 0,
                                          "'%s' contains album art URI" % tag)
            if require_album_artist:
                self.verifyTrueOrFailCase(hasattr(item, "albumArtist") and len(item.albumArtist) > 0,
                                          "'%s' contains album artist" % tag)
            if require_radio_show_md:
                self.verifyTrueOrFailCase(hasattr(item, "radioShowMd"),
                                          "'%s' contains radio show MD" % tag)
            if require_original_track_number:
                self.verifyTrueOrFailCase(hasattr(item, "originalTrackNumber"),
                                          "'%s' contains original track number" % tag)
            if require_creator:
                self.verifyTrueOrFailCase(hasattr(item, "creator") and len(item.creator) > 0,
                                          "'%s' contains creator" % tag)


    def _check_next_track(self, expectedNextTrackURI,
                          expected_class=None,
                          require_album_art=False,
                          require_album=False,
                          require_radio_show_md=False,
                          require_album_artist=False,
                          require_original_track_number=False,
                          require_creator=False):
        self.verifyTrueOrFailCase('NextTrackURI' in self.eventvars,
                                  "'NextTrackURI' should have been evented")
        if expectedNextTrackURI:
            self.verifyEqualOrFailCase(expectedNextTrackURI, self.eventvars['NextTrackURI'],
                                       "'NextTrackURI' should be same as the one set")

            # Check 'NextTrackMetaData' and its contents
            self.verifyTrueOrFailCase('NextTrackMetaData' in self.eventvars,
                                      "'NextTrackMetaData' should have been evented")
            didl_element = DIDLLite.DIDLElement.fromString(self.eventvars['NextTrackMetaData'])
            items = didl_element.getItems()
            if expectedNextTrackURI == "":
                self.verifyTrueOrFailCase(len(items) == 0, "'NextTrackMetaData' should be empty")
            else:
                self._check_track_metadata('NextTrackMetaData',
                                           expected_class,
                                           require_album_art,
                                           require_album,
                                           require_radio_show_md,
                                           require_album_artist,
                                           require_original_track_number,
                                           require_creator)

    def _check_before_eventing(self, zone):
        ret = zone.AVTransport.get_state_variable_and_time("LastChange")
        _, self.changeTimeBefore = ret

    def _check_after_eventing(self, zone):
        ret = zone.AVTransport.get_state_variable_and_time("LastChange")
        lastChange, changeTimeAfter = ret

        # Verify we've received a new LastChange event
        self.verifyGreaterOrFailCase(changeTimeAfter, self.changeTimeBefore,
                                     "LastChange event should have been fired after playing started")
        self._process_event_lastchange(lastChange)

    def test_on_playing_local_library_with_transitioning(self, zone):
        """
        Test: Verify that AVTransport events are being evented when (1) playing a track on the Queue;
              (2) skipping to the next track and then (3) stopping playback. This test will accept
              either a PLAYING -OR- TRANSITIONING event when transitioning between tracks. (This behavior
              was introduced in 22.0-63060)
        Devices: playback-capable && group coordinator
        Expected: success
        """
        # Set up initial conditions: stop playback, clear Queue and add two tracks
        # to the Queue, make sure repeat/suffle are disabled, and skip

        api_queue_management.setup_tracks_in_queue(zone, self.tracksAndMetadata, asNext=True)

        zone.AVTransport.SetPlayMode("NORMAL")

        #
        # Step 1 - Play local library track in the Queue
        #

        self.logger.info("- - - Step 1 - Play local library track in the Queue")

        # Start playing and wait until we AVTransport thinks it started playing
        self._check_before_eventing(zone)
        zone.AVTransport.use_queue()

        zone.AVTransport.Play()

        wait_until_true(lambda: zone.AVTransport.is_playing(), timeout_seconds=5)
        self._check_after_eventing(zone)

        # Check the contents of the LastChange event
        self._check_evented_var('TransportState', "PLAYING")
        self._check_evented_var('NumberOfTracks', "2")
        self._check_evented_var('CurrentTrack', "1")
        self._check_evented_var('CurrentTrackURI', self.libraryTrackURI_1)
        self._check_evented_var('EnqueuedTransportURI', self.libraryTrackURI_1)
        self._check_evented_var('EnqueuedTransportURIMetaData', self.metadataXML_1)
        self._check_current_track_duration()
        self._check_track_metadata(tag='CurrentTrackMetaData',
                                   expected_class='object.item.audioItem.musicTrack',
                                   require_album=True,
                                   require_album_artist=True,
                                   require_original_track_number=True)
        self._check_next_track(expectedNextTrackURI=self.libraryTrackURI_2,
                               expected_class='object.item.audioItem.musicTrack',
                               require_album=True,
                               require_album_artist=True,
                               require_original_track_number=True)


        #
        # Step 2 - Skip to next track
        #

        self.logger.info("- - - Step 2 - Skip to next track")

        # Skip to next track and wait until AVTransport thinks we've skipped
        self._check_before_eventing(zone)
        zone.AVTransport.Next()
        wait_until_true(lambda: zone.AVTransport.get_current_track() == 2, timeout_seconds=3)
        self._check_after_eventing(zone)

        # Check the contents of the LastChange event
        # self._check_evented_var('TransportState', "TRANSITIONING")
        self._check_evented_var('NumberOfTracks', "2")
        self._check_evented_var('CurrentTrack', "2")
        self._check_evented_var('CurrentTrackURI', self.libraryTrackURI_2)
        self._check_evented_var('EnqueuedTransportURI', self.libraryTrackURI_2)
        self._check_evented_var('EnqueuedTransportURIMetaData', self.metadataXML_2)
        self._check_current_track_duration()
        self._check_track_metadata(tag='CurrentTrackMetaData',
                                   require_album=True,
                                   require_album_artist=True,
                                   require_original_track_number=True)
        self._check_next_track(expectedNextTrackURI="",
                               require_album=True,
                               require_album_artist=True,
                               require_original_track_number=True)

        # Get the TransportState value and verify that it is either TRANSITIONING -OR- PLAYING.
        # TODO: Come back and fix this when we have a solid way to always provoke a TRANSITIONING event. This was added due to this test
        # Constantly failing after the 7/3 update to transitioning state in CL 103894 (PLAYER-5096)
        accepted_transport_states = ['PLAYING', 'TRANSITIONING']
        transport_state = zone.AVTransport.get_state_variable('TransportState')
        if transport_state not in accepted_transport_states:
            self.fail('%s is not an accepted transport state(expected PLAYING or TRANSITIONING)' % transport_state)

        #
        # Step 3 - Pause playback
        #

        self.logger.info("- - - Step 3 - Pause playback")

        # Pause playing and wait until we AVTransport thinks it stopped playing
        self._check_before_eventing(zone)
        zone.AVTransport.Pause()
        wait_until_true(lambda: zone.AVTransport.is_paused(), timeout_seconds=3)
        self._check_after_eventing(zone)

        # Check the contents of the LastChange event
        self._check_evented_var('TransportState', "PAUSED_PLAYBACK")
        self._check_evented_var('NumberOfTracks', "2")
        self._check_evented_var('CurrentTrack', "2")
        self._check_evented_var('CurrentTrackURI', self.libraryTrackURI_2)
        self._check_evented_var('EnqueuedTransportURI', self.libraryTrackURI_2)
        self._check_evented_var('EnqueuedTransportURIMetaData', self.metadataXML_2)
        self._check_current_track_duration()
        self._check_track_metadata(tag='CurrentTrackMetaData',
                                   require_album=True,
                                   require_album_artist=True,
                                   require_original_track_number=True)
        self._check_next_track(expectedNextTrackURI=None,
                               require_album=True,
                               require_album_artist=True,
                               require_original_track_number=True)

    def test_on_playing_internet_radio(self, zone):
        """
        Test: Verify that AVTransport events are being evented when playing and stopping an Internet radio station
        Devices: playback-capable && group coordinator
        Expected: success
        """
        # Set up initial conditions

        radio_metadata = ("<DIDL-Lite xmlns:dc=\"http://purl.org/dc/elements/1.1/\" "
                           "xmlns:upnp=\"urn:schemas-upnp-org:metadata-1-0/upnp/\" "
                           "xmlns:r=\"urn:schemas-rinconnetworks-com:metadata-1-0/\" "
                           "xmlns=\"urn:schemas-upnp-org:metadata-1-0/DIDL-Lite/\">"
                           "<item id=\"-1\" parentID=\"-1\" restricted=\"true\">"
                           "<dc:title>WBUR-FM</dc:title><upnp:class>object.item.audioItem.audioBroadcast"
                           "</upnp:class><desc id=\"cdudn\" "
                           "nameSpace=\"urn:schemas-rinconnetworks-com:metadata-1-0/\">SA_RINCON65031_"
                           "</desc></item></DIDL-Lite>")

        zone.AVTransport.SetAVTransportURI(self.radio_uri, radio_metadata)

        #
        # Play Radio Station
        #

        # Start playing and wait until we AVTransport thinks it started playing
        # Note: internet radio stations can take a variable (and sometimes long) amount of time
        # to start playing.
        # TODO: setup our own internal "internet" streaming server so we can better control
        # the variability and latency.
        self._check_before_eventing(zone)
        zone.AVTransport.Play()
        wait_until_true(lambda: zone.AVTransport.is_playing(), timeout_seconds=20)
        self._check_after_eventing(zone)

        # Check the relevant contents of the LastChange event
        # NOTE: track URI will be different from the one used for SetAVTransportURI() for Pandora,
        # it will contain the auth-token and even the album art encoded.
        self._check_evented_var('TransportState', "PLAYING")
        self._check_evented_var('CurrentTrackURI', None)
        self._check_track_metadata(tag='CurrentTrackMetaData',
                                   expected_class='object.item',
                                   require_album_art=True,
                                   require_radio_show_md=True)

        #
        # Stop Radio Station
        #

        # Stop playing and wait until we AVTransport thinks it stopped playing
        # Note:  it doesn't take long to stop playing an internet radio station, so
        # no need for an extra large timeout here.
        self._check_before_eventing(zone)
        zone.AVTransport.Stop()
        wait_until_true(lambda: zone.AVTransport.is_stopped(), timeout_seconds=3)
        self._check_after_eventing(zone)

        # Check the relevant contents of the LastChange event
        self._check_evented_var('TransportState', "STOPPED")
        self._check_evented_var('CurrentTrackURI', None)
        self._check_track_metadata(tag='CurrentTrackMetaData',
                                   expected_class='object.item',
                                   require_album_art=True,
                                   require_radio_show_md=True)
