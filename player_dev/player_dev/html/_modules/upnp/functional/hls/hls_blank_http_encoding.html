[docs]    @combinatorial('generate_testbed_unique_playback_devices')
    def test_blank_http_transfer_encoding(self, zp):
        """
        Play back M4A emulated radio stream and observe the player
        does not pay attention to the web server's reply which
        contains an empty HTTP transfer encoding.
        :param :class:`sonos.client.internal.SonosZoneComponent` zp:
        """
        self.setup_player(zp)
        zp.diag.watch_udp_log_for_string(
            "Bad HTTP Header: Transfer-Encoding:",
            lambda msg: self.fail(msg))
        self.verifyTrueOrFailCase(
            self.start_playback(zp, self.uri),
            "Expecting to play M4A radio station")
        avts = AVTransportStateTransitionTimer(zp.AVTransport)
        avts.start(CONTINUOUS_PLAYING_SEC,
                   TRANSITION_TO_PLAYING_SEC,
                   user_defined_start_states=[zp.AVTransport.PLAYING_STATE])
        self.verifyTrueOrFailCase(
            not avts.state_changed,
            "Played continuously {} seconds of radio stream '{}'".format(
                CONTINUOUS_PLAYING_SEC, zp.AVTransport.get_avtransport_uri()))
        self.cleanup_player(zp)
