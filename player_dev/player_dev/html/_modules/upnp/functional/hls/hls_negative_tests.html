[docs]    def setup_websrv(self):
        """
        Sets up the local web server.
        Fetches remote media files.
        Creates a handler dictionary for the web server.
        Starts web server.
        """
        self.fetch_web_content()
        self.abs_path = os.path.dirname(os.path.realpath(__file__))
        handlers = {}
        handlers[self.MASTER_HLS_NAME] = StaticFileHandler(self.MASTER_HLS_NAME)
        handlers[self.MEDIA_HLS_NAME] = StaticFileHandler(self.MEDIA_HLS_NAME)
        for data_file in self.data_files:
            handlers[data_file] = StaticFileHandler(data_file)
        self.websrv = WebServer(start_reactor=False)
        self.websrv.start(
            port=get_random_web_port_to_use(),
            path=self.abs_path,
            handlers=handlers)
        self.hls_uri = "hls-aac://{}/{}".format(
            self.websrv.host(),
            self.MASTER_HLS_NAME)


    def _cleanup_avt(self, zp):
        """
        Resets the state of the AV on the zone player.
        :param :class:`sonos.client.internal.SonosZoneComponent` zp:
        The zone component under test.
        """
        zp.AVTransport.clear_play_state()
        zp.AVTransport.clear_queue_and_wait()
        zp.RenderingControl.SetVolume(0)
        zp.AVTransport.SetPlayMode(zp.AVTransport.PLAYMODE_NORMAL)

    def _setup(self, zp):
        """
        Cleans AVTransport, starts persistent log
        """
        self._cleanup_avt(zp)
        zp_log = "{}_{}_playback.log".format(
            zp.modelNumber,
            datetime.now().strftime("%H_%M_%S"))
        zp.diag.start_udp_log_to_file(zp_log)
        for diagm in self.DIAG_LEVEL_DICT:
            zp.diag.set_diag_level(diagm, self.DIAG_LEVEL_DICT[diagm])

    def _cleanup(self, zp):
        """
        """
        self._cleanup_avt(zp)
        for diagm in self.DIAG_LEVEL_DICT:
            zp.diag.set_diag_level(diagm, 1)
        zp.diag.stop_udp_log_to_file()

[docs]    def start_playback(self, zp, timeout_seconds=10,
                       except_num=UPNP_OK):
        """
        """
        ret = zp.AVTransport.SetAVTransportURI(
            self.hls_uri,
            zp.AVTransport.generate_didl_lite())
        self.verifyTrueOrFailCase(
            ret,
            "Set AV transport URI")
        zp.RenderingControl.SetVolume(5)
        try:
            zp.AVTransport.Play_and_wait(timeout_seconds=timeout_seconds)
        except UPnPError as e:
            if except_num > UPNP_OK and except_num == e.upnp_error:
                self.logger.info("Caught expected exception {}".format(e))
                return False
            raise e
        except TimeoutError:
            return False
        return True



[docs]    def wait_on_diag(self, zp, watch_str, timeout_seconds=60):
        """
        :param :class:`sonos.client.internal.SonosZoneComponent` zp:
        :param str watch_str: UDP watch string.
        :param int timeout_seconds: Time in seconds to wait for diag.
        """
        try:
            wait_until_true(
                lambda: self.monitor_diagc.get_state(),
                iteration_delay=1,
                timeout_seconds=timeout_seconds)
        except TimeoutError:
            pass
        zp.diag.stop_watching_udp_log_for_string(watch_str)
        return self.monitor_diagc.get_state()

    def test_timeout_generic(self, zp, test_data):
        """
        Given a the test file, it artificially delays the HTTP GET response
        to solicit a time out from the player when the player accesses
        the file.
        1. Response on master playlist HTTP GET request
        delayed causing a timeout
        2. Response on media playlist HTTP GET request
        delayed causing a timeout
        3. Response on first audio segment HTTP GET request
        delayed causing a timeout
        :param :class:`sonos.client.internal.SonosZoneComponent` zp:
        :param tuple test_data: Test data pertinent to this test
        test_file, watch_str, timeout_sec, expect_play
        """
        test_file, watch_str, timeout_sec, expect_play = test_data
        self.set_udp_watch(zp, watch_str)
        self.websrv.remove_handler(test_file)
        self.websrv.add_handler(test_file,
                                SlowFileHandler(test_file, delay=timeout_sec))
        self.verifyEqualOrFailCase(
            expect_play,
            self.start_playback(zp),
            "Expected to play stream")
        self.verifyTrueOrFailCase(
            self.wait_on_diag(zp, watch_str, timeout_seconds=90),
            "Expected diag string '{}' from player".format(watch_str))
        zp.AVTransport.Stop_and_wait(timeout_seconds=30)
        self.websrv.remove_handler(test_file)
        self.websrv.add_handler(test_file, StaticFileHandler(test_file))

    def test_bad_uri(self, zp, test_data):
        """
        Given a test file to modify, the test replaces the contents
        of the test file with malformed data.
        :param :class:`sonos.client.internal.SonosZoneComponent` zp:
        :param tuple test_data: Test data pertinent to this test
        test_file, old, new, watch_str, expect_play
        """
        test_file, old, new, watch_str, expect_play = test_data
        # Read contents of the file under test
        with open(test_file, 'rb') as fh:
            contents = fh.read()
        # Replace contents with malformed URI
        with open(test_file, 'wb+') as fh:
            fh.write(contents.replace(old, new))
        self.set_udp_watch(zp, watch_str)
        self.verifyEqualOrFailCase(
            expect_play,
            self.start_playback(
                zp, except_num=UPNP_ERROR_701_TRANSITION_UNAVAILABLE),
            "Expected to play stream")
        ret = self.wait_on_diag(zp, watch_str)
        self.verifyTrueOrFailCase(
            ret,
            "Expected diag string '{}' from player".format(watch_str))
        # Restore original file contents
        with open(test_file, 'wb+') as fh:
            fh.write(contents)
        # If the  zone player has not generated the expected message, check if
        # the zone player has encountered the SWPBL-40033 bug in CHSRC.
        # An un-initialized time stamp which causes WATCHDOG to
        # fire because CHSRC is not responding.
        if not ret:
            self.monitor_diagc.set_state(False)
            watch_str = "WATCHDOG: chsrc unresponsive"
            zp.diag.watch_udp_log_for_string(
                watch_str,
                lambda msg: self.monitor_diagc.set_state(True))
            ret = self.wait_on_diag(zp, watch_str, timeout_seconds=180)
            response = zp.diag.get_thread_info()
            self.logger.debug("Thread Info: {}".format(response))
            if ret:
                self.logger.error("Detected unresponsive chsrc")
                # None of the UPnP commands will work when the player is
                # this state.
                # Recover player by rebooting; anacapad does not
                # pay attention to the termination signal at this point
                # when this symptom manifests hence we can't use
                # something like bounce_anacapa_and_wait()
                zp.reboot()
                self.stop("Stopping test due to unresponsive player")
        zp.AVTransport.Stop_and_wait(timeout_seconds=30)

    def test_uri_lengths(self, zp, test_data):
        """
        Given a test file, the test generates a URI based on the URI length
        and installs a static web handler to access the test_file.
        On failure, due to exceeding the URI length, the next file won't be
        retrieved by the player.
        :param :class:`sonos.client.internal.SonosZoneComponent` zp:
        :param tuple test_data: Test data pertinent to this test
        media_file, test_file, uri_length, str_expr exp_result
        """
        media_file, test_file, uri_length, str_expr, exp_result = test_data
        if exp_result:
            watch_str = str_expr.format(self.websrv.host())
        else:
            watch_str = str_expr
        self.logger.debug("watch_str='{}'".format(watch_str))
        test_uri_length = (uri_length -
                           len("http://{}/".format(self.websrv.host())))
        test_uri = "".join(random.choice(string.letters + string.digits)
                           for j in range(0, test_uri_length))
        self.logger.debug("Static file handler '{}' for test file {}".format(
                test_uri, test_file))
        self.websrv.add_handler(test_uri, StaticFileHandler(test_file))
        with open(media_file, 'rb') as fh:
            contents = fh.read()
        # Replace with URI and update file contents
        with open(media_file, 'wb+') as fh:
            fh.write(contents.replace(test_file, test_uri))
        self.set_udp_watch(zp, watch_str)
        self.verifyEqualOrFailCase(
            exp_result,
            self.start_playback(zp),
            "Expected to play stream")
        self.verifyTrueOrFailCase(
            self.wait_on_diag(zp, watch_str),
            "Expected diagnostic '{}' from player".format(watch_str))
        # Restore original file contents
        with open(media_file, 'wb+') as fh:
            fh.write(contents.replace(test_uri, test_file))
        self.logger.debug("Removing static file handler '{}'".format(test_uri))
        self.websrv.remove_handler(test_uri)
        zp.AVTransport.Stop_and_wait(timeout_seconds=30)

    def test_empty_data_file(self, zp, test_data):
        """
        This test confirms the player ignores an empty audio file.
        The player emits a diagnostic when attempting to digest the
        malformed audio file.
        :param :class:`sonos.client.internal.SonosZoneComponent` zp:
        :param tuple test_data: Test data pertinent to this test
        watch string

        self.set_udp_watch(zp, watch_str)
        fh = open(self.DATA_FILE_NAME, 'a')
        fh.close()
        self.verifyTrueOrFailCase(
            self.start_playback(zp),
            "Expected to play stream")
        # Should continue playing the rest of the audio files
        avts = AVTransportStateTransitionTimer(zp.AVTransport)
        avts.start(CONTINUOUS_PLAYING_SEC,
                   TRANSITION_TO_PLAYING_SEC,
                   user_defined_start_states=[zp.AVTransport.PLAYING_STATE])
        self.verifyTrueOrFailCase(
            not avts.state_changed,
            "Played continuously {} seconds".format(CONTINUOUS_PLAYING_SEC))
        self.verifyTrueOrFailCase(
            self.wait_on_diag(zp, watch_str),
            "Expected failure")
        zp.AVTransport.Stop_and_wait(timeout_seconds=30)
