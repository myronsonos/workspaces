    """
    Verify player conforms with the EXT-X-STREAM-INF tag.
    The tag indentifies that the next URI in the playlist is
    another playlist file.
    Confirm the player supports MAX_HLS_BITRATES (4) from
    all/oc/zone/zoneplayer/hlsplaylist.h and MAX_HLS_STREAMS_PER_BITRATE
    (4). For each unique bit rate, the player maintains up to 4
    URI for a potential 16 unique URI in the master playlist.
    Confirm the EXT-X-STREAM-INF tag is parsed when the line exceeds
    HLS_MAX_STREAM_INF (320) all/oc/zone/zoneplayer/hlsplaylist.h
    characters
    """
    TEST_TYPE = ["HLS_PLAYBACK"]
    TESTBED_SRV_PATH = "/srv/samba/Automation/HLS/bbc_live/"

                     "",
                     "data{:04d}.ts",

    )

[docs]    @combinatorial('stream_inf_data')
    def test_stream_inf_line(self, zp, master_content, watch_str):
        """
        :param :class:`sonos.client.internal.SonosZoneComponent` zp:
        :param str master_content: Master playlist content
        :param str watch_str: Expected diagnostic
        """
        with open(MASTER, "w+") as fh:
            fh.write(master_content)
        self.setup_player(zp)
        self.monitor_diagc.set_state(False)
        zp.diag.watch_udp_log_for_string(
            watch_str,
            lambda msg: self.monitor_diagc.set_state(True))
        self.verifyTrueOrFailCase(
            self.start_playback(zp, self.uri),
            "Expect to play stream {}".format(self.uri))
        self.verifyTrueOrStop(
            self.wait_on_diag(zp, watch_str),
            "Expect diagnostic {}".format(watch_str))
        self.cleanup_player(zp)
