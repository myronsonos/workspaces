        self.mock_server.start_service(smapi_properties=smapi_properties)
        wait_until_true(lambda: self.server_running(),
                        reason="Mock SMAPI service did not start",
                        timeout_seconds=10,
                        iteration_delay=1.0)
        self.logger.debug(
            "Run-time SMAPI properties=\"{}\"".format(
                self.mock_server.get_properties()))

                zp.diag.set_diag_level(diagm, self.DIAG_LEVEL_DICT[diagm])

        """
        context = re.search(ur"al:(\d+)", urllib.unquote(container))

[docs]    def get_track_title_and_id(self, zp):
        """
        Returns the track title and the track identifier.
        :param :class:`sonos.client.internal.SonosZoneComponent` zp:
        :return: Tuple string of track title, track identifier
        :rtype: :obj:`tuple`
        """
        pos_info = zp.AVTransport.GetPositionInfo()
        item = XMLAccessor(pos_info['TrackMetaData']).item
        track_title = item.title.value
        res = item.res.value
        track_id = urllib.unquote(re.search(ur':(\S+)\?', res).group(1))
        return (track_title, track_id)



[docs]    def wait_for_msg(self, zp, msg, stopped=True, timeout=10):
        """
        :param :class:`sonos.client.internal.SonosZoneComponent` zp:
        :param str watch_str: UDP watch string.
        :param int timeout: How long in seconds to wait
        :return: Has the zone generated the message ?
        :rtype: :obj:`bool`
        """
        self.logger.debug("Setting watch for \"{}\" on {}".format(msg, zp))
        self.event.clear()
        zp.diag.watch_udp_log_for_string(
            msg, lambda msg: self.event.set())
        if stopped:
            zp.AVTransport.Stop_and_wait()
        try:
            wait_until_true(
                lambda: self.event.isSet(),
                iteration_delay=1.0,
                timeout_seconds=timeout,
                reason="Unexpected timeout waiting for \"{}\"".format(msg))
        except TimeoutError:
            pass
        self.verifyTrueOrFailCase(
            self.event.isSet(), "Expect player to generate \"{}\"".format(msg))
