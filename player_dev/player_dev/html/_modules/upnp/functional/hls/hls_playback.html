    def _load_test_config(self):
        """
        Consume and make available the HLS_CONFIG file.

        config_fn = os.path.join(current_path, HLS_CONFIG)
        self.verifyTrueOrStop(os.path.exists(config_fn),
                    "expect test config file '{}' to exist".format(config_fn))
        config = load_workflow_execution_configuration_from_json(config_fn)

                "expect WorkflowExecutionConfiguration from '{}' "

        self.execution_configuration.as_dict().update(config.as_dict())

            hasattr(self.execution_configuration, "musicservices"),

[docs]    def monitor_playtime(self, msg):
        """
        Monitors the cumulative playtime on the HLS stream.
        :param str msg: Message from UDP logger.
        """
        m = re.search(ur"add: \d+ \d+ (\d+(?:\.\d+)?)", msg)
        if m:
            self.logger.debug("cumulative playtime {}".format(m.group(1)))
            self.cumulative_playtime = float(m.group(1))


    def _cleanup_avt(self, zp):
        """
        Restores the player's AVT state.
        :param :class:`sonos.client.internal.SonosZoneComponent` zp:
        """
        zp.DeviceProperties.cleanup_bonded_configuration()
        zp.AVTransport.SetAVTransportURI("", "")
        zp.AVTransport.clear_queue_and_wait()
        zp.AVTransport.clear_play_state()
        zp.AVTransport.SetPlayMode(zp.AVTransport.PLAYMODE_NORMAL)
        zp.RenderingControl.SetVolume(1)

    def _setup(self, zp):
        """
        Sets up the player's AVT state for play back and starts
        a UDP log on the player.

        :param :class:`sonos.client.internal.SonosZoneComponent` zp:
        """
        self._cleanup_avt(zp)
        zp.diag.start_udp_log()
        zp.diag.set_diag_level("hlsaudio", 4)
        zp.diag.set_diag_level("hlsmeta", 4)
        zp.diag.watch_udp_log_for_string(
            self.MONITOR_PLAYTIME_DIAGC,
            lambda msg: self.monitor_playtime(msg))

    def _cleanup(self, zp):
        """
        Resets UDP diagnostic levels to default.
        Stops UDP logging and restores player's AVT state.
        :param :class:`sonos.client.internal.SonosZoneComponent` zp:
        """
        zp.diag.set_diag_level("hlsaudio", 0)
        zp.diag.set_diag_level("hlsmeta", 0)
        zp.diag.stop_udp_log()
        self._cleanup_avt(zp)

        """
        Adds the specified accounts taken from the configuration file.
        """
        zp = self.test_zone
        for service_id in self.service_names:
            credentials = self.service_names[service_id]
            res = zp.SystemProperties.AddAccountX(service_id,
                                                  credentials.user,
                                                  credentials.password)
            self.verifyTrueOrFailCase(
                res,
                "Added service {} {}".format(service_id, credentials.name))



[docs]    @retry(attempts=3, expected_exception=UPnPError, retry_delay=5,
           exception_verifier=lambda e: e.upnp_error == UPNP_ERROR_402_INVALID_ARGUMENTS)
    def playback_stream(self, zp, uri, metadata, accum_play, accum_time):
        """
        Play a stream based on the vector.
        If the vector specifies accumulation play time, the
        test waits for the player to accumulate the specified play time
        and then plays back the play pipe line.

        :param :class:`sonos.client.internal.SonosZoneComponent` zp:
        :param str uri: URI string to set AVT.
        :param str metadata: Metadata string or empty string.
        :param bool accum_play: Wait for player to pre-fetch audio stream.
        :param float accum_time: Seconds of pre-fetch audio stream.
        """
        self._setup(zp)
        zp.AVTransport.SetAVTransportURI(uri, metadata)
        self.cumulative_playtime = 0.0
        zp.AVTransport.Play_and_wait()
        if accum_play:
            try:
                wait_until_true(
                    lambda: self.cumulative_playtime < accum_time,
                    iteration_delay=1.0,
                    timeout_seconds=int(accum_time))
            except TimeoutError:
                self.fail("Didn't accumulate enough play time")
        ret = zp.AVTransport.wait_for_avt_state(
            zp.AVTransport.STOPPED_STATE,
            timeout_seconds=int(accum_time))
        self.verifyFalseOrFailCase(
            ret,
            "Interrupted {} seconds of stream ?".format(accum_time))
        zp.AVTransport.Stop_and_wait()
        self._cleanup(zp)

[docs]    @combinatorial('siriusxm_hls_test_data')
    def test_siriusxm_hls_playback(self, zp, uri, metadata, accum_play, accum_time):
        """
        Play SiriusXM HLS stream.
        :param :class:`sonos.client.internal.SonosZoneComponent` zp:
        :param str uri: URI string to set AVT.
        :param str metadata: Metadata string or empty string.
        :param bool accum_play: Wait for player to pre-fetch audio stream.
        :param float accum_time: Seconds of pre-fetch audio stream.
        """
        if not self.have_siriusxm_service:
            self.skip(
                "Skip SiriusXM playback - service not available on test bed registered ID {}".format(
                    self.test_zone.SystemProperties.customer_id))
        account = self.test_zone.SystemProperties.get_service_account(SIRIUS_XM_SERVICE_ID)
        self.verifyTrueOrStop(
            len(account) > 0,
            "Expect Sirius-XM account to exist on player")
        self.verifyEqualOrStop(
            SIRIUS_XM_SERVICE_ID,
            int(account[0].service_id),
            "Expect Sirius-XM account ID to match from settings")
        self.logger.debug(
            "Using serial number '{}' in URI".format(account[0].serial_number))
        self.logger.debug(
            "Using service id '{}' user name '{}' in metadata".format(
                account[0].service_id,
                account[0].username))
        self.playback_stream(
            self.test_zone, uri.format(account[0].serial_number),
            metadata.format(account[0].service_id, account[0].username),
            accum_play, accum_time)
