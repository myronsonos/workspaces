    """
    Verifies SWPBL-57557
    Plays a track with OOB metadata & verifies CurrentURI has the correct flags
    for OOB metadata
    """
    TEST_TYPE = ["STATIC_HLS"]

    def _play_oob_metadata(self, zp):
        """
        Adds OOB metadata sample to queue & plays
        :param zp: Sonos Zone Component
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`
        """
        metadata = zp.AVTransport.generate_didl_lite(
            title="AOD Static HLS OOB",
            svc_id=self.svc_acct.service_id,
            acct_id=self.svc_acct.serial_number)
        zp.AVTransport.add_to_queue(CONTAINER, metadata)
        zp.RenderingControl.SetVolume(5)
        zp.AVTransport.set_queue_position(OOB_METADATA_TRACK_POS)
        zp.AVTransport.Play_and_wait()


[docs]    @skip("Skip until we're able to make use of a local mockSMAPI instance")
    @combinatorial('generate_testbed_unique_playback_devices')
    def test_static_hls_oob_metadata(self, zp):
        """
        1. Add OOB metadata track to queue
        2. Play
        3. Call AVTransport.GetMediaInfo()
        4. Verify CurrentURI flag has 9th bit set, which indicates that OOB
           metadata is enabled
        :param zp: Sonos Zone Component
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`
        """
        self._setUpTest(zp)
        self._play_oob_metadata(zp)
        position_info = zp.AVTransport.GetPositionInfo()
        self.logger.info(position_info)
        track_uri = position_info['TrackURI']
        self.logger.info(track_uri)
        flags_regex = u'(?<=flags=)(.*)(?=&)'
        res = re.search(flags_regex, track_uri)
        self.logger.info(res)
        flags = int(res.group(1))
        self.logger.info(flags)
        self.verifyNotEqualOrFailCase(
            0, flags & NINTH_BIT, 'bitwise AND should NOT equal 0')
