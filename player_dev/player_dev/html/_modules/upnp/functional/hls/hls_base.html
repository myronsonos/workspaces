[docs]    def web_files(self, stream_files):
        """
        Copies files from local server.
        Posts a web handler for each file.
        Note that this method to be used when the test changes the
        web content; playlist's modified, data files removed etc.
        :param tuple stream_files: tuple containing the names of the web
           files
        """
        for stream in stream_files:
            for file in stream[:2]:
                if file:
                    shutil.copyfile(self.TESTBED_SRV_PATH + file, file)
                    self.websrv.add_handler(
                        file,
                        StaticFileHandler(file))
                    self.data_files.append(file)
            audiofmt, beg, end = stream[2:]
            if audiofmt:
                for index in range(beg, end):
                    audio_file = audiofmt.format(index)
                    shutil.copyfile(
                        self.TESTBED_SRV_PATH + audio_file,
                        audio_file)
                    self.websrv.add_handler(
                        audio_file,
                        StaticFileHandler(audio_file))
                    self.data_files.append(audio_file)

[docs]    def tearDownFixture(self):
        """
        Terminates the master playlist thread.
        Shuts down the web server.
        Removes all files used as web content.
        """
        if self.master_thread:
            self.master_thread.shutdown()
        self.websrv.stop()
        for data_file in self.data_files:
            try:
                os.unlink(data_file)
            except OSError:
                pass
        for dir_name in self.dirs:
            shutil.rmtree(dir_name, ignore_errors=True)



[docs]    def create_playlist(self, name, files, remove=True):
        """
        Create a playlist file and populate with the files.
        :param str name: Playlist name
        :param tuple files: A tuple of file name(s) in playlist
        :param bool remove: Schedule for file removal at tear down.
        """
        with open(name, 'w+') as fh:
            filenum = 1
            fh.write('[playlist]\n')
            fh.write('NumberOfEntries={}\n'.format(len(files)))
            if remove:
                self.data_files.append(name)
            for filename in files:
                fh.write(
                    'File{}=http://{}/{}\n'.format(
                        filenum, self.websrv.host(), filename))
                filenum = filenum + 1



[docs]    def start_playback(self, zp, uri, timeout_seconds=15,
                       except_num=UPNP_OK):
        """
        :param :class:`sonos.client.internal.SonosZoneComponent` zp:
        :param str uri: URI string to set AV transport.
        :param int timeout_seconds: Maximum time in seconds to wait
           for playback to start.
        :param int except_num: Expected UPnPError exception number
        """
        ret = zp.AVTransport.SetAVTransportURI(
            uri,
            zp.AVTransport.generate_didl_lite())
        self.verifyTrueOrFailCase(
            ret,
            "Set AV transport URI {}".format(uri))
        zp.RenderingControl.SetVolume(5)
        try:
            zp.AVTransport.Play_and_wait(timeout_seconds=timeout_seconds)
        except UPnPError as e:
            if except_num > UPNP_OK and except_num == e.upnp_error:
                self.logger.info("Caught expected exception {}".format(e))
                return False
            raise e
        except TimeoutError:
            return False
        return True



[docs]    def start_queue_playback(self, zp, queue_id=0,
                             position=1, timeout_seconds=15):
        """
        Starts playback on desired queue using the first track.
        :param :class:`sonos.client.internal.SonosZoneComponent` zp:
        :param int queue_id: Default associated playback queue.
        :param int position: Defual track position in queue.
        :param int timeout_seconds: Default seconds to wait for playback.
        """
        zp.AVTransport.use_queue(queue_id=queue_id)
        zp.AVTransport.set_queue_position(
            position=position, instance_id=queue_id)
        self.verifyEqualOrFailCase(
            position,
            zp.AVTransport.get_queue_position(),
            "Track position for playback")
        zp.RenderingControl.SetVolume(5)
        played = True
        try:
            zp.AVTransport.Play_and_wait(timeout_seconds=timeout_seconds)
        except TimeoutError:
            played = False
        return played

        """
        :param :class:`sonos.client.internal.SonosZoneComponent` zp:
        :param str watch_str: UDP watch string.
        :param int timeout_seconds: Time in seconds to wait for diag.
        :return: The state of the call back. True if the particular
           watch string has been captured and the call back invoked
           or False.
        :rtype: :obj:`bool`
        """
        ret = self.wait_on_any_diag(zp, timeout_seconds=timeout_seconds)
        zp.diag.stop_watching_udp_log_for_string(watch_str)
        return ret



[docs]    def wait_on_any_diag(self, zp, timeout_seconds=60):
        """
        :param :class:`sonos.client.internal.SonosZoneComponent` zp:
        :param int timeout_seconds: Time in seconds to wait for diag.
        :return: The state of the call back. True if the particular
           watch string has been captured and the call back invoked
           or False.
        :rtype: :obj:`bool`
        """
        try:
            wait_until_true(
                lambda: self.monitor_diagc.get_state(),
                iteration_delay=1,
                timeout_seconds=timeout_seconds)
        except TimeoutError:
            pass
        return self.monitor_diagc.get_state()

[docs]    def fetch_files_from_server(self, remote_path, dir_list):
        """
        Copy list of directories
        TODO SWPBL-46746: Convert copying from local file system
        to copying from remote NAS server.
        :param str remote_path: Remote source path
        :param list dir_list: Directories to copy
        """
        if not os.path.exists(remote_path):
            return False
        for dir_name in dir_list:
            try:
                shutil.copytree(remote_path + dir_name, dir_name)
                self.dirs.append(dir_name)
            except OSError, e:
                if e.errno != os.errno.EEXIST:
                    raise
        return True

[docs]    def master_playlist_thread(self, name, medias):
        """
        Creates a thread which manages the master and media
        playlists for multi bit rate audio codecs.
        :param str name: Master playlist name
        :param tuples :obj:`MediaPlaylistFormat` medias:Tuples
        of media playlist(s)
        """
        self.master_thread = MasterPlaylistThread(name)
        for media in medias:
            m = MediaPlaylist(media.name,
                              media.uripat,
                              media.numsegs,
                              media.seglen,
                              media.filepat,
                              media.numfiles,
                              media.seqnum,
                              encrypted=media.encrypted,
                              bandwidth=media.bandwidth,
                              codecs=media.codecs)
            self.master_thread.push(m)
        self.master_thread.write()
        self.master_thread.start()
