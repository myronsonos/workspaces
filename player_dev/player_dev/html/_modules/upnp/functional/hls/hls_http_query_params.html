    reference audio file streams

    Verifies SWPBL-41171.

    AAC_MASTER_TEMPLATE = """#EXTM3U

[docs]    def setup_websrv(self):
        """
        Sets up the local web server.
        Creates a handler dictionary for the web server.
        Starts web server.
        """
        self.abs_path = os.path.dirname(os.path.realpath(__file__))
        self.websrv = WebServer(start_reactor=False)
        self.websrv.start(
            port=get_random_web_port_to_use(),
            path=self.abs_path)
        self.hls_uri = "hls-aac://{}/{}".format(
            self.websrv.host(),
            self.MASTER_HLS_NAME)
        for name in self.data_files:
            shutil.copyfile(self.TESTBED_SRV_PATH + name, name)
        # Add handlers for the master, media and audio files
        self.websrv.add_handler(self.MASTER_HLS_NAME,
                                StaticFileHandler(self.MASTER_HLS_NAME))
        self.websrv.add_handler(self.MEDIA_HLS_NAME,
                                StaticFileHandler(self.MEDIA_HLS_NAME))
        for data_file in self.data_files:
            self.websrv.add_handler(data_file,
                                    StaticFileHandler(data_file))


    def _setup_master_and_media_files(self, master, media, media_fmt,
                                      query_str, suffix):
        """
        Creates master and media files based on the supplied parameters.
        :param str master: Master playlist template.
        :param str media: Media template.
        :param str media_fmt: Media reference format.
        :param str query_str: HTTP query and params string
        :param str suffix: Audio file type, 'AAC' or 'TS'
        """
        with open(self.MASTER_HLS_NAME, 'wb+') as fh:
            fh.write(master.format(self.websrv.host(), query_str))
        with open(self.MEDIA_HLS_NAME, 'wb+') as fh:
            fh.write(media)
            for data_file in filter(lambda x: suffix in x, self.data_files):
                fh.write(media_fmt.format(
                    self.websrv.host(),
                    data_file,
                    query_str))

    def _cleanup_avt(self, zp):
        """
        Resets the state of the AV on the zone player.
        :param :class:`sonos.client.internal.SonosZoneComponent` zp:
        The zone component under test.
        """
        zp.AVTransport.clear_play_state()
        zp.AVTransport.clear_queue_and_wait()
        zp.RenderingControl.SetVolume(0)
        zp.AVTransport.SetPlayMode(zp.AVTransport.PLAYMODE_NORMAL)

    def _setup(self, zp):
        """
        """
        self._cleanup_avt(zp)
        zp.diag.start_udp_log()
        for diagm in self.DIAG_LEVEL_DICT:
            zp.diag.set_diag_level(diagm, self.DIAG_LEVEL_DICT[diagm])

    def _cleanup(self, zp):
        """
        :param :class:`sonos.client.internal.SonosZoneComponent` zp:
        """
        self._cleanup_avt(zp)
        for diagm in self.DIAG_LEVEL_DICT:
            zp.diag.set_diag_level(diagm, 0)
        zp.diag.stop_udp_log()

[docs]    def start_playback(self, zp, uri, timeout_seconds=15):
        """
        :param :class:`sonos.client.internal.SonosZoneComponent` zp:
        :param str uri: URI string
        :param int timeout_seconds: Maximum time in seconds to wait
        for playback to start.
        """
        ret = zp.AVTransport.SetAVTransportURI(
            uri,
            zp.AVTransport.generate_didl_lite())
        self.verifyTrueOrFailCase(
            ret,
            "Set AV transport URI {}".format(uri))
        zp.RenderingControl.SetVolume(5)
        played = True
        try:
            zp.AVTransport.Play_and_wait(timeout_seconds=timeout_seconds)
        except TimeoutError:
            played = False
        return played



[docs]    def wait_on_diag(self, zp, watch_str, timeout_seconds=60):
        """
        :return: The state of the call back. True if the particular
        watch string has been captured and the call back invoked
        or False.
        :rtype: :obj:`bool`
        :param :class:`sonos.client.internal.SonosZoneComponent` zp:
        :param str watch_str: UDP watch string.
        :param int timeout_seconds: Time in seconds to wait for diag.
        """
        try:
            wait_until_true(
                lambda: self.monitor_diagc.get_state(),
                iteration_delay=1,
                timeout_seconds=timeout_seconds)
        except TimeoutError:
            pass
        zp.diag.stop_watching_udp_log_for_string(watch_str)
        return self.monitor_diagc.get_state()



[docs]    @combinatorial('query_params_test_data')
    def test_http_query_params(self, zp, master, media, media_fmt,
                               query_str, suffix):
        """
        Given a set of input parameters, the test confirms the player
        plays the HLS stream.
        Build a master and media playlist files containing the test criteria.
        The test criteria are the whether or not there's a query string in the
        HTTP reference and the type of media stream, AAC or TS.
        :param :class:`sonos.client.internal.SonosZoneComponent` zp:
        :param str master: Master playlist template.
        :param str media: Media template.
        :param str media_fmt: Media reference format.
        :param str query_str: HTTP query and params string
        :param str suffix: Audio file type, 'AAC' or 'TS'
        """
        self._setup(zp)
        self._setup_master_and_media_files(
            master,
            media,
            media_fmt,
            query_str,
            suffix)
        self.monitor_diagc.set_state(False)
        watch_str = self.OPEN_STREAM_DIAGC.format(self.websrv.host())
        zp.diag.watch_udp_log_for_string(
            watch_str,
            lambda msg: self.monitor_diagc.set_state(True))
        self.verifyTrueOrFailCase(
            self.start_playback(zp, self.hls_uri),
            "Expect to play stream '{}' {} {}".format(
                self.hls_uri,
                zp.ip,
                zp.modelNumber))
        start = time.time()
        ret = zp.AVTransport.wait_for_avt_state(
            zp.AVTransport.STOPPED_STATE,
            timeout_seconds=35)
        end = time.time()
        self.verifyGreaterOrFailCase(
            end-start,
            20.0,
            "Played stream un-interrupted ?")
        self.verifyTrueOrFailCase(
            self.wait_on_diag(zp, watch_str, timeout_seconds=15),
            "Expected diagnostic string '{}' from player {}".format(
                watch_str, zp.ip))
        zp.AVTransport.Stop_and_wait()
        self._cleanup(zp)
