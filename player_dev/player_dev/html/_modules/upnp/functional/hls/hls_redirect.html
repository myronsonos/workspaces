        """
        Using absolute URI for now until we clarify what the HTTP spec
        governs on relative URI handling.

[docs]    def remove_redirects(self, websrv, redirect_tuple):
        """
        Removes the specified handlers in the tuple from the
        web server.
        :param :class:`sonos.websrv.WebServer` websrv:
        :param tuple redirect_tuple: Tuple of entries to be redirected
        """
        for key in redirect_tuple:
            websrv.remove_handler(key)



[docs]    def setup_redirects(self, websrv, redirect_tuple):
        """
        Sets up redirect handlers using the supplied web server object
        and the tuple containing the URI(s) to be redirected.
        Last entry in the tuple is the target URI after all the
        previous entries have been redirected.
        :param :class:`sonos.websrv.WebServer` websrv:
        :param tuple redirect_tuple: Tuple of entries to be redirected
        """
        self.remove_redirects(websrv, redirect_tuple)
        for x in range(0, len(redirect_tuple)-1):
            redirect_uri = self.REDIRECT_URI.format(
                websrv.host(),
                redirect_tuple[x+1])
            websrv.add_handler(
                redirect_tuple[x],
                RedirectHandler(
                    redirect_tuple[x],
                    redirect_uri=redirect_uri))
        # Add handler for the target URI
        websrv.add_handler(
            redirect_tuple[-1],
            StaticFileHandler(redirect_tuple[-1]))

        """
        Sets up a secondary web server whose function is to redirect
        the requests to the default web server.
        """
        self.redirect_dir = tempfile.mkdtemp()
        self.redirect_websrv = WebServer(start_reactor=False)
        self.redirect_websrv.start(
            port=self.REDIRECT_WEB_PORT,
            path=self.redirect_dir)

[docs]    def verify_redirect(self, zp, uri, websrv, redirect_tuple, play):
        """
        Common method to verify redirects when testing with a single
        web server. Sets up the AV transport with the specified URI,
        enables a watch on the appropriate diagnostic based on True
        or False, commences playback, verifies, removes handlers from
        the we server and cleans up the AV transport state.
        :param :class:`sonos.client.internal.SonosZoneComponent` zp:
        :param str uri: The URI to set AV transport.
        :param :class:`sonos.websrv.WebServer` websrv:
        :param tuple redirect_tuple: Contains URI for redirection
        :param bool play: Is player expected to play stream ?
        """
        self.setup_player(zp)
        self.monitor_diagc.set_state(False)
        watch_str = self.expect_diag[play]
        zp.diag.watch_udp_log_for_string(
            watch_str,
            lambda msg: self.monitor_diagc.set_state(True))
        self.setup_redirects(websrv, redirect_tuple)
        ret = self.start_playback(zp, uri)
        self.verifyEqualOrFailCase(
            play,
            ret,
            "Should match expected play state")
        self.verifyTrueOrFailCase(
            self.wait_on_diag(zp, watch_str, timeout_seconds=10),
            "Expected diagnostic '{}'".format(watch_str))
        self.remove_redirects(websrv, redirect_tuple)
        self.cleanup_player(zp)

[docs]    def verify_media_redirect(self, zp, watch_str, stop_test_str):
        """
        Verifies a redirected media playlist.
        The test expects playback to start on a stream which the
        media playlist has been redirected to another URI.
        Subsequent media open requests are directed to the resolved
        server handling the media playlist hence the test is expecting
        a single diagnostic redirect to be issued by the player.
        :param :class:`sonos.client.internal.SonosZoneComponent` zp:
        :param str watch_str: Diagnostic string to expect
        :param str stop_test_str: Diagnostic string to stop test.
        """
        uri = "hls-aac://{}/{}".format(
            self.redirect_websrv.host(),
            self.MASTER_NAME)
        self.setup_player(zp)
        self.media_redirects = 0
        self.monitor_diagc.set_state(False)
        zp.diag.watch_udp_log_for_string(
            stop_test_str,
            lambda msg: self.monitor_diagc.set_state(True))
        zp.diag.watch_udp_log_for_string(
            watch_str,
            lambda msg: self.media_redirect_cb(msg))
        self.verifyTrueOrFailCase(
            self.start_playback(zp, uri),
            "Expecting redirected stream to be played")
        self.verifyTrueOrFailCase(
            self.wait_on_diag(zp, stop_test_str, timeout_seconds=30),
            "Expected diagnostic '{}'".format(stop_test_str))
        self.verifyEqualOrFailCase(
            3,
            self.media_redirects,
            "Expecting 3 re-directs")
        zp.diag.stop_watching_udp_log_for_string(stop_test_str)
        self.remove_redirects(self.redirect_websrv,
                              self.REDIRECT_WEB_HANDLERS)
        self.cleanup_player(zp)

    def test_segments_redirect_websrv(self, zp):
        """
        Server A serves master and media playlists but one of the
        audio segments is redirected to server B.
        The temporary directory contains the master, media and all
        the non-redirected segments. The data0001.ts is redirected
        therefore it's not copied to the redirect server's directory.
        :param :class:`sonos.client.internal.SonosZoneComponent` zp:
        """
        self.cleanup_tempdir()
        shutil.copyfile(self.TESTBED_SRV_PATH + self.MASTER_NAME,
                        self.redirect_dir + '/' + self.MASTER_NAME)
        with open(self.redirect_dir + '/' + self.MEDIA_NAME, "wb+") as fh:
            fh.write(self.MEDIA_TEMPLATE)
        shutil.copyfile(self.TESTBED_SRV_PATH + "data0000.ts",
                        self.redirect_dir + '/' + "data0000.ts")
        # Skipping redirected file data0001.ts
        for index in range(2, 4):
            audio_file = "data{:04d}.ts".format(index)
            shutil.copyfile(self.TESTBED_SRV_PATH + audio_file,
                            self.redirect_dir + '/' + audio_file)
        name = self.SRV_REDIRECT_SEGMENT_NAME
        self.remove_redirects(self.redirect_websrv, self.REDIRECT_WEB_HANDLERS)
        redirect_uri = "http://{}/{}".format(self.websrv.host(), name)
        self.redirect_websrv.add_handler(
            name,
            RedirectHandler(name, redirect_uri=redirect_uri))
        watch_str = "redirect: http://{}/{} -> http://{}/{}".format(
            self.redirect_websrv.host(),
            name,
            self.websrv.host(),
            name)
        self.verify_server_redirect(zp, watch_str)
