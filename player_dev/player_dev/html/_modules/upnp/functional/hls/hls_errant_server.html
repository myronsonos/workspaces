    """
    A L{StaticProducer} that writes the entire file to the request.
    Modified to delay the final connection close after sending the
    last frame of the file in order to replicate a condition where
    the player is waiting for the close (which is delayed) causing
    underflow errors on chsnk.
    The player will initiate the close when it consumes the content
    length of the HTTP GET response (SWPBL-41450).
    """

        """
        Call back to gracefully stop the producer.
        Check if the request object exists, it may have been deleted
        if the player sent a close as expected.
        """
        if self.request:
            self.request.unregisterProducer()
            self.request.finish()
        self.stopProducing()



[docs]    def resumeProducing(self):
        """
        After sending all data in the file, delay the close
        on the connection by registering for a delayed call back.
        This emulates some HTTP servers that don't close the connection
        immediately after receiving acknowledgement from the client.
        """
        if not self.request:
            return
        data = self.fileObject.read(self.bufferSize)
        if data:
            # this .write will spin the reactor, calling .doWrite and then
            # .resumeProducing again, so be prepared for a re-entrant call
            self.request.write(data)
        else:
            # A timeout of 10 seconds per oc/zone/zoneplayer/audiodecoder.h
            # define RAF_IO_TIMEOUT 10000
            reactor.callLater(10, self.delay_stop)
            return server.NOT_DONE_YET

    """

        """

        :param fileObject: The file the contents of which to write to the
            request.
        :param int delay: Delay time
        :param int buffer_size: Size of buffer
        """
        super(NoRangeStaticProducerDelaySegment, self).__init__(
            request, fileObject)
        self.delay = delay
        if buffer_size:
            self.bufferSize = min(buffer_size, self.bufferSize)

[docs]    def resumeProducing(self):
        """
        Delay the last frame we're about to send to trigger
        an action by the player.
        """
        if not self.request:
            return
        data = self.fileObject.read(self.bufferSize)
        if data:
            # this .write will spin the reactor, calling .doWrite and then
            # .resumeProducing again, so be prepared for a re-entrant call
            # If we're about to send the last frame delay it to elicit
            # player behavior.
            if (self.bytesWritten + len(data)) < self.size:
                self.bytesWritten = self.bytesWritten + len(data)
                self.request.write(data)
            else:
                # A timeout of 10 seconds per oc/zone/zoneplayer/audiodecoder.h
                # define RAF_IO_TIMEOUT 10000
                if not self.sentLastFrameOnce:
                    reactor.callLater(self.delay, lambda: self.delay_frame_cb(data))
                    return server.NOT_DONE_YET
        else:
            self.request.unregisterProducer()
            self.request.finish()
            self.stopProducing()

[docs]    def resumeProducing(self):
        """
        Delay the last frame we're about to send to trigger
        an action by the player.
        """
        if not self.request:
            return
        data = self.fileObject.read(self.bufferSize)
        if data:
            # this .write will spin the reactor, calling .doWrite and then
            # .resumeProducing again, so be prepared for a re-entrant call
            # If we're about to send the last frame double its contents
            # in order to elicit player behavior.
            if (self.bytesWritten + len(data)) < self.size:
                self.bytesWritten = self.bytesWritten + len(data)
                self.request.write(data)
            else:
                self.bytesWritten = self.bytesWritten + len(data)
                # Force overwrite on the pipeline, extra data should be thrown
                # by the player.
                self.request.write(data[:] + data[:])
        else:
            self.request.unregisterProducer()
            self.request.finish()
            self.stopProducing()

[docs]    def makeProducer(self, request, fileForReading):
        """
        Make a L{StaticProducer} that will produce the body of this response.
        This method will also set the response code and Content-* headers.
        @param request: The L{Request} object.
        @param fileForReading: The file object containing the resource.
        @return: A L{StaticProducer}.  Calling C{.start()} on this will begin
            producing the response.
        """
        ret = super(StaticFileHandlerProducerOverride, self).makeProducer(
            request, fileForReading)
        for producer in self.static_range_producers:
            if isinstance(ret, producer):
                if isinstance(ret, SingleRangeStaticProducer):
                    return self.static_range_producers[producer](
                        request, fileForReading, ret.offset, ret.size)
                elif isinstance(ret, MultipleRangeStaticProducer):
                    return self.static_range_producers[producer](
                        request, fileForReading, ret.rangeInfo)
                elif isinstance(ret, NoRangeStaticProducer):
                    return self.static_range_producers[producer](
                        request, fileForReading)
        return ret




[docs]class HLSErrantServer(HLSBaseFixture):
    """
    Some HTTP servers when replying to a GET request to
    read a file don't terminate the connection after the
    last frame containing data has been acknowledged by
    the client (SWPBL-41450).
    This behavior results in underlow errors in chsnk
    while attempting to play an HLS stream hosted on such
    servers.
    Verify the player fetches data up to the content length
    specified in the server's reply and then the connection
    close is initiated from the player.
    Verify the player ignores extra data beyond the specified
    content length.
    Verify the player terminates the playback due to the server
    timing out on sending the data.
    """
    TEST_TYPE = ["HLS_REGRESSION"]
    TESTBED_SRV_PATH = "/srv/samba/Automation/HLS/bit-rate/"
    TESTBED_SRV_MP3_PATH = "/srv/samba/Automation/cpu/MaggieFarm/"
    TESTBED_SRV_MP3_PATH = "/srv/samba/Automation/HLS/"
    MP3_NAME = "01-EveryMorning.mp3"
    MASTER_HLS_NAME = "MaggieFarm.m3u8"
    MEDIA64_HLS_NAME = "MaggieFarm64k.m3u8"

    MASTER_TEMPLATE = """#EXTM3U
#EXT-X-STREAM-INF:AVERAGE-BANDWIDTH=64000,BANDWIDTH=64000,CODECS="mp4a.40.2"
MaggieFarm64k.m3u8
"""

    # Media template - EXT-X-MEDIASEQUENCE replaced with test data
    MEDIA_TEMPLATE = """#EXTM3U
#EXT-X-TARGETDURATION:10
#EXT-X-VERSION:2
#EXT-X-MEDIA-SEQUENCE:143419997
"""

    """
    Number of audio segments to use during test.
    """
    AUDIO_SEGMENTS = 12

    # Evil errors expected/not expected during playback.
    DIAGC_ERRORS = ("setoutput to begin failed, initial offset:",
                    "unable to write to stream error; written -4",
                    "large sync error triggered resync",
                    "underflow: error code 5:",
                    "LSE: stream underflow",
                    "stream reset on write",
                    "Transport error ERROR_LOST_CONNECTION",
                    "Transport error ERROR_BUFFERING",)

    STATIC_RANGE_PRODUCERS_OVERRIDES = (
        ({NoRangeStaticProducer: NoRangeStaticProducerDelayClose}, False),
        ({NoRangeStaticProducer: NoRangeStaticProducerOver}, False),
        ({NoRangeStaticProducer: NoRangeStaticProducerDelaySegment}, True),
        )

[docs]    def player_common_playback(self, zp, uri):
        """
        :param :class:`sonos.client.internal.SonosZoneComponent` zp:
        """
        self.setup_player(zp)
        self.verifyTrueOrFailCase(
            self.start_playback(zp, uri),
            "Expecting stream to be playing")
        playtime = 60.0
        start_play = time.time()
        ret = zp.AVTransport.wait_for_avt_state(
            zp.AVTransport.STOPPED_STATE,
            timeout_seconds=int(playtime))
        end_play = time.time()
        self.verifyFalseOrFailCase(
            ret,
            "Expect player not to switch AV states")
        self.verifyGreaterOrFailCase(
            end_play - start_play,
            playtime,
            "Played in excess of {} seconds".format(playtime))
        self.verifyFalseOrFailCase(
            self.monitor_diagc.get_state(),
            "Not expecting playback errors")
        self.cleanup_player(zp)
