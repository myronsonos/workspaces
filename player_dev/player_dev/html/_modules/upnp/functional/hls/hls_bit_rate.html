    """
    Conduct HLS bit rate adaptation tests.
    Based on available network bandwidth, the player selects
    the appropriate media playlist that can support the play back.
    Note that the initial implementation used the log inspector
    to confirm the player was opening (reading) the correct files
    that supported the available bandwidth.
    The problem with using the log inspector in this particular
    test is that it provided false positives/negatives since it
    was reading the contents of the anacapa trace which may have
    had the expected diagnostics from the current test or previous
    incarnations of tests which utilized the same playlists and
    just the right logging level.
    The UDP logger is more suitable to this test since it instantiates
    a logging session that is unique to the test.
    Addresses SWPBL-46155
    The script tests on specific players since not all players can
    switch to higher bit rates (when that rate is available) due to
    SWPBL-46264.
    """
    STREAM_FILES = (("",
                     "",
                     "",
                     0,
                     0),)
    TESTBED_SRV_PATH = '/srv/samba/Automation/HLS/bit-rate/'

    def _parse_args(self):
        parser = argparse.ArgumentParser()
        parser.add_argument(
            "--high_bitrate", required=False, type=int,
            default=DEFAULT_HIGH_BITRATE,
            help="Upper limit on bit rate - default".format(
                DEFAULT_HIGH_BITRATE))
        parser.add_argument(
            "--transition_timeout", required=False, type=int,
            default=DEFAULT_TRANSITION_TIMEOUT_SEC,
            help="Maximum wait for transition in seconds - default".format(
                DEFAULT_TRANSITION_TIMEOUT_SEC))
        args, unknown_args = parser.parse_known_args()
        self.high_bitrate = args.high_bitrate
        self.transition_timeout = args.transition_timeout
        self.logger.info(
            "Using high bit rate {} bps".format(self.high_bitrate))
        self.logger.info(
            "Using transition timeout {} seconds".format(
                self.transition_timeout))

    def _get_fixture_path(self):
        """
        Return the absolute path this module is executing from.
        """
        return os.path.dirname(
            os.path.abspath(inspect.getfile(self.__class__)))

    def _load_test_config(self):
        """
        Consume and make available the HLS_CONFIG file.
        """
        current_path = self._get_fixture_path()
        config_fn = os.path.join(current_path, HLS_CONFIG)
        self.verifyTrueOrStop(
            os.path.exists(config_fn),
            "expect test config file '{}' to exist".format(config_fn))
        config = load_workflow_execution_configuration_from_json(config_fn)
        self.verifyIsNotNoneOrStop(
            config,
            "expect WorkflowExecutionConfiguration from '{}' "
            "to contain content".format(config_fn))
        self.execution_configuration.as_dict().update(config.as_dict())
        self.verifyTrueOrStop(
            hasattr(self.execution_configuration, "traffic_shaper"),
            "expect 'traffic_shaper' section to be specified "
            "in test configuration after loading")
        self.traffic_shaper = self.execution_configuration.traffic_shaper
        self.logger.debug(
            "Traffic shaper {} iface {}".format(
                self.traffic_shaper.host, self.traffic_shaper.interface))

    def _set_bit_rate_and_monitor(
            self, zp, bit_rate, watch_exps, start_playback):
        self.tc.disable_rate_limit(self.traffic_shaper.interface)
        self.tc.enable_rate_limit(
            self.traffic_shaper.interface, zp, bit_rate, port=self.web_port)
        self.logger.debug(
            "Setting tc rate to {}bits playback={} on port {}".format(
                bit_rate, start_playback, self.web_port))
        [self.setup_diag_watch(zp, watch_exp) for watch_exp in watch_exps]
        if start_playback:
            self.verifyTrueOrFailCase(
                self.start_playback(zp, self.uri),
                "Expecting playback to succeed")
        self.verifyTrueOrStop(
            self.wait_on_any_diag(zp, timeout_seconds=self.transition_timeout),
            "{} should have been logged".format(watch_exps))
        # Remove watch
        [zp.diag.stop_watching_udp_log_for_string(watch_exp)
         for watch_exp in watch_exps]

    def _test_bit_rate_change(self, zp, bit_rate_list):
        """
        The player's HLS cached URI (last URI that was played) must
        be cleared prior to starting the bit rate tests.
        The playlist thread needs to reset the media playlist(s)
        """
        self.clear_hls_uri_cache(zp)
        self.master_playlist_reset()
        self.setup_player(zp)
        for index, (bit_rate, logmsg) in enumerate(bit_rate_list):
            start_playback = False
            if index == 0:
                start_playback = True
            self._set_bit_rate_and_monitor(
                zp, bit_rate, logmsg,
                start_playback)

    def test_low_to_high_bit_rate_low_high(self, zp):
        """
        The tests expects the player to use the 64K, 128K, 64K, 128K
        streams in that order.
        The bandwidth is adjusted to force the player to switch to
        the media that accomodates the available network bandwidth.
        The test confirms the player transitioned to the appropriate
        stream by monitoring the diagnostic that relates to playing
        the audio segment.
        """
        self.zp = zp
        zp.bounce_anacapa_and_wait()
        playlist_64 = "open: http://{}/audio64".format(self.websrv.host())
        playlist_128 = "open: http://{}/audio128".format(self.websrv.host())
        self._test_bit_rate_change(zp, ((self.high_bitrate, (playlist_64,)),
                                        (self.high_bitrate, (playlist_128,)),
                                        (72000, (playlist_64,)),
                                        (self.high_bitrate, (playlist_128,)),))

    def test_high_bit_rate_errors(self, zp):
        """
        The test initially expects the player to play back the 64K media
        playlist. Due to the reduced bandwidth, the player won't have enough
        buffered audio for play back and the player will generate
        errors and eventually will terminate the playback.
        """
        self.zp = zp
        zp.bounce_anacapa_and_wait()
        playlist_64 = "open: http://{}/audio64_000002.aac \(0x0\)".format(
            self.websrv.host())
        self._test_bit_rate_change(
            zp, ((80000, (playlist_64,)),
                 # Changed expected log message due to SWPBL-51800
                 # (58000, "Transport error ERROR_BUFFERING"),))
                 (58000, INSUFFICIENT_BANDWIDTH_ERRORS),))
