"""
Homogeneous bandwidth streams (same bit rate) with missing segments
These audio segments will force the player to switch to
the next alternate playlist which resolves the missing segment
There're four media playlists, the fifth the player should ignore
since it allows four media playlists per bit rate.
The expected playback of segments by the player:
Play from first media playlist segments 0 and 1.
Segment 2 (MaggieFarm256k2.aac) is missing from first media as well
as second and third.
Resolve segment 2 in fourth media playlist.
Play segments 2,3,4,5 from fourth media playlist.
Segment 6 (MaggieFarm256kfourth6.aac) is missing from fourth media and
first media.
Resolve segment 6 in second media playlist.
Play segments 6,7,8,9 from second playlist.
Segment 10 (MaggieFarm256ksecondary10.aac) is missing from second playlist.
Resolve segment 10 in third playlist.
Play segments 10,11,12 from third playlist.
Segment 13 (MaggieFarm256kthird13.aac) is missing from third, fourth and
first playlists.
Resolve segment 13 in second playlist.
Play segments 13 until LAST_FILE_TO_END_TEST
"""

[docs]    def add_missing_file_handler(self, resource_names):
        """
        Posts a handler for web server to return NOT_FOUND
        on HTTP GET request for the resource_names
        :param list resource_names: A list of resources the web
        server will respond with NOT_FOUND
        """
        for resource_name in resource_names:
            self.websrv.remove_handler(resource_name)
            self.logger.debug(
                "Adding missing segment handler {}".format(resource_name))
            self.websrv.add_handler(
                resource_name,
                ErrorPageHandler(resource_name,
                                 error_msg=HTTPErrorTuple(
                                     NOT_FOUND,
                                     "Resource not found",
                                     "Resource is not available")))

        """
        Add handlers for all the files the player is to open.
        Note that the MasterPlaylistThread removes these files on cleanup.
        """
        self.websrv.add_handler(
            PLAYLIST, StaticFileHandler(PLAYLIST))

            MEDIA_256K_SECONDARY, StaticFileHandler(MEDIA_256K_SECONDARY))
        self.websrv.add_handler(

            self.websrv.add_handler(audio_file, StaticFileHandler(audio_file))
            audio_file = AUDIO_FMT_256K_SECONDARY.format(audio_num)
            self.websrv.add_handler(audio_file, StaticFileHandler(audio_file))
            audio_file = AUDIO_FMT_256K_THIRD.format(audio_num)
            self.websrv.add_handler(audio_file, StaticFileHandler(audio_file))
            audio_file = AUDIO_FMT_256K_FOURTH.format(audio_num)
            self.websrv.add_handler(audio_file, StaticFileHandler(audio_file))

[docs]    def watch_for_missing_segment_cb(self, msg):
        """
        Updates a list comprised of missing segments the player encountered
        """
        try:
            segment_name = self.missing_segment_regex.search(msg).group(1)
            if segment_name not in self.missing_segment_order:
                self.missing_segment_order.append(segment_name)
        except AttributeError:
            self.fail("Regex failure on missing segment '{}'".format(msg))



[docs]    def watch_for_advancing_stream_cb(self, msg):
        """
        Updates a list comprised of media playlists the player opened

        try:

            self.advancing_stream_order.append(stream_name)
        except AttributeError:
            self.fail("Regex failure on advancing stream '{}'".format(msg))



[docs]    @combinatorial('generate_testbed_unique_playback_devices')
    def test_missing_segments(self, zp):
        """
        Play HLS live stream and monitor the player is able to
        select an alternate media playlist that has the missing
        audio segment.
        1. Play HLS live stream
        2. Register UDP message call backs to monitor the player
        (a) Attempts to open the expected missing segment and
        (b) Switches to the alternate media playlist to find the
        missing segment.
        3. Confirm the player progresses through the playback
        recovering on missing segments and media playlists
        :param :class:`sonos.client.internal.SonosZoneComponent` zp:
           The zone component under test.
        """
        if zp.is_virtual():
            self.skip("Skipping on emulator as anacapa bounce not supported.")
        self.clear_hls_uri_cache(zp)
        self.master_playlist_reset()
        self.setup_player(zp)
        self.new_playlist = []
        self.missing_segment_order = []
        self.advancing_stream_order = []
        self.monitor_diagc.set_state(False)
        zp.diag.watch_udp_log_for_string(
            self.missing_segment_msg, self.watch_for_missing_segment_cb)
        zp.diag.watch_udp_log_for_string(
            ADVANCING_STREAM_MSG, self.watch_for_advancing_stream_cb)

        self.verifyTrueOrFailCase(
            self.start_playback(zp, self.hls_uri),
            "Start playback of HLS stream '{}'".format(self.hls_uri))
        zp.RenderingControl.SetVolume(5)
        # Wait for the player to open the last segment for the test to complete
        watch_str = "open: http://{}/{}".format(
            self.websrv.host(), LAST_FILE_TO_END_TEST)
        self.monitor_diagc.set_state(False)
        zp.diag.watch_udp_log_for_string(
            watch_str,
            lambda msg: self.monitor_diagc.set_state(True))
        self.verifyTrueOrFailCase(
            self.wait_on_diag(zp, watch_str, timeout_seconds=180),
            "Last segment read '{}'".format(watch_str))

        for item in self.missing_segment_order:
            self.new_playlist.append(item)
        self.verifyEqualOrFailCase(
            list(MISSING_SEGMENTS),
            list(self.new_playlist),
            "Missing segment progression should be in order")

        self.verifyEqualOrFailCase(
            list(ALTERNATE_PLAYLIST_ORDER),
            self.advancing_stream_order,
            "Alternate playlist progression should be in order")
        self.logger.info("{}".format(self.advancing_stream_order))
        self.cleanup_player(zp)
