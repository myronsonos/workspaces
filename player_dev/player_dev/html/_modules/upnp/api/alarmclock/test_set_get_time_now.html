"""
Exercise the AlarmClock.SetTimeNow() and AlarmClock.GetTimeNow() function across the supported timezone indexes, 0-74, and
verify that the returned results contain proper values.
Verify that an attempt to access an invalid "index correctly fails and that the GetTimezoneRule()
function still returns proper values.

Note: The last four digits of a time zone rule are the Daylight Savings Time Bias, a hex ffc4  - see ac.cxx RTimeZone serialize.
The CurrentTimeZone returned from GetTimeNow() contain 0000 if the SetTimeZone AutoAdjustDst is set to False and ffc4 if True.
Certain time zone rules, such as 25 and 26, are equivalent with dst set to False.

        SetTimeNow()
        Set the current date/time used by this device.
        Arguments:
        time_now -- string, desired date/time to set on device
        timezone -- string, desired time zone rule

        Exceptions:
        UPNP_ERROR_402_INVALID_ARGUMENTS

        Returns:
        boolean -- successfully invoked action?

        GetTimeNow()
        Get the current time used by this device.

        Returns:
        dictionary -- current time

            Response Keys:
            'CurrentTimeZone'       -- rule for current device time zone
            'CurrentUTCTime'        -- current UTC time for device
            'CurrentLocalTime'      -- current localtime for device
            'CurrentTimeGeneration' -- current time generation
        Exceptions:
        UPNP_ERROR_800_TIME_NOT_SET - no test in these api tests. Requires device factory reset.


"""

        '''
        Test the comparator
        '''
        compare_vectors = [
                         ("2001-12-31 10:59:59", "2001-12-31 11:00:02"),
                         ("2001-12-31 23:59:59", "2002-01-01 00:00:02"),
                         ("2000-02-28 23:59:59", "2000-02-29 00:00:02"),

                         ]
        for vector in compare_vectors:

            self.verifyTrueOrFailCase(self._compare_date_time(exp, have), "Expected should match actual. exp: %s; have: %s" % (exp, have))



    def _compare_date_time(self, dt1, dt2):
        '''
        Date should match. Hours and minutes should match, Seconds role away.
        An expected time may also have a 59 count. That may roll up others as well.
        '''
        result = False
        sdt1 = dt1.split(":")
        sdt2 = dt2.split(":")
        if sdt1[0] == sdt2[0]:
            if sdt1[1] == sdt2[1]:
                return True
        if sdt1[2] == "59":
            dt = datetime.datetime.strptime(dt1, "%Y-%m-%d %H:%M:%S")
            td = dt + datetime.timedelta(0, 3)
            result = self._compare_date_time(str(td), dt2)
            if not result:
                self.logger.debug("Result of nested datetime compare was False. td: %s; dt2: %s" % (str(td), dt2))
        return result
