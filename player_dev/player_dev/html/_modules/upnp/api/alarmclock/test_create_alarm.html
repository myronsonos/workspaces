    def test_create_alarm (self, device1, startTime, duration, recurrence,
            enabled, roomUUID, programURI, programMetaData, playMode, volume,
            hasLinkedZones, setCST, exceptNum, TCID):
        self.logger.info("%s: %s" % (TCID, testCaseMap.get(TCID)))
        device1roomUUID = device1.raw_udn
        alarmId = "0"

        self.verifyIsNotNoneOrStop(device1, "device1 should be valid")
        self.verifyIsInstanceOrStop(device1, SonosZoneComponent, "device1 should be a SonosZoneComponent")

        alarmCount = self.alarm_manager.num_alarms
        self.verifyGreaterEqualOrFailCase(alarmCount, 0, "We had %s alarms at the beginning of the test." % alarmCount)
        have_failure = False
        try:
            alarmId = device1.AlarmClock.CreateAlarm(startTime, duration, recurrence, enabled, device1roomUUID, programURI, programMetaData, playMode, volume, hasLinkedZones)
            self.logger.debug("Created Alarm Id %s" % alarmId)
        except UPnPError, e:
            if exceptNum > 0 and exceptNum == e.upnp_error:
                self.logger.info("Correctly received expected UPnP Error %s: ", e)
                have_failure = True
            else:
                self.logger.debug("Create Alarm failed with exception: %s " % e)
                self.logger.debug("Start Time: %s; Duration: %s; Recurrence: %s; Enabled: %s; Device Id: %s; URI: %s; MD: %s; PM: %s; Volume: %s; hasLinkedZones: %s" \
                               % (startTime, duration, recurrence, enabled, device1roomUUID, programURI, programMetaData, playMode, volume, hasLinkedZones))
                self.verifyIsNoneOrStop(e, "Unexpected UPnPError: %s" % e)
        if exceptNum > 0:
            self.verifyTrueOrFailCase(have_failure, "Should receive expected exception: %d" % exceptNum)
        if int(alarmId) > 0:
            newAlarmCount = alarmCount + 1
            alarmCount = self.alarm_manager.num_alarms
            self.verifyEqualOrFailCase(alarmCount, newAlarmCount, "creating one alarm should increase the number of alarms in the system by 1")
            createdAlarm = self.alarm_manager.get_alarm(alarmId)
            self.verifyIsNotNoneOrFailCase(createdAlarm, "Should be able to get alarm created by id %s" % alarmId)
            wait_until_true(lambda: device1.AlarmClock.TimeGeneration != 0)
            # Loop through recurrence days/dates to keep using the same alarm and progress the day.
            if enabled:
                days = ['O']
                rec = recurrence.split('_')
                if len(rec) > 1:
                    days = list(rec[1])
                for d in days:
                    # Once is Monday 2001-1-1
                    fdate = dates.get(d)
                    self.verifyAlarmFiredAndIsInCorrectState(device1, alarmId, fdate, startTime, setCST, recurrence != 'ONCE')

            wait_until_true(lambda: zp.AVTransport.is_alarm_running() == True)
            zp.GroupRenderingControl.SetGroupMute(True)
            alarmProperties = zp.AVTransport.GetRunningAlarmProperties()

            zp.AVTransport.Stop()
            wait_until_true(lambda: zp.AVTransport.is_stopped() == True)
            wait_until_true(lambda: zp.AVTransport.is_alarm_running() == False)
