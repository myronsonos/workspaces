'''
    Test the AlarmClock Events API.

    + Clear alarms and verify that ListAlarms shows none
    + Get Time Format, Set it, verify changed value
    + Get Date Format, Set it, verify changed value
    + Get TimeZoneAndRule, Set TimeZone, verify changed value
    + Get Time Generation, SetTimeNow, verify changed value
    + Get Daily Index Refresh Time, Set it, Verify changed value
    + Create Alarm, verify alarm list version, verify values
    + Update Alarm, verify list version, verify values
    + Destroy Alarm, verify list version shows none


        Get data for the specified alarm

        Arguments:
        alarm_id -- string, ID of alarm to get

        Returns:
        dictionary -- information about the specified alarm

            Response Keys:
            'Volume' -- volume of alarm
            'IncludeLinkedZones' -- should this alarm play on linked devices?
            'RoomUUID' -- UDN of device this alarm should run on
            'Enabled -- is this alarm enabled?
            'Recurrence' -- which days should this alarm play on?
            'ProgramURI' -- URI of content alarm should play
            'ProgramMetaData' -- XML metadata for alarm content
            'PlayMode' -- device playmode (see PLAYMODE_OPTIONS)
            'Duration' -- duration this alarm should run for
            'StartTime' -- when this alarm should start



'''

[docs]    def test_combinatorial_update_alarm_data(self, zone_generator):
        '''
        Tests ListAlarms, AlarmData, AlarmListVersion
        '''
        global updateAlarmData
        global createAlarmData

        device1 = zone_generator
        (cstartTime, cduration, crecurrence, cenabled, roomUUID, cprogramURI, cprogramMetaData, cplayMode, cvolume, chasLinkedZones) = createAlarmData
        (startTime, duration, recurrence, enabled, roomUUID, programURI, programMetaData, playMode, volume, hasLinkedZones) = updateAlarmData
        self.logger.info("%s: %s" % (TCID,testCaseMap.get(TCID)))
        device1roomUUID = device1.raw_udn

        device2 = [device for device in suite.my_devices if device != device1][0]
        device2roomUUID = device2.raw_udn

        initialAlarmCount = self.alarm_manager.num_alarms
        before_alarms = device1.AlarmClock.ListAlarms()

        initialListVersion = device1.AlarmClock.AlarmListVersion
        before_list = before_alarms.get("CurrentAlarmList")
        self.verifyEqualOrStop(initialAlarmCount, len(before_list), "CurrentAlarmList should be empty.")
        beforeListVersion = before_alarms.get("CurrentAlarmListVersion")
        self.verifyEqualOrFailCase(initialListVersion, beforeListVersion, "AlarmClock state variable AlarmListVersion should match that from ListAlarms.")

        alarmId = device1.AlarmClock.CreateAlarm(cstartTime, cduration, crecurrence, cenabled, device1roomUUID, cprogramURI, cprogramMetaData, cplayMode, cvolume, chasLinkedZones)
        self.logger.debug("Created Alarm Id %s" % alarmId)
        afterCreate = device1.AlarmClock.ListAlarms()

        afterCreateAlarmListVersion = device1.AlarmClock.AlarmListVersion
        after_list = afterCreate.get("CurrentAlarmList")
        self.verifyEqualOrStop(len(after_list), len(before_list) + 1, "AlarmList after CreateAlarm should not be empty.")
        afterListVersion = afterCreate.get("CurrentAlarmListVersion")
        self.verifyEqualOrFailCase(afterListVersion, afterCreateAlarmListVersion, "AlarmClock state variable AlarmListVersion should match that from ListAlarms.")
        blv = beforeListVersion.split(":")
        alv = afterListVersion.split(":")
        self.verifyGreaterOrFailCase(int(alv[1]), int(blv[1]), "AlarmListVersion after CreateAlarm should be greater than before.")

        result = device1.AlarmClock.UpdateAlarm(alarmId, startTime, duration, recurrence, enabled, device2roomUUID, programURI, programMetaData, playMode, volume, hasLinkedZones)
        self.verifyTrueOrFailCase(result, "Alarm % should successfully update." % alarmId)

        expectedAlarmCount = initialAlarmCount + 1
        currentAlarmCount = self.alarm_manager.num_alarms
        self.verifyEqualOrFailCase(expectedAlarmCount, currentAlarmCount, "Current alarm count should match expected alarm count.")

        result = device1.AlarmClock.DestroyAlarm(alarmId)
        self.verifyTrueOrFailCase(result, "DestroyAlarm should succeed.")
        currentAlarmCount = self.alarm_manager.num_alarms
        self.verifyEqualOrFailCase(initialAlarmCount, currentAlarmCount, "Should have no net-change in alarms after DestroyAlarm.")

        afterDestroy = device1.AlarmClock.ListAlarms()

        afterDestroyAlarmListVersion = device1.AlarmClock.AlarmListVersion
        after_destroy_list = afterDestroy.get("CurrentAlarmList")
        self.verifyEqualOrStop(0, len(after_destroy_list), "AlarmList after DestroyAlarm should be empty.")
        afterListVersion = afterDestroy.get("CurrentAlarmListVersion")
        self.verifyEqualOrFailCase(afterListVersion, afterDestroyAlarmListVersion, "AlarmClock state variable AlarmListVersion should match that from ListAlarms.")
        adv = afterListVersion.split(":")
        self.verifyGreaterOrFailCase(int(adv[1]), int(alv[1]), "AlarmListVersion after DestroyAlarm should be greater than before.")



[docs]    def test_combinatorial_set_get_time_format(self, zone_generator):
        """
        Execute the Set/Get Time Format test with the set of playable devices.
        """
        device = zone_generator
        self.logger.info("%s: %s" % (TCID,testCaseMap.get(TCID)))
        self.verifyIsNotNoneOrStop(device, "device should be valid")
        self.verifyIsInstanceOrStop(device, SonosZoneComponent, "device1 should be a SonosZoneComponent")
        dateTimeData =[
                ("YMD", "12H"),
                ("DMY", "24H")
                ]
        for dt_format in dateTimeData:
            (dateFormat,timeFormat) = dt_format
            result = device.AlarmClock.SetFormat(timeFormat, dateFormat)

            self.verifyTrueOrFailCase(result, "Set return value should be true.")

            updatedFormat = device.AlarmClock.GetFormat()
            stateVarDateFormat = device.AlarmClock.get_state_variable('DateFormat')
            stateVarTimeFormat = device.AlarmClock.get_state_variable('TimeFormat')

            expectedDateFormat = dateFormat
            expectedTimeFormat = timeFormat
            self.verifyEqualOrFailCase(expectedDateFormat, updatedFormat.get('CurrentDateFormat'), "The date format should be properly set.")
            self.verifyEqualOrFailCase(expectedTimeFormat, updatedFormat.get('CurrentTimeFormat'), "The time format should be properly set.")
            self.verifyEqualOrFailCase(expectedDateFormat, stateVarDateFormat, "The date format State Variable should be properly set.")
            self.verifyEqualOrFailCase(expectedTimeFormat, stateVarTimeFormat, "The time format State Variable should be properly set.")



[docs]    def test_combinatorial_set_get_time_zone_and_rule(self, zone_generator):
        device = zone_generator
        tzi_data = [
                    (0,device.AlarmClock.GetTimeZoneRule(0)),
                    (34,device.AlarmClock.GetTimeZoneRule(34))
                    ]
        for tz_rules in tzi_data:
            (tzi,tzrule) = tz_rules
            startTimeGen = device.AlarmClock.TimeGeneration
            device.AlarmClock.SetTimeZone(tzi, True)
            results = device.AlarmClock.GetTimeZoneAndRule()
            device.AlarmClock.SetTimeNow("2001-01-30 00:00:01", tzrule)
            wait_until_true(lambda: device.AlarmClock.TimeGeneration > startTimeGen)
            retRule = results.get('CurrentTimeZone')
            retIndex = results.get('Index')
            expectedRule = tzrule
            expectedIndex = str(tzi)
            self.verifyEqualOrFailCase(expectedIndex, retIndex, "Retrieved index should be as expected.")
            self.verifyEqualOrFailCase(expectedRule, retRule, "Retrieved rule should be as expected.")
            self.logger.debug("Timezone index %s: rule %s;  " % (retIndex, str(retRule)))
