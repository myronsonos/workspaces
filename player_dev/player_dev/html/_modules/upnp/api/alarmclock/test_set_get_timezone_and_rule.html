    def test_get_timezone_and_set_get_timezone_and_rule(self, device):
        """
        Execute the get_timezone_"index test with the set of playable devices.
        """
        TCID = "P-10875"
        self.logger.info("{}: {}".format(TCID, testCaseMap.get(TCID)))
        self.verifyIsNotNoneOrStop(device, "device should be valid")
        self.verifyIsInstanceOrStop(device, SonosZoneComponent, "device should be a SonosZoneComponent")
        tz_rule_expectedRule = 0

        for index in range(76):
            for adjust_dst in [True, False]:
                device.AlarmClock.SetTimeZone(index, adjust_dst)
                device.settings.sync_alarm_settings()
                # Calls for results
                tz_results = device.AlarmClock.GetTimeZone()
                tz_rule_results = device.AlarmClock.GetTimeZoneAndRule()

                # GetTimeZone() data retrieval
                tz_retIndex = tz_results.get('Index')
                tz_retDst = bool(int(tz_results.get('AutoAdjustDst')))
                tz_expectedIndex = str(index)
                tz_expectedDST = adjust_dst

                # GetTimeZoneAndRule() data retrieval
                tz_rule_retRule = tz_rule_results.get('CurrentTimeZone')
                tz_rule_retIndex = tz_rule_results.get('Index')
                tz_rule_retDST = bool(int(tz_rule_results.get('AutoAdjustDst')))
                tz_rule_expectedRule = tz_rules[index].get("rule")
                tz_rule_expectedIndex = str(index)
                tz_rule_expectedDST = adjust_dst

                # Verifications
                self.verifyEqualOrFailCase(tz_expectedIndex, tz_retIndex, "Retrieved index should be as expected.")
                self.verifyEqualOrFailCase(tz_expectedDST, tz_retDst, "Retrieved adjust dst should be as expected.")
                self.logger.debug("Timezone index {}: Adjust to dst: {} ".format(tz_retIndex, tz_retDst))

                self.verifyEqualOrFailCase(tz_rule_expectedIndex, tz_rule_retIndex,
                                           "Retrieved index should be as expected.")
                self.verifyEqualOrFailCase(tz_rule_expectedRule, tz_rule_retRule,
                                           "Retrieved rule should be as expected.")
                self.verifyEqualOrFailCase(tz_rule_expectedDST, tz_rule_retDST,
                                           "Retrieved adjust dst should be as expected.")
                self.logger.debug("Timezone index {}: rule {}; Adjust to dst: {} ".format(tz_rule_retIndex, str(tz_rule_retRule), tz_rule_retDST))

    def test_set_get_timezone_and_rule_negative (self, device, adjust_dst,
            index):
        """
        Execute the get_timezone_"index test with the set of playable devices.
        """
        TCID = "P-10875"
        self.logger.info("%s: %s" % (TCID, testCaseMap.get(TCID)))
        self.verifyIsNotNoneOrStop(device, "device should be valid")
        self.verifyIsInstanceOrStop(device, SonosZoneComponent, "device should be a SonosZoneComponent")
        have_expected_exception = False
        previous_values = device.AlarmClock.GetTimeZoneAndRule()
        try:
            device.AlarmClock.SetTimeZone(index, adjust_dst)
        except UPnPError, e:
            if index == -1 and e.upnp_error == sonos.client.zone_player.UPNP_ERROR_402_INVALID_ARGUMENTS:
                have_expected_exception = True
                self.logger.info("Correctly received expected UPnP Error %s: ", e)
            else:
                self.verifyTrueOrFailCase(have_expected_exception, "Received unexpected_exception: %s" % e)
        device.settings.sync_alarm_settings()
        results = device.AlarmClock.GetTimeZoneAndRule()

        prevRule = previous_values.get('CurrentTimeZone')
        prevIndex = previous_values.get('Index')
        prevDst = bool(int(previous_values.get('AutoAdjustDst')))

        retRule = results.get('CurrentTimeZone')
        retIndex = results.get('Index')
        retDst = bool(int(results.get('AutoAdjustDst')))

        self.verifyEqualOrFailCase(prevIndex, retIndex, "Retrieved index should be as expected.")
        self.verifyEqualOrFailCase(prevRule, retRule, "Retrieved rule should be as expected.")
        self.verifyEqualOrFailCase(prevDst, retDst, "Retrieved adjust dst flag should be as expected.")
