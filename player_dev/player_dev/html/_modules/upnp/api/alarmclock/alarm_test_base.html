    def _setup_virtual_player(self, zone):
        ''' Setup virtual players: Registration and system properties.
            Arguments : Sonos Zone Component (emulator)
            Return : None
        '''
        zone.SystemProperties.register_device_to_test_user()
        zone.SystemProperties.SetString("R_UpdateInterval","5")
        self.zp.settings.sync_alarm_settings()
        zone.AlarmClock.SetFormat('12H', 'YMD')
        zone.AlarmClock.SetDailyIndexRefreshTime('00:00:00')
        zone.AlarmClock.SetTimeZone(TIMEZONE, False)

        for t in range(num_tracks):
            zone.AVTransport.add_to_queue(BUZZER)
        return zone.ContentDirectory.get_num_tracks_in_queue() == num_tracks



[docs]    def reset_zone_time_to_2001(self, zp):
        self.logger.info("resetting time on %s to Monday, 2001-01-01" % zp.serialNumber)
        zp.AlarmClock.SetTimeNow("2001-01-01 00:00:00", TIMEZONE)
        current_time = zp.AlarmClock.GetTimeNow()
        self.verifyTrueOrStop("2001-01-01" in current_time.get('CurrentUTCTime'), "failed to reset time to 2001-01-01 on %s" % zp.serialNumber)
