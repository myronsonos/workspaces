        username -- string, username for service
        password -- string, password for service

        Returns:
        boolean -- was invocation successful?

        Exceptions:
        UPNP_ERROR_402_INVALID_ARGUMENTS

        UPNP_ERROR_606_UNAUTHORIZED
        UPNP_ERROR_800_CANT_OVERWRITE
        UPNP_ERROR_802_NOT_ENOUGH_SPACE
        
    RemoveAccount
    
        Arguments:
        account_id -- integer, code for this service account
        username -- string, username used with this service account

        Returns:
        boolean -- was invocation successful?

"""

[docs]    def test_combinatorial_add_account_x(self, add_account_x_vector_generator):
        """
        Execute the Add Account X test with the set of devices.
        """
            
        (device, (internalAccountID, username, password, exceptNum)) = add_account_x_vector_generator
        self.logger.info("%s: %s" % (TCID,testCaseMap.get(TCID)))
        self.verifyIsNotNoneOrStop(device, "device should be valid")
        self.verifyIsInstanceOrStop(device, SonosZoneComponent, "device should be a SonosZoneComponent")
        
        haveException = False
        
        try:
            device.SystemProperties.AddAccountX(internalAccountID, username, password)
        except UPnPError as e:
            if exceptNum > 0 and exceptNum == e.upnp_error:
                self.logger.info("Correctly received expected Invalid Argument Error %s: ", e)
                haveException = True
            if e.upnp_error == 802 and ( device.is_bridge() or device.is_dock() ):
                self.logger.info("Correctly received expected Invalid Argument Error %s: ", e)
                haveException = True
            else:
                self.logger.debug("AddAccountX failed with exception: %s " % e)
                self.verifyIsNoneOrStop(e, "Unexpected UPnPError: %s" % e)
        
        
        if exceptNum > 0:
            self.verifyTrueOrStop(haveException, "Should receive exception %s" % exceptNum)
            
        wait_until_true( lambda: \
                         device.SystemProperties.does_service_account_exist(internalAccountID),
                         timeout_seconds=120,
                         iteration_delay=1)

        
        result = device.SystemProperties.does_service_account_exist(internalAccountID)
        self.verifyTrueOrStop(result, 'Failed to add test service account')
        
        device.SystemProperties.RemoveAccount(internalAccountID, 
                                                       username)
        wait_until_true(lambda: not device.does_service_account_exist(internalAccountID),
                        iteration_delay=1)

               

[docs]    def test_combinatorial_remove_account(self, remove_account_vector_generator):
        """
        Execute the Remove Account test with the set of devices.
        """
               
        (device, (internalAccountID, username, password, exceptNum)) = remove_account_vector_generator
        self.logger.info("%s: %s" % (TCID,testCaseMap.get(TCID)))
        self.verifyIsNotNoneOrStop(device, "device should be valid")
        self.verifyIsInstanceOrStop(device, SonosZoneComponent, "device should be a SonosZoneComponent")
        
        if device.is_bridge() or device.is_dock():
            self.skip("Skipping Bridge or Dock because AddAccountX produces UPnP 802 from either device")
        
        haveException = False
        
        if exceptNum == 0:
            try:
                device.SystemProperties.AddAccountX(internalAccountID, 
                                                    username, 
                                                    password)
            except UPnPError as e:
                self.logger.debug("AddAccountX failed with exception: %s " % e)
                self.verifyIsNoneOrStop(e, "Unexpected UPnPError: %s" % e)
            
            wait_until_true( lambda: device.SystemProperties.does_service_account_exist(internalAccountID),
                                 timeout_seconds=120, 
                                 iteration_delay=1)
            
        try:
            device.SystemProperties.RemoveAccount(internalAccountID, username)
            wait_until_true(lambda: not device.does_service_account_exist(internalAccountID),
                            iteration_delay=1)
        except UPnPError as e:
            if exceptNum > 0 and exceptNum == e.upnp_error:
                self.logger.info("Correctly received expected Invalid Argument Error %s: ", e)
                haveException = True
            else:
                self.logger.debug("RemoveAccount failed with exception: %s " % e)
                self.verifyIsNoneOrStop(e, "Unexpected UPnPError: %s" % e)
        
        self.logger.info("get_all_service_credentials() = %s", device.SystemProperties.get_all_service_credentials())
        
        if exceptNum > 0:
            self.verifyTrueOrStop(haveException, "Should receive exception %s" % exceptNum)
                   
        result = device.SystemProperties.does_service_account_exist(internalAccountID)
        self.verifyFalseOrStop(result, 'Failed to remove test service account')
