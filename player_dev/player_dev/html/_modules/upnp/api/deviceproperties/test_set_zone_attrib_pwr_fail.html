    """
    Tests updates to localsettings.txt file.
    TBF!!! Not part of automation since it requires that power
    is cut-off while this script is running.
    Following power restoration, confirm the localsettings.txt file
    has the zone name, icon and configuration intact.

    ZONE_ATTR_VECTORS = [
              (u'PowerOffZone', u'x-rincon-roomicon:woodshed', u'0'),
              (u'PowerOnZone', u'x-rincon-roomicon:attic', u'1'),
              (u'PowerZone', u'x-rincon-roomicon:basement', u'0'),
              (u'Play5White', u'x-rincon-roomicon:garage', u'1')]

    def _get_zone_attrib_state_variables(self, zp):
        name = zp.DeviceProperties.get_state_variable("ZoneName")
        icon = zp.DeviceProperties.get_state_variable("Icon")
        cfg = zp.DeviceProperties.get_state_variable("Configuration")
        return (name, icon, cfg)
