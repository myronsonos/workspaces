"""
    The DeviceProperties service has twenty-five state variables: 
         AutoplayIncludeLinkedZones,AutoplayRoomUUID, AutoplayUseVolume,
         AutoplayVolume, ChannelMapSet, Configuration, CopyrightInfo, 
         DisplaySoftwareVersion,ExtraInfo,HardwareVersion,HouseholdID,
         HTAudioIn,HTFreq, HTSatChanMapSet,Icon,IPAddress,
         IsZoneBridge,LEDState,MACAddress,SatRoomUUID,SerialNumber,
         SettingsReplicationState,SoftwareVersion, ZoneName

         SettingsReplicationState,, 
    
    Notes: 
        1.  SatRoomUUID is listed in DeviceSpy as a state variable. However,
            it is not readable via DeviceProperties.get_state_variable('SatRoomUUID')
            and times out. A grep of both python and C++ code only shows it as an
            argument to the RemoveHTSatellite() function call. Therefore this test
            does nothing with SatRoomUUID.
        
        2.  HTFreq is only readable via a Playbar. It always returns 0. I borrowed
            a piece of script from channel_change_2ghz.py and can see the channel
            change to the specified channel/frequency. I can see what appears to be
            correct changes in dmesg. However, HTFreq does not change on the S9 and 
            is not readable on the S1.
            e.g.
            (166261)sat: enable=1
            (166262)sat: saw primary on 2412 (htchan=5180)
            (166262)sat: freq=5180 (was 2412) 1
                    
    Test Cases:
    - Set/Get LEDState and compare to State variable LEDState
    - Set/Get HouseholdID and compare to State variable HouseholdID
    - Set/Get ZoneAttributes and compare to State variables ZoneName, 
      Configuration,and Icon.
    - Set/Get AutoplayUseVolume and compare to State variable AutoplayUseVolume
    - Set/Get AutoplayVolume and compare to State variable AutoplayVolume
    - Set/Get AutoplayLinkedZones and compare to State variable AutoplayLinkedZones
    - Set/Get AutoplayRoomUUID and compare to State variable to AutoplayRoomUUID
    - Add/Remove BondedZones and compare to State variable ChannelMapSet
    - Create/Separate StereoPair and compare to State variable ChannelMapSet
    - Add/Remove HTSatellite and compare to State variable HTSatChanMapSet
    - GetZoneInfo and compare to State variables CopyrightInfo,DisplaySoftwareVersion,
      ExtraInfo,HardwareVersion,HTAudioIn,IPAddress,MACAddress,SerialNumber,SoftwareVersion
    - Get all devices and compare zp.is_bridge() result to State variable IsZoneBridge
    - Get initial SettingsReplicationState variable. Change a system settings variable.
      Wait for settings to match. Compare Initial/Updated/Final result to State variable
      SettingsReplicationState.
"""

    def test_set_get_zone_attributes(self, device):
        """
        Execute the Set/Get ZoneAttributes and verify that the expected
        state variables update accordingly

        :param device: Unique testbed zone player.
        :type device: :class:`~sonos.client.internal.SonosZoneComponent`

                                self._get_zone_attrib_state_variables(device)

                                        "to be updated.")
            self.verifyEqualOrFailCase(updated_attribs['CurrentIcon'],
                                        updated_icon,
                                        "Expect state variable 'ZoneIcon' "
                                        "to be updated.")
            self.verifyEqualOrFailCase(device.diag.get_status_property('Configuration'),
                                       updated_cfg,
                                       "Expect state variable 'Configuration' "
                                       "to be updated.")

        """
        Execute the Add Remove Bonded Zones test with the set of playable devices.

                                  "Expect UDN of right channel "

        self.verifyTrueOrFailCase(sub.raw_udn in chmap,
                                  "Expect UDN of Sub in channelmap of "
                                  "bonded configuration")
        s1_left.DeviceProperties.cleanup_bonded_configuration()

    def _get_device_zone_info_state_variables(self, zp):
        """
        Get device info state variables.

        :param zp: zone player under test.
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`
        :return: tuple containing state variable value strings
        :rtype: :func:`tuple`

        """
        sn = zp.DeviceProperties.get_state_variable("SerialNumber")
        ver = zp.DeviceProperties.get_state_variable("SoftwareVersion")
        display_ver = zp.DeviceProperties.get_state_variable("DisplaySoftwareVersion")
        hw_ver = zp.DeviceProperties.get_state_variable("HardwareVersion")
        ip = zp.DeviceProperties.get_state_variable("IPAddress")
        mac = zp.DeviceProperties.get_state_variable("MACAddress")
        extra_info = zp.DeviceProperties.get_state_variable("ExtraInfo")
        htaudio = zp.DeviceProperties.get_state_variable("HTAudioIn")
        return (sn, ver, display_ver, hw_ver, ip, mac, extra_info, htaudio)

    def _set_wifi_mode_and_verify(self, zp, wifi_enable=True):
        """
        Set wifi mode via http and verify state variable WifiEnabled value

        :param zp: zone player.
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`
        :param bool wifi_enable: wifi enable or disable
        
        """
        if wifi_enable:
            # Turn on wifi. This causes anacapa restart.
            zp.wifi_enable()
            wait_until_true(lambda:zp.DeviceProperties.get_state_variable(
                                'WifiEnabled') == ENABLED, iteration_delay=1)
        else:
            # Turn off wifi (on devices which allow it).
            # This causes anacapa restart.
            if zp.modelNumber in SUPPORTS_WIRELESS_DISABLE:
                zp.wifi_disable()
                wait_until_true(lambda:zp.DeviceProperties.get_state_variable(
                    'WifiEnabled') == DISABLED, iteration_delay=1)

    def test_wifi_enable_disable_state(self, zp):
        """
        Verify WifiEnabled state variable is evented when enabling or disabling
        player wifi interface.
                
        :param zp: Unique testbed playback zone player.
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`
        
        """
        if zp.is_virtual():
            self.skip("This test is skipped for virtual players.")
        try:
            wifienabled_state = zp.DeviceProperties.get_state_variable("WifiEnabled")
        except InvalidArgumentsError as e:
                self.logger.warn("Unexpected exception for Rush and later release.")
                self.stop(e)
        if wifienabled_state == ENABLED:
            self._set_wifi_mode_and_verify(zp, wifi_enable=False)
            self._set_wifi_mode_and_verify(zp, wifi_enable=True)
        else:
            self._set_wifi_mode_and_verify(zp, wifi_enable=True)
