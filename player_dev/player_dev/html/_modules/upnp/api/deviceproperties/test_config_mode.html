'''
    Description of DeviceProperties action methods under current test

    EnterConfigMode(self, mode, options):
        """
        Enter Configuration Mode for Sonar, Set AVTransportURI
        to sonar calibration file. Also used to enter button config mode. Button
        config mode is used by controller to create stereo pair.

        :param str mode: Sonar and button config (stereo pairing) mode. Values
            'sonar-calibrate' for sonar and 'button-notify' for button config
        :param str options: Which version and sonar file to use for
            config mode?. For button config mode, this is empty string ""
        :return: Was invoke successful?
        :rtype: :obj:`bool`
        """
        args = [{'name':'Mode', 'value':mode, 'type':'remote'},
                {'name':'Options', 'value':options, 'type':'remote'}]
        self._invoke("EnterConfigMode", args)
        return self.zp.last_upnp_error_code == sonos.client.zone_player.UPNP_OK

    ExitConfigMode(self, options):
        """
        Exit Configuration Mode for Sonar. Set AVTransportURI
        to sonar complete calibration file. Also used to exit button config mode.

        :param int options: Was configuration mode successful?
            (success=1)
        :return: Was invoke Successful?
        :rtype: :obj:`bool`
        """
        args = [{'name':'Options', 'value':options, 'type':'remote'}]
        self._invoke("ExitConfigMode", args)
        return self.zp.last_upnp_error_code == sonos.client.zone_player.UPNP_OK

    GetButtonState(self):
        """
        Get button press stats when player is in button config, 'button-notify'
        mode. Format for results returned is as below:
        <Button #1>:<# of times button #1 pressed>:<time in seconds since button
        #1 was last pressed><space><Similar for format for next button>
        Volume+ = BUTTON1, Volume- = BUTTON2, Playpause/Join = BUTTON0
        Sample output: u'BUTTON1:5:9 BUTTON2:1:37 BUTTON0:2:9'
        When player is not in button config mode, UPnP error is raised.

        :raises UPNP_ERROR_401_INVALID_ACTION:
        :return: Button press stats
        :rtype: :obj:`str`
        """
        button_press_stats = self._invoke('GetButtonState').get('State').strip()
        return button_press_stats

    def test_get_button_state(self, zp):
        """
        Test button press states when in button config mode. The states are
        collected for 1 button action of all types with 2 sec interval.
        """
        #skip this test for emulators.
        if zp.is_virtual():
            self.skip("Skip button config mode test on emulators")
        # For encore since button meaning can change with respect to its
        # orientation, set the orientation to horizontal.
        if zp.is_encore():
            zp.cli.set_encore_orientation('horizontal')
            wait_until_true(lambda:zp.cli.get_encore_orientation() == 'horizontal',
                            iteration_delay=2)
        zp.DeviceProperties.EnterConfigMode(BUTTON_CONFIG_MODE, OPTION)
        zp.cli.volume_up()
        time.sleep(BUTTON_ACTION_INTERVAL)
        zp.cli.volume_down()
        time.sleep(BUTTON_ACTION_INTERVAL)
        zp.cli.volume_mute()
        time.sleep(BUTTON_ACTION_INTERVAL)
        zp.cli.button_press()
        time.sleep(BUTTON_ACTION_INTERVAL)
        button_state = zp.DeviceProperties.GetButtonState()
        self.logger.debug("Button state return from <{}> : {}"
                          .format(zp, button_state))
        if not button_state:
            self.fail("Expect button press data from <{}>".format(zp))
        states = zp.DeviceProperties.GetButtonState().split()
        # sample of button states returned
        # 'BUTTON1:2:3 BUTTON2:1:6 BUTTON0:2:3'
        for st in states:
            button, count, deltatime = tuple(part for part in st.split(':'))
            if zp.modelNumber in sonos.client.zone_player.CAP_TOUCH_DEVICES:
                expected_data = CAP_TOUCH_PLAYER_EXPECTED_STATS
            else:
                expected_data = BUTTON_PLAYER_EXPECTED_STATS
            self.verifyEqualOrFailCase(expected_data[button][COUNT], int(count),
                        "Expected count for button press on <{}>".format(zp))
            # for time delta, allow 2 seconds for some timing differences
            self.verifyTrueOrFailCase(expected_data[button][DELTA]
                                      <= int(deltatime) <=
                                      expected_data[button][DELTA] + 2,
                    "Expected delta time for button press on <{}>".format(zp))

    def test_no_button_action_when_in_button_config_mode(self, zp, action):
        """
        Verify when in button config mode, button press actions does not cause
        for example volume to increase or decrease on the player, or play/pause
        """
        #skip this test for emulators.
        if zp.is_virtual():
            self.skip("Skip button config mode test on emulators")
        initial_volume = 2
        zp.RenderingControl.SetVolume(initial_volume)
        zp.AVTransport.use_buzzer()
        zp.AVTransport.Play_and_wait()
        zp.DeviceProperties.EnterConfigMode(BUTTON_CONFIG_MODE, OPTION)
        getattr(zp.cli, action)()
        self.verifyTrueOrFailCase(all([initial_volume == zp.RenderingControl.GetVolume(),
                                       zp.AVTransport.is_playing()]),
                                      "Expect no effect from button action {} "
                                      "on <{}>".format(action, zp))
        zp.DeviceProperties.ExitConfigMode(BUTTON_CONFIG_MODE)
        zp.AVTransport.clear_play_state()

    def test_button_config_mode_timeout(self, zp):
        """
        Verify button mode fail-safe timeout of 5 minutes. For verification,
        using a testpoint which, when hit decrements the remaining time by 1
        minute
        """
        #skip this test for emulators.
        if zp.is_virtual():
            self.skip("Skip button config mode test on emulators")
        if not zp.diag.has_testpoint_support():
            self.skip("Test requires testpoint support, skipped.")

        zp.DeviceProperties.EnterConfigMode(BUTTON_CONFIG_MODE, OPTION)
        for i in range(5):
            zp.http.get_url(BUTTON_MODE_TIMEOUT_TEST_POINT.format(zp.ip))
        try:
            zp.DeviceProperties.GetButtonState()
        except UPnPError as e:
            if not (zp.last_upnp_error_code ==
                    sonos.client.zone_player.UPNP_ERROR_401_INVALID_ACTION):
                self.fail("Did not expect {} on <{}>".format(e, zp))
        else:
            self.fail("Expect {} not in config mode".format(zp))

    def test_reset_on_reentering_config_mode(self, zp):
        """
        Verify when reentering button config mode, previous states is reset
        """
        #skip this test for emulators.
        if zp.is_virtual():
            self.skip("Skip button config mode test on emulators")
        zp.DeviceProperties.EnterConfigMode(BUTTON_CONFIG_MODE, OPTION)
        #A no-op for a playback device with no volume control, like S23
        zp.cli.volume_up()
        time.sleep(BUTTON_ACTION_INTERVAL)
        zp.cli.button_press()
        time.sleep(BUTTON_ACTION_INTERVAL)
        states = zp.DeviceProperties.GetButtonState()
        self.verifyFalseOrStop(states == "", "Expect non empty button state"
                               " value on <{}>".format(zp))
        zp.DeviceProperties.EnterConfigMode(BUTTON_CONFIG_MODE, OPTION)
        self.verifyEqualOrFailCase("", zp.DeviceProperties.GetButtonState(),
                                    "Expect button states to reset")
