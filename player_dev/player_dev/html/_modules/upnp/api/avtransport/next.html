"""
 UPnP Test:  AVTransport.Next

 Positive test cases and some negative test cases (error codes 701, 711)
 for local library tracks and Pandora station.

 The JSON execution configuration must include music_accounts and
 music_shares entries, and music_shares must list a share called automation.

 Currently the track URI is hardcoded, but this will change soon.

"""

LIBRARY_TRACK_URI = ("m4a")

        for zp in self.playback_devices:

    def _generate_group_member(self):
        for gm in self.playback_devices:
            gc = [zp for zp in self.playback_devices if zp != gm][0]
            gm.AVTransport.join_group(gc)
            yield gm

    # Helper generator to pick a single random device
    def _generate_single(self):
        yield random.choice(self.playback_devices)

    # Generator for group coordinator devices capable of playback
    def _zone_generator(self):
        for zp in self.playback_devices:
            # skip if it's not a group coordinator
            if zp.AVTransport.is_group_member():
                continue
            yield zp

[docs]    def setupTracksInQueue(self, zone, trackURI, repeatCount):

        use_queue_and_empty(zone)

        queue_count = repeatCount

        tracksAndMD = []
        while(repeatCount > 0):
            tracksAndMD.append((trackURI, "<none/>"))
            repeatCount = repeatCount - 1

        #ret = setup_tracks_in_queue(zone, tracksAndMD)
        [zone.AVTransport.add_to_queue(*track_data) for track_data in tracksAndMD]

        wait_until_true(lambda: zone.AVTransport.get_number_of_tracks_in_queue() == queue_count,
                        timeout_seconds=10, iteration_delay=1)

        self.verifyEqualOrFailCase(zone.AVTransport.get_queue_position(), 1,
                                   "Current Track should be 1")

[docs]    @combinatorial('_zone_generator')
    def test_next_track_library_playing(self, zone):
        """
        Test: Next local library track in Queue while playing
        Devices: playback-capable && group coordinator
        Expected: success
        """
        self.setupTracksInQueue(zone, self.libraryTrackURI, 2)
        zone.AVTransport.use_queue()
        zone.AVTransport.Play_and_wait()

        with IgnoreInvokeErrors(zone):
            zone.AVTransport.Next()

        self.verifyEqualOrFailCase(zone.last_upnp_error_code, 0,
                                   "NextTrack should receive 0 response")

        try:
            zone.AVTransport.wait_for_state_variables('CurrentTrack', lambda ct: ct == 2,
                                                      timeout_seconds=2.0)
        except:
            pass
        self.verifyEqualOrFailCase(zone.AVTransport.get_queue_position(), 2,
                               "Current Track should be 2")

        zone.AVTransport.Stop_and_wait()
