"""
    ConfigureSleepTimer

        Configure the device sleep timer to run for this duration

        defined in oc/zone/zoneplayer/avt_impl.cxx

        Arguments:
        duration -- string, duration sleep timer should run for (ex: "00:20:25", run for 20 minutes, 25 seconds)

        Keyword Arguments:
        instance_id -- integer, default == 0, unused argument

        Returns:
        boolean

        Exceptions:
        UPNP_ERROR_402_INVALID_ARGUMENTS
        UPNP_ERROR_718_INVALID_INSTANCE_ID

    GetRemainingSleepTimerDuration

        Get the remaining duration for a sleep timer on this device.

        defined in oc/zone/zoneplayer/avt_impl.cxx

        Keyword Arguments:
        instance_id -- integer, default == 0, unused argument

        Returns:
        dictionary with GetRemainingSleepTimerDuration response.

        Exceptions:
        UPNP_ERROR_718_INVALID_INSTANCE_ID
        UPNP_ERROR_800_NOT_GROUP_COORDINATOR

"""

[docs]    def test_combinatorial_configure_sleep_timer_get_remaining_sleep_timer_duration(self, vector_generator):
        """
        Execute the ConfigureSleepTimer/GetRemainingSleepTimerDuration test
        """

        (test_zp, (instance_id, duration, exceptNum)) = vector_generator

        self.logger.info("%s : %s" % (TCID, testCaseMap.get(TCID)))
        self.verifyIsNotNoneOrStop(test_zp, "device should be valid")
        self.verifyIsInstanceOrStop(test_zp, SonosZoneComponent, "device should be a SonosZoneComponet")

        haveException = False
        test_zp.AVTransport.ConfigureSleepTimer('00:00:00', 0)
        try:
            initialSleepTimerDict = test_zp.AVTransport.GetRemainingSleepTimerDuration(instance_id)
            self.verifyTrueOrStop(isinstance(initialSleepTimerDict, dict), "GetRemainingSleepTimerDuration return value should be a Dict")

        except UPnPError as e:
            if exceptNum > 0 and exceptNum == e.upnp_error:
                self.logger.info("Correctly received expected Invalid Argument Error %s: ", e)
                haveException = True
            else:
                self.logger.debug("GetRemainingSleepTimerDuration failed with exception: %s " % e)
                self.fail("Unexpected UPnPError: %s" % e)

        # Verify exception 718 when calling GetRemainingSleepTimerDuration
        if exceptNum > 0 and exceptNum != sonos.client.zone_player.UPNP_ERROR_402_INVALID_ARGUMENTS:
            self.verifyTrueOrStop(haveException, "Should receive exception %s" % exceptNum)

        try:
            result = test_zp.AVTransport.ConfigureSleepTimer(duration, instance_id)
            self.verifyTrueOrFailCase(result, "ConfigureSleepTimer return value should be true.")
        except UPnPError as e:
            if exceptNum > 0 and exceptNum == e.upnp_error:
                self.logger.info("Correctly received expected Invalid Argument Error %s: ", e)
                haveException = True
            else:
                self.logger.debug("ConfigureSleepTimer failed with exception: %s " % e)
                self.verifyIsNoneOrStop(e, "Unexpected UPnPError: %s" % e)
        # Don't need to test ConfigureSleepTimer if GetRemainingSleepTimerDuration was tested for an exception
        if exceptNum != sonos.client.zone_player.UPNP_ERROR_718_INVALID_INSTANCE_ID:
            wait_until_true(lambda: self._getRemainingSleepTimerDuration(test_zp, instance_id) < duration)
            updatedSleepTimerDict = test_zp.AVTransport.GetRemainingSleepTimerDuration(instance_id)
            if exceptNum > 0 and haveException:
                self.verifyTrueOrStop(haveException, "Should receive exception %s" % exceptNum)
                self.verifyEqualOrFailCase(initialSleepTimerDict, updatedSleepTimerDict, "On exception the SleepTimerDict should not change.")
            else:
                self.verifyNotEqualOrFailCase(0, updatedSleepTimerDict['CurrentSleepTimerGeneration'], "Set CurrentSleepTimerGeneration should not be zero.")
                self.verifyLessOrFailCase(updatedSleepTimerDict['RemainingSleepTimerDuration'], duration, "RemainingSleepTimerDuration should be less than %s" % duration)
                test_zp.AVTransport.ConfigureSleepTimer('00:00:00', 0)


    def _getRemainingSleepTimerDuration(self, zp, instance_id):
        sleepTimerDict = zp.AVTransport.GetRemainingSleepTimerDuration(instance_id)
        return sleepTimerDict['RemainingSleepTimerDuration']

[docs]    def test_combinatorial_get_remaining_sleep_timer_duration_not_gc(self, device_generator):
        """
        Execute the GetRemainingSleepTimerDuration test
        """

        (test_zp) = device_generator
        exceptNum = sonos.client.zone_player.UPNP_ERROR_800_NOT_GROUP_COORDINATOR

        self._makeZPNotGC(test_zp)

        self.logger.info("%s : %s" % (TCID, testCaseMap.get(TCID)))
        self.verifyIsNotNoneOrStop(test_zp, "device should be valid")
        self.verifyIsInstanceOrStop(test_zp, SonosZoneComponent, "device should be a SonosZoneComponet")

        # Make test_zp not the group coordinator
        self._makeZPNotGC(test_zp)

        haveException = False

        try:
            res = test_zp.AVTransport.GetRemainingSleepTimerDuration()
            self.verifyTrueOrStop(isinstance(res, dict), "GetRemainingSleepTimerDuration return value should be a Dict")

        except UPnPError as e:
            if exceptNum > 0 and exceptNum == e.upnp_error:
                self.logger.info("Correctly received expected Invalid Argument Error %s: ", e)
                haveException = True
            else:
                self.logger.debug("GetRemainingSleepTimerDuration failed with exception: %s " % e)
                self.fail("Unexpected UPnPError: %s" % e)

        if exceptNum > 0:
            self.verifyTrueOrStop(haveException, "Should receive exception %s" % exceptNum)



    def _makeZPNotGC(self, test_player):
        if test_player.is_virtual():
            self.logger.info("Virtual devices: The test does not work with Sub. Investigate")
            candidatesforGC = [zp for zp in self.test_zones if zp != test_player and zp.modelNumber != 'Sub']
            group_coordinator =  candidatesforGC[0]
            # Make test_player part of group and not GC
            group_members = [zp for zp in self.test_zones if zp != group_coordinator and zp.modelNumber != 'Sub']
        else:
            candidatesforGC = [zp for zp in self.test_zones if zp != test_player]
            group_coordinator =  candidatesforGC[0]
            # Make test_player part of group and not GC
            group_members = [zp for zp in self.test_zones if zp != group_coordinator]

        self.logger.info("linking %d devices to group coordinator %s" % (len(group_members), group_coordinator))
        group_coordinator.AVTransport.SetAVTransportURI('', '')
        group_coordinator.AVTransport.link_zones(group_members)


[docs]    def test_sleep_timer(self):
        """
        Verify sleep timer pauses playback

        """
        self.short_description = "Test sleep timer on devices"
        for zp in self.test_zones:
            self.logger.info("starting sleep timer test for %s" % zp.serialNumber)
            self.helpers.playback.play_buzzer(zp)
            zp.AVTransport.ConfigureSleepTimer("00:00:15")
            wait_until_true(lambda: self.helpers.playback.is_paused(zp) == True, iteration_delay=1)
            self.alarm_manager.delete_all_alarms()
