[docs]def zone_generator_noncoordinator():

    gc = None
    for zp in suite.my_devices:

[docs]def playlist_files(files):
    def generate_m3u(request):
        host = request.getHost()
        urls = map(lambda x: "http://%s:%s/%s" % (host.host, host.port, x), files)
        return '\n'.join(urls)
    return generate_m3u

[docs]    @combinatorial(playable_zone_generator)
    def test_set_track_next_track_playlist(self, zone):
        """
        Test: with track, set Next AVT URI to a track, then playlist
        Expected: success
        """
        # ## SetAVT(empty) --> success
        zone.AVTransport.SetAVTransportURI_and_wait("", "")
        zone.AVTransport.SetPlayMode(zone.AVTransport.PLAYMODE_NORMAL)

        # ## SetAVT(track) --> success
        zone.AVTransport.SetAVTransportURI_and_wait(self.http_track, "")
        zone.AVTransport.wait_for_state_variables(AVT_STATE_VARS,
                                                  avt_state(self.http_track, 1, 1, ""))

        # ## SetNext(track) --> success (next is now track)
        zone.AVTransport.SetNextAVTransportURI_and_wait(self.http_track, "")
        zone.AVTransport.wait_for_state_variables(AVT_STATE_VARS,
                                                  avt_state(self.http_track, 1, 1, self.http_track))

        # ## SetNext(playlist) --> success (next is now playlist)
        zone.AVTransport.SetNextAVTransportURI_and_wait(self.http_playlist, "")
        zone.AVTransport.wait_for_state_variables(AVT_STATE_VARS,
                                                  avt_state(self.http_track, 1, 1, self.http_playlist))

        # ## Next() --> success (now playing first track of playlist; update track count)
        zone.AVTransport.Next()
        zone.AVTransport.wait_for_state_variables(AVT_STATE_VARS,
                                                  avt_state(self.http_playlist, self.http_playlist_count, 1, ""))

        # ## SetNext(track) --> success (playing playlist; next is track; no track count change)
        zone.AVTransport.SetNextAVTransportURI_and_wait(self.http_track, "")

        track_index = 1
        while track_index <= self.http_playlist_count:
            zone.AVTransport.wait_for_state_variables(AVT_STATE_VARS,
                                                      avt_state(self.http_playlist, self.http_playlist_count,
                                                                track_index, self.http_track))

            # ## Next() --> success (move to next track in playlist)
            zone.AVTransport.Next()
            track_index += 1

        # ## Should now be done with playlist and onto the next == track
        zone.AVTransport.wait_for_state_variables(AVT_STATE_VARS,
                                                  avt_state(self.http_track, 1, 1, ""))

        # ## SetNext(track) --> success (next is track; otherwise state unchanged)
        zone.AVTransport.SetNextAVTransportURI_and_wait(self.http_track, "")

    def test_shuffle_http_uri(self, zone):
        """
        Test: with track, try to set shuffle/repeat
        Expected: failure
        """
        # ## SetAVT(http) --> success
        zone.AVTransport.SetAVTransportURI_and_wait(self.http_track, "")

        # ## SetShuffle() --> success
        zone.AVTransport.SetPlayMode(zone.AVTransport.PLAYMODE_SHUFFLE)
        wait_until_true(lambda: zone.AVTransport.get_current_play_mode() == zone.AVTransport.PLAYMODE_SHUFFLE,
                        timeout_seconds=5)

        # ## SetAVT(queue) --> success
        zone.AVTransport.use_queue()

        # ## SetShuffle() --> success
        zone.AVTransport.SetPlayMode(zone.AVTransport.PLAYMODE_SHUFFLE_NO_REPEAT)
        wait_until_true(lambda: zone.AVTransport.get_current_play_mode() == zone.AVTransport.PLAYMODE_SHUFFLE_NO_REPEAT,
                        timeout_seconds=5)

        # ## SetAVT(http) --> success
        # This will change the PlayMode back to SHUFFLE since the AVT/Queue tracks it separately
        zone.AVTransport.SetAVTransportURI_and_wait(self.http_track, "")

        # ## VerifyShuffle()
        wait_until_true(lambda: zone.AVTransport.get_current_play_mode() == zone.AVTransport.PLAYMODE_SHUFFLE,
                        timeout_seconds=5)

        # ## SetNext(playlist) --> success
        # SetNextAVT will implicitly change the PlayMode to NORMAL
        zone.AVTransport.SetNextAVTransportURI_and_wait(self.http_playlist, "")
        zone.AVTransport.wait_for_state_variables(AVT_STATE_VARS,
                                                  avt_state(self.http_track, 1, 1, self.http_playlist))

        # ## VerifyShuffle()
        wait_until_true(lambda: zone.AVTransport.get_current_play_mode() == zone.AVTransport.PLAYMODE_NORMAL,
                        timeout_seconds=5)

        # ## SetShuffle() --> fail
        with IgnoreInvokeErrors(zone):
            zone.AVTransport.SetPlayMode(zone.AVTransport.PLAYMODE_SHUFFLE_NO_REPEAT)
        self.verifyEqualOrFailCase(712, zone.last_upnp_error_code, "Shuffle invalid with http URI")
