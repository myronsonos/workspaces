        '''
        Basic API test with good/bad argument vectors
        '''
        self.logger.info("execute GetMediaInfo using queue.")
        (test_zp, (instance_id , exceptNum)) = vector_generator
        haveException = False

        try:
            res = test_zp.AVTransport.GetMediaInfo(instance_id)
            self.verifyTrueOrStop(isinstance(res, dict), "GetMediaInfo return value should be a Dict")
        except sonos.exception.UPnPError, e:
            if exceptNum > 0 and exceptNum == e.upnp_error:
                self.logger.info("Correctly received expected Invalid Argument Error %s: ", e)
                haveException = True
            else:
                self.logger.debug("GetMediaInfo failed with exception: %s " % e)
                self.fail("Unexpected UPnPError: %s" % e)
                 
        if exceptNum > 0:
            self.verifyTrueOrStop(haveException, "Should receive exception %s" % exceptNum)

        

[docs]    def test_combinatorial_get_media_info_queue(self, zone_generator):
        '''
        GetMediaInfo does have return fields that vary based on AVT URI type. This test 
        works with queue data.
        '''
        self.logger.info("execute GetMediaInfo using queue.")
        test_zp = zone_generator
        test_zp.AVTransport.SetAVTransportURI("", "")
        wait_until_true(lambda: test_zp.AVTransport.get_state_variable("AVTransportURI") == '', 0.5, 10)
        
        test_zp.AVTransport.use_queue()
        test_zp.AVTransport.clear_queue()

        test_zp.AVTransport.add_to_queue(BUZZER_URI)
        test_zp.AVTransport.add_to_queue(BUZZER_URI)
        num_tracks = 2

        try:
            res = test_zp.AVTransport.GetMediaInfo()
            self.verifyTrueOrStop(isinstance(res, dict), "GetMediaInfo return value should be a Dict")
            self.verifyEqualOrFailCase(QUEUE_URI_TEMPLATE % test_zp.raw_udn, res.get('CurrentURI'), "Current URI should be as set.")
            self.verifyEqualOrFailCase(num_tracks, int(res.get('NrTracks')), "Number of tracks should match.")
            self.verifyEqualOrFailCase(u'', res.get('CurrentURIMetaData'),  "CurrentURIMetaData defaults to None.")
            self.verifyEqualOrFailCase(u'NOT_IMPLEMENTED', res.get('MediaDuration'), "MediaDuration defaults to None.")
            self.verifyEqualOrFailCase(u'', res.get('NextURI'), "NextURI defaults to None.")
            self.verifyEqualOrFailCase(u'', res.get('NextURIMetaData'), "NextURIMetaData defaults to None.")
            self.verifyEqualOrFailCase(u'NETWORK', res.get('PlayMedium'), "PlayMedium defaults to NETWORK.")
            self.verifyEqualOrFailCase(u'NOT_IMPLEMENTED', res.get('RecordMedium'), "PlayMedium defaults to None.")
            self.verifyEqualOrFailCase(u'NOT_IMPLEMENTED', res.get('WriteStatus'), "WriteStatus defaults to None.")
        except sonos.exception.UPnPError, e:
            self.logger.debug("GetMediaInfo failed with exception: %s " % e)
            self.fail("Unexpected UPnPError: %s" % e)
            
        test_zp.AVTransport.clear_queue()

                    

[docs]    def test_combinatorial_play_radio(self, zone_generator):
        '''
        GetMediaInfo does have return fields that vary based on AVT URI type. This test 
        works with radio data.
        '''
        test_zp = zone_generator

        try:
            api_queue_management.use_queue_and_empty(test_zp)
            test_zp.AVTransport.SetAVTransportURI(RADIO_URI, "")
            # Issue a Play action
            test_zp.AVTransport.Play()
            
            res = test_zp.AVTransport.GetMediaInfo()
            test_zp.AVTransport.Stop()
            self.verifyTrueOrStop(isinstance(res, dict), "GetMediaInfo return value should be a Dict")
            self.verifyEqualOrFailCase(RADIO_URI, res.get('CurrentURI'), "Current URI should be as set.")
            self.verifyEqualOrFailCase(u'', res.get('CurrentURIMetaData'),  "CurrentURIMetaData defaults to None.")
            self.verifyEqualOrFailCase(u'NOT_IMPLEMENTED', res.get('MediaDuration'), "MediaDuration defaults to None.")
            self.verifyEqualOrFailCase(u'', res.get('NextURI'), "NextURI should be as set.")
            self.verifyEqualOrFailCase(u'', res.get('NextURIMetaData'), "NextURIMetaData defaults to None.")
            self.verifyEqualOrFailCase(u'NETWORK', res.get('PlayMedium'), "PlayMedium defaults to NETWORK.")
            self.verifyEqualOrFailCase(u'NOT_IMPLEMENTED', res.get('RecordMedium'), "PlayMedium defaults to None.")
            self.verifyEqualOrFailCase(u'NOT_IMPLEMENTED', res.get('WriteStatus'), "WriteStatus defaults to None.")
        except sonos.exception.UPnPError, e:
