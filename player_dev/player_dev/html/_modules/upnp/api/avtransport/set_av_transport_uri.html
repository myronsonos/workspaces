[docs]def check_track_metadata(xmldata, title=None, artist=None, album=None, creator=None, length=None, res=None):
    failed = {}
    try:
        md = sonos.services.common.XMLAccessor(xmldata)
        if res is not None and md.item.res.value != res:
            failed['res'] = md.item.res.value
        if length is not None and md.item.res.duration != length:
            failed['length'] = md.item.res.duration
        if title is not None and md.item.title.value != title:
            failed['title'] = md.item.title.value
        if artist is not None and md.item.albumArtist.value != artist:
            failed['artist'] = md.item.albumArtist.value
        if album is not None and md.item.album.value != album:
            failed['album'] = md.item.album.value
        if creator is not None and md.item.creator.value != creator:
            failed['creator'] = md.item.creator.value
    except Exception, e:
        failed['exception'] = str(e)
    return failed

        # TODO: check that aupeo doesn't change around their URIs (otherwise title match will fail)

        uri = radio.valid_stream_uri
