                                    generate_testbed_unique_playback_devices):
        """
        Execute the GetSetPlayMode test and check for play modes

        (test_zp) = generate_testbed_unique_playback_devices
        try:
            test_zp.AVTransport.SetPlayMode("NORMAL", instance_id=100)
        except sonos.exception.UPnPError as e:
            if e.upnp_error == UPNP_ERROR_718_INVALID_INSTANCE_ID:
                self.logger.debug("caught expected UPNP error")
            else:
                raise
        self.verifyEqualOrFailCase(test_zp.last_upnp_error_code,
                                   UPNP_ERROR_718_INVALID_INSTANCE_ID,
                                   "expect UPNP 718 to be returned when "
                                   "passing invalid instance ID to SetPlayMode")

[docs]    def test_combinatorial_get_set_play_mode_invalid_mode(self,
                                    generate_testbed_unique_playback_devices):
        """
        Execute the GetSetPlayMode test and check for play modes
        """
        test_zp = generate_testbed_unique_playback_devices

        test_zp.AVTransport.use_queue()
        test_zp.AVTransport.clear_queue_and_wait()
        test_zp.AVTransport.SetAVTransportURI(self.RADIO_URI, "")

        try:
            test_zp.AVTransport.SetPlayMode("SHUFFLE_NOREPEAT")

        except sonos.exception.UPnPError as e:
            if e.upnp_error == UPNP_ERROR_712_PLAYMODE_UNSUPPORTED:
                self.logger.debug("caught expected UPNP error")
            else:
                raise
        else:
            self.fail("expect UPNP 712 error attempting to set "
                      "invalid play mode during radio station playback")
        finally:
            test_zp.AVTransport.clear_play_state()
