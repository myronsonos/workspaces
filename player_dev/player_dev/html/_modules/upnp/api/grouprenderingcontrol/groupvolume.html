[docs]class WaitForGroupRenderingControlSnapshotConvergence(object):
    """
    Wait for the snap shot group volume to converge on the
    group coordinator (GC) play zone.
    It monitors, using UDP log, a specific diagnostic the
    GC device will generate when all the group members have
    completed the group rendering registration with the GC.
    """
    def __init__(self, gc):
        """
        """
        self.gc = gc
        self.logger = logging.getLogger(
            make_log_name([self.__class__.__name__]))


[docs]    def diag_callback(self):
        """
        UDP logger call back to indicate that the GC has
        received all the group rendering subscriptions
        for the snap shot group volume.
        """
        self.grp_converged = True



[docs]    def wait_for_convergence(self, msg, retries=5, timeout=5):
        """

        :param str msg: Search string to match the UDP log.
        :param int retries: The number of times to attempt to get
        the snap shot group volume from all the players in the group.
        :param int timeout: Number of seconds to wait on the
        snap shot group volume command for convergence.
        :return: True is snap shot group volume command converged
        :rtype: :obj:`bool`
        """
        self.grp_converged = False
        self.gc.diag.watch_udp_log_for_string(
            msg,
            callback=lambda msg: self.diag_callback())
        attempts = 0
        while attempts < retries:
            res = self.gc.GroupRenderingControl.SnapshotGroupVolume()
            try:
                wait_until_true(
                    lambda: self.grp_converged == True,
                    .5,
                    timeout)
            except sonos.exception.TimeoutError:
                attempts = 1 + attempts
                self.logger.debug(
      "wait_for_convergence: no convergence yet ... attempt {} of {}".format(
                        attempts, retries))
            else:
                break
        return self.grp_converged


    def __enter__(self):
        """
        Increase the diagnostic level of the group rendering module (grc).
        Activate UDP logging.
        """
        self.gc.diag.set_diag_level('grc', 5)
        self.gc.diag.start_udp_log()
        return self

    def __exit__(self, type, value, traceback):
        """
        Restore the diagnostic level to default
        """
        self.gc.diag.set_diag_level('grc', 1)
        self.gc.diag.stop_udp_log()

    def _set_ungroup_volume(self):
        """
        Initializes all zone players to known states prior to conducting tests.
        Ungroups the zone player from a group.
        Sets the player's volume to a default.
        If the device supports fixed output, sets fixed output to false.
        """
        playablezones = [player
                         for player in self.my_devices
                         if player.is_playback_device()]
        for device in playablezones:
            device.AVTransport.BecomeCoordinatorOfStandaloneGroup()
            wait_until_true(
                lambda: device.AVTransport.is_group_member() == False,
                1.0,
                self.TIMEOUT_STANDALONE_ZP)
            if device.RenderingControl.GetSupportsOutputFixed():
                device.RenderingControl.SetOutputFixed(False)
            device.RenderingControl.SetVolume(self.DEFAULT_VOLUME)

[docs]    def calc_grp_vol(self, volumes):
        """
        Given an integer list of group's volumes (includes GC's volume),
        it returns a tuple fo the total group volume and the average of
        the group's volume as a float.
        :param list volumes: List of integers representing the volume
        of each device in the group.
        :return: A tuple total volume and the group's average volume.
        :rtype: :obj:`tuple`
        """
        return (sum(volumes), float(sum(volumes))/float(len(volumes)))



[docs]    def calc_mbr_volume(self, grp_vol, mbr_vol, new_grp_vol):
        """
        Calculate the group member's expected volume based on the current
        group snapshot volume, the current member volume and the
        desired new group volume. Implementation based on:
        https://confluence.sonos.com/display/PDSW/MediaRenderer%3AGroupRenderingControl
        :param float grp_vol: Current group volume
        :param int mbr_vol: Current member volume
        :param int new_grp_vol: Desired group volume
        :return: Expected volume of group member
        :rtype: :obj:`int`
        
        """
        offset = 100.0 if float(new_grp_vol) > float(grp_vol) else 0.0
        proportion = (offset - float(mbr_vol)) / (offset - float(grp_vol))
        computed_mbr_vol = float(mbr_vol) + (proportion * (float(new_grp_vol) - float(grp_vol)))
        self.logger.debug("gv={} mv={} ngr_v={} off={} prop={} comp_mv={}".format(
                grp_vol,
                mbr_vol,
                new_grp_vol,
                offset,
                proportion,
                computed_mbr_vol))
        return int(round(computed_mbr_vol))

	"""
	Determines whether devices are available on a given testbed to run
	tests against. If a device is available when a test calls on this
	method, it will pick it from a list and return it to test against. If a
	device is not available the test will then be skipped and output a log
	indicating what device is required.
	:param model: Model number of device to look for
	:type model: :obj:`str`
	"""
	device = self.my_models.get(model)
        device = device and device[0]
        self.verifyTrueOrSkip(device, "Test requires {}".format(model))
	return device

[docs]    @combinatorial('generate_testbed_unique_playback_devices')
    def test_SetGroupVolume_all_devices_same_starting_volume(self, zp):
        """
        Test: SetGroupVolume when all playback devices have the same initial volume
        Devices: All playback capable
        Expected: Success
        """
        if zp.is_virtual():
            self.skip("Skip this test on virtual players. Expect timing issues.")
        (gc, playablezones) = zp.AVTransport.create_zone_group_of_xnum_devices(
            self.my_devices,
            timeout_mbr_chk=self.TIMEOUT_STANDALONE_ZP,
            timeout_join=self.TIMEOUT_JOIN)
        # Set variable to number of player devices in the group, include GC
        grp_size = len(playablezones) + 1
        # Create the expected diagnostic the GC will emit when all the GMs
        # have completed their group rendering registration.
        # The diagnostic message is based on the number of devices in the group
        # along with the aggregate group volume.

        self.logger.debug("Getting snap shot group volume for {} zones aggregate volume {}".format(
                grp_size, grp_size*self.DEFAULT_VOLUME))
        with WaitForGroupRenderingControlSnapshotConvergence(gc) as w:
            self.verifyTrueOrFailCase(
                w.wait_for_convergence(self.SNAPSHOT_DIAG_STR.format(
                        grp_size,
                        grp_size)),
                "Expect rendering control to converge")

        self.verifyTrueOrFailCase(
            gc.GroupRenderingControl.SetGroupVolume(15),
            "Expect SetGroupVolume(15) to succeed")

        # Required quiet time reference
        # https://confluence.sonos.com/display/PDSW/Group+Rendering+Control+Guide
        time.sleep(2)

        # Verify that the group volume reaches the expected value.
        # A 'wait' is needed here because the GroupVolume value updates
        # asynchronously as the group coordinator ZP receives volume
        # events from each of the ZPs in the group.
        wait_until_true(
            lambda: gc.GroupRenderingControl.GetGroupVolume() == 15,
            1,
            10)
        self.verifyEqualOrFailCase(
            15,
            gc.GroupRenderingControl.GetGroupVolume(),
            "GroupVolume should be set to 15")



[docs]    @combinatorial('generate_testbed_unique_playback_devices')
    def test_SetGroupVolume_not_all_same_volumes(self, zp):
        """
        Test: SetGroupVolume when not all grouped devices have the same initial volume
        Devices: All playback capable
        Expected: Success
        """

        # Ask for a group of 3 devices and make sure we get 3 back
        (gc, playablezones) = zp.AVTransport.create_zone_group_of_xnum_devices(
            self.my_devices,
            timeout_mbr_chk=self.TIMEOUT_STANDALONE_ZP,
            timeout_join=self.TIMEOUT_JOIN)
        if len(playablezones) < 3:
            self.fail("Test requires at least 3 zones to run, only found {}".format(len(playablezones)))
        else:
            volumes = [10, 20, 35]
            # Set each zone volume individually
            for zone, volume in itertools.izip(playablezones, volumes):
                zone.RenderingControl.SetVolume(volume)
                wait_until_true(
                    lambda: zone.RenderingControl.GetVolume() == volume, 1, 5)

            # Derive the expected group volume.
            # The group volume is the average of all the volumes in the
            # group (includes GC).
            # Create an input list of integers that comprises the
            # individual volumes for each device in the group and
            # derive the expected group volume.
            (agg_grp_vol, expected_gvf) = self.calc_grp_vol(
                volumes[:] + [self.DEFAULT_VOLUME
                              for x in range(len(playablezones) - len(volumes) + 1)])
            # Convert to an integer - the player advertises the group volume as an integer
            expected_gv = int(expected_gvf)
            self.logger.debug("Expecting group volume int={} float={}".format(
                    expected_gv, expected_gvf))
            # wait for expected GV value before continuing
            try:
                wait_until_true(
                    lambda: gc.GroupRenderingControl.GetGroupVolume() == expected_gv,
                    1, 10)
            except sonos.exception.TimeoutError:
                self.verifyEqualOrFailCase(
                    gc.GroupRenderingControl.GetGroupVolume(),
                    expected_gv,
                    "Expect group volume to match")

            grp_size = len(playablezones) + 1
            with WaitForGroupRenderingControlSnapshotConvergence(gc) as w:
                self.verifyTrueOrFailCase(
                    w.wait_for_convergence(self.SNAPSHOT_DIAG_STR.format(
                            grp_size,
                            grp_size)),
                    "Expect capture of group's snap shot volume to succeed")

            self.verifyTrueOrFailCase(
                gc.GroupRenderingControl.SetGroupVolume(self.NEW_GRP_VOLUME),
                "Expect setting of new group volume to succeed")

            # Required quiet time reference
            # https://confluence.sonos.com/display/PDSW/Group+Rendering+Control+Guide
            time.sleep(2)

            newvolumes = []
            # Note that we pass the float value of the group's average volume
            # to derive the expected volume of the member.
            expectednewvolumes = [self.calc_mbr_volume(
                    expected_gvf, mbr_vol, self.NEW_GRP_VOLUME) for mbr_vol in volumes]
            self.logger.debug("expectednewvolumes={}".format(expectednewvolumes))

            for zone, volume in itertools.izip(playablezones, expectednewvolumes):
                try:
                    wait_until_true(
                        lambda: zone.RenderingControl.GetVolume() == volume,
                        1, 15)
                    newvolumes.append(zone.RenderingControl.GetVolume())
                except sonos.exception.TimeoutError:
                    newvolumes.append(zone.RenderingControl.GetVolume())

            self.verifyEqualOrFailCase(
                expectednewvolumes,
                newvolumes,
                "GroupVolume should be set correctly based on last snapshot value")



[docs]    def test_setgroupvolume_does_not_affect_headphone_only_when_GM(self):
        """
        Test: Verify a group member that is in headphone mode is not affected by GRC volume actions.
	If testbed does not contain device requirements, test will be skipped.
        Devices: Headphone mode capable (Play:5)
        Expected: Success
        """
	gc = self.get_or_skip(ZP120)
	play5 = self.get_or_skip(S5)
        play5.RenderingControl.SetMute(True, "SpeakerOnly")
        wait_until_true(lambda: play5.RenderingControl.GetMute("SpeakerOnly") == True, 1.0, 5)
        play5.AVTransport.join_group(gc)
        wait_until_true(play5.AVTransport.is_group_member, 1.0, 5)
        beforevol = play5.RenderingControl.GetVolume()

        gc.GroupRenderingControl.SetGroupVolume(35)

        aftervol = play5.RenderingControl.GetVolume()

        self.verifyEqualOrFailCase(beforevol, aftervol,
            "GroupVolume actions should not affect players in headphone mode")
        play5.RenderingControl.SetMute(False, "SpeakerOnly")

        """
        Test: Verify that players set to fixed volume output are not affected by SetGroupVolume when grouped
        Devices: Fixed output capable (CONNECT)
        Expected: Success

        fixedoutputcapable = [player
                              for player in self.my_devices

            beforevol = connect.RenderingControl.GetVolume()
            gc.GroupRenderingControl.SetGroupVolume(35)
            wait_until_true(lambda: gc.GroupRenderingControl.GetGroupVolume() == 35, 1.0, 5)

            aftervol = connect.RenderingControl.GetVolume()

            self.verifyEqualOrFailCase(beforevol, aftervol,
                "SetGroupVolume should not affect fixed output players when grouped")
