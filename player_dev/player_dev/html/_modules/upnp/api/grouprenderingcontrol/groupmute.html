[docs]class GroupMuteTests(WorkflowTestFixture):
    """
    GroupRenderingControl.SetGroupMute(mute_state, instanceID)
    GroupRenderingControl.GetGroupMute()
    """
    TEST_TYPE = ['NIGHTLY_API_GROUPRENDERINGCONTROL']
    """
    Maximum seconds to wait for the zone player to transition
    to a stand-alone player
    """
    TIMEOUT_STANDALONE_ZP = 20
    """
    Maximum seconds to wait for the zone player to join a group.
    """
    TIMEOUT_JOIN = 15

    def _set_ungroup_unmute(self):
        """
        Resets all playable devices to a known mute state prior to
        commencing tests.
        Ungroups the player device from a group.
        Set the mute state to False.
        """
        playablezones = filter(lambda x: x.is_playback_device(), self.my_devices)
        for device in playablezones:
            device.AVTransport.BecomeCoordinatorOfStandaloneGroup()
            wait_until_true(lambda: device.AVTransport.is_group_member() == False,
                            1.0,
                            self.TIMEOUT_STANDALONE_ZP)
            device.RenderingControl.SetMute(False, "Master")

    def test_SetGroupMute(self, zone, mute):
        """
        Test: Mute a Playable ZP on all channels - Master, RF, LF, SpeakerOnly
        Devices: Playable devices
        Expected: Success
        """
        zone.GroupRenderingControl.SetGroupMute(mute)
        self.verifyEqualOrFailCase(zone.last_upnp_error_code,
                                   sonos.client.zone_player.UPNP_OK,
            "SetGroupMute should receive UPNP_OK (0) response")
        self.verifyEqualOrFailCase(zone.GroupRenderingControl.GetGroupMute(), mute,
            "GetGroupMute should reflect the mute state (%s) set by the test" % mute)

[docs]    @combinatorial('generate_testbed_unique_playback_devices')
    def test_GetGroupMute_after_RC_SetMute_each_device(self, zp):
        """
        Test: GetGroupMute for GC when RenderingControl.SetMute was used on each device to set the mute state
        Devices: All playable devices
        Expected: Success - GRC GetGroupMute call should match the individual ZP mute states
        """
        (gc, playablezones) = zp.AVTransport.create_zone_group_of_xnum_devices(
            self.my_devices, timeout_mbr_chk=self.TIMEOUT_STANDALONE_ZP, timeout_join=self.TIMEOUT_JOIN)

        self.verifyTrueOrFailCase(
            gc.RenderingControl.SetMute(True),
            "Expect SetMute(True) to succeed on GC {}".format(
                gc.modelNumber))

        for zone in playablezones:
            self.verifyTrueOrFailCase(
                zone.RenderingControl.SetMute(True),
                "Expect SetMute(True) to succeed on GM {}".format(
                    zone.modelNumber))

            self.verifyTrueOrStop(zone.RenderingControl.GetMute(),
                                  "Failed to set the correct mute value "
                                  "for {}".format(zone))

        wait_until_true(lambda: gc.GroupRenderingControl.GetGroupMute() == True,
                        iteration_delay=1.0,
                        timeout_seconds=15)
        self.verifyEqualOrFailCase(True,
                                   gc.GroupRenderingControl.GetGroupMute(),
        "GroupMute state should reflect the overall state of the group")

    def test_GetGroupMute_after_RC_SetMute_random_device(self, zp):
        """
        Test: GetGroupMute state after SetMute on a few random devices
        Devices: All playback capable
        Expected: GroupMute state should return true iff ALL devices are currently muted
        """
        (gc, playablezones) = zp.AVTransport.create_zone_group_of_xnum_devices(
            self.my_devices, timeout_mbr_chk=self.TIMEOUT_STANDALONE_ZP, timeout_join=self.TIMEOUT_JOIN)

        if len(playablezones) < 3:
            self.fail("Not enough zones to mute 2 but not all, only found {}".format(
                    len(playablezones)))
        else:
            zone = choice(playablezones)
            playablezones.remove(zone)
            self.verifyTrueOrFailCase(
                zone.RenderingControl.SetMute(True),
                "Expect to set mute on GM {}".format(
                    zone.modelNumber))
            zone = choice(playablezones)
            playablezones.remove(zone)
            self.verifyTrueOrFailCase(
                zone.RenderingControl.SetMute(True),
                "Expect to set mute on GM {}".format(
                    zone.modelNumber))

            self.verifyEqualOrFailCase(
                gc.GroupRenderingControl.GetGroupMute(),
                False,
            "GroupMute state should report true iff all devices are currently muted")

    def test_SetGroupMute_true_on_GC_verify_on_all_devices(self, zp):
        """
        Test: SetGroupMute(True) using the GC - verify each device individually to make sure the RC service reports a muted state
        Devices: All playback capable
        Expected: Success - each device in the group should have its mute state set by the GRC command
        """
        (gc, playablezones) = zp.AVTransport.create_zone_group_of_xnum_devices(
            self.my_devices, timeout_mbr_chk=self.TIMEOUT_STANDALONE_ZP, timeout_join=self.TIMEOUT_JOIN)

        self.verifyTrueOrFailCase(
            gc.GroupRenderingControl.SetGroupMute(True),
            "Expect SetGroupMute(True) to succeed on GC {}".format(
                gc.modelNumber))

        wait_until_true(lambda: gc.GroupRenderingControl.GetGroupMute() == True,
                        iteration_delay=1.0,
                        timeout_seconds=15)
        individualmutestates = []
        for zone in playablezones:
            individualmutestates.append(zone.RenderingControl.GetMute())
        self.verifyNotInOrFailCase(False, individualmutestates,
            "All devices in a group should be muted when SetGroupMute(True) is called on the GC")
        self.verifyEqualOrFailCase(gc.GroupRenderingControl.GetGroupMute(), True,
            "GroupMute state should reflect the last set state of true")
