    def test_get_zone_group_state(self, device, exceptNum):
        """
        Execute the GetZoneGroupState test with the set of devices.

        :param device: SonosZoneComponent
        :type device: :class:`~sonos.client.internal.SonosZoneComponent`
        :param int exceptNum: the expected exception error number
        """
        self.logger.info("{}: {}".format(TCID,testCaseMap.get(TCID)))

        haveException = False
        try:
            result = device.ZoneGroupTopology.GetZoneGroupState()
            self.verifyIsOrFailCase(type(dict()), type(result), "Return value should be a dictionary.")
            self.verifyIsNotNoneOrFailCase(result['ZoneGroupState'], "ZoneGroupState should not be empty")
        except UPnPError as e:
            if exceptNum > 0 and exceptNum == e.upnp_error:
                self.logger.info("Correctly received expected Invalid Argument Error: {}".format(e))
                haveException = True
            else:
                self.logger.debug("GetZoneGroupState failed with exception: {}".format(e))
                self.verifyIsNoneOrStop(e, "Unexpected UPnPError: {}".format(e))

        if exceptNum > 0:
            self.verifyTrueOrStop(haveException, "Should receive exception {}".format(exceptNum))


    def _verify_voice_state(self, zp, expected_state):
        """
        Verify the voice state event in ZoneGroupState
        matches the expected state.

        :param zp: SonosZoneComponent
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`
        :param str expected_state: the expected voice state event value
        """
        try:
            wait_until_true(lambda: expected_state ==
                            zp.ZoneGroupTopology.get_voice_state())
        except TimeoutError:
            self.fail("Expect the voice state for {} to match the expected "
                      "state".format(zp))
