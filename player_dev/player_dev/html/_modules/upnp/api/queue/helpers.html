               does_service_account_exist(self.SMAPI_SERVER_INTERNAL_ID))
        self.verifyTrueOrStop(res, 'Mock SMAPI server should be added successfully')

    def _verify_tracks_in_queue(self, zone, queue_id, expected_track_list, start_position):
        failed_checks = 0

        # since Browse may return less than the total results, we may need more calls
        tracks_checked = 0
        while tracks_checked <= len(expected_track_list):
            result = zone.Queue.Browse(queue_id=queue_id,
                                       starting_index=start_position-1,
                                       requested_count=len(expected_track_list) - tracks_checked,
                                       convert_xml=False)

            md_items = DIDLLite.DIDLElement.fromString(result['Result']).getItems()
            if len(md_items) == 0:
                if tracks_checked != len(expected_track_list):
                    self.logger.debug("no results?! pos=%d, tc=%d, exp=%d" %
                                      (start_position, tracks_checked, len(expected_track_list)))
                    failed_checks += 1
                else:
                    self.logger.debug("no more results and none expected")
                break

            for item in md_items:
                track = expected_track_list[tracks_checked]
                self.logger.debug("%d/%d: track: %s -- item: %s" % (tracks_checked, start_position, track, item.toString()))

                # NOTE:
                #   Prior to ~2013-01-09, a ZP Browse() will return the data in ExtraMd, which is NOT
                #   what is sent as EnqueuedMD.  This means that the 'title' returned will likely be the filename
                #   and the 'albumArtURI' will be ZP-proxied (/getaa?u=...)
                #
                #   With the Browse() changes to support using the EnqueuedMD, the data returned should be the
                #   same as provided (except for item@id and item@parentID, which get replaced; the originals are
                #   stored in item@origID and item@origParentID).

                if not hasattr(item, "title") or item.title != track.title:
                    self.logger.debug("Browsed complete md should have matching title value")
                    failed_checks += 1
                if not hasattr(item, "album") or item.album != track.album:
                    self.logger.debug("Browsed complete md should have matching album value")
                    failed_checks += 1
                #TED: I hate python2.x and unicode!  (This is left behind to remind of the pain...)
                #print repr(item.creator), repr(track.artist), repr(track.artist.decode('utf-8'))
                if not hasattr(item, "creator") or item.creator != track.artist.decode('utf-8'):
                    self.logger.debug("Browsed complete md should have matching artist value")
                    failed_checks += 1
                #track.artwork_url

                tracks_checked += 1
                start_position += 1

        if failed_checks > 0:
            self.fail("Browsed Queue should match expected tracks (see debug log)")
        else:
            return True

[docs]    def generate_metadata_and_add_tracks_to_queue(self, zone, queue_id, track_list):
        count = 1
        for track in track_list:
            # generate valid DIDL metadata
            md = md_helper.generate_didl_lite(title=track.title,
                                                     svc_id=None,
                                                     acct_id=None,
                                                     upnp_class="object.item.audioItem.musicTrack",
                                                     artist=track.artist,
                                                     album=track.album,
                                                     aa_uri=track.artwork_url)
            update_id = self.get_private_queue_update_id(zone, queue_id)
            #Add the URI and metadata to the queue
            zone.Queue.AddURI(queue_id=queue_id,
                                    update_id=update_id,
                                    enqueueduri=track.media_url,
                                    enqueuedurimetadata=md,
                                    desiredfirsttracknumberenqueued=count,
                                    enqueueasnext=False)
            count += 1

[docs]    def create_uri_and_metadata_xml(self, track_list, md_list):
        uris = '<URIs>'
        for track, md in zip(track_list, md_list):
            uris += '<URI uri="%s">' % track.media_url
            uris += md
            uris += '</URI>'
        uris += '</URIs>'
        return uris
