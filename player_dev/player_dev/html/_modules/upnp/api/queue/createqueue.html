    def test_evented_owner_id_string_SetAVTransportURI(self, zone):
        """
        SWPBL-55180 Sonos CR can identify the app that created the playback session
        This tests checks that DirectControlClientId included in an UPnP event is the expected value
        before selecting the private queue, after selecting the private queue, and after clearing
        the session.
        The session is cleared in this test using AVTransport.SetAVTransportURI(uri="", metadata="").
        """
        # Clear session and wait for empty direct control client id
        zone.AVTransport.SetAVTransportURI(uri="", metadata="")
        wait_until_true(lambda: zone.AVTransport.get_state_variable("DirectControlClientID") == '', timeout_seconds=5)

        # Store current control id. This is extremely unlikely, but the value can change between the end of
        # wait_until_true and this statement, so verify that the string is the expected value.
        dcc_id_str0 = zone.AVTransport.get_state_variable("DirectControlClientID")
        self.verifyEqualOrFailCase(dcc_id_str0, '',
                                   "DirectControlClientId should be empty before creating private queue")

        # Create private queue
        my_owner_id = "my_owner_id"
        my_owner_context = "my_owner_context"
        private_queue_id = zone.Queue.CreateQueue(queueownerid=my_owner_id,
                                                  queueownercontext=my_owner_context)
        private_queue_uri = "x-rincon-queue:{}#{}".format(zone.raw_udn, private_queue_id)

        # Use private queue
        zone.AVTransport.SetAVTransportURI(uri=private_queue_uri, metadata="")

        # Wait for the direct control id to change
        wait_until_true(lambda: zone.AVTransport.get_state_variable("DirectControlClientID") != dcc_id_str0, timeout_seconds=5)

        # Get direct control client id after setting private queue (should be the owner id)
        dcc_id_str1 = zone.AVTransport.get_state_variable("DirectControlClientID")
        self.verifyEqualOrFailCase(dcc_id_str1, my_owner_id,
                                    "DirectControlClientId must match owner id after setting private queue")

        # Clear session
        zone.AVTransport.SetAVTransportURI(uri="", metadata="")

        # Wait for the direct control id to change
        wait_until_true(lambda: zone.AVTransport.get_state_variable("DirectControlClientID") != dcc_id_str1, timeout_seconds=5)

        # Get direct control client id after clearing play state and verify its value
        dcc_id_str2 = zone.AVTransport.get_state_variable("DirectControlClientID")
        self.verifyEqualOrFailCase(dcc_id_str2, '',
                                   "DirectControlClientId should be empty clearing play state")
