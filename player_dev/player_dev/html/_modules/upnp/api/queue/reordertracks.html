[docs]    @combinatorial(playable_zone_generator)
    def test_reordertracks_from_beginning(self, zone):
        """
        Verify reordering tracks in the queue causes the expected new order
        """
        zone.AVTransport.SetPlayMode(zone.AVTransport.PLAYMODE_NORMAL)
        zone.AVTransport.use_queue(queue_id=self.queue_id)
        zone.AVTransport.clear_queue_and_wait(instance_id=self.queue_id)

        track_list = []
        while len(track_list) < 9:
            track_list.append(random.choice(self.track_source.tracks[0:3]))

        self.add_http_tracks_to_queue(zone, track_list)
        res = zone.Queue.Browse(self.queue_id, 0, 0)
        update_id = int(res.get('UpdateID'))

        new_update_id = zone.Queue.ReorderTracks(self.queue_id,
                                 startingindex=1,
                                 numberoftracks=2,
                                 insertbefore=5,
                                 updateid=update_id)

        newtrackorder = [2, 3, 0, 1, 4, 5, 6, 7, 8]
        newtracklist = [track_list[i] for i in newtrackorder]

        self.verify_tracks_in_queue(zone, newtracklist, 1)
        self.verifyGreaterOrFailCase(new_update_id, update_id,
                                     "UpdateID should increment upon edit")

    def test_reordertracks_bad_update_id(self, zone):
        """
        Verify reordering tracks in the queue causes the expected new order
        """
        zone.AVTransport.SetPlayMode(zone.AVTransport.PLAYMODE_NORMAL)
        zone.AVTransport.use_queue(queue_id=self.queue_id)
        zone.AVTransport.clear_queue_and_wait(instance_id=self.queue_id)

        track_list = []
        while len(track_list) < 9:
            track_list.append(random.choice(self.track_source.tracks[0:3]))

        zone.ContentDirectory.Browse("Q:{}".format(self.queue_id),
                                     "BrowseMetadata", "", 0, 0)["UpdateID"]

        self.add_http_tracks_to_queue(zone, track_list)
        res = zone.Queue.Browse(self.queue_id, 0, 0)
        update_id = int(res.get('UpdateID'))

        # Checking zone.last_upnp_error_code seems error prone.
        # If the testrunner and an external service both make UPnP requests
        # at roughly the same time, we have a race.
        try:
            zone.Queue.ReorderTracks(self.queue_id,
                                     startingindex=1,
                                     numberoftracks=4,
                                     insertbefore=5,
                                     updateid=update_id * 2)
        except sonos.exception.UPnPError as e:
            self.verifyEqualOrFailCase(
                sonos.client.zone_player.UPNP_ERROR_1028_STALE_OBJECT,
                e.upnp_error,
                "Attempts to reorder with a bad updateid should "
                "fail with a 1028(stale data) error")
        else:
            self.fail("UPnP error not raised on reordering with bad updateid")
