# a player is ready if
# * it's the only device in its group
# * its AVT URI is set to its own queue
# * its queue is empty
# * its play mode is "NORMAL"
#
# this helper function checks this state so we can avoid a bunch of redundant UPNP calls to make players ready

[docs]def is_player_state_ready(zone, queue_id):
    ready = True
    queue_uri = "x-rincon-queue:{}#{}".format(zone.raw_udn, queue_id)
    ready = ready and zone.AVTransport.get_avtransport_uri() == queue_uri
    ready = ready and zone.ZoneGroupTopology.get_zone_player_uuids_in_group() == zone.raw_udn
    ready = ready and zone.AVTransport.get_number_of_tracks_in_queue(instance_id=queue_id) == 0
    ready = ready and zone.AVTransport.get_current_play_mode() == "NORMAL"
    return ready

    def test_add_multiple_uris(self, zone):
        """
        Test: Queue:AddMultipleURIs with 16 URIs
        Devices: All playable
        Expected: Success
        """
        track_list = []
        while len(track_list) < 16:
            track_list.append(random.choice(self.track_source.tracks[0:3]))
        track_list_md = helpers.create_didl_lite_md(track_list)
        enqueued_uris_and_metadata = helpers.create_uri_and_metadata_xml(track_list, track_list_md)
        res = zone.Queue.AddMultipleURIs(queue_id=self.queue_id,
                                         update_id=0,
                                         container_uri="",
                                         container_metadata="",
                                         desired_first_track_number_enqueued=1,
                                         enqueue_as_next=0,
                                         number_of_uris=len(track_list),
                                         enqueued_uris_and_metadata=enqueued_uris_and_metadata)

        self.verifyTrueOrFailCase(helpers._verify_tracks_in_queue(zone=zone,
                                                                  queue_id=self.queue_id,
                                                                  expected_track_list=track_list,
                                                                  start_position=1),
                                  "Expected tracks should be in the queue")
        self.verifyEqualOrFailCase(1, int(res['FirstTrackNumberEnqueued']),
                                   "First track enqueued should match expected.")

    def test_add_multiple_uris_enqueue_next(self, zone):
        """
        Test: Queue:AddMultipleURIs with shuffle on and using EnqueueAsNext
        Devices: All playable
        Expected: Success
        """
        track_list = []
        first_track_list = self.track_source.tracks[0:3]
        first_track_list_md = helpers.create_didl_lite_md(first_track_list)
        first_enqueued_uris_and_metadata = helpers.create_uri_and_metadata_xml(first_track_list,
                                                                               first_track_list_md)
        self.logger.debug("Track list: {}".format(first_track_list))
        self.logger.debug("Enqueued URIs and Metadata {}: "
                          .format(first_enqueued_uris_and_metadata))
        second_track_list = []
        second_track_list = self.track_source.tracks[3:]
        second_track_list_md = helpers.create_didl_lite_md(second_track_list)
        second_enqueued_uris_and_metadata = helpers.create_uri_and_metadata_xml(second_track_list, second_track_list_md)
        self.logger.debug("Second Enqueued URIs and Metadata {}: ".format(second_enqueued_uris_and_metadata))

        first_track_uri_list = []
        for track in first_track_list:
            first_track_uri_list.append(track.media_url)
        self.logger.debug("Track URI list: {}".format(first_track_uri_list))

        second_track_uri_list = []
        for track in second_track_list:
            second_track_uri_list.append(track.media_url)

        self.logger.debug("Second Track URI list: {}".format(second_track_uri_list))
        try:
            add_track_list_response = zone.Queue.AddMultipleURIs(queue_id=self.queue_id,
                                      update_id=0,
                                      container_uri="",
                                      container_metadata="",
                                      desired_first_track_number_enqueued=1,
                                      enqueue_as_next=0,
                                      number_of_uris=len(first_track_list),
                                      enqueued_uris_and_metadata=first_enqueued_uris_and_metadata)
            self.logger.debug("Add first track list response: {}".format(add_track_list_response))
        except sonos.exception.UPnPError, e:
            self.logger.debug("UPnP error adding first track list: {}".format(e))
            # Pass?
            pass


        wait_until_true(lambda: zone.AVTransport.
                        get_number_of_tracks_in_queue(instance_id=self.queue_id) ==
                        len(first_track_list), 0.5, 5)

        next_track = zone.AVTransport.get_state_variable("NextTrackURI")
        self.logger.debug("Next track after initial add to queue and prior to shuffle: {}".format(next_track))

        zone.AVTransport.SetPlayMode("SHUFFLE")

        # wait until track changes (or it might not!)
        try:
            wait_until_true(lambda: zone.AVTransport.get_state_variable("NextTrackURI") != next_track, 0.5, 5)
        # ...
        except sonos.exception.TimeoutError:
            pass
        next_track = zone.AVTransport.get_state_variable("NextTrackURI")
        self.logger.debug("Next track after shuffle and wait: {}".format(next_track))

        try:
            add_track_list_response = zone.Queue.AddMultipleURIs(
                                        queue_id=self.queue_id,
                                        update_id=0,
                                        container_uri="",
                                        container_metadata="",
                                        desired_first_track_number_enqueued=0,
                                        enqueue_as_next=1,
                                        number_of_uris=len(second_track_list),
                                        enqueued_uris_and_metadata=second_enqueued_uris_and_metadata)
            self.logger.debug("Add Second track list response: {}".format(add_track_list_response))
        except sonos.exception.UPnPError, e:
            self.logger.debug("UPnP error adding second track list: {}".format(e))
            pass

        # Ensure the full set of tracks are in the queue
        wait_until_true(lambda: zone.AVTransport.get_number_of_tracks_in_queue(instance_id=self.queue_id) ==
                        (len(first_track_list) + len(second_track_list)), 0.5, 5)

        # Setting the desired track as the NextTrackURI may take a moment.
        time.sleep(2)
        resulting_track = zone.AVTransport.get_state_variable("NextTrackURI")

        # For completeness in the prevalent case when no timeout occurs
        self.logger.debug("Resulting track after add with enqueue_as_next: {}".format(resulting_track))
        self.verifyInOrFailCase(resulting_track, second_track_uri_list,
            "EnqueueAsNext flag should cause the first track enqueued to be played next")

    def test_add_multiple_uris_dont_enqueue_next(self, zone):
        """
        Test: Queue:AddMultipleURIs with shuffle on and not using EnqueueAsNext
        Devices: All playable
        Expected: Success; next track does not change
        """

        track_list = []
        while len(track_list) < 3:
            track_list.append(random.choice(self.track_source.tracks[0:2]))
        track_list_md = helpers.create_didl_lite_md(track_list)
        enqueued_uris_and_metadata = helpers.create_uri_and_metadata_xml(track_list, track_list_md)

        second_track_list = []
        while len(second_track_list) < 3:
            second_track_list.append(random.choice(self.track_source.tracks[2:4]))
        second_track_list_md = helpers.create_didl_lite_md(second_track_list)
        second_enqueued_uris_and_metadata = helpers.create_uri_and_metadata_xml(second_track_list, second_track_list_md)

        track_uri_list = []
        for track in second_track_list:
            track_uri_list.append(track.media_url)

        try:
            zone.Queue.AddMultipleURIs(queue_id=self.queue_id,
                                       update_id=0,
                                       container_uri="",
                                       container_metadata="",
                                       desired_first_track_number_enqueued=1,
                                       enqueue_as_next=0,
                                       number_of_uris=len(track_list),
                                       enqueued_uris_and_metadata=enqueued_uris_and_metadata)
        except sonos.exception.UPnPError:
            pass

        # wait until we have a next track
        try:
            wait_until_true(lambda: zone.AVTransport.get_state_variable("NextTrackURI") != '', 0.5, 5)
        except sonos.exception.TimeoutError:
            pass

        next_track = zone.AVTransport.get_state_variable("NextTrackURI")
        self.verifyEqualOrFailCase(track_list[1].media_url, next_track,
                                   "Next track should be as added")

        # now shuffle, which could have changed the next track
        zone.AVTransport.SetPlayMode("SHUFFLE_NOREPEAT")
        wait_until_true(lambda: zone.AVTransport.get_state_variable("CurrentPlayMode") == "SHUFFLE_NOREPEAT", 0.5, 5)
        next_track = zone.AVTransport.get_state_variable("NextTrackURI")

        # Finally, add the 2nd batch of tracks
        try:
            zone.Queue.AddMultipleURIs(queue_id=self.queue_id,
                                       update_id=0,
                                       container_uri="",
                                       container_metadata="",
                                       desired_first_track_number_enqueued=4,
                                       enqueue_as_next=0,
                                       number_of_uris=len(second_track_list),
                                       enqueued_uris_and_metadata=second_enqueued_uris_and_metadata)
        except sonos.exception.UPnPError:
            pass

        self.verifyEqualOrFailCase(zone.AVTransport.get_state_variable("NextTrackURI"), next_track,
                                   "EnqueueAsNext=0 should not change the next track to be played")
