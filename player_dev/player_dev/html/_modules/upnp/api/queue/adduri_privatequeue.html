[docs]def create_didl_lite_md(zone, track_list):
    for track in track_list:
        # generate valid DIDL metadata
        md = zone.AVTransport.generate_didl_lite(title=track.title,
                                                 svc_id=None,
                                                 acct_id=None,
                                                 upnp_class="object.item.audioItem.musicTrack",
                                                 artist=track.artist,
                                                 album=track.album,
                                                 aa_uri=track.artwork_url)
        return md

[docs]    @combinatorial(playable_zone_generator)
    def test_adduri(self, zone):
        """
        Test: Add URI to queue
        Devices: All playable
        Expected: Success
        """
        if len(self.track_source.tracks) <= 1:
            self.stop("Needs 1 or more tracks {} to work with".
                      format(len(self.track_source.tracks)))

        zone.AVTransport.SetPlayMode(zone.AVTransport.PLAYMODE_NORMAL)
        zone.AVTransport.use_queue(queue_id=self.queue_id)
        zone.AVTransport.clear_queue_and_wait(instance_id=self.queue_id)

        track = []
        track.append(self.track_source.tracks[0])

        res = zone.Queue.Browse(queue_id=self.queue_ids[zone.ip],
                                starting_index=0,
                                requested_count=0)
        update_id = int(res.get('UpdateID'))

        md = create_didl_lite_md(zone, track)

        zone.Queue.AddURI(queue_id=self.queue_ids[zone.ip],
                          update_id=update_id,
                          enqueueduri=track[0].media_url,
                          enqueuedurimetadata=md,
                          desiredfirsttracknumberenqueued=1,
                          enqueueasnext=False)

        self.verifyEqualOrFailCase(1, queue_service_helper.get_number_of_tracks_in_queue(zone, self.queue_ids[zone.ip]),
                                   "Queue should contain the correct number of items")

        Test: Add URI to queue with invalid queue id
