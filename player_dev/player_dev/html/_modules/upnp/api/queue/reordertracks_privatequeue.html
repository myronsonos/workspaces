[docs]    @combinatorial(playable_zone_generator)
    def test_reordertracks(self, zone):
        """
        Verify reordering tracks in the queue causes the expected new order
        """
        track_list = []
        while len(track_list) < 9:
            track_list.append(random.choice(self.track_source.tracks[0:3]))

        helpers.generate_metadata_and_add_tracks_to_queue(zone, self.queue_ids[zone.ip], track_list)
        orig_update_id = helpers.get_private_queue_update_id(zone, self.queue_ids[zone.ip])
        new_update_id = zone.Queue.ReorderTracks(queueid=self.queue_ids[zone.ip],
                                                 startingindex=1,
                                                 numberoftracks=2,
                                                 insertbefore=5,
                                                 updateid=orig_update_id)

        self.verifyGreaterOrFailCase(new_update_id, orig_update_id,
                                     "UpdateID should increment upon edit")



[docs]    @combinatorial(playable_zone_generator)
    def test_reordertracks_bad_update_id(self, zone):
        """
        Verify reordering tracks in the queue causes the expected new order
        """
        track_list = []
        while len(track_list) < 9:
            track_list.append(random.choice(self.track_source.tracks[0:3]))

        helpers.generate_metadata_and_add_tracks_to_queue(zone, self.queue_ids[zone.ip], track_list)

        try:
            zone.Queue.ReorderTracks(queueid=self.queue_ids[zone.ip],
                                     startingindex=1,
                                     numberoftracks=2,
                                     insertbefore=5,
                                     updateid=123456789)
        except sonos.exception.UPnPError as e:
            self.verifyEqualOrFailCase(
                sonos.client.zone_player.UPNP_ERROR_1028_STALE_OBJECT,
                e.upnp_error,
                "ReorderTracks should succeed if your definition of success is "
                "a UPnP error (1028).")
        else:
            self.fail("ReorderTracks with bad update id should raise a "
                      "UPnP error (1028).")
        
        self.verifyEqualOrFailCase(9,
            helpers.get_number_of_tracks_in_queue(zone, self.queue_ids[zone.ip]),
            "Queue should still have expected number of tracks")
