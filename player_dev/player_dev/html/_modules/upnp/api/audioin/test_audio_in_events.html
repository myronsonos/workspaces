'''
    The AudioIn service has six state variables: AudioInputName, Icon, LeftLineInLevel, LineInConnected,
    Playing, and RightLineInLevel. LineInConnected can be spoofed via player debug hooks described at
    https://confluence.sonos.com/display/PDSW/Platform+Automation+and+Debug+Interface.

    Notes:
        - Playbar does not have AudioIn service. It uses the HTAudio service.
        - Play3 and SUB have no AudioIn service.

    Test Cases:
    Get AudioInAttributes, AudioInputName, Icon. Set them to new values. Read state variables and compare.
    Get LineIn levels, LeftLineInLevel and RightLineInLevel. Set them to new values. Read state variables and compare.
    Spoof LineInConnected. Read and compare LineInConnected state variable. Change value and verify. Un-spoof.

    Playing - Will not automate. A read always times out and Device Spy does not show Playing change. Manually performing
    the steps described below did not cause the Playing variable to change in device spy.

        Ted walked through player code: "Looks like it does get evented (but not with initial subscription, oddly).
        It also only updates when the iPod itself starts playing. So if you try to do anything via the Sonos CR,
        it doesn't update.  If you press the "play" button on iTunes/etc on the iPod, it does trigger. If the iPod is
        in 'accessory connected' mode, it doesn't event but there's also no actual "play/stop" button on the iPod then.

'''

    def test_audioin_attributes_events (self, device):
        """
        Verify the AudioInAttributes events with the set of playable devices.
        """
        self.logger.info("%s: %s" % (TCID, testCaseMap.get(TCID)))
        self.verifyIsNotNoneOrStop(device, "device should be valid")
        self.verifyIsInstanceOrStop(device, SonosZoneComponent, "device should be a SonosZoneComponent")
        initialAttributes = device.AudioIn.GetAudioInputAttributes()
        initialName = initialAttributes.get('CurrentName')
        initialIcon = initialAttributes.get('CurrentIcon')
        testName = "Bill"
        testIcon = "George"
        if initialName == testName:
            testName = "Sam"
        if initialIcon == testIcon:
            testIcon = "Sue"

        self.logger.debug("InitialName: %s; InitialIcon: %s." % (initialName, initialIcon))
        stateName = device.AudioIn.get_state_variable("AudioInputName")
        stateIcon = device.AudioIn.get_state_variable("Icon")
        self.verifyEqualOrFailCase(initialName, stateName, "The AudioInputName state variable should match the GetAudioInputAttributes return value.")
        self.verifyEqualOrFailCase(initialIcon, stateIcon, "The Icon state variable should match the GetAudioInputAttributes return value.")

        result = device.AudioIn.SetAudioInputAttributes(testName, testIcon)
        self.verifyTrueOrFailCase(result, "Set return value should be true.")

        updatedAttributes = device.AudioIn.GetAudioInputAttributes()
        updatedName = updatedAttributes.get('CurrentName')
        updatedIcon = updatedAttributes.get('CurrentIcon')
        stateName = device.AudioIn.get_state_variable("AudioInputName")
        stateIcon = device.AudioIn.get_state_variable("Icon")
        self.verifyEqualOrFailCase(testName, updatedName, "The updated name should be as set.")
        self.verifyEqualOrFailCase(testIcon, updatedIcon, "The updated icon should be as set.")
        self.verifyEqualOrFailCase(testName, stateName, "The state variable name should be as set.")
        self.verifyEqualOrFailCase(testIcon, stateIcon, "The state variable icon should be as set.")

        # Cleanup
        result = device.AudioIn.SetAudioInputAttributes(initialName, initialIcon)

    def test_audioin_linein_level_events (self, device):

        Execute the LineInLevel events with the set of playable devices.

            testLeft = 5
        if initialRight == testRight:

    def test_audioin_linein_connected_events (self, device):
        """
        Execute the LineInConnected test with the set of playable devices.
        """
        # skip the test if the device is virtual, needs access to /proc
        if device.is_virtual():
            self.skip("This test case cannot be executed on emulators")
        self.logger.info("%s: %s" % (TCID, testCaseMap.get(TCID)))
        self.verifyIsNotNoneOrStop(device, "device should be valid")
        self.verifyIsInstanceOrStop(device, SonosZoneComponent, "device should be a SonosZoneComponent")

        initialConnected = device.AudioIn.LineInConnected()
        testConnected = not initialConnected

        self.logger.debug("initialConnected: %s" % initialConnected)
        stateConnected = bool(int(device.AudioIn.get_state_variable("LineInConnected")))
        self.verifyEqualOrFailCase(initialConnected, stateConnected, "The LineInConnected state variable should match the LineInConnected return value.")

        device.cli.set_linein_state('on' if testConnected else 'off')
        wait_until_true(lambda: device.AudioIn.LineInConnected() == testConnected)
        updatedConnected = device.AudioIn.LineInConnected()

        stateConnected = bool(int(device.AudioIn.get_state_variable("LineInConnected")))
        self.verifyEqualOrFailCase(testConnected, updatedConnected, "The Updated Connected value should be as set.")
        self.verifyEqualOrFailCase(testConnected, stateConnected, "The stateConnected should be as set.")

        # Cleanup
        device.cli.set_linein_state('detect')
