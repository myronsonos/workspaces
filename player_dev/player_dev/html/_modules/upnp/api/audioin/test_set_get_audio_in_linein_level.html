[docs]    def test_combinatorial_set_get_audioin_level(self, 
                                     generate_testbed_unique_playback_devices): 
        """
        Execute the AudioInLevel test with the set of playable devices.
        """
        zp = generate_testbed_unique_playback_devices
        if zp.is_virtual():
            self.skip("Skip this test for i386 virtual devices")
        arch = zp.cli.get_arch()

        if arch not in self.LINE_IN_LEVELS:
            self.skip("skipping test audioin level test for {}".format(zp))

        min_level, max_level = self.LINE_IN_LEVELS[arch]
        
        for l,r in zip(range(min_level,max_level,1),
                       range(max_level,min_level,-1)):
            zp.AudioIn.SetLineInLevel(l,r)
            res = zp.AudioIn.GetLineInLevel()
            self.verifyEqualOrFailCase(res['CurrentLeftLineInLevel'],str(l),
                                      "expect left channel line-in level to "
                                      "match expected value")
            self.verifyEqualOrFailCase(res['CurrentRightLineInLevel'],str(r),
                                      "expect right channel line-in level to "
                                      "match expected value")
        zp.AudioIn.SetLineInLevel(0,0)
