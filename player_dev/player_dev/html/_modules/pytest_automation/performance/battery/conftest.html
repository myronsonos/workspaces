[docs]def pytest_addoption(parser):
    """
    Add test specific command line options.

    --playback_volume:
        Used to set the volume of a device during playback test cases.
    --collection_timeout:
        Used to abort battery metric collection after a set time,
        instead of running until the device is dead.
        e.g. pytest -v {file} --collection_timeout 60
        This command tells the collector to stop collecting data after
        60 seconds. This is not a precise setting, as the sampling interval
        and sleep operations may cause it to go beyond this time.
    --idle_discharge_minutes:
        This specifies how many minutes to run the battery drain test
        when the device is Idle (not playing). This value is in minutes,
        with a default of 30.
    """
    parser.addoption("--playback_volume", action="store",
                     dest="playback_volume", type=int, default=50,
                     help="Playback volume to set on device.")

    # Add option for timing out battery drain tests
    parser.addoption("--collection_timeout", action="store",
                     dest="collection_timeout", type=int, default=None,
                     help="Timeout to stop battery metric collection")

    # Add option for the idle discharge time
    parser.addoption("--idle_discharge_minutes", action="store",
                     dest="idle_discharge_minutes", type=int, default=30,
                     help="Minutes to collect battery data when device is idle")

def charge_until_alive():
    """
    This fixture is a charging function that will charge the DUT until it is
    responsive to the automation framework.

    This fixture is used by the `~charged_device` fixture to make sure the
    device is available for testing.

    This fixture may be overridden to support different implementations of the
    charging and is-alive indication. Pattern for override:

    ```
    @pytest.fixture(scope='function')
    def charge_until_alive():
        def charge_function(device):
            # Steps to charge device until it is available for
            # communication
        return charge_function

    Usage:

    def test_some_device(device, charge_until_alive):
        charge_until_alive(device)
        ...

    :return: A Function that takes a single device as an argument. This function
    will charge the device until it is alive.
    """

    def charge(device):
        # Since the device is dead at this point
        # we can't check to see if the device is charging or not.
        device.power.on()
        wait_until_true(lambda: device_is_alive(device),
                        iteration_delay=30,
                        timeout_seconds=CHARGE_UNTIL_ALIVE_SECONDS,
                        reason=("Timed out waiting for '{}' to come back "
                                "alive!"
                                .format(device)))

    return charge

def charged_device(charge_until_alive, unique_portable_device):
    """
    This fixture returns a device that is fully charged for test purposes.

    If the device is discharged, this fixture might take quite some time to
    execute before making the device available for testing.

    :return: A charged portable device
    :rtype: :class:`~sonos.client.internal.SonosZoneComponent`
    """
    pytest.logger.info("Attempting to fully charge '{}'".format(
        unique_portable_device))
    charge_until_alive(unique_portable_device)
    charge_device(unique_portable_device, MAX_CHARGE_PERCENTAGE)

    yield unique_portable_device

    # Make sure all power outlets are powered on during teardown
    unique_portable_device.power.on()

def discharged_device(unique_portable_device, playback_volume,
                      track_uri):
    """
    Parametrized fixture that returns a fully discharged
    :func:`~powered_configured_portable`.
    Plays audio at a volume set by the user to help decrease the amount of
    time it takes to discharge.

    :return: A discharged portable device
    :rtype: :class:`~sonos.client.internal.SonosZoneComponent`
    """

    def _cleanup_device():
        """
        Helper function for making sure
        device is powered on and not playing during failures
        and teardown.
        """
        # Stop playback, remove shares and disable speaker audio from the DUT
        stop_playback(device)
        device.disable_speaker_audio()

        # Make sure all power outlets are powered on during teardown
        device.power.on()

    device = unique_portable_device

    # Enable speaker audio from the DUT
    device.enable_speaker_audio()

    # Set volume and start playback to help discharge faster
    start_playback(device, playback_volume, track_uri)

    # Discharge device
    pytest.logger.info("Attempting to fully discharge '{}'".format(device))
    try:
        discharge_device(device, 0)
    except (AttributeError, TimeoutError):
        # If an error occurred it's possible the device ran
        # out of battery, as expected in this case. If the device
        # is alive it is a real issue.
        if device_is_alive(device):
            # Make sure all power outlets are powered on and no
            # audio is playing because teardown will not run in this case
            _cleanup_device()
            raise
    yield device

    # Stop audio playback and enable power outlets
    _cleanup_device()




[docs]def create_gc_gm_pair(gc, gm):
    """
    Helper function for creating a gc/gm pair of devices.

    :param gc: The device that should act as group coordinator
    :type gc: :obj:`~sonos.client.internal.SonosZoneComponent`
    :param gm: The device that should act as group memeber
    :type gc: :obj:`~sonos.client.internal.SonosZoneComponent`
    :returns: A group coordinator and group member pair
    :rtype: :obj:`tuple` of :obj:`~sonos.client.internal.SonosZoneComponent`
    :raises: :obj:`~sonos.exception.TimeoutError` if the GC/GM pair could
             not be created
    """
    gc.AVTransport.BecomeCoordinatorOfStandaloneGroup()
    wait_until_true(lambda: not gc.AVTransport.is_group_member())
    gm.AVTransport.join_group(gc)
    wait_until_true(lambda: gm.AVTransport.is_group_member())
    return gc, gm

[docs]def start_playback(device, playback_volume, uri):
    """
    Helper function to start repeated audio playback on a device
    and set the device's volume.

    :param device: The device to start playback on
    :type device: :obj:`~sonos.client.internal.SonosZoneComponent`
    :param playback_volume: The volume to use for playback
    :type playback_volume: :obj:`int`
    :param uri: the URI to the track to play
    :type uri: :obj:`str`
    """
    device.RenderingControl.SetEQ("VolumeScalingFactor", 100)
    device.RenderingControl.SetVolume(playback_volume)
    device.RenderingControl.SetMute(False)
    device.AVTransport.use_queue()
    device.AVTransport.clear_queue()
    device.AVTransport.add_to_queue(uri, '')
    device.AVTransport.SetPlayMode('REPEAT_ALL')
    device.AVTransport.Play_and_wait()




[docs]def stop_playback(device):
    """
    Helper function to stop repeated audio playback on a device,
    clear queue and remove any shares added.

    :param device: The device to start playback on
    :type device: :obj:`~sonos.client.internal.SonosZoneComponent`
    """
    device.AVTransport.Stop_and_wait()
    device.AVTransport.clear_play_state()
    device.AVTransport.clear_queue_and_wait()
    device.ContentDirectory.remove_all_shares()

def play_portable_audio(charged_device, playback_volume, track_uri):
    """
    Powers on/unmutes
    :func:`~pytest_automation.performance.battery.conftest.charged_device`
    amps, sets volume and starts playback. Stops playback during teardown and
    powers off amps. The parameter playback_volume allows the writer to specify
    a set of volume levels for playback.

    :return: A charged portable device
    :rtype: :class:`~sonos.client.internal.SonosZoneComponent`

    # Enable speaker audio from the DUT
    device.enable_speaker_audio()

    # Set volume and start playback
    start_playback(device, playback_volume, track_uri)
    yield device

    # Stop playback, remove shares and disable speaker audio from the DUT
    stop_playback(device)
    device.disable_speaker_audio()

[docs]@pytest.fixture(scope="function")
def battery_metrics_file_name(request):
    """
    Sets up a metrics writer to write battery percentage data to a file
    named after the test case being run. Yields a metrics collector that
    will be used to store battery percentage data as it is collected by
    the tests so the metrics writer can write this data.

    TODO: This value does not take parameters into account, so two identical
          tests with different parameters will write to the same file.
          It should be possible to get additional information from the
          request object specific to the test

    :return: A metrics collector object
    :rtype:
        :obj:`~metrics.metrics_battery_percentage_collector.BatteryPercentageMetricsCollector`
    """
    # Get test case and file name info
    test_case = request.node.originalname
    test_file_path = str(request.node.fspath)
    test_file_name_full = test_file_path.split('/')[-1]
    test_file_name = re.sub('.py', '', test_file_name_full)

    # Setup metrics writer and collector and remove
    # "test_" from the test case/file names for nicer reading
    metrics_file_name = '{}-{}'.format(
        re.sub("test_", "", test_file_name),
        re.sub("test_", "", test_case))
    pytest.logger.debug('Writing output to {}'.format(metrics_file_name))
    return metrics_file_name
