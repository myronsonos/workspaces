    def __init__(self, zp, event_test_utility, run_as_fast_as_possible=False):
        """
        Initialize the EventSequence with a specific zoneplayer
        This object is used to execute a set of actions sequentially on the
        associate zoneplayer object

        :param zp: The zoneplayer to associate with this sequence
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`

        :param event_test_utility: The zp's event-test app object
        :type event_test_utility: :obj:`EventTestUtility`

        :param run_as_fast_as_possible: If True, when calling run, all delays
            in the event sequence will be ignored. This is useful for running
            button press events in extremely rapid succession, but will break
            the expected result for actions that require a delay, like
            orientation changes or holding buttons
        :type run_as_fast_as_possible: :obj: `bool`
        """
        self.run_as_fast_as_possible = run_as_fast_as_possible
        self.action_sequence = defaultdict(list)
        self.expected_sources = defaultdict(list)
        self.expected_infos = defaultdict(list)
        self.zp = zp
        self.thresholds = zp.cli.get_temperature_thresholds()
        pytest.logger.info(
            "Temperature thresholds for {} are {}".format(zp, self.thresholds))
        self.event_test_utility = event_test_utility


[docs]    def run(self):
        """
        Iterate through self.action_sequence, executing each action on the
        associated zoneplayer. If `run_as_fast_as_possible=False` this
        method will also verify that the expected events for each action
        are being received by the event-test utility.
        """
        pytest.logger.info(
            "On zp {}, executing event sequence...\n{}".format(self.zp, self))
        expected_infos = []
        expected_sources = []
        for action in self.action_sequence:
            try:
                # For each action there can be multiple functions that
                # must be executed on the device to perform that action
                # and multiple expected events that may result from each
                # function. This executes each player function that is
                # part of the action and waits for the expected events
                # of each function to appear unless running with
                # self.run_as_fast_as_possible == True. See
                # self.actions.cpu_temp_warn_normal for an example of an
                # action with two functions and two sets of expected events -
                # action functions are a list and expected events
                # are stored in a list of lists.
                num_action_functions = len(self.action_sequence[action])
                for action_func in range(num_action_functions):
                    # Gather expected events for the function to be exectued
                    expected_infos.extend(
                        self.expected_infos[action][action_func])
                    expected_sources.extend(
                        self.expected_sources[action][action_func])

                    # Execute the action function
                    self.action_sequence[action][action_func](self.zp)

                    # Wait for the expected event/s to be logged
                    # before executing next function unless running
                    # as fast as possible
                    if not self.run_as_fast_as_possible:
                        self.wait_for_event_sequence(expected_infos,
                                                     expected_sources)
                    else:
                        pytest.logger.debug("Not waiting for events to be "
                                            "received - running as fast "
                                            "as possible")
            except TimeoutError as e:
                raise type(e), ("Timeout while evaluating {}. ".format(action)
                                + str(e))



[docs]    def verify_event_sequence(self, expected_infos, expected_sources):
        """
        Verifies that the sequence of events found in a device's
        event-test utility logs match the expected sequence of events.

        :param expected_infos: the expected infos that should have been logged
                               by the event-test app
        :type expected_infos: :obj:`list` of :obj:`Evt.Info`
        :param expected_sources: the expected sources that should
                                 have been logged by the event-test app
        :type expected_sources: :obj:`list` of :obj:`Evt.Source`
        :returns: Whether the actual event infos or sources do
                  not match the expected infos or sources
        :rtype: :obj:`bool`
        """
        actual_infos, actual_sources = (
            self.event_test_utility.get_logged_events())

        return (actual_infos == expected_infos and
                actual_sources == expected_sources)



[docs]    def wait_for_event_sequence(self, expected_infos, expected_sources):
        """
        This method is used to wait for events to be received
        by the event-test app on a device when an action or sequence of
        actions is executed and can be used to verify all expected
        events were received.

        :param expected_infos: the expected infos that should have been logged
                               by the event-test app
        :type expected_infos: :obj:`list` of :obj:`Evt.Info`
        :param expected_sources: the expected sources that should have
                                 been logged by the event-test app
        :type expected_sources: :obj:`list` of :obj:`Evt.Source`
        :raises: :exc:AssertionError if the actual event infos or sources do
                                     not match the expected infos or sources
        """
        try:
            wait_until_true(
                lambda: self.verify_event_sequence(expected_infos,
                                                   expected_sources),
                iteration_delay=0.5,
                timeout_seconds=MAX_EVENT_DELAY_SECONDS)
        except TimeoutError:
            actual_infos, actual_sources = (
                    self.event_test_utility.get_logged_events())

            # Even though the TimeoutError would fail a test, the asserts
            # here will cause pytest assert introspection to show the diff
            # nicely without us having to print it
            assert actual_infos == expected_infos, (
                "Unexpected event info/s found by the event-test app.")
            assert actual_sources == expected_sources, (
                "Unexpected event source/s found by the event-test app.")



[docs]    def consolidate_expected_events(self):
        """
        This method goes through all actions in the object's action
        sequence and combines all expected events for each action into
        an ordered list. This is useful when comparing all events that
        have been received in the event-test app with all actions that
        have been executed.

        :returns: The expected infos and sources for all actions that
                  have been added to self.action_sequence
        :rtype: (:obj:`list` of :obj:`Evt.Info`,
                 :obj:`list` of :obj:`Evt.Source`)
        """
        expected_infos = []
        expected_sources = []
        for action in self.action_sequence:
            for action_func in range(len(self.action_sequence[action])):
                expected_infos.extend(
                        self.expected_infos[action][action_func])
                expected_sources.extend(
                        self.expected_sources[action][action_func])
        return expected_infos, expected_sources

        :param action: the action to run on the zone player sequentially
        :type action: :obj:`EventSequence.action`
                        or :obj:`list` of :obj:`EventSequence.action`
        """
        # action can also be a list or tuple of EventSequence.action
        # what a wonderful opportunity for recursion
        if isinstance(action, list) or isinstance(action, tuple):
            for act in action:
                self.add_events(act)

        self.action_sequence_string += "{}\n".format(action)

        if action == self.actions.press_button_join:
            # BUTTON_JOIN: PRESSED, RELEASED
            self.expected_sources[action].append(
                [Evt.Source.BUTTON_JOIN.value,
                 Evt.Source.BUTTON_JOIN.value])
            self.expected_infos[action].append([Evt.Info.PRESSED.value,
                                                Evt.Info.RELEASED.value])
            self.action_sequence[action].append(
                 lambda pl: pl.cli.button_press())

        elif action == self.actions.press_button_mode:
            # BUTTON_MODE: PRESSED, RELEASED

                [Evt.Source.BUTTON_MODE.value,

                                                Evt.Info.RELEASED.value])
            self.action_sequence[action].append(
                 lambda pl: pl.cli.press_bluetooth_button())

        elif action == self.actions.press_button_power:
