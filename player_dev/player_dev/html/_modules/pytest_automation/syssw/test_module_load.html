    """
    This is a suite of tests for stress testing module loading and unloading
    on secure boot devices.
    """
    errors = {}  # Dictionary of devices with errors found
    iteration = 1  # Module load/unload iteration count
    power_coordinator_stopped_devices = []

    def _verify_free_mem(self, device, free_mem_before):
        """
        Verifies that the current free memory is greater than or equal to the
        free memory before the test. If not, it checks that the memory usage
        was less than the max allowed.

        :param device: The DUT
        :type device: :class:`~sonos.client.internal.SonosZoneComponent`
        :param free_mem_before: The free mem before the test, in bytes
        :type free_mem_before: :obj:`int`

        """
        one_kilo = 1024
        max_mem_usage_bytes = 100 * one_kilo

        free_mem_after = device.cli.get_free_mem()
        if free_mem_after >= free_mem_before:
            mem_used = 0
        else:
            mem_used = free_mem_before - free_mem_after

        assert mem_used <= max_mem_usage_bytes, ('The memory used should be '
                                                 'less than {}kB. Occurred on '
                                                 'iteration {}.'
                                                 .format(max_mem_usage_bytes /
                                                         one_kilo,
                                                         self.iteration))

    def _verify_no_errors(self, device, module):
        """
        Verifies that there are no errors on the device.

        :param device: The DUT
        :type device: :class:`~sonos.client.internal.SonosZoneComponent
        :param module: the module under test when error occurred.
        :type module: :obj:`str`

        """
        le = DmesgLogErrors(device)
        dmesg_errors = le.find_errors(SearchMethod.CLI)
        if len(dmesg_errors) != 0:
            self.errors[device] = True
            pytest.fail("Errors found on device for module {} during "
                        "iteration {}: {}"
                        .format(module, self.iteration, dmesg_errors))

    def _get_used_by_modules(self, device, module):
        """
        Returns a list of modules that depend on/use a given module.
        This is the "Used by" column from lsmod.

        :param device: The sonos zone component
        :type device: :class:`~sonos.client.internal.SonosZoneComponent
        :param module: The module to get dependents from
        :type module: :obj:str:
        :returns: A list of dependent modules
        :rtype: :obj:`list`
        """
        dependent_modules = []
        regex = "\\r\\n(?:{}) +\d+ +\d+ ((?:\[?\w\]?,?)+)".format(module)
        lsmod_output = device.cli.command('lsmod')

        # If there is a match, create a list of dependent modules
        match_found = re.search(regex, lsmod_output)
        if match_found:
            dependent_modules = match_found.groups()[0].split(',')

            # Older platforms may include brackets around each module, so
            # remove those
            for module in range(len(dependent_modules)):
                dependent_modules[module] = re.sub("\[|\]", "",
                                                   dependent_modules[module])
        return dependent_modules

    def _skip_module_testing(self, device, module_list):
        """
        On some platforms it is necessary to skip
        testing on certain modules due to restrictions around loading/
        unloading them. This helper method will remove modules that
        need to be skipped for each applicable platform along with the
        modules these skipped modules depend on.

        :param device: The sonos zone component
        :type device: :class:`~sonos.client.internal.SonosZoneComponent
        :param modules: The module list to remove modules from
        :type modules: :obj:`list` of :obj:`str`

        """
        # FIXME SWPBL-90789: don't skip lla and dependencies on El Rey/Hideout
        # FIXME SWPBL-102067: don't skip mtk_m4 on El Rey
        # FIXME SWPBL-110827: don't skip lla and dependencies on Bootleg
        # FIXME SWPBL-118750: don't skip sdd and dependencies
        # FIXME SWPBL-97232: don't skip lla and dependencies on IMX6 platforms
        # FIXME SWPBL-122730: don't skip unloading the ccg3 on Alpine

        # Reloading the ccg3_swd driver on S17 will reset the USB ethernet
        # connection and cause the device to lose ethernet, making it
        # unreachable. Skipping ccg3_swd and its dependencies.
        modules_to_skip = {"S6": ['sdd'],
                           "S11": ['sdd', 'lla'],
                           "S13": ['lla', 'sdd'],
                           "S14": ['mtk_m4', 'dspic', 'lla',
                                   'qca_ddfa', 'lp5562_led', 'sdd'],
                           "S15": ['lla'],
                           "S16": ['ice40_fpga', 'lla', 'ice40_led',
                                   'micctl', 'sdd'],
                           "S17": ['ccg3_swd', 'sdd', 'ampctl', 'ccg3'],
                           "S18": ['sdd', 'amperctl'],
                           "S22": ['sdd'],
                           "S23": ['sdd']}

        def safe_remove_from_list(item, item_list):
            """
            Helper function for removing items from a list and ignoring
            ValueErrors. A ValueError in the context of this function and test
            is usually caused by attempting to remove an item that
            has already been removed.
            """
            try:
                item_list.remove(item)
                pytest.logger.info("Skipping testing of '{}' on '{}'"
                                   .format(item, device))
            except ValueError:
                pass

        untouched_module_list = list(module_list)  # maintain a copy
        if device.modelNumber in modules_to_skip:
            # Remove the modules to skip and the modules they depend on
            for module_to_skip in modules_to_skip[device.modelNumber]:
                for module in untouched_module_list:
                    # Go through all modules and if a module is being used by
                    # the module to skip it will need to be skipped as well
                    if module_to_skip in self._get_used_by_modules(device,
                                                                   module):
                        safe_remove_from_list(module, module_list)
                safe_remove_from_list(module_to_skip, module_list)

    def device_modules(self, secure_boot_device, keep_devices_in_error_state):
        """
        This fixture will perform setup and teardown needed for the
        'test_module_load' test. This will yield a tuple containing the
        device and device modules to test on.

        :param secure_boot_device: The sonos zone component
        :type secure_boot_device:
            :obj:`~sonos.client.internal.SonosZoneComponent`
        :param keep_devices_in_error_state: Whether or not to reboot
                                            devices that have errors in dmesg
        :type keep_devices_in_error_state: :obj:`bool`
        :returns: A tuple containing the device and its modules to test
        :rtype: (:class:`~sonos.client.internal.SonosZoneComponent,
                 :obj:`list` of :obj:`str`)

        Setup:
        1. Kill anacapad
        2. Unload all modules in the correct order
        3. Set cpu scaling devices to high speed - needed by the PSoC

        Teardown:
        1. Restart anacapa

        """
        # Setup
        device = secure_boot_device
        self.errors[device] = False
        if device.cli.is_anacapa_running():
            device.cli.stop_anacapa_nicely()
        # Stop sonospowercoordinator for power coordinator
        # dependencies and errors
        if device.cli.has_power_coordinator():
            device.cli.stop_power_coordinator()
            self.power_coordinator_stopped_devices.append(device)
        device.cli.set_cpu_frequency_hz(device.high_cpu_speed)

        # Get and reverse module list to unload in proper order
        modules = KERNEL_MODULES[device.modelNumber].keys()
        modules.reverse()

        # Some modules cannot be removed or are not actually loaded
        # on different platforms. Skip testing on those modules
        # and modules they depend on.
        self._skip_module_testing(device, modules)

        try:
            for module in modules:
                device.cli.remove_module(module)
                self._verify_no_errors(device, module)
        except (TimeoutError, NameError) as e:
            # Reboot error devices unless running with
            # '--keep_devices_in_error_state'
            if not keep_devices_in_error_state:
                device.cli.reboot()
            raise e

        device.cli.clear_dmesg()

        # Reverse module list again so it is ordered correctly
        # and yield both the device and module list
        modules.reverse()
        yield device, modules

        # Teardown
        if not self.errors[device]:
            if device in self.power_coordinator_stopped_devices:
                device.cli.start_power_coordinator()
            if not device.cli.is_anacapa_running():
                device.cli.start_anacapa()
        elif not keep_devices_in_error_state:
            # Reboot error devices unless running with
            # '--keep_devices_in_error_state'
            device.cli.reboot()

        """
        Verifies that all expected module files are populated in /modules.

        :param secure_boot_device: The sonos zone component
        :type secure_boot_device:
            :obj:`~sonos.client.internal.SonosZoneComponent`

        1. ls /modules
        2. Parse the list of modules in /modules
        3. Compare the list to the modules defined in KERNEL_MODULES
        4. Verify all expected module files are populated

        """
        device = secure_boot_device
        modules = device.cli.listdir("/modules")
        expected_modules = KERNEL_MODULES[device.modelNumber].values()
        assert set(expected_modules) == set(modules), ('All expected modules '
                                                       'should be in /modules '
                                                       'on {}'.format(device))



[docs]    def test_module_load(self, device_modules, iterations):
        """
        Performs a module load/unload stress test and looks for memory leaks.
        Run using --iterations=<num_iterations (int)> - see ./conftest.py
        for more info.

        :param secure_boot_device: The sonos zone component
        :type secure_boot_device:
            :obj:`~sonos.client.internal.SonosZoneComponent`
        :param iterations: The number of iterations to load/unload
        :type iterations: :obj:`int`

        Test Steps:
        1. Load the module under test
        2. Verify the module under test is loaded
        3. Check dmesg for errors
        4. Check /proc/meminfo to get free memory
        5. Clear dmesg
        6. Unload the module under test
        7. Verify the module under test has been unloaded
        8. Check dmesg for errors
        9. Load the module under test
        10. Verify the module under test has been loaded
        11. Check dmesg for errors
        12. Get free memory, compare to step 4 and verify no leaks
        13. Repeat steps 1 - 12 for a number of iterations
        14. Repeat step 13 for all modules

        """
        device, modules = device_modules
        try:
            for module in modules:
                pytest.logger.info("Starting stress test of module: {}"
                                   .format(module))

                # Insert module under test
                module_path = ("/modules/" +
                               KERNEL_MODULES[device.modelNumber][module])
                device.cli.insert_module(module_path)
                self._verify_no_errors(device, module)

                # Repeat stress test for the given number of iterations
                for i in range(1, iterations+1):
                    self.iteration = i
                    device.cli.clear_dmesg()
                    free_mem_before = device.cli.get_free_mem()

                    # Verify module unload
                    device.cli.remove_module(module)
                    if module == "qca_ddfa":
                        # qca_ddfa requires amperctl to be reloaded so amperctl
                        # is in the correct state to load qca_ddfa module
                        # without causing an AMPCTL assertion error message
                        # "amp status not in PWRDOWN state, instead =2"
                        device.cli.remove_module("amperctl")
                    self._verify_no_errors(device, module)

                    # Verify module load
                    if module == "qca_ddfa":
                        # qca_ddfa requires amperctl to be reloaded so amperctl
                        # is in the correct state to load qca_ddfa module
                        # without causing an AMPCTL assertion error message
                        # "amp status not in PWRDOWN state, instead =2"
                        device.cli.insert_module("/modules/amperctl.ko")
                    device.cli.insert_module(module_path)
                    self._verify_no_errors(device, module)

                    # Verify no memory leaks
                    self._verify_free_mem(device, free_mem_before)
        except (TimeoutError, NameError) as e:
            self.errors[device] = True
            raise type(e), (str(e) + " - occurred on iteration {}"
                            .format(self.iteration))
