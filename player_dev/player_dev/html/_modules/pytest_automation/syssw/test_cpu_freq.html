"""
Tests around CPU Frequency Scaling in devices that support it (imx6 and newer)
Checks the CPU frequency on a player's SoC depending on different playback
and mic states.

Questions/concerns: SYSSW QA
"""

import pytest
import time
from pytest_automation.syssw.common import (wait_for_state,
                                            is_voice_enabled,
                                            set_play_buzzer_or_stop)
from sonos.client.zone_player import BlackboxMask, DEVICES_WITH_BUILT_IN_MICS


DIAG_LEVELS = [("amp", 5), ("power_ctl", 5), ("dapst", 3), ("rc_impl", 3),
               ("buttons", 5), ("upnp", 5), ("avt", 5)]
# Timeout timers are defined defined
# in all/oc/zone/zoneplayer/ampManager.cxx as DEFAULT_POWER_OFF_TIMER_SEC
DEFAULT_AMP_OFF = 120
FAST_AMP_OFF = 2
# Since the transition from low to high frequency is very fast, we use a low
# constant value for the timeout.
RISING_TIME = 2
BLACKBOX_AMPCTL_MASK = (BlackboxMask.AMPCTL | BlackboxMask.ERR |
                        BlackboxMask.DEBUG | BlackboxMask.WARNING |
                        BlackboxMask.INFO)

pytestmark = pytest.mark.usefixtures("disable_speaker_audio")

def device_in_testbed(cpu_scaling_device):
    """
    Yields a cpu scaling capable device on the testbed ready for testing.

    Setup:
        1) Setup blackbox mask
        2) Unmute device

    Teardown:
        1) Clear queue
        2) Clear play state
        3) Unmute
    """
    zp = cpu_scaling_device
    zp.cli.set_blackbox_mask(BLACKBOX_AMPCTL_MASK)
    zp.RenderingControl.SetMute(False)
    yield zp
    zp.AVTransport.clear_queue_and_wait()
    zp.AVTransport.clear_play_state()
    zp.RenderingControl.SetMute(False)

def voice_enabled(cpu_scaling_device):
    """
    If a device is voice capable and have the voice account setup,
    unmute the mic. Otherwise, keep the mic muted or it is a device without
    mic.
    """
    zp = cpu_scaling_device
    voice_enabled = is_voice_enabled(zp)
    if zp.modelNumber in DEVICES_WITH_BUILT_IN_MICS:
        if voice_enabled:
            zp.cli.unmute_mic()
    yield voice_enabled




[docs]@pytest.fixture(scope="module")
def wait_time(cpu_scaling_device):
    """
    wait_time to be passed to wait_until_true for each device.
    This wait time is the upper bound of approximately how long it would
    take for the amp power off which is the same time CPU freq would go from
    high to idle after playback stop.
    """
    zp = cpu_scaling_device
    # timeout timer is defined in all/oc/zone/zoneplayer/ampManager.cxx
    # as DEFAULT_POWER_OFF_TIMER_SEC
    wait_time = (FAST_AMP_OFF if zp.is_s6() or
                 zp.is_s11() or zp.is_s14() or
                 zp.is_s15() else DEFAULT_AMP_OFF)
    # Add small buffer to threshold
    # Without this buffer, we will see a lot of timeout errors.
    # TODO: Reduce this buffer after SWPBL-86458
    wait_time += 30
    yield wait_time

def iteration_delay(wait_time):
    """
    iteration delay based on wait_time.
    TODO: The delay will be reduced when the wait_time
    buffer is removed.
    """
    iteration_delay = 1 if wait_time <= 10 else 20
    yield iteration_delay

[docs]def test_idle_after_reboot(device_in_testbed, voice_enabled,
                           wait_time, iteration_delay):
    """
    Verify if a cpu scaling device enters idle speed state after reboot.

    Test Steps:
        1. Reboot
        2. Wait for anacapa to start
        3. Set blackbox mask
        4. Unmute device
        5. Stop playback
        6. Check if the cpu enters idle speed state
           Note: If it is a voice capable device with voice service setup,
           the cpu speed will not drop to idle. Timeouts are expected.

    """
    zp = device_in_testbed
    zp.reboot()
    wait_for_state(
        lambda: zp.cli.is_anacapa_running(), zp, timeout=3)
    zp.disable_speaker_audio()
    zp.cli.set_blackbox_mask(BLACKBOX_AMPCTL_MASK)
    for diag, level in DIAG_LEVELS:
        zp.diag.set_diag_level(diag, level)
    zp.RenderingControl.SetMute(False)
    set_play_buzzer_or_stop(zp, False)
    wait_for_state(
        lambda: zp.idle_cpu_speed == zp.cli.get_cpu_frequency_hz(),
        zp, timeout=wait_time, delay=iteration_delay,
        expect_timeout=voice_enabled)

[docs]def test_high_freq_after_stop_playing(device_in_testbed, wait_time,
                                      iteration_delay, override_diag_levels):
    """
    Verify if a cpu scaling device remains in high speed state
    for a short while after playback is just stopped.

    Test Steps:
        1. Start playback
        2. Stop playback
        3. Check if the cpu still hangs in high speed state
           shortly after playback stops

    # A time buffer between play and stop buzzer to avoid timeout errors.

    set_play_buzzer_or_stop(zp, False)
    wait_for_state(
        lambda: zp.high_cpu_speed == zp.cli.get_cpu_frequency_hz(),
        zp, timeout=wait_time, delay=iteration_delay)
