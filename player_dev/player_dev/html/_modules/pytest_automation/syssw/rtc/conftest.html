"""
RTC test command line options.
"""
import pytest

# Default long duration RTC drift test run time
DEFAULT_RTC_DRIFT_DURATION_SEC = 5 * 60 * 60

# Default long duration suspend test run time
DEFAULT_SUSPEND_DURATION_SEC = 2 * 60 * 60



[docs]def pytest_addoption(parser):
    """
    Add test specific command line options.

    --num_suspend_cycles
        The number of suspend/resume cycles to run for test cases
        that verify RTC behavior across multiple suspends/resumes

    --run_duration
        The time in seconds to run the RTC drift test

    --suspend_duration
        The time in seconds to suspend for for lond duration suspend testing
    """
    parser.addoption("--num_suspend_cycles", action="store",
                     dest="num_suspend_cycles", type=int, default=10,
                     help="Number of suspend/resume cycles")
    parser.addoption("--rtc_drift_duration", action="store",
                     dest="rtc_drift_duration", type=int,
                     default=DEFAULT_RTC_DRIFT_DURATION_SEC,
                     help="The time in seconds to run the RTC drift test for")
    parser.addoption("--suspend_duration", action="store",
                     dest="suspend_duration", type=int,
                     default=DEFAULT_SUSPEND_DURATION_SEC,
                     help=("The time in seconds to suspend for "
                           "for long duration suspend testing"))
