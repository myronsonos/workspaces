        """
        Returns top-most element of the queue without removing it from the queue
        :return: tuple
        """
        if not self.is_empty():
            return self.queue[0]
        else:
            raise ValueError("Queue is empty")

        """
        Returns top-most element of the queue and removes it from the queue
        :return: tuple
        """
        if not self.is_empty():
            return self.queue.popleft()

[docs]def verify_key_cache_show(zp, resp, expected_number_of_entries, expected_key_cache=None):
    """
    Verify expected_number_of_entries and expected_key_cache against REST response returned from testpoint
    :param zp:
    :param resp:
    :param expected_number_of_entries:
    :param expected_key_cache: Only keys are verified
    :return:
    """
    assert resp.status_code == 200, \
        "URL request to {}, resp code: {}, resp text: {}".format(zp.friendly_name,
                                                                 resp.status_code, resp.text)

    resp_number_of_entries = key_cache_helper.get_number_of_entries_from_key_cache(resp.text)
    assert expected_number_of_entries == resp_number_of_entries, \
        "Expected number of entries in key cache are {} but got {}".format(expected_number_of_entries,
                                                                           resp_number_of_entries)

    if expected_key_cache is not None:
        resp_key_cache = key_cache_helper.get_key_value_dict_from_key_cache(resp.text)
        resp_keys = resp_key_cache.keys()
        expected_keys = expected_key_cache.keys()
        assert expected_keys.sort() == resp_keys.sort(), \
            "Expected key cache is {} but got {}".format(expected_keys.sort(), resp_keys.sort())

def _verify_topology(resp, expected_zps, is_bonded, return_bool=False):
    """
    Verify raw_udn of ZPs found in resp(Groups Event)
    :param resp:
    :param expected_zps:
    :param is_bonded:
    :param return_bool:
    :return:
    """
    expected_devices = set([zp.raw_udn for zp in expected_zps])
    devices_found = set()

    for player in resp.body["players"]:
        devices_found.add(player["id"])
        # if bonded add devices found from GroupsEvent->players(master_zp)->deviceIds
        if is_bonded and player["id"] == expected_zps[0].raw_udn:
            for zp in player["deviceIds"]:
                devices_found.add(zp)

    if return_bool:
        return expected_devices.issubset(devices_found)
    else:
        return devices_found



[docs]def wait_for_consistent_topology_on_multiple_players(expected_zps, websocket_connections,
                                                     is_bonded=False, timeout=15):
    """
    Given a set of ZPs, verify all ZPs in the test are added in topology of the every ZP
    :param expected_zps:
    :param websocket_connections:
    :param is_bonded:
    :param timeout:
    :return:
    """
    for a_ws_client, a_zp in zip(websocket_connections, expected_zps):
        # Subscribe to Groups namespace and verify topology is stabilized
        a_ws_client.groups.subscribe()
        a_ws_client.waitForEvent(eventFilter=lambda e: (_verify_topology(e, expected_zps, is_bonded,
                                                                         return_bool=True)),
                                 eventType=GroupsEvent, timeout=timeout)




[docs]def verify_group_gc_and_members(ws_client, gc, expected_players_in_group):
    """
    Common groups event verifier. Checks for
    1) gc - group's gc
    2) expected_players_in_group - ZPs acting as group members
    :param ws_client:
    :param gc:
    :param expected_players_in_group:
    :return:
    """
    # Try up-to 3 times for matching playback state for respective zp
    for _ in range(3):
        try:
            ws_client.groups.getGroups()
            getGroups_resp = ws_client.waitForEvent(eventType=GroupsEvent)
            for group in getGroups_resp.body["groups"]:
                if group["coordinatorId"] == gc.raw_udn:
                    # If we found the group with the desired gc, verify the number of players in the group
                    assert len(group["playerIds"]) == len(expected_players_in_group),\
                        "Number of players in the GC {} group should be {}, got {}".format(gc.raw_udn,
                                                                                           len(expected_players_in_group),
                                                                                           len(group["playerIds"]))
                    # Verify that the playerIds in the group are as expected
                    for zp in expected_players_in_group:
                        assert zp.raw_udn in group["playerIds"],\
                            "PlayerId {} should be in GC {} group".format(zp.raw_udn, gc.raw_udn)
                    return
        except AssertionError:
            pass
    else:
        # Assert fail, no group found with the desired GC
        assert False, "Did not find playerId {} as a group GC".format(gc.raw_udn)




[docs]def verify_playback_state_from_groups_event(ws_client, zp_list, expected_playback_state_list):
    """
    Groups event playback state verifier. Checks for
    1) zp_list - list of SonosZoneComponent objects
    2) expected_playback_state_list - expected playback state list for corresponding zps
    :param ws_client: (MuseClientProtocol) defaulted to None -> ws_manager.muse
    :param zp_list: (list or SonosZoneComponent)
    :param expected_playback_state_list: (list or str)
    :return:
    """

    if type(zp_list) is not list:
        zp_list = [zp_list]

    if type(expected_playback_state_list) is not list:
        expected_playback_state_list = [expected_playback_state_list]

    assert len(zp_list) == len(expected_playback_state_list), \
        "Number of zps: {} != expected playback stateExpected length {}".format(len(zp_list),
                                                                                len(expected_playback_state_list))

    playback_matched = []
    # Try up-to 3 times for matching playback state for respective zp
    for _ in range(3):
        playback_matched = []
        ws_client.groups.getGroups()
        getGroups_resp = ws_client.waitForEvent(eventType=GroupsEvent)
        for zp, expected_playback_state in zip(zp_list, expected_playback_state_list):
            for group in getGroups_resp.body["groups"]:
                if group["coordinatorId"] == zp.raw_udn or zp.raw_udn in group["playerIds"]:
                    # If we found the group with the desired zp, verify the playback state
                    playback_matched.append(group["playbackState"] == expected_playback_state)
                    break
            else:
                # Desired zp not found in groups response
                playback_matched.append(False)

        if all(playback_matched) and len(playback_matched) == len(zp_list):
            return
    else:
        unmatched_zp = []
        unmatched_playback_state = []
        for match_status, zp, expected_playback_state in zip(playback_matched, zp_list, expected_playback_state_list):
            if not match_status:
                unmatched_zp.append(zp.friendly_name)
                unmatched_playback_state.append(expected_playback_state)

        assert len(unmatched_zp) == 0, \
            "Could not find ZP(s):{} with playback state: {} in 3 tries".format(unmatched_zp, unmatched_playback_state)
