def streamUrl(mock_cq_control, set_hosts_override, proto, track, library_path):
    """
    Start a mock CQ server to generate a HTTP/HTTPS URL to host a track
    :param mock_cq_control:
    :param set_hosts_override:
    :param proto: http or https
    :param track: track # like 0,1,2... to host
    :param library_path: directory to host tracks from
    :return: the stream URL
    :rtype: :obj:`str`
    """
    streamUrl = ''
    # Mock a CQ Server
    cq = mock_cq_control.start_cloud_queue_server(proto=proto, library_path=library_path)

    # Update streamUrl
    if proto == 'https':
        streamUrl += mock_cq_control.CQ_DOMAIN_URI

    elif proto == 'http':
        streamUrl += mock_cq_control.CQ_URI

    # Update streamUrl with path to the track
    if track:
        streamUrl = "{}/{}/{}".format(streamUrl, 'track/data', track)
    else:
        streamUrl = None

    yield streamUrl

    # Close/cleanup the mock cq server
    # Increment the next cq server port for /timePlayed reporting purposes
    mock_cq_control.host_port = mock_cq_control.host_port + 1
    mock_cq_control.stop_cloud_queue_server(cq)
    mock_cq_control.delete_all_logs()

    :param mock_cq_control:

        # Restore /etc/hosts

                                                                                                len(authorization))

                "httpAuthorization from medialog should be {} but got {}".format(httpAuthorization,
                                                                                 authorization)

def setup_verify_volume_unchanged_teardown(ws_client, vol, muted):
    """
    Set Volume to vol and Mute to muted before the test using
    the :func:`~pytest_automation.conftest.ws_manager` and
    verify Volume is vol and Mute is muted after the test then
    set volume to 0 and Mute to True.
    :param ws_client:
    :param vol:
    :param muted:
    :return:
    """
    # Setup - Set Volume to vol and Mute to muted before the test
    ws_client.playerVolume.setVolume(volume=vol, muted=muted)
    ws_client.waitForEvent(lambda e: e.header["response"] == "setVolume",
                           eventType=MuseSuccessEvent)

    # Subscribe to playerVolume event and verify Volume is vol and Mute is muted before the test
    ws_client.playerVolume.subscribe()
    ws_client.waitForEvent(lambda e: (e.body["volume"] == vol and e.body["muted"] == muted),
                           eventType=PlayerVolumeEvent)

    yield

    # Verify Volume is vol and Mute is muted after the test
    ws_client.playerVolume.getVolume()
    ws_client.waitForEvent(lambda e: (e.body["volume"] == vol and e.body["muted"] == muted),
                           eventType=PlayerVolumeEvent)

    # Teardown - Set Volume to 0 and Mute to True
    ws_client.playerVolume.setVolume(volume=vol, muted=muted)
    ws_client.waitForEvent(lambda e: e.header["response"] == "setVolume",
                           eventType=MuseSuccessEvent)

def wait_for_consistent_topology_on_player(ws_client, function_dut, subscribe_groups):
    """
    On a ZP subscribed to Groups namespace, retrieve the last Groups event then
    confirm ws_manager.zp shows up in players array
    :param ws_client:
    :param function_dut:
    :param subscribe_groups:
    :return:
    """

    if not isinstance(ws_client.allEvents[-1], GroupsEvent):
        ws_client.groups.getGroups()
        resp = ws_client.waitForEvent(eventType=GroupsEvent)

        'Player {} not found GroupsEvent players array: {}'.format(function_dut.friendly_name,

def cloudqueue_metrics(mock_cq_server, dut_manager):
    """
    Sets the eventmetrics URL on a ZP to send reports to mock Cloudqueue server
    :param mock_cq_server:
    :param dut_manager:
    :return:
    """
    # Set the metrics URL
    metrics_override = {EVENTMETRICS_URL_OVERRIDE_NAME: '{}/usageMetrics'.format(mock_cq_server.CQ_URI)}
    bounce_anacapa = False

    if not dut_manager.zp.SystemProperties.verifySysSettingStrings(metrics_override):
        dut_manager.zp.SystemProperties.SetString(EVENTMETRICS_URL_OVERRIDE_NAME,
                                                  metrics_override[EVENTMETRICS_URL_OVERRIDE_NAME])
        bounce_anacapa = True

    if dut_manager.zp.cli.does_metrics_config_file_exist():
        dut_manager.zp.cli.remove_metrics_config_file()
        bounce_anacapa = True

    if dut_manager.zp.cli.does_metrics_config_file_meta_exist():
        dut_manager.zp.cli.remove_metrics_config_file_meta()
        bounce_anacapa = True

    if bounce_anacapa:
        dut_manager.zp.bounce_anacapa_and_wait()
        wait_until_true(predicate=lambda: dut_manager.zp.muse_rest.getGroupId() is not None,
                        timeout_seconds=10, iteration_delay=1,
                        reason="Waiting for <{}> to get a groupId".format(dut_manager.zp.friendly_name))

    # Clear queues, play state and metrics
    if not dut_manager.zp.AVTransport.is_queue_empty():
        dut_manager.zp.AVTransport.clear_queue_and_wait()
    if dut_manager.zp.AVTransport.is_playing():
        dut_manager.zp.AVTransport.clear_play_state()

    player_common.trigger_job(dut_manager.zp, EVENT_UPLOAD)

    yield

    # Remove the metrics URL
    for zp in dut_manager.test_devices:
        if dut_manager.zp.cli.does_metrics_config_file_exist():
            dut_manager.zp.cli.remove_metrics_config_file()

        if dut_manager.zp.cli.does_metrics_config_file_meta_exist():
            dut_manager.zp.cli.remove_metrics_config_file_meta()

        # Remove the metrics URL
        try:
            zp.SystemProperties.Remove(EVENTMETRICS_URL_OVERRIDE_NAME)
        except:
            pass
        else:
            dut_manager.zp.bounce_anacapa_and_wait()
            wait_until_true(predicate=lambda: dut_manager.zp.muse_rest.getGroupId() is not None,
                            timeout_seconds=10, iteration_delay=1,
                            reason="Waiting for <{}> to get a groupId".format(dut_manager.zp.friendly_name))
