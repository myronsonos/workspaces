"""
[SWPBL-113352]
The purpose of these tests is to verify expected grouping behaviors of portable
players. These behaviors are tested against grouping of only portables as well
as grouping with a mix of portables and non-portable players
"""
import pytest
from sleep import guaranteed_sleep
from grouping_helpers import (
                        verify_playback,
                        verify_no_playback,
                        verify_group_members_and_coordinator,
                        verify_not_in_group,
                        ungroup_player,
                        group_from_zp)

pytestmark = pytest.mark.usefixtures('disable_speaker_audio',
                                     'clear_playstate',
                                     'ungroup_players')

[docs]def group_and_verify(gc, gms):
    """
    Shared test method for grouping one player to another and verifying
    that it was successful

    :param gc: the player we expect to be group coordinator
    :param gms: the players we expect to be group members
    """
    group_from_zp(gc, gms[0])
    verify_group_members_and_coordinator(gc, gms)




[docs]def group_standalone_stereo_and_verify(standalone, left, right, group_from_stereo):
    """
    Shared test method for grouping a standalone and stereo pair together and
    verifying it was successful

    :param standalone: a standalone player
    :param left: the left player in a stereo pair
    :param right: the right player in a stereo pair
    :param group_from_stereo: a Boolean if we should group from the stereo pair
    """
    if group_from_stereo:
        group_and_verify(left, [standalone])
    else:
        group_and_verify(standalone, [left, right])




[docs]def group_from_stereo_and_verify_playback(standalone, left, right):
    """
    Shared test method for grouping a standalone and stereo pair together and
    verifying it was successful and that the standalone starts playing

    :param standalone: a standalone player we expect to be a gm
    :param left: the left player in a stereo pair we expect to be gc
    :param right: the right player in a stereo pair
    """
    group_and_verify(left, [standalone])
    verify_playback([standalone, left])




[docs]def group_from_standalone_and_verify_playback(standalone, left, right):
    """
    Shared test method for grouping a standalone and stereo pair together and
    verifying it was successful and that the stereo pair starts playing back

    :param standalone: a standalone player we expect to be gc
    :param left: the left player in a stereo pair we expect to be a gm
    :param right: the right player in a stereo pair we expect to be a gm
    """
    group_and_verify(standalone, [left, right])
    verify_playback([standalone, left])




[docs]def ungroup_standalone_and_verify_playback_stops(standalone, left, right):
    """
    Shared test method for ungrouping a standalone player from it's group and
    verifying it was successful and stops playing

    :param standalone: a standalone player which will leave
    :param left: the left player in a stereo pair
    :param right: the right player in a stereo pair
    """
    ungroup_player(standalone)
    verify_not_in_group([standalone], left)
    verify_no_playback([standalone])
