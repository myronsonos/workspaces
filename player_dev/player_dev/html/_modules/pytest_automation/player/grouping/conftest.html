    """
    Returns a pair of portable devices capable of stereo pairing to each other

    :returns: a pair of portable stereo paired devices
    :rtype: :obj:`list` of :class:`~sonos.client.internal.SonosZoneComponent`
    """
    left = None
    right = None
    for L, R in all_stereo_pairable_devices:
        if L in all_portable_devices:
            left = L
            right = R
            break
    if left is None:
        pytest.skip("Test requires at least one portbale stereo pair of devices, "
                    "none found.")
    else:
        pytest.logger.info("Cleaning up any bonded configuration before the test")
        [zp.DeviceProperties.cleanup_bonded_configuration() for zp in [left, right]]

        pytest.logger.info("Configuring Stereo Pair: {} and {}"
                        .format(left, right))
        left.DeviceProperties.add_stereo_pair(right)

        yield [left, right]

        pytest.logger.info('Separating Stereo Pair after the test')
        left.DeviceProperties.cleanup_bonded_configuration()

def a_portable_power_managed_device_stereo_pair(all_stereo_pairable_devices):
    """
    Returns a portable device stereo pair, with one device in the pair
    managed via programmable power switch.

    :return: A portable device stereo pair, with one device managed via
    programmable power switch
    :rtype: :obj:`list` of :class:`~sonos.client.internal.SonosZoneComponent`
    """
    pairs = []
    for left, right in all_stereo_pairable_devices:
        if left.modelNumber in PORTABLE_DEVICES:
            if left.power or right.power:
                pairs.append((left, right))
    if not pairs:
        pytest.skip("Need at least 1 portable, stereo pairable device set, "
                    "with one player managed via programmable power switch")
    yield pairs[0]

[docs]def start_playback_from_queue(zp, webserver, track):
    """
    Start playback on zone player from queue.

    :param zp: Zone player
    :type: zp: :class:`~sonos.client.internal.SonosZoneComponent`
    :param webserver: Webserver fixture
    :param track: Track file to play
    :rtype: :obj:`str`
    """
    zp.AVTransport.clear_queue()
    zp.AVTransport.add_to_queue('http://{}/{}'.format(webserver.host(), track))
    zp.AVTransport.use_queue()
    zp.AVTransport.Play_and_wait()




[docs]def verify_playback(zones):
    """
    Verify all given players are playing. If not, assert and log player's AVT
    state.

    :param zones: Zone players whose playing state is to be verified
    :type: zones :obj:`list` of :class:`~sonos.client.internal.SonosZoneComponent`
    """
    for zp in zones:
        pytest.wait_for_assert(lambda: zp.AVTransport.is_playing(),
                               iteration_delay=1, timeout_seconds=15,
                               msg="Expected {} to be playing back, "
                                   "current AVT state, {}".
                               format(zp, zp.AVTransport.get_current_transport_state()))
