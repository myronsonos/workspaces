def pair_player_to_bluetooth_adapter(player, bluetooth_adapter):
    """
    Pair the given Bluetooth mode player to the given Bluetooth adapter

    :param player: zone player in bluetooth mode (normal or forced)
    :param bluetooth_adapter: a BluetoothAdapter object
    """
    bt_mac = player.bt_mac
    # if the device mac is not already in the list of scanned devices,
    # we need to scan for it.
    if not bluetooth_adapter.is_device_in_scan(bt_mac):
        player.cli.longpress_bluetooth_button()
        bluetooth_adapter.discover_devices(timeout_seconds=BT_SCAN_SECONDS)
    # we expect that the mac is in the list of devices that we've scanned for
    assert bluetooth_adapter.is_device_in_scan(bt_mac), \
        "Expected to discover bluetooth mac {} in device scan".format(bt_mac)
    # if the device is not already paired with the adapter, pair it
    if not bluetooth_adapter.is_device_paired(bt_mac):
        bluetooth_adapter.pair_with_device(bt_mac)

    """
    Unpair the given Bluetooth mode player from the given Bluetooth adapter

    :param player: zone player in bluetooth mode (normal or forced)
    :param bluetooth_adapter: a BluetoothAdapter object
    """
    bt_mac = player.bt_mac

    # if the device mac is not already in the list of scanned devices,
    # we're done
    if bluetooth_adapter.is_device_in_scan(bt_mac):
        pytest.logger.info("Unpairing BT device from test runner")
        bluetooth_adapter.unpair_from_device(bt_mac)
        pytest.wait_for_assert(
            lambda: not bluetooth_adapter.is_device_paired(bt_mac),
            timeout_seconds=5,
            msg="Expected device to unpair",
            iteration_delay=0.5)




[docs]def put_wifi_player_in_bluetooth_mode(zp, extract_detailed_logs=False):
    """
    Put a player that's in WiFi mode into Bluetooth mode

    :param zp: zone player in WiFi mode
    :param bool extract_detailed_logs: whether or not to extract the
        timestamps of all detailed logs
    :return: the transition time tracker
    :rtype: :obj:`BluetoothWiFiTransitionTimeTracker`
    """
    # put into bluetooth mode and wait for the relevant process to come up
    pytest.logger.info("Forcing serial cli")
    zp.force_serial_cli = True

    # clear the logs to start so that parsing is easier
    archive_and_clear_logs(zp, "pre_transition_wifi_to_bt")

    pytest.logger.info("Pressing Bluetooth button on player WiFi -> BT")
    zp.cli.press_bluetooth_button()

    bwttt = BluetoothWiFiTransitionTimeTracker(zp)
    bwttt.wait_for_all_wifi_to_bluetooth_messages()

    zp.cli.bt_mgr.wait_for_btc_bt_attach(
        iteration_delay=1,
        timeout_seconds=MAX_SECONDS_TRANSITION_TO_BLUETOOTH_MODE,
        reason="Waiting for btc_btattach to come up")

    if extract_detailed_logs:
        bwttt.extract_wifi_to_bluetooth_logs()
        pytest.logger.info("Waited {}s to transition to BT mode".format(
            bwttt.get_wifi_to_bluetooth_transition_time_seconds()))
    return bwttt




[docs]def put_bluetooth_player_in_wifi_mode(zp, ap_connected=True,
                                      extract_detailed_logs=False):
    """
    Put a player that's in Bluetooth mode into WiFi mode

    :param zp: zone player in Bluetooth mode
    :param bool ap_connected: whether or not an AP is connected to the player
    :param bool extract_detailed_logs: whether or not to extract the
        timestamps of all detailed logs
    :return: the transition time tracker
    :rtype: :obj:`BluetoothWiFiTransitionTimeTracker`
    """
    if zp.cli.bt_mgr.is_bt_attach_btc_running():
        # clear the logs before the transition to make it easier to parse
        archive_and_clear_logs(zp, "pre_transition_bt_to_wifi")
        bwttt = BluetoothWiFiTransitionTimeTracker(zp)

        # back to wifi mode and wait for it to go back to station mode
        pytest.logger.info("Pressing Bluetooth button on player BT -> WiFi")

        zp.cli.press_bluetooth_button()

        # wait for all appropriate logs
        bwttt.wait_for_all_bluetooth_to_wifi_messages()
        if extract_detailed_logs:
            bwttt.extract_bluetooth_to_wifi_logs()
            pytest.logger.info("Waited {}s to transition to WiFi mode".format(
                bwttt.get_bluetooth_to_wifi_transition_time_seconds()))

        zp.force_serial_cli = False

        pytest.logger.info("Waiting for player to come up in WiFi mode")

        pytest.subnet.wait_for_all_environment_devices(
            timeout_seconds=MAX_SECONDS_TRANSITION_TO_WIFI_MODE)
        zp.wait_for_station_mode(use_cli=not ap_connected)

        return bwttt

[docs]def suspend_player_for_duration(zp, seconds_to_sleep, ble_adapter,
                                wait_for_response=True):
    """
    Suspend player for N seconds

    :param zp: Zone Player
    :param seconds_to_sleep: Seconds to have player sleep
    :param ble_adapter: ble adapter fixture
    :param bool wait_for_response: whether or not to wait for a response to
        the suspend command. It seems that when using serial instead of
        telnet, it's less likely to get a response to a command prior to the
        player suspending, which can cause a test to time out.
    """
    pytest.logger.info("Suspending {}".format(zp))
    zp.cli.spoof_suspend(wait_for_response=wait_for_response)
    pytest.logger.info("Successfully suspended. Waiting for BLE ad "
                       "to confirm before proceeding")

    # note: we turn off the cli connection so that if we are using a serial
    # connection, we cease the polling over serial so that we don't wake up
    # the sleeping player. we don't need to reconnect, as this is handled
    # automatically whenever someone issues a zp.cli.command()
    zp.cli.disconnect()

    ble_adapter.turn_radio_on()

    pytest.wait_for_assert(
        lambda:
        get_power_state_from_ble_ads(zp, ble_adapter) == PowerState.SLEEP,
        reason="{} did not suspend".format(zp),
        iteration_delay=1,
        timeout_seconds=MAX_WAIT_FOR_SUSPEND_BLE_AD_SECS)
    pytest.logger.info("Confirmed player is sleeping. Waiting {}s before "
                       "proceeding".format(seconds_to_sleep))
    guaranteed_sleep(seconds_to_sleep)




[docs]def get_power_state_from_ble_ads(zp, ble_adapter, timeout_seconds=10):
    """
    Retrieve players BLE advertised power state

    :param zp: Zone Player
    :param ble_adapter: BLE adapter
    :param timeout_seconds: Max seconds to scan for
    """
    addr = ble_adapter.get_address_for_bt_name(zp.get_ble_name())
    pytest.logger.info("Device BLE MAC Address is {}".format(addr))
    ble_adapter.clear_saved_data()
    ble_adapter.save_data_from_mac(addr)
    ble_adapter.scan(timeout_seconds=timeout_seconds)
    ads = ble_adapter.get_sonos_manufacturer_specific_data()
    for ad in reversed(ads):
        if ad.packet_type == SonosBlePacketType.ADVERTISEMENT_STATUS:
            return ad.power_state
    # if we got no ads...
    return None

[docs]def spoof_autoconnect_list_with_testrunner_at_index(
        zp, index, n_autoconnect_devices, btc_adapter,
        base_device_data_container):
    """
    Spoof an autoconnect device on the zone player and include the test
    runner in the list of spoofed devices at the index given.

    :param zp: zone player to spoof the devices on
    :param index: the index in the autoconnect list to place the test runner
        Bluetooth MAC address or None (no test runner mac)
    :param n_autoconnect_devices: the number of devices to spoof
    :param btc_adapter: BluetoothAdapter object or None (no test runner mac)
    :param base_device_data_container: object containing necessary info to
        spoof a BlueZ device on the player
    :return: list of MAC addresses in autoconnect list
    :rtype: :obj:`list` of :obj:`str`
    """
    zp.cli.bt_mgr.clear_all_autoconnect_devices()
    if btc_adapter:
        test_runner_mac = btc_adapter.get_bluetooth_mac_address()
        # need to remove colons to add to the pairlist
        test_runner_mac = test_runner_mac.replace(":", "")
    for i in range(n_autoconnect_devices):
        list_idx = n_autoconnect_devices - i - 1
        if list_idx == index:
            zp.cli.bt_mgr.append_mac_to_front_of_pairlist(test_runner_mac)
        else:
            mac = SPOOFED_DEVICE_MAC_BASE.format(i)
            info_file_uri = base_device_data_container.info_file_uri
            cache_file_uri = base_device_data_container.cache_file_uri
            dev = DeviceDataContainer(mac_address=mac,
                                      info_file_uri=info_file_uri,
                                      cache_file_uri=cache_file_uri)
            zp.cli.bt_mgr.add_autoconnect_device(dev)
    autoconnect_list = zp.cli.bt_mgr.get_autoconnect_list()
    _complete_spoofing_autoconnect_devices(zp)
    return autoconnect_list
