"""
This module holds a class that's used to track the logs on a player and
return reports and info regarding the time that it took to transition between
WiFi mode and Bluetooth mode
"""
import datetime
from log_tracker import LogTracker, MessageFilter


MAX_SECONDS_TRANSITION_TO_BLUETOOTH_MODE = 120
MAX_SECONDS_TRANSITION_TO_WIFI_MODE = 120

ANACAPA_WIFI_BT_TRANS_START_MSG = MessageFilter(
    "\<bt_ctrl,1\> Beginning transition to BT mode",
    "Beginning transition to BT mode")

ANACAPA_BT_WIFI_TRANS_START_MSG = MessageFilter(
    "\<bt_ctrl,1\> Beginning transition to WiFi mode",
    "Beginning transition to WiFi mode")

ANACAPA_SHUTDOWN_TO_BT_MSG = MessageFilter(
    "\<mod_zp,0\> raising SIGTERM to shutdown anacapa: switch to bluetooth",
    "raising SIGTERM to shutdown anacapa")

ANACAPA_SHUTDOWN_TO_WIFI_MSG = MessageFilter(
    "\<mod_zp,0\> raising SIGTERM to shutdown anacapa: switch to wifi",
    "raising SIGTERM to shutdown anacapa")

ANACAPA_RESTART_MSG = MessageFilter(
    "\<main,0\> \*{30}",
    "\*")

ANACAPA_ALL_READY_MSG = MessageFilter(
    "\<mod_zp,1\> all ready, advertising on network \([\d]+\.[\d]+\)",
    "all ready, advertising on network")

BT_MANAGER_WIFI_BT_TRANS_START_MSG = MessageFilter(
    "\[BTC\] handling message 'switch to bt', ID: \([\d]+\)",
    "handling message \'switch to bt\', ID:")

BT_MANAGER_BT_WIFI_TRANS_START_MSG = MessageFilter(
    "\[BTC\] handling message 'switch to wifi', ID: \([\d]+\)",
    "handling message \'switch to wifi\', ID:")

BT_CLASSIC_READY_MSG = MessageFilter(
    "\[BTC\] BT Classic ready \([\d]+ attempts\)",
    "BT Classic ready")

BT_MANAGER_ANACAPA_IPC_MSG = MessageFilter(
    "\[BTC\] Received hello message from Anacapa",
    "Received hello message from Anacapa")

IP_ADDR_REGEX = r"\b(?:\d{1,3}\.){3}\d{1,3}\b"

ANACAPA_NEW_IP_SHUTDOWN_MSG = MessageFilter(
    r"\<zp,0\> shutting down for new IP addr, old = {}, new = {}"
    r"".format(IP_ADDR_REGEX, IP_ADDR_REGEX),
    "shutting down for new IP addr")

[docs]    def wait_for_all_wifi_to_bluetooth_messages(self):
        """
        Wait for all of the WiFi to Bluetooth transition messages that
        signal a complete transition to appear in the logs
        """
        # wait for the last messages first so that we don't have tons of
        # spam in the logs for every single preceding message - we will fail
        # in extract_wifi_to_bluetooth_logs() if any are missing
        self._wait_for_msg(
            self._atli,
            ANACAPA_ALL_READY_MSG,
            timeout_seconds=MAX_SECONDS_TRANSITION_TO_BLUETOOTH_MODE,
            delay=DEFAULT_ITERATION_DELAY_SECONDS)
        self._wait_for_msg(self._bmli, BT_MANAGER_ANACAPA_IPC_MSG)
        self._wait_for_msg(self._bmli, BT_CLASSIC_READY_MSG)

        """
        Wait for all of the Bluetooth to WiFi transition messages that

        """
        # wait for the last messages first so that we don't have tons of

        self._wait_for_msg(self._atli, ANACAPA_ALL_READY_MSG,

                           delay=DEFAULT_ITERATION_DELAY_SECONDS)
        self._wait_for_msg(self._bmli, BT_MANAGER_ANACAPA_IPC_MSG)



[docs]    def extract_bluetooth_to_wifi_logs(self):
        """
        Extract all of the relevant log messages from the logs and store the
        timestamps
        """

        # NOTE: we have seen instances where only one of these messages is
        # seen during the transition from Bluetooth mode to WiFi mode. We
        # really only care about when the transition starts, so we check for
        # both of them.
        start_msg = self._get_msg(log_inspector=self._atli,
                                  msg_filter=ANACAPA_BT_WIFI_TRANS_START_MSG,
                                  fail_on_not_found=False)
        if start_msg is None:
            start_msg = self._get_msg(log_inspector=self._atli,
                                      msg_filter=ANACAPA_SHUTDOWN_TO_WIFI_MSG,
                                      fail_on_not_found=False)
        if start_msg is None:
            self._dump_logs_with_timestamp_and_log(
                ANACAPA_SHUTDOWN_TO_WIFI_MSG)
            raise ValueError(
                'Nether of the BT => WiFi start transition messages were '
                'found in the anacapa log. Expected one of these log messages:'
                ' "{}" or "{}"'.format(ANACAPA_BT_WIFI_TRANS_START_MSG.regex,
                                       ANACAPA_SHUTDOWN_TO_WIFI_MSG.regex)
            )

        self._anacapa_start_transition_time = start_msg.time

        # we should see the BT manager signaling that it's transitioning
        # from BT to WiFi mode.
        self._get_msg(log_inspector=self._bmli,
                      msg_filter=BT_MANAGER_BT_WIFI_TRANS_START_MSG)

        # store all as time deltas
        anacapa_time_offset = self._anacapa_start_transition_time

        self._anacapa_shutdown_time = self._get_time_of_msg(
            self._atli, ANACAPA_SHUTDOWN_TO_WIFI_MSG) - anacapa_time_offset
        self._anacapa_restart_time = self._get_time_of_msg(
            self._atli, ANACAPA_RESTART_MSG) - anacapa_time_offset
        self._anacapa_all_ready_time = self._get_time_of_msg(
            self._atli, ANACAPA_ALL_READY_MSG) - anacapa_time_offset

        # adjust the start transition time to be 0 seconds in
        self._anacapa_start_transition_time = datetime.timedelta(0, 0, 0)



[docs]    def extract_wifi_to_bluetooth_logs(self):
        """
        Extract all of the relevant log messages from the logs and store the
        timestamps
        """
        self._bt_manager_start_transition_time = self._get_time_of_msg(
            self._bmli, BT_MANAGER_WIFI_BT_TRANS_START_MSG)

        # NOTE: sometimes, we hit a race condition where we change our IP
        # address to the autip and bounce anacapa instead of saying that we're
        # going to bluetooth mode and bouncing anacapa. so we look for
        # either string.
        start_msg = self._get_msg(log_inspector=self._atli,
                                  msg_filter=ANACAPA_WIFI_BT_TRANS_START_MSG,
                                  fail_on_not_found=False)
        if start_msg is None:
            start_msg = self._get_msg(log_inspector=self._atli,
                                      msg_filter=ANACAPA_NEW_IP_SHUTDOWN_MSG,
                                      fail_on_not_found=False)
        if start_msg is None:
            start_msg = self._get_msg(log_inspector=self._atli,
                                      msg_filter=ANACAPA_SHUTDOWN_TO_BT_MSG,
                                      fail_on_not_found=False)
        if start_msg is None:
            self._dump_logs_with_timestamp_and_log(
                ANACAPA_SHUTDOWN_TO_BT_MSG)
            raise ValueError(
                'Neither of the WiFi => BT start transition messages were '
                'found in the anacapa log. Expected one of these '
                'log messages: "{}" or "{}" or "{}"'.format(
                    ANACAPA_WIFI_BT_TRANS_START_MSG,
                    ANACAPA_NEW_IP_SHUTDOWN_MSG,
                    ANACAPA_SHUTDOWN_TO_BT_MSG))

        self._anacapa_start_transition_time = start_msg.time

        # store all as time deltas
        bt_manager_time_offset = self._bt_manager_start_transition_time
        anacapa_time_offset = self._anacapa_start_transition_time

        self._bluetooth_classic_ready_time = self._get_time_of_msg(
            self._bmli, BT_CLASSIC_READY_MSG) - bt_manager_time_offset
        self._bt_manager_anacapa_ipc_ready_time = self._get_time_of_msg(
            self._bmli, BT_MANAGER_ANACAPA_IPC_MSG) - bt_manager_time_offset
        self._anacapa_shutdown_time = self._get_time_of_msg(
            self._atli, ANACAPA_SHUTDOWN_TO_BT_MSG) - anacapa_time_offset
        self._anacapa_restart_time = self._get_time_of_msg(
            self._atli, ANACAPA_RESTART_MSG) - anacapa_time_offset
        self._anacapa_all_ready_time = self._get_time_of_msg(
            self._atli, ANACAPA_ALL_READY_MSG) - anacapa_time_offset

        # adjust the start transition time to be 0 seconds in
        self._anacapa_start_transition_time = datetime.timedelta(0, 0, 0)
        self._bt_manager_start_transition_time = datetime.timedelta(0, 0, 0)

    def _get_double_column_format(self, field_tuple):
        """
        Format a tuple of two element into two columns

        :param field_tuple: tuple of two elements to be put into the string
        :return: string with two columns and a newline
        """
        return "{0:<30}{1:<0}\n".format(field_tuple[0], field_tuple[1])


[docs]    def pretty_print_bluetooth_to_wifi_timing_report(self):
        """
        Return a string that contains formatted text denoting all relevant
        transition events and their relative timestamps

        :return: :obj:`str`
        """
        return "Bluetooth ---> WiFi Timing Report\n" + "".join([
            self._get_double_column_format(x) for x in [
                (self._anacapa_start_transition_time,
                 "Anacapa Start Transition"),
                (self._anacapa_shutdown_time,"Anacapa Shutdown"),
                (self._anacapa_restart_time, "Anacapa Restart"),
                (self._anacapa_all_ready_time, "Anacapa All Ready")
            ]
        ]) + "Total Time: {}s".format(
            self.get_bluetooth_to_wifi_transition_time_seconds())
