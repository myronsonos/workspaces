    def __str__(self):
        """
        Override __str__ for pretty printing

        :return: :obj:`str`
        """
        fmt_spec = "{0:^15}{1:^25}{2:^35}{3:^45}\n"
        return fmt_spec.format(
            "Index", "Timestamp", "MAC", "Connection Succeeded?") + \
            "".join([fmt_spec.format(
                idx,
                entry.timestamp,
                entry.mac,
                "True" if entry.connection_succeeded else "False",
            ) for idx, entry in enumerate(self._list)])

    def __eq__(self, other):
        """
        Override == so that comparisons are easier

        :param other: :obj:`AutoconnectAttemptList`
        :return: True/False
        """
        other_entries = other.get_autoconnect_attempt_info_list()
        if len(other_entries) != len(self._list):
            return False
        for entry_one, entry_two in zip(self._list, other_entries):
            if entry_one != entry_two:
                return False
        return True

    def __ne__(self, other):
        return not self.__eq__(other)




[docs]class AutoconnectAttemptInfo(object):
    """
    Class to hold info about a particular autoconnect attempt
    """
    def __init__(self, mac, connection_succeeded, timestamp="00:00:00"):
        """
        Initializer

        :param str mac: mac address of BT device XX:XX:XX:XX:XX:XX
        :param bool connection_succeeded: whether or not the connection worked
        :param str timestamp: free-field timestamp field for informational
            purposes only
        """
        self.mac = mac
        self.connection_succeeded = connection_succeeded
        self.timestamp = timestamp

    def __eq__(self, other):
        """
        Override of == so that we can simplify comparison

        :param other: :obj:`AutoconnectAttemptInfo`
        :return: True/False
        """
        return self.mac == other.mac and \
            self.connection_succeeded == other.connection_succeeded

    def __ne__(self, other):
        return not self.__eq__(other)

    def __str__(self):
        """
        Override of __str__ so that pretty printing is easier

        :return: :obj:`str`
        """
        return "MAC: {}\nConnection Succeeded: {}".format(
            self.mac, self.connection_succeeded)
