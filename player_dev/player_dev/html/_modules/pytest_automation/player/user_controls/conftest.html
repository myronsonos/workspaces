    """
    :returns: a dictionary of rms values for treble, bass, and volume before and after

    """
    rms_dict = {}

    for zp in all_playback_devices:
        rms_dict[zp.modelNumber] = {}
        rms_dict[zp.modelNumber]['pre-bass'] = 0
        rms_dict[zp.modelNumber]['post-bass'] = 0
        rms_dict[zp.modelNumber]['pre-treble'] = 0
        rms_dict[zp.modelNumber]['post-treble'] = 0
        rms_dict[zp.modelNumber]['pre-volume'] = 0
        rms_dict[zp.modelNumber]['post-volume'] = 0

    yield rms_dict
    with open('data.json', 'w') as outfile:
        json.dump(rms_dict, outfile, indent=4)

        unique_balance_capable_device):
    """
    Yield an already configured
    :func:`~pytest_automation.pytest_sonos_device_selection.unique_balance_capable_device`
    which supports balance controls.
    """
    configure_balanced_zps_for_user_controls_test(
        [unique_balance_capable_device])

    # also be sure to spoof the orientation if necessary
    zp = unique_balance_capable_device
    zp.RenderingControl.SetMute(False)
    start_orientation = None
    if zp.modelNumber in MODEL_NUMBER_TO_ORIENTATION:
        if zp.cli.get_device_orientation() != HORIZONTAL:
            start_orientation = zp.cli.get_device_orientation()
            pytest.logger.info("Device is {} oriented, spoofing "
                               "orientation".format(start_orientation))
            zp.cli.set_device_orientation(HORIZONTAL)
    yield unique_balance_capable_device
    if start_orientation:
        pytest.logger.info("Setting device orientation back to {}".format(
            start_orientation))
        zp.cli.set_device_orientation(start_orientation)

[docs]def prime_playback(zp, uri):
    """
    run some playback on the given player to allow any user control changes
    to propagate and take effect, so as to reduce the chances of
    transitional audio changes taking effect during a measured audiotap.
    :param zp: the primary zone player under test (ht master, left sp, etc.)
    :param uri: a music file URI to playback
    """
    pytest.logger.info("Priming playback on zp {} with track URI: {}".format(
        zp, uri))
    zp.AVTransport.SetAVTransportURI(uri, '')
    zp.AVTransport.Play_and_wait()
    guaranteed_sleep(DEFAULT_PLAYBACK_SEC)
    zp.AVTransport.Stop_and_wait()




[docs]def save_get_dsp(zp, get_dsp_filename):
    """
    saves the contents of <zp>:1400/getDSP
    :param zp: the zone player to collect the contents of <zp>:1400/getDSP from
    :param get_dsp_filename: the name of the output file
    """
    def format_and_save_xml(xml_string, xml_filename):
        xml_string = xml_string.encode('utf-8')
        xml_string = xml.dom.minidom.parseString(xml_string)
        xml_string = xml_string.toprettyxml(encoding='utf-8')
        with open(xml_filename, 'w') as xml_file:
            xml_file.write(xml_string)

    xml_string = str(zp.diag.get_dsp())
    chan_map_prefix = zp.modelNumber
    if zp.get_bonded_zones():
        for z in zp.get_bonded_zones().values():
            chan_map_prefix = "{}_{}".format(chan_map_prefix, z.modelNumber)

    fn = "{}-{}-{}".format(zp.modelNumber, chan_map_prefix, get_dsp_filename)
    format_and_save_xml(xml_string, fn)

    for z in zp.get_bonded_zones().values():
        xml_string = str(z.diag.get_dsp())
        fn = "{}-{}-{}".format(z.modelNumber, chan_map_prefix, get_dsp_filename)
        format_and_save_xml(xml_string, fn)
