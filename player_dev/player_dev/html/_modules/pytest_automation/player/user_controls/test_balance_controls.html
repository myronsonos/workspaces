"""
This test verifies the functionality of the balance control available in
rendering control by measuring the RMS levels across audiotap channels for
both standalone players which support balance control and stereo pair
configurations.
"""
import pytest
from audio_util import rms_helper
from sleep import guaranteed_sleep
from sonos.services.common import wait_until_true
from sonos.client.zone_player import MODEL_NUMBER_TO_ORIENTATION, VERTICAL, \
    HORIZONTAL
from conftest import ITERATION_DELAY_SEC, prime_playback

# the left right channel indices in the dspout audiotaps for each relevant
# model
CHANNEL_MAPS_BY_MODEL = {
    "ZP120": {
        "LF": 0,
        "RF": 1
    },
    "ZP90": {
        "LF": 0,
        "RF": 1
    },
    "S6": {
        "LF": 0,
        "RF": 2
    },
    "S5": {
        "LF": 1,
        "RF": 4
    },
    "S3": {
        "LF": 0,
        "RF": 1
    },
    "S16": {
        "LF": 0,
        "RF": 1
    }
}
REMOVE_AUDIOTAP_FILES = False

DEVICES_WITH_BALANCE_DEPENDENT_ORIENTATIONS = ["S3", "S6"]

# the max acceptable variation in the RMS levels for two channels that are
# supposed to be identical (in dB SPL)
EQ_BALANCE_DIFF_THRESH_DB_SPL = 2

pytestmark = [
    pytest.mark.usefixtures('disable_audio_output')
]



[docs]def verify_rms_levels_stereo_pair(zps, rms_levels, balance_levels):
    """
    Verify that the RMS levels captured via audiotap have the expected
    relationship given the balance levels that have been applied.
    :param list zps: the list of zone players [left, right]
    :param dict rms_levels: the RMS levels, grouped by serial number and
    channel
    :param list balance_levels: the balance levels [left, right]
    """
    left_levels = rms_levels[zps[0].serialNumber]
    right_levels = rms_levels[zps[1].serialNumber]

    pytest.logger.info("Given balance levels: {}".format(balance_levels))
    pytest.logger.info("Left Channel RMS Levels: {}\n"
                       "Right Channel RMS Levels: {}".format(left_levels,
                                                             right_levels))

    # if left is greater than right in settings
    if balance_levels[0] > balance_levels[1]:
        for ch in left_levels:
            assert left_levels[ch] > right_levels[ch], \
                "Left levels should be greater than right levels for " \
                "channel: {},\n {} vs {}".format(
                    ch, left_levels[ch], right_levels[ch])

    # else if right is greater than left
    elif balance_levels[0] < balance_levels[1]:
        for ch in left_levels:
            assert left_levels[ch] < right_levels[ch], \
                "Left levels should be less than right levels for " \
                "channel: {},\n {} vs {}".format(
                    ch, left_levels[ch], right_levels[ch])

    # else if they're equal
    else:
        for ch in left_levels:
            assert abs(left_levels[ch] - right_levels[ch]) < \
                   EQ_BALANCE_DIFF_THRESH_DB_SPL, \
                   "Left levels should be equal to right levels for " \
                   "channel: {},\n {} vs {}".format(
                        ch, left_levels[ch], right_levels[ch])




[docs]def verify_rms_levels_standalone(zp, rms_levels, balance_levels):
    """
    Verify that the RMS levels captured via audiotap have the expected
    relationship given the balance levels that have been applied.
    :param zp: the zone player under test
    :param dict rms_levels: the RMS levels, grouped by serial number and
    channel
    :param list balance_levels: the balance levels [left, right]
    """
    left_ch = rms_levels[zp.serialNumber][
        CHANNEL_MAPS_BY_MODEL[zp.modelNumber]["LF"]]
    right_ch = rms_levels[zp.serialNumber][
        CHANNEL_MAPS_BY_MODEL[zp.modelNumber]["RF"]]

    # if the player is vertically oriented, we expect the channels to be equal
    is_non_horizontal = False
    if zp.modelNumber in MODEL_NUMBER_TO_ORIENTATION:
        orientation = zp.cli.get_device_orientation()
        if orientation != HORIZONTAL:
            pytest.logger.info("Device orientation is {}!".format(orientation))
            is_non_horizontal = True
        else:
            pytest.logger.info("Device orientation is horizontal!")
    else:
        pytest.logger.info("Device does not support orientation")

    if is_non_horizontal:
        assert abs(left_ch - right_ch) < EQ_BALANCE_DIFF_THRESH_DB_SPL, \
            "Left channel should be equal to right channel: " \
            "{} vs {}".format(left_ch, right_ch)
        return

    # else... we check that the balance is expected based on the levels
    pytest.logger.info("Left: {} Right:{}".format(left_ch, right_ch))
    # if left is greater than right
    if balance_levels[0] > balance_levels[1]:
        assert left_ch > right_ch, \
            "Left channel should be greater than right channel: " \
            "{} vs {}".format(left_ch, right_ch)

    # if right is greater than left
    elif balance_levels[0] < balance_levels[1]:
        assert left_ch < right_ch, \
            "Left channel should be less than right channel: " \
            "{} vs {}".format(left_ch, right_ch)

    # if equal
    else:
        assert abs(left_ch - right_ch) < EQ_BALANCE_DIFF_THRESH_DB_SPL, \
            "Left channel should be equal to right channel: " \
            "{} vs {}".format(left_ch, right_ch)




[docs]def verify_rms_levels(zps, rms_levels, balance_levels):
    """
    Common verification method used for verifying RMS levels relative to
    balance settings
    :param list zps: list of ZPs under test
    :param dict rms_levels: the RMS levels, grouped by serial number and
    channel
    :param list balance_levels: the balance levels [left, right]
    """
    if len(zps) > 1:
        verify_rms_levels_stereo_pair(zps, rms_levels, balance_levels)
    else:
        verify_rms_levels_standalone(zps[0], rms_levels, balance_levels)




[docs]def run_configs_captures_verifications(zps, uri):
    """
    Primary test method, which sets the ZP levels, calls the audiotap
    capture mechanisms, and calls the verification mechanism
    :param list zps: list of ZPs under test
    :param str uri: the URI of the audio to be played
    """

    zp = zps[0]
    zp.RenderingControl.SetVolume(30)
    zp.RenderingControl.SetMute(False)
    zp.RenderingControl.SetEQ("VolumeScalingFactor", 100)
    max_vol = 100
    min_vol = 0
    # apply the following sequences of channel levels [left, right]
    balance_levels_settings = [
        [max_vol, max_vol],
        [max_vol, min_vol],
        [min_vol, max_vol],
        [max_vol, max_vol]
    ]

    rms_levs = {}

    for balance_levels in balance_levels_settings:

        pytest.logger.info("Setting left-right balance to {} and {}".format(
            balance_levels[0], balance_levels[1]))

        # set levels (configure)
        zp.RenderingControl.SetVolume(volume=balance_levels[0], channel='LF')
        zp.RenderingControl.SetVolume(volume=balance_levels[1], channel='RF')

        prime_playback(zp, uri)

        # note the levels in the audiotap filename
        note = "left={}_right={}".format(balance_levels[0], balance_levels[1])

        expect_all_nonzero_taps = \
            balance_levels[0] == balance_levels[1]

        # capture
        rms_levels = rms_helper.get_rms_levels_from_music_playback(
            zp=zp, zps=zps, uri=uri, notes=note, duration=3,
            remove_audiotap_files=REMOVE_AUDIOTAP_FILES,
            retry_if_tap_entirely_zeros=expect_all_nonzero_taps)

        k = "{}_{}".format(balance_levels[0], balance_levels[1])
        rms_levs[k] = rms_levels

        # verify
        verify_rms_levels(zps, rms_levels, balance_levels)

        guaranteed_sleep(ITERATION_DELAY_SEC)

    pytest.logger.info("RMS levels for all runs: {}".format(rms_levs))

def test_balance_control_standalone(unique_balance_capable_device,
                                    track_uri):
    """
    Test all standalone players which support balance control
    :param unique_balance_capable_device: a configured player with balance
    control
    :param track_uri: URI of test track
    """
    if unique_balance_capable_device.get_arch() == 'sh4':
        pytest.logger.info("Cannot take DSP audiotaps on sh4... skipping")
        return
    run_configs_captures_verifications([unique_balance_capable_device],
                                       track_uri)

    # by default, we will test in the horizontal orientation, but we should
    # also check the vertical orientation for the relevant players, ensuring
    #  that the balance doesn't change when vertically oriented
    mn = unique_balance_capable_device.modelNumber
    if mn in MODEL_NUMBER_TO_ORIENTATION:
        pytest.logger.info("PASSED for horizontal orientation")
        for orientation in MODEL_NUMBER_TO_ORIENTATION[mn]:
            if orientation == HORIZONTAL:
                continue
            pytest.logger.info("Testing {} configuration".format(orientation))
            unique_balance_capable_device.cli.set_device_orientation(
                orientation)
            run_configs_captures_verifications(
                [unique_balance_capable_device], track_uri)
