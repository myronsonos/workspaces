"""
This test module aims at testing the crossfade functionality on the players
by analyzing the audio output of the players' DSPOUT audiotaps.

The core of the test executes by queueing up 2 tracks in the players' queue.
One of the tracks has a sine tone playing at one frequency.
The other track has a sine tone playing at a different frequency.
Each of the tracks is > 30s in durations, so as to provide enough audio for
crossfade to execute.

The tracks are played back with and without crossfade enabled.
The analysis involves taking FFT windows throughout time slices of the
entire audiotap duration, which should be around 1 minute without crossfade
enabled and around 45s with crossfade enabled (15s of first, 15s of both,
15s of second).

The FFTs without crossfade should only one window where both tones are
present in the frequency spectrum of the audio.
The FFTs with crossfade should show 15s of the first tone, 15s where the
first tone progressively attenuates and the second tone progressively
increments, and 15s of the second tone.
"""

XFADE_ENABLED_TOTAL_DURATION_SEC = (2 * SINGLE_FILE_DURATION_SEC -
                                    XFADE_MIN_DURATION_SEC)
XFADE_DISABLED_TOTAL_DURATION_SEC = 2 * SINGLE_FILE_DURATION_SEC

FIRST_TONE_MAX_DURATION = SINGLE_FILE_DURATION_SEC
FIRST_TONE_MIN_DURATION = SINGLE_FILE_DURATION_SEC - XFADE_MAX_DURATION_SEC

[docs]def get_codec_transition_list():
    """
    function to compute a list of codec transitions to be tested for crossfade
    :return: list of codec transitions
    example list: [('1000Hz.wav','700Hz.wav'),('1000Hz.wav','700.ogg')]
    """
    codec_list = list(itertools.permutations(PLAYBACK_TONES, 2))
    final_list = []
    codec_transition_list = []

    for pair in codec_list:
        tone_1_hz, codec_1 = pair[0].split('.')
        tone_2_hz, codec_2 = pair[1].split('.')

        if tone_1_hz == tone_2_hz:
            continue
        if (codec_1, codec_2) in codec_transition_list:
            continue
        else:
            final_list.append(pair)
            codec_transition_list.append((codec_1, codec_2))

    return final_list




[docs]def add_dbs(dbs):
    """
    function to add an array of dB values and return their sum.
    # NOTE: this is for incohenrent signals' dB
    :param dbs: list of floats
    :return: float - the sum (in dB) of the dB values listed in the given list
    """
    # In case the values in dbs are 0, log10(0) would raise a RuntimeWarning
    # specifying a divide by zero event.
    # These warnings make the log look cluttered. As such, ignore log10
    # divide warnings if any
    np.seterr(divide='ignore')

    # if only 1 dB value, return it
    if len(dbs) == 1:
        return dbs[0]
    tot = dbs[0]
    for db in dbs[1:]:
        tot = 10*np.log10(10**(tot/10.0) + 10**(db/10.0))
    return tot




[docs]def get_index_for_freq(freqs, target_f):
    """
    Get the index of the frequency bin that fall within target_f +-
    deviation
    :param freqs: the list of frequencies in Hz for the given window in the
    spectrogram
    :param target_f: the target frequency to obtain the energy for, given in Hz
    :return: list with one entry - the frequency that comes closest to target_f
    """
    closest_bin = 0
    # arbitary starting diff
    diff = 10000

    for idx, f in enumerate(freqs):
        abs_diff = abs(f - target_f)
        if abs_diff < diff:
            diff = abs_diff
            closest_bin = idx
    return [closest_bin]




[docs]def get_indices_of_not_these_frequencies(freqs, target_fs):
    """
    Get the indices of the frequency bins that are not any of the target_fs
    :param freqs: the list of frequencies in Hz for the given window in the
    spectrogram
    :param target_fs: the target frequencies to obtain the energy for,
    given in Hz
    :return: list of indices in freqs which are not any of the target
    frequencies
    """
    target_fs_bins = []
    for target_f in target_fs:
        target_fs_bins.append(get_index_for_freq(freqs, target_f)[0])

    indices = []
    for idx, f in enumerate(freqs):
        if idx not in target_fs_bins:
            indices.append(idx)
    return indices




[docs]def get_energy_levels_for_frequencies_in_spectrograms(freqs, spectrograms):
    """
    given a set of target frequencies and a set of spectrograms for several
    players, return the energy levels each second for each target frequency
    for each player.
    :param freqs: list of frequencies that we're interested in obtaining the
    energy levels over time
    :param spectrograms: a dictionary of form
    {
        "player serial number":
        {
            "freqs": [the list of frequencies for the FFT on the player],
            "mags" [
                [list of magnitude values for each freq at time t=0],
                [list of magnitude values for each freq at time t=1],
                ...
                [list of magnitude values for each freq at time t=N]
            ]
        },
        ...
    }
    :return: dictionary detailing the energy over time for each frequency of
    interest and all other frequencies (noise):
    {
        "player serial number":
        {
            "frequency #1": [
                magnitude at time t=0,
                magnitude at time t=1,
                ...
                magnitude at time t=N
            ],
            "frequency #2": [
                ...
            ],
            ...
            "noise": [
                ....
            ]
        }
    }
    """
    energy_over_time = {}

    # store the list of bins for each frequency for each SN up front so that
    #  we don't have to do a look up each time (slow)
    bins_for_freqs = {}
    for sn in spectrograms:
        energy_over_time[sn] = {}

        # the bins that correspond to the frequencies within the tolerances
        # of the deviation for each target frequency
        bins_for_freqs[sn] = {}
        freq = spectrograms[sn]["freqs"]
        energy_over_time[sn]["noise"] = []
        bins_for_freqs[sn]["noise"] = get_indices_of_not_these_frequencies(
            freq, freqs)

        # initialize magnitudes of each frequency and noise to be empty
        for f in freqs:
            energy_over_time[sn][f] = []
            bins_for_freqs[sn][f] = get_index_for_freq(freq, f)

        # for each time slice
        for sec, mag in enumerate(spectrograms[sn]["mags"]):

            # get the magnitude for each frequency
            for f in freqs:

                energy_over_time[sn][f].append(
                    add_dbs([mag[b] for b in bins_for_freqs[sn][f]]))

            # get the energy in the noise band
            energy_over_time[sn]["noise"].append(
                add_dbs([mag[b] for b in bins_for_freqs[sn]["noise"]]))

    return energy_over_time




[docs]def is_xfade_engaged(energy_tone_1, prev_energy_tone_1, energy_tone_2,
                     prev_energy_tone_2):
    """
    Checker to determine if crossfade is currently engaged given the
    magnitude values of the two distinct frequencies that we're interested in
    :param energy_tone_1: the magnitude in dB of the starting tone
    :param prev_energy_tone_1: the magnitude in dB of the starting tone a
    window ago
    :param energy_tone_2: the magnitude in dB of the ending tone
    :param prev_energy_tone_2: the magnitude in dB of the ending tone a
    window ago
    :return: True/False depending on whether or not the logic determines
    that crossfade is occurring
    """
    # if the outgoing tone has more energy than the incoming tone, compare
    # the outgoing
    if energy_tone_1 > energy_tone_2:

        # if the difference in energy between the two tones is less than our
        #  threshold, we conclude that they are playing at the same time and
        #  that crossfade in engaged
        if (energy_tone_1 - energy_tone_2) < TONE_DIFFERENCE_THRESHOLD_DB:
            pytest.logger.info("Xfade engaged: Tone 1 - Tone 2 < diff thresh")
            return True

        # if the outgoing tone has less energy than it did 1s ago and the
        # difference in energy is greater than our threshold, conclude that
        # crossfade is engaged
        if prev_energy_tone_1 - energy_tone_1 > TONE_DECREASE_THRESHOLD_DB:
            pytest.logger.info("Xfade engaged: Tone 1 decrease above thresh")
            return True

        # otherwise, conclude crossfade is NOT engaged
        return False

    # if the incoming tone has more energy than the outgoing tone, compare
    # the incoming
    elif energy_tone_2 > energy_tone_1:

        # if the difference in energy between the two tones is less than our
        #  threshold, we conclude that they are playing at the same time and
        #  that crossfade in engaged
        if (energy_tone_2 - energy_tone_1) < TONE_DIFFERENCE_THRESHOLD_DB:
            pytest.logger.info("Xfade engaged: Tone 2 - Tone 1 < diff thresh")
            return True

        # if the incoming tone has more energy than it did 1s ago and the
        # difference in energy is greater than our threshold, conclude that
        # crossfade is engaged
        if energy_tone_2 - prev_energy_tone_2 > TONE_DECREASE_THRESHOLD_DB:
            pytest.logger.info("Xfade engaged: Tone 2 increase above thresh")
            return True

        # otherwise, conclude that crossfade is NOT engaged
        return False

    else:
        if energy_tone_2 > prev_energy_tone_2 and energy_tone_1 < \
                prev_energy_tone_1:
            # if they're equal, then it's definitely a crossfade
            pytest.logger.info("Xfade engaged - tones equal")
            return True
        # otherwise, this could be that both tones are playing equally
        # without changing amplitude
        return False




[docs]def verify_tone_1(sn, energy_tone_1, energy_tone_2, sec):
    """
    Set of assertions to run whenever the first tone is playing
    :param sn: playback device serial number
    :param energy_tone_1: the energy of the frequency of the first tone in
    dB for the current point in time
    :param energy_tone_2: the energy of the frequency of the second tone in
    dB for the current point in time
    :param sec: the number of seconds into playback that we are at
    :return: either an empty string if assertions pass or an error message
    """
    # the energy in the first tone frequency bin should be much higher than
    # the energy in the second tone frequency bin
    try:
        assert energy_tone_1 - energy_tone_2 > TONE_DIFFERENCE_THRESHOLD_DB
    except AssertionError:
        return "SN:{} - Tone 1 should be much higher than tone 2: {} vs {}".\
            format(sn, energy_tone_1, energy_tone_2)

    # the timestamp should come well before the max duration of the first tone
    try:
        assert sec < FIRST_TONE_MAX_DURATION
    except AssertionError:
        return "SN:{} - First tone should not be playing after {}s".format(
            sn, FIRST_TONE_MAX_DURATION)

    return ''

[docs]def verify_xfade_audio(sn, energy_tone_1, prev_energy_tone_1, energy_tone_2,
                       prev_energy_tone_2, sec):
    """
    Set of assertions to run whenever crossfade is engaged
    :param sn: playback device serial number
    :param energy_tone_1: the energy of the frequency of the first tone in
    dB for the current point in time
    :param prev_energy_tone_1: the energy of the frequency of the first tone in
    dB for previous second
    :param energy_tone_2: the energy of the frequency of the second tone in
    dB for the current point in time
    :param prev_energy_tone_2: the energy of the frequency of the second tone
    in dB for previous second
    :param sec: the number of seconds into playback that we are at
    :return: either an empty string if assertions pass or an error message
    """
    try:
        assert energy_tone_1 < prev_energy_tone_1
    except AssertionError:
        return "SN:{} - Energy should be decreasing in tone 1".format(sn)

    try:
        assert energy_tone_2 > prev_energy_tone_2
    except AssertionError:
        return "SN:{} - Energy should be increasing in tone 2".format(sn)

    return ''




[docs]def verify_xfade_enabled_in_frequency_bin(sn, levels, tone_1_hz, tone_2_hz):
    """
    Verify that the crossfade did actually engage on the zp for which we
    have the FFT levels at each second of the tones that we're interested in
    :param sn: playback device serial number
    :param levels: map of frequencies and their corresponding magnitude
    :param tone_1_hz: freq of first tone
    :param tone_2_hz: freq of second tone
    values for each second in time, of form:
    {
        "frequency1": [magnitude at 1s, magnitude at 2s, ... magnitude at Ns],
        "frequency2": [...],
        ...
        "frequencyN": [...]
    }
    :return: either an empty string if assertions pass or an error message
    """
    # boolean for tracking if we're currently in the crossfade
    in_xfade_section = False
    # boolean for tracking if we've ever been in the crossfade section
    xfade_was_detected = False
    n_sec_xfade_engaged = 0
    err_msg = ''

    # skip the first seconds and the last seconds to avoid noise floor in
    # the frequency bins
    for sec in range(SKIP_SECONDS_SPECTROGRAM_START,
                     len(levels["noise"]) - SKIP_SECONDS_SPECTROGRAM_END):

        # log current values for debugging purposes
        pytest.logger.info(
            "Time: {}s: Energy @{}Hz: {}dB, @{}Hz: {}dB".format(
                sec,
                tone_1_hz,
                levels[tone_1_hz][sec],
                tone_2_hz,
                levels[tone_2_hz][sec]
            )
        )

        # see if we think xfade is enabled from the audio
        if is_xfade_engaged(levels[tone_1_hz][sec],
                            levels[tone_1_hz][sec - 1],
                            levels[tone_2_hz][sec],
                            levels[tone_2_hz][sec - 1]):

            # if crossfade had engaged and then disengaged, then crossfade
            # should remain disengaged for the remainder of the test.
            # If crossfade was engaged > disengaged > engaged, there's
            # something very wrong. This assertion checks for that case.
            if xfade_was_detected:
                try:
                    assert in_xfade_section
                except AssertionError:
                    err_msg = "SN:{} - Xfade detection re-engaged".format(sn)
                    break

            in_xfade_section = True
            xfade_was_detected = True
            n_sec_xfade_engaged += 1
        else:
            if in_xfade_section:
                pytest.logger.info("Disengaging xfade")
            in_xfade_section = False

        # if before crossfade
        if not in_xfade_section and not xfade_was_detected:
            err_msg = verify_tone_1(sn, levels[tone_1_hz][sec],
                                    levels[tone_2_hz][sec], sec)
            if err_msg != '':
                break

        # if xfade is engaged
        elif in_xfade_section:
            err_msg = verify_xfade_audio(
                sn,
                levels[tone_1_hz][sec],
                levels[tone_1_hz][sec - 1],
                levels[tone_2_hz][sec],
                levels[tone_2_hz][sec - 1],
                sec
            )

            if err_msg != '':
                break

        # if xfade was on, but is not now (after crossfade)
        elif not in_xfade_section and xfade_was_detected:
            err_msg = verify_tone_2(sn, levels[tone_1_hz][sec],
                                    levels[tone_2_hz][sec], sec)

            if err_msg != '':
                break

            # Assertions around crossfade duration being within an
            # acceptable range
            try:
                assert n_sec_xfade_engaged >= XFADE_MIN_DURATION_SEC
            except AssertionError:
                err_msg = "SN:{} - Xfade should have been engaged for at " \
                          "least {}s".format(sn, XFADE_MIN_DURATION_SEC)
                break

            try:
                assert n_sec_xfade_engaged <= XFADE_MAX_DURATION_SEC
            except AssertionError:
                err_msg = "SN:{} - Xfade should not have been engaged for " \
                          "more than {}s".format(sn, XFADE_MAX_DURATION_SEC)
                break

    return err_msg




[docs]def verify_xfade_disabled_for_zp(sn, levels, tone_1_hz, tone_2_hz):
    """
    Verify that the crossfade did NOT engage on the zp for which we
    have the FFT levels at each second of the tones that we're interested in
    :param sn: playback device serial number
    :param levels: map of frequencies and their corresponding magnitude
    values for each second in time, of form:
    {
        "frequency1": [magnitude at 1s, magnitude at 2s, ... magnitude at Ns],
        "frequency2": [...],
        ...
        "frequencyN": [...]
    }
    :param tone_1_hz: freq of first tone
    :param tone_2_hz: freq of second tone
    :return: either an empty string if assertions pass or an error message
    """
    # because we are taking 1s spectrogram windows, there will be 1s in the
    # middle of the playback where it's likely that the difference between
    # the two tones will be less than the Xfade threshold - which is fine as
    # long as the closeness only lasts 1s
    n_counts_tone_diff_less_than_threshold = 0
    err_msg = ''

    # skip the first seconds and the last seconds to avoid noise floor
    for sec in range(SKIP_SECONDS_SPECTROGRAM_START,
                     len(levels["noise"]) - SKIP_SECONDS_SPECTROGRAM_END):

        pytest.logger.info(
            "Time: {}s: Energy @{}Hz: {}dB, @{}Hz: {}dB".format(
                sec, tone_1_hz, levels[tone_1_hz][sec], tone_2_hz,
                levels[tone_2_hz][sec]))

        if (abs(levels[tone_1_hz][sec] - levels[tone_2_hz][sec]) <
                TONE_DIFFERENCE_THRESHOLD_DB):
            n_counts_tone_diff_less_than_threshold += 1
            try:
                assert n_counts_tone_diff_less_than_threshold < 2
            except AssertionError:
                err_msg = "SN:{} - Tones should not be as close in energy " \
                          "for more than 1s".format(sn)
                break
            continue
        # check to see which tone is currently higher, and do asserts as
        # necessary
        if levels[tone_2_hz][sec] < levels[tone_1_hz][sec]:
            err_msg = verify_tone_1(sn, levels[tone_1_hz][sec],
                                    levels[tone_2_hz][sec], sec)

            if err_msg != '':
                break
        else:
            err_msg = verify_tone_2(sn, levels[tone_1_hz][sec],
                                    levels[tone_2_hz][sec], sec)

            if err_msg != '':
                break

    return err_msg




[docs]def verify_xfade_enabled_in_audio(spectrograms_xfade_on, tones_hz):
    """

    involved in the test to ensure that crossfade did engage at the correct
    time.

    :param tones_hz: list of tone frequencies under test

    levels = get_energy_levels_for_frequencies_in_spectrograms(
        [tone_2_hz, tone_1_hz], spectrograms_xfade_on)
    for sn in levels:
        pytest.logger.info("Verifying crossfade enabled for {}".format(sn))
        xfade_enabled_err_msgs.append(
            verify_xfade_enabled_in_frequency_bin(sn, levels[sn],
                                                  tone_1_hz, tone_2_hz))

    return xfade_enabled_err_msgs

def test_crossfade_standalone(all_unique_platform_playback_devices, webserver,
                              codec_pair, xfade_enabled):
    """
    Abstract test method to test cross any given number of ZPs, where the
    first zp is always the GC or the master in a bonded configuration
    :param all_unique_platform_playback_devices: one playback device per
    platform in this testbed
    :param webserver: a server hosting some files for playback
    :param codec_pair: codec transition under test
    :param xfade_enabled: crossfade is enabled or disabled
    """
    zps = [zp for zp in all_unique_platform_playback_devices if
           zp.get_arch() not in ['sh4']]

    # Flag to check if all processes start
    p_start_flag = True

    # Get tone file names
    # Get tone frequencies for the codec pair
    tone_1_file, tone_2_file = codec_pair
    tone_1_hz = int(re.findall(r'\d+', tone_1_file)[0])
    tone_2_hz = int(re.findall(r'\d+', tone_2_file)[0])

    # Get codec for each tone to be used in audiotap file name
    tone_1_codec = tone_1_file.split('.')[1]
    tone_2_codec = tone_2_file.split('.')[1]

    first_tone_uri = 'http://{}/{}'.format(
        webserver.host(), tone_1_file)
    second_tone_uri = 'http://{}/{}'.format(
        webserver.host(), tone_2_file)

    pytest.logger.info('Crossfade test - Tone 1:{}, '
                       'Tone 2:{}'.format(tone_1_file, tone_2_file))

    for z in zps:
        pytest.logger.info(z)

        z.cli.amp_power_off()

        z.RenderingControl.SetVolume(TEST_VOLUME)
        z.RenderingControl.SetMute(False)
        z.RenderingControl.SetEQ("VolumeScalingFactor", 100)

        # in order to enable crossfade, we must be using the queue
        z.AVTransport.use_queue()

        # enable/disable crossfade
        z.AVTransport.SetCrossfadeMode(xfade_enabled)

        # set queue up
        z.AVTransport.clear_queue()
        z.AVTransport.add_to_queue(first_tone_uri)
        z.AVTransport.add_to_queue(second_tone_uri)
        z.AVTransport.use_queue()

    if xfade_enabled:
        duration = XFADE_ENABLED_TOTAL_DURATION_SEC
        note = 'xfade_on_{}_{}'.format(tone_1_codec, tone_2_codec)
    else:
        duration = XFADE_DISABLED_TOTAL_DURATION_SEC
        note = 'xfade_off_{}_{}'.format(tone_1_codec, tone_2_codec)

    # Setup a multiprocessing queue for parallel execution on all devices
    spectrograms_xfade_dct = {}
    zps_sn_modelnumber_dct = {}
    proc = []
    res_q_xfade = Queue()

    for z in zps:
        zps_sn_modelnumber_dct[z.serialNumber] = z.modelNumber
        z.AVTransport.Play_and_wait()
        p = Process(target=fft_helper.play_and_get_spectrograms, args=(
            [z], res_q_xfade, note, duration,
            SAMPLE_RATE, 0, False))
        proc.append(p)
        p.start()

    # Check if all processes actually started. If a process didn't start,
    # change the value of p_start_flag to False
    for p in proc:
        if not p.is_alive():
            pytest.logger.error('Failed to start process')
            proc.remove(p)
            p_start_flag = False

    # Wait for the active processes to finish
    # Each process returns the device serialNumber and its spectrogram
    # Store this data in dictionary spectrograms_xfade_dct
    for p in proc:
        spec = res_q_xfade.get()
        sn, spect = spec.items()[0]
        key = '{}:{}'.format(sn, zps_sn_modelnumber_dct[sn])
        spectrograms_xfade_dct[key] = spect

    for z in zps:
        z.AVTransport.Stop_and_wait()

    # If a process failed to start, wait for the other processes to finish,
    # then delete the audiotaps and raise an assertion
    if not p_start_flag:
        gld_filelist = [f for f in os.listdir(LOCAL_PATH) if
                        f.endswith('.gld')]
        for f in gld_filelist:
            os.remove(f)
        raise AssertionError('Failed to start process')

    pytest.logger.info(spectrograms_xfade_dct.keys())

    # do verifications
    if xfade_enabled:
        xfade_err_msgs = \
            verify_xfade_enabled_in_audio(spectrograms_xfade_dct,
                                          [tone_1_hz, tone_2_hz])
    else:
        xfade_err_msgs = \
            verify_xfade_disabled_in_audio(spectrograms_xfade_dct,
                                           [tone_1_hz, tone_2_hz])

    # Check for errors
    assert xfade_err_msgs.count('') == len(xfade_err_msgs), \
        display_xfade_errors(xfade_err_msgs)
