"""
This module tests various test cases pertaining to streams in standalone
playback devices.
"""

import pytest
from test_streams_standalone import get_rms_levels_for_stream, \
    skip_if_unsupported_stream, setup_player_rc, HIGH_VOLUME, LOW_VOLUME, \
    PLAYBACK, EXT_AUDIO, FEEDBACK, channel_is_silent, DIFFERENCE_THRESH_DB, \
    ITERATION_DELAY_SEC
from conftest import prime_playback, save_get_dsp
from sonos.exception import TimeoutError
from sleep import guaranteed_sleep
from sonos.services.common import wait_until_true


HTM_PREFIX = "HTM_"
SAT_PREFIX = "SAT_"
HTM_FEEDBACK = HTM_PREFIX + FEEDBACK
HTM_EXT_AUDIO = HTM_PREFIX + EXT_AUDIO
SAT_FEEDBACK = SAT_PREFIX + FEEDBACK
SAT_EXT_AUDIO = SAT_PREFIX + EXT_AUDIO

HT_BONDED_TEST_STREAMS = [
    HTM_FEEDBACK,
    HTM_EXT_AUDIO,
    SAT_FEEDBACK,
    SAT_EXT_AUDIO,
    PLAYBACK
]

pytestmark = [
    pytest.mark.usefixtures('disable_audio_output')
]



[docs]def get_playback_capture_muse_from_bonded_stream(bonded_stream,
                                                 htm_muse, left_muse, htm,
                                                 left, right):
    """
    get the player that should do the playback, the players that should have
    audiotaps captured, and the muse connection for the given stream test
    :param str bonded_stream: one of HT_BONDED_TEST_STREAMS
    :param htm_muse: muse connection to the HT master
    :param left_muse: muse connection to the left surround
    :param htm: ht master
    :param left: left surround
    :param right: right surround
    :return: (playback zone player, list of zone players to capture
        audiotaps for, muse connection to playback zone player)
    """
    # depending on the stream, the zone players whose RMS levels should vary
    # with volume varies (i.e. SAT_FEEDBACK => only the satellite's levels
    # should be looked at and expected to change. others should not play
    # back at all).
    if bonded_stream.startswith(HTM_PREFIX):
        capture_zps = [htm]
        playback_zp = htm
        muse = htm_muse

    elif bonded_stream.startswith(SAT_PREFIX):
        capture_zps = [left]
        playback_zp = left
        muse = left_muse

    elif bonded_stream == PLAYBACK:
        capture_zps = [htm, left, right]
        playback_zp = htm
        muse = htm_muse

    else:
        raise NotImplementedError("Unrecognized stream: {}".format(
            bonded_stream))

    return playback_zp, capture_zps, muse




[docs]def translate_bonded_stream_to_stream(bonded_stream):
    """
    convert the  stream from one of HT_BONDED_TEST_STREAMS to one of the
    STREAMS used in the standalone stream's test's get_rms_levels_for_stream
    :param str bonded_stream: one of HT_BONDED_TEST_STREAMS
    :return str: one of STREAMS
    """
    if FEEDBACK in bonded_stream:
        stream = FEEDBACK

    elif EXT_AUDIO in bonded_stream:
        stream = EXT_AUDIO

    elif PLAYBACK == bonded_stream:
        stream = bonded_stream

    else:
        raise NotImplementedError("Unrecognized stream: {}".format(
            bonded_stream))

    return stream




[docs]def run_test(track_uri, bonded_stream, muses_for_50_system):
    """
    This is the common test method called by the various test methods to
    test that RMS levels are as expected across the audiotaps of various HT
    bonded players for various stream playbacks.

    :param str track_uri: the uri to an audio file for playback
    :param str bonded_stream: one of HT_BONDED_TEST_STREAMS
    :param muses_for_50_system: muse connection configured to HT
        master and left surround. also the zone players for 50 system of form
        (htm_muse, left_muse, htm, left, right)
    """
    # zps to capture audio on
    htm_muse, left_muse, htm, left, right = muses_for_50_system
    zps = [htm, left, right]

    playback_zp, capture_zps, mc = \
        get_playback_capture_muse_from_bonded_stream(
            bonded_stream, htm_muse, left_muse, htm, left, right)

    stream = translate_bonded_stream_to_stream(bonded_stream)
    skip_if_unsupported_stream(playback_zp, stream)

    # set the volume and other rendering control settings and prime the
    # playback
    pytest.logger.info("Setting up bonded zone's RC configs via HT Master")
    setup_player_rc(zp=htm, volume=HIGH_VOLUME)
    prime_playback(htm, track_uri)

    # get rms levels for the reference settings to compare against other taps
    notes = "base_bonded_stream={}_{}_{}_{}".format(stream, htm.modelNumber,
                                                    left.modelNumber,
                                                    right.modelNumber)
    base_notes = notes
    save_get_dsp(playback_zp, "{}_getDSP.xml".format(notes))
    rms_levels_base = get_rms_levels_for_stream(
        zp=playback_zp, uri=track_uri, notes=notes, stream=stream,
        capture_zps=capture_zps,
        muse=mc)

    # verification => make sure that we get timeout errors trying to get
    # audiotap files from other ZPs if we're doing external audio or
    # feedback on one of the ZPs
    for zp in [z for z in zps if z not in capture_zps]:
        with pytest.raises(TimeoutError) as e:
            get_rms_levels_for_stream(
                zp=playback_zp, uri=track_uri, notes=notes, stream=stream,
                capture_zps=[zp], muse=mc)
            guaranteed_sleep(ITERATION_DELAY_SEC)

    # change level on satellite to lower level
    pytest.logger.info("Setting volume on HT master via Satellite's RC")
    left.RenderingControl.SetVolume(LOW_VOLUME)
    prime_playback(htm, track_uri)

    # take rms levels and verify
    notes = "sat_lowered_bonded_stream={}_{}_{}_{}".format(
        stream, htm.modelNumber, left.modelNumber, right.modelNumber)
    notes_lowered = notes
    save_get_dsp(playback_zp, "{}_getDSP.xml".format(notes))
    rms_levels_lowered = get_rms_levels_for_stream(
        zp=playback_zp, uri=track_uri, notes=notes, stream=stream,
        capture_zps=capture_zps,
        muse=mc)

    # change level on ht master to higher level
    pytest.logger.info("Setting volume on HT Master via HT Master's RC")
    htm.RenderingControl.SetVolume(HIGH_VOLUME)
    prime_playback(htm, track_uri)

    # take rms levels and verify
    notes = "htm_raised_bonded_stream={}_{}_{}_{}".format(
        stream, htm.modelNumber, left.modelNumber, right.modelNumber)
    notes_raised = notes
    save_get_dsp(playback_zp, "{}_getDSP.xml".format(notes))
    rms_levels_raised = get_rms_levels_for_stream(
        zp=playback_zp, uri=track_uri, notes=notes, stream=stream,
        capture_zps=capture_zps,
        muse=mc)

    debug_notes = "{}\n{}\n{}".format(base_notes, notes_lowered, notes_raised)

    # verifications of RMS levels that should have changed
    for zp in capture_zps:
        sn = zp.serialNumber
        for ch in rms_levels_base[sn]:

            if channel_is_silent(rms_levels_raised[sn][ch]):
                assert channel_is_silent(rms_levels_lowered[sn][ch]) and \
                    channel_is_silent(rms_levels_base[sn][ch]), \
                    "zp: {} ch: {} Expect all to have silent RMS. base: {}," \
                    "lowered: {} raised: {}\n" \
                    "Audiotap notes:\n{}".format(
                        zp, ch,
                        rms_levels_base, rms_levels_lowered,
                        rms_levels_raised, debug_notes)
                continue

            assert rms_levels_raised[sn][ch] > rms_levels_lowered[sn][ch], \
                "zp: {} ch: {} Expect levels to be higher:" \
                "base: {}, lowered: {}, raised: {}\n" \
                "Audiotap Notes:\n{}".format(
                    zp, ch, rms_levels_base, rms_levels_lowered,
                    rms_levels_raised, debug_notes)

            assert abs(rms_levels_raised[sn][ch] -
                       rms_levels_base[sn][ch]) < DIFFERENCE_THRESH_DB, \
                "zp: {} ch: {} Expect levels to be same for raised vs base, " \
                "base: {}, lowered: {}, raised: {}\n" \
                "Audiotap notes:\n{}".format(
                    zp, ch, rms_levels_base, rms_levels_lowered,
                    rms_levels_raised, debug_notes)




[docs]def bump_diag_levels_on_players(zps):
    """
    Increase the logging level on the given zp's for modules useful for
    debugging

    :param list zps: list of zone players to bump the diags on
    """
    for zp in zps:
        zp.diag.set_diag_level('sound_device', 4)




[docs]def wait_for_zone_channel_count(muses_for_50_system):
    """
    Wait until we get a message telling us that the zone channel count has
    been updated

    :param muses_for_50_system: muse connection configured to HT
        master and left surround. also the zone players for 50 system
    """
    htm_muse, left_muse, htm, left, right = muses_for_50_system

    if not htm.is_voice_capable():
        pytest.skip("Non-voice capable HT master")

    bump_diag_levels_on_players([htm])

    msg = "Zone channel count has been updated to 5"
    htm.diag.start_udp_log(reset=True)
    htm.diag.udp_logger.add_search(
        query=msg,
        zp=htm
    )
    try:
        wait_until_true(
            lambda: htm.diag.udp_logger.get_search_results(htm, msg),
            reason="Waiting for channel count to change on zp: {}".format(htm),
            timeout_seconds=15)
    except TimeoutError:
        pytest.logger.info("Did not see log message for increased channel "
                           "count on HT master - is possible that channel "
                           "count was set prior to increased logging. "
                           "Proceeding under assumption that it has been set "
                           "already.")
    # remove the query
    htm.diag.udp_logger.remove_search(zp=htm, query=msg)
