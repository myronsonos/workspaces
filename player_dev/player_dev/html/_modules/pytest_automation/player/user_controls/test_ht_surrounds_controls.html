"""
This module is intended to test the various surround controls that are
available to users in the controller by toggling the controls via
RenderingControl's UPnP interface and assessing the output audio quality
from the DSP. The tests measure if the surround levels are changed
accordingly for TV and Music when each is playing and also test the negative
cases (TV volume does not change when music volume is adjusted). This test
also tests the surrounds enable/disable feature by inspecting the DSP output.

This test also verifies the ambient vs. full mode of the satellites during
music playback by checking the values in the getDSP config page.
"""
from audio_util import rms_helper
import pytest
from sonos.services.common import wait_until_true
from sleep import guaranteed_sleep
from conftest import LOCAL_TEST_FILE_PATH, TEST_FILE_METADATA, \
    ITERATION_DELAY_SEC, prime_playback, dump_settings, save_get_dsp
from sonos.exception import TimeoutError
import os

# The baseline level is the controller setting level at which the test
# measurements are taken.
BASELINE_CONTROLLER_LEVEL = 15
NUMBER_OF_ITERATIONS = 5
REMOVE_AUDIOTAP_FILES = False

# NOTE: because we're taking an audiotap as our main form of verification,
# we don't have any variables that are exposed for which we can call
# wait_until_true() to wait before performing a verification. As such,
# we use a propagation delay here (heuristic) to allow ample time for the
# surround EQ levels to be adjusted after hitting a RenderingControl point.
N_SECONDS_EQ_CHANGE_PROPAGATION_DELAY = 3

pytestmark = [
    pytest.mark.usefixtures('disable_audio_output')
]



[docs]def surround_mode_is_full(zp):
    """
    Check the values in getDSP to view the current surround mode setting
    :param zp: the HT Master Sonos Zone Component
    :return: boolean - True if 'full' - False if 'ambient'
    """
    block = zp.diag.get_dsp_block('Upmixer')
    are_full = [b.value for b in block.parameter.param if b.name ==
                "fullRangeSurrounds"][0]
    pytest.logger.info("Surround Mode is Full - {}".format(are_full))
    return are_full == "Yes"

def test_surround_music_playback_mode(ht_master_capable_device,
                                      track_uri):
    """
    Test to verify that getDSP is set when playing music content
    and toggling the ambient/full surround mode switches
    :param ht_master_capable_device: an HT Master capable ZP
    :param track_uri: URI to test track
    """
    zp = ht_master_capable_device
    zp.AVTransport.SetAVTransportURI(track_uri, '')
    zp.AVTransport.Play_and_wait()
    for i in range(0, NUMBER_OF_ITERATIONS):
        zp.RenderingControl.set_surround_mode('full')
        pytest.logger.info("Set Surround Mode to Full")
        wait_until_true(
            predicate=lambda: surround_mode_is_full(zp),
            timeout_seconds=5,
            reason="Waiting for surround mode to switch to full"
        )
        zp.RenderingControl.set_surround_mode('ambient')
        pytest.logger.info("Set Surround Mode to Ambient")
        wait_until_true(
            predicate=lambda: not surround_mode_is_full(zp),
            timeout_seconds=5,
            reason="Waiting for surround mode to switch to ambient"
        )
    zp.AVTransport.Pause_and_wait()




[docs]def set_levels_and_wait_until_set(surround_mode, tv_level, music_level,
                                  surrounds_enabled, zp):
    """
    Helper function to set the TV and music surround levels, the
    surround enable/disable setting, and the surround mode (ambient/full)
    :param str surround_mode: either 'ambient' or 'full'
    :param int tv_level: an int between -15 and 15
    :param int music_level: an int between -15 and 15
    :param bool surrounds_enabled: True - surrounds on , False - surrounds off
    :param zp: Sonos Zone Component to apply settings to
    """
    zp.RenderingControl.set_surround_mode(surround_mode)

    pytest.wait_for_assert(
        lambda: zp.RenderingControl.get_surround_mode() == surround_mode,
        timeout_seconds=N_SECONDS_EQ_CHANGE_PROPAGATION_DELAY,
        iteration_delay=1,
        msg="Waiting for surround mode to change to {} on {}".format(
            surround_mode, zp
        ))

    zp.RenderingControl.set_toslink_surround_level(tv_level)

    pytest.wait_for_assert(
        lambda: zp.RenderingControl.get_toslink_surround_level() == tv_level,
        timeout_seconds=N_SECONDS_EQ_CHANGE_PROPAGATION_DELAY,
        iteration_delay=1,
        msg="Waiting for tv surround level to change to {} on {}".format(
            tv_level, zp
        ))

    zp.RenderingControl.set_surround_level(music_level)

    pytest.wait_for_assert(
        lambda: zp.RenderingControl.get_surround_level() == music_level,
        timeout_seconds=N_SECONDS_EQ_CHANGE_PROPAGATION_DELAY,
        iteration_delay=1,
        msg="Waiting for music surround level to change to {} on {}".format(
            music_level, zp
        ))

    zp.RenderingControl.set_surround_enable(surrounds_enabled)

    pytest.wait_for_assert(
        lambda:
        zp.RenderingControl.are_surrounds_enabled() == surrounds_enabled,
        timeout_seconds=N_SECONDS_EQ_CHANGE_PROPAGATION_DELAY,
        iteration_delay=1,
        msg="Waiting for surrounds enabled to change to {} on {}".format(
            surrounds_enabled, zp
        ))

    guaranteed_sleep(N_SECONDS_EQ_CHANGE_PROPAGATION_DELAY)

@pytest.mark.parametrize('toslink_source',
                         [(LOCAL_TEST_FILE_PATH,
                           TEST_FILE_METADATA['mplayer_codec'])],
                         indirect=['toslink_source'])
def test_surround_levels_and_enable_disable(
        user_controls_test_configured_ht_50_system,
        toslink_source,
        track_uri):
    """
    Given a 5.0 HT configuration of players, ensure that the surround level
    controls and enable-disable controls directly affect the RMS level of
    the audiotap captured during both music and TV playback in an expected way.
    :param user_controls_test_configured_ht_50_system:
    :param toslink_source: source of optical content
    :param track_uri: URI to test track
    """
    pytest.logger.info(user_controls_test_configured_ht_50_system)
    zp = user_controls_test_configured_ht_50_system[0]
    lsat = user_controls_test_configured_ht_50_system[1]
    rsat = user_controls_test_configured_ht_50_system[2]
    sats = [lsat, rsat]
    zps = [zp, lsat, rsat]
    zp.RenderingControl.SetVolume(70)

    # set TV level
    set_levels_and_wait_until_set(surround_mode='full',
                                  tv_level=BASELINE_CONTROLLER_LEVEL,
                                  music_level=BASELINE_CONTROLLER_LEVEL,
                                  surrounds_enabled=True,
                                  zp=zp)
    prime_playback(zp, track_uri)

    # obtain baselines
    tv_baseline, tv_audiotap_names = \
        rms_helper.get_rms_levels_from_toslink_playback(
            ht_master=zp, zps=zps, toslink_source=toslink_source,
            notes="tv_baseline", remove_audiotap_files=REMOVE_AUDIOTAP_FILES,
            return_audiotap_names=True)
    guaranteed_sleep(ITERATION_DELAY_SEC)
    music_baseline, music_audiotap_names = \
        rms_helper.get_rms_levels_from_music_playback(
            zp=zp, zps=zps, uri=track_uri, notes="music_baseline",
            return_audiotap_names=True)

    # establish test parameters
    sources = ["tv", "music"]
    tv_levels = [-15, BASELINE_CONTROLLER_LEVEL]
    music_levels = [-15, BASELINE_CONTROLLER_LEVEL]
    surrounds_settings = [True, False]

    # iterate over all permutations
    for source in sources:
        for tv_level in tv_levels:
            for music_level in music_levels:
                for surrounds_on in surrounds_settings:

                    dump_settings(zp)

                    notes = "ML={}_TL={}_S={}_SE={}".format(
                        music_level, tv_level, source, surrounds_on
                    )

                    # note test case
                    pytest.logger.info(
                        "Testing Source: {},\n"
                        "Surrounds TV Level: {},\n"
                        "Surrounds Music Level: {},\n"
                        "Surrounds Enabled: {}".format(
                            source, tv_level, music_level, surrounds_on))
                    # setup as necessary
                    set_levels_and_wait_until_set(
                        surround_mode='full',
                        tv_level=tv_level,
                        music_level=music_level,
                        surrounds_enabled=surrounds_on,
                        zp=zp)
                    prime_playback(zp, track_uri)
                    save_get_dsp(zp, "{}_getDSP.xml".format(notes))
                    # if surrounds are not on, we expect an exception to be
                    # raised when trying to capture an audiotap.
                    if not surrounds_on:
                        with pytest.raises(TimeoutError) as e:
                            if source == "music":
                                rms_helper.get_rms_levels_from_music_playback(
                                        zp=zp, zps=zps, uri=track_uri,
                                        notes=notes,
                                        remove_audiotap_files=REMOVE_AUDIOTAP_FILES,
                                        return_audiotap_names=True)
                            else:
                                rms_helper.get_rms_levels_from_toslink_playback(
                                        ht_master=zp, zps=zps,
                                        toslink_source=toslink_source,
                                        notes=notes,
                                        remove_audiotap_files=REMOVE_AUDIOTAP_FILES,
                                        return_audiotap_names=True)
                        guaranteed_sleep(ITERATION_DELAY_SEC)
                        continue

                    if source == "music":
                        rms_levels, audiotap_names = \
                            rms_helper.get_rms_levels_from_music_playback(
                                zp=zp, zps=zps, uri=track_uri, notes=notes,
                                remove_audiotap_files=REMOVE_AUDIOTAP_FILES,
                                return_audiotap_names=True)

                    else:
                        rms_levels, audiotap_names = \
                            rms_helper.get_rms_levels_from_toslink_playback(
                                ht_master=zp, zps=zps,
                                toslink_source=toslink_source, notes=notes,
                                remove_audiotap_files=REMOVE_AUDIOTAP_FILES,
                                return_audiotap_names=True)

                    comparison_audiotap_names = music_audiotap_names if \
                        source == "music" else tv_audiotap_names
                    all_audiotap_names = audiotap_names + \
                                         comparison_audiotap_names

                    rms_helper.verify_rms_levels(
                      ht_master=zp,
                      sats=sats,
                      source=source,
                      music_level=music_level,
                      toslink_level=tv_level,
                      surrounds_enabled=surrounds_on,
                      rms_levels=rms_levels,
                      baseline_music_levels=music_baseline,
                      baseline_tv_levels=tv_baseline,
                      baseline_level=BASELINE_CONTROLLER_LEVEL,
                      audiotap_names=all_audiotap_names)

                    guaranteed_sleep(ITERATION_DELAY_SEC)
