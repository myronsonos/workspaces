"""
This module is intended to host tests for verifying the functionality of
various bonded subwoofer controls in a standalone + sub configuration,
where the standalone player is one without built-in speakers, namely a
connect:amp or a hideout
"""
import pytest
from audio_util import rms_helper
from sleep import guaranteed_sleep
from conftest import (LOCAL_SERVER_PATH, ITERATION_DELAY_SEC, prime_playback,
                      save_get_dsp)

TEST_VOLUME = 80
TEST_DURATION_SEC = 15
CONTROL_PROPAGATION_DELAY_SEC = 5
DEFAULT_SUB_XOVER_FREQ = 500
TRANSITION_SUB_XOVER_FREQ = 30
DEFAULT_SPEAKER_SIZE = 2
TRANSITION_SPEAKER_SIZE = 0
LOW_FREQ_TONE_HZ = 30
HIGH_FREQ_TONE_HZ = 150
LOW_FREQUENCY_DIFF_THRESHOLD_DB = 3
REMOVE_TAPS = False

pytestmark = [
    pytest.mark.usefixtures('disable_audio_output')
]

[docs]def run_rms_based_freq_test(zp, zps, webserver,
                            speaker_size_start, speaker_size_end,
                            crossover_start, crossover_end,
                            test_str):
    """
    This common method is used to set the given zp and bonded sub's
    crossover and speaker size, play back a higher frequency and lower
    frequency tone, and verify that after changing the speaker size and/or
    crossover frequency that the RMS values decrease as expected for the
    different frequency tones.
    :param zp: the HT master or other ZP that the sub is bonded to
    :param zps: both the HT master and the sub [zp, sub]
    :param webserver: the webserver hosting the content
    :param speaker_size_start: the value to initialize the speaker size at
    :param speaker_size_end: the value to set the speaker size to for the
    second measurement
    :param crossover_start: the value to initialize the sub crossover at
    :param crossover_end: the value to set the sub crossover to for the
    second measurement
    :param str test_str: test string to be included in audiotap names
    """
    # if it's an SH4 player, don't capture
    if zp.get_arch() == "sh4":
        pytest.skip("Skipping - SH4 has no DSP out audio tap")

    # generate the tone URIs
    high_freq_tone_uri = 'http://{}/{}'.format(
        webserver.host(), '{}hz.wav'.format(HIGH_FREQ_TONE_HZ))
    low_freq_tone_uri = 'http://{}/{}'.format(
        webserver.host(), '{}hz.wav'.format(LOW_FREQ_TONE_HZ))

    # get the sub and its sn
    sub = [z for z in zps if z.is_sub()][0]
    sub_sn = sub.serialNumber

    # setup
    zp.RenderingControl.SetVolume(TEST_VOLUME)
    zp.RenderingControl.SetMute(False)
    zp.RenderingControl.SetEQ("VolumeScalingFactor", 100)
    zp.cli.amp_power_off()
    sub.cli.amp_power_off()

    # set crossover and speaker size to initial values, allowing for
    # adequate propagation time
    zp.RenderingControl.set_sub_crossover(crossover_start)
    prime_playback(zp, low_freq_tone_uri)
    guaranteed_sleep(CONTROL_PROPAGATION_DELAY_SEC)

    # if it's an HT master, then we can't set the speaker size directly on
    # the sub. we must do it on the ZP.
    if zp.is_ht_master_capable():
        zp.RenderingControl.set_speaker_size(speaker_size_start)
    else:
        sub.RenderingControl.set_speaker_size(speaker_size_start)
    guaranteed_sleep(CONTROL_PROPAGATION_DELAY_SEC)

    # specify the notes to include in the filename and take the taps
    notes = get_notes(speaker_size_start, crossover_start,
                      zp.modelNumber, HIGH_FREQ_TONE_HZ, test_str)
    save_get_dsp(zp, "{}_getDSP.xml".format(notes))
    rms_levels_start_high_freq = \
        rms_helper.get_rms_levels_from_music_playback(
            zp=zp, zps=zps,
            uri=high_freq_tone_uri, notes=notes,
            duration=TEST_DURATION_SEC, remove_audiotap_files=REMOVE_TAPS,
            retry_if_tap_entirely_zeros=True,
            skip_start_seconds=1.0,
            skip_end_seconds=1.0)
    guaranteed_sleep(ITERATION_DELAY_SEC)

    # specify the notes to include in the filename and take the taps
    notes = get_notes(speaker_size_start, crossover_start,
                      zp.modelNumber, LOW_FREQ_TONE_HZ, test_str)
    rms_levels_start_low_freq = \
        rms_helper.get_rms_levels_from_music_playback(
            zp=zp, zps=zps,
            uri=low_freq_tone_uri, notes=notes,
            duration=TEST_DURATION_SEC, remove_audiotap_files=REMOVE_TAPS,
            retry_if_tap_entirely_zeros=True,
            skip_start_seconds=1.0,
            skip_end_seconds=1.0
        )
    guaranteed_sleep(ITERATION_DELAY_SEC)

    # set crossover and speaker size to ending values
    # NOTE: the player does not seem to respond to having both its crossover
    #  and speaker size toggled simultaneously. to account for this, we only
    #  change the values if we absolutely need to for the test.
    if crossover_end != crossover_start:
        zp.RenderingControl.set_sub_crossover(crossover_end)
    if speaker_size_end != speaker_size_start:
        if zp.is_ht_master_capable():
            zp.RenderingControl.set_speaker_size(speaker_size_end)
        else:
            sub.RenderingControl.set_speaker_size(speaker_size_end)

    # NOTE: we play briefly here to "flush" the "woof" out of the following
    # dspout tap. the woof severely throws off the RMS measurements for the
    # audiotap on the bonded sub tests. SWPBL-105948
    prime_playback(zp, low_freq_tone_uri)

    guaranteed_sleep(CONTROL_PROPAGATION_DELAY_SEC)

    # specify the notes to include in the filename and take the taps
    notes = get_notes(speaker_size_end, crossover_end,
                      zp.modelNumber, HIGH_FREQ_TONE_HZ, test_str)
    save_get_dsp(zp, "{}_getDSP.xml".format(notes))
    rms_levels_end_high_freq = \
        rms_helper.get_rms_levels_from_music_playback(
            zp=zp, zps=zps,
            uri=high_freq_tone_uri, notes=notes,
            duration=TEST_DURATION_SEC, remove_audiotap_files=REMOVE_TAPS,
            retry_if_tap_entirely_zeros=True,
            skip_start_seconds=1.0,
            skip_end_seconds=1.0
        )
    guaranteed_sleep(ITERATION_DELAY_SEC)

    # specify the notes to include in the filename and take the taps
    notes = get_notes(speaker_size_end, crossover_end,
                      zp.modelNumber, LOW_FREQ_TONE_HZ, test_str)
    rms_levels_end_low_freq = \
        rms_helper.get_rms_levels_from_music_playback(
            zp=zp, zps=zps,
            uri=low_freq_tone_uri, notes=notes,
            duration=TEST_DURATION_SEC, remove_audiotap_files=REMOVE_TAPS,
            retry_if_tap_entirely_zeros=True,
            skip_start_seconds=1.0,
            skip_end_seconds=1.0
        )
    guaranteed_sleep(ITERATION_DELAY_SEC)

    # run verifications now
    for ch in rms_levels_start_high_freq[sub_sn]:

        r1 = rms_levels_start_low_freq[sub_sn][ch]
        r2 = rms_levels_end_low_freq[sub_sn][ch]

        pretty_print_rms_results(LOW_FREQ_TONE_HZ, crossover_start,
                                 speaker_size_start, r1)
        pretty_print_rms_results(LOW_FREQ_TONE_HZ, crossover_end,
                                 speaker_size_end, r2)

        # the low frequency tones should not vary by much at all
        assert abs(r1 - r2) < LOW_FREQUENCY_DIFF_THRESHOLD_DB, \
            "Expect low frequency levels to be closer to one another:" \
            "{} vs {}".format(r1, r2)

        r1 = rms_levels_start_high_freq[sub_sn][ch]
        r2 = rms_levels_end_high_freq[sub_sn][ch]

        pretty_print_rms_results(HIGH_FREQ_TONE_HZ, crossover_start,
                                 speaker_size_start, r1)
        pretty_print_rms_results(HIGH_FREQ_TONE_HZ, crossover_end,
                                 speaker_size_end, r2)

        # the high frequency tone in the ending capture should be noticeably
        # lower than the start since the high frequencies have been attenuated
        assert r1 > r2 + 10, "Expect high frequencies to attenuate: " \
                             "{} vs {}".format(r1, r2)




[docs]@pytest.mark.parametrize('webserver', [LOCAL_SERVER_PATH], indirect=True)
@pytest.mark.rms
@pytest.mark.crossover
def test_amp_sub_crossover_controls(user_controls_test_configured_amp_and_sub,
                                    webserver):
    """
    This test verifies that the sub crossover controls attenuate higher
    frequency signals when lowered and do not attenuate lower frequency
    tones regardless of crossover
    :param user_controls_test_configured_amp_and_sub: a list of a zp and a
    bonded sub
    :param webserver: a server, hosting playback files
    """
    zp = user_controls_test_configured_amp_and_sub[0]
    sub = user_controls_test_configured_amp_and_sub[1]

    # vary only the crossover frequency
    run_rms_based_freq_test(zp=zp, zps=[zp, sub],
                            webserver=webserver,
                            speaker_size_start=DEFAULT_SPEAKER_SIZE,
                            speaker_size_end=DEFAULT_SPEAKER_SIZE,
                            crossover_start=DEFAULT_SUB_XOVER_FREQ,
                            crossover_end=TRANSITION_SUB_XOVER_FREQ,
                            test_str="test_xover")
