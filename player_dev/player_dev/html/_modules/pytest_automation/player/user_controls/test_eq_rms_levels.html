from sonos.services.common import wait_until_true

TIMEOUT_EQ_CHANGE = 5

# max tolerance
MAX_CHANNEL_DIFF = 1.5  # dB rule-of-thumb estimate
MIN_VOL_DIFF = 10.0  # dB rule-of-thumb estimate, from ui 50 to 100

# start/end eq settings for bass and treble
EQ_START_LEVEL = -9
EQ_END_LEVEL = 9
VOL_START_LEVEL = 50
VOL_END_LEVEL = 100
RMS_TEST_DURATION = 3  # seconds of playback

pytestmark = [
    pytest.mark.usefixtures('disable_speaker_audio')

[docs]def get_volume_level(zp):
    """
    Get the level of the volume in DB from the DSP of the ZP
    :param zp: Sonos Zone Component
    :return: float value representing the treble level on the ZP
    """
    if zp.get_arch() == "sh4":
        params = zp.diag.get_dsp_block('DSPSystemSH4').get(
            'parameter').get('param')
        vol = [p for p in params if p.name == "Volume"][0]
        return float(vol.value)

    vol = zp.diag.get_dsp_block('Volume')
    vol_params = vol.get('parameter').get('channel_0')
    vol_gain = [p for p in vol_params if p.name == 'targetGaindB_0'][0]
    volume_level = vol_gain.value
    pytest.logger.info("Volume level: {}".format(volume_level))
    return float(volume_level)




[docs]def set_volume_to_level_and_wait(zp, level, exp):
    """
    Set the bass level to the level given and wait for the value grabbed
    from the DSP config to match exp.
    This function will throw a timeout error if it fails
    :param zp: Sonos Zone Component
    :param level: the level to call the RenderingControl method with
    :param exp: the level to expect when parsing the DSP config
    """
    zp.RenderingControl.SetVolume(level)
    pytest.logger.info("Waiting for bass to go to {}".format(exp))
    wait_until_true(predicate=lambda: zp.RenderingControl.GetVolume() == exp,
                    timeout_seconds=TIMEOUT_EQ_CHANGE,
                    reason="Waiting for level to adjust")

def test_treble_rms(playback_device, track_uri, rms_dict, rms_ref):
    """
    look at rms of target channel, make sure there are less than 1.5 dB differences on
    each channel. if the channel is empty and the result is negative infinity, make sure
    it is identical to the reference
    :param playback_device: pytest fixture playback device selection
    :param track_uri: pytest fixture track playback URI
    :param rms_dict: pytest fixture dictionary of rms values to store
    :param rms_ref: pytest fixture of rms values to compare to
    """
    zp = playback_device
    uri = track_uri

    zp.RenderingControl.ResetBasicEQ()
    play_music_on_player(zp, track_uri)

    ref = rms_ref[zp.modelNumber]['pre-treble']
    test = get_pre_treble_rms(zp, uri)
    rms_dict[zp.modelNumber]['pre-treble'] = test
    pre_diffs = analyze_diffs(test, ref)

    ref = rms_ref[zp.modelNumber]['post-treble']
    test = get_post_treble_rms(zp, uri)
    rms_dict[zp.modelNumber]['post-treble'] = test
    post_diffs = analyze_diffs(test, ref)

    assert not np.any(pre_diffs + post_diffs)

def test_volume_control(playback_device, track_uri, rms_dict, rms_ref):
    """

    :param track_uri: pytest fixture track playback URI

[docs]def analyze_diffs(test, ref):
    """
    this for loop goes through all the channels, looks at the
    differences. if the differences in any channel is more than the thresh,
    mark it as a difference found with a 1 in your list
    prepost should be 'pre' or 'post'
    :param test: dict of test rms values per channel
    :param ref: dict of reference rms values per channel
    :rtype: :obj:`dict`
    """
    diffs_found = np.zeros(len(test))

    # add len(test) to indexing to account for pre-post
    for k in test:
        if np.isneginf(test[k]) and (test[k] != ref[str(k)]):
            diffs_found[int(k)] = 1
        elif (test[k] - ref[str(k)]) > MAX_CHANNEL_DIFF:  # dB
            diffs_found[k] = 1
        else:
            continue

    return diffs_found




[docs]def get_pre_treble_rms(zp, uri):
    """ get treble rms under test
    :param zp: Sonos Zone Component
    :param uri: Test track URI shared through *webserver*
    :rtype: :obj:`dict`
    """
    zp.RenderingControl.SetTreble(EQ_START_LEVEL)
    wait_until_true(
        predicate=lambda: get_treble_level(zp) <= (EQ_START_LEVEL + 1),
                    timeout_seconds=TIMEOUT_EQ_CHANGE,
                    reason="Waiting treble drop, actual {}".format(
                        get_treble_level(zp)))

    note = "treble={}".format(EQ_START_LEVEL)
    before_rms = rms_helper.get_rms_levels_from_music_playback(
        ht_master=zp, zps=[zp], uri=uri, notes=note, duration=RMS_TEST_DURATION)

    return before_rms.values()[0]
