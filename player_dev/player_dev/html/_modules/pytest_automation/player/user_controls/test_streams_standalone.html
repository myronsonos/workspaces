[docs]def setup_player_rc(zp, volume=LOW_VOLUME, volume_limit=100, loudness=False,
                    left_level=LOW_VOLUME, right_level=LOW_VOLUME, mute=False):
    """
    Generic function to set the rendering control settings up as desired for
    the given player

    :param zp: zone player to set the rendering control levels for
    :param int volume: normal volume
    :param int volume_limit: volume scaling factor (max volume limit)
    :param bool loudness: whether or not to turn loudness on
    :param int left_level: left channel volume level
    :param int right_level: right channel volume level
    :param bool mute: whether or not to mute player
    """
    zp.RenderingControl.SetVolume(volume)
    zp.RenderingControl.SetEQ('VolumeScalingFactor', volume_limit)
    zp.RenderingControl.SetLoudness(loudness)
    zp.RenderingControl.SetVolume(left_level, 'LF')
    zp.RenderingControl.SetVolume(right_level, 'RF')
    zp.RenderingControl.SetMute(mute)




[docs]def skip_if_unsupported_stream(zp, stream):
    """
    Skip if the stream is unsupported by the zone player

    :param zp: zone player
    :param str stream: which stream is to be tested
    """
    if stream == FEEDBACK and zp.modelNumber not in CAP_TOUCH_DEVICES:
        pytest.skip("Device does not support feedback: {}".format(zp))
    if stream == EXT_AUDIO and zp.modelNumber not in \
            SUPPORTS_EXTERNAL_AUDIO_STREAM_DEVICES:
        pytest.skip("Device does not support external audio "
                    "stream: {}".format(zp))
    if zp.get_arch() == "sh4":
        pytest.skip("SH4 not supported arch for audiotaps")




[docs]def get_rms_levels_for_stream(zp, uri, notes, stream, muse,
                              capture_zps):
    """
    Get the RMS levels for playback of a certain stream

    :param zp: the zone player to obtain the RMS levels for
    :param str uri: URI to an audio file to play back for music or ext audio
    :param str notes: string to include in audiotap filename
    :param str stream: the stream to capture audio for, one of STREAMS
    :param muse: muse object for handling sending of muse commands
    :param list capture_zps: the list of zone players to capture audiotaps for
    :return: a dictionary of RMS values for the ZP, where the
        serial number is the key and the value is a float representing the RMS
        value across all channels in the audiotap file.
    """

    skip_if_unsupported_stream(zp, stream)
    pytest.logger.info("Getting RMS levels for stream: {}".format(stream))

    if stream == PLAYBACK:
        rms_levels = rms_helper.get_rms_levels_from_music_playback(
            zp=zp, zps=capture_zps, uri=uri,
            notes=notes, remove_audiotap_files=REMOVE_TAPS,
            duration=PLAYBACK_DURATION)

    elif stream == FEEDBACK:

        rms_levels = rms_helper.get_rms_levels_from_feedback(
            zp=zp, zps=capture_zps, duration=PLAYBACK_DURATION,
            notes=notes, remove_audiotap_files=REMOVE_TAPS,
            skip_start_seconds=0.8)

    elif stream == EXT_AUDIO:

        rms_levels = rms_helper.get_rms_levels_from_external_audio(
            zp=zp, zps=capture_zps, uri=uri, muse=muse,
            notes=notes, remove_audiotap_files=REMOVE_TAPS,
            duration=PLAYBACK_DURATION)

    else:
        raise NotImplementedError(
            "Unrecognized stream type: {}".format(stream))

    return rms_levels




[docs]def channel_is_silent(rms_level):
    """
    General function to check if RMS level is within silence threshold

    :param int rms_level: the RMS level in question
    :return: True if less than threshold
    """
    return rms_level < SILENCE_THRESH_DB

@pytest.mark.mute
def test_mute_stream(track_uri, mute_state, stream, muse_and_zp):
    """
    This test checks to ensure that the playback stream gets muted by the
    mute command and the other streams (external audio and feedback) do not
    get muted by the mute command.

    :param muse_and_zp: muse and test zone player tuple
    :param str track_uri: the URI to an audio file for playback
    :param bool mute_state: the mute state to test (true is muted)
    :param str stream: the stream to test (playback, feedback or ext audio)
    """
    mc, zp = muse_and_zp
    skip_if_unsupported_stream(zp, stream)
    mute = mute_state == "Mute"
    setup_player_rc(zp=zp, volume=HIGH_VOLUME, mute=mute)
    notes = 'mute={}_stream={}'.format(mute, stream)
    prime_playback(zp, track_uri)
    save_get_dsp(zp, "{}_getDSP.xml".format(notes))

    rms_levels = get_rms_levels_for_stream(zp, track_uri, notes, stream, mc,
                                           [zp])
    sn = zp.serialNumber

    # go through each channel of the tap and check for any nonzero samples
    nonzero_audio_in_tap = False
    for ch in rms_levels[sn]:
        if rms_levels[sn][ch] > NEG_INF:
            nonzero_audio_in_tap = True

    # if it's a mute test for a playback stream, we should not see any
    # nonzero samples in the tap
    if mute and stream == PLAYBACK:
        assert not nonzero_audio_in_tap, \
            "Channel should be muted: {}".format(rms_levels)

    # if it's not a playback stream or if it is an unmuted playback stream
    # there should be some nonzero samples in the tap
    else:
        assert nonzero_audio_in_tap, \
            "Channel should have nonzero signal: {}".format(rms_levels)

@pytest.mark.volume
def test_volume_stream(muse_and_zp, track_uri, stream):
    """
    Test that volume changes affect the output RMS levels of each stream as
    expected.

    :param muse_and_zp: muse and test zone player tuple
    :param str track_uri: the URI to an audio file for playback
    :param str stream: the stream to test (playback, feedback or ext audio)
    """
    mc, zp = muse_and_zp
    skip_if_unsupported_stream(zp, stream)

    # check at three distinct volume that the levels consistently increase
    start_volume = 25
    volumes = [100]
    setup_player_rc(zp=zp, volume=start_volume)
    prime_playback(zp, track_uri)
    guaranteed_sleep(1)
    notes = 'vol={}_stream={}'.format(start_volume, stream)
    save_get_dsp(zp, "{}_getDSP.xml".format(notes))
    rms_levels_low = get_rms_levels_for_stream(zp, track_uri, notes, stream,
                                               mc, [zp])

    # check for each of the volume levels
    for volume in volumes:

        # need to sleep between audiotaps to allow socket disconnection
        guaranteed_sleep(ITERATION_DELAY_SEC)

        zp.RenderingControl.SetVolume(volume)
        prime_playback(zp, track_uri)
        notes = 'vol={}_stream={}'.format(volume, stream)
        save_get_dsp(zp, "{}_getDSP.xml".format(notes))
        rms_levels_high = get_rms_levels_for_stream(zp, track_uri, notes,
                                                    stream, mc, [zp])

        # check each channel of the audiotaps to verify that the channels
        # from the tap with the higher volume have consistently higher RMS
        # levels
        sn = zp.serialNumber
        for ch in rms_levels_high[sn]:

            # if it's an empty audiotap, then confirm the RMS level of the
            # other tap is also empty
            if channel_is_silent(rms_levels_high[sn][ch]):
                assert channel_is_silent(rms_levels_low[sn][ch]), \
                    "volume: {} - Expect both channels to be silent " \
                    "low volumes: {}, high volumes: {}".format(
                        volume, rms_levels_low, rms_levels_high)
                continue

            assert rms_levels_high[sn][ch] > rms_levels_low[sn][ch], \
                "volume: {} - Ch {} lower on low volume tap: " \
                "low volumes: {}, high volumes: {}".format(
                    volume, ch, rms_levels_low, rms_levels_high)

        rms_levels_low = rms_levels_high

@pytest.mark.loudness
def test_loudness_stream(muse_and_zp, track_uri, stream):
    """
    Verify that volume of the playback stream varies with loudness but that
    the volume of the feedback and external audio streams do not vary with
    loudness.

    :param muse_and_zp: muse and test zone player tuple
    :param str track_uri: the URI to an audio file for playback
    :param str stream: the stream to test (playback, feedback or ext audio)
    """
    # volume changes level of each stream
    mc, zp = muse_and_zp
    skip_if_unsupported_stream(zp, stream)
    test_volume = 30
    setup_player_rc(zp=zp, volume=test_volume, loudness=False)
    prime_playback(zp, track_uri)
    wait_for_loudness(zp, False)

    notes = 'loudness={}_vol={}_stream={}'.format(False, test_volume, stream)
    save_get_dsp(zp, "{}_getDSP.xml".format(notes))
    rms_levels_no_loud = get_rms_levels_for_stream(zp, track_uri, notes,
                                                   stream, mc, [zp])

    zp.RenderingControl.SetLoudness(True)
    zp.RenderingControl.SetVolume(test_volume)
    prime_playback(zp, track_uri)
    wait_for_loudness(zp, True)

    notes = 'loudness={}_vol={}_stream={}'.format(True, test_volume, stream)
    save_get_dsp(zp, "{}_getDSP.xml".format(notes))
    rms_levels_loud = get_rms_levels_for_stream(zp, track_uri, notes,
                                                stream, mc, [zp])

    # for each channel of the taps, verify that the RMS levels have
    # increased for the loudness enabled taps for playback streams but have
    # remained consistent for the other streams
    sn = zp.serialNumber
    for ch in rms_levels_no_loud[sn]:
        if channel_is_silent(rms_levels_no_loud[sn][ch]):
            assert channel_is_silent(rms_levels_loud[sn][ch]), \
                "Expect both channels to be silent if one is " \
                "loudness off: {}, loudness on: {}".format(
                    rms_levels_loud, rms_levels_no_loud)
            continue

        if stream == PLAYBACK:
            assert rms_levels_loud[sn][ch] > rms_levels_no_loud[sn][ch], \
                "Ch {} lower on loudness off tap: " \
                "loudness off: {}, loudness on: {}".format(
                    ch, rms_levels_no_loud, rms_levels_loud)
        else:
            assert abs(rms_levels_loud[sn][ch] -
                       rms_levels_no_loud[sn][ch]) < DIFFERENCE_THRESH_DB, \
                "Ch {} different level for stream on tap with regard to " \
                "Loudness: loudness off: {}, loudness on: {}".format(
                    ch, rms_levels_no_loud, rms_levels_loud)

    :param str stream: the stream to test (playback, feedback or ext audio)

    prime_playback(zp, track_uri)

                                                   HIGH_VOLUME, stream)
    save_get_dsp(zp, "{}_getDSP.xml".format(notes))

                                                    stream, mc, [zp])

    zp.RenderingControl.SetEQ("VolumeScalingFactor", low_vol_limit)
    prime_playback(zp, track_uri)

    sn = zp.serialNumber
    for ch in rms_levels_no_limit[sn]:
        if channel_is_silent(rms_levels_no_limit[sn][ch]):
            assert channel_is_silent(rms_levels_with_limit[sn][ch]), \
                "Expect both channels to be silent if one is " \
                "limited: {}, unlimited: {}".format(
                    rms_levels_with_limit, rms_levels_no_limit)
            continue

        assert rms_levels_no_limit[sn][ch] > rms_levels_with_limit[sn][ch], \
            "Ch {} lower on limited tap: " \
            "limited: {}, unlimited: {}".format(
                ch, rms_levels_with_limit, rms_levels_no_limit)

@pytest.mark.balance
def test_balance_stream(muse_and_zp, track_uri, stream):
    """
    Verify that balance control changes do not affect feedback or external
    audio stream.

    :param muse_and_zp: muse and test zone player tuple
    :param str track_uri: the URI to an audio file for playback
    :param str stream: the stream to test (playback, feedback or ext audio)
    """
    mc, zp = muse_and_zp
    skip_if_unsupported_stream(zp, stream)
    if zp.modelNumber not in BALANCE_CAPABLE_MODELS:
        pytest.skip("Player does not support balance adjustment")

    # NOTE: hideout is a feedback capable player whose balance is actually
    # modified by balance control. So for hideout, we are okay with the
    # feedback levels getting adjusted to left and right channels.
    if zp.is_hideout() and stream == FEEDBACK:
        pytest.skip("Skipping hideout feedback balance - not expected to be "
                    "impervious to balance")

    # as an effect of the above, the only player which does check feedback
    # balance is then encore. We must force its orientation to horizontal
    # though to test it.
    if zp.is_encore():
        zp.cli.set_device_orientation('horizontal')

    # reference balance settings - equal balance
    setup_player_rc(zp=zp, volume=HIGH_VOLUME, left_level=HIGH_VOLUME,
                    right_level=HIGH_VOLUME)
    prime_playback(zp, track_uri)
    notes = "left={}_right={}_stream={}".format(HIGH_VOLUME, HIGH_VOLUME,
                                                stream)
    save_get_dsp(zp, "{}_getDSP.xml".format(notes))
    rms_levels_balanced = get_rms_levels_for_stream(zp, track_uri, notes,
                                                    stream, mc, [zp])

    # iterate through each of the following balance variations
    balance_levels_settings = [
        [HIGH_VOLUME, LOW_VOLUME],
        [LOW_VOLUME, HIGH_VOLUME]]

    sn = zp.serialNumber
    for left, right in balance_levels_settings:

        # set the balance levels
        zp.RenderingControl.SetVolume(left, "LF")
        zp.RenderingControl.SetVolume(right, "RF")
        prime_playback(zp, track_uri)

        notes = "left={}_right={}_stream={}".format(left, right, stream)
        save_get_dsp(zp, "{}_getDSP.xml".format(notes))
        rms_levels_test = get_rms_levels_for_stream(zp, track_uri, notes,
                                                    stream, mc, [zp])
        guaranteed_sleep(ITERATION_DELAY_SEC)

        for ch in rms_levels_test[sn]:

            if channel_is_silent(rms_levels_balanced[sn][ch]):
                assert channel_is_silent(rms_levels_test[sn][ch]), \
                    "Expect both channels to be silent if one is " \
                    "balanced: {}, unbalanced: {}".format(
                        rms_levels_balanced, rms_levels_test)
                continue

            assert abs(rms_levels_test[sn][ch] -
                       rms_levels_balanced[sn][ch]) < DIFFERENCE_THRESH_DB, \
                "Levels should not change for non-playback streams: " \
                "balanced: {}, unbalanced: {}".format(
                    rms_levels_balanced, rms_levels_test)

    """
    Wait for the EQ field in local settings to change
    :param zp: zone player to wait on
    :param str eq_start: the EQ value at the start
    """
    wait_until_true(
        predicate=lambda:

            zp.cli.get_localsettings_as_dict()['EQ'] != eq_start,
        timeout_seconds=100,
        iteration_delay=33,
        reason="Waiting for EQ to change from {}".format(eq_start)
    )




[docs]def get_boot_volume_from_localsettings(zp):
    """
    Parse the currently stored master volume level for the given zone player
    from localsettings.txt
    :param zp: the zone player to check
    :return: int the current volume
    """
    eq = zp.cli.get_localsettings_as_dict()['EQ']
    amv = re.findall(r'AMV[0-9]*', eq)
    assert len(amv) > 0, "No AMV param in localsettings EQ: {}".format(eq)
    assert len(amv[0]) > 3, "Unexpected AMV format: {}".format(amv[0])
    return int(amv[0][3:])

def test_retain_settings_for_stream_on_restart(muse_and_zp):
    """
    Verify that volume balance, and loudness settings are retained for each
    stream during a restart

    :param muse_and_zp: muse and test zone player tuple
    """
    mc, zp = muse_and_zp

    # add a wait for the EQ to be present in the localsettings.txt file
    wait_for_eq_to_change(zp, eq_start="dummystring")

    setup_player_rc(zp, volume=LOW_VOLUME)

    # note: we need to make sure that we change at least one of the EQ
    # settings of the player so that the EQ in localsettings.txt will
    # actually change. we do this by adding one to the volume and modulo 50
    start_vol = (get_boot_volume_from_localsettings(zp) + 1) % 50
    right_level = 25
    left_level = 10
    loudness = True

    eq_start = zp.cli.get_localsettings_as_dict()['EQ']
    zp.RenderingControl.SetVolume(start_vol)
    zp.RenderingControl.SetVolume(left_level, "LF")
    zp.RenderingControl.SetVolume(right_level, "RF")
    zp.RenderingControl.SetLoudness(loudness)

    pre_vol = zp.RenderingControl.GetVolume()
    pre_left = zp.RenderingControl.GetVolume("LF")
    pre_right = zp.RenderingControl.GetVolume("RF")

    # there is a propagation delay in the volume and other EQ values being
    # saved to jffs/localsettings. we need to wait for this to occur before
    # we expect this to pass.
    wait_for_eq_to_change(zp, eq_start)

    # wait for device to begin reboot, then wait for it to come online
    zp.reboot()
    guaranteed_sleep(5)
    assert pytest.subnet.wait_for_all_environment_devices(), \
        "Waiting for devices to come back online"

    post_vol = zp.RenderingControl.GetVolume()
    post_left = zp.RenderingControl.GetVolume("LF")
    post_right = zp.RenderingControl.GetVolume("RF")

    log_str = "Master Volume - Before: {}, After: {}\n" \
              "Left Volume - Before: {}, After: {}\n" \
              "Right Volume - Before: {}, After: {}\n" \
              "Loudness - Before: {}, After: {}".format(
                pre_vol, post_vol, pre_left, post_left, pre_right, post_right,
                loudness, zp.RenderingControl.GetLoudness())

    assert zp.RenderingControl.GetVolume() == pre_vol, \
        "Volume should remain same on anacapa restart\n{}".format(log_str)

    assert zp.RenderingControl.GetVolume("LF") == pre_left, \
        "Left level should remain same on anacapa restart\n{}".format(log_str)

    assert zp.RenderingControl.GetVolume("RF") == pre_right, \
        "Right level should remain same on anacapa restart\n{}".format(log_str)

    assert zp.RenderingControl.GetLoudness() == loudness, \
        "Loudness should remain same on anacapa restart\n{}".format(log_str)

@pytest.mark.min_volume
def test_min_volume_per_stream(muse_and_zp, track_uri, stream):
    """
    Verify that the minimum volume for playback is 0 and that the minimum
    volume for feedback and external audio is 16. This is done by taking RMS
    measurements for each stream at volume levels 16 and 0 on the ZP and doing
    comparisons.

    :param muse_and_zp: muse and test zone player tuple
    :param str track_uri: the URI to an audio file for playback
    :param str stream: the stream to test (playback, feedback or ext audio)
    """
    mc, zp = muse_and_zp
    skip_if_unsupported_stream(zp, stream)
    # verify that the levels for cap feedback and streams are similar when
    # volume is set at 16 and 0. We set it at 14 so that there' wiggle room
    # for the volume variations that happen during the cap touch button presses
    # going from 16 to 17 for example
    test_volume = 14

    setup_player_rc(zp, volume=test_volume)
    prime_playback(zp, track_uri)

    notes = "volume={}_stream={}_minvoltest".format(test_volume, stream)
    save_get_dsp(zp, "{}_getDSP.xml".format(notes))
    rms_levels_base = get_rms_levels_for_stream(zp, track_uri, notes,
                                                stream, mc, [zp])
    guaranteed_sleep(ITERATION_DELAY_SEC)

    change_volume = 0
    zp.RenderingControl.SetVolume(change_volume)
    prime_playback(zp, track_uri)

    notes = "volume={}_stream={}_minvoltest".format(change_volume, stream)
    save_get_dsp(zp, "{}_getDSP.xml".format(notes))
    rms_levels_zero = get_rms_levels_for_stream(zp, track_uri, notes,
                                                stream, mc, [zp])

    sn = zp.serialNumber
    non_silent_audio_detected = False
    for ch in rms_levels_base[sn]:
        if channel_is_silent(rms_levels_base[sn][ch]):
            assert channel_is_silent(rms_levels_zero[sn][ch]), \
                "Expect both channels to be silent if one is " \
                "base: {}, zero: {}".format(rms_levels_base,
                                            rms_levels_zero)
            continue

        non_silent_audio_detected = True

        # if it's a playback stream, then the levels should have attenuated
        # with the drop in volume level
        if stream == PLAYBACK:
            assert rms_levels_base[sn][ch] > rms_levels_zero[sn][ch], \
                "Ch {} lower on limited tap: " \
                "base: {}, zero: {}".format(ch, rms_levels_base,
                                            rms_levels_zero)

        # otherwise, the values should more or less be the same
        # for beam, we skip this check because the external audio levels can
        #  actually vary below 14 when the volume is changed.
        elif not zp.is_elrey():
            assert abs(rms_levels_base[sn][ch] -
                       rms_levels_zero[sn][ch]) < DIFFERENCE_THRESH_DB, \
                "Levels should not change for non-playback streams: " \
                "base: {}, zero: {}".format(rms_levels_base,
                                            rms_levels_zero)

    assert non_silent_audio_detected, \
        "Detected no audio - low: {}, zero: {}".format(
            rms_levels_base, rms_levels_zero)




[docs]@pytest.mark.parametrize("extension", ["mp3"], ids=["FileType=.mp3"])
@pytest.mark.parametrize("stream", [PLAYBACK, EXT_AUDIO],
                         ids=[PLAYBACK, EXT_AUDIO])
@pytest.mark.normalization
def test_normalization_stream(muse_and_zp, webserver, stream,
                              extension):
    """
    Verify that volume normalization (replay gain) is applied when playing
    tracks with replay gain on the playback or external audio streams.
    In this test case, we apply -12dB of replay gain to the track (pink
    noise) and check that the RMS levels drop during playback.

    :param muse_and_zp: muse and test zone player tuple
    :param webserver: a server hosting audio files
    :param str stream: the stream to test (playback, feedback or ext audio)
    :param str extension: the file extension to use for the pink noise test
        track (any codec that supports replay gain / normalization)
    """

    mc, zp = muse_and_zp
    skip_if_unsupported_stream(zp, stream)
    setup_player_rc(zp, volume=HIGH_VOLUME)

    no_gain_uri = "http://{}".format(
        os.path.join(webserver.host(),
                     'pink.{}'.format(extension)))

    # -12dB replay gain
    gain_uri = "http://{}".format(
        os.path.join(webserver.host(),
                     'pink_norm.{}'.format(extension)))

    # TODO: this is retry logic that's been baked in because something is
    # intermittently causing the normalization not to be applied to the
    # playback stream in some instances. we give it a few tries before
    # saying that the test failed entirely.
    passed = False
    max_iterations = 3
    n_iterations = max_iterations
    while not passed and n_iterations > 0:

        pytest.logger.info("Attempt #{}".format(max_iterations - n_iterations))
        prime_playback(zp, no_gain_uri)

        notes = "no_gain_stream={}_codec={}".format(stream, extension)
        rms_no_gain = get_rms_levels_for_stream(zp, no_gain_uri, notes, stream,
                                                mc, [zp])

        zp.AVTransport.clear_play_state()
        guaranteed_sleep(ITERATION_DELAY_SEC)

        notes = "gain_stream={}_codec={}".format(stream, extension)
        rms_gain = get_rms_levels_for_stream(zp, gain_uri, notes,
                                             stream, mc, [zp])

        pytest.logger.info("Checking results with rms's\n"
                           "No Gain: {}\n"
                           "Gain:    {}\n".format(rms_no_gain, rms_gain))

        passed = True
        sn = zp.serialNumber
        for ch in rms_no_gain[sn]:
            if rms_no_gain[sn][ch] == NEG_INF:
                # if the channel with no negative gain has an RMS of -inf,
                # the channel with negative gain should also be -inf
                if NEG_INF != rms_gain[sn][ch]:
                    passed = False
                    break
                continue

            # otherwise, we should see at least a 9dB difference in the
            # output RMS level
            if rms_no_gain[sn][ch] < rms_gain[sn][ch] + 9:
                passed = False
                break

        n_iterations -= 1
        guaranteed_sleep(ITERATION_DELAY_SEC)

    # run formal assertions here to confirm pass/fail
    sn = zp.serialNumber
    for ch in rms_no_gain[sn]:
        if rms_no_gain[sn][ch] == NEG_INF:
            assert NEG_INF == rms_gain[sn][ch], \
                "Expect both channels to be {} if one is " \
                "no gain: {}, gain: {}".format(NEG_INF, rms_no_gain, rms_gain)
            continue

        assert rms_no_gain[sn][ch] > rms_gain[sn][ch] + 9, \
            "Ch {} is lower RMS level on tap with no replay gain: " \
            "no gain: {}, gain: {}".format(ch, rms_no_gain, rms_gain)

[docs]@pytest.mark.parametrize("stream", [EXT_AUDIO], ids=[EXT_AUDIO])
@pytest.mark.downmix
def test_external_audio_downmix_ht(muse_and_zp, webserver, stream):
    """
    Verify that the external audio signal is downmixed on HT players to the
    center channel.

    :param muse_and_zp: muse and test zone player tuple
    :param webserver: a server hosting audio files
    :param str stream: the stream to test (playback, feedback or ext audio)
    """
    mc, zp = muse_and_zp
    skip_if_unsupported_stream(zp, stream)
    if zp.modelNumber not in DOWNMIX_CHANNELS:
        pytest.skip("Device does not support external audio downmix")

    channel_map = DOWNMIX_CHANNELS[zp.modelNumber]

    no_right_uri = "http://{}".format(
        os.path.join(webserver.host(),
                     'PinkUncorrelated2Minutes_no_right.wav'))
    no_left_uri = "http://{}".format(
        os.path.join(webserver.host(),
                     'PinkUncorrelated2Minutes_no_left.wav'))

    setup_player_rc(zp, volume=HIGH_VOLUME)
    # NOTE: we just need to give the player something to flush the changes
    # that have been applied via rendering control. it doesn't matter what
    # we play, so we play the no_left_uri track.
    prime_playback(zp, no_left_uri)

    notes = "left_stream={}".format(stream)
    rms_left = get_rms_levels_for_stream(zp, no_left_uri, notes, stream, mc,
                                         [zp])

    guaranteed_sleep(ITERATION_DELAY_SEC)

    notes = "right_stream={}".format(stream)
    rms_right = get_rms_levels_for_stream(zp, no_right_uri, notes, stream, mc,
                                          [zp])

    sn = zp.serialNumber
    pytest.logger.info("no left: {}\nno right:{}".format(
        rms_right[sn], rms_left[sn]))

    # when a left-only signal is played, it should get down-mixed and there
    # should be equal RMS levels between the left and right channels.
    for ch in channel_map['lefts']:
        rms_diff = abs(rms_left[sn][ch] - rms_right[sn][ch])
        assert rms_diff < DIFFERENCE_THRESH_DB, \
            "Left channel should be about equal to right channel. ch {} " \
            "left only: {}, right only: {}".format(ch, rms_left, rms_right)

    # when a right-only signal is played, it should get down-mixed and there
    # should be equal RMS levels between the left and right channels.
    for ch in channel_map['rights']:
        rms_diff = abs(rms_left[sn][ch] - rms_right[sn][ch])
        assert rms_diff < DIFFERENCE_THRESH_DB, \
            "Left channel should be about equal to right channel. ch {} " \
            "left only: {}, right only: {}".format(ch, rms_left, rms_right)

    # the center channel taps of both the left-only and right-only signals
    # should be approximately equal.
    for ch in channel_map['centers']:
        assert abs(rms_left[sn][ch] -
                   rms_right[sn][ch]) < DIFFERENCE_THRESH_DB, \
            "Left channel should be ~= right channels. ch {} " \
            "left only: {}, right only: {}".format(ch, rms_left, rms_right)
