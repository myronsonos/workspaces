[docs]def generate_tap_name(player, note):
    """
    Common method for generating audiotap file names.

    :param player: zone player object
    :param str note: custom note to attach
    :return: str the filename generated
    """
    return "{}_{}_{}_{}.gld".format(player.modelNumber,
                                    player.serialNumber,
                                    note, time.time())




[docs]def setup_primary_zp(zp):
    """
    Common setup process for the primary zp in any bonded/standalone config

    :param zp: zone player object
    """
    zp.RenderingControl.SetVolume(100)
    zp.RenderingControl.SetMute(False)
    if zp.is_ht_master_capable():
        zp.RenderingControl.set_surround_level(15)
        zp.RenderingControl.set_toslink_surround_level(15)
        zp.RenderingControl.set_sub_gain(10)
        zp.RenderingControl.set_sub_enable(True)
        zp.RenderingControl.set_surround_enable(True)

def audiotap_running_on_playback_device(request, playback_device, tap_name,
                                        playback_sources):
    """
    Yield an audiotap running on a standalone
    :func:`~pytest_automation.pytest_sonos_device_selection.playback_device`
    using :func:`~pytest_automation.player.fade.conftest.tap_name` and
    :func:`~pytest_automation.player.fade.conftest.playback_sources`.
    :return: the audiotap instance, the playback SonosZoneComponent,
        the parametrized source value, and MaskingTestRunner
    :rtype: :class:`~audio_util.audiotap.AudioTap`,
        :class:`~sonos.client.internal.SonosZoneComponent`,
        :class:`~pytest_automation.player.fade.fade_common.MaskingTestRunner`
    """
    if playback_device.modelNumber == 'ZP80':
        pytest.skip("Skipping ZP80 since does not support audiotap")
    setup_primary_zp(playback_device)
    source = request.param
    note = "{}_{}".format(tap_name, source.name)
    output_fn = generate_tap_name(playback_device, note)
    audiotap = AudioTap(zp=playback_device, output_fn=output_fn,
                        audiotap_type=TAP_TYPE)
    test_runner = MaskingTestRunner(zp=playback_device,
                                    playback_sources=playback_sources,
                                    test_source=source)
    yield audiotap, playback_device, source, test_runner
    sample = audiotap.stop()
    if sample:
        sample.to_wav()
    guaranteed_sleep(AUDIOTAP_TERMINATION_PROPAGATION_DELAY_SEC)
    playback_device.AVTransport.leave_group()
    playback_device.AVTransport.clear_play_state()




[docs]def generic_bonded_setup(source, bonded_system, tap_name, bonded_type,
                         playback_sources):
    """
    Generic setup for 3.1 and 5.0 bonded configurations

    :param source: of type SourceEnum
    :param bonded_system: tuple of (HT master, sub) or (HT master, left, right)
    :param str tap_name: fixture that yields tap name notes string
    :param str bonded_type: type of bonded config (3.0 or 5.1)
    :param playback_sources: Sources object
    :return: list of AudioTap objects, Zone Player object, Zone Player
        object(s), SourceEnum object, MaskingTestRunner object
    """
    if bonded_type == "bonded_sub" or bonded_type == "31":
        primary, sub = bonded_system
        sats = [sub]
        all_zones = [primary, sub]
    elif bonded_type == "50":
        primary, left, right = bonded_system
        sats = [left, right]
        all_zones = [primary, left, right]
        if left == right:
            all_zones = [primary, left]
    else:
        raise NotImplementedError("No setup for {}".format(bonded_type))

    setup_primary_zp(primary)
    note = "{}_{}".format(tap_name, source.name)
    note += "_ht_master={}".format(primary.modelNumber)
    audiotaps = []

    for zone in all_zones:

        output_fn = generate_tap_name(zone, note)
        audiotaps.append(AudioTap(zp=zone, output_fn=output_fn,
                                  audiotap_type=TAP_TYPE))

    test_runner = MaskingTestRunner(zp=primary,
                                    playback_sources=playback_sources,
                                    test_source=source,
                                    satellite=sats[0])

    if bonded_type == "bonded_sub" or bonded_type == "31":
        return audiotaps, primary, sats[0], source, test_runner
    elif bonded_type == "50":
        return audiotaps, primary, sats[0], sats[1], source, test_runner




[docs]def generic_bonded_tear_down(audiotaps, primary):
    """
    Generic tear down method for bonded configurations

    :param list audiotaps: list of audiotaps running on players
    :param primary: zone player object for primary zp
    """
    for audiotap in audiotaps:
        sample = audiotap.stop()
        if sample:
            sample.to_wav()
    guaranteed_sleep(AUDIOTAP_TERMINATION_PROPAGATION_DELAY_SEC)
    primary.AVTransport.clear_play_state()
