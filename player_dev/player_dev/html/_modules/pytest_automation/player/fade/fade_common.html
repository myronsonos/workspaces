def _skip_if_source_not_in_list(source, source_list):
    """
    Call pytest.skip() if the given source is not in the list of sources given
    :param source: SourceEnum
    :param list source_list: list of type SourceEnum
    """
    if source not in source_list:
        pytest.skip("{} not in list of valid source: {}".format(
            source, source_list))

    """

    def __init__(self, zp, playback_sources, test_source, satellite=None):
        """
        Initializer

        :param zp: zone player object
        :param playback_sources: Sources object
        :param test_source: SourceEnum type
        :param satellite: zone player configured as satellite
        """
        self.zp = zp
        self.playback_sources = playback_sources
        self.test_source = test_source
        self.control_zp = self.zp
        if self.test_source == SourceEnum.GC:
            self.control_zp = self.playback_sources.get_gc_player_for_zp(zp)
        self.satellite = satellite

[docs]    def begin_tv_playback(self):
        """
        Begin playback of TV on self.zp
        """
        if not self.zp.ht_master_capable:
            pytest.skip("Device doesn't support TV input")
        self.zp.AVTransport.use_spdif()
        self.playback_sources.toslink_source.start()


    def _verify_presence_of_line_in(self, zp):
        """
        Verify that we have an 'AudioIn' attribute if the player supports
        line-in. Otherwise, raise an exception.
        :param zp: zone player to check
        """
        if zp.has_line_in() and not hasattr(zp, 'AudioIn'):
            raise Exception("Missing expected line-in on player {}".format(
                zp))


[docs]    def begin_line_in_playback(self):
        """
        Begin playback of local line-in on self.zp
        """
        if not self.zp.has_line_in():
            pytest.skip("Device doesn't support line-in")
        self._verify_presence_of_line_in(self.zp)
        # NOTE: assuming that all line-in autoplay should be disabled here
        self.playback_sources.line_in_source.start()
        guaranteed_sleep(LINE_IN_PLAYBACK_PROPAGATION_DELAY_SECONDS)
        self.zp.AVTransport.use_line_in(self.zp)
        self.zp.AVTransport.Play_and_wait()

[docs]    def begin_playback(self):
        """
        Generic begin playback method which calls whichever playback method is
        appropriate for self.test_source
        """
        if self.test_source == SourceEnum.MUSIC:
            self.begin_music_playback()
        elif self.test_source == SourceEnum.TV:
            self.begin_tv_playback()
        elif self.test_source == SourceEnum.LINEIN:
            self.begin_line_in_playback()
        elif self.test_source == SourceEnum.REMOTE_LINE_IN:
            self.begin_remote_line_in_playback()
        elif self.test_source == SourceEnum.GC:
            self.begin_gc_playback()
        else:
            raise NotImplementedError("Source {} not recognized".format(
                self.test_source))



[docs]    def end_playback(self):
        """
        Generic end playback method which calls whichever end playback
        method is approapriate for self.test_source
        """
        if self.test_source == SourceEnum.MUSIC:
            pass
        elif self.test_source == SourceEnum.TV:
            self.end_tv_playback()
        elif self.test_source in [SourceEnum.LINEIN,
                                  SourceEnum.REMOTE_LINE_IN]:
            self.end_line_in_playback()
        elif self.test_source == SourceEnum.GC:
            pass
        else:
            raise NotImplementedError("Source {} not recognized".format(
                self.test_source))



[docs]    def test_repeated_next_prev(self):
        """
        Setup the player's AVT to do a series of Next/Previous commands and
        then execute the Next/Previous commands repeatedly
        """
        self.zp.AVTransport.clear_queue()
        self.zp.AVTransport.add_to_queue(self.playback_sources.uri)
        self.zp.AVTransport.add_to_queue(self.playback_sources.uri)
        self.zp.AVTransport.use_queue()
        self.zp.AVTransport.Play_and_wait()
        repeater = PlayerRepeater(player=self.zp, sec_between_repeats=3,
                                  playback_source=self.test_source,
                                  n_repeats=5)
        repeater.next_previous()
        self.end_playback()
