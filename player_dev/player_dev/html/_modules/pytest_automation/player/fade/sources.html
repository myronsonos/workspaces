[docs]class AutoNumber(Enum):
    """
    helper to automatically generate the numbers for an int enum
    """
    def __new__(cls):
        value = len(cls.__members__) + 1
        obj = object.__new__(cls)
        obj._value_ = value
        return obj

[docs]class Sources(object):
    """
    Class to hold various playback sources to be used for tests to reduce
    the number of sources that must be passed from test case to test case.
    """

    def __init__(self, uri, toslink_source, line_in_source, freq_hz,
                 line_in_players, all_players):
        """
        Initializer (assumes same file is loaded to uri, toslink source,
        and line-in source)

        :param str uri:
        :param toslink_source: MplayerAc3Playback object already configured
            with a test file
        :param line_in_source: MplayerLineOutPlayback object already
            configured with a test file
        :param float freq_hz: the frequency in Hz of the test signal
        :param list line_in_players: list of zone players with local line-in
        :param list all_players: list of all players on the test bed
        """
        self.uri = uri
        self.toslink_source = toslink_source
        self.line_in_source = line_in_source
        self.freq_hz = freq_hz
        self._line_in_players = line_in_players
        self._all_players = all_players


[docs]    def stop_sources(self):

        """
        self.toslink_source.stop()
        self.line_in_source.stop()



[docs]    def get_line_in_player_for_zp(self, zp):
        """
        Return a player from the list of line-in capable players as long as
        it is not the given player

        :param zp: zone player NOT to return
        :return: zone player object
        """
        return [z for z in self._line_in_players if z is not zp][0]



[docs]    def get_gc_player_for_zp(self, zp):
        """
        Return a player from the list of all players as long as it is not
        the given player

        :param zp: zone player NOT to return
        :return: zone player object
        """
        return [z for z in self._all_players if z is not zp and not
                z.is_sub()][0]
