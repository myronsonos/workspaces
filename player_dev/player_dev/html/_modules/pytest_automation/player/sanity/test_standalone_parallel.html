[docs]def get_channel_index_for_frequency_and_model(frequency_hz, model_number):
    """
    Given a frequency of a pure tone and a model number of a zone player,
    return the index to an appropriate DSPOUT audiotap channel that will be
    playing the frequency.

    :param float frequency_hz: the frequency of the tone
    :param str model_number: the model number of the zone player
    :return: intteger denoting the index of the channel
    """
    if frequency_hz > 1000:
        return get_tweeter_channel_index(model_number)
    else:
        return get_woofer_channel_index(model_number)




[docs]def time_is_up(timers, duration_sec):
    """
    Given a map of form {'serial_number': start time }, return True when the
    current time minus the start_time for each value in the map is greater
    than the given duration.

    :param dict timers: dictionary of playback start times by player
    :param float duration_sec: number of seconds that must expire before True
    :return: True/False
    """
    for k in timers:
        if time.time() - timers[k] < duration_sec:
            return False
    return True




[docs]def get_zps_for_source(zps, source):
    """
    Return all of the valid zone players for the given source

    :param list zps: list of zone players
    :param str source: the source that we're testing
    :return: list of zone players that are appropriate for that source
    """
    valid_zps = []
    for zp in zps:
        if source == TV and not zp.ht_master_capable:
            pytest.logger.info("Skipping TV capture on zp: {}".format(zp))
            continue
        valid_zps.append(zp)
    return valid_zps




[docs]def start_audiotaps(zps, source, line_in_mapping):
    """
    Start audiotap capture on all given zone players for the given source type

    :param list zps: zone players to start audiotap capture on
    :param str source: identifier of source (music, tv, or line-in)
    :param dict or None line_in_mapping: either None or a dictionary that
        maps each zone player to its line-in source
    :return: tuple - a list of filenames for audiotaps and a dictionary of
        serial numbers to audiotap objects and a dictionary of input audiotaps
    """
    # filenames (fns) and audiotaps (ats)
    fns = []
    ats = {}
    input_taps = {}

    # For each player
    for zp in zps:

        # don't capture TV audiotaps for non-TV capable players
        if source == TV and not zp.ht_master_capable:
            pytest.logger.info("Skipping TV capture on zp: {}".format(zp))
            continue

        # create a filename for the tap and append to list
        base_filename = "{}_{}_{}".format(zp.get_arch(),zp.modelNumber, source)
        if line_in_mapping:
            line_in_arch = line_in_mapping[zp].get_arch()
            base_filename += "_line-in_source={}".format(line_in_arch)
        fn = base_filename + "_standalone.raw"
        fns.append(fn)

        # also start input tap capture
        if source == LINEIN and zp.is_hideout():
            in_fn = '{}_{}_{}_line_in_tap.gld'.format(
                zp.get_arch(), zp.modelNumber, zp.raw_udn)
            input_taps[zp] = AudioTap(
                zp=zp, audiotap_type=AudioTap.AUDIOTAP_LINEIN,
                output_fn=in_fn)
        elif source == TV and zp.ht_master_capable:
            in_fn = '{}_{}_{}_spdif_tap.gld'.format(
                zp.get_arch(), zp.modelNumber, zp.raw_udn)
            input_taps[zp] = AudioTap(
                zp=zp, audiotap_type=AudioTap.AUDIOTAP_SPDIF,
                output_fn=in_fn)

        # Start audiotap for zp
        ats[zp] = AudioTap(zp=zp, audiotap_type=AudioTap.AUDIOTAP_DSP,
                           output_fn=fn)

    # return filenames and audiotap dictionary
    return fns, ats, input_taps




[docs]def wait_for_timers(timers, duration_per_source_seconds, iteration_delay_sec):
    """
    Wait for time_is_up() to return True for the given timers dictionary,
    duration. Check if it's True every iteration_delay_sec seconds.

    :param dict timers: dictionary of playback start times by player
    :param float duration_per_source_seconds: number of seconds that must
        expire
    :param float iteration_delay_sec: seconds to wait between checks
    """
    while not time_is_up(timers, duration_per_source_seconds):
        pytest.logger.info("sleeping {} seconds".format(
            iteration_delay_sec))
        guaranteed_sleep(iteration_delay_sec)




[docs]def capture_playback_for_standalone_players(zps, source, track_uri,
                                            line_in_source, toslink_source,
                                            duration_sec):
    """
    Capture audiotaps for each of the given zone players while playing back
    content in the form of the given source. Return the list of audiotap
    filenames.

    :param list zps: the list of zone players to capture audiotaps for
    :param str source: the source to playback
    :param str track_uri: if music playback, the URI to use
    :param line_in_source: if line-in playback, the line-in source to use
    :param toslink_source: if TV playback, the toslink source to use
    :param int duration_sec: the number of seconds to play a source
    :return: list of filenames to analyze
    """

    # amount of time to wait between checks of a wait_until_true
    iteration_delay_sec = 0.5

    # zps that support line-in
    line_in_zps = [z for z in zps if z.has_line_in()]

    line_in_zp_mapping = None
    # map the player to the line-in zp that it will get line-in from
    if source == LINEIN:
        line_in_zp_mapping = {}
        for zp in zps:
            line_in_zp = [z for z in line_in_zps if z != zp][0]
            line_in_zp_mapping[zp] = line_in_zp

    # get filenames and audiotap objects
    valid_zps = get_zps_for_source(zps, source)

    timers = {}

    # start playback section
    if source == TV:
        toslink_source.start()
        guaranteed_sleep(SECONDS_AFTER_STARTING_MPLAYER_TO_SELECT_SOURCE)
        for zp in valid_zps:
            zp.AVTransport.use_spdif()
        for zp in valid_zps:
            timers[zp] = time.time()
    elif source == LINEIN:
        line_in_source.start()
        guaranteed_sleep(SECONDS_AFTER_STARTING_MPLAYER_TO_SELECT_SOURCE)
        for zp in valid_zps:
            # we want to play remote line-in. choose a zp that isn't this one
            line_in_zp = line_in_zp_mapping[zp]

            # set the gain on the line-in zp to not max (12, 15) but close
            # to it so as to prevent clipping but ensure good SNR
            # NOTE: the values of 5 and 7 were shown to yield a reasonably
            # high SNR (20dB)
            line_in_gain = 5 if line_in_zp.get_arch() != 'sh4' else 7
            pytest.logger.info("Setting line-in source to {} on zp {}".format(
                line_in_zp, zp))

            zp.AVTransport.use_line_in(line_in_zp)
            line_in_zp.AudioIn.SetLineInLevel(
                left_level=line_in_gain, right_level=line_in_gain)
            zp.AVTransport.Play_and_wait()
        for zp in valid_zps:
            timers[zp] = time.time()
    elif source == MUSIC:
        for zp in valid_zps:
            zp.AVTransport.SetAVTransportURI(track_uri, '')
            zp.AVTransport.Play_and_wait()
            timers[zp] = time.time()
    else:
        raise NotImplementedError("Unknown audio source: {}".format(source))

    fns, ats, input_taps = start_audiotaps(zps, source,
                                           line_in_mapping=line_in_zp_mapping)

    # playback section
    wait_for_timers(timers, duration_sec,
                    iteration_delay_sec)

    # end playback section
    if source == TV:
        toslink_source.stop()
    elif source == LINEIN:
        line_in_source.stop()
    elif source == MUSIC:
        [zp.AVTransport.Stop_and_wait() for zp in zps]

    # Stop each audiotap as its duration is met
    zps_missing_taps = []
    for zp in ats:
        sample = ats[zp].stop()
        if sample:
            sample.to_wav()
        else:
            pytest.logger.info("No samples captured for audiotap on "
                               "{}".format(zp))
            zps_missing_taps.append(zp)
    if len(zps_missing_taps) > 0:
        raise Exception("Missing audiotaps for these players: {}".format(
            zps_missing_taps))

    # stop all input taps
    for zp in input_taps:
        input_taps[zp].stop()

    # return audiotap filenames
    return fns




[docs]def remove_starting_ending_noise(signal, threshold=0.2):
    """
    Remove any audio at the start and end of the file that, when normalized, is
    less than the given threshold.
    :param signal: numpy array of audio samples
    :param float threshold: threshold from 0.0 to 1.0
    :return: numpy array of audio samples from start index to end index
    """
    # divide by max to normalize -1 to 1
    normalized_signal = signal / max(signal)
    start_idx = 0
    end_idx = -1

    # get start index as first sample above threshold and end index as last
    # sample above threshold
    for idx in range(0, len(normalized_signal)):
        start_idx = idx
        if normalized_signal[start_idx] > threshold:
            break

    for idx in range(0, len(normalized_signal)):
        end_idx = len(normalized_signal) - 1 - idx
        if normalized_signal[end_idx] > threshold:
            break

    pytest.logger.info("Selected start/end index of audiotap to be {} {} for "
                       "line in source".format(start_idx, end_idx))

    if end_idx != -1:
        return signal[start_idx: end_idx + 1]
    return signal[start_idx:]




[docs]def account_for_corrupted_test_runner_signal(signal, source, fs):
    """
    ignore the first second of audio for some sources

    :param list signal: list of audio samples
    :param str source: the name of the input source
    :param int fs: samples per second
    :return: list of samples with first second removed if a signal that may
        have been corrupted by test runner output
    """
    if source in SOURCES_CORRUPTED_AT_START_ON_TEST_RUNNER:
        signal = signal[fs:]
    return signal




[docs]def verify_audiotap(audiotap_file, test_signal_to_noise_db,
                    test_frequency_hz):
    """
    Verify that the given audiotap file:
    1. exists
    2. has a high signal to noise ratio for the test signal frequency

    Note: we do not throw asserts in this test because we want to ensure
    that we execute for each file for each player. Otherwise, we would stop
    on the first failure.

    :param str audiotap_file: path to the audiotap file
    :param int test_signal_to_noise_db: the minimum dB difference that can
        exist between the signal energy and noise energy before we fail
    :param int test_frequency_hz: the frequency (in Hz) of the signal that
        we're looking for
    :return: string denoting whether or not test passed for given audiotap.
    """
    # get the base filename and extract the model from it
    fname = os.path.basename(audiotap_file)
    params = fname.split('_')
    model = params[1]
    source = params[2]
    return run_verification(model, source, audiotap_file, test_frequency_hz,
                            test_signal_to_noise_db)




[docs]def run_verification(model, source, audiotap_file, test_frequency_hz,
                     test_signal_to_noise_db):
    """
    Run the actual verification of the audiotaps captured

    :param str model: model number of player
    :param str source: the name of the playback source
    :param str audiotap_file: path to the audiotap file
    :param int test_frequency_hz: the expected signal frequency in Hertz
    :param int test_signal_to_noise_db: the acceptable Signal to Noise Ratio
    :return: str with results of verification
    """
    # whether or not the audiotap is padded with 8 bits of zeros
    padded = model in DEVICES_WITH_8_BIT_ZERO_PADDING_IN_DSPOUT_AUDIOTAPS

    # if the file does not exist, return a file not found error
    if not os.path.exists(audiotap_file):
        return ERROR_FILE_NOT_FOUND

    # get the normalized audio from a file
    norm_audio = rms_helper.get_normalized_audio_from_file(
        fname=audiotap_file, padded=padded,
        n_channels=AudioTap.AUDIOTAP_DSP_CHANNELS[model])

    # sampling frequency depends on the player - sub is lower
    if model != "SUB":
        fs = 44100
        # downmix the audio to one channel
        norm_audio = norm_audio[get_channel_index_for_frequency_and_model(
            test_frequency_hz, model)]
    else:
        fs = 8138.02

    norm_audio = account_for_corrupted_test_runner_signal(norm_audio,
                                                          source, fs)

    # remove the leading and trailing zeros from the signal
    norm_audio = rms_helper.get_signal_removed_starting_zeros(norm_audio)

    # there can be nonzero samples before the signal actually begins
    # playback and after playback ends. Need to remove that.
    norm_audio = remove_starting_ending_noise(norm_audio)

    norm_audio = account_for_corrupted_test_runner_signal(norm_audio,
                                                          source, fs)

    # get the spectrogram for the given normalized audio
    # in this case, we don't do any overlap and use sampling frequency sized
    # windows so as to have a high degree of frequency granularity at each
    # window
    specgram = fft_helper.get_spectrogram_from_signal(
        signal=norm_audio, fs=fs, overlap=0, n=fs)

    # get the index of the test signal's frequency bin
    idx = test_crossfade.get_index_for_freq(freqs=specgram['freqs'],
                                            target_f=test_frequency_hz)[0]

    # get the indices of the noise frequency bins (not the signal's)
    n_idx = test_crossfade.get_indices_of_not_these_frequencies(
        freqs=specgram['freqs'], target_fs=[test_frequency_hz])

    # track the number of windows where the SNR is low
    n_windows_snr_too_low = 0
    total_n_winows = len(specgram['mags']) - 2

    # don't start from first window or end at last window as they may
    # contain noise - we just want to make sure that the audio playback
    # works generally, we don't really care about masking/fade in this test.
    first_second_with_noise = None
    for s in range(1, len(specgram['mags']) - 1):

        # get the magnitudes of the signal and the noise bins
        sig_mag = specgram['mags'][s][idx]
        # the noise magnitude is the sum of all bins besides the signal bin
        n_mag = test_crossfade.add_dbs([
            specgram['mags'][s][x] for x in n_idx])

        pytest.logger.info("@{}Hz: {}".format(test_frequency_hz, sig_mag))
        pytest.logger.info("@noise: {}".format(n_mag))

        # if even one window has noise with higher magnitude than the
        # signal, we return that there was an error with the audio signal
        if sig_mag <= n_mag + test_signal_to_noise_db:
            n_windows_snr_too_low += 1
            pytest.logger.info("sig to noise should be higher. {} windows "
                               "with low SNR".format(n_windows_snr_too_low))
            if first_second_with_noise is None:
                first_second_with_noise = s

    ret = OK

    percent_windows_low_snr = 100.0 * n_windows_snr_too_low / total_n_winows
    pytest.logger.info("{}% windows had low SNR".format(
        percent_windows_low_snr))
    if percent_windows_low_snr > 0.0:
        ret = ERROR_AUDIO_SIGNAL.format(first_second_with_noise)

    return ret

                         [(LOCAL_TEST_FILE_PATH, TEST_FILE_METADATA[
                             'mplayer_codec'])], indirect=['toslink_source'])
@pytest.mark.parametrize('line_in_source', [[LOCAL_TEST_FILE_PATH]],
                         indirect=['line_in_source'])
@pytest.mark.parametrize("source", [MUSIC, TV, LINEIN],
                         ids=[MUSIC, TV, LINEIN])
@pytest.mark.standalone
def test_standalone_concurrently(sanity_test_configured_standalone_players,
                                 source, toslink_source, line_in_source,
                                 track_uri, request):
    """
    This test verifies that the signal to noise ratio for a given playback
    source is at least some given value. The goal of this test is to confirm
    that the most basic playback scenarios outputs some correct audio in a
    standalone configuration (i.e. no silence, no huge amounts of noise).

    :param list sanity_test_configured_standalone_players: list of players
        that have been configured to run this test, along with the duration
        of the test and the max percentage of FFT windows that can have a
        low SNR before we fail the test
    :param str source: the source that will be used for playback for this test
    :param toslink_source: a toslink source
    :param line_in_source: a line-in source
    :param str track_uri: a URI to a music file for playback
    :param request: pytest request
    """
    skip_line_in = request.config.getoption('skip_line_in')
    if skip_line_in and source == LINEIN:
        pytest.skip("Skipping line-in as per command line specification")
    zps, duration = sanity_test_configured_standalone_players
    guaranteed_sleep(PROPAGATION_DELAY_FOR_AUDIOTAP_SEC)

    # capture audiotaps for playback from the given source on the given players
    files = capture_playback_for_standalone_players(
        zps=zps, source=source, toslink_source=toslink_source,
        line_in_source=line_in_source, track_uri=track_uri,
        duration_sec=duration)

    # go through each audiotap file and verify if it passed
    pass_state_per_player = {}
    for f in files:
        pytest.logger.info("file: {}".format(f))
        pass_state_per_player[f] = verify_audiotap(f, TEST_SIGNAL_TO_NOISE_DB,
                                                   TEST_FREQUENCY_HZ)

    # we do the assertions around pass/fail separately so that we can get
    # the results for all players simultaneously
    for f in pass_state_per_player:
        if pass_state_per_player[f] != OK:
            zp = zps[0]
            diag_id = zp.diag.submit_direct_diag()
            pytest.fail("Failed test: {}. Diag ID {}".format(
                pass_state_per_player, diag_id))
