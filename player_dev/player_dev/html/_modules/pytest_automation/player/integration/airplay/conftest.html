def cleanup_virtual_groupID(dut_manager, ws_client, groups_in_hh):
    """
    Clear the 'VirtualLineInGroupID' value at the start and end of each test
    for each device used in the test using the
    :func:`~pytest_automation.player.conftest.dut_manager`.
    :func: groups_in_hh - return number of groups originally reported in the HH
    :func: ws_client - active webscoket connection with dut_manager.zp

    :return: the dut_manager
    :rtype: :func:`~pytest_automation.player.conftest.dut_manager`
    """
    if player_common.is_locked_build():
        pytest.skip("This current build is locked, airplay tests can only be run on unlocked builds")

    player_common.set_verify_zp_vli_all_devices("", dut_manager)

    yield dut_manager

    player_common.set_verify_zp_vli_all_devices("", dut_manager)

    # Force player to standalone if it is not its own gc or is bonded
    for zp in dut_manager.test_devices:
        zp.diag.set_vli_setBecomeStandaloneDelay()
        dut_manager.zp_cleanup(zp)

    # Finally wait for up to 30 secs for the number of groups in the hh to match the value set at the start
    ws_client.groups.getGroups()
    verify_number_of_groups(ws_client, groups_in_hh, new_group_count=0)

def itunes_tracks_setup(mac_config):
    """
    Copies iTunes tracks from the test_tracks directory to the mac using the
    :func:`~pytest_automation.player.airplay.conftest.mac_config`.

    :return: True
    :rtype: :obj:`bool`
    """
    hostname = getattr(mac_config, 'hostname', None)

    if hasattr(mac_config, 'username_var'):
        username = os.getenv(getattr(mac_config, 'username_var'))
    else:
        username = getattr(mac_config, 'username', None)

    if hasattr(mac_config, 'password_var'):
        password = os.getenv(getattr(mac_config, 'password_var'))
    else:
        password = getattr(mac_config, 'password', None)

    sftp = SonosPlatformSFTP(hostname, username, password)
    sftp.backup_files_by_pattern(
        LOCAL_TRACKS_DIR, TARGET_TRACKS_DIR.format(username), "*")

    tracks = [TARGET_TRACKS_DIR.format(username) + t
              for t in os.listdir(LOCAL_TRACKS_DIR)]

    pytest.logger.info("Adding tracks " + ", ".join(tracks))

    ITunesAddTracks(config=mac_config, username=username, password=password).add(tracks)

    return True

def setup_cleanup_itunes_airplay(itunes_tracks_setup, itunes_player, dut_manager, system_control):
    """
    Performs following functions:
    - setup a mp3 track in iTunes
    - switch the airplay target to a zp
    - switch back to mac during teardown
    - log any airplay failures encountered
    :param itunes_tracks_setup:
    :param itunes_player:
    :param dut_manager:
    :param system_control:
    """
    # Assert if the mp3 track setup failed
    assert itunes_tracks_setup, (
        "Test mp3s should have been copied to the mac"
    )

    # If itunes is currently playing, stop it and switch the airplay target to the mac
    itunes_player.stop()
    itunes_player.create_playlist("QA")
    itunes_player.enable_playlist("QA", "true")
    if itunes_player.get_devices() != {"Computer"}:
        common.set_itunes_airplay_target("Computer", itunes_player)
        wait_until_true(predicate=lambda: itunes_player.get_devices() == {"Computer"},
                        timeout_seconds=6, iteration_delay=1,
                        reason="Waiting for <Computer> to be selected as Airplay target "
                               "but got <{}>".format(dut_manager.zp.friendly_name, itunes_player.get_devices()))
    yield

    # Swap airplay target back to mac if its not already and clear the zp playstate
    if itunes_player.get_devices() != {"Computer"}:
        common.set_itunes_airplay_target("Computer", itunes_player)
        wait_until_true(predicate=lambda: itunes_player.get_devices() == {"Computer"},
                        timeout_seconds=6, iteration_delay=1,
                        reason="Waiting for <Computer> to be selected as Airplay target "
                               "but got <{}>".format(dut_manager.zp.friendly_name, itunes_player.get_devices()))

    # Itunes and test devices cleanup
    itunes_player.delete_playlist("QA")
    itunes_player.stop()
    dut_manager.zp_cleanup_all_test_devices()

def setup_cleanup_no_airplayIncludeLinked(dut_manager):
    """
    Sets the R_AirplayIncludeLinked string value to 0 initially using the
    :func:`~pytest_automation.player.conftest.dut_manager`, then returns to original
    value after the test is finished.
    """

    try:
        orig_val = dut_manager.zp.SystemProperties.GetString(common.INCLUDE_LINKED_KEY)
    except UPnPError as e:
        if e.upnp_error == UPNP_ERROR_800_NO_SETTING:
            # If the value has not been initialized yet, an error is thrown. Default behavior when
            # R_AirplayIncludeLinked is not set is to treat it as '0', so replicate that here
            orig_val = 0
        else:
            raise e

    common.set_airplay_includeLinked(dut_manager.zp, 0)
    yield
    common.set_airplay_includeLinked(dut_manager.zp, int(orig_val))  # put everything back how we found it
