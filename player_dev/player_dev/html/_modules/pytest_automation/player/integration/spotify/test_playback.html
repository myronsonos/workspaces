def test_verify_tracks_in_sync_for_several_skips_forward(sca_playback):
    """
    1. Select sonos device in Spotify
    2. Start playback
    3. Skip forward for via Spotify several times
    4. Verify currently playing track's metadata matches up on Sonos and
       Spotify side
    """
    sca = sca_playback()

    zp_tracks = []
    spot_tracks = []
    expected_tracks = []
    for track in sca.playlist['items'][:5]:
        expected_tracks.append((track['track']['name'], track['track']['track_number']))
        curr_track_sp = sca.spot.currently_playing_track()
        zp_track_info = sca.zp.AVTransport.GetPositionInfo()
        zp_tracks.append(
            (sca.zp.AVTransport.get_current_track_title(), int(zp_track_info['Track'])))
        spot_tracks.append((curr_track_sp['item']['name'], curr_track_sp['item']['track_number']))
        skipped_forward = sca.spot.next()
        wait_until_true(lambda: (sca.zp.AVTransport.is_playing() and
                                sca.zp.AVTransport.GetPositionInfo()["RelTime"] >= "0:00:03" and
                                 sca.spot.currently_playing_track()['item']['name'] != curr_track_sp['item']['name']),
                        iteration_delay=1,
                        timeout_seconds=10, reason="ZP<{}> not playing".format(sca.zp.friendly_name))
        assert skipped_forward, 'failed to skip to next track'

    assert len(spot_tracks) == len(expected_tracks), 'list of tracks do not match'
    assert spot_tracks == expected_tracks, 'Tracks didn\'t skip forward in order. Expected:<{}> Received:<{}>'\
                                            .format(expected_tracks, spot_tracks)

def test_verify_tracks_in_sync_for_several_skips_backwards(sca_playback, get_playlist):
    """
    1. Start playback
    2. Skip backward for via Spotify several times
    3. Verify currently playing track's metadata matches up on Sonos and
       Spotify side
    """
    # start at the end of playlist
    sca = sca_playback(offset={'position': get_playlist.playlist['total'] - 1})

    skip_success = 0
    zp_tracks = []
    spot_tracks = []
    expected_tracks = [t['track']['name'] for t in reversed(sca.playlist['items'])]
    for i in range(6):
        curr_track_sp = sca.spot.currently_playing_track()
        # add tracks to compare later
        zp_tracks.append(sca.zp.AVTransport.get_current_track_title())
        spot_tracks.append(curr_track_sp['item']['name'])
        skipped_backward = sca.spot.previous()
        wait_until_true(lambda: (sca.zp.AVTransport.is_playing() and
                                sca.zp.AVTransport.GetPositionInfo()["RelTime"] >= "0:00:03" and
                                 sca.spot.currently_playing_track()['item']['name'] != curr_track_sp['item']['name']),
                        iteration_delay=1,
                        timeout_seconds=10, reason="ZP<{}> not playing".format(sca.zp.friendly_name))
        if skipped_backward:
            skip_success += 1

    assert skip_success == 6, 'At least one skip to previous track failed'
    assert spot_tracks == expected_tracks[:6], 'Tracks didn\'t skip backward in order. Expected:<{}> Received:<{}>' \
                                        .format(expected_tracks[:6], spot_tracks)

def test_verify_song_added_to_queue_is_present_in_playlist(spotify_zp_standalone, get_playlist):
    """
    1. Start playback with playlist via Spotify
    2. Add a song to the playlist
    3. Verify that song was added to the end of the queue for that playlist
    """
    from random import randint
    the_beatles_uri = 'spotify:artist:3WrFJ7ztbogyGnTHbHJFl2'
    beatles_tracks = spotify_zp_standalone.spot.artist_top_tracks(the_beatles_uri)['tracks']
    track_uris = [beatles_tracks[randint(0, len(beatles_tracks) - 1)]['uri']]

    # add song to playlist & verify
    add_track = spotify_zp_standalone.spot.user_playlist_add_tracks(
        spotify_zp_standalone.spot.me()['id'], get_playlist.playlist['id'], track_uris)
    assert add_track['snapshot_id'] != '', 'track was not added to queue'
    updated_tracks = spotify_zp_standalone.spot.user_playlist_tracks(
        spotify_zp_standalone.spot.me()['id'], get_playlist.playlist['id'],
        fields='items,uri,name,id,total', market='us')
    assert get_playlist.playlist['total'] + 1 == updated_tracks['total'], 'total number of tracks did not update'

    # code add: get updated queue from sonos api
    spotify_zp_standalone.spot.play(context_uri=get_playlist.playlist['uri'],
                                    offset={'position': updated_tracks['total'] - 1})
    wait_until_true(lambda: spotify_zp_standalone.zp.AVTransport.is_playing()
                and spotify_zp_standalone.zp.AVTransport.GetPositionInfo()["RelTime"] >= "0:00:03", iteration_delay=1,
                    timeout_seconds=10, reason="ZP<{}> not playing".format(spotify_zp_standalone.zp.friendly_name))
    curr_track_sp = spotify_zp_standalone.spot.current_playback_info(market='us')
    curr_track_zp = spotify_zp_standalone.zp.AVTransport.get_current_track_title()

    # get a list of tracks and compare the names and uris
    assert track_uris[0] == curr_track_sp['item']['uri'], 'Spotify track in queue not matching with expected ZP track. '\
                                      'Expected:<{}> Received:<{}>'.format(track_uris[0], curr_track_sp['item']['uri'])
    assert curr_track_zp in curr_track_sp['item']['name'], 'Spotify <{}> and Sonos <{}> tracks are not in sync'\
                                                    .format(curr_track_sp['item']['name'], curr_track_zp)

def test_verify_song_removed_from_queue_is_gone_from_playlist(spotify_zp_standalone, get_playlist):
    """
    1. Start playback with playlist via Spotify
    2. Remove a song from the playlist
    3. Verify that song was removed from the queue (at the end)
    """
    before_playlist = spotify_zp_standalone.spot.user_playlist_tracks(spotify_zp_standalone.spot.me()['id'],
                                get_playlist.playlist['id'], fields='items,uri,name,id,total', market='us')
    last_track_num = before_playlist['total']
    last_track = before_playlist['items'][last_track_num - 1]
    expected_track = before_playlist['items'][last_track_num - 2]

    # start playback
    spotify_zp_standalone.spot.play(context_uri=get_playlist.playlist['uri'], offset={'position': 5})
    wait_until_true(lambda: spotify_zp_standalone.zp.AVTransport.is_playing()
                and spotify_zp_standalone.zp.AVTransport.GetPositionInfo()["RelTime"] >= "0:00:03", iteration_delay=1,
                    timeout_seconds=10, reason="ZP<{}> not playing".format(spotify_zp_standalone.zp.friendly_name))
    # Remove track
    removed_track = spotify_zp_standalone.spot.user_playlist_remove_specific_occurrences_of_tracks(
        spotify_zp_standalone.spot.me()['id'], get_playlist.playlist['id'],
        [{'uri': last_track['track']['uri'], 'positions': [last_track_num - 1]}])
    assert removed_track['snapshot_id'] != '', 'track was not removed'
    # time.sleep(request.config.getoption("--time_delay") + 2)
    updated_tracks = spotify_zp_standalone.spot.user_playlist_tracks(
        spotify_zp_standalone.spot.me()['id'], get_playlist.playlist['id'],
        fields='items,uri,name,id,total', market='us')
    new_last_track = updated_tracks['items'][-1]
    assert expected_track['track']['name'] == new_last_track['track']['name']
    assert last_track_num - 1 == updated_tracks['total'], 'total number of tracks did not update'
    assert last_track != new_last_track, 'last track in queue should be the previous track'

    spotify_zp_standalone.spot.play(context_uri=get_playlist.playlist['uri'],
                                    offset={'position': updated_tracks['total'] - 1})
    wait_until_true(lambda: spotify_zp_standalone.zp.AVTransport.is_playing()
                and spotify_zp_standalone.zp.AVTransport.GetPositionInfo()["RelTime"] >= "0:00:03", iteration_delay=1,
                    timeout_seconds=10, reason="ZP<{}> not playing".format(spotify_zp_standalone.zp.friendly_name))
    curr_track_sp = spotify_zp_standalone.spot.current_playback_info(market='us')
    curr_track_zp = spotify_zp_standalone.zp.AVTransport.get_current_track_title()
    assert curr_track_sp['item']['id'] != last_track['track']['id'], 'track ids should be different'
    assert curr_track_sp['item']['name'] != last_track['track']['name'], 'track names should be different'
    assert curr_track_zp in curr_track_sp['item']['name'], 'Spotify <{}> and Sonos <{}> tracks are not in sync.'\
                                                    .format(curr_track_sp['item']['name'], curr_track_zp)

def test_remove_currently_playing_track(spotify_zp_standalone, get_playlist):
    """
    1. Start playback with playlist via Spotify (pick a track in the middle)
    2. Remove the track that is currently playing
    3. Verify that the next track starts playing (original doesn't work,
       so verify current track is playing)
    4. Skip to previous x2 and verify that is the track before the deleted
       track
    """
    position = 5
    before_playlist = spotify_zp_standalone.spot.user_playlist_tracks(spotify_zp_standalone.spot.me()['id'],
                                get_playlist.playlist['id'], fields='items,uri,name,id,total', market='us')
    spotify_zp_standalone.spot.play(context_uri=get_playlist.playlist['uri'], offset={'position': position})
    wait_until_true(lambda: spotify_zp_standalone.zp.AVTransport.is_playing()
                            and spotify_zp_standalone.zp.AVTransport.GetPositionInfo()["RelTime"] >= "0:00:03",
                    iteration_delay=1,
                    timeout_seconds=10, reason="ZP<{}> not playing".format(spotify_zp_standalone.zp.friendly_name))
    # Grab first playing track info
    first_playing = spotify_zp_standalone.spot.current_playback_info(market='us')
    first_track_zp = spotify_zp_standalone.zp.AVTransport.get_current_track_title()
    track_ids = [first_playing['item']['uri']]
    # Remove the track
    removed_track = spotify_zp_standalone.spot.user_playlist_remove_all_occurrences_of_tracks(
        spotify_zp_standalone.spot.me()['id'], get_playlist.playlist['id'], track_ids)
    assert removed_track['snapshot_id'] != '', 'removing currently playing track failed'
    # Skip to the next track and verify
    spotify_zp_standalone.spot.next()
    wait_until_true(lambda: spotify_zp_standalone.zp.AVTransport.is_playing()
                and spotify_zp_standalone.zp.AVTransport.GetPositionInfo()["RelTime"] >= "0:00:03"
                    and spotify_zp_standalone.spot.currently_playing_track()['item']['name'] != first_playing['item']['name'],
                    iteration_delay=3,
                    timeout_seconds=10, reason="ZP<{}> not playing".format(spotify_zp_standalone.zp.friendly_name))
    # Grab second playing track info
    second_playing = spotify_zp_standalone.spot.current_playback_info(market='us')
    second_track_zp = spotify_zp_standalone.zp.AVTransport.get_current_track_title()

    # skip to previous track by calling previous twice
    skip_backward = spotify_zp_standalone.spot.previous()
    skip_backward_1 = spotify_zp_standalone.spot.previous()
    wait_until_true(lambda: spotify_zp_standalone.zp.AVTransport.is_playing()
                            and spotify_zp_standalone.zp.AVTransport.GetPositionInfo()["RelTime"] >= "0:00:03" and
                spotify_zp_standalone.spot.currently_playing_track()['item']['name'] != second_playing['item']['name'],
                    iteration_delay=3,
                    timeout_seconds=10, reason="ZP<{}> not playing".format(spotify_zp_standalone.zp.friendly_name))
    # Grab final playing track info
    final_playing = spotify_zp_standalone.spot.current_playback_info(market='us')
    final_track_zp = spotify_zp_standalone.zp.AVTransport.get_current_track_title()
    num_tracks_left = spotify_zp_standalone.spot.user_playlist_tracks(
        spotify_zp_standalone.spot.me()['id'], get_playlist.playlist['id'])['total']
    # Verify
    assert num_tracks_left == before_playlist['total'] - 1, 'the playlist/queue size did not reduce by 1'
    assert first_playing['item']['name'] != final_playing['item']['name'], 'track to be removed still exists'
    assert final_playing['is_playing'] is True, 'playback did not continue after removing track'
    assert first_track_zp in first_playing['item']['name'], 'First Spotify <{}> and Sonos <{}> track are not in sync.'\
                                                .format(first_playing['item']['name'], first_track_zp)
    assert second_track_zp in second_playing['item']['name'], 'Second Spotify <{}> and Sonos <{}> track are not in sync.'\
                                                .format(second_playing['item']['name'], second_track_zp)
    assert final_track_zp in final_playing['item']['name'], 'Final Spotify <{}> and Sonos <{}> track are not in sync.'\
                                                .format(final_playing['item']['name'], final_track_zp)
