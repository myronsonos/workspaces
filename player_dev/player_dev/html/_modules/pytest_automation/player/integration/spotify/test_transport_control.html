def test_seekRelative_from_sonos(spotify_zp_standalone, get_playlist):
    """
    SWPBL-125439
    Verify that muse seekRelative and upnp TIME_DELTA seek commands works with a Spotify DC stream
    :param spotify_zp_standalone:
    :param get_playlist:
    :return:
    """
    spotify_zp_standalone.spot.play(device_id=spotify_zp_standalone.device_id, position_ms=30000,
                                    context_uri=get_playlist.playlist['uri'])
    integ_common.wait_for_some_playback(spotify_zp_standalone.zp, 3, 10)

    # seekRelative forward
    spotify_zp_standalone.zp.muse_rest.playback.seekRelative(10000)
    wait_until_true(lambda: (spotify_zp_standalone.zp.AVTransport.is_playing() and
                             "0:00:43" <= spotify_zp_standalone.zp.AVTransport.GetPositionInfo()["RelTime"] <= "0:00:49"),
                    iteration_delay=1, timeout_seconds=10,
                    reason="ZP<{}> playhead not between expected 0:43 and 0:49".format(spotify_zp_standalone.zp.friendly_name))

    # seekRelative backward
    spotify_zp_standalone.zp.muse_rest.playback.seekRelative(-10000)
    wait_until_true(lambda: (spotify_zp_standalone.zp.AVTransport.is_playing() and
                             "0:00:33" <= spotify_zp_standalone.zp.AVTransport.GetPositionInfo()["RelTime"] <= "0:00:39"),
                    iteration_delay=1, timeout_seconds=10,
                    reason="ZP<{}> playhead not between expected 0:33 and 0:39".format(spotify_zp_standalone.zp.friendly_name))

    # upnp seek forward via TIME_DELTA
    spotify_zp_standalone.zp.AVTransport.Seek("TIME_DELTA", "+00:00:20")
    wait_until_true(lambda: (spotify_zp_standalone.zp.AVTransport.is_playing() and
                             "0:00:53" <= spotify_zp_standalone.zp.AVTransport.GetPositionInfo()["RelTime"] <= "0:00:59"),
                    iteration_delay=1, timeout_seconds=10,
                    reason="ZP<{}> playhead not between expected 0:53 and 0:59".format(spotify_zp_standalone.zp.friendly_name))

    # upnp seek forward via TIME_DELTA
    spotify_zp_standalone.zp.AVTransport.Seek("TIME_DELTA", "-00:00:20")
    wait_until_true(lambda: (spotify_zp_standalone.zp.AVTransport.is_playing() and
                             "0:00:33" <= spotify_zp_standalone.zp.AVTransport.GetPositionInfo()["RelTime"] <= "0:00:39"),
                    iteration_delay=1, timeout_seconds=10,
                    reason="ZP<{}> playhead not between expected 0:33 and 0:39".format(spotify_zp_standalone.zp.friendly_name))

def test_verify_zp_respects_esdk_transport_limitations(sca_playback, ws_client):
    """
    Any kind of transport control limitations for a Spotify Free experience is imposed on the player via
    their eSDK. All the transport policies are tracked together so they're no value in verifying
    each individual action. For this test case, we know that the eSDK does not allow skipping back
    if you're playing the first track in a list(free or paid), as there is no previous track.
    The test verifies via Muse that the ZP advertises that the "canSkipBack" action is not allowed and returns the expected Error event
    when we try to skip back.
    :param sca_playback:
    :param ws_client:
    :return:
    """
    # Start Spotify playback
    sca = sca_playback()
    integ_common.wait_for_some_playback(sca.zp, 3, 10)

    # Verify canSkipBack is disabled
    ws_client.playback.subscribe()
    ws_client.waitForEvent(lambda e: (e.body["playbackState"] == PlaybackState.PLAYING and
                                      not e.body["availablePlaybackActions"]["canSkipBack"]),
                           eventType=PlaybackStatusEvent)

    # Verify attempting to skip back results in expected error
    ws_client.playback.skipToPreviousTrack()
    ws_client.waitForEvent(lambda e: (e.body["errorCode"] == ErrorState.DISALLOWED_POLICY),
                           eventType=PlaybackErrorEvent)

def test_skip_to_next_from_spotify(sca_playback):
    """
    1. Initiate playback via Spotify
    2. Skip to next song in queue, verify next song is correct on Sonos side
    """
    # pos = random.randint(1, 2)
    sca = sca_playback(offset={'position': 0})
    integ_common.wait_for_some_playback(sca.zp, 3, 10)
    expected_next_track = sca.playlist['items'][1]['track']
    curr_track_zp = sca.zp.AVTransport.get_current_track_title()
    curr_track_sp = sca.spot.currently_playing_track()

                             sca.zp.AVTransport.GetPositionInfo()["RelTime"] >= "0:00:03" and

    next_track_sp = sca.spot.currently_playing_track()
    # Verify
    assert next_track_zp in next_track_sp['item']['name'], (
        'spotify and sonos track not in sync')
    assert next_track_zp in expected_next_track['name'], (
        'expected next track did not match up on zone player')
    assert next_track_sp['item']['name'] in expected_next_track['name'], \
        'expected next track did not match up on spotify'

    """
    1. Start playback on last song in a playlist
    2. Skip to next song at the end of the playlist
    3. Verify Spotify and Sonos player skips to the first song in queue and
       pauses there
    """

                             sca.spot.currently_playing_track()['item']['name'] == sca.playlist['items'][0]['track']['name']),
