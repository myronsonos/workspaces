[docs]def test_sonos_player_to_local_device_and_back_context(
        sca_playback, all_spotify_computer_type_devices):
    """
    1. Start playback on Sonos player via Spotify
    2. Switch to a local device from Spotify device picker
    3. Press the Play button on the original Sonos player
    4. Verify playback resumed from where it left off on the zone player
    """
    local_device = all_spotify_computer_type_devices[0]
    sca = sca_playback(offset={'position': 0})

    # get zone player state
    prev_zp_info = sca.zp.AVTransport.GetPositionInfo()
    prev_zp_md = XMLAccessor(prev_zp_info['TrackMetaData'])
    prev_spot_info = sca.spot.current_playback_info(market='us')

    # Wait until track played for 3 seconds
    wait_until_true(lambda: sca.zp.AVTransport.GetPositionInfo()["RelTime"] >= "0:00:03", timeout_seconds=10)

    # switch to a local device, no state needed
    common.transfer_playback_to_device(local_device, sca.spot, play=False)
    local_device_info = sca.spot.current_playback_info(market='us')

    # press Play via zone player API and get zone player state
    zp_resumed = sca.zp.AVTransport.Play()
    wait_until_true(lambda: sca.spot.is_spotify_dc_active_on_device(sca.zp),
                    timeout_seconds=10, iteration_delay=1, start_delay=3,
                    reason="Waiting for <{}> to become the active spotify DC target"
                    .format(sca.zp.friendly_name))

    # Wait until track played for 6 seconds
    wait_until_true(lambda: sca.zp.AVTransport.GetPositionInfo()["RelTime"] >= "0:00:06", timeout_seconds=10)

    curr_spot_info = sca.spot.current_playback_info(market='us')
    curr_zp_info = sca.zp.AVTransport.GetPositionInfo()
    curr_zp_md = XMLAccessor(curr_zp_info['TrackMetaData'])
    # less 9 secs to account for the wait time for refreshing state variables
    same_progress = abs(curr_spot_info['progress_ms']//1000 -
                        prev_spot_info['progress_ms']//1000) <= 9

    assert zp_resumed, 'Sonos failed to resume after switch from local device'
    assert prev_spot_info['item']['name'] == curr_spot_info['item']['name'], \
        'Spotify track names do not match after switching from local to Sonos'
    assert prev_zp_md.item.title.value == curr_zp_md.item.title.value, \
        'Sonos track names do match after switching from local to Sonos'
    assert prev_zp_md.item.title.value in curr_spot_info['item']['name'], \
        'Sonos and Spotify track names should be in sync after resuming context'
    assert local_device_info['device']['id'] != curr_spot_info['device']['id'],\
        'Local device is not suppose to be playing'
    assert prev_spot_info['device']['id'] == curr_spot_info['device']['id'], \
        'Playback did not resume on the same Sonos player'
    assert curr_spot_info['is_playing'], 'Sonos is not playing on Spotify'
    assert same_progress, \
        'Playback did not resume from the same position after context switch'




[docs]def test_sonos_player_to_sonos_player_in_different_household_context(
        sca_playback, all_playback_devices):
    """
    1. Start playback on Sonos player via Spotify
    2. In Spotify device picker, pick a Sonos device in a different household
    3. Press Play on Sonos Player A, verify the playback state(context)
    4. Press Play on Sonos Player B, verify the playback state(context)
    """
    sca = sca_playback(offset={'position': 0})

    # Wait until track played for 3 seconds
    wait_until_true(lambda: sca.zp.AVTransport.GetPositionInfo()["RelTime"] >= "0:00:03", timeout_seconds=10)

    # get a player in a different household and check if spotify can discover it
    player_b = [device for device in all_playback_devices
                if device.muse_hhid != sca.zp.muse_hhid][0]
    other_device = None
    for d in sca.spot.get_devices():
        if d['name'] == player_b.roomName:
            other_device = d
            break
    assert other_device, 'Player in a different household not found via Spotify'
    # collect previous zp state before transferring session
    prev_zp_info = sca.zp.AVTransport.GetPositionInfo()
    prev_zp_md = XMLAccessor(prev_zp_info['TrackMetaData'])
    prev_spot_info = sca.spot.current_playback_info(market='us')

    spot_a_info = sca.spot.current_playback_info(market='us')

    # transfer session to another player in a different household

    # transfer session to another player in a different household
    common.transfer_playback_to_device(player_b, sca.spot)

    # Wait until track played for 6 seconds
    wait_until_true(lambda: player_b.AVTransport.GetPositionInfo()["RelTime"] >= "0:00:06", timeout_seconds=10)

    # SWPBL - 110863
    assert not sca.zp.AVTransport.is_playing(),\
        "<{}> should not be playing".format(sca.zp)

    spot_b_info = sca.spot.current_playback_info(market='us')

    # less 12 secs to account for the wait time for refreshing state variables
    same_progress = abs(prev_spot_info['progress_ms']//1000 -
                        spot_b_info['progress_ms']//1000) <= 12

    assert same_progress, 'Playback did not resume from the same position'
    assert spot_a_info['is_playing'], \
        'Music should be playing on Spotify for Player A'
    assert spot_b_info['is_playing'], \
        'Music should be playing on Spotify for Player B'
    assert spot_a_info['item']['name'] == spot_b_info['item']['name'], \
        'Spotify track did not remain the same from Player A to Player B'
    assert spot_a_info['device']['id'] != spot_b_info['device']['id'], \
        'SCA Direct Control Session did not change to Player B'
    assert prev_spot_info['item']['name'] == spot_a_info['item']['name'], \
        'Spotify track is not the same on Player A'
    assert prev_spot_info['item']['name'] == spot_b_info['item']['name'], \
        'Spotify track is not the same on Player B'
    assert prev_zp_md.item.title.value in prev_spot_info['item']['name'], \
        'Sonos track is not matching Spotify track (not in sync)'




[docs]def test_sonos_player_to_sonos_player_in_same_household_context(
        sca_playback_two_zp_household):
    """
    1. Start playback on Sonos player via Spotify
    2. In Spotify device picker, pick a Sonos device in the same household
    3. Press Play on Sonos Player A, verify the playback state(context)
    4. Press Play on Sonos Player B, verify the playback state(context)
    """
    sca = sca_playback_two_zp_household(offset={'position': 0})

    # Wait until track played for 3 seconds
    wait_until_true(lambda: sca.zp.AVTransport.GetPositionInfo()["RelTime"] >= "0:00:03", timeout_seconds=10)

    player_b = sca.hh_zps[1]

    # get player A current state
    player_a_zp_info = sca.zp.AVTransport.GetPositionInfo()
    player_a_zp_md = XMLAccessor(player_a_zp_info['TrackMetaData'])
    player_a_spot_info = sca.spot.current_playback_info(market='us')

    # switch to zone player B
    common.transfer_playback_to_device(player_b, sca.spot)

    # Wait until track played for 6 seconds
    wait_until_true(lambda: player_b.AVTransport.GetPositionInfo()["RelTime"] >= "0:00:06", timeout_seconds=10)

    # get player B current state
    player_b_zp_info = player_b.AVTransport.GetPositionInfo()
    player_b_zp_md = XMLAccessor(player_b_zp_info['TrackMetaData'])
    player_b_spot_info = sca.spot.current_playback_info(market='us')

    final_player_a_zp_info = sca.zp.AVTransport.GetPositionInfo()
    final_player_a_spot_info = sca.spot.current_playback_info(market='us')

    # same compared to less than or equal to 6 and 12 secs to account for the
    # wait time for refreshing state variables
    same_progress = abs(player_a_spot_info['progress_ms']//1000 -
                        player_b_spot_info['progress_ms']//1000) <= 6

    assert same_progress, \
        'Track did not resume at same position after transferring playback'
    assert final_player_a_spot_info['is_playing'], \
        'Spotify doesn\'t show Player A as playing'
    assert player_a_zp_md.item.title.value == player_b_zp_md.item.title.value, \
        'Track name is not the same on both Player A and B'
    assert player_a_spot_info['item']['name'] == player_b_spot_info['item']['name'], \
        'Track name changed from Player A to Player B context'

    assert player_a_spot_info['device']['id'] == sca.device_id, \
        'Player A device ID does not match player in SCA session'
    assert final_player_a_spot_info['device']['id'] != sca.device_id, \
        'Player A device ID does match player in SCA session'




[docs]def test_pull_sonos_context_while_local_device_paused(
        sca_playback, all_spotify_computer_type_devices):
    """
    1. Start playback on Sonos player via Spotify
    2. Allow playback to continue for a few seconds
    3. In the Device Picker, select the local device
    4. Pause playback in the Spotify App
    5. Press Play/Pause on Sonos Player A
    """
    local_device = all_spotify_computer_type_devices[0]
    sca = sca_playback(offset={'position': 0})

    # Wait until track played for 3 seconds
    wait_until_true(lambda: sca.zp.AVTransport.GetPositionInfo()["RelTime"] >= "0:00:03", timeout_seconds=10)

    # let playback continue for a few seconds
    time.sleep(random.randint(1, 10))

    # get player's previous state
    prev_zp_info = sca.zp.AVTransport.GetPositionInfo()
    prev_zp_md = XMLAccessor(prev_zp_info['TrackMetaData'])
    prev_spot_info = sca.spot.current_playback_info(market='us')

    # switch to a local device
    common.transfer_playback_to_device(local_device, sca.spot, play=False)

    local_spot_info = sca.spot.current_playback_info(market='us')
    zp_a_stopped = sca.zp.AVTransport.is_playing()

    # pause local playback
    sca.spot.pause(device_id=local_device['id'])

    # resume on Player A via Play API command
    zp_a_played = sca.zp.AVTransport.Play()
    wait_until_true(lambda: sca.spot.is_spotify_dc_active_on_device(sca.zp),
                    timeout_seconds=10, iteration_delay=1, start_delay=3,
                    reason="Waiting for <{}> to become the active spotify DC target"
                    .format(sca.zp.friendly_name))

    # Wait until track played for 6 seconds
    wait_until_true(lambda: sca.zp.AVTransport.GetPositionInfo()["RelTime"] >= "0:00:06", timeout_seconds=10)

    curr_zp_info = sca.zp.AVTransport.GetPositionInfo()
    curr_zp_md = XMLAccessor(curr_zp_info['TrackMetaData'])
    curr_spot_info = sca.spot.current_playback_info(market='us')
    # will account for 3 second delay for each delay to get state variables
    same_progress = abs(curr_spot_info['progress_ms']//1000 -
                        local_spot_info['progress_ms']//1000) <= 6

    assert local_spot_info['is_playing'] is False, \
        'Local device should not be playing'
    assert zp_a_stopped is False, 'Player A should be paused'
    assert local_spot_info['device']['id'] != curr_spot_info['device']['id'], \
        'Local device should not be playing'
    assert local_spot_info['device']['type'] == 'Computer', \
        'Local device is expected to the a computer'
    assert prev_spot_info['device']['type'] == 'Speaker', \
        'Sonos speaker should be playing'
    assert curr_spot_info['device']['type'] == 'Speaker', \
        'Sonos speaker should be playing after pressing Play button'
    assert zp_a_played, 'Player A is expected to resume playback on Spotify'
    assert prev_spot_info['item']['name'] == curr_spot_info['item']['name'], \
        'Spotify track is not the same after direct control session resumed'
    assert prev_spot_info['device']['id'] == curr_spot_info['device']['id'], \
        'Playback did not resume on Player A as expected'
    assert prev_zp_md.item.title.value == curr_zp_md.item.title.value, \
        'Same track did not resume on Player A'
    assert same_progress, 'Playback did not resume from the same position'
