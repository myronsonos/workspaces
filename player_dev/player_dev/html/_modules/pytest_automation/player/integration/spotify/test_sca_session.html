[docs]def time_to_sec(t):
    """
    Take a datetime obj and converts the time to seconds.
    :param t: datetime object
    :return: Seconds
    """
    return t.hour * 3600 + t.minute * 60 + t.second

    """
    1. Start playback on SCA
    2. Verify that the direct control session is not suspended
    """
    sca = sca_playback(offset={'position': 0})
    # Wait for some seconds of playback
    integ_common.wait_for_some_playback(sca.zp, 3, 10)

    # collect current playback info
    curr_dc_is_suspended = sca.zp.AVTransport.get_directcontrol_is_suspended()
    zp_info = sca.zp.AVTransport.GetPositionInfo()
    zp_md = XMLAccessor(zp_info['TrackMetaData'])

                                                '%H:%M:%S').time())
    spot_info = sca.spot.current_playback_info(market='us')
    same_progress = (spot_info['progress_ms']//1000 <=
                     zp_state_timeout + zp_progress)

    assert curr_dc_is_suspended[0] == '0', 'SCA Session should not be suspended'
    assert sca.zp.AVTransport.is_playing(), 'Player is expected to be playing'
    assert spot_info['is_playing'], 'Spotify is expected to be playing'
    assert zp_md.item.title.value in spot_info['item']['name'], \
        'SCA session did not resume with same track name'
    assert same_progress, \
        'Track progress between player and spotify not matching'




[docs]def test_verify_sca_session_is_suspended(sca_playback, zp_state_timeout,
                                         all_spotify_computer_type_devices):
    """
    1. Switch to local device from Spotify device picker
    2. Pause playback
    3. Verify that the session is suspended
    """
    assert len(all_spotify_computer_type_devices) > 0, 'No local devices found'
    local_device = all_spotify_computer_type_devices[0]
    sca = sca_playback(offset={'position': 0})
    # Wait for some seconds of playback
    integ_common.wait_for_some_playback(sca.zp, 3, 10)
    prev_spot_title = sca.spot.currently_playing_track()['item']['name']

    # suspend session by transferring playback to non-sonos device
    spot_common.transfer_playback_to_device(local_device, sca.spot, play=False)
    dc_is_suspended = sca.zp.AVTransport.get_directcontrol_is_suspended()
    spot_info = sca.spot.current_playback_info(market='us')

    assert dc_is_suspended[0] == '1', 'SCA Session should be suspended'
    assert not spot_info['is_playing'], 'Spotify should not be playing'
    assert sca.zp.AVTransport.get_current_transport_state() == 'PAUSED_PLAYBACK', \
        'Playback state should be paused due to the session being suspended'
    assert spot_info['item']['name'] == prev_spot_title, \
        'Spotify current track does not match previous track before suspending SCA session'

def test_verify_sca_session_resumes_from_previous_paused_state(
        sca_playback, all_spotify_computer_type_devices, zp_state_timeout,
        session_delay):
    """
    1. Verify that the session resumes play back on the Sonos player from
    where it was paused on the Local device
    2. Verify that the session is not suspended in the sessionStore on the
    SCA Worker
    """
    local_device = all_spotify_computer_type_devices[0]
    sca = sca_playback(offset={'position': random.randint(1, 8)})
    # Wait for some seconds of playback
    integ_common.wait_for_some_playback(sca.zp, 3, 10)

    # collect previous zp state before suspending session
    prev_zp_info = sca.zp.AVTransport.GetPositionInfo()
    prev_zp_progress = time_to_sec(datetime.strptime(
        prev_zp_info['RelTime'], '%H:%M:%S').time())
    prev_zp_md = XMLAccessor(prev_zp_info['TrackMetaData'])
    prev_spot_info = sca.spot.current_playback_info(market='us')

    # suspend session by transferring playback to non-sonos device
    spot_common.transfer_playback_to_device(local_device, sca.spot, play=False)
    prev_dc_is_suspended = sca.zp.AVTransport.get_directcontrol_is_suspended()

    # resume session by transferring playback to the same sonos device
    spot_common.transfer_playback_to_device(sca.zp, sca.spot, play=False)
    # Wait for some seconds of playback
    integ_common.wait_for_some_playback(sca.zp, 3, 10)

    # collect current playback info
    curr_dc_is_suspended = sca.zp.AVTransport.get_directcontrol_is_suspended()
    zp_info = sca.zp.AVTransport.GetPositionInfo()
    zp_md = XMLAccessor(zp_info['TrackMetaData'])
    curr_zp_progress = time_to_sec(datetime.strptime(zp_info['RelTime'],
                                                     '%H:%M:%S').time())
    spot_info = sca.spot.current_playback_info(market='us')
    # <= 12 represents each timeout (3 secs) to get state variable added up
    same_progress = abs(prev_zp_progress - curr_zp_progress) <= 12
    same_progress_spot = abs(prev_spot_info['progress_ms']//1000 -
                             spot_info['progress_ms']//1000) <= 12

    assert prev_dc_is_suspended == '1', 'SCA Session should be suspended'
    assert curr_dc_is_suspended == '0', 'SCA Session should not be suspended'
    assert sca.zp.AVTransport.is_playing(), 'Player is expected to be playing'
    assert spot_info['is_playing'], 'Spotify is expected to be playing'
    assert zp_md.item.title.value in prev_spot_info['item']['name'], \
        'Track name changed after session resumed, expected to be the same'
    assert prev_zp_md.item.title.value in spot_info['item']['name'], \
        'Track names are expected to remain the same between player and spotify'
    assert zp_md.item.title.value in prev_zp_md.item.title.value, \
        'Spotify and Sonos tracks not matching after resumed'
    assert zp_md.item.title.value in spot_info['item']['name'], \
        'SCA session did not resume with same track name'
    assert same_progress, 'Track did not start at the same position on player'
    assert same_progress_spot, 'Spotify track did not start at same position'




[docs]def test_verify_sca_session_resumes_on_another_household(dut_manager,
        sca_playback, all_playback_devices, all_spotify_computer_type_devices,
        zp_state_timeout, session_delay, disable_speaker_audio):
    """
    1. Verify that the session resumes play back on the Sonos player from
    where it was paused on the Local device
    2. Verify that the session resumes playing back on another zone player
    from where it was paused on a previous zone player
    3. Verify that the session is not suspended on the new zone player
    """
    local_device = all_spotify_computer_type_devices[0]
    sca = sca_playback(offset={'position': 0})
    # get a player in a different household and check if spotify can discover it
    zp_diff_hh = dut_manager.get_another_device(
        device_filter=lambda zp: zp.muse_hhid != sca.zp.muse_hhid)
    other_device = None
    for d in sca.spot.get_devices():
        if d['name'] == zp_diff_hh.roomName:
            other_device = d
            break
    assert other_device, 'Player in a different household not found via Spotify'

    # Wait for some seconds of playback
    integ_common.wait_for_some_playback(sca.zp, 3, 10)

    # collect previous zp state before suspending session
    prev_zp_info = sca.zp.AVTransport.GetPositionInfo()
    prev_zp_progress = time_to_sec(datetime.strptime(
        prev_zp_info['RelTime'], '%H:%M:%S').time())
    prev_zp_md = XMLAccessor(prev_zp_info['TrackMetaData'])
    prev_spot_info = sca.spot.current_playback_info(market='us')
    prev_dc_is_suspended = sca.zp.AVTransport.get_directcontrol_is_suspended()

    # suspend session by transferring playback to non-sonos device
    spot_common.transfer_playback_to_device(local_device, sca.spot, play=False)

    # verify that session is suspended when switched to a local device
    local_dc_is_suspended = sca.zp.AVTransport.get_directcontrol_is_suspended()

    # resume session by transferring playback to different device in another
    # household
    spot_common.transfer_playback_to_device(zp_diff_hh, sca.spot)
    # Wait for some seconds of playback
    integ_common.wait_for_some_playback(zp_diff_hh, 3, 10)

    # collect current playback info
    curr_dc_is_suspended = zp_diff_hh.AVTransport.get_directcontrol_is_suspended()
    zp_info = zp_diff_hh.AVTransport.GetPositionInfo()
    zp_md = XMLAccessor(zp_info['TrackMetaData'])
    curr_zp_progress = time_to_sec(datetime.strptime(zp_info['RelTime'],
                                                     '%H:%M:%S').time())
    curr_zp_playing = zp_diff_hh.AVTransport.get_current_transport_state()
    spot_info = sca.spot.current_playback_info(market='us')
    # 5 * zp_state_timeout represents the number of delays for state variable
    # update. So all the delays are added up.
    same_progress = abs(curr_zp_progress - prev_zp_progress) <= (
            5 * zp_state_timeout)
    same_progress_spot = abs(prev_spot_info['progress_ms'] // 1000 -
                             spot_info['progress_ms'] // 1000) <= (
            5 * zp_state_timeout)

    assert prev_dc_is_suspended == '0', \
        'SCA Session should not be suspended on zone player'
    assert local_dc_is_suspended == '1', \
        'SCA session should be suspended on zone player'
    assert curr_dc_is_suspended == '0', 'SCA Session should not be suspended'
    assert curr_zp_playing in ('PLAYING', 'TRANSITIONING'), \
        'zone player is expected to be playing'
    assert zp_md.item.title.value in prev_spot_info['item']['name'], \
        'Spotify and Sonos tracks not matching after session transfer'
    assert prev_zp_md.item.title.value in spot_info['item']['name'], \
        'Sonos track is different from current Spotify playback'
    assert prev_zp_md.item.title.value in zp_md.item.title.value, \
        'Tracks are different after changing playback to another household'
    assert same_progress, 'Track did not start at the same position on player'
    assert same_progress_spot, 'Spotify track did not start at same position'
