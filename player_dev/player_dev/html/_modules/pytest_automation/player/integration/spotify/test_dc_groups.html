def test_avt_restores_correctly_on_group_gc_after_delegation(get_spotify_sn, spotify_two_zp_household):
    """
    SWPBL-119555
    Verify that the original group GC reverts its AVT and is able to play it
    :param spotify_two_zp_household:
    :return:
    """
    # Play tunein on GC
    spotify_two_zp_household.zp.muse_rest.playback.loadStream(stationId=TUNEIN_STATIONID,
                                                              playOnCompletion=True)
    # Wait for right metadata and playstate
    wait_until_true(lambda: "river" in spotify_two_zp_household.zp.muse_rest.playbackMetadata.
                    getMetadataStatus()["container"]["name"].lower() and
                    spotify_two_zp_household.zp.AVTransport.is_playing(),
                    timeout_seconds=10, iteration_delay=1,
                    reason="Waiting for The River station metadata to show up")
    # Wait for a bit of playback
    integ_common.wait_for_some_playback(spotify_two_zp_household.zp, 3, 10)

    # Switch to spotify DC
    spot_common.login_and_play_spotify_dc_session(spotify_two_zp_household.zp, get_spotify_sn,
                                                  pause_after_dc_start=False, autostart_dc=True)

    # get spotify client track info
    spot_api_playback_track_name = spotify_two_zp_household.spot.current_playback_info(market='us')['item']['name']

    # Create the group
    spotify_two_zp_household.hh_zps[1].AVTransport.join_group(spotify_two_zp_household.zp)

    # Wait for playing state on the member
    wait_until_true(lambda: spotify_two_zp_household.hh_zps[1].AVTransport.is_playing(),
                    timeout_seconds=10, iteration_delay=1,
                    reason="Waiting for <{}> to start playing".format(spotify_two_zp_household.hh_zps[1].friendly_name))
    # Wait for some more playback after grouping
    integ_common.wait_for_some_playback(spotify_two_zp_household.zp, 3, 10)

    # GC delegates and leaves the group, previous AVT source (tunein is expected in this case) is restored
    # the GC should not be playing
    spotify_two_zp_household.zp.AVTransport.BecomeCoordinatorOfStandaloneGroup(timeout_seconds=10)
    wait_until_true(lambda: "service" in
                            spotify_two_zp_household.zp.muse_rest.playbackMetadata.getMetadataStatus()
                            ["container"]
                            and
                            "tunein" in
                            spotify_two_zp_household.zp.muse_rest.playbackMetadata.getMetadataStatus()
                            ["container"]["service"]["name"].lower()
                            and
                            not spotify_two_zp_household.zp.AVTransport.is_playing(),
                    timeout_seconds=10, iteration_delay=1,
                    reason="Waiting for tunein service metadata to show up again on <{}>"
                    .format(spotify_two_zp_household.zp.friendly_name))

    # Verify the original group member's md is still playing spotify
    wait_until_true(lambda: "spotify" in spotify_two_zp_household.hh_zps[1].muse_rest.playbackMetadata.
                    getMetadataStatus()["container"]["name"].lower() and
                    spotify_two_zp_household.hh_zps[1].AVTransport.is_playing(),
                    timeout_seconds=10, iteration_delay=1,
                    reason="Waiting for Spotify station metadata to still be on <{}>"
                    .format(spotify_two_zp_household.hh_zps[1].friendly_name))

    zp_track_name = spotify_two_zp_household.hh_zps[1].muse_rest.playbackMetadata. \
           getMetadataStatus()["currentItem"]["track"]["name"]

    assert zp_track_name == spot_api_playback_track_name, \
        "Track name mismatch between Spotify App and {}".format(spotify_two_zp_household.hh_zps[1].friendly_name)

    # Attempt to play original GC
    spotify_two_zp_household.zp.AVTransport.Play_and_wait()

    # The AVT should have reverted back to original tuneIn source
    wait_until_true(lambda: "river" in spotify_two_zp_household.zp.muse_rest.playbackMetadata.
                    getMetadataStatus()["container"]["name"].lower(),
                    timeout_seconds=10, iteration_delay=1,
                    reason="Waiting for The River station metadata to show up again on <{}>"
                    .format(spotify_two_zp_household.zp.friendly_name))




[docs]def verify_group_volumes_match(spot_gc, spot_session, exp_volume):
    """
    Waits for Sonos and Spotify volumes to match expected, +/- 1

    :param spot_gc:
    :param spot_session:
    :param exp_volume:
    :return:
    """

    # Because of SWPBL-119646, the spotify and sonos volumes may be off by up to 1%
    upper = exp_volume + 1
    lower = exp_volume - 1
    format_msg = "{} group volume incorrect for {}. Actual value <{}> was not within acceptable range <{} - {}>".format

    wait_until_true(lambda:
                    lower <=
                    spot_gc.muse_rest.groupVolume.getVolume()["volume"]
                    <= upper,
                    iteration_delay=1, timeout_seconds=10,
                    reason=format_msg("Sonos",
                                      spot_gc.friendly_name,
                                      spot_gc.muse_rest.groupVolume.getVolume()["volume"],
                                      upper, lower))

    wait_until_true(lambda:
                    lower <=
                    spot_session.current_playback_info()['device']['volume_percent']
                    <= upper,
                    iteration_delay=1, timeout_seconds=15,
                    reason=format_msg("Spotify",
                                      spot_gc.friendly_name,
                                      spot_session.current_playback_info()['device']['volume_percent'],
                                      upper, lower))

def test_grouped_volume(spotify_two_zp_household, direction, volume_controller):
    """
    SWPBL-121565: [Spotify ESDK] [Automation] Verify grouped volume

    Verify that sonos and spotify group volume controls match
    :param spotify_two_zp_household:
    :param direction:
    :param volume_controller:
    :return:
    """

    # Start a Spotify session on the GC
    if not spotify_two_zp_household.hh_zps[0].AVTransport.is_playing():
        spotify_two_zp_household.hh_zps[0].AVTransport.Play_and_wait()
    integ_common.wait_for_some_playback(spotify_two_zp_household.hh_zps[0], 3, 10)

    player_volume = [10, 30]
    group_volume = sum(player_volume) / len(player_volume)
    vol_delta = 10  # Amount to set up/down

    # Before grouping, set initial player volumes on second zp to something other than the first players volume
    for zp, volume in zip(spotify_two_zp_household.hh_zps, player_volume):
        zp.muse_rest.playerVolume.setVolume(volume)

    # Group the ZPs
    spotify_two_zp_household.hh_zps[1].AVTransport.join_group(spotify_two_zp_household.hh_zps[0])
    wait_until_true(lambda: not spotify_two_zp_household.hh_zps[1].GroupManagement.is_group_coordinator_local(),
                    iteration_delay=1, timeout_seconds=10,
                    reason="{} did not become GM".format(spotify_two_zp_household.hh_zps[1].friendly_name))

    # Make sure the group volume is immediately correct
    verify_group_volumes_match(spotify_two_zp_household.hh_zps[0], spotify_two_zp_household.spot, group_volume)

    # Change the the volume and apply
    group_volume = group_volume + vol_delta if "up" in direction else group_volume - vol_delta

    prev_spot_volume = spotify_two_zp_household.spot.current_playback_info()['device']['volume_percent']
    new_spot_volume = prev_spot_volume + vol_delta if "up" in direction else prev_spot_volume - vol_delta
    if "sonos" in volume_controller:
        # Set Sonos-side
        spotify_two_zp_household.zp.muse_rest.groupVolume.setVolume(group_volume)
    else:
        # Set Spotify-side
        assert spotify_two_zp_household.spot.set_volume(new_spot_volume), 'Spotify API call failed to change volume'

    group_volume = new_spot_volume
    # Check the group volume followed suit
    verify_group_volumes_match(spotify_two_zp_household.hh_zps[0], spotify_two_zp_household.spot, group_volume)

def test_grouping_esdk_suspended_zp(spotify_two_zp_household):
    """
    SWPBL-120804
    Verify adding a ZP that is suspended (Spotify Connect) into a group playing Spotify Connect
    does not ungroup unexpectedly
    :param spotify_two_zp_household:
    :return:
    """
    # Start Spotify DC on GC if GC is not active Spotify DC target
    if not spotify_two_zp_household.zp.AVTransport.is_playing():
        spotify_two_zp_household.zp.AVTransport.Play_and_wait()

    # Wait until track played for 3 seconds on GC
    integ_common.wait_for_some_playback(spotify_two_zp_household.zp, 3, 7)

    # Create the group and verify GM is grouped with GC and is playing
    spotify_two_zp_household.hh_zps[1].AVTransport.join_group(spotify_two_zp_household.zp)
    common.verify_zp_is_group_member_and_playing(spotify_two_zp_household.hh_zps[1])

    # Wait for some more playback after grouping on GC
    integ_common.wait_for_some_playback(spotify_two_zp_household.zp, 5, 10)

    # Verify GM is still grouped with GC
    assert spotify_two_zp_household.hh_zps[1].GroupManagement.is_group_coordinator_local() is False, \
        "GM: <{}> ungrouped from GC: <{}>".format(spotify_two_zp_household.hh_zps[1].raw_udn,
                                                  spotify_two_zp_household.zp.raw_udn)

    # Verify GC is still active Spotify DC target
    integ_common.verify_track_name_on_zp_and_spotify(spotify_two_zp_household.zp, spotify_two_zp_household.spot)

def test_VLI_Group_ID_delegation(spotify_two_zp_household, get_spotify_sn):
    """
    SWPBL-121733
    Verify removing GC from a group of 2 ZPs playing Spotify DC sets new VLI Group ID starting with "spotify" in the
    new GC of the group and the new GC continues playing Spotify DC
    :param spotify_two_zp_household:
    :param get_spotify_sn:
    :return:
    """
    gc = spotify_two_zp_household.hh_zps[0]
    gm = spotify_two_zp_household.hh_zps[1]

    # Start Spotify DC on GC if GC is not active Spotify DC target, wait for some playback
    spot_common.login_and_play_spotify_dc_session(gc, get_spotify_sn, autostart_dc=True,
                                                  pause_after_dc_start=False)

    # Create the group and verify GM is grouped with GC and is playing
    gm.AVTransport.join_group(gc)
    common.verify_zp_is_group_member_and_playing(gm)

    # Verify GC's VLI Group ID contains spotify
    assert "spotify" in player_common.get_zp_vli(gc), \
        "Expected \"spotify\" in VLInGroupID: <{}> on {}".format(player_common.get_zp_vli(gc), gc)

    # Ungroup GC from the existing Spotify DC group
    gc.AVTransport.leave_group()

    # Wait for some more playback after un-grouping on GM(new GC)
    integ_common.wait_for_some_playback(gm, 5, 10)

    # Verify GC is not grouped with GM
    assert gc.GroupManagement.is_group_coordinator_local(), "GC: <{}> still grouped with GM: <{}>".format(gc, gm)

    # Verify GM(new GC) is still active Spotify DC target
    integ_common.verify_track_name_on_zp_and_spotify(gm, spotify_two_zp_household.spot)

    # Verify GM(new GC)'s VLI Group ID contains spotify
    assert "spotify" in player_common.get_zp_vli(gm), \
        "Expected \"spotify\" in VLInGroupID: <{}> on {}".format(player_common.get_zp_vli(gm), gm)

def test_volume_updated_to_groupVolume_after_delegation(spotify_zp_standalone, dut_manager):
    """
    Verify spotify client volume and sonos groupVolume is updated to the new groupVolume
    when the GC is delegated to a new group with 2 zps instead of the 3 originally
    :param dut_manager:
    :param spotify_zp_standalone:
    :return:
    """

    # Get 3 zps
    zp1 = spotify_zp_standalone.zp
    zp2 = dut_manager.test_devices[1]
    zp3 = dut_manager.get_another_device()

    # Start a Spotify session on the GC
    if not zp2.AVTransport.is_playing():
        zp2.AVTransport.Play_and_wait()
    integ_common.wait_for_some_playback(zp2, 3, 10)

    player_volume = [10, 30, 50]
    group_volume = sum(player_volume) / len(player_volume)

    # Before grouping, set initial player volumes on zps to be different from one another
    for a_zp, volume in zip([zp1, zp2, zp3], player_volume):
        a_zp.muse_rest.playerVolume.setVolume(volume)

    # Group all the ZPs
    zp1.AVTransport.join_group(zp2)
    wait_until_true(lambda: not zp1.GroupManagement.is_group_coordinator_local(),
                    iteration_delay=1, timeout_seconds=10,
                    reason="{} did not become GM".format(zp1.friendly_name))

    zp3.AVTransport.join_group(get_gc_of_group([zp2, zp1]))
    wait_until_true(lambda: not zp3.GroupManagement.is_group_coordinator_local(),
                    iteration_delay=1, timeout_seconds=10,
                    reason="{} did not become GM".format(zp3.friendly_name))

    # Check groupVolume is the same on Sonos and Spotify
    verify_group_volumes_match(get_gc_of_group([zp1, zp2, zp3]), spotify_zp_standalone.spot, group_volume)

    # Remove GC from the group (zp2) -> New group with delegated GC and GM
    zp2.AVTransport.leave_group()
    wait_until_true(lambda: zp2.GroupManagement.is_group_coordinator_local(),
                    iteration_delay=1, timeout_seconds=10,
                    reason="{} did not become GM".format(zp2.friendly_name))

    # Re-calculate group volume with new group
    player_volume = [10, 50]
    group_volume = sum(player_volume) / len(player_volume)

    # Check the groupVolume is the same on Sonos and Spotify
    verify_group_volumes_match(get_gc_of_group([zp1, zp3]), spotify_zp_standalone.spot, group_volume)
