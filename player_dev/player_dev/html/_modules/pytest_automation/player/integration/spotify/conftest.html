[docs]def pytest_addoption(parser):
    import argparse

    class SpotifyConfigAction(argparse.Action):
        def __init__(self, option_strings, dest, nargs=None, **kwargs):
            if nargs is not None:
                raise ValueError("nargs not allowed")
            super(SpotifyConfigAction, self).__init__(option_strings, dest,
                                                      **kwargs)

        def __call__(self, parser, namespace, values, option_string=None):
            # read JSON file and set spotify_config attribute
            with open(values, 'r') as json_data:
                d = json.load(json_data)
                setattr(namespace, self.dest, d)

    spot_default_config = {
            'access_token': os.getenv('SPOT_ACCESS_TOKEN'),
            'refresh_token': os.getenv('SPOT_REFRESH_TOKEN'),
            'client_id': os.getenv('SPOT_CLIENT_ID'),
            'client_secret': os.getenv('SPOT_CLIENT_SECRET')
        }
    parser.addoption("--spotify_config", action=SpotifyConfigAction,
                     default=spot_default_config,
                     help="file with spotify client_id, client_secret, tokens")

    parser.addoption("--time_delay", action="store", type=int, default=3,
                     help="time to delay for zone player state variables")
    parser.addoption("--session_sleep", action="store", type=int, default=60,
                     help="sleep time after start/stop direct control session")
    parser.addoption("--mitc_env", action="store", type=str, default="prod",
                     choices=TESTING_ENVIRONMENTS,
                     help="change MITC environment")
    parser.addoption("--spotify_sn", action="store", type=str, default=None,
                     help="spotify SerialNum as defined in accounts.xml")

def get_spotify_sn(request, dut_manager, spotify_sn_cmdline):
    """
    Return the spotify account if there's only 1 or a specific account if there's multiple on the hh
    :param request:
    :param dut_manager:
    :param spotify_sn_cmdline:
    :return:
    """
    # Skip the test case if we're running on locked builds since it won't have access to the testpoint later on
    player_common.skip_locked_build("testpoints")

    spot_account = dut_manager.all_playback_devices[0].SystemProperties.get_service_account(
        spot_common.SPOTIFY_SERVICE_ID)

    if len(spot_account) == 1:
        return spot_account[0]
        # If there's not spotify accounts on the hh, fail
    elif len(spot_account) > 1:
        # If there's more then one Spotify accounts,
        # try to figure out which one to use in the following order:
        #
        # 1. Python module level delcaration: "SPOTIFY_SN=1"
        # 2. Linux shell environmental variable: "export SPOTIFY_SN=1"
        # 3. Pytest command line option: "pytest --spotify_sn 1"
        if hasattr(request.node.module, "SPOTIFY_SN"):
            spotify_sn = request.node.module.SPOTIFY_SN
        elif "SPOTIFY_SN" in os.environ:
            spotify_sn = os.environ["SPOTIFY_SN"]
        elif spotify_sn_cmdline is not None:
            spotify_sn = spotify_sn_cmdline
        else:
            pytest.fail("More than one Spotify SerialNum exist on this hh; "
                        "please set SPOTIFY_SN or use --spotify_sn arg")

        for acct in spot_account:
            if acct.serial_number == spotify_sn:
                return acct
        else:
            # If the specified sn is not found, fail
            pytest.fail("SPOTIFY_SN/--spotify_sn value does not match any Spotify SerialNum in accounts.xml")
    else:
        pytest.fail("No Spotify SMAPI account exists on this hh")

def spotify_zp_standalone(get_spotify_sn, ungroup_unbond_cleanup, get_module_dut,
                          login_esdk_via_testpoint, spotify_client, all_spotify_speaker_type_devices):
    """
    Returns one zone player with no other group members in sync with spotify
    connect adapter.
    :param get_spotify_sn:
    :param ungroup_unbond_cleanup:
    :param get_module_dut:
    :param login_esdk_via_testpoint:
    :param spotify_client:
    :param all_spotify_speaker_type_devices:
    :return:
    """
    # The esdk login fixture should make players show up in the /devices output
    assert len(all_spotify_speaker_type_devices) > 0, 'Spotify available device list is empty.'

    # TO-DO: Add condition to switch the environment on the players if necessary
    for device in all_spotify_speaker_type_devices:
        if get_module_dut.DeviceProperties.get_zone_name() == device['name']:
            # Force start the DC session on the zp
            spot_common.login_and_play_spotify_dc_session(get_module_dut, get_spotify_sn, autostart_dc=True)

            # Reset the client
            spotify_client.set_volume(10)
            spotify_client.pause()
            spotify_client.shuffle("false")
            spotify_client.repeat_mode("off")

            def format_playback(current_playback_info):
                return "Failed to reset spotify client state.\n\t" \
                       "volume_percent: Expected  <10>,    Actual <{}>\n\t" \
                       "is_playing:     Expected  <False>, Actual <{}>\n\t" \
                       "shuffle_state:  Expected  <False>, Actual <{}>\n\t" \
                       "repeat_state:   Expected  <off>,   Actual <{}>"\
                       .format(current_playback_info['device']['volume_percent'],
                               current_playback_info['is_playing'],
                               current_playback_info['shuffle_state'],
                               current_playback_info['repeat_state'])

            wait_until_true(lambda: (
                    abs(spotify_client.current_playback_info()['device']['volume_percent']-10) <= 2 and
                    not spotify_client.current_playback_info()['is_playing'] and
                    not spotify_client.current_playback_info()['shuffle_state'] and
                    spotify_client.current_playback_info()['repeat_state'] == "off"),
                            iteration_delay=1, timeout_seconds=10,
                            reason=format_playback(spotify_client.current_playback_info()))

            # Verify the active speaker
            wait_until_true(lambda: spotify_client.is_spotify_dc_active_on_device(get_module_dut),
                            timeout_seconds=10, iteration_delay=1,
                            reason="Waiting for <{}> to become the active spotify DC target".format(
                                get_module_dut.friendly_name))

            yield SCASession(spot=spotify_client, zp=get_module_dut,
                             hh_zps=None, device_id=device['id'])

            spotify_client.pause()
            return
    else:
        assert False,\
            'Could not find <{}> in SpotifyAPI /devices output'.format(get_module_dut.friendly_name)

    :return: SCAPlaylist with playlist as a dict and longest track
    :rtype: :obj: `namedtuple SCAPlaylist`

    user_id = get_spotify_user['id']
    # Get a list of newly released albums
    # Use artist's IDs to seed recommendations to return 20 tracks
    new_releases = spotify_client.new_releases(country='us', limit=5)
    artist_ids = [album['artists'][0]['id']
                    for album in new_releases['albums']['items']]

        'Artist ID list is empty. Needed for recommendations.'
    # get track recommendations
    rec = spotify_client.recommendations(seed_artists=artist_ids, limit=20)
    assert rec is not None and 'tracks' in rec.keys(), \
        'No tracks returned from Spotify recommendations.'

    playlist = spotify_client.user_playlist_create(user_id,name,public=True)
    uris = [t['uri'] for t in rec['tracks']]
    total_tracks = len(rec['tracks'])
    longest_track = max([t['duration_ms'] for t in rec['tracks']])
    # Create new playlist and add tracks
    add_tracks = spotify_client.user_playlist_add_tracks(user_id,
                                                    playlist['id'], uris)
    time.sleep(1)
    # Add tracks and update playlist
    updated_playlist = spotify_client.user_playlist_tracks(user_id,
            playlist['id'], market='us')
    # Remove tracks that are not playable
    updated_playlist['items'] = [t for t in updated_playlist['items']
                            if t['track']['is_playable'] is True]
    playlist.update(updated_playlist)
    playlist.pop('tracks', None)  # tracks is empty when playlist is created
    p = SCAPlaylist(longest=longest_track, playlist=playlist)
    yield p
    # delete playlist
    spotify_client.user_playlist_unfollow(user_id, playlist['id'])

def sca_playback(spotify_zp_standalone, get_playlist):
    """
    Initializes spotifyc direct control playback with a specific playlist.

    :param spotify_zp_standalone: namedtuple with spotify client and zone player
    :param dict: dict of a spotify created playlist
    :param disable_speaker_audio:
    :return: namedtuple that contains a zone player, playlist and spotify client
    object.
    :rtype: :obj: `function`
    """
    zp = spotify_zp_standalone.zp
    spot = spotify_zp_standalone.spot
    SCA = namedtuple('SCA', 'spot zp device_id playlist')

    def start_playback(device=zp, context_uri=get_playlist.playlist['uri'],
                       offset={'position': 0}, position_ms=0, throw_error=False):

        # Play the spotify DC session
        played = spot.play(device_id=spotify_zp_standalone.device_id, context_uri=context_uri,
                           offset=offset, position_ms=position_ms, throw_error=throw_error)
        assert played, 'spotify playback did not start in test fixture'

        # Wait until our ZP starts playing
        wait_until_true(lambda: device.AVTransport.is_playing(),
                        timeout_seconds=10, iteration_delay=1,
                        reason="Waiting for <{}> to be playing".format(device.friendly_name))

        # Wait until the Spotify side reports playhead is in the expected position
        # This check is currently disabled due to issues with the Spotify backend keeping in sync
        # with our ZP state: https://jira.sonos.com/browse/SPOTIFY-180
        wait_until_true(lambda: (spot.current_playback_info(market='us')['is_playing'] and
                                 abs(int(spot.current_playback_info(market='us')['progress_ms'])-position_ms) < 3000),
                        timeout_seconds=10, iteration_delay=1, start_delay=1,
                        reason="Waiting for Spotify client to start playing at {} ms".format(position_ms))

        return SCA(spot=spot, zp=zp, playlist=get_playlist.playlist,
                   device_id=spotify_zp_standalone.device_id)

    yield start_playback
    spot.pause()

def cloudqueue_metrics_and_force_esdk_login(mock_cq_server_module, spotify_zp_standalone, zp_state_timeout,
                                            get_spotify_sn):
    """
    Sets the eventmetrics URL on a ZP to send reports to mock Cloudqueue server
    and then forces eSDK login on the ZP after anacapa restart
    :param mock_cq_server_module:
    :param spotify_zp_standalone:
    :param zp_state_timeout:
    :param get_spotify_sn:
    :return:
    """
    # Set the metrics URL
    metrics_override = {EVENTMETRICS_URL_OVERRIDE_NAME: '{}/usageMetrics'.format(mock_cq_server_module.CQ_URI)}
    if not spotify_zp_standalone.zp.SystemProperties.verifySysSettingStrings(metrics_override):
        spotify_zp_standalone.zp.SystemProperties.SetString(
            EVENTMETRICS_URL_OVERRIDE_NAME, metrics_override[EVENTMETRICS_URL_OVERRIDE_NAME])
        spotify_zp_standalone.zp.bounce_anacapa_and_wait()

        # Wait for AVTransport Service to be re-initialized
        spotify_zp_standalone.zp.AVTransport.get_state_variable('AVTransportURI', wait=True, timeout=zp_state_timeout)
        # eSDK login after anacapa restart
        spot_common.login_and_play_spotify_dc_session(spotify_zp_standalone.zp, get_spotify_sn, autostart_dc=False)
