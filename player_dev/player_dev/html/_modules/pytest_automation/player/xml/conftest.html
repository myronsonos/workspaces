def zp_status_endpoint_urls(unique_platform_devices):
    """
    crawl http://<ip>:1400/status for additional
    :param unique_platform_devices: each unique platform playback
    device available on the testrunner.
    :return: string for all URLs that stem from the http://<ip>:1400/status
    base URL
    """
    zp = unique_platform_devices
    base_url = "http://{}:1400/status".format(zp.ip)
    resp = requests.get(base_url)
    soup = BeautifulSoup(resp.text, 'html.parser')
    links = soup.find_all('a')
    urls = ["http://{}:1400".format(zp.ip) + ext
            for ext in [link['href']
                        for link in links if link.has_attr('href')]]
    # Append the XSL style sheet used to render XML date in a web browser
    xml_stylesheet = "http://{}:1400/xml/review.xsl".format(zp.ip)
    urls.append(xml_stylesheet)
    yield urls
