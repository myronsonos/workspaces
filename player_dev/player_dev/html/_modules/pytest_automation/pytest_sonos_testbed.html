def pytest_configure(config):
    """
    Pytest hook which runs before a session starts. Performs testbed cleanup.
    """
    if not hasattr(pytest, "testbed_config"):
        setattr(pytest, 'testbed_config', load_EXECUTION_ENVIRONMENT())
        setattr(pytest, 'logger', logging.getLogger())

    config.option.json = pytest.testbed_config
    if config.option.clean_at_start:
        _add_build_info_to_pytest_config(config)
        log_fn = setup_validate_testbed_logger("cleanup_testbed")
        pytest.logger.info("Starting Testbed Cleanup")
        try:
            TestbedCleanup.create_from_namespace(config.option).run()
        except Exception:
            # If we caught an exception cleaning up a Jenkins testbed, record
            # it in our output junit XML file and don't run any tests.
            if running_on_jenkins_testbed():
                _record_testbed_error(config, "cleanup")
                config.pluginmanager.getplugin("session").shouldstop = True
            else:
                raise
        stop_validate_testbed_logger(log_fn)
        pytest.logger.info("Ending Testbed Cleanup")



[docs]@pytest.hookimpl(tryfirst=True)
def pytest_sessionfinish(session, exitstatus):
    """
    Pytest hook which runs when a Pytest session finishes.

    Performs testbed scanning, and ensures a zero exit code when running
    from Jenkins.
    """
    config = session.config
    if config.option.scan_at_end:
        _add_build_info_to_pytest_config(config)
        log_fn = setup_validate_testbed_logger("check_testbed")
        try:
            pytest.logger.info("Starting Testbed Check")
            TestbedChecker.create_from_namespace(config.option).run()
        except Exception:
            if running_on_jenkins_testbed():
                # If we caught an exception scanning the testbed after running
                # automation, AND we're running on a Jenkins testbed, record
                # the exception in our output junit XML file.
                _record_testbed_error(config, "check")
                pytest.logger.exception("Caught exception checking testbed")
            else:
                raise
        stop_validate_testbed_logger(log_fn)
        pytest.logger.info("Ending Testbed Check")

def pytest_unconfigure(config):
    if running_on_jenkins_testbed():
        # Change exit code to zero if we failed when running on a Jenkins
        # testbed. Jenkins marks jobs with non-zero exit codes as failures. If
        # the job passes, but testcases are recorded as failed in the Junit XML
        # file, Jenkins will mark the job 'unstable'. We prefer to designate
        # our failing automation as 'unstable', rather than failed.
        session = config.pluginmanager.get_plugin("session")
        if session and session.exitstatus != PYTEST_EXIT_OK:
            session.exitstatus = PYTEST_EXIT_OK


def _record_testbed_error(config, stage):
    """
    Writes testbed cleanup/check exceptions to the test JUnit XML file.
    """
    if config.getoption("xmlpath", None):
        junit = config.pluginmanager.getplugin("junitxml")
        xml = config._xml
        reporter = xml.node_reporter('internal')
        reporter.attrs.update(classname='pytest', name='internal')
        reporter._add_simple(
                junit.Junit.error, "testbed {} error".format(stage),
                traceback.format_exc())
        pytest.logger.exception("Caught exception checking testbed")

def _add_build_info_to_pytest_config(config):
    """
    Create a "build_info" section in the testbed configuration, so that
    TestbedChecker knows where to look for test artifacts.
    """
    if (getattr(config.option, "json") and not
            hasattr(config.option.json, "build_info")):
        version = config.getoption("update_version", None)
        branch = config.getoption("update_branch", None)
        update_url = config.getoption("update_url", None)
        try:
            config.option.json.build_info = SonosBuildServerUtil(
                    url=update_url,
                    version=version,
                    branch=branch).as_build_info()
        except (requests.exceptions.RequestException, SonosBuildServerError):
            config.option.json.build_info = (
                    SonosBuildServerUtil().as_build_info())
