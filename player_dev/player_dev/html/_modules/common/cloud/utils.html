[docs]class CloudUtilsMixin(object):
    """
    Mixin class for parsing common arguments exclusive to cloud fixtures and
    providing general utility functions.
    """

    # Default overrides for establishing cloud connections to different
    # services
    SERVICE_OVERRIDE = L7_OVERRIDE_NAME
    SERVICE_OVERRIDE_URL = None
    TEST_ENV = MITC_ENVIRONMENT

    def __init__(self, *args, **kwargs):
        args = self._get_cmdline_args()
        self.secure_register = not args.no_secure_reg
        self.supports_lechmere = not args.no_lechmere
        self.use_overrides = not args.no_overrides
        self.override_exceptions = FrozenAttributeDict(args.exc_overrides or [])
        super(CloudUtilsMixin, self).__init__()

    def _get_cmdline_args(self):
        """
        Parse command line arguments. Places all arguments back into sys.argv

        :return: Namespace of parsed arguments
        :rtype: :class:`argparse.Namespace`
        """
        # Store a copy of command-line args
        cmdline_args = sys.argv[:]

        # Parse known args
        parser = argparse.ArgumentParser(description="{} arguments".
            format(self.__class__.__name__))
        parser.add_argument("--no_secure_reg", dest="no_secure_reg",
            action="store_true", default=False,
            help="Secure registration should not be used "
            "in tests that require cloud connections.")
        parser.add_argument("--no_lechmere_support", dest="no_lechmere",
            action="store_true", default=False,
            help="Firmware under test does not support Lechmere connections."
            " When possible, skip verifications that require Lechmere.")
        parser.add_argument("--no_overrides", dest="no_overrides",
            action="store_true", default=False,
            help="Firmware under test should not set overrides. This is useful "
            "for testing firmware 'out-of-the-box', as many tests rely heavily "
            "on using overrides to change endpoints to point to different "
            "service endpoints.")
        parser.add_argument("--override_exceptions", dest="exc_overrides",
            action=StoreKeyValuePair, nargs="*",
            help="Exceptions to the above 'no_overrides' argument. Defaults to "
            "This argument consumes an arbitrary number of key=value pairs.")
        args, _ = parser.parse_known_args()

        # Place all arguments back onto the command-line for other fixtures
        sys.argv = cmdline_args
        return args

    @property
    def overrides_dict(self):
        """
        Default dictionary of overrides.
        """
        if self.SERVICE_OVERRIDE_URL is None:
            d = {}
        else:
            d = {self.SERVICE_OVERRIDE: self.SERVICE_OVERRIDE_URL}

        if self.TEST_ENV != 'prod':
            d.update({'O_SONOS_CLOUD_API_TEST_ENV': self.TEST_ENV})
        return d


[docs]    def get_overrides(self, custom_url=None):
        """
        Get the list of overrides. Allow specification of a custom
        url for the service pointer override for negative testing.

        :param str custom_url: Optional string to use for url override

        :return: Immutable dictionary of the form override:value
        :rtype: :class:`~custom_dicts.FrozenAttributeDict`
        """
        overrides = {}

        if self.use_overrides:
            # Use a copy of the overrides dictionary in case we need to
            # replace a value.
            overrides.update(self.overrides_dict)
            # Handle case where custom_url is specified
            if custom_url != self.SERVICE_OVERRIDE_URL and custom_url != None:
                overrides[self.SERVICE_OVERRIDE] = custom_url
            return FrozenAttributeDict(overrides)
        else:
            overrides.update(self.override_exceptions)
            # Handle case where custom_url is specified
            if custom_url != self.SERVICE_OVERRIDE_URL and custom_url != None:
                # Log a warning here if the cert override is not in the exception
                # list, but otherwise allow it to be set. This allows the testing
                # of certain negative cases where the url is pointed at an
                # invalid endpoint, but is otherwise unchanged as a part of normal
                # functionality
                if self.SERVICE_OVERRIDE not in self.override_exceptions:
                    self.logger.warn("{} not in specified override exceptions {}, "
                        "use case may result in unexpected behavior.".format(
                        self.SERVICE_OVERRIDE, self.override_exceptions))
                # Copy the dict into a mutable object as override_exceptions
                # is immutable
                overrides[self.SERVICE_OVERRIDE] = custom_url

        return FrozenAttributeDict(overrides)



[docs]    def set_string_overrides(self, zp, url=None, enable=True):
        """
        Set or remove the appropriate string overrides for the service url

        :param zp: ZonePlayer
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`
        :param str url: Url to set service string override, uses default if None
        :param bool enable: Add override if true, else remove it

        :return bool: UPnP rc OK?
        """
        if url is None:
            url = self.SERVICE_OVERRIDE_URL

        overrides = self.get_overrides(url)
        svc_override_present = self.SERVICE_OVERRIDE in overrides.keys()

        if enable:
            self.logger.info("Setting overrides: {}".format(overrides))
            return (all(map(zp.SystemProperties.SetString,
                overrides.keys(), overrides.values())) and
                    # Always explicitly remove the service override if setting
                    # overrides when the service name is not included. This protects
                    # against the player having a bad override value after a negative
                    # test or on boot.
                    (zp.SystemProperties.Remove(self.SERVICE_OVERRIDE) if not
                svc_override_present else True)
                    and MPCSC.remove_mutual_ssl_overrides(zp))
        else:
            # Always remove the service override even if it is not included
            # in the list of overrides. This protects against having a bad
            # override after a negative test regardless of override_exemptions
            override_keys = overrides.keys()[:]
            if self.SERVICE_OVERRIDE not in override_keys:
                override_keys.append(self.SERVICE_OVERRIDE)

            # Don't remove testenv
            try:
                override_keys.remove('O_SONOS_CLOUD_API_TEST_ENV')
            except ValueError:
                pass

            self.logger.info("Removing overrides: {}".format(override_keys))
            return all(map(zp.SystemProperties.Remove, override_keys)) and MPCSC.remove_mutual_ssl_overrides(zp)



[docs]    def set_string_overrides_and_bounce_anacapa(self, zps, url=None, enable=True):
        """
        Set or remove the appropriate string overrides for the cert service url

        :param zp: ZonePlayer
        :type zp: :obj:`list` of :class:`~sonos.client.internal.SonosZoneComponent`
        :param str url: Url to set cert string override to, uses default if None
        :param bool enable: Add override if true, else remove it

        :return bool: Operation was successful (upnp return codes OK and anacapa bounced?)
        """
        if not self.set_string_overrides(zps[0], url=url, enable=enable):
            return False

        sleep(2)

        run_concurrent_zone_player_processes(
            SonosZoneComponent.bounce_anacapa_and_wait, zps)

        return True



[docs]    @contextmanager
    def temp_non_secure_registered_cm(self, zp):
        """
        Contextmanager to temporarily move the registration cert, bounce
        anacapa, then move the registration cert back and bounce anacapa
        again on exit.

        Basically does nothing if you're not secure registered.

        :return:
        """
        had_cert = False
        try:
            if zp.cli.has_reg_cert():
                had_cert = True
                zp.cli.remove_file(TEMP_REG_CERT_DIR)
                zp.cli.command('mv -f {} {}'.format(zp.cli.get_reg_cert_root_dir(), TEMP_REG_CERT_DIR))
                zp.bounce_anacapa_and_wait()
            yield
        finally:
            if had_cert:
                zp.cli.command('mv -f {} {}'.format(TEMP_REG_CERT_DIR, zp.cli.get_reg_cert_root_dir()))
                zp.cli.remove_file(TEMP_REG_CERT_DIR)
                zp.bounce_anacapa_and_wait()



[docs]    @contextmanager
    def temp_copy_secure_registered_cm(self, zp):
        """
        Contextmanager to temporarily copy the registration cert

        :return:
        """
        try:
            zp.cli.remove_file(TEMP_REG_CERT_DIR)
            zp.cli.command('cp -rf {} {}'.format(zp.cli.get_reg_cert_root_dir(), TEMP_REG_CERT_DIR))
            yield
        finally:
            zp.cli.remove_reg_certs()
            zp.cli.command('mv -f {} {}'.format(TEMP_REG_CERT_DIR, zp.cli.get_reg_cert_root_dir()))

        apilog = "{}/{}/logs".format(base_url, type)
        data = requests.get(url=apilog, verify=False).text
        payload = json.loads(data, encoding='utf-8')
        return payload
