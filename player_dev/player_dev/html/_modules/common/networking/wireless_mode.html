        """
        Enable power on one or more zone components

        :param zones:one or more zone components
        :type zones: :class:`~sonos.client.internal.SonosZoneComponent`
        """
        for zone in zones:
            if zone.power and zone.power.is_off():
                zone.power.on()
        self._disabled_zones = self.disabled_zones - DevicesSet(zones)

    def _force_device_removal(self, zone):
        try:
            zone.stop_anacapa()
        except TimeoutError:
            zone.start_anacapa()

[docs]    def disable_zones(self, *zones):
        """
        Usubscribe UPnP events and power off one or more zone components
        :param zones:one or more zone components
        :type zones: :class:`~sonos.client.internal.SonosZoneComponent`
        """
        for zone in zones:
            self._force_device_removal(zone)
            zone.cli.disconnect()
            zone.power.off()
        self.disabled_zones.update(zones)


    def _get_zones_to_disable(self, exclude_zones):
        """
        Return all zone components in testbed that are not in exclude_zones and
        are enabled

        :param zones:one or more zone components
        :type zones: :obj:`list`
        :return: zone components not in zones
        :rtype: :obj:`list`
        """
        return list(DevicesSet(self._all_testbed_devices) -
                    DevicesSet(exclude_zones) -
                    self.disabled_zones)


[docs]    def transition_zones_to_station_mode(self, *zones):
        """
        Disable wires on zones and disconnect all other devices in
        testbed

        :param zones:one or more zone players to remove ethernet wires
            from
        :type zones: :class:`~sonos.client.internal.SonosZoneComponent`
        """
        sats = DevicesSet(sum((z.get_ht_satellites() for z in zones), []))
        station_capable_in_zones = DevicesSet([zone for zone in zones
            if not zone.is_bridge()])
        zones_set = DevicesSet([zone for zone in zones
            if zone in station_capable_in_zones and zone not in sats])
        zones_to_disable = self._get_zones_to_disable(station_capable_in_zones)
        is_already_in_station = all(run_concurrent_zone_player_processes(
            SonosZoneComponent.is_in_station_mode, zones_set))
        if is_already_in_station:
            logger.info("All expected zones already in station mode, "
                "short-circuiting station mode transition")
            self.wait_for_zones_in_station_mode(zones_set)
            self.wait_for_zones_in_sonosnet_mode(sats)
            return
        logger.info("Doing station mode transition")
        self.enable_zones(*(zones_set | sats))
        self.disable_zones(*zones_to_disable)
        self.disable_ethernet_wires_on_zones(*(zones_set | sats))
        self.subnet.clear_and_wait_for_devices(zones_set | sats,
            timeout_seconds=MAX_TRANSITION_TIME)
        self.wait_for_zones_in_station_mode(zones_set)
        self.wait_for_zones_in_sonosnet_mode(sats)



[docs]    def transition_zones_to_wired_sonosnet_mode(self, *zones):
        """
        Enable wires and atheros radio's on one or more zone players to
        transition the HH into Sonosnet mode

        :param zones:one or more zone players to remove ethernet wires
            from
        :type zones: :class:`~sonos.client.internal.SonosZoneComponent`
        """
        if all([z.wire.is_enabled() for z in zones]):
            logger.info("All expected zones in sonosnet mode, "
                "short-circuiting wired Sonosnet transition")
            self.wait_for_zones_in_sonosnet_mode(zones)
            return
        logger.info("Performing wired sonosnet transition on zones: {}".format(
                                                                        zones))
        self.disable_zones(*self._get_zones_to_disable(zones))
        self.enable_zones(*zones)
        self.enable_ethernet_wires_on_zones(*zones)
        self.subnet.clear_and_wait_for_devices(zones,
            timeout_seconds=MAX_TRANSITION_TIME)
        self.wait_for_zones_in_sonosnet_mode(zones)


    @staticmethod
    def _get_bridges(zones):
        return [zone for zone in zones if zone.is_bridge()]

    def _get_wired_device(self, zones):
        bridges = self._get_bridges(zones)
        if bridges:
            wired_zone = bridges[0]
            zones.remove(wired_zone)
        else:
            wired_zone = zones.pop()
        return wired_zone

[docs]    def transition_zones_to_wireless_sonosnet_mode(self, *wireless_devices,
                                                   **kwargs):
        """
        Unplug *wireless_devices* from ethernet and make sure at least 1 device
        in the testbed that is not one of said devices has a wire. More than one
        device may be wired given a kwarg *wired_devices* which is inclusive of
        the kwarg *first_wire* device. All other devices will be disabled (
        powered off)

        .. note::

            if no first wire is given then this method first attempts to
            find other devices not passed to it and make a first wire from
            one of those devices.  If no other devices exist then it
            attempts to make a bridge first wire

        :param wireless_devices: one or more zone players to remove ethernet
            wires from
        :type wireless_devices:
            :class:`~sonos.client.internal.SonosZoneComponent`
        :param first_wire: the sonos device that supplies first wire
        :type first_wire: :class:`~sonos.client.internal.SonosZoneComponent`
        :param wired_devices: the sonos devices to be wired passed as a kwarg
        :type wired_devices: :class:`~sonos.client.internal.SonosZoneComponent`
        """
        wired_devices = DevicesSet()
        if 'wired_devices' in kwargs:
            wired_devices.update(kwargs['wired_devices'])
        if 'first_wire' in kwargs:
            wired_devices.add(kwargs['first_wire'])
        wireless_devices = DevicesSet(wireless_devices) - wired_devices
        other_devices = DevicesSet(self._get_zones_to_disable(
            wired_devices | wireless_devices))
        if not wired_devices:
            wired_devices.add(self._get_wired_device(
                other_devices or wireless_devices))
            other_devices -= wired_devices
        if not wireless_devices:
            wireless_devices.update(self._get_zones_to_disable(wired_devices))
        if not wired_devices:
            raise InvalidArgumentsError(
                'transition_zones_to_wireless_sonosnet_mode requires a wired '
                'device, one device must be excluded from the set of '
                'wireless_zones + first_wire')
        # use lists to make sure the order is correct
        enabled_devices = list(wired_devices) + list(wireless_devices)
        expected_wire_state = ([False] * len(wired_devices) +
            [True] * len(wireless_devices))
        actual_wire_state = [x.wire.is_enabled() for x in enabled_devices]
        if expected_wire_state == actual_wire_state:
            logger.info("All zones in expected wired/sonosnet mode, "
                "short-circuiting wireless sonosnet transition")
            self.wait_for_zones_in_sonosnet_mode(enabled_devices)
            return
        logger.info("Performing wireless sonosnet transition")
        # enabling devices and wire before disabling minimizes topology churn
        self.enable_zones(*enabled_devices)
        self.enable_ethernet_wires_on_zones(*wired_devices)
        self.disable_zones(*other_devices)
        self.disable_ethernet_wires_on_zones(*wireless_devices)
        self.wait_for_zones_in_sonosnet_mode(enabled_devices)
        [Brctl(zc).wait_for_stp_stable() for zc in enabled_devices]
        self.subnet.clear_and_wait_for_devices(enabled_devices,
            timeout_seconds=MAX_TRANSITION_TIME)

    def _toggle_wire(self, wire_enabled_state, dut, expect_station_mode, wait):
        """
        Helper function to enable/disable wire on a sonos device
        and wait for the device to settle on the specified mode

        :param wire_enabled_state: true to enable wire, or
            false to disable wire
        :type wire_enabled_state: :obj:`bool`
        :param dut: the sonos device
        :type dut: :class:`~sonos.client.internal.SonosZoneComponent`
        :param expect_station_mode: flag to expect station mode
            or sonosnet mode
        :type expect_station_mode: :obj:`bool`
        :param wait: flag to skip the wait for the device to change
            network operating modes.
        :type wait: :obj:`bool`
        """
        if wire_enabled_state:
            dut.wire.enable()
        else:
            dut.wire.disable()
        if not wait:
            return
        if expect_station_mode and not(dut.is_bridge() or
                                       dut.is_satellite()):
            dut.wait_for_station_mode()
        else:
            dut.wait_for_sonosnet_mode()
