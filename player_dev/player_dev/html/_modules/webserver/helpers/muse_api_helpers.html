[docs]    def make_muse_session(self, muse_connection, group_id=None, hhid=None, test_fixture=None, accountId=None):
        if test_fixture:
            app_id=test_fixture.__class__.__name__
            app_context=test_fixture._testMethodName
        else:
            app_id=self.__class__.__name__
            app_context=self._testMethodName

        group_id, hhid = self._get_group_id_and_household_id(group_id, hhid, test_fixture)

        muse_connection.create_session(group_id=group_id, hhid=hhid,
                                       app_id=app_id,
                                       app_context=app_context,
                                       accountId=accountId)

        if test_fixture:
            test_fixture.verify_event(SessionStatusEvent, connection=muse_connection,
                          event_verifier=lambda e: (e.session_state == SessionState.CONNECTED))
        else:
            self.verify_event(SessionStatusEvent, connection=muse_connection,
                          event_verifier=lambda e: (e.session_state == SessionState.CONNECTED))

        # added here to subscribe to playbackSession so that eviction test will pass after SWPBL-41194
        muse_connection.subscribe_session(session_id=muse_connection.session_id, hhid=hhid)
        if test_fixture:
            test_fixture.verify_event(Success, connection=muse_connection)
        else:
            self.verify_event(Success, connection=muse_connection)

    """
    Helper methods to get player hardware via REST
    """
    def __init__(self, zp, logger=None):
        """
        Hardware status helper init.
        :param :class:`sonos.client.internal.SonosZoneComponent` zp:
        :param :class: logging
        """
        if logger:
            self.logger = logger
        else:
            self.logger = logging.getLogger('HardwareStatusHelper')

        self.zp = zp
        self.muse_helper = MuseApiHelpers()
        self.rest_ep = HARDWARE_STATUS_EP.format(self.zp.raw_udn)
        self.req_url_stub = MUSE_REST_STUB.format(self.zp.ip)

    def _get_player_hardware_status(self, attribute):
        """
        Get player setting using REST endpoint The method
        expects OAUTH_USER, OAUTH_PW and MITC_ENVIRONMENT environment
        variables else throws HTTPError exception.
        :param str attribute: Hardware attribute of player to get status
        :return: dictionary containing status code and message
        :rtype::obj:`dict`
        """
        token = OAuth.get_oauth_token(scope=OAUTH_HH_ADMIN_SCOPE).token
        rest_ep = self.rest_ep + attribute
        resp = self.muse_helper.hit_rest_endpoint(self.zp,
                                                  rest_endpoint=rest_ep,
                                                  req_url_stub=self.req_url_stub,
                                                  is_secure=True,
                                                  token=token,
                                                  req_method="GET")
        return json.loads(resp.text)

    """
    Helper methods to get/set player settings via REST
    """
    def __init__(self, zp, token=None, logger=None):
        if logger:
            self.logger = logger
        else:
            self.logger = logging.getLogger('PlayerSettingsHelper')

        self.token = token or OAuth.get_oauth_token(scope=OAUTH_HH_ADMIN_SCOPE).token

        self.zp = zp
        self.muse_helper = MuseApiHelpers()
        self.rest_ep = PLAYER_SETTING_EP.format(self.zp.raw_udn)
        self.req_url_stub = MUSE_REST_STUB.format(self.zp.ip)

    def _perform_player_setting(self,
                                player_setting=None,
                                oauth_scope=None,
                                exp_http_return_code=HTTPStatusCode.ok):
        """
        Perform setting on player using REST. The method
        expects OAUTH_USER, OAUTH_PW and MITC_ENVIRONMENT environment
        variables else throws HTTPError exception.
        :param dict player_setting: Player setting
        :param str oauth_scope: OAUTH scope
        :param int exp_http_return_code: expected return code.
        :return: Is status code same as the expected status code?
            Default expected status code = 200 (HTTPStatusCode.ok)
        """
        self.zp.diag.set_diag_level('httpauth', 5)
        self.zp.diag.set_diag_level('certval', 5)
        self.zp.diag.set_diag_level('muse', 8)

        # if oauth_scope is not provided, use as defined in init, else use
        # the provided oauth scope to get the token
        if oauth_scope is not None:
            self.token = OAuth.get_oauth_token(scope=oauth_scope).token
        resp = self.muse_helper.hit_rest_endpoint(self.zp,
                                                  rest_endpoint=self.rest_ep,
                                                  req_url_stub=self.req_url_stub,
                                                  is_secure=True,
                                                  token=self.token,
                                                  req_method="POST",
                                                  req_json=player_setting)
        self.logger.info("Player setting: {} respose: {}"
                         .format(player_setting, resp.text))
        return resp.status_code == exp_http_return_code

    def _get_player_setting(self):
        """
        Get player setting using REST endpoint The method
        expects OAUTH_USER, OAUTH_PW and MITC_ENVIRONMENT environment
        variables else throws HTTPError exception.
        :return: request response containing status code and message content
        """
        self.zp.diag.set_diag_level('httpauth', 5)
        self.zp.diag.set_diag_level('certval', 5)
        self.zp.diag.set_diag_level('muse', 8)

        resp = self.muse_helper.hit_rest_endpoint(self.zp,
                                                  rest_endpoint=self.rest_ep,
                                                  req_url_stub=self.req_url_stub,
                                                  is_secure=True,
                                                  token=self.token,
                                                  req_method="GET")
        return resp

[docs]    def hit_rest_endpoint(self, zp, rest_port=None,
                          rest_endpoint="", is_secure=False, api_key=X_SONOS_API_KEY, token=None, params=None,
                          req_url_stub=None, req_method="GET", req_headers=None, req_data="", req_json=None):
        """
        Helper to build a REST call to the player.
        :param zp:
        :param rest_endpoint:
        :param is_secure:
        :param api_key:
        :param token:
        :param req_url_stub:
        :param req_method:
        :param req_headers:
        :param req_data:
        :param req_json:
        :return:
        """
        if req_headers is None:
            req_headers = {}
        req_headers["X-Sonos-Api-Key"] = api_key
        if rest_port is None:
            rest_port = int(zp.ssl_port if is_secure else zp.port)
        if req_url_stub is None:
            req_url_stub = "{}://{}:{}/api/v1/households/{}".format('https' if is_secure else 'http',
                                                                    zp.ip, rest_port, zp.muse_hhid)
        req_url = "{}/{}".format(req_url_stub, rest_endpoint)
        if token is not None:
            req_headers["Authorization"] = 'Bearer {}'.format(token)

        req = requests.Request(req_method, req_url, params=params,
                               data=req_data, headers=req_headers, json=req_json).prepare()
        self.logger.info("REST req: <{}> <{}> headers: <{}>, body: <{}>".
                          format(req.method, req.url, req.headers, req.body))

        with requests.Session() as s:
            resp = s.send(req, verify=False)
            self.logger.info("REST endpoint: <{}>, response status_code: <{}>, response text: <{}>".
                             format(rest_endpoint, resp.status_code, resp.text.encode('utf-8')))
            return resp



[docs]    def verify_rest_response(self, fixture, resp, expected_resp_code=None, expected_resp_text=None):
        """
        Verify the response code/text is as expected
        :param fixture:
        :param resp:
        :param expected_resp_code:
        :param expected_resp_text:
        :return:
        """
        if expected_resp_code is not None:
            fixture.verifyTrueOrFailCase(resp.status_code == expected_resp_code,
                                         "Response code should be {} and was {}".format(expected_resp_code,
                                                                                        resp.status_code))

        if expected_resp_text is not None:
            fixture.verifyTrueOrFailCase(resp.text == expected_resp_text,
                                         "Response text should be '{}' and was '{}'".format(expected_resp_text,
                                                                                            resp.text))
