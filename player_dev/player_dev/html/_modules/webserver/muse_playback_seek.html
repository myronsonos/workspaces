[docs]    def test_playbackSeek_parameter_fuzzer(self):
        """
        Cycles through each input param to playback.Seek with various values and verify the response event
        :return:
        """
        # create dict of all the args the command takes
        params = {"hhid": self.zp.muse_hhid, "groupId": self.gid, "positionMillis": 15000,
                  "itemId": "0"}
        # Define which ones are optional
        optional_params = ("itemId")

        # load a cloudqueue so the queue isn't empty
        self.muse.load_cloud_queue(session_id=self.muse.session_id,
                                   hhid=self.zp.muse_hhid,
                                   queue_base_url=self.cq_control.CQ_PLAYBACK_URI,
                                   track_metadata=None,
                                   play_on_completion=False)
        self.verify_event(PlaybackStatusEvent, timeout=1)

        for param in params:
            for value in ("True", 99999, None):
                self.clear_message_queue()
                self.logger.debug("Setting {} to {}".format(param, value))
                # make a copy of the params dict so it can be reset after iteration
                new_params = dict(params)
                new_params[param] = value
                self.muse.playback_seek(**new_params)
                # Sending a null value for an optional param should result in a success response
                # Sending an oob positionMillis should get clamped by the player and result in a success response
                if (isinstance(value, int) and param == "positionMillis") or (param in optional_params and value is None):
                    self.verify_event(Success, timeout=1, iterations=1)
                elif param == "groupId" and value is not None:
                    self.verify_event(GroupCoordinatorChangedEvent, timeout=1, iterations=1)
                # All other cases should result in one of these 3 errors
                else:
                    self.verify_event(Error, timeout=1, iterations=1,
                                      event_verifier=lambda e: (e.error_code in (ErrorEnum.GLOBAL_MISSING_PARAMETERS.value,
                                                                                 ErrorEnum.GLOBAL_INVALID_OBJECT_ID.value,
                                                                                 ErrorEnum.GLOBAL_INVALID_PARAMETER.value)))



[docs]    def test_playbackSeekRelative_parameter_fuzzer(self):
        """
        Cycles through each input param to playback.SeekRelative with various values and verify the response event
        """
        # Set the player to use the shared queue and clear it. I want the player to have no content.
        self.zp.AVTransport.use_queue()
        self.zp.AVTransport.clear_queue()

        # create dict of all the args the command takes
        params = {"hhid": self.zp.muse_hhid, "groupId": self.gid, "deltaMillis": 15000,
                  "itemId": "0"}
        # Define which ones are optional
        optional_params = ("itemId")

        for param in params:
            for value in ("True", 99999, None):
                self.clear_message_queue()
                self.logger.debug("Setting {} to {}".format(param, value))
                # make a copy of the params dict so it can be reset after iteration
                new_params = dict(params)
                new_params[param] = value
                self.muse.playback_seekRelative(**new_params)
                # Sending a null value for an optional param should result in a success response but since the
                # queue is empty, a NO_CONTENT error should result
                # Same thing for oob deltaMillis value
                if (param in optional_params and value is None):
                    self.verify_event(Error, timeout=1, iterations=1,
                                      event_verifier=lambda e: (e.error_code == PlaybackError.NO_CONTENT))
                elif param == "groupId" and value is not None:
                    self.verify_event(GroupCoordinatorChangedEvent, timeout=1, iterations=1)
                # All other cases should result in one of these 3 errors
                else:
                    self.verify_event(Error, timeout=1, iterations=1,
                                      event_verifier=lambda e: (e.error_code in (ErrorEnum.GLOBAL_MISSING_PARAMETERS.value,
                                                                                 ErrorEnum.GLOBAL_INVALID_OBJECT_ID.value,
                                                                                 ErrorEnum.GLOBAL_INVALID_PARAMETER.value)))



[docs]    def test_seek_track_position(self):
        """
        Use seek and seekRelative to scrub around in a track and verify that the playback status reports the
        expected positionMillis
        :return:
        """
        negative_pos = -5000
        positive_pos = 10000
        self.muse.load_cloud_queue(session_id=self.muse.session_id,
                                   hhid=self.zp.muse_hhid,
                                   queue_base_url=self.cq_control.CQ_PLAYBACK_URI,
                                   track_metadata=None,
                                   play_on_completion=False)
        self.verify_event(PlaybackStatusEvent, timeout=1)

        # Seek a positive quantity with the absolute seek
        self.muse.playback_seek(groupId=self.gid, hhid=self.zp.muse_hhid, positionMillis=positive_pos,
                                itemId="0")
        self.verify_event(PlaybackStatusEvent, timeout=1,
                          event_verifier=lambda e: (e.position_millis == positive_pos))

        # Seek a negative quantity with the absolute seek. The player should clamp this value to 0.
        self.muse.playback_seek(groupId=self.gid, hhid=self.zp.muse_hhid, positionMillis=negative_pos,
                                itemId="0")
        self.verify_event(PlaybackStatusEvent, timeout=1,
                          event_verifier=lambda e: (e.position_millis == 0))

        # Seek a positive quantity twice with the relative seek. The scrub action should be cumulative.
        for x in (1,2):
            self.muse.playback_seekRelative(groupId=self.gid, hhid=self.zp.muse_hhid, deltaMillis=positive_pos,
                                            itemId="0")
            self.verify_event(PlaybackStatusEvent, timeout=1,
                              event_verifier=lambda e: (e.position_millis == positive_pos*x))

        # Seek a negative quantity twice with the relative seek. The scrub action should be cumulative.
        for y in (1,2):
            self.muse.playback_seekRelative(groupId=self.gid, hhid=self.zp.muse_hhid, deltaMillis=negative_pos,
                                            itemId="0")
            self.verify_event(PlaybackStatusEvent, timeout=1,
                              event_verifier=lambda e: (e.position_millis == (positive_pos*x+negative_pos*y)))

        self.muse.playback_seekRelative(groupId=self.gid, hhid=self.zp.muse_hhid, deltaMillis=-99999,
                                        itemId="0")
        self.verify_event(PlaybackStatusEvent, timeout=1,
                          event_verifier=lambda e: (e.position_millis == 0))



[docs]    def test_seek_track_id(self):
        """
        Verify that the players returns an error if the user tries to seek to an itemId that's not
        the currently heard track
        :return:
        """
        negative_pos = -5000
        positive_pos = 15000

        # run this test while the player is playing so its a little different from the previous test
        self.muse.load_cloud_queue(session_id=self.muse.session_id,
                                   hhid=self.zp.muse_hhid,
                                   queue_base_url=self.cq_control.CQ_PLAYBACK_URI,
                                   track_metadata=None,
                                   play_on_completion=True)
        self.verify_event(PlaybackStatusEvent, timeout=1,
                          event_verifier=lambda e: (e.playback_state == PlaybackState.PLAYING))

        # verify again that the absolute seek works while playing
        self.muse.playback_seek(groupId=self.gid, hhid=self.zp.muse_hhid, positionMillis=positive_pos,
                                itemId="0")
        self.verify_event(PlaybackStatusEvent, timeout=1,
                          event_verifier=lambda e: (e.playback_state == PlaybackState.PLAYING and
                                                    e.item_id == "0" and
                                                    e.position_millis >= positive_pos))

        # if the itemId doesn't match the current playing track, the player should send an error response
        # playback should not stop
        self.muse.playback_seek(groupId=self.gid, hhid=self.zp.muse_hhid, positionMillis=positive_pos,
                                itemId="1")
        self.verify_event(Error, timeout=1,
                          event_verifier=lambda e: (e.error_code == Error.INVALID_OBJECT_ID))
        self.muse.playback.get_playback_status(group_id=self.gid, hhid=self.zp.muse_hhid)
        self.verify_event(PlaybackStatusEvent, timeout=1,
                          event_verifier=lambda e: (e.playback_state == PlaybackState.PLAYING and
                                                    e.item_id == "0" and
                                                    e.position_millis >= positive_pos))

        # verify again that the relative seek works while playing
        for x in (1,2):
            self.muse.playback_seekRelative(groupId=self.gid, hhid=self.zp.muse_hhid, deltaMillis=negative_pos,
                                            itemId="0")
            self.verify_event(PlaybackStatusEvent, timeout=1,
                              event_verifier=lambda e: (e.playback_state == PlaybackState.PLAYING and
                                                        e.item_id == "0" and
                                                        e.position_millis >= positive_pos+negative_pos*x))

        # verify that the error response is sent, playback has not stopped and the position has not reset
        self.muse.playback_seekRelative(groupId=self.gid, hhid=self.zp.muse_hhid, deltaMillis=positive_pos,
                                        itemId="1")
        self.verify_event(Error, timeout=1,
                          event_verifier=lambda e: (e.error_code == Error.INVALID_OBJECT_ID))
        self.muse.playback.get_playback_status(group_id=self.gid, hhid=self.zp.muse_hhid)
        self.verify_event(PlaybackStatusEvent, timeout=1,
                          event_verifier=lambda e: (e.playback_state == PlaybackState.PLAYING and
                                                    e.item_id == "0" and
                                                    e.position_millis >= positive_pos+negative_pos*x))
