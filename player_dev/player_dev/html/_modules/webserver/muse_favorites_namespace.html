    def _get_favoriteId(self, items=1, name=smapi_title):
        """
        Get the favoriteId of the desired track
        :return: 
        """
        self.muse.getFavorites(hhid=self.zp.muse_hhid)
        self.verify_event(FavoritesListEvent, timeout=1,
                          event_verifier=lambda e: (len(e.items) == items))

        for item in self.muse.test_events[-1].items:
            if name in item["name"]:
                return item["id"]
        else:
            self.fail("Did not find the requested track")

    def _start_webserver_playback(self, autoplay=True):
        """
        Start playback from the webserver
        :param autoplay: 
        :return: 
        """
        self.zp.AVTransport.use_queue()
        self.zp.AVTransport.clear_queue_and_wait()
        track_list = []
        for track in self.track_source.tracks:
            track_list.append(track)
        track_list_md = queue_helpers.create_didl_lite_md(track_list)
        enqueued_uris_and_metadata = queue_helpers.create_uri_and_metadata_xml(track_list, track_list_md)
        self.zp.Queue.AddMultipleURIs(queue_id=0,
                                         update_id=0,
                                         container_uri="",
                                         container_metadata="",
                                         desired_first_track_number_enqueued=1,
                                         enqueue_as_next=0,
                                         number_of_uris=len(track_list),
                                         enqueued_uris_and_metadata=enqueued_uris_and_metadata)
        if autoplay:
            self.zp.AVTransport.Play_and_wait(timeout_seconds=5)

[docs]    def test_subscribe_favorites(self):
        """
        Verify that the favorites namespace will event out a versionChanged event
        if you subscribe to it and add an item to favorites.
        The version should also change with the addition.
        :return: 
        """
        self.muse.getFavorites(hhid=self.zp.muse_hhid)
        self.verify_event(FavoritesListEvent, timeout=1,
                          event_verifier=lambda e: (e.version is not None and
                                                    len(e.items) == 1))
        # Store the old version value
        old_ver = self.muse.test_events[-1].version
        self.muse.subscribe_favorites(hhid=self.zp.muse_hhid)
        self.zp.ContentDirectory.create_favorite(self.tuneInStation)
        # After adding to favorites, the namespace should send an event
        # with a different version value
        self.verify_event(VersionChangedEvent, timeout=1,
                          event_verifier=lambda e: (e.version != old_ver))
        new_ver = self.muse.test_events[-1].version
        # The favoritesList event version value should match the namespace
        # versionChanged event
        self.muse.getFavorites(hhid=self.zp.muse_hhid)
        self.verify_event(FavoritesListEvent, timeout=1,
                          event_verifier=lambda e: (e.version == new_ver and
                                                    len(e.items) == 2))
        # Verify that the tunein station was the one that was added
        items = self.muse.test_events[-1].items
        for item in items:
            if "WGBH" in item["name"]:
                break
        else:
            self.fail("Did not find WGBH after it was added to favorites")

        """
        Verify that the player no longer sends out versionChanged events
        after the client unsubscribes to the namespace
        :return: 
        """
        self.zp.ContentDirectory.remove_all_favorites()

        old_ver = self.muse.test_events[-1].version
        self.muse.subscribe_favorites(hhid=self.zp.muse_hhid)
        self.zp.ContentDirectory.create_favorite(self.tuneInStation)
        self.verify_event(VersionChangedEvent, timeout=1,
                          event_verifier=lambda e: (e.version != old_ver))
        # verify that once you unsubscribe, you shouldn't get versionChanged events
        # after changing the favorites list
        self.muse.unsubscribe_favorites(hhid=self.zp.muse_hhid)
        self.zp.ContentDirectory.create_favorite(self.smapi_track)
        self.verify_event(VersionChangedEvent, timeout=1, expect_event=False)



[docs]    def test_loadFavorite(self):
        """
        Verifies that loadFavorite loads the correct track and doesn't start playback
        if playOnCompletion is not specified.
        :return: 
        """
        self.zp.AVTransport.clear_play_state()
        self.zp.AVTransport.clear_queue()
        favoriteId = self._get_favoriteId()
        self.muse.loadFavorite(hhid=self.zp.muse_hhid, groupId=self.gid,
                               favoriteId=favoriteId)
        self.muse.subscribe_playback(self.gid, hhid=self.zp.muse_hhid)
        # load without specifying autoplay should start the player at IDLE
        self.verify_event(PlaybackStatusEvent, timeout=1,
                          event_verifier=lambda e: (e.playback_state == PlaybackState.IDLE))
        # Verify that we can play with a play command
        self.muse.play(group_id=self.gid, hhid=self.zp.muse_hhid)
        self.verify_event(PlaybackStatusEvent, timeout=1,
                          event_verifier=lambda e: (e.playback_state == PlaybackState.PLAYING))
        # Use metadata to verify the correct track is played
        self.muse.subscribe_metadata(group_id=self.gid, hhid=self.zp.muse_hhid)
        self.verify_event(MetadataStatusEvent, timeout=1,
                          event_verifier=lambda e: (e.current_item['track']['name'] == 'Old Enough'))

[docs]    def test_loadFavorite_fails_on_member_of_group(self):
        """
        Verify that playback fails if you try to load on a member of a group and an GC moved is sent.
        :return: 
        """
        # Group self.zp to another player
        self.verifyTrueOrSkip(len(self.my_devices) > 0,
                              "need at least 2 ZPs to run this test")
        self.zp2 = ifilter(lambda device: (device.is_playback_device()
                                          and not device.is_bonded()
                                          and device != self.zp),
                           self.my_devices).next()
        if self.zp2.AVTransport.is_playing():
            self.zp2.AVTransport.clear_play_state()
            self.zp2.AVTransport.clear_queue_and_wait()
        self.zp.AVTransport.join_group(self.zp2, timeout_seconds=5)

        favoriteId = self._get_favoriteId()
        # Attempt to load but should get a group moved event
        self.muse.loadFavorite(hhid=self.zp.muse_hhid, groupId=self.gid,
                               favoriteId=favoriteId, playOnCompletion=True)
        self.verify_event(GroupCoordinatorChangedEvent, timeout=1,
                          event_verifier=lambda e: (e.group_status == "GROUP_STATUS_MOVED"))
        # GC should not be playing after load failure
        wait_until_true(lambda: not self.zp2.AVTransport.is_playing(),
                        timeout_seconds=10, iteration_delay=1)
        # clean up the groups
        self.zp.AVTransport.BecomeCoordinatorOfStandaloneGroup()
