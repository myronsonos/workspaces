    def _test_apikey_in_CQ_header(self, target_zp):
        """
        Creates CloudQueue server and issues 'creates session' with matched accountId.
        Verifies CloudQueue server will only work when account is specified that contains APIKey in Manifest
        """

        target_zp.cli.clear_anacapa_trace()
        self.cq_control.clear_log()

        accounts = target_zp.SystemProperties.get_service_accounts()
        self.verifyTrueOrStop(len(accounts) == 0,
                              "Expect no accounts in the household (1). There are {} accounts in the household.".format(len(accounts)))

        manifest_rs_version = target_zp.settings.get_music_service_manifests_version()

        museConnection = self.make_connection(target_zp)

        # Attempt to match new mockSMAPI account - this should always create an account because
        # we remove all accounts at the end of this function.
        resp_msg = museConnection.ms_accounts.match(hhid=target_zp.muse_hhid, user_hash="", nickname="service",
                                                         service_id=str(self.sid), link_code="MAGIC_LINK_CODE",
                                                         linkDeviceId="MAGIC_LINK_DEVICE_ID")

        self.verify_event(musicServiceAccountsEvent, connection=museConnection, timeout=1,
                          event_verifier=lambda e: (e.nickname == "NICKNAME" and
                                                    e.service.id == str(self.sid) and
                                                    e.userIdHashCode is not None and
                                                    e.isGuest is True))

        self.accountId = resp_msg.id

        # wait for guest account update to replicate throughout the system
        self.logger.info("Waiting for replication....")
        target_zp.settings.sync_account_settings()
        # target_zp.settings.sync_music_service_manifests(expected_ver=manifest_rs_version+1)
        target_zp.settings.sync_music_service_manifests()
        self.logger.info("... replication done")

        # Sanity check that the manifest end-point is valid
        manifest_get_uri = "http://{}:{}/manifest".format(smapi_ctl.host_ip, self.smapi_port)
        wait_until_true(lambda: (urllib2.urlopen(manifest_get_uri).read() is not None and
                                 urllib2.urlopen(manifest_get_uri).read() != ""),
                        timeout_seconds=30, iteration_delay=1,
                        reason="Waiting until manifest hosted on Node_SMAPI returns something")
        contents = urllib2.urlopen(manifest_get_uri).read()
        self.logger.info("manifest hosted on Node_SMAPI: {}".format(contents))
        expected_manifest = json.loads(contents)
        expected_apikeyzp = expected_manifest["apiKey"]["zp"]
        self.verifyGreaterOrFailCase(len(expected_apikeyzp), 128, "apiKey.zp exists in manifest source")

        # Verify manifest file has been replicated and is correct.
        wait_until_true(lambda: (target_zp.settings.get_music_service_manifests() is not None and
                                 target_zp.settings.get_music_service_manifests() != ""),
                        timeout_seconds=30, iteration_delay=1,
                        reason="Waiting until music service manifest returns something")
        manifest_rs = target_zp.settings.get_music_service_manifests()
        manifest = json.loads(manifest_rs)
        entries = manifest["manifests"]
        manifest_rs_version = manifest["version"]
        self.verifyEqualOrFailCase(len(entries), 1, "Expect one manifest in the household for version {}".format(manifest_rs_version))
        for entry in entries:
            sid = entry["sid"]
            actual_manifest = entry["manifest"]
            self.verifyEqualOrFailCase("X-MockSMAPI-ApiKey:David did not start the fire", actual_manifest["apiKey"]["zp"], "Found expected apiKey")

        book_endpoint = "{}book/{}/".format(self.cq_control.CQ_PLAYBACK_URI, AUDIOBOOK_OBJECT_ID)

        self.logger.info("Creating Session for account {}".format(self.accountId))
        groupId = target_zp.GroupManagement.get_local_group_uuid()
        museConnection.create_session(group_id=groupId, hhid=target_zp.muse_hhid,
                                  app_id="auto_new_session", app_context="one_time",
                                  accountId=self.accountId)

        self.logger.info("Session created: {}".format(museConnection.session_id))

        museConnection.load_cloud_queue(session_id=museConnection.session_id,
                                    hhid=target_zp.muse_hhid,
                                    queue_base_url=book_endpoint,
                                    play_on_completion=False,
                                    httpAuthorization='')

        self.logger.info("subscribe playback gid={}".format(groupId))
        museConnection.playback.subscribe(group_id=groupId, hhid=target_zp.muse_hhid)
        self.verify_event(PlaybackStatusEvent, connection=museConnection, timeout=2)

        museConnection.metadata.subscribe(group_id=groupId, hhid=target_zp.muse_hhid)
        self.verify_event(MetadataStatusEvent, connection=museConnection, timeout=2)

        # Play, wait and verify item 0 is playing
        museConnection.play(group_id=groupId, hhid=target_zp.muse_hhid, session_id=museConnection.session_id)
        wait_until_true(lambda: target_zp.AVTransport.is_playing(), timeout_seconds=10, iteration_delay=0.5)
        self.verify_event(PlaybackStatusEvent, connection=museConnection, timeout=2,
                          event_verifier=lambda e: e.playback_state == PlaybackState.PLAYING and e.item_id == "0")

        museConnection.pause(group_id=groupId, hhid=target_zp.muse_hhid, session_id=museConnection.session_id)
        wait_until_true(lambda: target_zp.AVTransport.is_paused(), timeout_seconds=10, iteration_delay=0.5)
        self.verify_event(PlaybackStatusEvent, connection=museConnection, timeout=2,
                          event_verifier=lambda e: e.playback_state == PlaybackState.PAUSED)

        museConnection.skip_to_next(group_id=groupId, hhid=target_zp.muse_hhid,
                                session_id=museConnection.session_id)
        self.verify_event(PlaybackStatusEvent, connection=museConnection, timeout=2,
                          event_verifier=lambda e: e.playback_state == PlaybackState.PAUSED and e.item_id == "1")

        museConnection.play(group_id=groupId, hhid=target_zp.muse_hhid, session_id=museConnection.session_id)
        wait_until_true(lambda: target_zp.AVTransport.is_playing(), timeout_seconds=10, iteration_delay=0.5)
        self.verify_event(PlaybackStatusEvent, connection=museConnection, timeout=2,
                          event_verifier=lambda e: e.playback_state == PlaybackState.PLAYING and e.item_id == "1")


        # Playing is good indication that /context and /itemWindow worked correctly. However, we still need to check
        # for /timePlayed and /version
        cq_logs = self.cq_control.wait_for_filtered_log(type='api',
                                                        fn=lambda l: (l['path'] == '/book/bk1/version' and
                                                                      l['headers'][
                                                                          'x-mocksmapi-apikey'] == 'David did not start the fire'),
                                                        timeout=10,
                                                        expected=1)
        self.verifyGreaterEqualOrFailCase(len(cq_logs), 1, "was apiKey sent to /version?")

        cq_logs = self.cq_control.wait_for_filtered_log(type='api',
                                                        fn=lambda l: (l['path'] == '/book/bk1/timePlayed' and
                                                                      l['headers'][
                                                                          'x-mocksmapi-apikey'] == 'David did not start the fire'),
                                                        timeout=10,
                                                        expected=1)
        self.verifyGreaterEqualOrFailCase(len(cq_logs), 1, "was apiKey sent to /timePlayed")

        museConnection.playback.unsubscribe(group_id=groupId, hhid=target_zp.muse_hhid)
        museConnection.metadata.unsubscribe(group_id=groupId, hhid=target_zp.muse_hhid)

        #
        # Now create a session without a matched account and expect it to fail because the CQ server expects the api key
        #
        book_endpoint = "{}book/{}/".format(self.cq_control.CQ_PLAYBACK_URI, NEG_TEST_AUDIOBOOK_OBJECT_ID)

        logger.info("Running test without match account")
        museConnection.create_session(group_id=groupId, hhid=target_zp.muse_hhid,
                                  app_id="auto_new_session", app_context="one_time")
        self.logger.info("Session created: {}".format(museConnection.session_id))

        self.logger.info("subscribe playback gid={}".format(groupId))
        museConnection.playback.subscribe(group_id=groupId, hhid=target_zp.muse_hhid)
        self.verify_event(PlaybackStatusEvent, connection=museConnection, timeout=2)

        museConnection.load_cloud_queue(session_id=museConnection.session_id,
                                    hhid=target_zp.muse_hhid,
                                    queue_base_url=book_endpoint,
                                    play_on_completion=False)


        museConnection.play(group_id=groupId, hhid=target_zp.muse_hhid, session_id=museConnection.session_id)
        self.verify_event(PlaybackErrorEvent, connection=museConnection, timeout=2,
                          event_verifier=lambda e: (e.error_code == "ERROR_PLAYBACK_NO_CONTENT" and
                                                    e.reason == "ERROR_NO_CONTENT"))

        # We are done with this zp, clear its play state, remove accounts and wait so the next
        # model ZP gets to do the same test.
        target_zp.AVTransport.clear_play_state()
        self.verifyTrueOrStop(target_zp.SystemProperties.remove_all_service_accounts(), "Removed all service accounts")
        self.logger.info("Waiting for replication(2)....")
        target_zp.settings.sync_account_settings()
        target_zp.settings.sync_music_service_manifests()
        self.logger.info("... replication(2) done")
        self.logger.info("Closing muse connection")
        self.disconnect(museConnection)
