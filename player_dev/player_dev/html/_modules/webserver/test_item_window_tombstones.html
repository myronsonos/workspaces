[docs]    def test_window_isEnd_false_delete_remaining_tracks(self):
        """
        SWPBL-70016 - Verify that the player continues playing after trying to play a tombstone
        in a window that has isBeginning=false and isEnd=false.
        """
        # Add more tracks to queue (30)
        for _ in xrange(10):
            for x in range(3):
                muse_session_mixin.cq_control.add_track_to_end(track_id=x)
        # Refresh and load the CQ
        self.muse.refresh_cloud_queue(session_id=self.muse.session_id, hhid=self.zp.muse_hhid)
        muse_session_mixin.start_cloud_queue_client(self.muse, group_id=self.gid, hhid=self.zp.muse_hhid,
                                                    auto_playback=False, test_fixture=self, my_item_id='15')
        # Skip to near the end of the current track
        self.muse.skip_to_item(session_id=self.muse.session_id,
                               hhid=self.zp.muse_hhid,
                               item_id=15,
                               position_millis=27000,
                               play_on_completion=False)
        # Delete tracks 16-25
        for t in xrange(16, 26):
            muse_session_mixin.cq_control.delete_track(track_id=t)

        # Play and verify player plays track 15, followed by track 26 (in the next window)
        self.muse.play(group_id=self.gid, hhid=self.zp.muse_hhid)
        self.verify_event(PlaybackStatusEvent,
                          event_verifier=lambda e: e.playback_state == PlaybackState.PLAYING
                          and e.item_id == str(t + 1))

[docs]    def test_load_and_play_on_empty_window404(self):
        """
        Verify ERROR_NO_PLAYBACK_CONTENT event returned from a muse.play command sent after the server has
        returned a 404 on get itemWindow. Subsequently verify that refreshCloudQueue will update the queue
        after new tracks have been added. Regression test for SWPBL-49012
        """
        # Force a 404 on the initial /itemWindow fetch
        muse_session_mixin.cq_control.arm_error_script(call="itemWindow", error=404)

        muse_session_mixin.start_cloud_queue_client(self.muse, group_id=self.gid, hhid=self.zp.muse_hhid,
                                                    auto_playback=False, test_fixture=self, track_metadata=None)
        self.muse.play(group_id=self.gid, hhid=self.zp.muse_hhid)
        # expect a playback error, NO_CONTENT
        self.verify_event(PlaybackErrorEvent, event_verifier=lambda e: e.error_code == PlaybackError.NO_CONTENT)

        # wait until we get two /itemWindow fetches to ensure the IDLE message comes after the playback error
        wait_until_true(lambda: muse_session_mixin.cq_control.wait_for_filtered_log(fn=lambda logEntry: logEntry['path'] == '/itemWindow',
                                                                                    timeout=2,
                                                                                    expected=2),
                        timeout_seconds=10)

        # change the queue and skipToItem to force a new itemWindow fetch and verify a playback status event
        muse_session_mixin.cq_control.add_track_to_end("0")
        self.muse.skip_to_item(session_id=self.muse.session_id,
                               hhid=self.zp.muse_hhid,
                               item_id="0",
                               play_on_completion=False)
        self.verify_event(PlaybackStatusEvent, timeout=1,
                          event_verifier=lambda e: (e.playback_state == PlaybackState.IDLE and
                                                    e.item_id == '0'))

        """
        Verify ERROR_NO_PLAYBACK_CONTENT event returned from a muse.play command sent after the server has
        returned a 200 empty window on get itemWindow. Subsequently verify that skipToItem will update the queue
        after new tracks have been added. Regression test for SWPBL-49012:
        """

        for x in range(0,9):

        muse_session_mixin.start_cloud_queue_client(self.muse, group_id=self.gid, hhid=self.zp.muse_hhid,

        self.verify_event(PlaybackErrorEvent, event_verifier=lambda e: e.error_code == PlaybackError.NO_CONTENT)
        # add tracks back into the cloud queue
        muse_session_mixin.cq_control.reset_server()
        self.muse.skip_to_item(session_id=self.muse.session_id, hhid=self.zp.muse_hhid, item_id=0)
        self.verify_event(PlaybackStatusEvent, event_verifier=lambda e: e.item_id == '0')
