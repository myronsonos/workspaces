[docs]    def test_previousPositionMillis_do_not_change_while_streaming(self):
        """
        SWPBL-78857
        Verify that the previousPositionMillis and previousItemId are accurate when you play/pause
        on a radio stream and as the client switches between a stream and cloudqueue playback.
        :return:
        """
        # start cloudqueue playback
        self.muse.load_cloud_queue(session_id=self.muse.session_id,
                                   hhid=self.zp.muse_hhid,
                                   queue_base_url=self.cq_control.CQ_PLAYBACK_URI,
                                   content_type=None,
                                   httpAuthorization=None,
                                   play_on_completion=True)
        self.verify_event(PlaybackStatusEvent, timeout=1,
                          event_verifier=lambda e: (e.playback_state == PlaybackState.PLAYING))
        wait_until_true(lambda: self.zp.AVTransport.GetPositionInfo()["RelTime"] >= "0:00:02",
                        iteration_delay=1, timeout_seconds=4)

        # switch to a stream, the previous parameters should be where the cloudqueue playback
        # was interrupted
        self.muse.load_stream_url(session_id=self.muse.session_id,
                                  hhid=self.zp.muse_hhid,
                                  play_on_completion=True,
                                  stream_url=BEATS1,
                                  station_metadata=BEATS1_MD,
                                  item_id="BEATS1")
        wait_until_true(lambda: self.zp.AVTransport.GetPositionInfo()["RelTime"] >= "0:00:03",
                        iteration_delay=1, timeout_seconds=6)
        self.verify_event(PlaybackStatusEvent, timeout=1,
                          event_verifier=lambda e: (e.playback_state == PlaybackState.PLAYING and
                                                    e.item_id == "BEATS1" and
                                                    e.previousItemId == "0" and
                                                    abs(e.previousPositionMillis - 2000) < 1500))

        # pause the stream and verify the previous parameters are when the stream was paused
        self.muse.pause(group_id=self.gid, hhid=self.zp.muse_hhid)
        self.verify_event(PlaybackStatusEvent, timeout=1,
                          event_verifier=lambda e: (e.playback_state == PlaybackState.IDLE and
                                                    e.item_id == "BEATS1" and
                                                    e.previousItemId == "BEATS1" and
                                                    abs(e.previousPositionMillis - 4000) < 1500))

        # play the stream and verify the previous parameters are still from when the stream
        # was paused
        self.muse.play(group_id=self.gid, hhid=self.zp.muse_hhid)
        wait_until_true(lambda: self.zp.AVTransport.GetPositionInfo()["RelTime"] >= "0:00:04",
                        iteration_delay=1, timeout_seconds=8)
        self.verify_event(PlaybackStatusEvent, timeout=1,
                          event_verifier=lambda e: (e.playback_state == PlaybackState.PLAYING and
                                                    e.item_id == "BEATS1" and
                                                    e.previousItemId == "BEATS1" and
                                                    abs(e.previousPositionMillis - 4000) < 1500))

        # load the cloudqueue again and verify the previous parameters are from when the stream
        # was interrupted
        self.muse.load_cloud_queue(session_id=self.muse.session_id,
                                   hhid=self.zp.muse_hhid,
                                   queue_base_url=self.cq_control.CQ_PLAYBACK_URI,
                                   content_type=None,
                                   httpAuthorization=None,
                                   play_on_completion=True,
                                   item_id="2")
        self.verify_event(PlaybackStatusEvent, timeout=1,
                          event_verifier=lambda e: (e.playback_state == PlaybackState.PLAYING and
                                                    e.item_id == "2" and
                                                    e.previousItemId == "BEATS1" and
                                                    abs(e.previousPositionMillis - 4000) < 1500))

        # Added for SWPBL-68110
        # Verify that we get a /timePlayed report for the cloudqueue track played in between
        log = self.cq_control.wait_for_filtered_log(type='api',
                                                                  fn=lambda l: (l['path'] == '/timePlayed'),
                                                                  timeout=15)
        self.verifyEqualOrFailCase(1, len(log),
                                   "Should have gotten 1 /timePlayed report for the first cloudqueue track played")

        """
        SWPBL-70169
        Verify that the AVT is restored if you leave the session after starting
        playback with loadStreamUrl
        :return:
        """
        # Preload an AV Transport URI to kick the logic that saves the previous AV Transport URI
        # This reliably makes the player store the shared queue AV Transport URI on the next line. I found
        # during testing that use_queue() was failing unexpectedly.
        my_metadata = """
        <DIDL-Lite xmlns:dc="http://purl.org/dc/elements/1.1/"
           xmlns:upnp="urn:schemas-upnp-org:metadata-1-0/upnp/"
           xmlns:r="urn:schemas-rinconnetworks-com:metadata-1-0/"
           xmlns="urn:schemas-upnp-org:metadata-1-0/DIDL-Lite/">
         <item id="-1" parentID="-1" restricted="true">
          <dc:title>custom radio</dc:title>
          <upnp:class>object.item.audioItem.audioBroadcast</upnp:class>
          <desc id="cdudn" nameSpace="urn:schemas-rinconnetworks-com:metadata-1-0/">SA_RINCON1543_X_#Svc1543-0-Token</desc>
         </item>
        </DIDL-Lite>"""
        self.verifyTrueOrStop(self.zp.AVTransport.SetAVTransportURI_and_wait(uri='x-rincon-mp3radio://' + CUSTOM_MP3_RADIO_STREAM,
                                                                             metadata=my_metadata,
                                                                             timeout_seconds=10),
                              "Attempt to activate AVT queue")

        self.verifyTrueOrStop(self.zp.AVTransport.use_queue(queue_id=0), "Attempt to activate shared queue")
        avt = self.zp.AVTransport.get_avtransport_uri()

        # Load a stream
        self.muse_session_helper.make_muse_session(self.muse, hhid=self.zp.muse_hhid, test_fixture=self)
        self.muse.subscribe_playback(self.gid, hhid=self.zp.muse_hhid)
        self.muse.load_stream_url(session_id=self.muse.session_id,
                                  hhid=self.zp.muse_hhid,
                                  play_on_completion=True,
                                  stream_url=BEATS1,
                                  station_metadata=BEATS1_MD)
        self.verify_event(PlaybackStatusEvent, timeout=1,
                          event_verifier=lambda e: (e.playback_state == PlaybackState.PLAYING))

        # Leave the session and verify that the avt was reverted
        self.muse.leave_session(session_id=self.muse.session_id,
                                hhid=self.zp.muse_hhid)
        self.verifyTrueOrFailCase(self.zp.AVTransport.wait_for_avtransport_uri(avt),
                                   "Player should revert AV Transport URI")



[docs]    def test_loadStreamUrl_is_not_restored_after_delegation(self):
        """
        SWPBL-87729
        Verify that the streamUrl is not restored after delegating and a failed loadCloudQueue
        command.
        :return:
        """
        # Select a second ZP to use
        try:
            self.zp2 = ifilter(lambda d: (d.is_playback_device() and d != self.zp), self.my_devices).next()
        except ValueError:
            self.zp2 = None

        if not self.zp2:
            self.skip("No second playable ZP found")

        # Create a group with zp (gc) and zp2 using Muse createGroup()
        pids = [self.zp.raw_udn, self.zp2.raw_udn];
        self.muse.createGroup(hhid=self.zp.muse_hhid, playerIds=pids)
        self.verify_event(GroupInfoEvent, timeout=1,
                          event_verifier=lambda e: (e.group['coordinatorId'] in self.zp.raw_udn and
                                                    len(e.group['playerIds']) == 2))

        # Create a playback session
        self.muse.create_session(group_id=self.gid,
                                 hhid=self.zp.muse_hhid,
                                 app_id=self.__class__.__name__,
                                 app_context=self._testMethodName,
                                 custom_data=None)
        self.verify_event(SessionStatusEvent, timeout=1,
                          event_verifier=lambda e: ((e.session_created == True) and
                                                    (e.session_state == SessionState.CONNECTED)))

        # Subscribe to playback, call loadStreamUrl on zp, and verify playback starts
        self.muse.subscribe_playback(self.gid, hhid=self.zp.muse_hhid)
        self.muse.load_stream_url(session_id=self.muse.session_id,
                                  hhid=self.zp.muse_hhid,
                                  play_on_completion=True,
                                  stream_url=BEATS1,
                                  station_metadata=BEATS1_MD)
        self.verify_event(PlaybackStatusEvent, timeout=1,
                          event_verifier=lambda e: (e.playback_state == PlaybackState.PLAYING))

        # Save zp's AVTransport URI
        self.original_avt = self.zp.AVTransport.get_avtransport_uri()

        # Force zp to delegate to zp2
        self.zp2.AVTransport.join_group(self.zp)
        self.zp.AVTransport.BecomeCoordinatorOfStandaloneGroup()

        # Create a connection to zp2 and create session, group is zp's original group ID
        self.second_muse_client = self.make_connection(self.zp2)
        self.gid2 = self.zp2.GroupManagement.get_local_group_uuid()
        self.second_muse_client.create_session(group_id=self.gid2,
                                               hhid=self.zp2.muse_hhid,
                                               app_id=self.__class__.__name__,
                                               app_context=self._testMethodName,
                                               custom_data=None)
        self.verify_event(SessionStatusEvent, timeout=1, connection=self.second_muse_client,
                          event_verifier=lambda e: ((e.session_created == True) and
                                                    (e.session_state == SessionState.CONNECTED)))

        # Call loadCloudQueue() on zp2 with a cloudqueue error
        self.cq_control.arm_error_script(call="context", error=404)
        self.second_muse_client.load_cloud_queue(session_id=self.second_muse_client.session_id,
                                                 hhid=self.zp2.muse_hhid,
                                                 queue_base_url=self.cq_control.CQ_PLAYBACK_URI,
                                                 httpAuthorization=None,
                                                 item_id=0,
                                                 play_on_completion=True,
                                                 track_metadata=None)
        self.verify_event(Success, timeout=1, connection=self.second_muse_client)

        # Check that playback stops on the second player
        wait_until_true(lambda: not self.zp2.AVTransport.is_playing(),
                        timeout_seconds=5, iteration_delay=1)

        # Verify that zp2's AVTransport URI is different from zp's original AVTransport URI
        wait_until_true(lambda: self.zp2.AVTransport.get_avtransport_uri() != self.original_avt,
                        reason="Waiting for AVTransportURI mismatch on second player",
                        timeout_seconds=5, iteration_delay=1)
