[docs]    def test_playerSettings_get_post_volumeMode_setting(self):
        """
        SWPBL-95185
        volumeMode was the first playerSetting implemented so we're using it to verify the REST playerSetting
        endpoint is working
        :return:
        """
        # volumeMode only works on connects so we need to make sure self.zp is a connect
        connect_zps = filter(lambda d: (d.is_zp80() or d.is_zp90() or d.is_s15()),
                             self.my_devices)
        if self.zp not in connect_zps:
            self.mock_lechmere.remove_mutual_ssl_overrides(self.zp)
            self.zp = random.choice(connect_zps)
        if not self.zp.DeviceProperties.is_device_secure_registered():
            self.set_string_overrides_and_bounce_anacapa([self.zp])
            self.verifyTrueOrStop(PR.reg_zp(self.zp),
                                  "ZP <{}> successfully registered?".format(self.zp))
        self.mock_lechmere.set_player_connect_override_mutual_ssl(self.zp)

        # Get a valid auth token and setup the playerSetting endpoint
        player_endpoint = "player"
        playerId_playerSettings_stem = self.playerSettings_stem.format(self.zp.ip,
                                                                       self.zp.ssl_port,
                                                                       self.zp.raw_udn)

        # Verify we are able to change the volumeMode to PASS_THROUGH
        resp = muse_helper.hit_rest_endpoint(zp=self.zp, is_secure=True, req_url_stub=playerId_playerSettings_stem,
                                             req_method="POST", rest_endpoint=player_endpoint,
                                             req_json={"volumeMode": "PASS_THROUGH"}, token=self.config_token)
        resp = muse_helper.hit_rest_endpoint(zp=self.zp, is_secure=True, req_url_stub=playerId_playerSettings_stem,
                                             req_method="GET", rest_endpoint=player_endpoint)
        muse_helper.verify_rest_response(fixture=self, resp=resp,
                                         expected_resp_code=HTTPStatusCode.OK,
                                         expected_resp_text=None)
        self.verifyTrueOrFailCase(json.loads(resp.text)["volumeMode"] == "PASS_THROUGH",
                                  "{} volumeMode: {}".format(self.zp.friendly_name,
                                                             json.loads(resp.text)["volumeMode"]))

        # Verify we return 401 if the token is empty.
        resp = muse_helper.hit_rest_endpoint(zp=self.zp, is_secure=True, req_url_stub=playerId_playerSettings_stem,
                                             req_method="POST", rest_endpoint=player_endpoint,
                                             req_json={"volumeMode":"FIXED"})
        muse_helper.verify_rest_response(fixture=self, resp=resp,
                                         expected_resp_code=HTTPStatusCode.UNAUTHORIZED,
                                         expected_resp_text="")

        # playerSettings requires HH_CONFIG write auth. Verify we return 403 if the token isn't the right scope
        token = self.mock_lechmere.create_user_token(self.user_id, Scopes.AS_ADD_PLAYER.string)
        resp = muse_helper.hit_rest_endpoint(zp=self.zp, is_secure=True, req_url_stub=playerId_playerSettings_stem,
                                             req_method="POST", rest_endpoint=player_endpoint,
                                             req_json={"volumeMode": "FIXED"}, token=token)
        muse_helper.verify_rest_response(fixture=self, resp=resp,
                                         expected_resp_code=HTTPStatusCode.FORBIDDEN,
                                         expected_resp_text="")

        # Verify the previous failed POST attempts didn't actually change the volumeMode then return to VARIABLE mode
        resp = muse_helper.hit_rest_endpoint(zp=self.zp, is_secure=True, req_url_stub=playerId_playerSettings_stem,
                                             req_method="GET", rest_endpoint=player_endpoint)
        muse_helper.verify_rest_response(fixture=self, resp=resp,
                                         expected_resp_code=HTTPStatusCode.OK,
                                         expected_resp_text=None)
        self.verifyTrueOrFailCase(json.loads(resp.text)["volumeMode"] == "PASS_THROUGH",
                                  "{} volumeMode: {}".format(self.zp.friendly_name,
                                                             json.loads(resp.text)["volumeMode"]))

        resp = muse_helper.hit_rest_endpoint(zp=self.zp, is_secure=True, req_url_stub=playerId_playerSettings_stem,
                                             req_method="POST", rest_endpoint=player_endpoint,
                                             req_json={"volumeMode": "VARIABLE"}, token=self.config_admin_token)
        resp = muse_helper.hit_rest_endpoint(zp=self.zp, is_secure=True, req_url_stub=playerId_playerSettings_stem,
                                             req_method="GET", rest_endpoint=player_endpoint)
        muse_helper.verify_rest_response(fixture=self, resp=resp,
                                         expected_resp_code=HTTPStatusCode.OK,
                                         expected_resp_text=None)
        self.verifyTrueOrFailCase(json.loads(resp.text)["volumeMode"] == "VARIABLE",
                                  "{} volumeMode: {}".format(self.zp.friendly_name,
                                                             json.loads(resp.text)["volumeMode"]))

    def _wait_and_verify_cloud_connected(self, zp, timeout_seconds=30):
        """
        Wait for ZP to connect
        :param zp: zp
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`
        :param int timeout_seconds: Time to wait.
        :return:
        """
        try:
            zp.diag.wait_for_cloud_connected(timeout_seconds=timeout_seconds)
        except TimeoutError:
            pass

        # Verify player indicates cloud connected
        self.logger.error(zp.diag.get_cloud_as_dict())
        self.logger.error('{}'.format(zp.diag.get_raw_anacapa_trace()))
        self.verifyTrueOrStop(zp.diag.get_cloud_connected(),
                              "ZP <{}> successfully connected to Lechmere?".format(zp))


[docs]    def test_setting_explicitContentFiltering_via_rest_and_verifying_getRestrictedAdminSettings(self):
        '''
        SWPBL-88393: Expose Radio Location and Explicit Setting for Find and Play
        Check if restrictedAdminSettings muse call returns correct value of explicitContentFiltering attribute
        :return:
        '''
        # Get a valid auth token and setup the explicitContentFiltering_endpoint
        token = self.mock_lechmere.create_user_token(self.user_id, Scopes.AS_HH_CONFIG_ADMIN.string)
        explicitContentFiltering_endpoint = "explicitContentFiltering"

        # GET original value of explicitContentFiltering
        resp = muse_helper.hit_rest_endpoint(zp=self.zp, is_secure=True, req_url_stub=self.restrictedAdminSettings_stem,
                                             req_method="GET", rest_endpoint=explicitContentFiltering_endpoint,
                                             req_data="true")
        muse_helper.verify_rest_response(fixture=self, resp=resp, expected_resp_code=HTTPStatusCode.OK)
        original_value = resp.text

        cmd = [{"namespace": "settings",
                "command": "getRestrictedAdminSettings",
                "householdId": self.zp.muse_hhid},
               {}]

        # Send the muse command to the mock lechmere
        resp = self.mock_lechmere.send_lechmere_command(cmd=cmd,
                                                        serial=self.zp.serialNumber,
                                                        content_type="application/json",
                                                        response_json=False,
                                                        request_type="POST",
                                                        command_type="muse",
                                                        request_json=True)

        # stuff the response from the mock lechmere into the self.muse.test_events object
        # so that we can use self.verify_event to look at the response event
        self.muse = self.make_connection(self.zp)
        event = self.muse._handle_message(resp)
        self.muse.test_events.append(event)
        self.verify_event(restrictedAdminSettingsSuccess, timeout=1, connection=self.muse,
                          event_verifier=lambda e: (e.body['explicitContentFiltering'] is False))

        # POST True to explicitContentFiltering using REST endpoint
        resp = muse_helper.hit_rest_endpoint(zp=self.zp, is_secure=True, req_url_stub=self.restrictedAdminSettings_stem,
                                             req_method="POST", rest_endpoint=explicitContentFiltering_endpoint,
                                             token=token, req_data="true")
        muse_helper.verify_rest_response(fixture=self, resp=resp,
                                         expected_resp_code=HTTPStatusCode.OK,
                                         expected_resp_text="")

        # Verify Muse getter returns True as explicitContentFiltering value
        self.muse.get_restrictedAdminSettings(hhid=self.zp.muse_hhid)
        self.verify_event(restrictedAdminSettingsSuccess, timeout=1, connection=self.muse,
                          event_verifier=lambda e: (e.body['explicitContentFiltering'] is True))

        # POST original_value to explicitContentFiltering using REST endpoint
        resp = muse_helper.hit_rest_endpoint(zp=self.zp, is_secure=True, req_url_stub=self.restrictedAdminSettings_stem,
                                             req_method="POST", rest_endpoint=explicitContentFiltering_endpoint,
                                             token=token, req_data=original_value)
        muse_helper.verify_rest_response(fixture=self, resp=resp,
                                         expected_resp_code=HTTPStatusCode.OK,
                                         expected_resp_text="")



[docs]    def test_setting_radioLocation_via_rest_and_verifying_via_getPublicSettings(self):
        '''
        SWPBL-88393: Expose Radio Location and Explicit Setting for Find and Play
        Check if publicSettings muse call returns correct value of radioLocation attribute
        :return:
        '''
        # Setup the radioLocation_endpoint
        radioLocation_endpoint = "R_RadioLocation"

        # GET original value of radioLocation
        resp = muse_helper.hit_rest_endpoint(zp=self.zp, is_secure=True, req_url_stub=self.publicSettings_stem,
                                             req_method="GET", rest_endpoint=radioLocation_endpoint)

        # Check if radioLocation setting exists
        if (resp.status_code == HTTPStatusCode.NOT_FOUND):
            # Setting does not exist
            original_value = None
        else:
            # Setting exists
            muse_helper.verify_rest_response(fixture=self, resp=resp, expected_resp_code=HTTPStatusCode.OK)
            original_value = resp.text

        # POST DEFAULT_LOCATION to radioLocation
        resp = muse_helper.hit_rest_endpoint(zp=self.zp, is_secure=True, req_url_stub=self.publicSettings_stem,
                                             req_method="POST", rest_endpoint=radioLocation_endpoint,
                                             req_data=DEFAULT_LOCATION)

        muse_helper.verify_rest_response(fixture=self, resp=resp,
                                         expected_resp_code=HTTPStatusCode.OK,
                                         expected_resp_text="")

        # Verify Muse getter returns DEFAULT_LOCATION as radioLocation value
        self.muse.get_publicSettings(hhid=self.zp.muse_hhid)
        self.verify_event(publicSettingsSuccess, timeout=1, connection=self.muse,
                          event_verifier=lambda e: (e.body['radioLocation'] == DEFAULT_LOCATION))

        # Revert back to original setting
        if(original_value is not None):
            # POST original_value to radioLocation
            resp = muse_helper.hit_rest_endpoint(zp=self.zp, is_secure=True, req_url_stub=self.publicSettings_stem,
                                                 req_method="POST", rest_endpoint=radioLocation_endpoint,
                                                 req_data=original_value)
            muse_helper.verify_rest_response(fixture=self, resp=resp,
                                             expected_resp_code=HTTPStatusCode.OK,
                                             expected_resp_text="")
        else:
            # Remove radioLocation setting
            self.zp.SystemProperties.Remove(radioLocation_endpoint)
