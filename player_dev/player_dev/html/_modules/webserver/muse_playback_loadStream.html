                                      enqueued_uris_and_metadata=self.enqueued_uris_and_metadata)

        # Start playback and reset the player playmodes back to a known state
        self.muse.play(hhid=self.zp.muse_hhid, group_id=self.gid)

    def _verify_tracks_in_shared_queue(self, expected_num_tracks="2"):

    def _autoplay_test_flow(self, autoplay):
        """
        Common test code for the following 3 tests.
        :param action:
        :param tracks:
        :param autoplay:
        :return:
        """
        # Load and verify the number of tracks in the queue
        self.muse.load_stream(hhid=self.zp.muse_hhid, groupId=self.gid,
                              stationId=tunein_stationId,
                              playOnCompletion=autoplay)
        self._verify_tracks_in_shared_queue()
        self.clear_message_queue()

        # Verify the current play state based on the autoplay value
        # True should always play, false should always pause, null doesn't change current state
        self.muse.playback.get_playback_status(group_id=self.gid, hhid=self.zp.muse_hhid)
        if autoplay:
            self.verify_event(PlaybackStatusEvent, timeout=1,
                              event_verifier=lambda e: (e.playback_state == PlaybackState.PLAYING))
        else:
            self.verify_event(PlaybackStatusEvent, timeout=1,
                              event_verifier=lambda e: (e.playback_state == PlaybackState.IDLE))

        # If the player is stopped, send a play command
        if self.zp.AVTransport.is_stopped():
            self.clear_message_queue()
            self.muse.play(hhid=self.zp.muse_hhid, group_id=self.gid)
            self.verify_event(PlaybackStatusEvent, timeout=2,
                              event_verifier=lambda e: (e.playback_state == PlaybackState.PLAYING))

        self.muse.metadata.subscribe(group_id=self.gid, hhid=self.zp.muse_hhid)
        self.verify_event(MetadataStatusEvent, timeout=1,
                          event_verifier=lambda e: ("92.5" in e.container["name"] and
                                                    e.container["type"] == "station" and
                                                    e.container["id"] == tunein_stationId and
                                                    e.container["service"]["name"] == "TuneIn"))

[docs]    def test_loadStream_missing_parameters(self):
        """
        Cycles through each input param to loadContainer and verify the response event
        """
        # create dict of all the args the command takes
        params = {"hhid": self.zp.muse_hhid, "groupId": self.gid,
                  "stationId": tunein_stationId, "playOnCompletion": False}
        # Define which ones are optional
        optional_params = ("playOnCompletion")

        for param in params:
            for value in ("True", 123, None):
                self.logger.debug("Setting {} to {}".format(param, value))

                # make a copy of the params dict so it can be reset after iteration
                new_params = dict(params)
                new_params[param] = value
                resp = self.muse.load_stream(**new_params)

                # Sending a null value for an optional param should result in a success response
                if value is None and param in optional_params:
                    self.verifyIsInstanceOrFailCase(resp, Success, "Got success response?")
                elif param == "groupId" and value is not None:
                    self.verifyIsInstanceOrFailCase(resp, GroupCoordinatorChangedEvent,
                                                    "Got GroupCoordinatorChangedEvent response?")
                # All other cases should result in one of these 3 errors
                else:
                    self.verifyIsInstanceOrFailCase(resp, Error, "Got error response?")
                    self.verifyInOrFailCase(resp.body.errorCode,
                                    (ErrorEnum.GLOBAL_MISSING_PARAMETERS.value,
                                     ErrorEnum.GLOBAL_INVALID_OBJECT_ID.value,
                                     ErrorEnum.GLOBAL_INVALID_PARAMETER.value),
                                    "Got expected error code?")



[docs]    def test_loadStream_stationId_error_check(self):
        """
        Cycle through each parameter in the containerId object and verify the correct error/
        playback event
        :param param:
        :param value:
        :return:
        """
        for param in tunein_stationId:
            for value in (None, "sn_0", 254):
                # Inject the containerId object with different values
                self.logger.debug("Setting {} to {}".format(param, value))

                # Load and verify the response events
                new_params = dict(tunein_stationId)
                new_params[param] = value
                self.muse.load_stream(hhid=self.zp.muse_hhid, groupId=self.gid,
                              stationId=new_params,
                              playOnCompletion=True)
                if param == "accountId" and not isinstance(value, int):
                    # The player should be able to load if the serviceId is valid and the accountId is null or valid
                    self.verify_event(PlaybackStatusEvent, timeout=1,
                              event_verifier=lambda e: (e.playback_state == PlaybackState.PLAYING))
                elif param == "objectId" and value is not None:
                    # Incorrect objectIds will result in failed playback
                    self.verify_event(PlaybackErrorEvent, timeout=1,
                              event_verifier=lambda e: (e.error_code == "ERROR_PLAYBACK_FAILED"))
                elif param == "serviceId" and isinstance(value, str):
                    self.verify_event(Error, timeout=1,
                              event_verifier=lambda e: (e.error_code == "ERROR_ACCOUNT_INVALID_ID" and
                                                        param in e.reason))
                elif param == "objectId" and value is None:
                    self.verify_event(Error, timeout=1,
                              event_verifier=lambda e: (e.error_code == "ERROR_MISSING_PARAMETERS" and
                                                        param in e.reason))
