[docs]    def test_verify_groupVolume_reports_correct_volume_level_after_reboot(self):
        """
        SWPBL-63287
        Verify that after a reboot, groupVolume events report the correct volume level,
        whatever that level was set before the reboot
        :return:
        """
        gid_cached = self.gid
        self.zp.GroupRenderingControl.SetGroupVolume(self.volume)
        self.zp.GroupRenderingControl.SetGroupMute(False)
        wait_until_true(lambda: (self.zp.RenderingControl.GetVolume() == self.volume
                                 and self.zp.RenderingControl.GetMute() is False),
                        timeout_seconds=5, iteration_delay=1)
        self.zp.bounce_anacapa_and_wait()
        self.zp.cli.wait_for_anacapad(iteration_delay=1, timeout_seconds=10,
                                      reason="Timed out waiting for 'anacapad' process to start after reboot")
        wait_until_true(lambda: self.gid == gid_cached,
                        timeout_seconds=30, iteration_delay=1)
        self.muse = self.make_connection(self.zp)
        self.muse.groupVolume.subscribe(group_id=gid_cached, hhid=self.zp.muse_hhid)
        self.verify_event(GroupVolumeEvent, timeout=1,
                          event_verifier=lambda e: (e.volume == self.volume and
                                                    not e.muted))



[docs]    def verify_groupvolume_state(self,
                                 expected_volume=None,
                                 expected_mute=None):
        """
        if expected_volume is not None, look for an event with that volume value
        if expected_mute is not None, look for an event with that mute state
        if both are provided, look for both
        if both are None, return the first GroupVolumeEvent
        """
        last_volume = None
        last_mute = None

        self.logger.debug("Verifying group volume state (expected_volume={}), "
            "(expected_mute={})".format(expected_volume, expected_mute))

        for event in self.muse.generate_events(timeout=5):
            self.logger.debug("Muse Event: {}".format(event))
            if isinstance(event, GroupVolumeEvent):
                last_volume = event.volume
                last_mute = event.muted
                self.logger.debug("GroupVolumeEvent: (volume={}, mute={})".
                                  format(last_volume, last_mute))
                if ((expected_volume is None
                     or event.volume == expected_volume)
                    and (expected_mute is None
                         or event.muted == expected_mute)):
                    break
        else:
            pass

        if ((expected_volume is None and expected_mute is None)
            or (expected_volume is not None and expected_mute is not None)):
            return last_volume, last_mute
        elif expected_volume is not None:
            return last_volume
        else:
            return last_mute

        self.muse.groupVolume.set_volume(group_id=self.gid,
                                         hhid=self.zp.muse_hhid,
                                         volume=self.volume)
        volume = self.verify_groupvolume_state(expected_volume = self.volume)

        # SWPBL-62009
        # Verifies that the player will not set the volume to 0 if that field is missing from the

                                       hhid=self.zp.muse_hhid,
                                       muted=True)
        volume = self.verify_groupvolume_state(expected_volume = self.volume,
                                               expected_mute=True)
        self.verifyEqualOrFailCase((self.volume, True), volume,
                                   "Evented volume should match set volume after mute")

[docs]    def test_unsubscribe_after_gc_gone(self):
        """
        Test unsubscribe to groupvolume namespace on after gc changed
        """
        # Other group coordinator
        diff_gc = self.get_playback_gc()

        # Subscribe to group volume on zp
        self.muse.groupVolume.subscribe(group_id=self.gid, hhid=self.zp.muse_hhid)
        volume, mute = self.verify_groupvolume_state()

        # Cache GID
        gid_cached = self.gid;

        # Join other group, zp is no longer group coordinator, this will trigger group coordinator gone
        self.zp.AVTransport.join_group(diff_gc)
        diff_gc.AVTransport.BecomeCoordinatorOfStandaloneGroup()
        self.verify_event(GroupCoordinatorChangedEvent,
                          event_verifier=lambda e: (e.group_status == GroupStatus.GONE))

        # Send playback unsubscribe command
        self.muse.playback.unsubscribe(group_id=gid_cached, hhid=self.zp.muse_hhid)

        # Verify the error is global error with code INVALID_OBJECT_ID
        self.verify_event(Error, event_verifier=lambda e: e.error_code == Error.INVALID_OBJECT_ID)

        """
        Test unsubscribe to groupvolume namespace on after gc moved
        """
        # Other group coordinator
        diff_gc = self.get_playback_gc()

        # Subscribe to group volume on zp
        self.muse.groupVolume.subscribe(group_id=self.gid, hhid=self.zp.muse_hhid)
        volume, mute = self.verify_groupvolume_state()

        # Cache GID
        gid_cached = self.gid;

        # Have different player join zp and become standalone, this will trigger group coordinator moved
        diff_gc.AVTransport.join_group(self.zp)

        self.zp.AVTransport.BecomeCoordinatorOfStandaloneGroup()
        self.verify_event(GroupCoordinatorChangedEvent,
                          event_verifier=lambda e: (e.group_status == GroupStatus.MOVED))

        # Send playback unsubscribe command
        self.muse.playback.unsubscribe(group_id=gid_cached, hhid=self.zp.muse_hhid)

        # Verify the error is global error with code INVALID_OBJECT_ID
        self.verify_event(Error, event_verifier=lambda e: e.error_code == Error.INVALID_OBJECT_ID)
