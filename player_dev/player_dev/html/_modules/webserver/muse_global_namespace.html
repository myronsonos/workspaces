[docs]    def test_bad_gid(self):
        # Generate a changed event for the bad GID
        bad_gid = 'bad_gid'
        self.muse.set_muted_groupvolume(muted=True, group_id=self.gid, hhid=self.zp.muse_hhid)
        self.muse.play(group_id=bad_gid, hhid=self.zp.muse_hhid)

        # Confirm group coordinator changed event gets returned
        self.verify_event(GroupCoordinatorChangedEvent,
                          event_verifier=(lambda e: e.group_status == GroupStatus.GONE and e.header.householdId == self.zp.muse_hhid and e.header.groupId == bad_gid))

        # Try playing with a *good* group ID
        self.muse.play(group_id=self.gid, hhid=self.zp.muse_hhid)
        self.verify_event(Success)

        # Confirm no group coordinator changed events
        self.verify_event(GroupCoordinatorChangedEvent, expect_event=False, timeout=1)



[docs]    def test_group_move_on_delegation(self):
        """
        Verify that GROUP_GONE does not sneak in if a command is ordered soon
        after delegation.
        @see PTS-2543
        """
        self.verifyNotEqualOrFailCase(self.gid, self.zp_2.GroupManagement.get_local_group_uuid(),
                                      "different GC has same GID!")
        self.zp_2.AVTransport.join_group(self.zp)

        self.muse.set_muted_groupvolume(muted=True, group_id=self.gid, hhid=self.zp.muse_hhid)
        self.verify_event(Success)

        # We want to ask the muse endpoint a question during delegation in
        # order to stress the code in question. Load two normal HTTP tracks
        # because cloud queues don't support cross fading.
        self.zp.AVTransport.SetCrossfadeMode(True)
        self.muse.play(group_id=self.gid, hhid=self.zp.muse_hhid)
        self.verify_event(Success)
        self.zp.AVTransport.seek_to_position_before_track_end(20)

        # wait until song transition to delegate (actively cross fading)
        # currently, delegation is delayed up to 6 seconds while cross
        # fading.
        track_duration = self.zp.AVTransport.get_current_track_duration()
        track_duration_seconds = reltime_str_to_seconds(track_duration)

        #TODO fix target_millis if the crossfade time changes
        target_millis = (track_duration_seconds * 1000) - 6000
        wait_until_true(lambda: self.muse.get_playback_status(self.gid, self.zp.muse_hhid).position_millis >= target_millis,
                        iteration_delay=1, timeout_seconds=track_duration_seconds + 10)

        # delegate!
        original_gid = self.gid
        self.zp.AVTransport.leave_group()

        # try to pause the old GID
        self.muse.playback.pause(group_id=original_gid, hhid=self.zp.muse_hhid)
        # I don't use self.verify_event here because I want to verify
        # that STATUS_GONE never arrives
        gotStatusCount = 0
        for event in self.muse.generate_events():
            if isinstance(event, GroupCoordinatorChangedEvent):
                self.verifyNotEqualOrFailCase(event.group_status, GroupStatus.GONE,
                                              'group should have moved')
                gotStatusCount += 1

        # This is only expected once because we never subscribed for any events.
        # Therefore, the event version of the GroupCoordinatorChangedEvent is
        # never delivered. Use "Greater than or equal" in case the implementation
        # changes and clients are always signed up for GroupCoordinatorChangedEvent
        # events.
        self.verifyGreaterEqualOrFailCase(gotStatusCount, 1, "should move at least once")

[docs]    def test_group_status_updated_only_sent_when_player_is_gc(self):
        """
        Verify the GROUP_STATUS_UPDATED event is only sent once a player is the GC.
        Verify the websocket URL is not truncated on a GC change.
        This is a regression test for SWPBL-43196 and SWPBL-82646.
        """
        self.muse.subscribe_playback(group_id=self.gid, hhid=self.zp.muse_hhid)
        self.zp_2.AVTransport.join_group(self.zp)
        self.zp.AVTransport.BecomeCoordinatorOfStandaloneGroup()
        zp2_muse = self.make_connection(self.zp_2)
        zp2_muse.subscribe_playback(group_id=self.zp_2.ZoneGroupTopology.GetZoneGroupAttributes()['CurrentZoneGroupID'],
                                    hhid=self.zp.muse_hhid)
        wait_until_true(lambda: self.zp_2.GroupManagement.is_group_coordinator_local())
        self.verify_event(event_type=GroupCoordinatorChangedEvent,
                          expect_event=True,
                          event_verifier=lambda e: (e.group_status == GroupStatus.MOVED and
                                                    (e.websocket_url == WEBSOCKET_URL.format(self.zp_2.ip))))

    def test_no_body_error(self, *args):
        """
        Regression test for SWPBL-47738. This test verifies the player will return an errror
        if a command without a body is sent whether or not that command has required body
        parameters. These tests on pts_team use an additional 'reason' check but this feature
        is NYI on mainline until SWPBL-47625 is done and merged.
        """

        namespaces, commands = args

        payload = '[{{"groupId" : "{}", "householdId" : "{}",' \
                  '"namespace" : "{}:1",' \
                  '"command" : "{}"}}]'.format(self.gid, self.zp.muse_hhid, namespaces, commands)

        msg = WebsocketMessageRaw(payload)

        self.muse.send_message_raw(msg)
        self.verify_event(Error, event_verifier=lambda e: e.error_code == Error.MISSING_PARAM)
