    def _verify_metadata_and_playback(self, objectId="1", accountId="sn_0", item_id="3"):
        '''
        Verifies the metadata event has the correct track.id information corresponding with
        use of track 1 from the node.js mock SMAPI server.
        :param objectId: objectId of the track from the SMAPI service. "1" is track 1 from the mock SMAPI server.
        :param accountId: accountId of the account from the SMAPI server. "sn_0" is the default anon Id used.
        :return:
        '''
        self.verify_event(PlaybackStatusEvent, timeout=1,
                          event_verifier=lambda e: (e.playback_state == PlaybackState.PLAYING and
                                                    e.item_id == item_id))
        self.muse.metadata.subscribe(group_id=self.gid, hhid=self.zp.muse_hhid)
        self.verify_event(MetadataStatusEvent,
                          event_verifier=lambda e: "id" in e.current_item["track"] and
                                                   e.current_item["track"]["id"]["serviceId"] == str(self.sid) and
                                                   e.current_item["track"]["id"]["objectId"] == objectId and
                                                   e.current_item["track"]["id"]["accountId"] == accountId and
                                                   # Service object should be in current track
                                                   "service" in e.current_item["track"] and
                                                   e.current_item["track"]["service"]["name"] == self.service_name,
                          timeout=1)


[docs]    def test_verify_player_resumes_from_frame_cache_on_v2_servers(self):
        """
        If the CQ server is running the V2 endpoint, the player should resume from the frame cache
        regardless of how much has current track has been cached in the frame cache
        :return:
        """
        if self.zp.is_legacy_device():
            self.skip("This test needs a device with a frame cache")
        for x in range(3,1,-1):
            self.cq_control.add_smapi_track_to_end(track=str(x), smapi_ctl=smapi_ctl,
                                                   accountId=self.muse_account_id,
                                                   serviceId=self.sid)
        self.muse.load_cloud_queue(session_id=self.muse.session_id,
                                   hhid=self.zp.muse_hhid,
                                   queue_base_url=self.cq_control.CQ_PLAYBACK_URI,
                                   play_on_completion=True,
                                   item_id="3")
        self.verify_event(PlaybackStatusEvent, timeout=1,
                          event_verifier=lambda e: e.playback_state == PlaybackState.PLAYING)
        self.muse.pause(session_id=self.muse.session_id, hhid=self.zp.muse_hhid,
                        group_id=self.gid)
        # Sleeping here to allow the frame cache to fill up with the current track
        sleep(10)
        self.verify_event(PlaybackStatusEvent, timeout=1,
                          event_verifier=lambda e: e.playback_state == PlaybackState.PAUSED)
        self.muse.play(session_id=self.muse.session_id, hhid=self.zp.muse_hhid,
                       group_id=self.gid)
        self.verify_event(PlaybackStatusEvent, timeout=1,
                          event_verifier=lambda e: e.playback_state == PlaybackState.PLAYING)
        logs, time = smapi_ctl.wait_for_filtered_log(smapi=smapi_ctl,
                                                     calls=["getMediaURI"],
                                                     expected_items=1,
                                                     item_id="3")
        self.verifyLessEqualOrFailCase(len(logs), 1,
                                       "SMAPI track should resume from frame cache using CQ V2")



[docs]    def test_verify_buffered_audio_is_saved_if_itemWindow_is_unchanged(self):
        """
        SWPBL-63191
        Verify that if the upcoming itemId does not change once the player starts buffering it,
        a queue replacement operation will not cause the player to discard what is already buffered
        :return:
        """
        if is_locked_build():
            self.skip('locked builds does not support withWindow commands on the LAN')

        for x in range(1,3):
            self.cq_control.add_smapi_track_to_end(track=str(x), smapi_ctl=smapi_ctl,
                                                   accountId=self.muse_account_id,
                                                   serviceId=self.sid)
        window = self.cq_control.get_itemwindow_endpoint()
        self.muse.load_cloud_queue(session_id=self.muse.session_id,
                                   hhid=self.zp.muse_hhid,
                                   queue_base_url=self.cq_control.CQ_PLAYBACK_URI,
                                   play_on_completion=True,
                                   item_id="3",
                                   position_millis=10000)
        self.verify_event(PlaybackStatusEvent, timeout=1,
                          event_verifier=lambda e: e.playback_state == PlaybackState.PLAYING)
        # Let the player play to buffer the next track
        wait_until_true(lambda: self.zp.AVTransport.GetPositionInfo()["RelTime"] >= "0:00:27",
                        start_delay=15, timeout_seconds=5, iteration_delay=.5)
        # skipToItemWithWindow clears the queue, pushes the itemWindow, unchanged, to the player
        self.muse.skip_to_item_with_window(hhid=self.zp.muse_hhid, session_id=self.muse.session_id,
                                           item_id="3", window=window)
        self.verify_event(PlaybackStatusEvent, timeout=1,
                          event_verifier=lambda e: (e.playback_state == PlaybackState.PLAYING and
                                                    e.item_id == "4"))

        # Verify that we only see one /getMediaURI call to track 2 on the smapi server,
        # meaning the buffered data has been retained
        logs, time = smapi_ctl.wait_for_filtered_log(smapi=smapi_ctl,
                                                     calls=["getMediaURI"],
                                                     expected_items=1,
                                                     item_id="2",
                                                     timeout=5)
        self.verifyEqualOrFailCase(1, len(logs),
                                   "There should be only 1 /getMediaURI call for track 2 on the smapi server")

[docs]    def test_player_blacklists_server_with_small_windows(self):
        """
        SWPBL-88592: Verify that player blacklist logic works when cloud queue provides small windows

        The player blacklists a server after five consecutive failures
        Expect:
        1. playback error on the last track to play: track_list[4] (zero based index)
        2. the player to pause on the next track, track_list[5]
        """
        self.cq_control.clear_queue()
        item_list = []
        for _ in range (20):
            my_item = self.cq_control.add_smapi_track_to_end(track="1", smapi_ctl=smapi_ctl,
                                                             accountId=self.muse_account_id,
                                                             serviceId=self.sid)
            item_list.append(my_item)

        smapi_ctl.stop_node_smapi_server(self.smapi)
        self.cq_control.set_window_limits({'tracksBeforeLimit':0,'tracksAfterLimit':1})

        sample_window = self.cq_control.get_itemwindow_endpoint(itemid=item_list[10]['id'], window_width=20)
        self.verifyEqualOrStop(len(sample_window['items']), 2,
                               "verify that window size limiting is configured correctly")

        self.muse.load_cloud_queue(session_id=self.muse.session_id,
                                   hhid=self.zp.muse_hhid,
                                   queue_base_url=self.cq_control.CQ_PLAYBACK_URI,
                                   httpAuthorization=None,
                                   item_id=item_list[0]['id'],
                                   play_on_completion=True,
                                   track_metadata=None)

        self.verify_event(PlaybackErrorEvent, timeout=5,
                          event_verifier=lambda e: (e.error_code == "ERROR_PLAYBACK_FAILED" and
                                                    e.item_id == item_list[4]['id']))
        self.verify_event(PlaybackStatusEvent, timeout=1, past_events=True,
                          event_verifier=lambda e: (e.playback_state == PlaybackState.PAUSED and
                                                    e.item_id == item_list[5]['id']))

        # verify that "playing" resets the blacklist count and it triggers again.
        self.muse.play(hhid=self.zp.muse_hhid, group_id=self.gid)
        self.verify_event(PlaybackErrorEvent, timeout=5,
                          event_verifier=lambda e: (e.error_code == "ERROR_PLAYBACK_FAILED" and
                                                    e.item_id == item_list[9]['id']))
        self.verify_event(PlaybackStatusEvent, timeout=1, past_events=True,
                          event_verifier=lambda e: (e.playback_state == PlaybackState.PAUSED and
                                                    e.item_id == item_list[10]['id']))

        # concerned with tqueue.
        for module in ("cqfsm", "cloudqueue", "chsrc", "muse", "cssc", "blacklist", "sonosapi", "avt_impl"):
            self.zp.diag.set_diag_level(module, 1)
        for module in ("spot", "cssc"):

        try:
            self.zp.diag.set_diag_level("tqueue", 5)
            self.cq_control.clear_queue()
            item_list = []

                                                                 accountId=self.muse_account_id,
                                                                 serviceId=self.sid)
                item_list.append(my_item)

            smapi_ctl.stop_node_smapi_server(self.smapi)
            for _ in xrange(20):
                for i in xrange(2, 10, 2):

                                               hhid=self.zp.muse_hhid,

                                               item_id=item_list[0]['id'],

                                               track_metadata=None)

                    # If the window size is 5 or greater, verify the player blacklists and pauses after 5 tracks
