[docs]    def test_blacklist_playback_errors(self):
        """
        Verify that the player doesn't continuously event any playbackErrors
        after being blacklisted and doesn't event UNSUPPORTED_FORMAT errors
        when failing to play SMAPI tracks.
        :return:
        """
        self.zp.AVTransport.clear_play_state()
        self.zp.AVTransport.clear_queue()
        self._start_smapi_server()
        try:
            self.zp.AVTransport.use_queue()

            num_tracks = 0
            # Add alternating tracks to queue for a total of 10 tracks (5 required for blacklist)
            for _ in xrange(5):
                for track_id in xrange(1, 3):
                    self.zp.Queue.AddURI(queue_id=0,
                                         update_id=0,
                                         enqueueduri=smapi_ctl.SMAPI_SERVER_TRACK_URIS[str(track_id)],
                                         enqueuedurimetadata=smapi_ctl.SMAPI_TRACK_METADATA[str(track_id)],
                                         desiredfirsttracknumberenqueued="",
                                         enqueueasnext="")
                    num_tracks += 1

            # Wait for the queue status to be reflected in the zp
            wait_until_true(lambda: int(self.zp.Queue.Browse(queue_id=0,
                                                             starting_index=0,
                                                             requested_count="")["NumberReturned"]) == num_tracks,
                            iteration_delay=.5,
                            timeout_seconds=5)
            # Clear the muse event history and stop server
            self.muse.clear_test_events()
            # Subscribe
            self.muse.subscribe_playback(hhid=self.zp.muse_hhid, group_id=self.gid)
            # Stop the node SMAPI server.
            smapi_ctl.stop_node_smapi_server(self.smapi)
            # Play
            self.muse.play(hhid=self.zp.muse_hhid, group_id=self.gid)

            # Verify no unsupported format playback failed errors appear
            self.verify_event(PlaybackErrorEvent, expect_event=False, timeout=1,
                event_verifier=lambda e: e.error_code == PlaybackError.PLAYBACK_FAILED and
                                         e.reason == PlaybackError.UNSUPPORTED_FORMAT)

            # Verify player is in the correct state
            self.muse.playback.get_playback_status(hhid=self.zp.muse_hhid, group_id=self.gid)
            self.verify_event(PlaybackStatusEvent, timeout=1,
                              event_verifier=lambda e: (e.playback_state == PlaybackState.IDLE))

            # Filter through previous events to make sure we got the expected number
            errors = filter(lambda e: isinstance(e, PlaybackErrorEvent), self.muse.test_events)
            self.verifyInRangeOrFailCase(len(errors), 1, 5,
                "<{}> got at least 1 and at most 5 errors?".format(self.zp))

            # Verify all errors were the expected type
            for e in errors:
                self.verifyEqualOrFailCase(PlaybackError.PLAYBACK_FAILED, e.error_code,
                    "<{}> got expected error code for <{}>".format(self.zp, e))
                self.verifyEqualOrFailCase(PlaybackError.UNREACHABLE, e.reason,
                    "<{}> got expected reason for <{}>".format(self.zp, e))
        finally:
            self._stop_smapi_server()
