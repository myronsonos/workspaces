"""
This suite contains tests for the SSDP SpeakerGroup:1 Search Target (ST). The SSDP messages
in this suite are parsed in /core/src/sonos/group_ssdp/notification.py where the correct
ST is set in __init__(). All verifications within this script are comparing a device UDN
from self.my_devices to the latest 'alive' or 'byebye' received with the SpeakerGroup:1 ST or NT.

Currently this script relies on the Workflow framework to manage the initial set of
players to test against, which uses responses from the existing SSDP ST `ZonePlayer:1`
and then compares that to a list of M-Search responses returned for Speakergroup:1. This
may be non-obvious to people not familiar with the framework and new search target and
issues have been seen in the past where more devices were in self.my_devices than were
returned by searching for SpeakerGroup:1 (this can be seen in some tests where we filter
the SUB out of the my_devices list)
"""

def _cleanup_all(*devices):
    """
    Returns all the devices back into a solo, ungrouped state
    :param devices: Devices to revert back to normal state
    """
    for zp in devices:
        MPCSC.remove_mutual_ssl_overrides(zp)
        if zp.modelNumber in PLAYBACK_DEVICES:
            zp.DeviceProperties.cleanup_bonded_configuration()
            zp.AVTransport.clear_play_state()


def _escape_http_quoted_string(quoted_string):
    return quoted_string.replace('\\', '\\\\').replace('"', '\\"')

    def _validate_msearch(self, devices):
        # Typically, we should send three to five multicast and broadcast
        # msearch requests, one second apart from one another. Unfortunately,
        # coherence does not allow us to do this. Let's settle for the next
        # best thing, sending five multicast msearch requests two and a half
        # seconds apart from one other.
        msearcher = MSearcher()
        for _ in xrange(5):
            msearcher.send_msearch()
            time.sleep(SSDP_MX * 2 + 0.5)

        # Collect the results of the MSearches
        try:
            expected_udns = {zp.raw_udn for zp in devices
                            if (zp.playback_device) }
        except TypeError:
            expected_udns = [devices.raw_udn, ]

        results = {response.uuid: response
                   for response in msearcher.generate_responses()}
        # Verify results
        for expected_udn in expected_udns:
            self.verifyInOrStop(expected_udn, set(results.keys()),
                                    "Expected playback device {} to send a response"
                                    .format(expected_udn))

        for uuid, response in results.iteritems():
            self.logger.info("UUID: {} Response: <{}>".format(uuid, response))
            self.logger.info("UUID: {}  raw: <{}>".format(uuid, response.data))
            zp = self.subnet.get_device_by_udn(uuid)
            self.verifyTrueOrStop(zp is not None, 'ZP was not found with UUID:{}'.format(uuid))

            self.verifyEqualOrStop(
                zp.api_version,
                response.api_version,
                "Expected zone group api version to match response. Expected: {} Got: {}".format(zp.api_version, response.api_version))
            self.verifyEqualOrStop(
                zp.min_api_version,
                response.min_api_version,
                "Expected zone group min api version to match response. Expected: {} Got: {}".format(zp.min_api_version, response.min_api_version))

            zg = zp.ZoneGroupTopology.GetZoneGroupAttributes()
            self.verifyEqualOrStop(
                zg['CurrentZoneGroupID'],
                response.group_info.id,
                "Expected zone group ID to match response. Expected: {} Got: {}".format(zg['CurrentZoneGroupID'], response.group_info.id))
            self.verifyEqualOrStop(
                zg['CurrentZoneGroupName'],
                response.group_info.name,
                "Expected zone group name to match response. Expected: {} Got: {}".format(zg['CurrentZoneGroupName'], response.group_info.name))
            self.verifyEqualOrStop(
                zp.GroupManagement.is_group_coordinator_local()
                and zp.is_visible(),
                response.group_info.is_coordinator,
                "Expected zone group coordinatorship to match response. Expected: {} Got: {}".format
                (zp.GroupManagement.is_group_coordinator_local(), response.group_info.is_coordinator))

[docs]    def test_not_in_household(self):
        """Test that msearches are returned by everything in the household"""
        self.logger.info("Starting msearch discovery test")
        original_devices = self.subnet.allDevices[:]

        # Get some arbitrary ZP on its own and remove it from the household
        lone_zp = random.choice([
            zp for zp in self.my_devices
            if (zp.is_playback_device()
                and not zp.is_sub()
                and not zp.is_bonded())])
        # It is possible that lone_zp might be none if none of the above if
        # conditions are met. If this occurs, let's stop the test.
        self.verifyIsNotNoneOrStop(lone_zp, "lone_zp is not equal to none")
        self.logger.info("Selected {} as lone_zp".format(lone_zp))
        # Remove the player from the subnet manually and then back up the
        # localsettings and netsettings files. This is essentially what
        # the backup_settings() context manager does but that was causing
        # issues with this test when/if lone_zp didn't reset correctly
        lone_zp.force_subnet_removal()
        for prefix in ['net', 'local']:
            lone_zp.cli.command('cp /jffs/{prefix}settings.txt '
                '/jffs/{prefix}settings.{suffix}'.format(prefix=prefix,
                    suffix='old'))
        lone_zp.cli.command('rm /jffs/*settings.txt')
        # Poke anacapa to restart here with no HH affiliation
        lone_zp.bounce_anacapa_and_wait()
        self.verifyEqualOrStop('', lone_zp.hhid,
                                   "Expected the lone zp to be removed from "
                                   "the household and have no hhid")
        original_devices.remove(lone_zp)
        self._validate_msearch(original_devices)
        # Restore the previous HH settings, again mimicking the end
        # of the backup_settings context manager manually
        for prefix in ['net', 'local']:
            lone_zp.cli.command('mv /jffs/{prefix}settings.old '
                '/jffs/{prefix}settings.{suffix}'.format(prefix=prefix,
                    suffix='txt'))
        # Poke anacapa again and wait for the device to come back
        lone_zp.bounce_anacapa_and_wait()

    def _reboot_and_wait_many(self, zps):
        pool = ThreadPool()
        pool.map(self._reboot_and_wait, zps)
        pool.close()
        pool.join()

    def _do_rename_test(self, gc, members=None, name_suffix='renamed'):
        self.logger.info("Setting up rename test")
        zps = [gc] + (members or [])
        for zp in zps:
            zp.AVTransport.clear_play_state()

            def reset_name(z, zf, n):
                def f():
                    self.logger.debug("Resetting {} to {}".format(z, n))
                    zf(n)
                return f

            self.addCleanup(reset_name(zp,
                                       zp.DeviceProperties.set_zone_name,
                                       zp.DeviceProperties.get_zone_name()))

        with group_notification_monitor() as monitor:
            self.logger.info("Restart anacapa to get byebye message")
            self._reboot_and_wait_many(zps)

            if members:
                self.logger.info("Setting ZP {} to be a group coordinator with {}"
                             "".format(gc, members))
            else:
                self.logger.info("Setting ZP {} to be a standalone group"
                             "".format(gc))

            if members:
                gc.AVTransport.link_zones(members)
                for member in members:
                    gc.topology.wait_for_topologies_to_match(member)

            final_alives, final_byebyes = monitor.get_notifications(zps)

            for zp in zps:
                self.verifyLessOrStop(
                    final_byebyes[zp.raw_udn].boot_id,
                    final_alives[zp.raw_udn].boot_id,
                    "Expected byebye for {} to precede alive".format(zp))

            expected_boot_ids = {
                uuid: n.boot_id
                for uuid, n in final_alives.iteritems()
            }
            previous_config_ids = {
                uuid: n.config_id
                for uuid, n in final_alives.iteritems()
            }

            self.logger.info("Renaming zp and verifying alive message")
            new_names = {
                zp: "{}-{}".format(
                    zp.DeviceProperties.get_zone_name()[:56],
                    name_suffix)
                for zp in zps
            }
            for zp, new_name in new_names.iteritems():
                zp.DeviceProperties.set_zone_name(new_name)

            expected_group_name = new_names[gc]
            # SWPBL-12010: Escape the special characters (backslash and DQUOTE) in the name
            expected_group_name = _escape_http_quoted_string(expected_group_name)
            if members:
                expected_group_name += " + {}".format(len(members))

            a, b = monitor.get_notifications(zps)
            final_alives.update(a)
            final_byebyes.update(b)

            for zp in zps:
                self.verifyEqualOrStop(
                    expected_boot_ids[zp.raw_udn],
                    final_alives[zp.raw_udn].boot_id,
                    "Expected boot id not to change.")
                if zp == gc:
                    self.verifyTrueOrStop(final_alives[zp.raw_udn].group_info[0],
                                          "Expect GC to report in ALIVE")
                    self.verifyLessOrStop(
                        previous_config_ids[zp.raw_udn],
                        final_alives[zp.raw_udn].config_id,
                        "Expected config id to grow.")
                    self.verifyEqualOrStop(
                        expected_group_name,
                        final_alives[zp.raw_udn].group_info.name,
                        "Expected name to match new name.")
                else:
                    self.verifyFalseOrStop(final_alives[zp.raw_udn].group_info[0],
                                           "Expect member to not report GC in ALIVE")
                    self.verifyEqualOrStop(
                        previous_config_ids[zp.raw_udn],
                        final_alives[zp.raw_udn].config_id,
                        "Expected config id not to grow for non-gc.")
                    self.verifyEqualOrStop(
                        "", final_alives[zp.raw_udn].group_info.name,
                        "Expected group name to be blank.")

            self.logger.info("Restarting anacapa to verify byebye message and "
                            "alive message on restart")
            self._reboot_and_wait_many(zps)
            a, b = monitor.get_notifications(zps)
            final_alives.update(a)
            final_byebyes.update(b)

            for zp in zps:
                self.verifyEqualOrStop(
                    expected_boot_ids[zp.raw_udn],
                    final_byebyes[zp.raw_udn].boot_id,
                    "Expected prior alive to match final byebye")
                self.verifyLessOrStop(
                    final_byebyes[zp.raw_udn].boot_id,
                    final_alives[zp.raw_udn].boot_id,
                    "Expected final byebye to precede final alive")
                self.verifyEqualOrStop(
                    _escape_http_quoted_string(new_names[zp]),
                    final_alives[zp.raw_udn].group_info.name,
                    "Expected name to match new name still.")

        """Tests that group and ungroup actions send alive and byebye messages
        as expected from the group coordinator and members.
        """
        self.logger.info("Preparing for group/ungroup testing")
        # Gather group coordinator and member and create zps collection as a
        # handy reference to all DUTs
        zps = gc, member = [d for d in self.my_devices
                            if d.is_playback_device()
                            and not d.is_secondary_bonded_zone()][:2]

        # Reset ZPs to be standalone
        for zp in zps:
            zp.AVTransport.clear_play_state()

        # Watch for SSDP notifications
        with group_notification_monitor() as monitor:
            self.logger.info(
                "Reboot all devices under test and validate SSDP notifications")
            # Reboot all ZPs to reset boot id on notifications and make sure
            # everyone is alone. Then collect notifications.
            self._reboot_and_wait_many(zps)
            for zp in zps:
                zp.AVTransport.clear_play_state()

            final_alives, final_byebyes = monitor.get_notifications(zps)

            # For each ZP validate that the boot id incremented and that for
            # each ZP a group is advertised with the zp as group coordinator and
            # the group name matching the ZP's name
            for zp in zps:
                self.verifyLessOrStop(
                    final_byebyes[zp.raw_udn].boot_id,
                    final_alives[zp.raw_udn].boot_id,
                    "Expected byebye for {} to precede alive".format(zp))
                self.verifyTrueOrStop(
                    final_alives[zp.raw_udn].group_info.is_coordinator,
                    "Expected to be a group coordinator for a standalone group")
                self.verifyEqualOrStop(
                    zp.DeviceProperties.get_zone_name(),
                    final_alives[zp.raw_udn].group_info.name,
                    "Expected group name to match standalone coordinator name")

            # Keep track of the boot ids and config ids for later
            expected_boot_ids = {
                uuid: n.boot_id
                for uuid, n in final_alives.iteritems()
            }
            previous_config_ids = {
                uuid: n.config_id
                for uuid, n in final_alives.iteritems()
            }

            self.logger.info("Create group and validate SSDP notifications")
            # Create the group and capture subsequent notifications
            gc.AVTransport.link_zones([member])
            gc.topology.wait_for_topologies_to_match(member)
            a, b = monitor.get_notifications(zps)
            final_alives.update(a)
            final_byebyes.update(b)

            # Validate that the new group is named correctly, that boot ids are
            # correct, that config ids increment, that the group coordinator

            expected_group_name = "{} + 1".format(

                expected_boot_ids[gc.raw_udn],
                final_alives[gc.raw_udn].boot_id,
