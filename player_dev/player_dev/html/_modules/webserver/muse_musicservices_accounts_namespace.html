[docs]    def missing_param_gen(self):
        """
        Generate all possible combinations of arguments with
        at least 1 missing parameter. The actual value for
        each parameter is set to 'test_str' in all cases.
        """
        # Remove implicit self arg
        if hasattr(self, 'muse') and self.muse is not None:
            args = inspect.getargspec(self.muse.ms_accounts.match).args[1:]

            for i in xrange(1, len(args)):
                it = combinations(args, i)
                for element in it:
                    x = dict(zip(element, ['test_str'] * len(element)))
                    yield x



[docs]    def test_missing_parameters(self):
        """
        Verify one or more missing parameters, regardless of the value
        of said parameters, results in a MISSING_PARAMETERS_ERROR
        """
        for params in self.missing_param_gen():
            resp = self.muse.ms_accounts.match(**params)
            self.verifyIsInstanceOrFailCase(resp, Error, "Got error response?")
            self.verifyInOrFailCase(resp.body.errorCode,
                                    (ErrorEnum.GLOBAL_MISSING_PARAMETERS.value,
                                     ErrorEnum.GLOBAL_INVALID_OBJECT_ID.value),
                                    "Got expected error code?")

        # Verify that the mock lechmere received the expected login request

[docs]    def test_match_guest_account_with_deviceId(self):
        """
        SWPBL-83258
        Verify that link deviceId is required to match if it is enabled on the server
        :return:
        """
        # restart the smapi server and enable the link code and device id
        smapi_ctl.stop_node_smapi_server(self.smapi)
        self.smapi = smapi_ctl.start_node_smapi_server(stdout=sys.stdout,
                                                       stderr=sys.stderr,
                                                       server_port=3200,
                                                       magicLinkCode=True,
                                                       magicLinkDeviceId=True)
        # Add the node smapi service but no account to the player
        self.verifyTrueOrStop(smapi_ctl.smapi_account_setup(self.zp,
                                                            self.sid,
                                                            add_account=False,
                                                            auth_type='DeviceLink'),
                              "Added custom service for smapi server to <{}>".format(self.zp))

        # Attempt to match with no device id, expect an error
        self.muse.match_ms_accounts(hhid=self.zp.muse_hhid, user_hash="", nickname="service",
                                    service_id=str(self.sid), link_code="MAGIC_LINK_CODE")
        self.verify_event(Error, timeout=1,
                          event_verifier=lambda e: (e.error_code == "ERROR_COMMAND_FAILED"))

        # Attempt to match with the right device id, should get a guest account back
        self.muse.match_ms_accounts(hhid=self.zp.muse_hhid, user_hash="", nickname="service",
                                    service_id=str(self.sid), link_code="MAGIC_LINK_CODE",
                                    linkDeviceId="MAGIC_LINK_DEVICE_ID")
        self.verify_event(musicServiceAccountsEvent, timeout=1,
                          event_verifier=lambda e: (e.nickname == "NICKNAME" and
                                                    e.service.id == str(self.sid) and
                                                    e.userIdHashCode is not None and
                                                    e.isGuest is True))
