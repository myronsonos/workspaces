    def _clear_play_state_and_ungroup(self, zp=None):
        if not zp:
            zp = self.zp
            gid = self.gid
            raw_udn = self.zp.raw_udn
        else:
            gid = zp.GroupManagement.get_local_group_uuid()
            # protect incase the player gid is None
            if gid is None:
                gid = ""
            raw_udn = zp.raw_udn

        # If self.zp's player uuid doesn't match its groupId, explicitly make it standalone
        if not zp.GroupManagement.is_group_coordinator_local():
            zp.AVTransport.BecomeCoordinatorOfStandaloneGroup(timeout_seconds=10)
            wait_until_true(lambda: raw_udn in zp.GroupManagement.get_local_group_uuid(),
                            iteration_delay=1,
                            timeout_seconds=10)

        # Clear the play state and queue if the player is previously playing
        if not zp.AVTransport.is_stopped():
            zp.AVTransport.clear_play_state()
            zp.AVTransport.clear_queue_and_wait()
            wait_until_true(lambda: zp.AVTransport.is_stopped(),
                            timeout_seconds=10, iteration_delay=1)

[docs]    def verify_event(self, event_type, expect_event=True,
                     event_verifier=None, connection=None,
                     past_events=False, timeout=5, iterations=6):
        '''
        :param event_type: Event class to match
        :param expect_event: Boolean value whether the above event should be expected or not, aka, we do NOT expect this event_type
        :param event_verifier: Optional event verifier function
        :param connection: Muse connection
        :param past_events: Boolean value whether to look at the events queue or muse.test_events to match against
                            Set to 'False' if the event_type should be in the events queue at the time of the verify_event call
                            Set to 'True' if the event_type is already in muse.test_events or are uncertain
        :param timeout: Time for the generate_events function to wait if the queue is empty
        :return:
        '''
        # get the caller's filename and line number
        # this makes it easier to debug verify_event failures
        # because the error message includes the call site.
        try:
            mystack = inspect.stack()
            # get the caller's frame
            myframe = mystack[1]
            fmtargs = myframe[1], myframe[2], event_type
        except:
            fmtargs = '<unknown>', '<unknown>', event_type
        finally:
            # Delete these variables, as recommended by the Python documentation
            # https://docs.python.org/2/library/inspect.html#the-interpreter-stack
            del mystack
            del myframe

        if connection is None:
            connection = self.muse

        discarded_events = "Discarded Muse Messages:\n"
        evt_fmt_str = '{}\t* {}\n'
        events_generated = []

        verifier_str = None
        try:
            verifier_str = ' '.join([l.strip() for l in inspect.getsource(event_verifier).splitlines()])
        except:
            pass

        def event_source(past_events=False, timeout=5):
            # Determine whether we are looking at current or past events
            if past_events:
                return self.muse.test_events
            else:
                return connection.generate_events(timeout=timeout)

        # if we are looking for a single event, put it into a list
        if not isinstance(event_type, list):
            event_type = [event_type]

        # The main loop will go through a max of 6 iterations to look for the right event
        # When it loops, it will update the muse.test_events list with whatever is on the queue
        # The inner loop does the actual comp, it looks at all the events the user sent and
        # compares that to every event in the current event queue or the muse.test_events list one-by-one
        iter = 0
        start_time = datetime.now()
        while iter < iterations:
            iter = iter + 1
            for event in event_source(past_events, timeout):
                # iterate through every event in the event_source queue
                myfmtargs = fmtargs + (event, verifier_str)
                events_generated.append(event)
                for search_event in event_type:
                    # iterate through each event we are trying to match
                    if isinstance(event, search_event):
                        if not expect_event and not event_verifier:
                            # Immediately fail the test if there's an unexpected event match
                            event_type.remove(search_event)
                            self.fail("[{}:{}] Found unexpected {} event: {}, verifier: {}".format(*myfmtargs))
                            return False
                        elif not expect_event and event_verifier(event):
                            # If there is an event_verifier, only fail if the event passes the verifier
                            event_type.remove(search_event)
                            self.fail("[{}:{}] Found unexpected {} event: {}, verifier: {}".format(*myfmtargs))
                            return False
                        elif not event_verifier or event_verifier(event):
                            # Break out of the search_event loop, we have a match
                            event_type.remove(search_event)
                            break
                        elif not event_verifier(event):
                            # Reject the potential match if it fails the event_verifier
                            discarded_events = evt_fmt_str.format(discarded_events, event)
                            self.logger.debug("[{}:{}] Looking for type {}; rejected {}, verifier: {}".format(*myfmtargs))
                    else:
                        # Drop the event if the type doesn't match
                        discarded_events = evt_fmt_str.format(discarded_events, event)
                        self.logger.debug("[{}:{}] Looking for type {}; dropped {}, verifier: {}".format(*myfmtargs))
                if not event_type:
                    # Return true if all the event_Type list is exhausted
                    return True

                # We exhausted the polling timeout without finding the event.
                if (datetime.now() - start_time).total_seconds() > (timeout * iterations):
                    self.logger.debug("[{}:{}] Looking for type {}; reached timeout polling on {}, verifier: {}".format(*myfmtargs))
                    # Set iter to max iterations to break out of the outer loop.
                    iter = iterations
                    break

            if past_events is True:
                # Sets the events source to the events queue if we don't find what we're looking in the muse.test_events list
                past_events = False
        else:
            # Only execute if the outer while loops exhausts iterations
            if expect_event:
                if events_generated:
                    fmtargs = fmtargs + (verifier_str, discarded_events.rstrip('\n'))
                    self.fail("[{}:{}] Couldn't find a {} event, verifier: {} \n {}".format(*fmtargs))
                else:
                    self.fail("Expected response of type {} from connection but got none!".format(event_type))
                    # raise NoResponseException("Expected response of type {} "
                    #                           "from connection but got none!".format(event_type))
        # Return false all iterations are exhausted and event_type list has not been exhausted
        if expect_event:
            return False
        else:
            return True

[docs]    def remove_track_from_queue(self, zp, queueId, updateId, startingIndex,
                                numberOfTracks, totalTracks, timeout=5, attempts=2):
        """
        Remove the second of the 1 second tracks and verify the new number of total tracks
        startingIndex is 1-based
        :param zp: 
        :param queueId: 
        :param updateId: 
        :param startingIndex: 
        :param numberOfTracks: 
        :param totalTracks: 
        :param timeout: 
        :return: 
        """
        for attempt in range(0, attempts):
            zp.Queue.RemoveTrackRange(queueid=queueId, updateid=updateId,
                                  startingindex=startingIndex, numberoftracks=numberOfTracks)
            if lambda: zp.Queue.Browse(queue_id=queueId,
                                       starting_index=startingIndex,
                                       requested_count="")["NumberReturned"] == totalTracks:
                return True
            sleep(timeout)
        else:
            return False

[docs]    def verify_no_open_websocket_connections(self, zp):
        """
        Validate that there are no open websocket connections remaining on a given player.
        Warnings are logged with the IP of the player and IP:PORT of the websocket client
        if an open connection is found.

        Compares open connection to the known cloud connection source of 'MuseCloudClient' and
        do not count that as a reportable violation. This keeps us from logging warnings with
        empty data on tests that have players connected to the cloud.
        """
        response = requests.get("http://{}:{}/status/api".format(zp.ip, zp.port))
        connections = ET.ElementTree(ET.fromstring(response.text)).getroot().findall("Muse")
        if connections:
            connections = connections[0].findall("Active")
            if len(connections[0]) > 0:
                count = 0
                for conn in connections[0]:
                    # "0" is the lechmere (cloud) connection
                    # "1" is the AVS2 connection on Alexa microphone enabled speakers
                    if "id" not in conn.attrib or (conn.attrib["id"] != "0" and conn.attrib["id"] != "1"):
                        self.warn(
                            "FOUND A WEBSOCKET CONNECTION LEFT OPEN TO {} FROM {}".format(zp.ip, connections[0][count][0].text))
                        count += 1

[docs]    @contextmanager
    def establish_client_connection(self, auth_header=False,
                                    auth_token=None, wait_l7_seconds=5):
        """
        Establish a client connection, terminate on __exit__
        :param bool auth_header: Use an Authorization header for this client?
        :param str auth_token: Token for auth header
        """
        # TODO: Remove wait_l7 variable when SWPBL-43810 resolved
        headers = {}
        if auth_header:
            headers = {"Authorization": "bearer {}".format(auth_token)}

        muse = self.make_connection(self.zp, headers,
            incl_auth_header=False, wait_l7_seconds=wait_l7_seconds)
        new_client = self.get_connection(muse)

        self.logger.info("Opened client connection: {}".format(new_client))

        try:
            yield self.get_connection(muse), muse
        finally:
            self.logger.info("Disconnecting client {}".format(new_client))
            self.disconnect(muse)

    def authorized_client_connection(self, auth_token,
                                     expect_success=True, wait_l7_seconds=0):
        """
        Yield an authorized client connection
        :param str auth_token: Token to use for auth
        :param bool expect_success: Do we expect the authentication to succeed?
        """
        # TODO: Remove wait_l7 variable when SWPBL-43810 resolved
        event_verifier = None
        event = AuthenticationSuccessResponse

        if not expect_success:
            event_verifier = (lambda e:
                e.body.errorCode == Error.NOT_ALLOWED.value
                and e.reason == "Not authorized")
            event = AuthenticateError

        with self.establish_client_connection(wait_l7_seconds=wait_l7_seconds) as (c, m):
            m.authentication.authenticate(auth_token)
            self.verify_event(event, event_verifier=event_verifier, connection=m)
            yield c, m
