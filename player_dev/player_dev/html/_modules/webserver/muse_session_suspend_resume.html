[docs]    def setup_cq(self):
        """
        Helper function for performing common options to setup the CQ server.

        :param dict kwargs: Keyword arguments for start_cloud_queue_server. If
            cq_port is not specified, it will be increased by one each time the server
            is restarted.
        """
        self._restart_cq_server(endpoint_version=self.endpoint_version,
                                cq_port=self.cq_port)
        # Start playback from the webserver
        self.zp.AVTransport.use_queue()
        self.zp.AVTransport.clear_queue_and_wait()
        track_list = []
        for track in self.track_source.tracks:
            track_list.append(track)
        track_list_md = queue_helpers.create_didl_lite_md(track_list)
        enqueued_uris_and_metadata = queue_helpers.create_uri_and_metadata_xml(track_list, track_list_md)
        self.zp.Queue.AddMultipleURIs(queue_id=0,
                                      update_id=0,
                                      container_uri="",
                                      container_metadata="",
                                      desired_first_track_number_enqueued=1,
                                      enqueue_as_next=0,
                                      number_of_uris=len(track_list),
                                      enqueued_uris_and_metadata=enqueued_uris_and_metadata)
        self.zp.AVTransport.Play_and_wait(timeout_seconds=10)
        self.avt = self.zp.AVTransport.get_avtransport_uri()

        self.muse_session_helper.make_muse_session(self.muse, group_id=self.gid,
                                                   hhid=self.zp.muse_hhid, test_fixture=self,
                                                   accountId=self.muse_account_id)
        self.verify_event(SessionInfoEvent, timeout=1,
                          event_verifier=lambda e: e.suspended is False)
        # Subscribe to playback
        self.muse.subscribe_playback(group_id=self.gid, hhid=self.zp.muse_hhid)
        self.verify_event(PlaybackStatusEvent, timeout=1)


    def _load_play_and_suspend(self, itemId='0', modify_polling=True):
        """
        Common test setup, loads and plays a CQ and then suspends. Verifies that
        player returns a sessionInfo event with "suspended=True"
        :return:
        """
        if self.zp.diag.has_testpoint_support() and modify_polling:
            # Change the version poll timer to 3 seconds and verify that the new settings are being used
            self.cq_control.set_cloudqueue_poll_timers(zp=self.zp,
                                                       playing_timer=3,
                                                       paused_timer=3)
        else:
            self.skip("Test case needs testpoint support")

        self.muse.load_cloud_queue(session_id=self.muse.session_id,
                                   hhid=self.zp.muse_hhid,
                                   queue_base_url=self.cq_control.CQ_PLAYBACK_URI,
                                   httpAuthorization=None,
                                   item_id=itemId,
                                   play_on_completion=True,
                                   track_metadata=None)
        self.verify_event(PlaybackStatusEvent, timeout=1,
                          event_verifier=lambda e: e.playback_state == PlaybackState.PLAYING)
        
        if modify_polling:
            log = self.cq_control.wait_for_filtered_log(fn=lambda l: l['path'] == '/version',
                                                        expected=2, timeout=15)
            self.verifyGreaterEqualOrFailCase(len(log), 2,
                                              "At least 2 version requests should have been made after playing interval change")
        self.cq_control.clear_log()

        self.muse.suspend_session(hhid=self.zp.muse_hhid, sessionId=self.muse.session_id)
        self.verify_event(Success, timeout=1,
                          event_verifier=lambda e: (e.response == "suspend"))
        self.verify_event(PlaybackStatusEvent, timeout=1,
                          event_verifier=lambda e: (e.playback_state == PlaybackState.IDLE and
                                                    e.item_id is None))
        self.verify_event(SessionInfoEvent, timeout=1, past_events=True,
                          event_verifier=lambda e: e.suspended is True)

        # Verify correct upnp values
        wait_until_true(lambda: self.zp.AVTransport.get_state_variable_and_time("DirectControlIsSuspended")[0] == "1",
                        timeout_seconds=10, iteration_delay=1, reason="Waiting for DirectControlIsSuspended==1")
        self.verifyEqualOrFailCase(self.__class__.__name__,
                                   self.zp.AVTransport.get_state_variable_and_time("DirectControlClientID")[0],
                                   "DirectControlClientID should have expected value")
        self.verifyEqualOrFailCase(self.muse_account_id,
                                   self.zp.AVTransport.get_state_variable_and_time("DirectControlAccountID")[0],
                                   "DirectControlAccountID should have expected value")

    def _play_and_verify_avt_restored(self, zp=None, connection=None):
        """
        SWPBL-70169
        Verify that the previous AVT source can be restored and played after a second 
        upnp play is sent after a session has been suspended
        :return: 
        """
        if zp is None:
            zp = self.zp
        # Verify that second play restores the AVT
        with IgnoreInvokeErrors(zp):
            zp.AVTransport.Play_and_wait(timeout_seconds=10)
        self.verify_event(PlaybackStatusEvent, timeout=1,
                          event_verifier=lambda e: e.playback_state == PlaybackState.PLAYING,
                          connection=connection)
        self.verify_event(SessionErrorEvent, timeout=1, past_events=True,
                          event_verifier=lambda e: (e.error_code == "ERROR_SESSION_EVICTED"),
                          connection=connection)
        wait_until_true(lambda: zp.AVTransport.get_avtransport_uri() == self.avt,
                        reason="Waiting for AVT to revert after second play command",
                        timeout_seconds=10, iteration_delay=1)

        """
        Verify that when the player suspends, the following events occur:
        1. previous item data reflects when the suspend occurred while a track is playing
        2. current itemId is None
        3. The /timePlayed report is accurate to when the suspend occurred
        :return:
        """
        self.zp.diag.set_diag_level("tpm", 5)
        self.muse.load_cloud_queue(session_id=self.muse.session_id,
                                   hhid=self.zp.muse_hhid,
                                   queue_base_url=self.cq_control.CQ_PLAYBACK_URI,
                                   httpAuthorization=None,
                                   play_on_completion=True,
                                   track_metadata=None)
        self.verify_event(PlaybackStatusEvent, timeout=1,
                          event_verifier=lambda e: e.playback_state == PlaybackState.PLAYING)
        wait_until_true(lambda: self.zp.AVTransport.GetPositionInfo()["RelTime"] >= "0:00:03",
                        start_delay=2, timeout_seconds=5, iteration_delay=.5)
        self.cq_control.clear_log()
        self.muse.suspend_session(hhid=self.zp.muse_hhid, sessionId=self.muse.session_id)
        self.verify_event(PlaybackStatusEvent, timeout=1,
                          event_verifier=lambda e: (e.playback_state == PlaybackState.IDLE and

                                                    e.previousPositionMillis >= 3000 and
                                                    e.item_id is None))
        log = self.cq_control.wait_for_filtered_log(fn=lambda l: l['path'] == '/timePlayed')
        self.verifyGreaterEqualOrFailCase(len(log), 1,
                                          "Should have one /timePlayed report after suspend")
        self.verifyAlmostEqualOrFailCase(3000, log[0]["body"]["items"][0]["durationPlayedMillis"],
                                         "durationPlayedMillis should be 3 seconds", delta=2000)
        self.verifyAlmostEqualOrFailCase(3000, log[0]["body"]["items"][0]["positionMillis"],
                                         "positionMillis should be 3 seconds", delta=2000)
        self.zp.diag.set_diag_level("tpm", 1)

    def test_verify_upnp_play_after_suspend(self, version):
        """
        Verify that "suspended=false" after a PLAY is sent to the player and that the session
        will be evicted and AVT reverted after a second play command
        :return:

        self._restart_cq_server(endpoint_version=version)
        self._load_play_and_suspend()

            # SWPBL-67824

            # an actual track list

        self.zp.AVTransport.Play()
        self.verify_event(SessionInfoEvent, timeout=1,

        # the zp should not be playing
        wait_until_true(lambda: not self.zp.AVTransport.is_playing(), timeout_seconds=10,
                            iteration_delay=1)
        # Verify correct upnp events
        wait_until_true(lambda: self.zp.AVTransport.get_state_variable_and_time("DirectControlIsSuspended")[0] == "0",
                            timeout_seconds=10, iteration_delay=1, reason="Waiting for DirectControlIsSuspended==0")
        self.verifyEqualOrStop(self.__class__.__name__,
                                   self.zp.AVTransport.get_state_variable_and_time("DirectControlClientID")[0],
                                   "DirectControlClientID should have expected value")
        self.verifyEqualOrFailCase(0, self.zp.AVTransport.get_state_variable_and_time("CurrentTrack")[0],
                                       "CurrentTrack should have expected value")

        # A second play command within 5 seconds of the first one is ignored (SWPBL-72370)
        sleep(6)

        # Verify second play restores the AVT
        self._play_and_verify_avt_restored()
