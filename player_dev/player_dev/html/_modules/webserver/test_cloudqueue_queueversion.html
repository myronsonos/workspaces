    def _verify_query_params(self, log, query_param, expected_results, is_post=False):
        if log:
            if not is_post:
                for key in log:
                    path = key['path']
                    try:
                        param_value = key['queryParams'][query_param]
                        if not (param_value == '' and path == '/context'):
                            if path == '/itemWindow':
                                self.verifyTrueOrFailCase((expected_results[0] == param_value or expected_results[1] == param_value),
                                                          "Verify expected {} for {} returned - {}".format(query_param, path, param_value))
                            else:
                                self.verifyTrueOrFailCase(expected_results in param_value,
                                                          "Verify expected {} for {} returned - {}".format(query_param, path, param_value))
                        else:
                            self.logger.warning("No {} found for {}".format(query_param, path))
                    except KeyError:
                        self.logger.warning("Unable to verify {}. Not found in {} log entry.".format(query_param, path))
            else:
                for key in log:
                    path = key['path']
                    try:
                        param_value = key['body']['items'][0][query_param]
                        self.verifyTrueOrFailCase(expected_results in param_value,
                                                      "Verify expected {} for {} returned - {}".format(query_param, path, param_value))
                    except KeyError:
                        try:
                            param_value = key['body'][query_param]
                            self.verifyTrueOrFailCase(expected_results in param_value,
                                                          "Verify expected {} for {} returned - {}".format(query_param, path,
                                                                                                    param_value))
                        except KeyError:
                            self.logger.warning("Unable to verify {}. Not found in {} log entry.".format(query_param, path))
        else:
            self.logger.error("Unable to verify {}".format(query_param))

    def _verify_context_and_queue_version(self, path, queue_result, context_result, is_post=False):
        # Verify contextVersion and queueVersion

            self._verify_query_params(log, 'queueVersion', expected_results=queue_result, is_post=True)
            self._verify_query_params(log, 'contextVersion', expected_results=context_result, is_post=True)
        else:
            self._verify_query_params(log, query_param='queueVersion', expected_results=queue_result)
            self._verify_query_params(log, query_param='contextVersion', expected_results=context_result)

[docs]    def test_loadcloudqueue_with_queue_version_and_metadata(self):
        """
        Test that passing a queueVersion parameter on loadCloudQueue with
        metadata results in the first playbackStatus event generated returning
        the queueVersion.

        After the itemWindow request, the player should update the queueVersion.

        See https://confluence.sonos.com/x/uFZPB
        """
        # Generate a random string for queue version
        queue_version = cq_helper.generate_random_queueversion_string()
        self.logger.debug("Generated queueVersion string <{}>".
            format(queue_version))

        # Clear the API log
        self.cq_control.clear_log('api')

        # Generate some metadata
        md = self.cq_control.get_track_metadata("0")

        # Load a cloudqueue and specify the queueversion string
        self.muse.load_cloud_queue(session_id=self.muse.session_id,
            hhid=self.zp.muse_hhid,
            queue_base_url=self.cq_control.CQ_PLAYBACK_URI,
            content_type=None, httpAuthorization=None,
            item_id="0",
            play_on_completion=False,
            track_metadata=md,
            queue_version=queue_version)

        # Verify the playbackStatus event triggered after loadCQ has
        # the correct metadata
        self.verify_event(PlaybackStatusEvent, timeout=1,
                          event_verifier=lambda e: (e.body.queueVersion == queue_version))

        # Wait for the player to make itemWindow request and get the
        # log from the Mock CQ server.
        log = self.cq_control.wait_for_filtered_log(fn=lambda l: l['path'] == '/itemWindow')[0]

        # Verify the queueVersion in the itemWindow fetch matches the
        # queueVersion passed to loadCQ
        self.verifyEqualOrFailCase(queue_version, log['queryParams']['queueVersion'],
            "<{}> queueVersion in itemWindow fetch matches expected?".format(log))

        # Verify that the queue version is updated after the itemWindow request
        self.verify_event(PlaybackStatusEvent, timeout=1,
                          event_verifier=lambda e: (e.body.queueVersion != queue_version and
                                                    e.body.queueVersion is not None))

        # Get the new playbackStatus so we can extract the new queueVersion from it
        status = self.muse.get_playback_status(session_id=self.muse.session_id,
                                               group_id=self.gid, hhid=self.zp.muse_hhid)

        # Clear the API log to get rid of the previous event, then refresh
        # CloudQueue to force the player to fetch a new itemWindow event.
        self.cq_control.clear_log('api')
        self.muse.refresh_cloud_queue(session_id=self.muse.session_id, hhid=self.zp.muse_hhid)

        # Wait for the player to make a subsequent itemWindow request
        # with the new queueVersion
        log = self.cq_control.wait_for_filtered_log(fn=lambda l: l['path'] == '/itemWindow')[0]

        # Verify the player used the new queueVersion in the request
        self.verifyEqualOrFailCase(status.body.queueVersion,
            log['queryParams']['queueVersion'],
            "<{}> queueVersion in itemWindow fetch updated?".format(log))

        '''
        Verify that the expected playback_id is present in HTTP headers
        when playback is moved from one player to another in the household
        :return:
        '''
        # Verify that there are at least 2 players in the household
        unique_device_collection = {}
        for zp in self.my_devices:
            if zp.is_playback_device() and zp.modelNumber not in unique_device_collection:
                unique_device_collection[zp.modelNumber] = zp
                continue
        self.non_duplicate_players = unique_device_collection.values()
        self.verifyTrueOrStop(len(self.non_duplicate_players) > 1,
                              "Verify that there are at least 2 players in the household")
        for player in self.non_duplicate_players:
            if player == self.zp:
                self.non_duplicate_players.remove(player)
        player_b = self.non_duplicate_players[0]

        #load cloud queue and begin playback on player A
        self.muse.load_cloud_queue(session_id=self.muse.session_id,
                                   hhid=self.zp.muse_hhid,
                                   queue_base_url=self.cq_control.CQ_PLAYBACK_URI,
                                   content_type=None,
                                   httpAuthorization=None,
                                   play_on_completion=True,
                                   track_metadata=None,
                                   item_id='0')
        self.verify_event(PlaybackStatusEvent, timeout=1,
                          event_verifier=lambda e: e.playback_state == PlaybackState.PLAYING)

        self.cq_control.wait_for_filtered_log(playback_id=self.zp.GroupManagement.get_local_group_uuid(), clear=True)
        #Move playback to player B in the household, and then back to player A
        self.zp.AVTransport.link_zones([player_b])
        self.cq_control.wait_for_filtered_log(playback_id=self.zp.GroupManagement.get_local_group_uuid(), clear=True)
        self.zp.AVTransport.Next()
        self.zp.AVTransport.BecomeCoordinatorOfStandaloneGroup()
        player_b.AVTransport.Next()
        self.cq_control.wait_for_filtered_log(playback_id=player_b.GroupManagement.get_local_group_uuid(), clear=True)
        player_b.AVTransport.link_zones([self.zp])
        player_b.AVTransport.BecomeCoordinatorOfStandaloneGroup()
        self.zp.AVTransport.Next()
        self.cq_control.wait_for_filtered_log(playback_id=self.zp.GroupManagement.get_local_group_uuid(), clear=True)
