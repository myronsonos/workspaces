[docs]    def test_loadTrackList(self):
        """
        Verify that various bits of metadata that is fed in via the track list is reflected
        in the muse metadata event after the load
        :return:
        """
        self.muse.load_trackList(hhid=self.zp.muse_hhid, groupId=self.gid,
                                 tracks=self.track_list,
                                 containerMetadata=container_metadata,
                                 playOnCompletion=True)
        # Verify that playback starts at the beginning of the track
        self.verify_event(PlaybackStatusEvent, timeout=1,
                          event_verifier=lambda e: (e.playback_state == PlaybackState.PLAYING and
                                                    e.position_millis < 3000))
        # Verify current and next item track names show up
        self.verify_event(MetadataStatusEvent, timeout=1, past_events=True,
                          event_verifier=lambda e: (e.current_item is not None and
                                                    e.current_item["track"]["name"] == "SMAPI_TRACK" and
                                                    e.next_item is not None and
                                                    e.next_item["track"]["name"] == "CQ_TRACK"))
        # Verify that the queue has 2 tracks
        self._verify_tracks_in_shared_queue()

        # Verify that seeking works
        self.muse.playback_seek(hhid=self.zp.muse_hhid, groupId=self.gid,
                                positionMillis=15000)
        self.verify_event(PlaybackStatusEvent, timeout=1,
                          event_verifier=lambda e: (e.playback_state == PlaybackState.PLAYING and
                                                    e.position_millis >= 15000))

        # Verify that skipping works and that current item metadata is updated with the new track
        self.muse.skip_to_next(hhid=self.zp.muse_hhid, group_id=self.gid)
        self.verify_event(PlaybackStatusEvent, timeout=1,
                          event_verifier=lambda e: (e.playback_state == PlaybackState.PLAYING and
                                                    e.position_millis < 3000 and
                                                    e.previousPositionMillis > 15000))
        self.verify_event(MetadataStatusEvent, timeout=1, past_events=True,
                          event_verifier=lambda e: (e.current_item is not None and
                                                    e.current_item["track"]["name"] == "CQ_TRACK" and
                                                    e.current_item["track"]["durationMillis"] == 30000 and
                                                    e.current_item["track"]["tags"][0] == "TAG_EXPLICIT"))



[docs]    def test_load_large_tracklist(self):
        """
        Verify that we're able to load a track list > 20 in size, which is the CQ limit
        :return:
        """
        # Add a bunch of tracks to the track list
        track = self.track_list[0]
        for _ in range(0,25):
            self.track_list.append(track)

        # Load and verify the size of the shared queue
        self.muse.load_trackList(hhid=self.zp.muse_hhid, groupId=self.gid,
                                 tracks=self.track_list,
                                 containerMetadata=container_metadata,
                                 playOnCompletion=False)
        self.verify_event(PlaybackStatusEvent, timeout=1,
                          event_verifier=lambda e: (e.playback_state == PlaybackState.IDLE))
        self._verify_tracks_in_shared_queue(expected_num_tracks="27")

        # Verify playback starts
        self.muse.play(hhid=self.zp.muse_hhid, group_id=self.gid)
        self.verify_event(PlaybackStatusEvent, timeout=1,
                          event_verifier=lambda e: (e.playback_state == PlaybackState.PLAYING))

        # Verify we are able to skip to the last track in the queue and playback continues
        self.zp.AVTransport.Seek(unit="TRACK_NR", target="27")
        wait_until_true(lambda: self.zp.AVTransport.GetPositionInfo()["Track"] == "27",
                        timeout_seconds=5,
                        iteration_delay=1,
                        reason="Current track: {}".format(self.zp.AVTransport.GetPositionInfo()["Track"]))
        self.verify_event(PlaybackStatusEvent, timeout=1,
                          event_verifier=lambda e: (e.playback_state == PlaybackState.PLAYING))



[docs]    def test_loadTrackList_contextId(self):
        """
        Verify that SMAPI tpm calls report the correct contextId, which should the smapi track id
        since it is loaded a single track not a container
        :return:
        """
        # Load and play the smapi track for 5 seconds
        self.muse.load_trackList(hhid=self.zp.muse_hhid, groupId=self.gid,
                                 tracks=self.track_list,
                                 containerMetadata=container_metadata,
                                 playOnCompletion=True)
        self.verify_event(PlaybackStatusEvent, timeout=1,
                          event_verifier=lambda e: (e.playback_state == PlaybackState.PLAYING))
        wait_until_true(lambda: self.zp.AVTransport.GetPositionInfo()["RelTime"] >= "0:00:05",
                        iteration_delay=1, timeout_seconds=10,
                        reason="Current track position: {}".format(self.zp.AVTransport.GetPositionInfo()["RelTime"]))
        # Skip to generate usage reports and clear the log
        self.muse.skip_to_next(hhid=self.zp.muse_hhid, group_id=self.gid)
        smapi_ctl.clear_smapi_server_logs()

        # Verify the setPlaySeconds report has the correct contextId value
        log_values, time_taken_secs = smapi_ctl.wait_for_filtered_log(smapi=smapi_ctl, timeout=30,
                                                                      fn=lambda l: (l["args"]["contextId"] == "1" and
                                                                                    l["args"]["seconds"] >= 4),
                                                                      item_id="1")
        self.verifyGreaterEqualOrFailCase(len(log_values), 1,
                                          "Expected 1 setPlayedSeconds reported, got {}".format(len(log_values)))
        self.verifyEqualOrFailCase('1', log_values[0]["args"]["contextId"],
                                   "setPlayedSeconds contextId should be 1, got {}".format(log_values[0]["args"]["contextId"]))

        # Verify the reportPlayStatus report has the correct contextId value
        log_values, time_taken_secs = smapi_ctl.wait_for_filtered_log(smapi=smapi_ctl, timeout=30,
                                                                      calls=["reportPlayStatus"],
                                                                      fn=lambda l: (l["args"]["contextId"] == "1" and
                                                                                    l["args"]["status"] == "skippedTrack"),
                                                                      item_id="1")
        self.verifyGreaterEqualOrFailCase(len(log_values), 1,
                                          "Expected 1 reportPlayStatus reported, got {}".format(len(log_values)))
        self.verifyEqualOrFailCase('1', log_values[0]["args"]["contextId"],
                                   "reportPlayStatus contextId should be 1, got {}".format(log_values[0]["args"]["contextId"]))

    def test_loadTrackList_track_id_fuzzer(self, param):
        """
        The smapi track should play with either a valid serviceId or accountId but
        should fail if the objectId is missing
        :param param:
        :return:
        """
        # Remove one of the track.id params
        smapi_track = [self.track_list[0]]
        del smapi_track[0]["track"]["id"][param]

        # Load and verify error or successful playback depending on which track.id param was removed
        self.muse.load_trackList(hhid=self.zp.muse_hhid, groupId=self.gid,
                                 tracks=smapi_track,
                                 containerMetadata=container_metadata,
                                 playOnCompletion=True)
        if param == "objectId":
            self.verify_event(Error, timeout=1,
                              event_verifier=lambda e: (e.error_code == "ERROR_MISSING_PARAMETERS" and
                                                        param in e.reason))
        else:
            self.verify_event(PlaybackStatusEvent, timeout=1,
                              event_verifier=lambda e: (e.playback_state == PlaybackState.PLAYING))

    def _verify_tracks_in_shared_queue(self, expected_num_tracks="2"):
        """
        Helper to verify the number of tracks in the shared queue
        :param expected_num_tracks:
        :param zp:
        :return:
        """
        wait_until_true(lambda: self.zp.Queue.Browse(queue_id=0,
                                                     starting_index=0,
                                                     requested_count="")["NumberReturned"] == expected_num_tracks,
                        iteration_delay=1,
                        timeout_seconds=5,
                        reason="Shared queue returned {} items".format(self.zp.Queue.Browse(queue_id=0,
                                                                                            starting_index=0,
                                                                                            requested_count="")
                                                                       ["NumberReturned"]))
