    def setup_media_server(self, version=2.1):
        """
        Setup for media error by restarting CQ. Return the CQ token.

        :return: auth token
        :rtype: :obj:`str`
        """
        # Add hosts entry
        host = "{}     {}     {}\n".format(self.cq_control.host_ip,
                                           "mock.dev.ws.sonos.com",
                                           "mock.dev.ws.sonos.com")
        self.zp.cli.command("echo '{}' > /etc/hosts".format(ETC_HOSTS + host))
        try:
            # New setup steps needed after changes from SWPBL-81671
            self.stop_cloud_queue_server()
            self.start_cloud_queue_server(authTokenExpiration="1",
                                          authToken='12345',
                                          proto="https",
                                          endpoint_version=version)
            self.md0 = self.cq_control.get_track_metadata("0")
            self.host = self.cq_control.host_ip + ":" + str(self.cq_control.host_port)
            self.muse_session_helper.make_muse_session(self.muse, group_id=self.gid, hhid=self.zp.muse_hhid, test_fixture=self)
            self.muse.subscribe_playback(group_id=self.gid, hhid=self.zp.muse_hhid)
            self.verify_event(PlaybackStatusEvent, timeout=1,
                              event_verifier=lambda e: (e.playback_state == "PLAYBACK_STATE_IDLE"))
            self.cq_control.set_cloudqueue_poll_timers(zp=self.zp, playing_timer=3, paused_timer=3)

            self.cq_control.expire_tokens()
            yield 'playon=' + self.cq_control.get_current_auth_token()
        # Cleanup
        finally:
            self.zp.cli.command("echo '{}' > /etc/hosts".format(ETC_HOSTS))

    def test_media_error_on_load_from_idle_with_md(self, http_error, cq):
        """
        Verifies player behavior when it seems a http error coming from a media fetch
        404 - Playback error, no eviction
        Everything else - Playback error, eviction
        SWPBL-86272 added combinatorial to make sure that the player only evicts if using
        cq v1.0 and getting 401/403 media errors
        :param http_error:
        :return:
        """
        self.stop_cloud_queue_server()
        self.start_cloud_queue_server(endpoint_version=cq)
        self.cq_control.arm_error_script(call="mediaError", error=http_error)
        self.muse.load_cloud_queue(session_id=self.muse.session_id,
                                   hhid=self.zp.muse_hhid,
                                   queue_base_url=self.cq_control.CQ_PLAYBACK_URI,
                                   httpAuthorization=None,
                                   item_id="0",
                                   play_on_completion=True,
                                   track_metadata=self.md0)
        self.verify_event(PlaybackErrorEvent, timeout=1,
                          event_verifier=lambda e: (e.http_status == http_error and
                                                    e.host == self.host and
                                                    e.hostIp in e.host and
                                                    e.item_id == "0"))

        if http_error in (401,403) and cq == 1.0:
            self.verify_event(SessionErrorEvent, timeout=1,
                              event_verifier=lambda e: (e.error_code == "ERROR_SESSION_EVICTED"), past_events=True)
        else:
            self.verify_event(SessionErrorEvent, expect_event=False, timeout=1, past_events=True, iterations=3,
                              event_verifier=lambda e: (e.error_code == "ERROR_SESSION_EVICTED"))


    def _req_res_testflow(self, request, response):
        """
        Verifies player behavior to specific requests responses from the CQ server
        401/403/404 - playback error, session eviction except 404/itemWindow which results in player going IDLE
        503, anything else - Player retries and continues playing
        :param req:
        :param res:
        :return:
        """
        wait_until_true(lambda: self.cq_control.wait_for_filtered_log(fn=lambda l: (l['path'] == '/{}'.format(request)),
                                                                      timeout=5),
                        timeout_seconds=15,
                        iteration_delay=1)

        self.logger.debug("Server responded to {} request with {}".format(request, response))
        if request == 'itemWindow' and response == 404:
            self.verify_event(PlaybackStatusEvent, timeout=1,
                              event_verifier=lambda e: (e.playback_state == PlaybackState.IDLE))
            self.verify_event(SessionErrorEvent, past_events=True, timeout=1, expect_event=False,
                              event_verifier=lambda e: (e.error_code == "ERROR_SESSION_EVICTED"))
        elif response in (401, 403, 404):
            self.verify_event(PlaybackErrorEvent, timeout=1,
                              event_verifier=lambda e: (e.http_status == response))
            self.verify_event(SessionErrorEvent, timeout=1, past_events=True,
                              event_verifier=lambda e: (e.error_code == "ERROR_SESSION_EVICTED"))
        # SWPBL-61327
        else:
            self.verify_event(SessionErrorEvent, past_events=True, timeout=1, expect_event=False, iterations=3,
                              event_verifier=lambda e: (e.error_code == "ERROR_SESSION_EVICTED"))
            # If not playing, start playback and verify that the player can continue to play after the retry
            if not self.zp.AVTransport.is_playing():
                self.muse.play(group_id=self.gid, hhid=self.zp.muse_hhid)
            wait_until_true(lambda: self.zp.AVTransport.is_playing(),
                            timeout_seconds=10, iteration_delay=1,
                            reason="ZP should be playing after a {} res to {} req retry".format(response, request))
            wait_until_true(lambda: self.cq_control.wait_for_filtered_log(fn=lambda l: (l['path'] == '/{}'.format(request)),
                                                                          timeout=5,
                                                                          expected=2),
                            timeout_seconds=15,
                            iteration_delay=1)

        """
        Verifies player behavior when it seems 4 503 errors, which exceeds the retry limit, while playing.
        Changed with SWPBL-93610:
        Current behavior is that the player continues to play out the current song and then silently stop.
        :return:
        """
        self.muse.load_cloud_queue(session_id=self.muse.session_id,
                                   hhid=self.zp.muse_hhid,
                                   queue_base_url=self.cq_control.CQ_PLAYBACK_URI,
                                   httpAuthorization=None,
                                   play_on_completion=True,
                                   track_metadata=None,
                                   item_id="2",
                                   position_millis=20000)
        self.verify_event(PlaybackStatusEvent, timeout=1,
                          event_verifier=lambda e: (e.playback_state==PlaybackState.PLAYING))
        wait_until_true(lambda: self.cq_control.wait_for_filtered_log(fn=lambda l: (l['path'] == '/itemWindow'),
                                                                      timeout=5),
                        timeout_seconds=15,
                        iteration_delay=1)
        self.cq_control.clear_log()

        # The player should finish playing out the current track once the retries are exhausted.
        # No error or eviction should be sent.
        self.cq_control.arm_error_script(call="itemWindow",
                                         error=503, num_errors=4,
                                         retry_after_time=1)
        self.muse.refresh_cloud_queue(session_id=self.muse.session_id, hhid=self.zp.muse_hhid)
        wait_until_true(lambda: self.cq_control.wait_for_filtered_log(fn=lambda l: (l['path'] == '/itemWindow'),
                                                                      timeout=5, expected=4),
                        timeout_seconds=15,
                        iteration_delay=1)
        self.verify_event(PlaybackErrorEvent, timeout=1, iterations=3, expect_event=False, past_events=True)
        self.verify_event(SessionErrorEvent, timeout=1, iterations=3, expect_event=False, past_events=True,
                          event_verifier=lambda e: (e.error_code == "ERROR_SESSION_EVICTED"))
        self.cq_control.clear_log()
        self.verify_event(PlaybackStatusEvent, timeout=2, past_events=True,
                          event_verifier=lambda e: (e.playback_state==PlaybackState.PAUSED and
                                                    e.item_id == "2" and
                                                    e.position_millis == 30000))

        # Verify that a refresh cloudqueue will not reset the FSM
        self.muse.refresh_cloud_queue(session_id=self.muse.session_id, hhid=self.zp.muse_hhid)
        self.verify_event(Error, timeout=1,
                          event_verifier=lambda e: (e.error_code == "ERROR_COMMAND_FAILED"))

        # Playback can be restarted with a skipToItem/loadCloudqueue
        self.muse.skip_to_item(hhid=self.zp.muse_hhid, session_id=self.muse.session_id,
                               item_id="0", track_metadata=None, play_on_completion=True)
        self.verify_event(PlaybackStatusEvent, timeout=1,
                          event_verifier=lambda e: (e.playback_state == PlaybackState.PLAYING and
                                                    e.item_id == "0"))

        """
        Verifies player behavior to make sure the 503 retry counter is reset after a successful fetch

        This test originally used Pause() commands to force the player to send out a version check request
        to the CQ /version endpoint. The bug fix for SWPBL-68185 tweaked the player enforcement of network calls
        when there isn't enough playback data buffered such that these pauses caused this test to fail. This test
        now progresses naturally through the items in the CQ and runs reliably without the extra pauses.

        See CL 295386 for all affected test changes based on this bugfix.
        """
        self.muse.load_cloud_queue(session_id=self.muse.session_id,

                                         queue_base_url=self.cq_control.CQ_PLAYBACK_URI,
                                         httpAuthorization=None,
                                         play_on_completion=True,
                                         track_metadata=None)
        self.verify_event(PlaybackStatusEvent, timeout=1,
                          event_verifier=lambda e: (e.playback_state==PlaybackState.PLAYING))
        for x in (1,2):
            self.cq_control.arm_error_script(call="version", error=503, num_errors=3, retry_after_time=0)
            self.cq_control.clear_log()
            filtered_logs = self.cq_control.wait_for_filtered_log(type='api',
                                                                  fn=lambda l: l['path'] == '/version',
                                                                  timeout=15,
                                                                  expected=4)

            self.verifyGreaterEqualOrFailCase(len(filtered_logs[0]), 4,
                                              "Loop {}: Expected 4 version requests".format(x))
        self.verifyTrueOrFailCase(self.zp.AVTransport.is_playing(), "Player should still be in the playing state")
