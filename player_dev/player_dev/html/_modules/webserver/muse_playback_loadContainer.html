    def _autoplay_test_flow(self, action, tracks, autoplay):
        """
        Common test code for the following 3 tests.
        :param action:
        :param tracks:
        :param autoplay:
        :return:
        """
        # Get the play mode state, map upnp strings to muse strings
        playbackState = self.zp.AVTransport.get_transport_state()
        if playbackState == "PAUSED_PLAYBACK":
            playbackState = "PAUSED"
        elif playbackState == "STOPPED":
            playbackState = "IDLE"

        # Load and verify the number of tracks in the queue
        self.clear_message_queue()
        self.muse.load_container(hhid=self.zp.muse_hhid, groupId=self.gid,
                                 containerId=self.containerId,
                                 containerMetadata=container_metadata,
                                 playOnCompletion=autoplay,
                                 action=action)
        self._verify_tracks_in_shared_queue(expected_num_tracks=tracks)

        # Verify the current play state based on the autoplay value
        # True should always play, false should always pause, null doesn't change current state
        self.muse.playback.get_playback_status(group_id=self.gid, hhid=self.zp.muse_hhid)
        if autoplay:
            self.verify_event(PlaybackStatusEvent, timeout=1,
                              event_verifier=lambda e: (e.playback_state == PlaybackState.PLAYING))
        elif autoplay is None:
            self.verify_event(PlaybackStatusEvent, timeout=1,
                              event_verifier=lambda e: (e.playback_state == "PLAYBACK_STATE_{}".format(playbackState)))
        elif not autoplay and playbackState is not "IDLE":
            self.verify_event(PlaybackStatusEvent, timeout=1,
                              event_verifier=lambda e: (e.playback_state == PlaybackState.PAUSED))
        else:
            self.verify_event(PlaybackStatusEvent, timeout=1,
                              event_verifier=lambda e: (e.playback_state == PlaybackState.IDLE))

        # For "REPLACE" or "INSERT", the playhead should be on the first smapi track
        self.muse.metadata.subscribe(group_id=self.gid, hhid=self.zp.muse_hhid)
        if action == "REPLACE" or action == "INSERT":
            self.verify_event(MetadataStatusEvent, timeout=1,
                          event_verifier=lambda e: (e.current_item["track"]["name"] == self.smapi_track_list[0] and
                                                    "service" in e.current_item["track"] and
                                                    e.current_item["track"]["service"]["name"] == "node-js SMAPI server" and
                                                    e.container["name"] == "TEST_CONTAINER",
                                                    e.container["type"] == "station"))
        # For "APPEND", the playhead should still be on the first webserver track
        elif action == "APPEND":
            self.verify_event(MetadataStatusEvent, timeout=1,
                              event_verifier=lambda e: (e.current_item["track"]["name"] == "First Song"))
        # For "INSERT_NEXT", the playhead should be on the first webserver track and the next track should
        # be the first smapi track
        else:
            self.verify_event(MetadataStatusEvent, timeout=1,
                              event_verifier=lambda e: (e.current_item["track"]["name"] == "First Song" and
                                                        e.next_item["track"]["name"] == self.smapi_track_list[0]))

    def test_loadContainer_playModes_enable_disable(self, playMode, state):
        """
        Verify that each playmode can be enabled/disabled with loadContainer
        :param playMode:
        :param state:
        :return:
        """
        self.muse.load_container(hhid=self.zp.muse_hhid, groupId=self.gid,
                                 containerId=self.containerId,
                                 containerMetadata=container_metadata,
                                 playModes={playMode: state})
        self.verify_event(PlaybackStatusEvent, timeout=1,
                          event_verifier=lambda e: (e.playModes[playMode] == state and
                                                    e.playback_state == PlaybackState.PLAYING))

        # Change the play state and load again with the playmode set to null.
        # This should not change the playmode condition
        self.zp.AVTransport.clear_queue()
        self.verify_event(PlaybackStatusEvent, timeout=1,
                          event_verifier=lambda e: (e.playback_state == PlaybackState.IDLE))
        self.muse.load_container(hhid=self.zp.muse_hhid, groupId=self.gid,
                                 containerId=self.containerId,
                                 containerMetadata=container_metadata,
                                 playModes={playMode: None})
        self.verify_event(PlaybackStatusEvent, timeout=1,
                          event_verifier=lambda e: (e.playModes[playMode] == state and
                                                    e.playback_state == PlaybackState.IDLE))

        # Load once more with the playmode False and verify that the playmode is indeed false
        self.muse.load_container(hhid=self.zp.muse_hhid, groupId=self.gid,
                                 containerId=self.containerId,
                                 containerMetadata=container_metadata,
                                 playModes={playMode: False},
                                 playOnCompletion=True)
        self.verify_event(PlaybackStatusEvent, timeout=1,
                          event_verifier=lambda e: (e.playModes[playMode] == False and
                                                    e.playback_state == PlaybackState.PLAYING))

        # Finally load with an invalid playmode condition and verify the error response
        self.muse.load_container(hhid=self.zp.muse_hhid, groupId=self.gid,
                                 containerId=self.containerId,
                                 containerMetadata=container_metadata,
                                 playModes={playMode: "TEST_STR"})
        self.verify_event(Error, timeout=1,
                          event_verifier=lambda e: (e.error_code == "ERROR_INVALID_PARAMETER" and
                                                    playMode in e.reason))


    def _shuffle_test_flow(self, action, tracks, shuffle):
        """
        Test flow for the following two test cases
        :param action:
        :param tracks:
        :param shuffle:
        :return:
        """
        # Get the play mode state, map upnp paused string to muse paused string
        playbackState = self.zp.AVTransport.get_transport_state()
        if playbackState == "PAUSED_PLAYBACK":
            playbackState = "PAUSED"

        # Remove the second webserver track for ease of testing
        self.zp.Queue.RemoveTrackRange(queueid=0,
                                       updateid=0,
                                       startingindex=2,
                                       numberoftracks=1)
        self._verify_tracks_in_shared_queue(expected_num_tracks="1")
        self.verify_event(PlaybackStatusEvent, timeout=1,
                          event_verifier=lambda e: (e.playback_state == "PLAYBACK_STATE_{}".format(playbackState)))

        # Load the container with the provided shuffle mode state and action
        self.clear_message_queue()
        self.muse.load_container(hhid=self.zp.muse_hhid, groupId=self.gid,
                                 containerId=self.containerId,
                                 containerMetadata=container_metadata,
                                 playModes={"shuffle": shuffle},
                                 action=action)
        self.muse.playback.get_playback_status(group_id=self.gid, hhid=self.zp.muse_hhid)
        self.verify_event(PlaybackStatusEvent, timeout=1,
                          event_verifier=lambda e: (e.playback_state == "PLAYBACK_STATE_{}".format(playbackState)))
        self._verify_tracks_in_shared_queue(expected_num_tracks=tracks)

        # We want the playhead to be on first smapi track so we should skipToNext if the action is
        # "APPEND" or "INSERT_NEXT"
        if action == "APPEND" or action == "INSERT_NEXT":
            self.muse.skip_to_next(group_id=self.gid, hhid=self.zp.muse_hhid)
            self.verify_event(PlaybackStatusEvent, timeout=1,
                              event_verifier=lambda e: (e.playback_state == "PLAYBACK_STATE_{}".format(playbackState)))
            wait_until_true(lambda: self.zp.AVTransport.GetPositionInfo()["Track"] == "2",
                            iteration_delay=1,
                            timeout_seconds=5)

        # Get the metadata event and parse it for the currentItem and nextItem names
        self.muse.metadata.subscribe(group_id=self.gid, hhid=self.zp.muse_hhid)
        self.verify_event(MetadataStatusEvent, timeout=1,
                          event_verifier=lambda e: ("currentItem" in e.body and
                                                    "nextItem" in e.body and
                                                    e.container["name"] == "TEST_CONTAINER"))
        md_track_list = []
        # protection in case the last event in the test_events queue isn't a metadataStatus event
        for event in self.muse.test_events[::-1]:
            if isinstance(event, MetadataStatusEvent):
                for item in ("currentItem", "nextItem"):
                    md_track_list.append(event.body[item]["track"]["name"])
            # we only care about the most recent metadataStatus event so stop after processing it
                break

        if shuffle:
            # If shuffle is on when the smapi container is load, it is certain that the second track will
            # be different from what was it was on the initial server load
            self.verifyNotEqualOrFailCase(self.smapi_track_list, md_track_list,
                                      "The current and next items in the metadata event should differ from the"
                                      "original smapi playlist since shuffle is on")
        else:
            # If shuffle is off, we should expect the currentItem/nextItem order to match the order from
            # when the smapi server was first loaded
            self.verifyEqualOrFailCase(self.smapi_track_list, md_track_list,
                                       "The current and next items in the metadata event should be the same as the "
                                       "original smapi playlist since shuffle is off")

[docs]    def test_loadContainer_missing_parameters(self):
        """
        Cycles through each input param to loadContainer and verify the response event
        """
        # create dict of all the args the command takes
        params = {"hhid": self.zp.muse_hhid, "groupId": self.gid, "containerId": self.containerId,
                  "containerMetadata": container_metadata, "action": "REPLACE",
                  "playModes": default_playModes, "playOnCompletion": True}
        # Define which ones are optional
        optional_params = ("action", "playModes", "playOnCompletion")

        for param in params:
            for value in ("True", 123, None):
                self.logger.debug("Setting {} to {}".format(param, value))
                # make a copy of the params dict so it can be reset after iteration
                new_params = dict(params)
                new_params[param] = value
                resp = self.muse.load_container(**new_params)
                # Sending a null value for an optional param should result in a success response
                if value is None and param in optional_params:
                    self.verifyIsInstanceOrFailCase(resp, Success, "Got success response?")
                elif param == "groupId" and value is not None:
                    self.verifyIsInstanceOrFailCase(resp, GroupCoordinatorChangedEvent,
                                                    "Got GroupCoordinatorChangedEvent response?")
                # All other cases should result in one of these 3 errors
                else:
                    self.verifyIsInstanceOrFailCase(resp, Error, "Got error response?")
                    self.verifyInOrFailCase(resp.body.errorCode,
                                    (ErrorEnum.GLOBAL_MISSING_PARAMETERS.value,
                                     ErrorEnum.GLOBAL_INVALID_OBJECT_ID.value,
                                     ErrorEnum.GLOBAL_INVALID_PARAMETER.value),
                                    "Got expected error code?")
