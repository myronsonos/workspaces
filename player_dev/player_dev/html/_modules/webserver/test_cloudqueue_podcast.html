    def _verifyDIDLLiteTrackMetadata(self, didl_element, track):
        """
        Verifies that the track metadata in a DIDL-Lite element
        matches the corresponding metadata in a Muse track object.
        """

        items = didl_element.getItems()
        for item in items:
            self.verifyEqualOrFailCase("object.item.audioItem.podcast", item.upnp_class,
                                        "UPnP class should be 'object.item.audioItem.podcast'")
            self.verifyEqualOrFailCase(track["name"], item.title,
                                        "UPnP title should match track name")
            self.verifyEqualOrFailCase(track["podcast"]["producer"]["name"], item.creator,
                                        "UPnP creator should match producer name")
            if "releaseDate" in track:
                release_date = parse(track["releaseDate"]).replace(microsecond=0)
                release_date = release_date.strftime("%Y-%m-%dT%H:%M:%SZ")
                self.verifyEqualOrFailCase(release_date, item.releaseDate,
                                            "UPnP releaseDate should match track releaseDate")

    def _verifyDIDLLiteMetadataElementFromXML(self, xml, element, expectedValue):
        """
        Verifies non-standard elements in the DIDL-Lite by parsing the XML
        and comparing with the expected value
        """
        for child in utils.parse_xml(xml).getroot():
            if "DIDL-Lite" in child.tag:
                for subchild in child:
                    if element in subchild.tag:
                        self.verifyEqualOrFailCase(str(expectedValue), subchild.text,
                                                   "Element {} does not match expected value".format(element))
                        break
                else:
                    self.fail("Failed to find element {} in the xml".format(element))

    def _customVerifyDictEqualOrFailCase(self, expected, actual):
        """
        Normalizes the dicts representing the CQ Muse object
        and the Muse object evented from the player. This adjusts
        and corrects certain expected differences in the data.
        """

        if "track" in actual:
            actual = actual["track"]
            if "imageUrl" in actual:
                del actual["imageUrl"]
            if "releaseDate" in actual:
                release_date = parse(actual["releaseDate"]).replace(microsecond=0)
                release_date = release_date.strftime("%Y-%m-%dT%H:%M:%SZ")
                actual["releaseDate"] = release_date

        if "track" in expected:
            expected = expected["track"]
            if "podcast" in expected:
                if "producer" in expected["podcast"]:
                    # Related IDs are not evented in Muse.
                    if "id" in expected["podcast"]["producer"]:
                        del expected["podcast"]["producer"]["id"]
                # Related IDs are not evented in Muse.
                if "id" in expected["podcast"]:
                    del expected["podcast"]["id"]
            # Muse doesn't event policies.
            if "policies" in expected:
                del expected["policies"]
            if "durationMillis" in expected:
                expected['durationMillis'] = int(expected['durationMillis'])
            if "imageUrl" in expected:
                del expected["imageUrl"]
            if "mediaUrl" in expected:
                del expected["mediaUrl"]
            if "contentType" in expected:
                del expected["contentType"]
            if "releaseDate" in expected:
                release_date = parse(expected["releaseDate"]).replace(microsecond=0)
                release_date = release_date.strftime("%Y-%m-%dT%H:%M:%SZ")
                expected["releaseDate"] = release_date

        # SWPBL-74358 - exclude incidental tags from the comparison
        if "tags" not in actual:
            if "tags" in expected:
                del expected["tags"]

        return self.verifyDictEqualOrFailCase(expected, actual,
                                              "Evented metadata should match expected metadata")

    def _get_cloudqueue_container(self, cq_endpoint):
        """
        Get the container metadata from the cloud queue playlist endpoint
        :return metadata: :list: a list of the decoded JSON container metadata set on the CQ playlist
        """
        try:
            resp = requests.get(url=cq_endpoint, verify=False)
            resp.raise_for_status()
            data = json.loads(resp.text, encoding='utf8')
            self.verifyTrueOrStop("context" in data, "context object is required")
            self.verifyTrueOrStop("container" in data["context"], "container object is required")
            return data["context"]["container"]
        except requests.RequestException as e:
            self.logger.warning("Could not get container metadata, server returned {}".format(e))

    def _get_cloudqueue_items(self, cq_endpoint):
        """
        Get the list of items from the cloud queue playlist (in our case podcast) endpoint
        :return metadata: :list: a list of the decoded JSON track items from the playlist
        """
        try:
            resp = requests.get(url=cq_endpoint, verify=False)
            resp.raise_for_status()
            data = json.loads(resp.text, encoding='utf8')
            self.verifyTrueOrStop("items" in data, "items object is required")
            return data["items"]
        except requests.RequestException as e:
            self.logger.warning("Could not get container metadata, server returned {}".format(e))

    def _set_podcast_window_playhead(self, podcastId, resume_position, resume_episode="0"):
        setWindowPlayhead_uri = "{}podcast/{}/windowPlayhead".format(self.cq_control.CQ_PLAYBACK_URI, podcastId)
        resume_data = {'itemId': resume_episode, 'positionMillis': resume_position}
        payload = json.dumps(resume_data)

        try:
            res = requests.post(url=setWindowPlayhead_uri,
                                headers=self.cq_control.POLICY_HEADERS,
                                data=payload, verify=False)
            res.raise_for_status()
        except requests.RequestException as e:
            self.logger.error("Received HTTP {0} error attempting to add track to CQ server".format(e))

    def test_podcast_container_metadata(self, zone):
        """
        Verify that the playbackMetadata Muse container has the expected
        podcast type and the Podcast object metadata from the CQ context.
        Verify that the UPNP event has the correct enqueued URI metadata
        UPNP class and podcast name.
        """

        podcast_endpoint = "{}podcast/{}/".format(self.cq_control.CQ_PLAYBACK_URI, PODCAST_OBJECT_ID)
        container_metadata = self._get_cloudqueue_container(podcast_endpoint)

        self.muse.load_cloud_queue(session_id=self.muse.session_id,
                                  hhid=self.zp.muse_hhid,
                                  queue_base_url=podcast_endpoint,
                                  play_on_completion=True)
        self.verify_event(PlaybackStatusEvent, timeout=2,
                          event_verifier=lambda e: e.playback_state == PlaybackState.PLAYING)

        # Verify evented Muse container metadata matches cloud queue container metadata
        self.muse.metadata.subscribe(group_id=self.gid, hhid=self.zp.muse_hhid)
        self.verify_event(MetadataStatusEvent, timeout=2,
                          event_verifier=lambda e: (self._customVerifyDictEqualOrFailCase(container_metadata, e.container)))

        # Verify EnqueuedTransportURIMetaData UPNP class and title are what is expected for an podcast
        didl_element = DIDLLite.DIDLElement.fromString(self.zp.AVTransport.get_state_variable("EnqueuedTransportURIMetaData"))
        items = didl_element.getItems()
        for item in items:
            self.verifyEqualOrFailCase(item.upnp_class, "object.container.podcast",
                                        "UPnP class should be 'object.container.podcast'")

            self.verifyEqualOrFailCase(item.title, container_metadata["name"],
                                       "DIDL-Lite <dc:title> should match container name")
            self.verifyEqualOrFailCase(item.albumArtURI, container_metadata["imageUrl"],
                                       "DIDL-Lite <upnp:albumArtURI> should match container imageUrl.")

    def test_podcast_track_metadata(self, zone):
        """
        Verify that the current and next item in the playbackMetadata Muse event
        matches the track metadata returned in the cloud queue item window
        Verify that the UPNP event CurrentTrackMetaData and NextTrackMetaData
        contain the expected podcast-related metadata
        """
        podcast_endpoint = "{}podcast/{}/".format(self.cq_control.CQ_PLAYBACK_URI, PODCAST_OBJECT_ID)
        cq_items = self._get_cloudqueue_items(podcast_endpoint)

        self.muse.load_cloud_queue(session_id=self.muse.session_id,
                                   hhid=self.zp.muse_hhid,
                                   queue_base_url=podcast_endpoint,
                                   play_on_completion=True)
        self.verify_event(PlaybackStatusEvent, timeout=2,
                          event_verifier=lambda e: e.playback_state == PlaybackState.PLAYING)

        # Verify current item and next item in the Muse event match the 1st and 2nd items from the CQ item window
        self.muse.metadata.subscribe(group_id=self.gid, hhid=self.zp.muse_hhid)
        self.verify_event(MetadataStatusEvent, timeout=2,
                          event_verifier=lambda e: (self._customVerifyDictEqualOrFailCase(cq_items[0], e.current_item) and
                                                    self._customVerifyDictEqualOrFailCase(cq_items[1], e.next_item)))

        # Verify UPNP CurrentTrackMetaData matches the track metadata from the 1st CQ item
        md1 = cq_items[0]["track"]
        didl_element = DIDLLite.DIDLElement.fromString(self.zp.AVTransport.get_state_variable("CurrentTrackMetaData"))
        self._verifyDIDLLiteTrackMetadata(didl_element, md1)

        # Verify episodeNumber in the UPNP CurrentTrackMetaData DIDL-Lite matches the one in the 1st CQ item
        ''' IX-Content: Revisit in SWPBL-114138
        self._verifyDIDLLiteMetadataElementFromXML(self.zp.AVTransport.get_state_variable("CurrentTrackMetaData"),
                                                    "episodeNumber",
                                                    md1["episodeNumber"])
        '''

        # Verify UPNP NextTrackMetaData matches the track metadata from the 2nd CQ item
        md2 = cq_items[1]["track"]
        didl_element = DIDLLite.DIDLElement.fromString(self.zp.AVTransport.get_state_variable("NextTrackMetaData"))
        self._verifyDIDLLiteTrackMetadata(didl_element, md2)

        # Verify episodeNumber in the UPNP NextTrackMetaData DIDL-Lite matches the one in the 2nd CQ item
        ''' IX-Content: Revisit in SWPBL-114138
        self._verifyDIDLLiteMetadataElementFromXML(self.zp.AVTransport.get_state_variable("NextTrackMetaData"),
                                                    "episodeNum",
                                                    md2["episodeNumber"])
        '''

        # Skip to the next item
        self.muse.skip_to_item(session_id=self.muse.session_id,
                                    hhid=self.zp.muse_hhid,
                                    item_id=cq_items[2]["id"],
                                    play_on_completion=True,
                                    track_metadata=None)

        # Verify current item in the Muse event matches the CQ item to which we skipped
        self.verify_event(MetadataStatusEvent, timeout=2,
                          event_verifier=lambda e: (self._customVerifyDictEqualOrFailCase(cq_items[2], e.current_item)))

    def test_podcast_invalid_resume_position_millis(self, zone):
