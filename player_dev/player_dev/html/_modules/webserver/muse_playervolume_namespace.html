[docs]    def verify_playervolume_state(self,
                                  expected_volume=None,
                                  expected_muted=None):
        """
        if expected_volume is not None, look for an event with that volume value
        if expected_muted is not None, look for an event with that mute state
        if both are provided, look for both
        if both are None, return the first PlayerVolumeEvent
        """
        last_volume = None
        last_muted = None
        self.logger.debug("Verifying player volume state (expected_volume={}), "
            "(expected_muted={})".format(expected_volume, expected_muted))

        for event in self.muse.generate_events(timeout=5):
            self.logger.debug("Muse Event: {}".format(event))
            if isinstance(event, PlayerVolumeEvent):
                last_volume = event.volume
                last_muted = event.muted
                self.logger.debug("PlayerVolumeEvent: (volume={}, muted={})".
                                  format(last_volume, last_muted))
                if ((expected_volume is None
                     or event.volume == expected_volume)
                    and (expected_muted is None
                         or event.muted == expected_muted)):
                    break
        else:
            pass

        if ((expected_volume is None and expected_muted is None)
            or (expected_volume is not None and expected_muted is not None)):
            return last_volume, last_muted
        elif expected_volume is not None:
            return last_volume
        else:
            return last_muted

[docs]    def verify_playerAndGroupVolume_state(self,
                                          expected_group_volume=None,
                                          expected_group_muted=None,
                                          expected_player_volume=None,
                                          expected_player_muted=None):
        last_group_volume = None
        last_group_muted = None
        last_player_volume = None
        last_player_muted = None

        # Store last volume event data
        for event in self.muse.generate_events(timeout=5):
            if isinstance(event, PlayerVolumeEvent):
                last_player_volume = event.volume
                last_player_muted = event.muted
            if isinstance(event, GroupVolumeEvent):
                last_group_volume = event.volume
                last_group_muted = event.muted

        # Verify all provided events
        if expected_group_volume != None:
            self.verifyEqualOrFailCase(expected_group_volume, last_group_volume,
                                       "Expected and actual group volume should match")
        if expected_group_muted != None:
            self.verifyEqualOrFailCase(expected_group_muted, last_group_muted,
                                       "Expected and actual group mute should match")
        if expected_player_volume != None:
            self.verifyEqualOrFailCase(expected_player_volume, last_player_volume,
                                       "Expected and actual player volume should match")
        if expected_player_muted != None:
            self.verifyEqualOrFailCase(expected_player_muted, last_player_muted,
                                       "Expected and actual player mute should match")

        # Log what each player's playerVolume and muted values are when a test case fails
        if self.fail_stop != int(self.verify_stop_count) + int(self.verify_fail_count):
            for zp, muse in self.muses.iteritems():
                playerId = zp.raw_udn
                muse.playerVolume.get_volume(player_id=playerId, hhid=zp.muse_hhid)
                for event in muse.generate_events(timeout=1):
                    if isinstance(event, PlayerVolumeEvent):
                        self.warn("ZP {} - playerVolume: {}, muted: {}".format(zp,
                                                                             event.volume,
                                                                             event.muted))
