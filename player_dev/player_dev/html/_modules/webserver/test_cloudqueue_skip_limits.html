[docs]    def setSkipEnforcementPolicy(self, type=None):
        """
        Convenience function to set different playback policies to mimic partner radios
        :param type: :string: 'DMCA' or CQ Server defaults plus limitedSkips:true, canSkip: true
        """

        if type == 'ad' :
            playback_policies = {"canSkip": False,
                                 "canSkipBack" : False,
                                 "limitedSkips": True,
                                 "canSkipToItem": False}
        else:
            playback_policies = {"canSkip": True,
                                 "limitedSkips": True,
                                 "canSkipBack": False,
                                 "canSkipToItem": True}

        self.cq_control.set_playback_policy(playback_policies)
        expected_playback_policies = self.cq_control.get_playback_policies()
        for policy in playback_policies:
            self.verifyEqualOrFailCase(expected_playback_policies[policy],
                        playback_policies[policy],
                        "For {0}, container {1} policy should be equal to {2}".format(type, policy, playback_policies[policy]))




[docs]    def setLimitedSkipsStateAndLoadCloudQueue(self, itemId="0", remaining=0, maxSkips=6,
                                              ttl=3600, position_millis=None):
        """
        Set the cloud queue server's last report ID, skip limit, and TTL, create a Muse session,
        load a cloud queue, and subscribe to the playback namespace.

        :param remaining: :int: the number of skips the server allows.
        :param maxSkips: :int: the maximum number of skips allowed by this container
        :param restoreAfterSec: :int: time after which the cloudqueue server will restore 'maxSkips'
                                      skips back to the user
        """
        self.cq_control.set_skip_enforcement(remaining, maxSkips, ttl)
        skip_enforcement = self.cq_control.get_skip_enforcement()
        self.verifyEqualOrFailCase(remaining,
                                   skip_enforcement['remaining'],
                                   "The skip limit should be {}.".format(str(remaining)))

        self.muse_session_helper.make_muse_session(self.muse,
                                                   group_id=self.gid,
                                                   hhid=self.zp.muse_hhid,
                                                   test_fixture=self)
        self.muse.load_cloud_queue(session_id=self.muse.session_id,
                                   hhid=self.zp.muse_hhid,
                                   queue_base_url=self.cq_control.CQ_PLAYBACK_URI,
                                   play_on_completion=False,
                                   position_millis=position_millis,
                                   item_id=itemId)
        self.muse.subscribe_playback(group_id=self.gid,
                                     hhid=self.zp.muse_hhid)
        self.verify_event(PlaybackStatusEvent, timeout=1,
                          event_verifier=lambda e: (e.header.householdId == self.zp.muse_hhid and
                                                    e.header.groupId == self.gid and
                                                    e.playback_state == PlaybackState.IDLE and
                                                    e.item_id == itemId))



[docs]    def verifyCanSkip(self, num_skips, next_item_id):
        """
        Verify that we are able to skip to the next N tracks.

        :param num_skips: :int: number of times to skip.
        :param next_item_id: :string: the item id of the next track.
        """
        for i in range(0, num_skips):
            self.cq_control.clear_log()
            self.muse.skip_to_next(group_id=self.gid,
                                   hhid=self.zp.muse_hhid,
                                   session_id=self.muse.session_id)
            self.verify_event(PlaybackStatusEvent,
                              event_verifier=lambda e: (e.item_id == str(next_item_id)),
                              timeout=1)
            if not self.cq_control.wait_for_filtered_log(type='api',
                                                         fn=lambda l: (l['path'] == '/timePlayed' and
                                                                       l['body']['items'][0]['id'] == str(next_item_id - 1))):
                self.fail("Did not get a /timePlayed event after the skip")
            next_item_id += 1

    def test_time_played_post_skip_information(self, ver, policy):
        """
        Test that players send a skip.reportId with timePlayed POSTs when skipping a track
        """
        self.stop_cloud_queue_server()
        self.start_cloud_queue_server(cq_port=self.cq_port,
                                      endpoint_version=ver)
        self.cq_control.set_reports_policy({'sendPlaybackActions': policy})
        self.setLimitedSkipsStateAndLoadCloudQueue(remaining=1)
        self.muse.play(group_id=self.gid, hhid=self.zp.muse_hhid)
        wait_until_true(lambda: self.zp.AVTransport.GetPositionInfo()["RelTime"] >= "0:00:05",
                        timeout_seconds=5, iteration_delay=1, start_delay=5)
        self.verifyCanSkip(num_skips=1, next_item_id=1)

        # SWPBL-66771
        # Verify that the /timePlayed report has the correct format for CQ ver and whether
        # "sendPlaybackActions" is enabled or not
        if ver == 2.1:
            logs = self.cq_control.wait_for_filtered_log(fn=lambda l: (l['path'] == '/timePlayed' and
                                                                       "skip" in l["body"]["items"][0]),
                                                         timeout=15, expected=1)
        elif ver == 2.2 and policy:
            logs = self.cq_control.wait_for_filtered_log(fn=lambda l: (l['path'] == '/timePlayed' and
                                                                       "skip" in l["body"]["items"][0]["actions"][0] and
                                                                       l["body"]["items"][0]["actions"][0]["skip"][0]["positionMillis"] >= 4000),
                                                         timeout=15, expected=1)
        else:
            logs = self.cq_control.wait_for_filtered_log(fn=lambda l: (l['path'] == '/timePlayed' and
                                                                       "actions" not in l["body"]["items"][0]),
                                                         timeout=15, expected=1)
        self.verifyGreaterEqualOrFailCase(len(logs), 1,
                                          "Should have found a /timePlayed report with expected format")

[docs]    def test_skip_limit_after_delegation(self):
        '''
        Covers SWPBL-54064 and SWPBL-54021
        Tests whether the reportId and skip enforcement is carried over from one player to
        another after delegation.
        '''
        self.zp.diag.set_diag_level("tpm", 5)
        for x in range(0, 2):
            for x in range(0, 3):
                self.cq_control.add_track_to_end(track_id=x)

        self.setSkipEnforcementPolicy()
        self.setLimitedSkipsStateAndLoadCloudQueue(remaining=3, maxSkips=4, ttl=60)
        self.muse.play(group_id=self.gid,
                       hhid=self.zp.muse_hhid,
                       session_id=self.muse.session_id)
        # Wait for playback confirmation
        wait_until_true(lambda: self.zp.AVTransport.is_playing(),
                        timeout_seconds=2)

        # Now skip 2 times (1 skip left)
        self.zp.AVTransport.Next()
        self.zp.AVTransport.Next()
        # last skip
        self.zp.AVTransport.Next()
        wait_until_true(lambda: self.zp.AVTransport.GetPositionInfo()["RelTime"] >= "0:00:02",
                        iteration_delay=1,
                        timeout_seconds=5)

        # Get another ZP and delegate to it
        self.zp2 = ifilter(lambda device: (device.is_playback_device()
                                           and not device.is_bonded()
                                           and device != self.zp),
                           self.my_devices).next()
        self.zp2.AVTransport.BecomeCoordinatorOfStandaloneGroup(timeout_seconds=10)

        # Join group
        self.zp2.AVTransport.join_group(self.zp, timeout_seconds=5)

        # Leave group
        self.zp.AVTransport.BecomeCoordinatorOfStandaloneGroup(timeout_seconds=5)
        wait_until_true(lambda: self.zp.GroupManagement.is_group_coordinator_local())

        # Try to skip via UPnP and verify we can not
        with IgnoreInvokeErrors(self.zp2):
            self.zp2.AVTransport.Next()
        upnpErr = self.zp2.last_upnp_error_code
        self.verifyEqualOrFailCase(upnpErr, 800, "Skip should not be allowed")

        # Now try to skip using MUSE
        self.muse2 = self.make_connection(self.zp2)
        gid2 = self.zp2.GroupManagement.get_local_group_uuid()
        self.muse2.subscribe_playback(group_id=gid2, hhid=self.zp2.muse_hhid)
        self.verify_event(PlaybackStatusEvent, timeout=1, connection=self.muse2,
                          event_verifier=lambda e: (e.header.householdId == self.zp2.muse_hhid and
                                                    e.header.groupId == gid2))
        self.muse2.skip_to_next(group_id=gid2,
                                hhid=self.zp2.muse_hhid,
                                session_id=self.muse2.session_id)
        self.verify_event(Error, timeout=1, connection=self.muse2,
                          event_verifier=lambda e: (e.error_code == Error.SKIP_LIMIT_REACHED and
                                                    e.header.groupId == gid2))

        # Make sure we get 4 timePlayed reports
        # 3 from the skips, 1 from the original GC for partial playback of item 3 before delegation
        for x in range(0,4):
            timePlayed_log = self.cq_control.wait_for_filtered_log(type='api',
                                                                   fn=lambda l: (l['path'] == '/timePlayed' and
                                                                                 l['body']['items'][0]['id'] == str(x)),
                                                                   timeout=10,
                                                                   expected=1)
            self.verifyEqualOrFailCase(1, len(timePlayed_log),
                                   "Should have gotten /timePlayed report for itemId {}".format(x))

        self.zp.diag.set_diag_level("tpm", 1)

        # Verify we still can't skip
        self.verifySkipLimitReached()

        # Give it time to transition naturally to the next track which should be item id 2
        self.verify_event(PlaybackStatusEvent, timeout=2,
                          event_verifier=lambda e: (e.playback_state == PlaybackState.PLAYING and
                                                    e.item_id == "2"))
