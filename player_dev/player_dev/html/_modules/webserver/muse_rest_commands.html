[docs]class MuseRestCommands(MuseRestClientFixtureMixin):
    """
    Tests for RESTful Muse commands. These tests cover basic use cases to ensure all REST handlers in the player work and
    also SSL validations in the namespaces that require it. SWPBL-88608 removed the SSL requirement for some namespaces as
    captured in the table below.

    Many tests in this suite are run as combinatorial cases to test both secure and insecure connections to the different
    namespaces. In those cases the Muse connection object is re-established within the test as the combinatorial vectors
    cannot be passed into setUpTest().

    Namespace | SSL Required

    favorites:1             |   Yes
    groups:1                |   Yes
    groupVolume:1           |   No
    musicServiceAccounts:1  |   Yes
    playback:1              |   No
    playbackExtended:1      |   Yes
    playbackMetadata:1      |   No
    playbackSession:1       |   Yes
    playerVolume:1          |   No
    households:1            |   Yes

    # TODO: SWPBL-12525 add subscription command tests

    See https://confluence.sonos.com/x/8FiwAw for documentation.
    """
    TEST_TYPE = ['NIGHTLY_MUSE_MISC_NAMESPACE']

[docs]    @combinatorial("generate_connection_type")
    def test_groups_command(self, connection_type):
        """
        Verify a command routed by the /groups/groupId path.
        Verify both a successful command an an unsuccessful command.

        :return:
        """
        resp = muse_helper.hit_rest_endpoint(zp=self.zp, is_secure=connection_type,
                                             rest_endpoint=self.groupVolume_mute,
                                             req_method="POST", req_json={"muted":True})
        muse_helper.verify_rest_response(fixture=self, resp=resp,
                                         expected_resp_code=requests.codes.OK,
                                         expected_resp_text=None)

        resp = muse_helper.hit_rest_endpoint(zp=self.zp, is_secure=connection_type,
                                             rest_endpoint=self.groupVolume_stem(self.groups_stem, ""),
                                             req_method="GET")
        muse_helper.verify_rest_response(fixture=self, resp=resp,
                                         expected_resp_code=requests.codes.OK,
                                         expected_resp_text=None)
        self.verifyTrueOrFailCase(json.loads(resp.text)["muted"] is True,
                                  "Resp text: {}".format(resp.text))

        resp = muse_helper.hit_rest_endpoint(zp=self.zp, is_secure=connection_type,
                                             rest_endpoint=self.groupVolume_stem("groups/BAD_GID",
                                                                                 "/mute"),
                                             req_method="POST", req_json={"muted":True})
        muse_helper.verify_rest_response(fixture=self, resp=resp,
                                         expected_resp_code=requests.codes.client_closed_request,
                                         expected_resp_text=None)
        self.verifyTrueOrFailCase(json.loads(resp.text)["groupStatus"] == "GROUP_STATUS_GONE",
                                  "Resp text: {}".format(resp.text))

    def test_sessions_command(self, connection_type):
        """
        Verify a command routed by the /playbackSessions/sessionId path.
        Verify both a successful command an an unsuccessful command.

        Note that most commands that target sessions won't work since the session
        is removed immediately after being created on REST.

        :return:
        """
        resp = muse_helper.hit_rest_endpoint(zp=self.zp, is_secure=connection_type,
                                             rest_endpoint=self.playbackSession_stem(self.groups_stem, ""),
                                             req_method="POST", req_json={"appId":self.__class__.__name__,
                                                                          "appContext":self._testMethodName})
        if connection_type:
            muse_helper.verify_rest_response(fixture=self, resp=resp,
                                             expected_resp_code=requests.codes.OK,
                                             expected_resp_text=None)

            resp = muse_helper.hit_rest_endpoint(zp=self.zp, is_secure=connection_type,
                                                 rest_endpoint=self.playbackSession_stem("groups/BAD_GID",
                                                                                         ""),
                                                 req_method="POST", req_json={"appId":self.__class__.__name__,
                                                                              "appContext":self._testMethodName})
            muse_helper.verify_rest_response(fixture=self, resp=resp,
                                             expected_resp_code=requests.codes.client_closed_request,
                                             expected_resp_text=None)
            self.verifyTrueOrFailCase(json.loads(resp.text)["groupStatus"] == "GROUP_STATUS_GONE",
                                      "Resp text: {}".format(resp.text))
        else:
            muse_helper.verify_rest_response(fixture=self, resp=resp,
                                             expected_resp_code=requests.codes.FORBIDDEN,
                                             expected_resp_text=None)
