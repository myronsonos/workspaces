    def _conver_str_time_to_millis(self, str_time):
        """
        :param str_time:
        :return:
        """
        t = time.strptime(str_time, "%H:%M:%S")
        return int(1000 * datetime.timedelta(seconds=t.tm_sec).total_seconds())

[docs]    def verify_request_param(self, value, param="reason", fetches=1,
                             clear=True, timeout=10, path="/itemWindow"):
        """
        Helper function to verify a specific reason value in /itemWindow fetches
        :param reason:
        :param fetches:
        :return:
        """
        logs = self.cq_control.wait_for_filtered_log(type='api', timeout=timeout, expected=fetches,
                                                     fn=lambda l: (l['path'] == path and
                                                                   param in l['queryParams']))

        if len(logs) >= fetches:
            if fetches > 0:
                for l in logs:
                    self.verifyEqualOrWarn(value, l["queryParams"][param],
                                           "Looking for {} request with {} {} found".format(path, param,
                                                                                            value))
                    return l["queryParams"]
            else:
                self.verifyEqualOrFailCase(fetches, len(logs),
                                           "No {} request with {} {} should have been found".format(path,
                                                                                                    param,
                                                                                                    value))
                return None
        else:
            self.fail("Expected {} {} requests with {} {}, got {}".format(path, fetches, param,
                                                                          value, len(logs)))

        if clear:
            self.cq_control.clear_log()

    def test_verify_playback_resumes_after_suspend(self, qv_enable):
        """
        SWPBL-67824
        With a >2.1 CQ server, the player should resume playback of the cq content after a single
        play command while suspended. The result /itemWindow request should have the current contextVersion
        and the queueVersion if it was passed into with the suspend command. The itemId should be empty.
        :return:
        """
        # Load and suspend. Gather the contextVersion and queueVersion values.
        self.muse.load_cloud_queue(session_id=self.muse.session_id,
                                   hhid=self.zp.muse_hhid,
                                   queue_base_url=self.cq_control.CQ_PLAYBACK_URI,
                                   httpAuthorization=None,
                                   play_on_completion=True,
                                   track_metadata=None)
        self.verify_event(PlaybackStatusEvent, timeout=1,
                          event_verifier=lambda e: (e.playback_state == PlaybackState.PLAYING and
                                                    e.queue_version is not None))
        versions = self.cq_control.get_version_endpoint()
        if qv_enable:
            qv = versions["queueVersion"]
        else:
            qv = ""
        self.muse.suspend_session(hhid=self.zp.muse_hhid, sessionId=self.muse.session_id,
                                  queueVersion=qv)
        self.verify_event(SessionInfoEvent, timeout=1,
                          event_verifier=lambda e: e.suspended is True)
        self.cq_control.clear_log()

        # Send a play command, the player should restart playback
        self.zp.AVTransport.Play()
        self.verify_event(PlaybackStatusEvent, timeout=1,
                          event_verifier=lambda e: (e.playback_state == PlaybackState.PLAYING))

        # Verify the params in the /itemWindow request with reason=resume
        params = self.verify_request_param(value="resume")
        self.verifyEqualOrFailCase(qv, params["queueVersion"],
                                   "queueVersion should be {}".format(qv))
        self.verifyEqualOrFailCase(versions["contextVersion"], params["contextVersion"],
                                   "contextVersion should match the value on the server")
        self.verifyTrueOrFailCase(not params["itemId"],
                                  "itemId should be empty")
