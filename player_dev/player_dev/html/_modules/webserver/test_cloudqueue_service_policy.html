        '''
        Make sure that when delegation occurs, the new GC picks up the current policies and playmodes
        from the original GC.
        :return:
        '''
        # Set desired server policies and expected policy and playmodes
        # canSkip = False
        # canSkipBack = True
        # REPEAT_ALL = True
        # SHUFFLE_NOREPEAT = True
        # CROSSFADE = False
        policy = {"canShuffle": True, "canCrossfade": False, "canRepeatOne": True,
                  "canRepeat": True, "canStop": True, "canPause": True, "canSeek": False,
                  "canSkipBack": True, "canSkip": False}
        expectedPlaybackActions = {"canShuffle": True, "canCrossfade": False, "canRepeatOne": True,
                  "canRepeat": True, "canStop": True, "canPause": True, "canSeek": False,
                  "canSkipBack": True, "canSkip": False}
        expectedPlaymodes = {"crossfade": False, "repeat": True, "shuffle": False, "repeatOne": False}

        self.zp2 = self.get_playback_gc()
        self.cq_control.set_playback_policy(policy)
        self.muse.load_cloud_queue(session_id=self.muse.session_id,
                                   hhid=self.zp.muse_hhid,
                                   queue_base_url=self.cq_control.CQ_PLAYBACK_URI,
                                   item_id="1",
                                   track_metadata=None,
                                   play_on_completion=True)
        self.zp.AVTransport.SetPlayMode("REPEAT_ALL")
        self.verify_event(PlaybackStatusEvent, timeout=1,
                          event_verifier=lambda e: (e.playback_state == PlaybackState.PLAYING and
                                                    e.availablePlaybackActions == expectedPlaybackActions and
                                                    e.playModes == expectedPlaymodes))
        self.muse.unsubscribe_playback(group_id=self.gid, hhid=self.zp.muse_hhid)
        wait_until_true(lambda: self.zp.AVTransport.GetPositionInfo()['RelTime'] == "0:00:02",
                        iteration_delay=.5,
                        timeout_seconds=5)

        # Add second player to the group
        self.zp2.AVTransport.join_group(self.zp, timeout_seconds=10)
        wait_until_true(lambda: self.zp.AVTransport.GetPositionInfo()['RelTime'] == "0:00:04",
                        iteration_delay=.5,
                        timeout_seconds=5)

        # ZP2 is now the new GC, establish muse connection to it
        self.zp.AVTransport.BecomeCoordinatorOfStandaloneGroup()
        self.muse2 = self.make_connection(self.zp2)
        gid2 = self.zp2.GroupManagement.get_local_group_uuid()
        self.muse2.subscribe_playback(group_id=gid2, hhid=self.zp2.muse_hhid)

        # Expect ZP2 to be playing and its availablePlaybackActions and playmodes
        # to match what was ZP1 before delegation
        self.verify_event(PlaybackStatusEvent, timeout=1,
                          event_verifier=lambda e: (e.playback_state == PlaybackState.PLAYING and
                                                    e.gid == gid2 and
                                                    e.availablePlaybackActions == expectedPlaybackActions and
                                                    e.playModes == expectedPlaymodes),
                          connection=self.muse2)

        # Checks to see skipping is disallowed
        self.muse2.skip_to_next(group_id=gid2,
                               hhid=self.zp2.muse_hhid,
                               session_id=self.muse2.session_id)
        self.verify_event(Error,
                          event_verifier=lambda e: (e.error_code == Error.DISALLOWED_POLICY),
                          timeout=1,
                          connection=self.muse2)

        # Checks to see skipping backwards is allowed
        self.muse2.skip_to_previous(group_id=gid2,
                               hhid=self.zp2.muse_hhid,
                               session_id=self.muse2.session_id)
        self.verify_event(PlaybackStatusEvent,
                          event_verifier=lambda e: (e.item_id == "0" and
                                                    e.playback_state == PlaybackState.PLAYING),
                          timeout=1,
                          connection=self.muse2)

        # Checks to see enabling crossfade is disallowed
        with IgnoreInvokeErrors(self.zp2):
            self.zp2.AVTransport.SetCrossfadeMode(True)
        self.verifyEqualOrFailCase(712, self.zp2.last_upnp_error_code,
                                   "Player should return a 712 UPnP code")

        # Checks to see enabling SHUFFLE_NOREPEAT is allowed
        with IgnoreInvokeErrors(self.zp2):
            self.zp2.AVTransport.SetPlayMode("SHUFFLE_NOREPEAT")
        self.verifyEqualOrFailCase(0, self.zp2.last_upnp_error_code,
                                   "Player should return a 0 UPnP code")

        '''
        Cloud Queue /context verfication test: The player should use the default value for any policies that are
        omitted from the received context.

        This test assumes the value of R_TRACKQ_POLICY_CLOUDQ_DEFAULT in the player (playbackpolicies.h). It
        specifically assumes that crossfade is enabled by default.

        Playback Policies are immutable per "loadCloudQueue session." That means that the test needs to reload
        the cloud queue whenever the playback policies are changed on the server.

        @see SWPBL-53761, SWPBL-52992
        '''
        # 1. Set initial conditions to have crossfade off
        self.cq_control.set_playback_policy({'canCrossfade': False})
        playback_policy = self.cq_control.get_playback_policies()
        self.verifyFalseOrStop(playback_policy['canCrossfade'],
                               "Container canCrossfade policy should be False")

        self.muse.load_cloud_queue(session_id=self.muse.session_id,
                                   hhid=self.zp.muse_hhid,
                                   queue_base_url=self.cq_control.CQ_PLAYBACK_URI,
                                   item_id="0",
                                   track_metadata=None,
                                   play_on_completion=True)
        context_log = self.cq_control.wait_for_filtered_log(type='api',
                                                            fn=lambda l: l['path'] == '/context',
                                                            timeout=5,
                                                            history=False)
        with IgnoreInvokeErrors(self.zp):
            self.zp.AVTransport.SetCrossfadeMode(True)

        self.verifyEqualOrFailCase(712, self.zp.last_upnp_error_code, "Player should not allow crossFade")

        # 2. The player should not respect this change because loadCloudQueue was not called
        self.cq_control.set_playback_policy({'canCrossfade': True})
        playback_policy = self.cq_control.get_playback_policies()
        self.verifyTrueOrStop(playback_policy['canCrossfade'],
                              "Container canCrossfade policy should be True")
        self.muse.refresh_cloud_queue(session_id=self.muse.session_id)
        context_log = self.cq_control.wait_for_filtered_log(type='api',
                                                            fn=lambda l: l['path'] == '/context',
                                                            timeout=5,
                                                            history=False)
        with IgnoreInvokeErrors(self.zp):
            self.zp.AVTransport.SetCrossfadeMode(True)

        self.verifyEqualOrFailCase(712, self.zp.last_upnp_error_code, "Player should not allow crossFade")

        # 3. Now, drop "canCrossfade" from the CQ server policies and make sure that the player restores the default.
        self.cq_control.delete_playback_policy('canCrossfade')
        playback_policy = self.cq_control.get_playback_policies()
        self.verifyFalseOrStop('canCrossfade' in playback_policy,
                               "Container canCrossfade policy should be omitted")

        self.muse.load_cloud_queue(session_id=self.muse.session_id,
                                   hhid=self.zp.muse_hhid,
                                   queue_base_url=self.cq_control.CQ_PLAYBACK_URI,
                                   item_id="0",
                                   track_metadata=None,
                                   play_on_completion=True)
        context_log = self.cq_control.wait_for_filtered_log(type='api',
                                                            fn=lambda l: l['path'] == '/context',
                                                            timeout=5,
                                                            history=False)
        with IgnoreInvokeErrors(self.zp):
            self.zp.AVTransport.SetCrossfadeMode(True)

        self.verifyEqualOrFailCase(0, self.zp.last_upnp_error_code, "Player should allow crossfade")
        self.zp.AVTransport.SetCrossfadeMode(False)

        # 4. Check that the player still downloads the playback policy. This case is not covered by
        #    number one when the player is the initial state (i.e. after reboot).
        self.cq_control.set_playback_policy({'canCrossfade': False})
        playback_policy = self.cq_control.get_playback_policies()
        self.verifyFalseOrStop(playback_policy['canCrossfade'],

                                   queue_base_url=self.cq_control.CQ_PLAYBACK_URI,

                                                            timeout=5,
                                                            history=False)
        with IgnoreInvokeErrors(self.zp):
