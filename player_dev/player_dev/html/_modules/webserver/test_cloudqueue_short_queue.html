                                                        self.my_devices)))
        except IndexError:
            test_devices["SH4"] = None

        for type in test_devices:
            yield type, test_devices[type]

    def _setup(self, type, zp):
        # Gracefully skip the test if the desired arch type isn't on the testbed.
        if not zp:
            self.skip("Couldn't find playback device type {}, skipping test".format(type))
        else:
            # Replace the current self.zp and self.zp to be targeted to the new DUT
            # Also replace the failure log file to be targeted to the new ZP
            self.zp.diag.stop_udp_log_to_file()
            if os.path.exists(self.zp_logger.fail_log):
                os.remove(self.zp_logger.fail_log)
            self.disconnect_all()
            self.zp = zp
            self.muse = self.make_connection(self.zp)
            self.zp.diag.start_udp_log_to_file(self.zp_logger.fail_log)

        muse_session_mixin.make_muse_session(self.muse,
                                             group_id=self.gid,
                                             hhid=self.zp.muse_hhid,
                                             test_fixture=self)
        self.muse.set_muted_groupvolume(group_id=self.gid, hhid=self.zp.muse_hhid, muted=True)
        self.muse.subscribe_playback(group_id=self.gid,
                                     hhid=self.zp.muse_hhid)

        self.zp.diag.set_diag_level("cloudqueue", 5)
        self.zp.diag.set_diag_level("muse", 5)
        self.zp.diag.set_diag_level("cqfsm", 5)
        self.zp.diag.set_diag_level("chsrc", 3)
        self.zp.diag.set_diag_level("chsnk", 3)
        self.zp.diag.set_diag_level("cssc", 4)

    def test_player_does_not_repeat_track_at_end_of_queue(self, type, zp):
        """
        SWPBL-12550
        Verifies that the player does not repeat the track if the last track is deleted
        during the last few seconds of playback.
        Due to the different archs we have, we would like to test on sh4, ppc, playbar and connectx.
        :param device:
        :param zp:
        :return:
        """
        self._setup(type, zp)

        md = muse_session_mixin.cq_control.get_track_metadata("0")
        self.muse.load_cloud_queue(session_id=self.muse.session_id,
                                   hhid=self.zp.muse_hhid,
                                   queue_base_url=muse_session_mixin.cq_control.CQ_PLAYBACK_URI,
                                   play_on_completion=True,
                                   item_id="0",
                                   position_millis=20000)
        self.verify_event(PlaybackStatusEvent, timeout=1,
                          event_verifier=lambda e: (e.playback_state == PlaybackState.PLAYING))
        wait_until_true(lambda: self.zp.AVTransport.GetPositionInfo()["RelTime"] >= "0:00:27",
                        start_delay=6, timeout_seconds=5, iteration_delay=.5)
        muse_session_mixin.cq_control.delete_track("1")
        # Use a skipToItem command with track MD to increase stress on the cqfsm
        self.muse.skip_to_item(session_id=self.muse.session_id,
                               hhid=self.zp.muse_hhid,
                               item_id="0",
                               play_on_completion=True,
                               track_metadata=md)
        self.verify_event(PlaybackStatusEvent, timeout=1,
                          event_verifier=lambda e: (e.playback_state == PlaybackState.PAUSED and
                                                    e.position_millis >= 30000))

    def test_player_buffering_unavailable_last_track(self, type, zp):
        """
        SWPBL-63825
        Verify that the player enters the paused state under the following circumstances:
            - player is playing second to last track
            - player attempts to start buffering last track and fails (track unreachable)
            - player stops playback and the playhead is at the top of the queue (changed by SWPBL-63057)
        """
        self._setup(type, zp)

        # Add an inacessible track at an invalid URL
        muse_session_mixin.cq_control.add_custom_track_to_end(name='badtrack',
                                                              mediaUrl=INVALID_HLS_TRACK_URL,
                                                              contentType='audio/mp3')
        # Load the CQ pointing at the first track
        self.muse.load_cloud_queue(session_id=self.muse.session_id,
                                   hhid=self.zp.muse_hhid,
                                   queue_base_url=muse_session_mixin.cq_control.CQ_PLAYBACK_URI,
                                   play_on_completion=True,
                                   item_id="1",
                                   position_millis=20000)
        # Verify the player is playing
        self.verify_event(PlaybackStatusEvent, timeout=1,
                          event_verifier=lambda e: (e.playback_state == PlaybackState.PLAYING))

        # Wait until the player attempts to request the next track
        self.zp.diag.wait_for_anacapa_log_statement(
            '{} lookup failed'.format(INVALID_HLS_TRACK_NAME), 10)

        # Verify that the player is IDLE and at the top of the queue.
        self.verify_event(PlaybackStatusEvent, timeout=5,
            event_verifier=lambda e: (e.playback_state == PlaybackState.IDLE and
                                      e.item_id == "0" and
                                      e.position_millis == 0))

    def _queue_setup(self, num_tracks, start_track, play_on_completion=True, policies=None):
        """
        Helper function for setting up the queue.

        :param int num_tracks: Number of tracks to add to the queue.
        :param int start_track: Starting track # to play on
        :param bool play_on_completion: Play on completion?
        :param dict policies: (Optional) Playback policies for CQ
        """
        zp = self.generators.generate_testbed_playback_device().next()
        self._setup(zp.modelNumber, zp)

        # Set policies.
        muse_session_mixin.cq_control.set_playback_policy(policies or {})

        # Add the specified number of tracks (tracks are chosen randomly).
        for _ in xrange(num_tracks):
            muse_session_mixin.cq_control.add_track_to_end(str(random.randint(0, 2)))

        # Load the CQ pointing at the last track (track list: 3, ..., 3 + num_tracks - 1)
        self.muse.load_cloud_queue(session_id=self.muse.session_id,
                                   hhid=self.zp.muse_hhid,
                                   queue_base_url=muse_session_mixin.cq_control.CQ_PLAYBACK_URI,
                                   play_on_completion=play_on_completion,
                                   item_id=str(start_track),
                                   position_millis=20000)

        if play_on_completion:
            # Wait for player to start playing
            self.verify_event(PlaybackStatusEvent,
                              event_verifier=lambda e: (e.playback_state == PlaybackState.PLAYING and
                              e.item_id == str(start_track)))


[docs]    def test_player_replays_pauseAtEnd_queue_with_play_when_paused_at_end_and_last_track_deleted(self):
        """
        Load a cloud queue w/ pauseAtEndOfQueue=true, allow the player to pause at the end:
            - Delete the last track and send a "play" command.
            - Verify that the player starts playing from the (logical) beginning of the cloud queue.
        :return:
        """
        # Clear queue
        muse_session_mixin.cq_control.clear_queue()

        # Indices for tracks, note the first track is 3 because the CQ is initialized with
        # 3 tracks that we remove when we clear the queue.
        first_track = 3
        num_tracks = 30
        last_track = first_track + num_tracks - 1

        self._queue_setup(num_tracks=num_tracks, start_track=last_track, policies={"pauseAtEndOfQueue": True})

        # Wait for player to pause (at end)
        self.verify_event(PlaybackStatusEvent,
                          event_verifier=lambda e: (e.playback_state == PlaybackState.PAUSED and
                                                    e.item_id == str(last_track) and
                                                    e.position_millis >= 30000))
        # Delete the current track.
        muse_session_mixin.cq_control.delete_track(str(last_track))

        # Play and verify the player restarts at the beginning of the queue
        self.muse.playback.play(hhid=self.zp.muse_hhid, group_id=self.gid, session_id=self.muse.session_id)
        self.verify_event(PlaybackStatusEvent,
                          event_verifier=lambda e: (e.playback_state == PlaybackState.PLAYING and
