[docs]    def test_no_new_token_while_paused(self):
        """
        Regression testing for SWPBL-48044:
        CQ Tokens are incorrectly refreshed when a refreshCloudQueue() command is processed in the paused/stopped state
        This test case makes sure that we are ignoring new tokens if they expire while the player is paused
        :return:
        """
        # Set the auth token poll rate while paused to 5 seconds on the player
        if not self.zp.diag.has_testpoint_support():
            self.skip('set_cloudqueue_poll_timers needs builds with testpoint support')
        else:
            self.cq_control.set_cloudqueue_poll_timers(zp=self.zp,
                                                       playing_timer=600,
                                                       paused_timer=5)

        self.cq_control.expire_tokens()
        token = 'playon=' + self.cq_control.get_current_auth_token()
        self.muse.load_cloud_queue(session_id=self.muse.session_id,
                                   hhid=self.zp.muse_hhid,
                                   queue_base_url=self.cq_control.CQ_PLAYBACK_URI,
                                   httpAuthorization=token,
                                   play_on_completion=True,
                                   track_metadata=None,
                                   useHttpAuthorizationForMedia=True)
        self.verify_event(PlaybackStatusEvent, timeout=1,
                          event_verifier=lambda e: (e.playback_state == PlaybackState.PLAYING))

        # We need to let the player buffer track data while playing otherwise it won't make the
        # required network calls below. This is a subtle change introduced with SWPBL-68185
        wait_until_true(lambda: self.zp.AVTransport.GetPositionInfo()["RelTime"] >= '0:00:10',
                        timeout_seconds=5,
                        iteration_delay=1,
                        start_delay=10)

        # Wait for at least one /version fetch after the pause so that the player gets
        # the new token from the server
        self.cq_control.clear_log()
        self.muse.pause(group_id=self.gid,
                        hhid=self.zp.muse_hhid)
        self.verify_event(PlaybackStatusEvent, timeout=1,
                          event_verifier=lambda e: (e.playback_state == PlaybackState.PAUSED))
        filtered_logs = self.cq_control.wait_for_filtered_log(fn=lambda x: x['path'] == '/version')
        self.verifyGreaterEqualOrFailCase(len(filtered_logs), 1,
                               "Should have gotten an /version fetch after pause")
        paused_token = 'playon=' + self.cq_control.get_current_auth_token()

        # Expire the current token again and wait for another /version so that the player
        # has the most current token
        self.cq_control.expire_tokens()
        self.cq_control.clear_log()
        filtered_logs = self.cq_control.wait_for_filtered_log(fn=lambda x: x['path'] == '/version')
        self.verifyGreaterEqualOrFailCase(len(filtered_logs), 1,
                                          "Should have gotten at least one /version fetch to poll for token changes")

        # Refresh to force an /itemWindow fetch, make sure that fetch uses the token from the pause
        self.cq_control.clear_log()
        self.muse.refresh_cloud_queue(session_id=self.muse.session_id,
                                      hhid=self.zp.muse_hhid)
        filtered_logs = self.cq_control.wait_for_filtered_log(fn=lambda x: x['path'] == '/itemWindow')
        self.verifyEqualOrStop(1, len(filtered_logs),
                               "Should have gotten an /itemWindow fetch after a CQ refresh")
        self.verifyEqualOrStop(paused_token, filtered_logs[0]["authorization"],
                               "Current auth token should be the same as after pause")

    def test_time_played_uses_new_auth_token(self, requireAuth):
        """
        Check that new tokens are used when reporting time played of cloud queue tracks. The player
        should use the most recent token when reporting.
        SWPBL-83584 added combinatorial test to verify that the player sends timePlayed report with authToken
        no matter what the track "requireAuth" policy is set to.
        """
        self.cq_control.start_cloud_queue_server(authToken='12345',
                                                 proto="https",
                                                 mediaUnsecured=True,
                                                 authTokenExpiration="1")

        self.cq_control.clear_log(type='api')
        self.cq_control.expire_tokens()
        token = 'playon=' + self.cq_control.get_current_auth_token()
        self.muse.load_cloud_queue(session_id=self.muse.session_id,
                                   hhid=self.zp.muse_hhid,
                                   queue_base_url=self.cq_control.CQ_PLAYBACK_URI,
                                   httpAuthorization=token,
                                   play_on_completion=True,
                                   track_metadata=None,
                                   useHttpAuthorizationForMedia=requireAuth)
        self.verify_event(PlaybackStatusEvent, timeout=1,
                          event_verifier=lambda e: (e.item_id == '0' and
                                                    e.playback_state == "PLAYBACK_STATE_PLAYING"))

        # Let the player buffer enough track data that it will make network calls while in the PAUSED state.
        wait_until_true(lambda: self.zp.AVTransport.GetPositionInfo()["RelTime"] >= "0:00:10",
                        iteration_delay=1, start_delay=10, timeout_seconds=5)

        self.muse.pause(group_id=self.gid, hhid=self.zp.muse_hhid)
        self.verify_event(PlaybackStatusEvent, timeout=1,
                          event_verifier=lambda e: (e.playback_state == "PLAYBACK_STATE_PAUSED"))

        # make sure at least 2 /version fetches has occurred
        self.verifyGreaterEqualOrFailCase(len(self.cq_control.wait_for_filtered_log(type='api',
                                                                                    fn=lambda logEntry:
                                                                                    logEntry['path'] == '/version',
                                                                                    timeout=10,
                                                                                    expected=2)),
                                          1, "should have at least 1 /version fetch after the pause")

        self.muse.play(group_id=self.gid, hhid=self.zp.muse_hhid)
        self.verify_event(PlaybackStatusEvent, timeout=1,
                          event_verifier=lambda e: (e.playback_state == "PLAYBACK_STATE_PLAYING"))

        new_token = 'playon=' + self.cq_control.get_current_auth_token()

        # verify that tokens are different
        self.verifyNotEqualOrFailCase(token, new_token, "tokens must differ")

        # skip to force a /timePlayed post without waiting for the song to finish
        self.cq_control.clear_log(type='api')
        self.muse.skip_to_next(group_id=self.gid, hhid=self.zp.muse_hhid)
        self.verify_event(PlaybackStatusEvent,
                          event_verifier=lambda e: (e.item_id == '1' and
                                                    e.playback_state == "PLAYBACK_STATE_PLAYING"),
                          timeout=1)

        # expect that timePlayed uses new token
        time_played_logs = self.cq_control.wait_for_filtered_log(type='api',
                                                                 fn=lambda logEntry:
                                                                 logEntry['path'] == '/timePlayed',
                                                                 timeout=30)
        self.verifyGreaterEqualOrFailCase(len(time_played_logs), 1,
                                          "need at least one log")
        for tpl in time_played_logs:
            self.verifyEqualOrFailCase(new_token, tpl['authorization'], "must use new token")

    def test_auth_token_in_media_request(self, cq, requireAuth):
        """
        SWPBL-82079
        Verify that media requests do or do not have the authorization param based on CQ version
        and the "requireAuth" track-level policy
        :param cq:
        :param requireAuth:
        :return:
        """
        token = "AUTH_TOKEN"
        # Start the server with the request cq version
        self.cq_control.start_cloud_queue_server(proto="https", endpoint_version=cq)
        if requireAuth:
            md = self.cq_control.get_track_metadata(item_id="0")
            # set the urls to use the DNS name
            for url in ("imageUrl", "mediaUrl"):
                md[url] = md[url].replace(self.cq_control.host_ip,
                                          "mock.dev.ws.sonos.com")
        else:
            md = None

        self.muse.load_cloud_queue(session_id=self.muse.session_id,
                                   hhid=self.zp.muse_hhid,
                                   queue_base_url=self.cq_control.CQ_PLAYBACK_URI,
                                   httpAuthorization=token,
                                   play_on_completion=True,
                                   track_metadata=md,
                                   item_id="0",
                                   useHttpAuthorizationForMedia=requireAuth)
        self.verify_event(PlaybackStatusEvent, timeout=1,
                          event_verifier=lambda e: (e.playback_state == PlaybackState.PLAYING and
                                                    e.item_id == "0"))

        # The authorization param should be present if we're using cq version 1.0 and useHttpAuthorizationForMedia
        # is True or None.
        # cq version 2 defaults to false so the only time we should see the authorization param
        # is when useHttpAuthorizationForMedia is set to true on the track
        if (cq == 1.0 and requireAuth is None) or requireAuth is True:
            logs = self.cq_control.wait_for_filtered_log(fn=lambda l: ("authorization" in l and
                                                                       l["authorization"] == token),
                                                         type="media", timeout=5)
            self.verifyGreaterEqualOrFailCase(len(logs), 1,
                                              "Media requests should have auth token if requireAuth is True or CQ version v1")
        else:
            logs = self.cq_control.wait_for_filtered_log(fn=lambda l: "authorization" in l,
                                                         type="media", expected=0, timeout=3)
            self.verifyEqualOrFailCase(0, len(logs),
                                       "Media requests should not have auth token if requireAuth is False and CQ version v2.x"
                                       .format(requireAuth, str(cq)))
