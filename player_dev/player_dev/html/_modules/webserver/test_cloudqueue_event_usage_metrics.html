[docs]    def override_eventmetrics_url(self, zp, url=None):
        """
        Override the Event Metrics url to the specified value. Restart anacapa for the changes
        to take effect
        :param zp: The device under test
        :type zone: :class:`~sonos.client.internal.SonosZoneComponent`
        """
        if url is None:
            url = '{}usageMetrics'.format(self.cq_control.CQ_PLAYBACK_URI)
        zp.SystemProperties.SetString(self.EVENTMETRICS_URL_OVERRIDE, url)
        wait_until_true(lambda: zp.SystemProperties.GetString(self.EVENTMETRICS_URL_OVERRIDE) == url,
                        timeout_seconds=5,
                        iteration_delay=1)
        zp.bounce_anacapa_and_wait()

[docs]    def collect_usage_metric_events(self, envelopes, name, filt):
        """
        This function collects all of the 'name' events from a list of usage
        metric envelopes into a single envelope. This normalizes comparison
        of when events are split up into different envelopes by the player.
        """
        self.logger.debug("Inspecting {}".format(envelopes))
        my_event_envelope = self.del_event_content(envelopes[0], filt)
        all_control_events = []
        for evt in envelopes:
            self.logger.debug("Adding event: {}".format(evt))
            if name in evt:
                all_control_events = all_control_events + list(evt[name])

        my_event_envelope[name] = tuple(all_control_events)
        self.logger.debug("Combined envelope: {}".format(my_event_envelope))
        return my_event_envelope

[docs]    def get_actual_event_content(self, timeout=30,
                                 event=('playback', 'largeSyncError', 'transportError', 'control', 'cloudQueueError'),
                                 targetId_param='pqid', targetId=None):
        """
        Gets the expected events and formats it so that the test cases can use it for comparison
        :param timeout:
        :param event:
        :return:
        """
        logs = self.cq_control.wait_for_filtered_log(type='api',
                                                     fn=lambda l: (l['path'] == '/usageMetrics' and
                                                                   'UsageMetrics' in l['body']),
                                                     timeout=timeout, iteration_delay=10,
                                                     flush=True, zp=self.zp)
        all_content = []
        for p in logs:
            # This entire section is to take the output from the api log and
            # format it to match the expected structure of the tests.
            content = {}
            content['customerNumber'] = str(p['body']['UsageMetrics']['cu'])
            content['hhid'] = str(p['body']['UsageMetrics']['hh'])
            content['modelNumber'] = str(p['body']['UsageMetrics']['md'])
            content['serialNumber'] = str(p['body']['UsageMetrics']['sn'])

            # check for lost events
            if 'LostEvents' in p['body']['UsageMetrics']:
                self.logger.warning("The player dropped {} events".format(p['body']['UsageMetrics']['LostEvents']))

            # Loops through each 'event' in the report
            current_event_list = []
            if not isinstance(p['body']['UsageMetrics']['events']['event'], list):
                p['body']['UsageMetrics']['events']['event'] = [p['body']['UsageMetrics']['events']['event']]
            for e in p['body']['UsageMetrics']['events']['event']:
                if ('name' in e):
                    current_event_list.append(e)
            if isinstance(current_event_list, dict):
                self.logger.debug("Wrapping current event {} to a list: {}".format(type(current_event_list),
                                                                                   current_event_list))
                current_event_list = [current_event_list, ]
            for x in current_event_list:
                event_found = False
                name = x['name']
                try:
                    for en in x['entry']:
                        if en['key'] == 'type':
                            name = str(en['value'])
                except TypeError:
                    self.logger.warning("Unexpected type for name ({}) in: {}".format(x, p))
                    continue

                if name in event:
                    entry = {}
                    # Loops through each 'entry' under an 'event'
                    for y in x['entry']:
                        # Discard if the key has no associated value
                        if 'value' in y:
                            key = str(y['key'])
                            value = str(y['value'])
                            # Convert string playtime value to float then integer and
                            # back to str to make sure we have a whole number
                            if key == 'ps':
                                value = str(int(float(value)))

                            entry[key] = value

                    # If pqid is not None, check that the id of the event matches
                    if targetId is not None:
                        if targetId_param in entry and targetId in entry[targetId_param]:
                            event_found = True
                        else:
                            self.logger.debug("Dropping event that does not match "
                                "expected {} {}: {}".format(targetId_param, targetId, x))
                    # Otherwise assume the entry belongs in the event
                    else:
                        event_found = True

                    # Add the entry to the event if a match was found
                    if event_found:
                        if name not in content:
                            content[name] = list()

                        content[name].append(entry)

                else:
                    self.logger.debug("Dropping unwanted event {}: {}".format(name, x))

            # converts each event list to tuples to match expectedEvent data type format
            for item in event:
                if item in content:
                    content[item] = tuple(content[item])

            all_content.append(content)

        self.logger.debug("all_content: {}".format(all_content))
        return filter(lambda x: len(set(x.keys()).intersection(set(event))) > 0, all_content)

[docs]    def test_muse_basic_control_events(self):
        """
        Tests that the player sends control events for:
        * createSession
        * leaveSessionExplicitly
        * loadCloudQueue
        * skipToItem

        The following were added with SWPBL-63941
        * skipToNext
        * skipToPrevious
        * seek
        * pause
        * play
        """
        self.zp.diag.set_diag_level("muse", 5)
        self._start_playback()

        zp_udp = self.zp.raw_udn
        create_session_event = 'createSession'
        load_cloud_queue_event = 'loadCloudQueue'
        skip_to_next_event = 'skipToNextTrack'
        skip_to_previous_event = 'skipToPreviousTrack'
        pause_event = 'pause'
        play_event = 'play'
        seek_event = 'seek'
        skip_to_item_event = 'skipToItem'
        leave_session_event = 'leaveSession'
        museCommandHandler_events = (create_session_event,
                                     load_cloud_queue_event,
                                     skip_to_next_event,
                                     skip_to_previous_event,
                                     pause_event,
                                     play_event,
                                     seek_event,
                                     skip_to_item_event,
                                     leave_session_event)
        create_event = 'create'
        leave_event = 'leave'
        end_event = 'evict'
        museSessionChange_events = (create_event,
                                    leave_event,
                                    end_event)
        event_types = {"museSessionChange": museSessionChange_events,
                       "museCommandHandler": museCommandHandler_events}
        command_action_types = {"museSessionChange": "action",
                                "museCommandHandler": "command"}
        id_types = {"museSessionChange": "sessionId",
                    "museCommandHandler": "targetId"}

        self.muse.skip_to_next(group_id=self.gid, hhid=self.zp.muse_hhid)
        self.verify_event(PlaybackStatusEvent, timeout=1,
                          event_verifier=lambda e: (e.playback_state == PlaybackState.PLAYING and
                                                    e.item_id == "1"))
        self.muse.skip_to_previous(group_id=self.gid, hhid=self.zp.muse_hhid)
        self.verify_event(PlaybackStatusEvent, timeout=1,
                          event_verifier=lambda e: (e.playback_state == PlaybackState.PLAYING and
                                                    e.item_id == "0"))
        self.muse.pause(group_id=self.gid, hhid=self.zp.muse_hhid)
        self.verify_event(PlaybackStatusEvent, timeout=1,
                          event_verifier=lambda e: (e.playback_state==PlaybackState.PAUSED and
                                                    e.item_id == "0"))
        self.muse.play(group_id=self.gid, hhid=self.zp.muse_hhid)
        self.verify_event(PlaybackStatusEvent, timeout=1,
                          event_verifier=lambda e: (e.playback_state==PlaybackState.PLAYING and
                                                    e.item_id == "0"))
        self.muse.seek(session_id=self.muse.session_id, hhid=self.zp.muse_hhid,
                       item_id="0", position_millis=15000)
        self.verify_event(PlaybackStatusEvent, timeout=1,
                          event_verifier=lambda e: (e.playback_state==PlaybackState.PLAYING and
                                                    e.item_id == "0" and
                                                    e.position_millis >= 15000))
        self.muse.skip_to_item(session_id=self.muse.session_id,
                               hhid=self.zp.muse_hhid, queue_version='', item_id='2',
                               position_millis=0, play_on_completion=True)
        self.verify_event(PlaybackStatusEvent, timeout=1,
                          event_verifier=lambda e: (e.playback_state == PlaybackState.PLAYING and
                                                    e.item_id == "2"))
        self.muse_session_helper.remove_muse_session(self.muse, group_id=self.gid,
                                                     hhid=self.zp.muse_hhid, test_fixture=self)

        for type in event_types.keys():
            wait_until_true(lambda: len(self.get_actual_event_content(event=(type,), targetId=zp_udp,
                                                                      targetId_param=id_types[type])) > 0,
                            timeout_seconds=60, iteration_delay=1,
                            reason='Waiting for control events ({})'.format(type))
            actualEventContent = self.get_actual_event_content(event=(type,),
                                                               targetId=zp_udp,
                                                               targetId_param=id_types[type])
            self.verifyGreaterOrFailCase(len(actualEventContent), 0,
                                         "Expect at least one control event")

            if len(actualEventContent) > 0:
                myEvent = self.collect_usage_metric_events(actualEventContent, type,
                                                           ('largeSyncError', 'playback',
                                                            'transportError', 'cloudQueueError'))
                for expected_event in event_types[type]:
                    for actual_event in myEvent[type]:
                        if expected_event == actual_event[command_action_types[type]]:
                            self.verifyEqualOrFailCase(expected_event, actual_event[command_action_types[type]],
                                                       "Expected and actual event command should match")
                            break
                    else:
                        self.fail("Did not find expected usage metrics event "
                                  "after issuing command ({}: <{}>)".format(command_action_types[type],
                                                                            expected_event))

[docs]    def test_interval_tq_reporting(self):
        """
        Verify that the player only sends a track queue size event to the
        report uploader no more than once per reporting interval
        (default 1 minute).

        Multiple queue edits within 60 seconds will be combined into
        a single event, which will not be reported until another
        queue event occurs after the 60 second period.

        See https://confluence.sonos.com/display/PDSW/Reporting+track+queue+summary+event-based+metrics
        """
        # Set the version poll rate while paused/playing to 3 seconds on the player
        if not self.zp.diag.has_testpoint_support():
            self.skip('set_cloudqueue_poll_timers needs builds with testpoint support')
        else:
            self.cq_control.set_cloudqueue_poll_timers(zp=self.zp,
                                                       playing_timer=3,
                                                       paused_timer=3)
        # cleanup
        self.zp.AVTransport.clear_play_state()
        self.zp.AVTransport.clear_queue()

        # Clear the api log
        self.cq_control.clear_log('api')

        # Perform various edit operations on the track (add, delete)
        # First enqueue five tracks. (ehw=5)
        self.enqueue_five_tracks_in_shared(self.zp, self.track_source)
        # Delete the first two (3 tracks left)
        self.zp.AVTransport.RemoveTrackRangeFromQueue(1, 2)
        # Add one, the total number in the queue should be 4 (eu=4)
        self.zp.AVTransport.add_to_queue(uri=
            self.track_source.generator_uri().next())

        # Sleep for the tqueue reporting interval + 20 second leeway to
        # avoid race
        sec = self.TQUEUE_REPORTING_INTERVAL + 10
        self.logger.info("Sleeping {} seconds to wait for reporting "
            "interval to pass...".format(sec))
        guaranteed_sleep(sec)

        # Add one track so the previous report is sent out.
        self.zp.AVTransport.add_to_queue(uri=
            self.track_source.generator_uri().next())

        # Find the event
        wait_until_true(lambda: len(self.get_actual_event_content(event=('trackQueueSummary',))) > 0,
                        timeout_seconds=30,
            reason='Waiting for track queue summary events (trackQueueSummary)')
        event_content = self.get_actual_event_content(event=('trackQueueSummary',))

        # Makes ure we got at least one event
        self.verifyGreaterEqualOrFailCase(len(event_content), 1,
            "Expected number of UsageMetrics reports generated? (at least 1)")

        if len(event_content) > 0:
            # Collect fields from all UM events and find the trackQueueSummary field
            event = self.collect_usage_metric_events(event_content, 'trackQueueSummary',
                ('ehw', 'eu', 'strhw', 'stru'))['trackQueueSummary'][0]
            # Convert string fields into integers
            for key in event:
                event[key] = int(event[key])

        self.verifyEqualOrFailCase(5, event['ehw'], "Queue entries high "
            "watermark for trackQueueSummary is correct?")
        self.verifyEqualOrFailCase(4, event['eu'], "Number of queue entries "
            "used correct?")
        self.verifyGreaterOrFailCase(event['strhw'], 0, "String table size "
            "high watermark is non-zero?")
        self.verifyGreaterOrFailCase(event['stru'], 0, "String table size "
            "used is non-zero?")
