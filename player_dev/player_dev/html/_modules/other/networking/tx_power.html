        self.supported_devices = [
            zp for zp in self.my_devices if not

                                      "requires at least 1x Amphenol Play:3")
        self.verifyGreaterEqualOrStop(len([zp for zp in p3_list
                                           if not zp.is_play3_amphenol()]), 1,
                                      "requires at least 1x whaYu Play:3")
        # ensure we have Play:1 with AN and IAC radios
        p1_list = [zp for zp in self.supported_devices if
                   zp.is_s1() or zp.is_s12()]
        self.verifyGreaterEqualOrStop(

            "requires at least 3x Play:1s (2 S1 and 1 S12)")

                                      "requires at least 1x IAC Play:1")
        self.verifyGreaterEqualOrStop(
            len([zp for zp in p1_list if not zp.is_play1_iac()]),
            2,
            "requires at least 2x non IAC Play:1 (1 S1 and 1 S12)")

    def _get_tx_power_register(self, zp, wireless_mode, netif):
        """
        Retrieves the TX Power register for given WiFi radio and wireless mode

        :param zp: desired zp to get tx power register
        :type zp: :class:'sonos.client.internal.SonosZoneComponent'
        :param wireless_mode: 802.11 WiFi mode (B/G/N)
        :type wireless_mode: :obj:'str'
        :param netif: wireless interface ath0 or ath1
        :type netif: :obj:'str'
        :returns: TX Power register for given radio and 802.11 wireless mode
        :rtype: :obj:'str'
        """
        radio = zp.get_radio_type(netif)
        if radio in RADIO_TYPE_9xxx:
            self.logger.info("{} has a 95xx or 93xx radio".
                             format(zp.modelName))
            if wireless_mode == WIFI_80211_B:
                return RADIO_9xxx_80211_B_REG
            elif wireless_mode == WIFI_80211_G:
                return RADIO_9xxx_80211_G_REG
            elif wireless_mode == WIFI_80211_N:
                return RADIO_9xxx_80211_N_REG
        elif radio is RADIO_5416:
            self.logger.info("{} has a 5416 radio".format(zp.modelName))
            if wireless_mode == WIFI_80211_B:
                return RADIO_5416_80211_B_REG
            elif wireless_mode == WIFI_80211_G:
                return RADIO_5416_80211_G_REG
            elif wireless_mode == WIFI_80211_N:
                return RADIO_5416_80211_N_REG
        elif radio is RADIO_5212:
            self.logger.info("{} has a 5212 radio".format(zp.modelName))
            if wireless_mode == WIFI_80211_B:
                return RADIO_5212_80211_B_REG
            elif wireless_mode == WIFI_80211_G:
                return RADIO_5212_80211_G_REG
        else:
            self.logger.error("{} - {} tx power register not available".
                              format(zp.modelName, wireless_mode))

    def _get_freq_table(self, zp, region, wireless_mode):
        """
        Returns the frequency table for given ZP, region and 802.11 mode

        :param zp: desired zp to get the frequency table
        :type zp: :class:'sonos.client.internal.SonosZoneComponent'
        :param region: region code to change (codes -- 1:US, 2:EU, 3:CN, 4:JP)
        :type region: :class:`~sonos.client.cli.mdputil.MdpRegion`
        :param wireless_mode: 802.11 WiFi mode (B/G/N)
        :type wireless_mode: :obj:'str'
        :returns: frequency table for given ZP, region and 802.11 mode
        :rtype: :class: 'sonos.environment.WorkflowExecutionConfigurationList'
        """
        freqs = getattr(self.execution_configuration.freqs_2ghz,
            region.name)[:]
        if zp.is_5ghz_supported() and wireless_mode is WIFI_80211_N:
            freqs += getattr(
                self.execution_configuration.freqs_5ghz,
                region.name)
        return freqs

[docs]    def set_channel(self, zp, region_ch_freq, netif):
        """
        Sets the channel frequency

        :param zp: desired zp to set the channel
        :type zp: :class:'sonos.client.internal.SonosZoneComponent'
        :param region_ch_freq: The freq of the desired region
        :type region_ch_freq: :obj:'int'
        :param netif: wireless interface ath0 or ath1
        :type netif: :obj:'str'
        """
        region = zp.cli.mdp.region
        self.logger.info("ZP: {} - Setting wifi channel to: {}".format(zp,
            region_ch_freq))
        # allow up to 100 ms for channel change, for ht_masters in 5GHz give a
        # bit more time to stop intermittent ioctl failures
        if zp.is_ht_master_capable() and is_5ghz_channel(region_ch_freq):
            time.sleep(0.4)
        else:
            # this may be dirty but it prevents ath1 calls on ath0 only zps
            netif = 'ath0'
        for attempt in range(1, 4):
            if zp.cli.change_wireless_channel(netif, region_ch_freq):
                break
            elif attempt == 3:
                self.fail('Failed {} times to set; {} channel: {}, region:'
                          ' {}, netif: {}'.format(attempt, zp, region_ch_freq,
                                                  region, netif))



[docs]    def get_power_setting(self, zp, region, wireless_mode, netif, freq):
        """
        Gets the tx power setting for the given zp, frequency, and wireless
        mode

        :param zp: desired zp to verify the channel change
        :type zp: :class:'sonos.client.internal.SonosZoneComponent'
        :param region: region code to change (codes -- 1:US, 2:EU, 3:CN, 4:JP)
        :type region: :class:`~sonos.client.cli.mdputil.MdpRegion`
        :param wireless_mode: 802.11 WiFi mode (B/G/N)
        :type wireless_mode: :obj:'str'
        :param freq: frequency to be verified
        :type freq: :obj:'int'
        :param netif: wireless interface ath0 or ath1
        :type netif: :obj:'str'
        :returns: The current power setting
        :rtype: :obj:'int'
        """
        # getting register and values from ZP
        reg = self._get_tx_power_register(zp, wireless_mode, netif)
        self.logger.info("Verifying tx power on {} - region: {!r}, freq: {}, "
                         "wireless_mode: {}, register: {}"
                         .format(zp.modelName, region, freq,
                                 wireless_mode, reg))
        if zp.is_ht_master_capable():
            Athconfig(zp).psenable(INTERFACE_5GHZ, 0)
        reg_utility = Reg(zp)
        dbm = 0
        for _ in range(3):
            dbm = reg_utility.get_reg_at_offset(netif,
                                                safe_convert_value(reg),
                                                as_int=True)
            dbm = (dbm >> 24) / float(2)
            # Dock register values are offset by 5
            if zp.cli.is_dock():
                dbm = -5
            if dbm:
                return dbm
            self.logger.warning('{} returned power value {} for freq: {}, '
                'region: {!r}, mode: {}, netif: {}, retrying'.format(zp, dbm,
                    freq, region, wireless_mode, netif))
            time.sleep(1)


    def _get_expected_tx_power(self, model, orientation, region, wifi_mode,
            frequency):
        return self.execution_configuration.tx_power.dict_tree_copy(
            )[model][orientation][region.name][wifi_mode][str(frequency)]

    def _verify_tx_power(self, zp, region, wireless_mode, orientation, freq,
                         netif):
        """
        Verify the tx power in the ZP for given freq and 802.11 wireless mode
        against defined dbm values in the config file

        :param zp: desired zp to verify the channel change
        :type zp: :class:'sonos.client.internal.SonosZoneComponent'
        :param region: region code to change (codes -- 1:US, 2:EU, 3:CN, 4:JP)
        :type region: :class:`~sonos.client.cli.mdputil.MdpRegion`
        :param wireless_mode: 802.11 WiFi mode (B/G/N)
        :type wireless_mode: :obj:'str'
        :param orientation: device orientation to verify tx power of
        :type orientation: :obj:'str'
        :param freq: frequency to be verified
        :type freq: :obj:'int'
        :param netif: wireless interface ath0 or ath1
        :type netif: :obj:'str'
        """
        # getting defined dbm values from config file
        if zp.is_play3_amphenol():
            model = zp.modelNumber.lower() + '_amphenol'
        elif zp.is_play1_iac():
            model = zp.modelNumber.lower() + '_iac'
        elif zp.cli.is_connectx_connect():
	    model = zp.modelNumber.lower() + '_connectx'
	else:
            model = zp.modelNumber.lower()
        expected = self._get_expected_tx_power(model,
            orientation,
            region,
            wireless_mode,
            freq)
        # int 0 means we have no tx power data
        if expected is 0:
            self.logger.info(
                "DO NOT HAVE TX POWER DATA FOR {} - {}, {}".format(
                    zp.modelName, freq, wireless_mode))
        elif expected is not None:
            if (zp.cli.is_ppc() or zp.cli.is_sh4()) and all([int(freq) == 2442,
                    wireless_mode == WIFI_80211_B, region == EU]):
                dbm = self.get_power_setting(zp, region, wireless_mode, netif,
                                             freq)
                self.verifyEqualOrFailCase(13.0, dbm,
                    '{}: {}: {}: {}: legacy ppc and sh4: tx power for '
                    'region 2, 11b, ch7 is 13'.format(zp, region, freq,
                                                      orientation))
            else:
                dbm = self.get_power_setting(zp, region, wireless_mode, netif,
                                             freq)
                self.verifyEqualOrFailCase(expected, dbm,
                    "{}: {}: {}: {}: tx power should match values in "
                    "tx power data".format(zp, region, freq, orientation))
        else:
            self.stop("Failed to load expected data for {}".format(
                zp.modelName, wireless_mode, freq))

    def _perform_tx_power_test(self, zp, region):
        """
        Perform the tx power test

        :param zp: desired zp to verify the channel change
        :type zp: :class:'sonos.client.internal.SonosZoneComponent'
        :param region: region code to change (codes -- 1:US, 2:EU, 3:CN, 4:JP)
        :type region: :class: '__main__.Region'
        """
        # make sure we can set to DFS channels for ht_masters
        if zp.is_ht_master_capable():
            zp.cli.athconfig_cmd('nolenable ath1 0')
        # make sure that we get default power from br200 as valiadate test bed
        # now reduces it for brick test beds
        if zp.is_br200():
            zp.cli.athconfig_cmd('settxpower ath0 0')
        zp_orientations = self.execution_configuration.\
            tx_power.as_dict()[zp.modelNumber.lower()].as_dict().keys()
        for orientation in zp_orientations:
            if orientation != 'default':
                self.logger.info('setting zone {} orientation to {} for '
                                 '{!r}'.format(zp, orientation, region))
                zp.cli.set_device_orientation(orientation)
            for mode in WIFI_MODES:
                self.logger.info("Initiating {} tx power test".format(mode))
                freq_table = self._get_freq_table(zp, region, mode)
                self.logger.debug(freq_table)
                for freq in freq_table:
                    # handle playbar, solbase differently as they have 2
                    # wireless interfaces
                    if is_5ghz_channel(freq) and zp.is_ht_master_capable():
                        wifi_interface = INTERFACE_5GHZ
                    else:
                        wifi_interface = INTERFACE_2GHZ
                    self.set_channel(zp, safe_convert_value(freq),
                                     wifi_interface)
                    self._verify_tx_power(zp, region, mode, orientation,
                                          safe_convert_value(freq),
                                          wifi_interface)
        self.logger.info("Completed 802.{} for {}".format(mode, zp))
