    def _wait_for_cac(self, freq):
        """
        Private method that waits for Channel Availability Check (CAC).
        The maximum CAC time is 60 sec.
        Anything after 65 sec indicates the channel was forcefully changed
        during CAC.
        Notifies if radar is detected.

        Arguments:
        freq -- str, channel frequency

        Keyword Arguments:
        None

        Returns:
        None
        """
        timeout = 65
        start_time = time.time()
        stop_time = start_time
        while self.s9.cli.get_advertising_channel(freq) == NOTHING:
            self.s9.cli.get_advertising_channel(freq)
            self.logger.info("CAC in progress...")
            if self.s9.cli.detect_radar(freq) != NOTHING:
                self.logger.warn("Radar detected on {}".format(freq))
                break
            new_time = time.time()
            delta = new_time - stop_time
            if delta > timeout:
                now_channel = self.s9.cli.get_wireless_channel('ath1')
                self.verifyLessOrFailCase(delta,
                    timeout,
                    ("Channel should not change during CAC. Should be: "
                        "{}, but is now {}".format(freq, now_channel)))
                break

    def _verify_channel_change(self, freq):
        """
        Private method that verifies channel change.  If channel change is not complete, wait and verify again until timeout is reached.

        Arguments:
        freq -- str, channel frequency

        Keyword Arguments:
        None

        Returns:
        None
        """
        timeout = 10
        start_time = time.time()
        stop_time = start_time
        advertising_channel = self.s9.cli.get_advertising_channel(freq)
        get_channel = [self.sat_lr.cli.get_wireless_channel(INTERFACE_2GHZ),
            self.sat_rr.cli.get_wireless_channel(INTERFACE_2GHZ)]
        if advertising_channel != NOTHING:
            for i in range(len(get_channel)):
                while get_channel[i] != advertising_channel:
                    get_channel = [
                        self.sat_lr.cli.get_wireless_channel(INTERFACE_2GHZ),
                        self.sat_rr.cli.get_wireless_channel(INTERFACE_2GHZ)]
                    self.logger.debug(
                        "waiting for channel to change on satellite...")
                    new_time = time.time()
                    delta = new_time - stop_time
                    if str(get_channel[i]) in advertising_channel:
                        self.logger.debug("Satellite channel == Advertising Channel")
                        break
                    if delta > timeout:
                        self.logger.debug("Channel change performance failed")
                        break
                self.verifyEqualOrFailCase(str(freq) in advertising_channel, str(freq) in str(get_channel[i]),
                                          'Channels should match on sats: {} == {}'.format(advertising_channel, get_channel[i]))
        else:
            self.fail("advertising channel did not change")

    def _set_channel(self, region_ch_freq, interface):
        """
        Private method that sets desired channel, waits for CAC, detect if radar is present and verifies
        channel change

        Arguments:
        region_ch_freqs -- str, list of the available channel frequencies for
        partiular region

        Keyword Arguments:
        None

        Returns:
        None
        """
        # disable Link Monitoring so channels do not change on us if quality of
        # wireless link is bad
        self.s9.cli.athconfig_cmd('acslmenable ath1 0')
        for ch in range(len(region_ch_freq)):
            region_ch_freq[ch] = str(region_ch_freq[ch])
            self.logger.info("Setting wifi channel to: {}".format(
                region_ch_freq[ch]))
            # change from unicode to str as change_wireless_channel requires str
            if self.s9.cli.change_wireless_channel(interface,
                    region_ch_freq[ch]):
                self._wait_for_cac(region_ch_freq[ch])
                # verify channel change on the sats
                self._verify_channel_change(region_ch_freq[ch])
            else:
                self.logger.warn("{} is on NOL".format(region_ch_freq[ch]))

    def _change_satellite_region(self, region):
        """
        Private method that sets the region of the satellite

        Arguments:
        region_ch_freqs -- str, list of the available channel frequencies for
        partiular region

        Keyword Arguments:
        None

        Returns:
        None
        """
        # for country in REGIONS:
        for sat in self.sats:
            self.logger.info("Setting satellite mdp region to "
                "{!r} on {}".format(region, sat))
            if sat.cli.mdp.set_region(region):
                self.logger.info("mdp region changed on {}".format(sat))
            else:
                self.logger.error("Failed to change region code to "
                    "{!r} on {}".format(region, sat))
            self.subnet.reboot_devices(self.sats)

    def _perform_mixed_region_test(self, zp, region):
        freqs_5ghz = getattr(self.execution_configuration.freqs_5ghz,
                                region.name)
        self.change_home_region(zp, region)
        for region in MdpRegion:
            self._change_satellite_region(region)
            self._set_channel(freqs_5ghz, INTERFACE_5GHZ)
