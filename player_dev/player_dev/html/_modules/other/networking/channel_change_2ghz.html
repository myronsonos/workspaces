[docs]    def check_ch12_and_ch13(self, zp, interface):
        """
        Verifies channel 12 and 13 are not supported in the wifi driver for
        USA and CHINA

        Arguments:
        zp -- SonosZoneComponent, DUT
        interface -- str, wifi interface

        Keyword Arguments:
        None

        Returns:
        boolean
        """
        res = False
        self.logger.info("\n\nZP: %s - Verifying channels 12 and 13" % zp)
        supported_channels = zp.cli.get_supported_channels(interface)
        self.logger.debug(supported_channels)
        if CH12 and CH13 not in supported_channels:
            res = True
        else:
            return res
        return res

[docs]    def verify_channel_change(self, zp, channel):
        """
        Verifies if channel change was successful

        Arguments:
            SonosZoneComponent - zp
            string - interface

        Keyword Arguments:
            None

        Returns:
            None
        """
        get_operating_freq = zp.cli.get_wireless_channel(INTERFACE_2GHZ)
        self.verifyTrueOrFailCase(channel in get_operating_freq,
            "expect desired channel: %s == operating freq: %s" % (channel,
                get_operating_freq))



[docs]    def set_channel(self, zp, region_ch_freq):
        """
        Sets the desired channel frequency

        Arguments:
            SonosZoneComponent - zp
            string - region_ch_freq, region's frequency

        Keyword Arguments:
            None

        Returns:
            None
        """
        #change from unicode to str as change_wireless_channel requires str
        region_ch_freq = str(region_ch_freq)
        self.logger.info("ZP: %s - Setting wifi channel to: %s" % (zp,
            region_ch_freq))
        if zp.cli.change_wireless_channel('ath0', region_ch_freq):
            #allow 2 sec for channel change to happen
            time.sleep(2)
            self.verify_channel_change(zp, region_ch_freq)
        else:
            self.fail("%s channel is not available" % region_ch_freq)



[docs]    def change_home_region(self, zp, region):
        """
        Sets the home region of the Sonos device

        Arguments:
            SonosZoneComponent - zp
            MdpRegion - region, region code.  1 - USA, 2 - Europe, 3 - China,
                4 - Japan

        Keyword Arguments:
            None

        Returns:
            None
        """
        self.logger.info("ZP: %s - Setting home mdp region to: %s" % (zp,
            region))
        if zp.cli.mdp.set_region(region):
            #reboot needed after changing mdp region
            self._reboot(zp)
        else:
            self.logger.error("Failed to change region code to %s on %s" % (
                region, zp))


    def _get_radio(self, zp):
        """
        Return the wifi radio

        Arguments:
            SonosZoneComponent - zp

        Returns:
            string - wifi radio number
        """

        if zp.is_br200():
            return RADIO_9558
        elif zp.is_s1():
            return RADIO_9582
        elif zp.is_s3() or zp.is_soundbar():
            return RADIO_9380
        elif zp.is_sub():
            return RADIO_9382
        elif zp.is_s5() or zp.is_zp120() or zp.is_zp90() or zp.is_dock():
            return RADIO_5416
        elif zp.is_zp80() or zp.is_zp100() or zp.is_br100():
            return RADIO_5212
        else:
            self.logger.error("ZP has no supported radio")

    def _verify_tx_power(self, zp, region, freq, netif):
        """
        Verify the tx power in the ZP for given freq and mode against defined dbm values in the config file

        Arguments:
            SonosZoneComponent - zp
            string - mode, rate (ex: 11b, 11g, 11n)
            string - netif, interface

        Returns:
            None
        """
        #type casting freq to a string
        freq = str(freq)
        #We have different tx register values for 93xx/95xx radios
        if (self._get_radio(zp) is RADIO_9582 or
                self._get_radio(zp) is RADIO_9380 or
                self._get_radio(zp) is RADIO_9382):
            self.logger.info("%s has a 93xx or 95xx radio" % zp)
            reg_list = [{WIFI_80211_B:RADIO_9xxx_80211_B_REG},
                {WIFI_80211_G:RADIO_9xxx_80211_G_REG},
                {WIFI_80211_N:RADIO_9xxx_80211_N_REG}]
            for i in range(len(reg_list)):
                mode = reg_list[i].keys()[0]
                reg = reg_list[i].values()[0]
                self.logger.info("Verifying tx power on %r: %s, freq: %s, "
                    "mode: %s, register: %s" % (zp.modelName,
                        region,
                        freq,
                        mode,
                        reg))
                tx_power = zp.cli.get_tx_power(freq, netif, reg)
                dbm = tx_power[0]['dbm']

                expected = self._get_expected_dbm(zp, region, mode, freq)
                #int 0 means we have no tx power data
                if expected is 0:
                    self.logger.info("DO NOT HAVE TX POWER DATA FOR "
                        "%s - %s, %s" % (zp.modelName, freq, mode))
                elif expected is not None:
                    self.verifyEqualOrFailCase(expected, dbm,
                        "tx power should match values in tx power data")

        elif self._get_radio(zp) is RADIO_5416:
            #We have different tx register values for 5416 radios
            self.logger.info("%s has a 5416 radio" % zp)
            reg_list = [{WIFI_80211_B:RADIO_5416_80211_B_REG},
                {WIFI_80211_G:RADIO_5416_80211_G_REG},
                {WIFI_80211_N:RADIO_5416_80211_N_REG}]
            for i in range(len(reg_list)):
                mode = reg_list[i].keys()[0]
                reg = reg_list[i].values()[0]
                self.logger.info("Verifying tx power on %s: %r, freq: %s, "
                    "mode: %s, register: %s" % (zp.modelName,
                        region,
                        freq,
                        mode,
                        reg))
                tx_power = zp.cli.get_tx_power(freq, netif, reg)
                dbm = tx_power[0]['dbm']

                expected = self._get_expected_dbm(zp, region, mode, freq)
                #Hook to handle data we do not have.
                if expected is 0:
                    self.logger.warn("DO NOT HAVE TX POWER DATA FOR "
                        "%s - %s, %s" % (zp.modelName, freq, mode))
                elif expected is not None:
                    self.verifyEqualOrFailCase(expected, dbm,
                        "tx power should match values in tx power data")

        elif self._get_radio(zp) is RADIO_9558:
