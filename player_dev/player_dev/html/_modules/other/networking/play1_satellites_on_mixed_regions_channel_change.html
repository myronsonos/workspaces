        # get all HT devices
        test_models = getattr(self.sat_config, "test_models", None)
        self.verifyIsNotNoneOrStop(test_models,
            "no 'test_models' config found in 'satellites'")

        self.satellites = [ zp for zp in self.ht_devices

        # remove any existing ht configs
        self.helpers.zones.cleanup_device_configurations(self.ht_devices)
        # Create HT system according to bonded config
        self._ht_config(self.bonded_configs)

    def _reboot(self, zp):
        self.logger.info("rebooting: %s" % zp)
        self.subnet.reboot_devices([zp])

    def _ht_config(self, htconfigs):
        """
        Private method that creates a 5.1 Home Theater configuration

        Arguments:
        htconfigs -- WorkFlowExecutionConfiguration list

        Keyword Arguments:
        None

        Returns:
        boolean
        """
        self.logger.info("Adding HT Satellites")
        for zp in htconfigs:
            self.logger.debug("setting up configuration: %s" % zp)
            if zp.is_bonded():
                self.helpers.zones.setup_device_configuration(zp,
                    self.ht_devices)
            else:
                self.logger.error("Incorrect bonded config")

    def _wait_for_cac(self, freq):
        """
        Private method that waits for Channel Availability Check (CAC).
        The maximum CAC time is 60 sec.
        Anything after 65 sec indicates the channel was forcefully changed
        during CAC.
        Notifies if radar is detected.

        Arguments:
        freq -- str, channel frequency

        Keyword Arguments:
        None

        Returns:
        None
        """
        timeout = 65
        start_time = time.time()
        stop_time = start_time
        while self.s9.cli.get_advertising_channel(freq) is None:
            advertising_channel = self.s9.cli.get_advertising_channel(freq)
            self.logger.debug(advertising_channel)
            self.logger.debug("CAC in progress...")
            time.sleep(5)
            if self.s9.cli.detect_radar(freq):
                self.logger.warn("Radar detected on {}".format(freq))
                break
            new_time = time.time()
            delta = new_time - stop_time
            if delta > timeout:
                now_channel = self.s9.cli.get_wireless_channel('ath1')
                self.verifyLessOrFailCase(delta, timeout,
                    "Channel should not change during CAC. Should be: {}, "
                    "but is now {}".format(freq, now_channel))
                break

    def _verify_channel_change(self, freq, timeout=10):
        """
        Private method that verifies channel change.  If channel change is
        not complete, wait and verify again until timeout is reached.

        Arguments:
        freq -- str, channel frequency

        Keyword Arguments:
        None

        Returns:
        None
        """
        start_time = time.time()
        advertising_channel = self.s9.cli.get_advertising_channel()
        sat_sub_get_channel = [
            ('LR', lambda: self.sat_lr.cli.get_wireless_channel('ath0')),
            ('RR', lambda: self.sat_rr.cli.get_wireless_channel('ath0')),
            ('SUB', lambda: self.sub.cli.get_wireless_channel('ath0'))]
        if advertising_channel is not None:
            for sat_sub_id, get_channel in sat_sub_get_channel:
                while get_channel() != advertising_channel:
                    self.logger.debug("Waiting for channel to change to {} on "
                        "{}".format(advertising_channel, sat_sub_id))
                    if time.time() - start_time > timeout:
                        if not self.assertEqual(advertising_channel,
                                get_channel(), 'Verify {} advertising channel '
                                'change'.format(sat_sub_id)):
                            return
            self.logger.log(AssertionLevel.PASS, "All HT members changed "
                "to advertising channel <{}>".format(advertising_channel))
        else:
            self.fail("advertising channel did not change")

    def _set_wifi_channels(self, region_ch_freqs):
        """
        Private method that sets desired channel, waits for CAC, detect if
        radar is present and verifies channel change

        Arguments:
        region_ch_freqs -- str, list of the available channel frequencies for
            partiular region

        Keyword Arguments:
        None

        Returns:
        None
        """
        # disable Link Monitoring so channels do not change on us if quality
        # of wireless link is bad
        self.s9.cli.athconfig_cmd('acslmenable ath1 0')
        for freq in region_ch_freqs:
            self.logger.info("Setting wifi channel on {} to: {}".format(
                self.s9, freq))
            if self.s9.cli.change_wireless_channel('ath1', freq):
                self._wait_for_cac(freq)
                self._verify_channel_change(freq)
            else:
                self.logger.warn("{} is on NOL" .format(freq))

    def _change_satellite_region(self, region_ch_freqs):
        """
        Private method that sets the region of the satellite

        :param region_ch_freqs: available channel frequencies for
            particular region
        :type region_ch_freqs: :obj:`list`
        """
        for region in MdpRegion:
            for sat in self.satellites:
                self.logger.info("Setting satellite mdp region to {} on "
                    "{}".format(region, sat))
                if sat.cli.mdp.set_region(region):
                    self.logger.info("mdp region changed on {}".format(sat))
                else:
                    self.logger.error("Failed to change region code to {} on "
                    "{}".format(region, sat))
            # reboot needed after changing mdp region
            self.subnet.reboot_devices(self.satellites)
            self._set_wifi_channels(region_ch_freqs)
            self.logger.info("Completed region: {} on satellites {}".format(
                region, self.satellites))
            # Rebooting PLAYBAR to reset CAC and NOL for DFS channels
            self._reboot(self.s9)

    def _change_home_region(self, zp, region):
        """
        Private method that sets the home region of the Sonos device

        Arguments:
        :param region: region code.  1 - USA, 2 - Europe, 3 - China, 4 - Japan

        """
        self.logger.info("Setting home mdp region to {} on {}".format(region,
            zp))
        if zp.cli.mdp.set_region(region):
            self.logger.info("home mdp successfully set, rebooting...")
            # reboot home device
            self._reboot(self.s9)
        else:
            self.logger.error("Failed to change region code to {} on {}".format(
                region, zp))
