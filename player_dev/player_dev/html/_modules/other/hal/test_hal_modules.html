[docs]    def get_dmesg_errors(self, zp):
        """
        Get dmesg errors.
        
        Arguments:
            SonosZoneComponent zp
        
        Returns:
            dict - dictionary of dmesg error : error text
        """
        dmesg_buffer = zp.diag.get_dmesg()
        dmesg_errors = []
        for entry in dmesg_buffer:
            for error in DMESG_ERRORS:
                if error in entry ['log']:
                    dmesg_errors.append(entry.copy())
        
        # Remove duplicate entries
        dmesg_errors = [ dict(tupleized) for tupleized in set(tuple(
            item.items()) for item in dmesg_errors) ]
        for error in dmesg_errors:
            self.logger.info("Error found in dmesg log on {}:{} {}".format(
                             zp, error['t_stamp'], error['log']))
    
        return dmesg_errors

    

[docs]    def verify_no_dmesg_errors(self, zp):
        """
        Verify no dmesg errors.
        
        Arguments:
            SonosZoneComponent zp
        """
        dmesg_errors = self.get_dmesg_errors(zp)
        has_no_dmesg_errors = bool(len(dmesg_errors) == 0)
        self.verifyTrueOrFailCase(has_no_dmesg_errors, 'Verify no dmesg'
                                  ' errors for DUT <{}>'.format(zp))
        if not has_no_dmesg_errors:
            self.dump_dmesg(zp)

[docs]    def module_order_cmp(self, a, b):
        """
        Compare a and b, returning < 0 if a > b, 0 if a == b, 1 if a > b.
        Used to sort the list of modules returned by lsmod.
        
        Arguments:
            SonosZoneComponent zp
        """
        # Order in which modules must be installed (reverse for removing modules)
        ath_modules = ['ath_driver', 'ath_hal', 'dfs', 'asf', 'adf']
        return ath_modules.index(a) - ath_modules.index(b)

    

[docs]    def get_configurable_modules(self, zp):
        """
        Get all modules that have .ko files in wifi/N/
        
        Arguments:
            SonosZoneComponent zp
        
        Returns:
            list - list of .ko files (not including the .ko extension)
        """
        return re.findall('.+(?=\.ko)', zp.cli.command('ls {} | grep .ko'.format(MODULE_DIR)))[1:]

    

[docs]    def get_installed_modules(self, zp):
        """
        Get module names that appear in lsmod
        
        Arguments:
            SonosZoneComponent zp
        
        Returns:
            list - list of module names
        """
        # This could probably be less complicated
        return map(lambda mod: mod.group('mod'),
                   filter(lambda mod: mod if mod != None else None,
                          map(lambda mod: re.search('(?P<mod>^[^ #]+)', mod), 
                              zp.cli.command('lsmod').splitlines()[1:])))

    

[docs]    def install_module(self, zp, module):
        """
        Install the module if (a) the .ko file exists and (b) it is not already installed.
        Verify the module is installed by checking lsmod after calling insmod.
        
        Arguments:
            SonosZoneComponent zp
            string module - module name
        """
        if module not in self.get_configurable_modules(zp):
            self.logger.error('<{}>: module {} does not exist in {}'.format(zp, module, MODULE_DIR))
        elif module in self.get_installed_modules(zp):
            self.logger.error('<{}>: module {} is already installed'.format(zp, module))
        else:
            self.logger.info('<{}>: installing module {}'.format(zp, module))
            zp.cli.command('insmod {}{}{}'.format(MODULE_DIR, module, MODULE_EXTENSION))
            self.verifyTrueOrFailCase(module in self.get_installed_modules(zp),
                                      '<{}>: module {} successfully installed?'.format(zp, module))

    

[docs]    def remove_module(self, zp, module):
        """
        Uninstall the module if (a) the .ko file exists and (b) it is installed.
        Verify the module is uninstalled by checking lsmod after calling insmod.
        
        Arguments:
            SonosZoneComponent zp
            string module - module name
        """
        if module not in self.get_configurable_modules(zp):
            self.logger.error('<{}>: module {} does not exist in {}'.format(zp, module, MODULE_DIR))
        elif module not in self.get_installed_modules(zp):
            self.logger.error('<{}>: module {} is not installed'.format(zp, module))
        else:
            zp.cli.command('rmmod {}'.format(module))
            self.verifyTrueOrFailCase(module not in self.get_installed_modules(zp),
                                      '<{}>: module {} successfully uninstalled?'.format(zp, module))

    

[docs]    def install_configurable_modules(self, zp):
        """
        Install all modules (in the correct order) that have .ko files.
        
        Arguments:
            SonosZoneComponent zp
        """
        configurable_modules = self.get_configurable_modules(zp)
        
        # Install modules in reverse order
        configurable_modules.sort(cmp=self.module_order_cmp)
        configurable_modules.reverse()
        
        for module in configurable_modules:
            self.install_module(zp, module)



[docs]    def remove_configurable_modules(self, zp):
        """
        Uninstall all modules (in the correct order) that have .ko files.
        
        Arguments:
            SonosZoneComponent zp
        """
        configurable_modules = self.get_configurable_modules(zp)
        
        configurable_modules.sort(cmp=self.module_order_cmp)
        
        for module in configurable_modules:
            self.remove_module(zp, module)

[docs]    def test_combinatorial_reinstall_modules(self, generator_zp):
        """
        Clear dmesg ring buffer, then remove and readd all modules. After this, verify
        there are no dmesg errors. Runs for each zp in self.my_devices.
        """
        self.clear_dmesg_ring_buffer(generator_zp)
        
        self.remove_configurable_modules(generator_zp)
        self.install_configurable_modules(generator_zp)
        
        self.verify_no_dmesg_errors(generator_zp)
