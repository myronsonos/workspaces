[docs]    def setUpFixture(self):
        """
            Simple set up fixture. Establishes a 'master' zone player based on
            the config file containing a zone player with the
            key:value pair master:True. It reads a list of channels to test and
            creates a dictionary of optional arguments to pass to the utility.
        """
        self.logger = logging.getLogger(make_log_name([__name__]))
        self.logger.info('Setting up Pertest Test')

        # Get test master info from json
        self.logger.info('Getting Master')
        self.master = [zp for zp in
            self.execution_configuration.get_testbed_zone_players()
            if getattr(zp, 'master', None) == True]
        for zp in self.my_devices:
            if zp.serialNumber == self.master[0].serial_number:
                self.master_zp = zp
                break
        self.verifyGreaterOrStop(len(self.master), 0,
            'Expect at least one master')

        # Get 2Ghz frequencies from json
        self.two_ghz = self.execution_configuration.two_ghz_freq
        # Get common optional args as dict
        self.com_optionals = (
            self.execution_configuration.common_optional.as_dict())
        # Get receiver Optional args as dict
        self.r_optionals = self.execution_configuration.r_optional.as_dict()
        self.r_optionals = dict(self.r_optionals.items() +
            self.com_optionals.items())
        # Get transmitter optional args as dict
        self.t_optionals = self.execution_configuration.t_optional.as_dict()
        self.t_optionals = dict(self.t_optionals.items() +
            self.com_optionals.items())
        # Get DUT
        self.duts = [zp for zp in self.my_devices if zp != self.master_zp]

[docs]    def get_per_tool(self, zp):
        """
        Creates a directory to hold the tool and downloads the tool to this
        directory.

        Checks for a telnet connection before attempting to get the tool and
        will establish a connection if none exists.

        :param zp: The ZP needing the tool
        :type  zp: :class:`~sonos.client.internal.SonosZoneComponent`
        :return: True if the file exists on the ZP, otherwise False
        :rtype: :obj: 'bool'
        """
        tool = '{}pertest_{}'.format(PERTEST_WEB_LOCATION, zp.cli.get_arch())
        zp.cli.command('mkdir /jffs/per_test')
        zp.cli.command('wget -O {} {}'.format(PERTEST_ZP_LOCATION, tool))
        got_file = zp.cli.does_file_exist(PERTEST_ZP_LOCATION)
        if not got_file:
            raise sonos.exception.DeviceInformationError(
                'Could not download utility')
        zp.cli.command('chmod a+x {}'.format(PERTEST_ZP_LOCATION))




[docs]    def run_per_tool(self, zp, dtype, ath, channels, rates, tx_mac='',
            roomname='', mcs='', nframes='', framesize='', tx_power='',
            interval='', nsignalframes='', orient=''):
        """
        Runs the PER tool on the transmitter ZP.

        :param zp: The receiver ZP
        :type: zp: :class:`~sonos.client.internal.SonosZoneComponent`
        :param str dtype: The device type (transmitter or receiver)
            running the tool. '-t' for transmitter, '-r' for receiver
        :param str ath: Netif to use for tx/rx (ath0 or ath1)
        :param str channels: 2.4 GHz channels to tx/rx on (CSV) <ex. 2412,2417>
        :param str rates: Data rates to test (CSV) <ex.-54,0,1,15>
            Or <n> for n rates, <bg> for bg rates or <bgn> for all.
        :param: str tx_mac: The mac address of the applicable netif of the
            transmitting device
        :param: str roomname: The room name to use for the csv output file
            Must be =< 16 characters
        :param str mcs: MCS to use for signal frame <ex. -1 for 1 mbps (G)
            0 for MCS 0 (N-capable only)>, <default=-1>
        :param str nframes: Number of frames to tx/rx <default=200>
        :param str framesize: Size of frames to tx/rx <default=220>
        :param str tx_power: txpower values to test <ex. 18,17.5>
            If specified, must match on receiver and transmitter.
            ALTERNATIVELY: Enter <start_value>,<end_value>,<decrement>
            <ex. 18,10,-1> would test txpower 18,17,16,15,...,10
        :param str interval: Delay in us between sending frames <default=10>.
            It is not recommended to set this too low.
        :param str nsignalframes: Number of signal frames to send.
            Sending less signal frames will speed up the test but may make it
            less reliable.
        :param: str orient: The orientation (angle in degrees).
            Defaults to 0 if given empty string. Must be 0-360
        """

        # Verify the tool is installed
        if not (zp.cli.does_file_exist(PERTEST_ZP_LOCATION) or
                zp.cli.does_file_exist('per_test')):
            self.get_per_tool(zp)
        per_command = ['/jffs/per_test/per_test {} -a {} -c {} -d {}'.format(
            dtype, ath, channels, rates)]
        # Format the params to be used as cmdline args
        if roomname:
            per_command.append(' -R {}'.format(roomname))
        if tx_mac:
            per_command.append(' -m {}'.format(tx_mac))
        if mcs:
            per_command.append(' -M {}'.format(mcs))
        if nframes:
            per_command.append(' -n {}'.format(nframes))
        if framesize:
            per_command.append(' -f {}'.format(framesize))
        if tx_power:
            per_command.append(' -T {}'.format(tx_power))
        if orient:
            per_command.append(' -o {}'.format(orient))
        if interval:
            per_command.append(' -i {}'.format(interval))
        if nsignalframes:
            per_command.append(' -s {}'.format(nsignalframes))

        try:
            zp.cli.command('cd /jffs/per_test')
            zp.cli.command(''.join(per_command), wait_for_response=False)
            """
            Since this method is called back to back, once for the receiver and
            once for the transmitter, give time to ensure that the receiver
            is running before starting the transmitter.
            """
            if dtype == '-r':
                sleep(1)
        except Exception as e:
            self.logger.exception('Caught exception {} while attempting to '
                'run PER tool on {}'.format(e, zp))
            self.fail("run_per_tool failed on {}".format(zp))



[docs]    def parse_results(self, zp, name):
        '''
            Cats the csv file that is created on the receiver ZP. Parses the out
            and creates lists for each category.

            :param zp: component that has the csv file
            :type:
            :param str name: The 'roomname' passed to the perf_tool when the
                             test was run
            :return: A dictionary containing a PASS/FAIL result and lists of
                     containing failing PER, MCS, frame size, and RA values
            :rtype: :obj: 'dict'
        '''
        per_fail = []
        mcs_fail = []
        frame_size_fail = []
        ra_avg_fail = []
        failures_dict = {"results":"PASS", "per":per_fail, "mcs":mcs_fail,
                         "frame_size":frame_size_fail, "ra_avg":ra_avg_fail}

        # Verify the csv exists
        if not (zp.cli.does_file_exist('/jffs/per_test/RX_{}_0.csv'.format(
                name)) or zp.cli.does_file_exist('RX_{}_0.csv'.format(name))):
            raise sonos.exception.DeviceInformationError(
                'RX_{}_0.csv file does not exist'.format(name))

        results = zp.cli.command('cat /jffs/per_test/RX_{}_0.csv'.format(name))
        parsed_results = results.splitlines()
        parsed_results = parsed_results[2:-1]

        for item in parsed_results:
            str_item = str(item)
            str_item = str_item.split(',')
            per = float(str_item[2])

            if per > ACCEPTABLE_PER:
                per_fail.append(per)
                mcs_fail.append(int(str_item[4]))
                frame_size_fail.append(int(str_item[6]))
                ra_avg_fail.append(int(str_item[8]))
                failures_dict["results"] = "FAIL"
        return failures_dict

    def test_rx(self, two_ghz):
        """
        Cominatorial test to run the tool with the DUT as the receiving device

        :param two_ghz: A dict with the region under test and a list of
                        channels to test
        :type: two_ghz: `dict`

        """
        master_mac = self.master_zp.cli.get_mac_address(INTERFACE_2GHZ).strip()
        channel_list = two_ghz["channels"]
        region = two_ghz["region"]

        for item in product(self.duts, channel_list):
            zp = item[0]
            channel = str(item[1])
            self.change_home_region(zp, region)
            self.change_home_region(self.master_zp, region)
            output_file = OUTPUT_FILE + channel

            if zp.is_n_capable():
                data_rate = DATARATES_BGN
                mcs_value = MCS_0
            else:
                data_rate = DATARATES_BG
                mcs_value = ''

            self.run_per_tool(zp, '-r', INTERFACE_2GHZ, channel, data_rate,
                master_mac, output_file, mcs_value, **self.r_optionals)
            self.run_per_tool(self.master_zp, '-t', INTERFACE_2GHZ, channel,
                          data_rate, mcs=mcs_value, **self.t_optionals)
            results = self.parse_results(zp, output_file)
            self.verifyTrueOrFailCase(results["results"] == "PASS",
                'PER should be under {} percent'.format(ACCEPTABLE_PER * 100))
            if results["results"] == "FAIL":
                self.logger.info('Failure in {!r} on {}'.format(region,
                    zp))
                for i in range(len(results["per"])):
                    self.logger.info(
                        ('Failures seen at freq: {} PER: {:<9} MCS: {:>3} '
                                'FRAME_SIZE: {:<4} RA AVG: {} ').format(
                            channel,
                            results["per"][i],
                            results["mcs"][i],
                            results["frame_size"][i],
                            results["ra_avg"][i]))
