[docs]    def verify_regs(self, zp, registers, dumpregs, dmp_regs_option=''):
        """
        Verifies that the given register information is returned when dumpregs
        is run.

        :param zp: The zone player under test
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`
        :param registers: The list of registers to check
        :type registers: :obj:`list` of :obj:`str`
        :param dumpregs: An instance of the dumpregs class, used to run the
                        utility
        :type dumpregs: :class:`~sonos.client.dumpregs.Dumpregs`
        """
        list_to_check = dumpregs.get_dump_regs_as_list(dmp_regs_option=dmp_regs_option)
        for reg in registers:
            self.verifyTrueOrFailCase(reg in list_to_check,
                                      '{} register should be found'.format(reg))



[docs]    def get_baseband_reg_range(self, zp, netif, dumpregs):
        """
        Gets the min and max hex representation of the available regs.

        :param zp: The zone player under test
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`
        :param dumpregs: An instance of the dumpregs class, used to run the
                        utility
        :type dumpregs: :class:`~sonos.client.dumpregs.Dumpregs`
        :returns: A list of strings containing the min and max reg
        :rtype: :obj:`list`
        """
        results = dumpregs.get_dump_regs_as_list(netif=netif,
                                                 dmp_regs_option='-b')
        min_result = int(results[0], 16)
        max_result = int(results[-2], 16)
        return [min_result, max_result]



[docs]    def verify_baseband_reg_range(self, zp, netif, dumpregs):
        """
        Verifies that the max and min available registers match the expected.

        :param zp: The zone player under test
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`
        :param dumpregs: An instance of the dumpregs class, used to run the
                        utility
        :type dumpregs: :class:`~sonos.client.dumpregs.Dumpregs`
        """
        reg_range = self.get_baseband_reg_range(zp, netif, dumpregs)
        self.verifyEqualOrFailCase(hex(EXPECTED_MIN_BASEBAND_REG),
                                   hex(min(reg_range)),
                                   '{} should be the min reg'.
                                   format(hex(EXPECTED_MIN_BASEBAND_REG)))
        self.verifyEqualOrFailCase(hex(EXPECTED_MAX_BASEBAND_REG),
                                   hex(max(reg_range)),
                                   '{} should be the max reg'.
                                   format(hex(EXPECTED_MAX_BASEBAND_REG)))



[docs]    def get_reg_values(self, zp, reg_list, netif, option='-a'):
        """
        Iterate through the list of regs and get their values. Return a named
        tuple of regs and their values.

        :param zp: The zone player under test
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`
        :param reg_lst: A list of strings consisting of registers of interest
        :type reg_list: :obj:`list` of :obj:`str`
        :param option: The optional argument to be passed to the dumpregs
                        utility
        :type option: :obj:`str`
        :returns: A list of named tuples. Each tuple holds the name of the
                register, as a string, and an int representing the value held
                in the register.
        :rtype: :obj:`list` of :obj:`collections.namedtuple`
        """
        try:
            dumpreg = DumpRegs(
                zp, branch=self.execution_configuration.build.branch_name)
        except AttributeError as e:
            self.logger.info('Missing attribute error, did you forget to pass'
                             ' in --branch, --update_url and --version?')
            raise e
        regs_with_values = []
        list_to_check = dumpreg.get_dump_regs_as_list(netif, option)
        for reg in reg_list:
            value = int(list_to_check[list_to_check.index(reg) + 1], 16)
            regs_with_values.append(self.reg_tuple(reg, value))
        return regs_with_values



[docs]    def verify_summary_reg_values(self, zp, netif, dumpreg):
        """
        Verifies that the registers under test hold the expected values

        :param zp: The zone player under test
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`
        :param dumpregs: An instance of the dumpregs class, used to run the
                        utility
        :type dumpregs: :class:`~sonos.client.dumpregs.Dumpregs`
        """
        self.logger.debug('MAC: {}'.format(zp.cli.get_mac_address('ath0')))
        ath_config = Athconfig(zp)
        chains_before = ath_config.getrfchains(netif)["tx"]["bitmap"]
        ath_config.setrfchains(netif, bitmap=[0, 1, 0])
        self.logger.debug('MAC: {}'.format(zp.cli.get_mac_address('ath0')))
        phy_rx_chainmask_reg = dumpreg.get_reg_from_friendly_name('PHY_RX_CHAINMASK')
        phy_cal_chainmask_reg = dumpreg.get_reg_from_friendly_name('PHY_CAL_CHAINMASK')
        reg_of_interest = ['HPRXDP', 'LPRXDP', 'STA_ID0', 'STA_ID1']
        actual_reg_values = self.get_reg_values(zp, reg_of_interest, netif,
                                                option=' ')
        actual_reg_values.extend(self.get_reg_values(zp,
                                 [phy_rx_chainmask_reg,
                                  phy_cal_chainmask_reg], netif, '-b'))
        actual_reg_values.extend(self.get_reg_values(zp, ['9E24'], netif, '-b'))
        for reg in actual_reg_values:
            if reg.name in [phy_rx_chainmask_reg, phy_cal_chainmask_reg]:
                self.verifyEqualOrFailCase(2,
                                           reg.value, 'Value in {} should '
                                           'match expected'.format(reg.name))
            else:
                self.verifyEqualOrFailCase(hex(dumpreg.get_expected_reg_value(reg.name, netif)),
                                           hex(reg.value),
                                           'Value in {} should match '
                                           'expected'.format(reg.name))
        ath_config.setrfchains(netif, bitmap=chains_before)
        self.logger.debug('MAC: {}'.format(zp.cli.get_mac_address('ath0')))

    def test_verify_dumpregs(self, zp):
        """
        Combinatorial test that verifies the registers under test are
        available. It then verifies the total range of available registers,
        the values held in the registers under test, and the the high priority
        state has not been enabled.

        :param zp: The zone player under test
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`
        """
        zp.cli.enable_all_wifi_interfaces()
        self.logger.debug('MAC: {}'.format(zp.cli.get_mac_address('ath0')))
        self.duts.append(zp)
        try:
            dumpregs = DumpRegs(
                zp, branch=self.execution_configuration.build.branch_name)
        except AttributeError as e:
            self.logger.info('Missing attribute error, did you forget to pass'
                             ' in --branch, --update_url and --version?')
            raise e
        netif = [INTERFACE_2GHZ]
        if zp.is_ht_master_capable():
            netif.append(INTERFACE_5GHZ)
        self.verify_regs(zp, SUMMARY_REGS, dumpregs)
        self.verify_regs(zp, INTERRUPT_REGS, dumpregs, dmp_regs_option='-i ')
        self.verify_regs(zp, KEY_CACHE_REGS, dumpregs, dmp_regs_option='-k ')
        for interface in netif:
            self.verify_baseband_reg_range(zp, interface, dumpregs)
            self.verify_summary_reg_values(zp, interface, dumpregs)
            self.verify_high_priority_queue_disabled(zp, interface, dumpregs)
        self.logger.debug('MAC: {}'.format(zp.cli.get_mac_address('ath0')))
        zp.cli.disable_all_wifi_interfaces()
