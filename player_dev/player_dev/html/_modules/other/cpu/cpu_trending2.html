    def __init__(self, avg_directory, update, configurations, resources, trend_name, ver_range=None,
                 suppress_graphs=False):
        """
        Initialize the object

        Arguments:
        :param str avg_directory: top directory to search for cpu data in. Typically ends with
            branch name, e.g. /path/to/files/floyd
        :param configurations: configurations to be analyzed. Same format used with
            cpu_performance.py, e.g. ['stereo-sub-S5', 'standalone']
        :type configurations: list of str
        :param resources: resources to be analyzed, e.g. ['stereo-S1', 'stereo-sub-S1']
        :type resources: list of str
        :param trend_name: identifying name for the data set
        :type trend_name: str or None
        :param date_range: -- start and end dates time frame to analyze.Format MM/DD/YYYY. If None,
            all dates accepted.
        :type date_range: list of str or None
        :param suppress_graphs: if True graphs will not be raised for viewing as they are created,
            only saved to png files.
        :type suppress_graphs: bool or False

        """
        self.avg_dir = avg_directory
        self.update_file = update
        self.avg_data = []
        self.ver_range = ver_range
        if trend_name:
            self.trend_name = trend_name
        else:
            self.trend_name = ""
        self.configurations = configurations
        self.resources = resources
        self.suppress_graphs = suppress_graphs
        # Recursively defined dict allows calls to nonexistent keys multiple levels deep not to fail
        # Useful for checking if data has already been written for
        # resource/config/device combos
        recursive_default_dict = lambda: collections.defaultdict(
            recursive_default_dict)
        self.compiled_dirs = recursive_default_dict()
        self.result_objects = []
        self.versions = []
        self.timestamps = []

    @staticmethod
    def _generate_valid_rootnames(dirname):
        """
        Walk through directory and sub directories, yielding directories that contain cpu data

        :param str dirname: topmost directory
        :return: directory with cpu data
        :rtype: str

        """
        for root, dirnames, _ in os.walk(dirname):
            # when correct directory is found, prune irrelevant directories to
            # speed up os.walk
            if 'cpu' in dirnames and CPUResultUtility._get_version_num_from_cpu_dirname(
                    root):
                dirnames[:] = ['cpu']
            elif 'averages' in dirnames:
                dirnames[:] = ['averages']

            if ('cpu' not in root) or ('averages' not in root):
                continue
            yield root


[docs]    def get_data(self):
        """
        Walks through given directory and collects data for the specified resources and
        configurations.

        The data is saved on the instance as self.result_objects

        """
        branch_url = 'http://camb-sqa-nas2/AutomationServer/{}'.format(
            self.avg_dir)
        r = requests.get(branch_url)

        versions = re.findall('\d+.\d+-\d+', r.text)
        versions = sorted(set(versions))

        cpu_results = {}

        for v in versions:
            if self.ver_range:
                if v < self.ver_range[0] or v > self.ver_range[1]:
                    continue

            cpu_result_url = 'http://camb-sqa-nas2.rinconnetworks.com/AutomationServer/{branch}/{ver}/cpu/averages/{ver}-CPU-averages.json'.format(
                branch=self.avg_dir, ver=v)
            resp = requests.get(cpu_result_url)
            if resp.ok:
                cpu_result = json.loads(resp.text)
                cpu_results[v] = cpu_result

        self.cpu_results = cpu_results

    def collate_directory_average_files(dirname, collated_avg_fn, resources=None,
                                        configs=None, check_for_compiled=True):
        """
        Combines all cpu json data files in a directory (corresponding to a build version) into a
        CPUResult object

        :param str dirname: string, the directory to combine files in
        :param str collated_avg_fn: base filename of json compiled average files
        :param resources: resources of interest. If None, all considered.
        :type resource: list of str or None
        :param configs: configurations of interest. If None, all considered.
        :type configs: list of str or None
        :param check_for_compiled: if True, a compiled cpu data json file is checked for
            before going through individual json files
        :type check_for_compiled: bool or True

        :return: a CPUResult object
        :rtype: CPUResult

        """
        timestamp = os.stat(dirname).st_mtime
        ver_num = CPUResultUtility._get_version_num_from_cpu_dirname(
            os.path.abspath(dirname))
        compiled_avg_file = ver_num + '-' + collated_avg_fn

        result = CPUResult(ver_num, timestamp)

        if os.path.exists(compiled_avg_file) and check_for_compiled:
            result.add_result_from_json(compiled_avg_file)
            return result
        else:
            for fn in os.listdir(dirname):
                if not fn.endswith("-averages.json"):
                    continue
                if resources:
                    if not any([resource in fn for resource in resources]):
                        continue
                if configs:
                    if not any([config in fn for config in configs]):
                        continue
                fn_abspath = os.path.join(dirname, fn)
                result.add_result_from_json(fn_abspath)
            return result

[docs]    def graph_and_print_running_averages(self, resource, config):
        """
        Prepares single resource/config data for graphing and printing by creating lists by device
        from data stored in dict, then graphs and prints it

        :param str resource: resource of interest
        :param str config: configuration of interest

        """
        avg_data, all_versions = self._prepare_for_graph_and_print(
            resource, config)
        self.avg_data = avg_data
        self.versions = all_versions

        self._print_data_to_console(
            avg_data, all_versions, resource, config)
        self._graph_data(avg_data, all_versions, resource, config)


    def _prepare_for_graph_and_print(self, resource, config):
        """
        Compiles data by device into lists, which are stored by device in a dict. Also returns all
        timestamps and all versions that were stored (no one device is guaranteed to have all)

        :param str resource: resource of interest
        :param str config: configuration of interest
        :return: the data itself. Keys are devices. Data is stored as list of tuples
            (timestamp, avg)
        :rtype: dict
        :return: all timestamps for which data was found
        :rtype: list of int
        :return: all versions for which data was found
        :rtype: list of str

        """
        avg_versions = []
        devices = set()

        for ver in self.cpu_results:
            data = self.cpu_results[ver]
            if resource not in data or config not in data[resource]:
                continue
            avg_versions.append(ver)
            rc_combo = data[resource][config]
            devices.update(rc_combo.keys())

        avg_versions.sort()

        avg_data = {d: [] for d in devices}
        for ver in avg_versions:
            data = self.cpu_results[ver]
            rc_combo = data[resource][config]
            for device in devices:
                if device not in data[resource][config]:
                    avg_data[device].append(self.NO_DATA)
                else:
                    avg = data[resource][config][device]
                    avg_data[device].append(avg)

        return avg_data, avg_versions

    def _graph_data(self, avg_data, all_versions, resource, config):
        """
        Graphs the data, formats the graph, and saves it as a png

        :param dict avg_data: the data to plot
        :param all_versions: list of all versions in avg_data
        :type all_version: list of str
        :param all_timestamps: list of all timestamps in avg_data
        :type all_timestamps: list of int
        :param str resource
        :param str config

        """
        devices = sorted([key for key in avg_data.keys()])
        ver_ix = numpy.arange(len(all_versions))
        for d in devices:
            data = avg_data[d]
            data = numpy.array([a if a != self.NO_DATA else -1 for a in data])
            x_ax = ver_ix[data > -1]
            plt.plot(x_ax, data[data > -1], label=d)

        plt.grid(which='both')

        try:
            tick_ix = numpy.linspace(ver_ix[0], ver_ix[-1],
                                     min(25, len(ver_ix))).astype(numpy.int32)
        except IndexError:  # catches bad resource/config pairs
            return
        tick_labels = [all_versions[ix] for ix in tick_ix]
        plt.xticks(tick_ix, tick_labels, rotation=90)

        ax = plt.gca()
        box = ax.get_position()
        ax.set_position([box.x0, box.y0 + box.height * 0.15,
                         box.width * 0.85, box.height * 0.85])
        ax.legend(
            loc='center left',
            bbox_to_anchor=(
                1,
                0.5),
            prop={
                'size': 9})
        plt.title('{} {} {}'.format(self.trend_name, config, resource))
        plt.ylim([0, 100])
        plt.ylabel('% CPU Utilization')

        if self.ver_range:
            ver_range = self.ver_range
        else:
            ver_range = [all_versions[0], all_versions[1]]

        png_fn = "{}_{}-{}_{}_to_{}.png".format(self.trend_name,
                                                resource, config,
                                                ver_range[0],
                                                ver_range[1])
        self.logger.info("saving figure as {}".format(png_fn))
        plt.savefig(png_fn, dpi=300)
        if not self.suppress_graphs:
            plt.show()
        else:
            plt.close()

[docs]    def do_update_averages(self, resource, config):
        """
        From avg_data, compute the median, update the averages for the \
        given resource and config
        """
        avg_data = self.avg_data
        versions = self.versions
        devices = [key for key in avg_data.keys()]

        deviceMedian = {}

        for key in devices:
            col = avg_data[key][0:len(versions)]
            col = numpy.asarray(col)
            try:
                col = col[col != 'no data']
            except IndexError:
                continue
            col = col.astype(numpy.float)
            deviceMedian.update({key: numpy.median(col)})

        # must start with properly formatted json
        infile = open(self.update_file, 'r')
        testConfig = json.load(infile)
        # devices is sorted but testConfig is not

        for idx, key in enumerate(testConfig[resource][config]):
            try:
                rounded_avg = round(deviceMedian[key], 2)
                testConfig[resource][config][key] = rounded_avg
            except KeyError:  # S11, etc.
                continue

        with open(self.update_file, 'w+') as f:
            json.dump(testConfig, f, sort_keys=True, indent=4,
                      separators=(',', ': '))




[docs]def parse_args():
    parser = argparse.ArgumentParser()
    parser.add_argument(
        "input_dir",
        help="generate JSON averages from these files")
    parser.add_argument("-v", "--ver_range", nargs=2, default=None,
                        help="31.0-12345 format, start and end version range to be considered."
                        " Considers all versions if no range is specified.")
    parser.add_argument("--suppress_graphs", action='store_true',
                        help="don't show graphs, just save them")
    parser.add_argument("-c", "--configurations", nargs="*", help="Get cpu data for these "
                        "configurations. Use same format expected by cpu_performance.py, "
                        "e.g. stereo-sub-S5", default=None)
    parser.add_argument("-r", "--resources", nargs="*", help="Get cpu data for these resources. Use"
                        " same format expected by cpu_performance.py, e.g. library-aif",
                        default=None)
    parser.add_argument('--trend_name', default=None,
                        help='Identifying name of the data. Branch'
                        'name will make sense in many cases. This name'
                        'will be inluded in the title of the graphs'
                        'and filenames.')
    parser.add_argument('-u', '--update', default=None,
                        help="Automatically update JSON CPU averages contained. You may need to CHMOD 666 the json file (change permissions to writable)"
                        "in specified file. e.g. cpu_trending_2.py -u=cpu_test.json")
    return parser.parse_args()




[docs]def main():
    # logging.basicConfig(level=logging.DEBUG)
    logger = logging.getLogger('CPU Trending')
    args = parse_args()

    if args.resources and args.configurations:
        util = CPUResultUtility(args.input_dir, args.update, args.configurations, args.resources,
                                args.trend_name, args.ver_range, args.suppress_graphs)
        util.get_data()
        for resource, config in util.resource_config_generator(
                args.resources, args.configurations):
            try:
                util.graph_and_print_running_averages(resource, config)
            except KeyError:
                logger.error("Resource / config key pair invalid")
                continue
            if args.update:
                try:  # many configs don't support htaudio
                    util.do_update_averages(resource, config)
                except KeyError:
                    continue

    else:
        logger.info('If Resources and configs are not specified, all'
                    'will be run. Use flags -r and -c, e.g. -r '
                    'library-aif -c stereo-sub-S5. Run this script'
                    'with flag -h for more help.')



if __name__ == '__main__':
