Module for graphing CPU results sample data from web interface.

An example "raw data" file is:
http://camb-sqa-nas2/AutomationServer/PlayerDev/33.7-28170/cpu/samples/CPU-library-AAC-standalone-40684-S1-LF,RF-samples.xml
where:
    CPU-<resource>-<configuration>-40684-<zp-channel>-samples.xml

Note:
    Not all resources and configurations are available for all versions.
    Phase 1: As it is loading raw samples via http, it can be slow.
    Phase 1.1: Include GroupBy(branch)
    Phase 2: Put all raw samples into a persistent highly-available database suited for massive queries.
    Phase 3: Use aggregation functionality if possible from DB.

[docs]def get_page_id(
        page,
        space=DEFAULT_CONF_SPACE,
        conf_rest_api_root_url=CONF_REST_API_ROOT_URL,
        auth_headers=AUTH_HEADERS
):
    """
    Get the unique page ID for page in space.

    :param page: Page title.
    :type page: :obj: `str`

    :param space: Confluence space
    :type space: :obj: `str`

    :param conf_rest_api_root_url: Authentication headers to be passed to
    requests
    :type conf_rest_api_root_url: :obj: `dict`

    :param auth_headers: Root REST API URL
    :type auth_headers: :obj: `str`

    :return: tuple of page ID of space:page and current version or
    (None, None) if space:page DNE
    (page creation is not supported by this module as per design)
    """
    CONF_UPLOADER_LOGGER.info("Getting pageId for Space: {}, Page: {}".format(
        space, page))

    get_content_resource = \
        "/content?spaceKey={}&title={}&expand=version".format(
            space,
            urllib2.quote(page)
        )
    r = requests.get(
        conf_rest_api_root_url + get_content_resource,
        headers=auth_headers)

    if r.ok and len(json.loads(r.text)["results"]):
        resp = json.loads(r.text)
        pageId = resp["results"][0]["id"]
        curVersion = resp["results"][0]["version"]["number"]
        CONF_UPLOADER_LOGGER.info("{} - {} page id: {}, current version: {}".format(
            space, page, pageId, curVersion))
    else:
        pageId = None
        curVersion = None

    return pageId, curVersion

[docs]def upload_attachment(
        pageId,
        attachment_name,
        comment,
        conf_rest_api_root_url=CONF_REST_API_ROOT_URL,
        auth_headers=AUTH_HEADERS
):
    """
    Uploads an attachment to confluence.
    
    :param conf_rest_api_root_url: The confluence REST API root URL.
    :param auth_headers: Auth headers of REST request
    :param pageId: Page ID for page to attach attachment
    :param attachment_name: The attachment name
    :param comment: Comment for the action.

    :return: Tuple of attachment version and id
    """
    create = True
    get_attachment_resource = "/content/{}/child/attachment?filename={}".format(
        pageId, attachment_name)
    r = requests.get(conf_rest_api_root_url + get_attachment_resource,
                     headers=auth_headers)
    if r.ok and len(json.loads(r.text)["results"]):
        resp = json.loads(r.text)
        create = False
    if create:
        create_attachment_resource = "/content/{}/child/attachment".format(
            pageId)
        files = {'file': open(attachment_name, 'rb'), "comment": comment}
        r = requests.post(conf_rest_api_root_url + create_attachment_resource,
                          headers=auth_headers, files=files)
        if r.ok and len(json.loads(r.text)["results"]):
            resp = json.loads(r.text)
            CONF_UPLOADER_LOGGER.info("Created attachment at {}".format(
                create_attachment_resource))
            new_ver = resp["results"][0]["version"]["number"]
            attId = resp["results"][0]["id"]
        else:
            CONF_UPLOADER_LOGGER.info("Cannot create attachment at {}".format(
                create_attachment_resource))
            new_ver = None
            attId = None
    else:
        attId = resp["results"][0]["id"]
        update_attachment_resource = \
            "/content/{}/child/attachment/{}/data".format(
                pageId,
                attId
            )
        files = {'file': open(attachment_name, 'rb'), "comment": comment}
        r = requests.post(conf_rest_api_root_url + update_attachment_resource,
                          headers=auth_headers, files=files)
        if r.ok:
            resp = json.loads(r.text)
            CONF_UPLOADER_LOGGER.info("Updated attachment at {}".format(
                update_attachment_resource))
            new_ver = resp["version"]["number"]
        else:
            CONF_UPLOADER_LOGGER.info("Cannot update attachment at {}".format(
                update_attachment_resource))
        files['file'].close()

    CONF_UPLOADER_LOGGER.info("Attachment version: {}".format(new_ver))
    CONF_UPLOADER_LOGGER.info("Attachment id: {}".format(attId))
    return new_ver, attId

    def iter_samples(samples, branch, version, resource, config, zp):
        samples_attr = samples.attrib
        for sample in samples.iterfind('.//sample'):
            sample_dict = samples_attr.copy()
            sample_dict.update(sample.attrib)
            sample_dict['cpu_percent'] = float(sample.find("cpu_percent").text)
            sample_dict['user_cpu_percent'] = float(sample.find("user_cpu_percent").text)
            sample_dict['branch'] = branch
            sample_dict['version'] = version
            sample_dict['resource'] = resource
            sample_dict['config'] = config
            sample_dict['zp'] = zp
            yield sample_dict



[docs]    def get_data(self, branches, ver_range, resources, configurations, max_versions_lookback):
        """
        Walks through given directory and collects data for the specified resources and
        configurations.

        """
        cpu_results = {'averages': {}, 'samples_df': pd.DataFrame()}

        for branch in branches:
            cpu_results['averages'][branch] = {}
            for resource in resources:
                for config in configurations:
                    try:
                        branch_url = 'http://camb-sqa-nas2/AutomationServer/{}'.format(branch)
                        r = requests.get(branch_url)
                        if r.ok:

                            # versions = re.findall('(\d+).(\d+)-(\d+)', r.text)
                            versions = re.findall('\d+.\d+-\d+', r.text)
                            versions = sorted(set(versions))
                            # Get past n releases
                            past_n_major_releases = []
                            for a in sorted(list(set(map(lambda x: x.split('-')[0], versions))))[
                                     -max_versions_lookback - 1:-1]:
                                major_minor_versions_list_reversed_sorted = sorted(
                                    filter(lambda x: x.startswith(a), versions), reverse=True)
                                # The latest of each might not have cpu-samples, so we need to find out the last version
                                # that ran the test.
                                for try_get_samples_ver in major_minor_versions_list_reversed_sorted:
                                    try_get_samples_ver_url = 'http://camb-sqa-nas2/AutomationServer/{}/{}/cpu/samples'.format(
                                        branch,
                                        try_get_samples_ver)
                                    try_get_samples_ver_url_resp = requests.get(try_get_samples_ver_url)
                                    if try_get_samples_ver_url_resp.ok:
                                        self.logger.info("Last version from " \
                                                "major.minor string \"{}\" that has CPU samples dir: {}".format(
                                            try_get_samples_ver.split('-')[0],
                                            try_get_samples_ver))
                                        # past_n_major_releases.append(max(filter(lambda x: x.startswith(a), versions)))
                                        past_n_major_releases.append(try_get_samples_ver)
                                        break

                            for v in versions:
                                if ver_range:
                                    if not (ver_range[0] <= v <= ver_range[1] or v in past_n_major_releases):
                                        continue

                                cpu_result_url = 'http://camb-sqa-nas2.rinconnetworks.com/AutomationServer/{branch}/{ver}/cpu/averages/{ver}-CPU-averages.json'.format(
                                    branch=branch, ver=v)
                                cpu_result_url_resp = requests.get(cpu_result_url)
                                if cpu_result_url_resp.ok:
                                    # cpu_result = json.loads(cpu_result_url_resp.text)
                                    # cpu_results['averages'][branch][v] = cpu_result
                                    # No need averages file to play what is required.
                                    pass

                                samples_url = 'http://camb-sqa-nas2/AutomationServer/{}/{}/cpu/samples'.format(branch,
                                                                                                               v)
                                samples_url_resp = requests.get(samples_url)
                                if samples_url_resp.ok:
                                    samples = re.findall(
                                        "<a href=\"(CPU-{}-{}-\d+-(\w+-\w+,\w+)-samples.xml)\">".format(resource,
                                                                                                        config),
                                        samples_url_resp.text)
                                    if len(samples):
                                        samples = sorted(set(samples))
                                        for s in samples:
                                            samples_xm_url = "{}/{}".format(samples_url, s[0])
                                            samples_xm_url_resp = requests.get(samples_xm_url)
                                            if samples_xm_url_resp.ok:
                                                samples_tree = ET.fromstring(samples_xm_url_resp.text)
                                                samples_df = pd.DataFrame(
                                                    list(self.iter_samples(samples_tree, branch, v, resource, config,
                                                                           s[1])))
                                                cpu_results['samples_df'] = pd.concat(
                                                    [cpu_results['samples_df'], samples_df])
                    except KeyError:
                        self.logger.info("Resource:{}/config:{} key pair "
                                         "invalid".format(resource, config))
                        continue

        cpu_results['samples_df'].set_index([range(cpu_results['samples_df'].shape[0])], inplace=True)
        return cpu_results

[docs]def main(branches, configurations, resources, zps_list, ver_range, max_versions_lookback):
    """

    :param branches: Directory name of the CPU samples
    :type branches: str
    :param configurations: List of configurations we want to plot.  The length determines the subplot's x-size
    :type configurations: list of str
    :param resources: The resources we want to plot.  Each resource plot is on a different confluence page (pre-created)
    :type resources: str
    :param zps_list: The length of zps_list determines the y-size.  The length of zps_list[0] must match len(configurations)
    :type zps_list: list of list.
    :param ver_range: The version range to plot
    :param max_versions_lookback: Maximum number of major.minor versions to lookback, exluding current major verion.
    :type max_versions_lookback: :obj: `int`
    :type ver_range: list (of 2 elements)
    :return: cpu_results, confluence_upload_data_dict
    :rtype: tuple
    """
    logger = logging.getLogger(make_log_name(['CPUTrending']))
    confluence_upload_data_dict = {}

    util = CPUSamplesUtility()

    if ver_range[0] == "0":
        _, minor, _ = get_version_string_tuple(ver_range[1])
        ver_range[0] = get_start_range_version(ver_range[1], major_back=0, minor_back=minor)

    logger.info("ver_range: {}".format(ver_range))

    cpu_results = util.get_data(branches, ver_range, resources, configurations, max_versions_lookback)
    group_by = np.array(['version', 'branch', 'config', 'resource', 'zp'])
    if cpu_results['samples_df'].shape[0]:
        grouped = cpu_results['samples_df'].groupby(group_by.tolist())
        versions = sorted(set(map(lambda x: x[0], grouped.groups.keys())))

        x_iterables = branches if len(branches) > 1 else configurations
        for resource in resources:
            fig, ax = plt.subplots(len(zps_list), len(x_iterables),
                                   figsize=(CPUSamplesUtility.FIGURE_WIDTH_PER_COLUMN * len(x_iterables),
                                            CPUSamplesUtility.FIGURE_HEIGHT_PER_ROW * len(zps_list)), sharey=True)
            for i, x_iter in enumerate(x_iterables):
                for j, zp in enumerate(zps_list):
                    axis = ax[j][i] if (len(zps_list) > 1 and len(x_iterables) > 1) else (
                        ax[j] if (len(zps_list) > 1 and len(x_iterables) == 1) else (ax[i] if (
                            len(zps_list) <= 1 < len(x_iterables)) else ax))
                    t = axis.set_title(
                        "{} in {} branch playing {}".format(zp[i], x_iter, resource) if len(branches) > 1 else
                        "{} in {} configuration playing {}".format(zp[i], x_iter, resource),
                        fontsize='large')
                    # axis.boxplot([grouped.get_group(k).cpu_percent.values for k in sorted(
                    #     filter(lambda x: config in x[1] and resource in x[2] and zp[i] in x[3], grouped.groups.keys()))],
                    #                  labels=map(lambda x: x[0], sorted(
                    #                      filter(lambda x: config in x[1] and resource in x[2] and zp[i] in x[3],
                    #                             grouped.groups.keys()))))
                    filtered_rows = filter(
                        lambda x: x_iter in x[np.where(group_by == "branch")[0].tolist()[0] if len(branches) > 1 else \
                            np.where(group_by == "config")[0].tolist()[0]] and resource in x[
                            np.where(group_by == "resource")[0].tolist()[0]] and zp[i] in x[
                            np.where(group_by == "zp")[0].tolist()[0]],
                        grouped.groups.keys())
                    if len(filtered_rows):
                        values = [grouped.get_group(k).cpu_percent.values for k in sorted(filtered_rows)]
                        axis.boxplot(values)
                        # for k, v in enumerate(values):
                        #     axis.text((k + 1) + 0.5, np.median(v), '%d' % len(v), ha='center', va='bottom',
                        #               fontsize='x-small', rotation=45)
                        axis.set_xticklabels(map(
                            lambda x: '({}) {}'.format(len(grouped.get_group(x).cpu_percent.values),
                                                       x[np.where(group_by == "version")[0].tolist()[0]]), sorted(
                                filter(lambda x: x_iter in x[
                                    np.where(group_by == "branch")[0].tolist()[0] if len(branches) > 1 else
                                    np.where(group_by == "config")[0].tolist()[0]] and resource in x[
                                    np.where(group_by == "resource")[0].tolist()[0]] and zp[i] in x[
                                    np.where(group_by == "zp")[0].tolist()[0]],
                                       grouped.groups.keys()))), rotation=45,
                            fontsize='small')
                        axis.set_xlabel('Release versions', fontsize='medium')
                        axis.set_ylabel('CPU Percent', fontsize='medium')
                        """
                        Explicitly set the y-limit.  On the later matplotlib, with subplots, when we share y and when
                        the left plot is empty, it doesn't set the y ticks, hence the right plots might be empty.
                        """
                        axis.set_ylim(0, 100)

                        # axis.legend(loc='center left', bbox_to_anchor=(0.8, 0.9))
                        axis.grid(which='both', axis='y')

            # png = "{}_{}-{}_vs_{}_{}_to_{}.png".format(branch, resource, ht_configs[0], ht_configs[1], versions[0], versions[-1:][0])
            png = "{}_{}-{}.png".format(configurations[0], resource, "_vs_".join(branches)) if len(branches) > 1 else \

    else:
        logger.info("No matching rows found in data!")

    return cpu_results, confluence_upload_data_dict

[docs]def get_version_string_tuple(version_str):
    major = version_str.split(".")[:1][0]
    major = int(major)
    minor, build = tuple(version_str.split(".")[1].split("-"))
    minor, build = int(minor), int(build)
    return major, minor, build
