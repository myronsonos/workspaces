    """
    Stores cpu test results for one build/version

    """

    logger = logging.getLogger("CPUResult")

    def __init__(self, version, timestamp, data_dict=None):
        """
        :param str version: version number e.g. 27.2-19180
        :param int timestamp: time the test was run, in epoch time
        :param data_dict: optional dict of data to add to the object as it is initialized. Data can
            also be added later using the methods below
        :type data_dict: dict or None

        """
        self.version = version
        self.timestamp = timestamp
        recursive_default_dict = lambda: collections.defaultdict(recursive_default_dict)
        self.data = recursive_default_dict()
        if data_dict:
            self.add_result_from_dict(data_dict)

        """
        Add data to the object from a json file

        :param str path_to_file: path to the json file holding the data

        with open(path_to_file, 'r') as fh:

[docs]    def add_result_from_dict(self, data):
        """
        Add data to the object from a dict

        :param dict data: the data to be added. Should have the structure
            data[resource][config][device]

        """
        for r, c, d, avg in self.json_avg_generator(data):
            if not self.data[r][c][d]:
                self.data[r][c][d] = avg
            else:
                self.logger.error("already recorded data for {} '{}/{}/{}':"
                                  " skipping {}".format(self.version, r, c, d, avg))



[docs]    @staticmethod
    def json_avg_generator(avg_dict):
        """
        Generates all resource, config, device, data combinations in a dictionary of cpu data

        :param dict avg_dict: dictionary of cpu test data
        :return: resource, config, device, and cpu data in a tuple
        :rtype: tuple

        """
        for resource in avg_dict:
            for config in avg_dict[resource]:
                for device in avg_dict[resource][config]:
                    avg = avg_dict[resource][config][device]
                    yield (resource, config, device, avg)

[docs]    def generate_compiled_averages(self):
        """
        Generate a dict of data about each captured average

        """
        for resource in self.data:
            for config in self.data[resource]:
                for device in self.data[resource][config]:
                    avg = self.data[resource][config][device]
                    model, channel = device.split('-')
                    event_dict = {'version':self.version,
                                  'resource':resource,
                                  'config':config,
                                  'channel':device,
                                  'average':avg,
                                  'timestamp':self.timestamp}
                    yield event_dict



[docs]class CPUResultUtility(object):
    """
    CPU test result utility that can plot CPU test data over time and print it to the console.
    """

    # base filename for compiled average JSON files
    COLLATED_AVG_FN = "CPU-averages.json"

    logger = logging.getLogger("CPUResultUtil")

    def __init__(self, avg_directory, configurations, resources, trend_name, date_range=None,
                 supress_graphs=False):
        """
        Initialize the object

        Arguments:
        :param str avg_directory: top directory to search for cpu data in. Typically ends with
            branch name, e.g. /path/to/files/floyd
        :param configurations: configurations to be analyzed. Same format used with
            cpu_performance.py, e.g. ['stereo-sub-S5', 'standalone']
        :type configurations: list of str
        :param resources: resources to be analyzed, e.g. ['stereo-S1', 'stereo-sub-S1']
        :type resources: list of str
        :param trend_name: identifying name for the data set
        :type trend_name: str or None
        :param date_range: -- start and end dates time frame to analyze.Format MM/DD/YYYY. If None,
            all dates accepted.
        :type date_range: list of str or None
        :param supress_graphs: if True graphs will not be raised for viewing as they are created,
            only saved to png files.
        :type supress_graphs: bool or False

        """
        self.avg_dir = avg_directory
        if date_range:
            self.date_range = [datetime.datetime.strptime(d, "%m/%d/%Y").date() for d in date_range]
            self.date_range.sort()
        else:
            self.date_range = None
        if trend_name:
            self.trend_name = trend_name
        else:
            self.trend_name = ""
        self.configurations = configurations
        self.resources = resources
        self.supress_graphs = supress_graphs
        # Recursively defined dict allows calls to nonexistent keys multiple levels deep not to fail
        # Useful for checking if data has already been written for resource/config/device combos
        recursive_default_dict = lambda: collections.defaultdict(recursive_default_dict)
        self.compiled_dirs = recursive_default_dict()
        self.result_objects = []
        self.versions = []
        self.timestamps = []

    @staticmethod
    def _generate_valid_rootnames(dirname):
        """
        Walk through directory and sub directories, yielding directories that contain cpu data

        :param str dirname: topmost directory
        :return: directory with cpu data
        :rtype: str

        """
        for root, dirnames,_ in os.walk(dirname):
            # when correct directory is found, prune irrelevant directories to speed up os.walk
            if 'cpu' in dirnames and CPUResultUtility._get_version_num_from_cpu_dirname(root):
                dirnames[:] = ['cpu']
            elif 'averages' in dirnames:
                dirnames[:] = ['averages']

            if ('cpu' not in root) or ('averages' not in root):
                continue
            yield root


[docs]    def get_data(self):
        """
        Walks through given directory and collects data for the specified resources and
        configurations.

        The data is saved on the instance as self.result_objects

        """

        for root in self._generate_valid_rootnames(self.avg_dir):
            timestamp = os.stat(root).st_mtime
            if not self._is_date_in_range(datetime.date.fromtimestamp(timestamp)):
                continue
            else:
                self.logger.debug("getting averages in '{}'".format(root))
                cpu_result = self.collate_directory_average_files(root,
                                                                  self.COLLATED_AVG_FN,
                                                                  self.resources,
                                                                  self.configurations)
                # if a directory with no data made it through, don't save that empty dict
                if cpu_result.data and cpu_result.version:
                    self.result_objects.append(cpu_result)



[docs]    @staticmethod
    def collate_directory_average_files(dirname, collated_avg_fn, resources=None,
                                        configs=None, check_for_compiled=True):
        """
        Combines all cpu json data files in a directory (corresponding to a build version) into a
        CPUResult object

        :param str dirname: string, the directory to combine files in
        :param str collated_avg_fn: base filename of json compiled average files
        :param resources: resources of interest. If None, all considered.
        :type resource: list of str or None
        :param configs: configurations of interest. If None, all considered.
        :type configs: list of str or None
        :param check_for_compiled: if True, a compiled cpu data json file is checked for
            before going through individual json files
        :type check_for_compiled: bool or True

        :return: a CPUResult object
        :rtype: CPUResult

        """
        timestamp = os.stat(dirname).st_mtime
        ver_num = CPUResultUtility._get_version_num_from_cpu_dirname(os.path.abspath(dirname))
        compiled_avg_file = ver_num+'-'+collated_avg_fn

        result = CPUResult(ver_num, timestamp)

        if os.path.exists(compiled_avg_file) and check_for_compiled:
            CPUResultUtility.logger.debug("using averages already combined"
                                          " in file: {}".format(compiled_avg_file))
            result.add_result_from_json(compiled_avg_file)
            return result
        else:
            for fn in os.listdir(dirname):
                if not fn.endswith("-averages.json"):
                    continue
                if resources:
                    if not any([resource in fn for resource in resources]):
                        continue
                if configs:
                    if not any([config in fn for config in configs]):
                        continue
                fn_abspath = os.path.join(dirname, fn)
                result.add_result_from_json(fn_abspath)
            return result

    def resource_config_generator(resources, configs):
        """
        Generates resource/config combinations

        :param resources
        :type resources: list of str
        :param configs
        :type configs: list of str

        :return: resource and config

        for resource in resources:

    def _sort_result_objects_in_time(self):
        """
        Sorts the list of CPUResult objects according to their timestamps

        """
        timestamps = [result.timestamp for result in self.result_objects]
        sort_ix = numpy.argsort(timestamps)
        sorted_results = [self.result_objects[index] for index in sort_ix]
        self.result_objects = sorted_results


[docs]    def graph_and_print_running_averages(self, resource, config):
        """
        Prepares single resource/config data for graphing and printing by creating lists by device
        from data stored in dict, then graphs and prints it

        :param str resource: resource of interest
        :param str config: configuration of interest

        """

        avg_data, all_timestamps, all_versions = self._prepare_for_graph_and_print(resource, config)

        # if no date range was defined, figure out what the first and last dates were
        if not self.date_range:
            self.date_range = [datetime.date.fromtimestamp(all_timestamps[0]),
                                datetime.date.fromtimestamp(all_timestamps[-1])]

        self._print_data_to_console(avg_data, all_versions, all_timestamps, resource, config)
        self._graph_data(avg_data, all_versions, all_timestamps, resource, config)


    def _prepare_for_graph_and_print(self, resource, config):
        """
        Compiles data by device into lists, which are stored by device in a dict. Also returns all
        timestamps and all versions that were stored (no one device is guaranteed to have all)

        :param str resource: resource of interest
        :param str config: configuration of interest
        :return: the data itself. Keys are devices. Data is stored as list of tuples
            (timestamp, avg)
        :rtype: dict
        :return: all timestamps for which data was found
        :rtype: list of int
        :return: all versions for which data was found
        :rtype: list of str

        """
        all_timestamps = []
        all_versions = []
        avg_data = {}
        self._sort_result_objects_in_time()
        for result in self.result_objects:
            data = result.data
            if resource not in data or config not in data[resource]:
                continue
            rc_combo = data[resource][config]
            all_timestamps.append(result.timestamp)
            all_versions.append(result.version)
            for device in rc_combo:
                if device not in avg_data:
                    avg_data[device] = []
                avg = data[resource][config][device]
                avg_data[device].append((result.timestamp, avg))
        return avg_data, all_timestamps, all_versions

    def _graph_data(self, avg_data, all_versions, all_timestamps, resource, config):
        """
        Graphs the data, formats the graph, and saves it as a png

        :param dict avg_data: the data to plot
        :param all_versions: list of all versions in avg_data
        :type all_version: list of str
        :param all_timestamps: list of all timestamps in avg_data
        :type all_timestamps: list of int
        :param str resource
        :param str config

        """
        graph = Graph(avg_data, xlabel='Version Number', ylabel='CPU %',
                      title=self.trend_name + ' ' + resource + ' ' + config + ' ' +
                      "\n{:%b-%d-%Y} to {:%b-%d-%Y}".format(self.date_range[0], self.date_range[1]),
                      sort_keys=True)

        graph.reduce_graph_height(0.15)
        plt.ylim(0,100)
        graph.axes.yaxis.set_ticks([ y for y in range(0,105,5) ])
        n_labels = min(10, len(all_versions))
        label_ix = numpy.linspace(0, len(all_timestamps)-1,
                                   n_labels).astype(numpy.int64)

        new_ticks = [all_timestamps[ix] for ix in label_ix]
        ticklabels = [all_versions[ix] for ix in label_ix]
        plt.xticks(new_ticks, ticklabels, rotation=90)

        png_fn = "{}_{}-{}_{:%b-%d-%Y}_to_{:%b-%d-%Y}.png".format(self.trend_name, resource, config,
                                                                  self.date_range[0],
                                                                  self.date_range[1])
        self.logger.info("saving figure as {}".format(png_fn))
        graph.save_graph(png_fn)
        if not self.supress_graphs:
            graph.show_graph()
        else:
            graph.close_graph()

    def _print_data_to_console(self, avg_data, versions, timestamps, resource, config):
        """
        Prints the data to the console

        :param dict avg_data: the data to plot
        :param all_versions: list of all versions in avg_data
        :type all_version: list of str
        :param all_timestamps: list of all timestamps in avg_data
        :type all_timestamps: list of int
        :type str resource
        :type str config

        """
        devices = [key for key in avg_data.keys()]
        devices.sort()
        top_row_format ="{:>12}" * (len(devices) + 1)
        row_format = "{:>12}"+"{:>12}"*len(devices)

        print
        print('Average CPU Test Result between {:%b %d, %Y}'
              ' and {:%b %d, %Y}'.format(*self.date_range))
        print('{:>15}{:<15}'.format('Resource: ', resource))
        print('{:>15}{:<15}'.format('Configuration: ', config))
        print
        print(top_row_format.format("version", *devices))

        device_timestamps = dict(zip(devices, [[]]*len(devices)))
        for d in devices:
            device_timestamps[d] = [a[0] for a in avg_data[d]]

        for version, master_timestamp in zip(versions, timestamps):
            print_data = []
            for d in devices:
                if  master_timestamp in device_timestamps[d]:
                    ix = device_timestamps[d].index(master_timestamp)
                    avg = avg_data[d][ix][1]
                    print_data.append(round(avg, 2))
                else:
                    print_data.append('no data')
            print(row_format.format(version, *print_data))

    def _is_date_in_range(self, date):
        """
        Determine if a date is in the specified range

        :param date: date of interest
        :type date: datetime.date

        :return: True - in range, False - out of range
        :rtype: bool

        """
        if self.date_range:
            return (date >= self.date_range[0]) and (date <= self.date_range[1])
        else:
            return True

    @staticmethod
    def _get_version_num_from_cpu_dirname(dirname):
        """
        determine version number from path

        :param str dirname: path to use
        :return: the version number, or None if couldn't determine
        :rtype: str or None

        """
        path = os.path.abspath(dirname)
        if 'cpu' in path:
            names = dirname.split('/')
            return names[names.index('cpu') - 1]
        else:
            return None

[docs]def combine_loose_averages_to_json(dirname, filename=None):
    """
    Walks through given directory and sub-directories looking for loose cpu json files to
    combine into one. Each directory (corresponding to a version), gets it's own json file.

    :param str dirname: the topmost directory to start the search in
    :param filename: optional name of the file to be written. If not specified, a unique name will
        be created based on the version.
    :type filename: str or None

    :return: CPUResult object with compiled test data
    :rtype: CPUResult or None 

    """
    result = None
    for root in CPUResultUtility._generate_valid_rootnames(dirname):
        CPUResultUtility.logger.debug("compiling averages in '{}'".format(root))
        version = CPUResultUtility._get_version_num_from_cpu_dirname(root)
        result = CPUResultUtility.collate_directory_average_files(
            root, version, CPUResultUtility.COLLATED_AVG_FN, check_for_compiled=False)
        if result.data:
            if filename:
                compiled_avg_fn = filename
            else:
                compiled_avg_fn = result.version+'-'+CPUResultUtility.COLLATED_AVG_FN
            result.write_compiled_averages_to_file(compiled_avg_fn)
    return result

[docs]def parse_args():
    parser = argparse.ArgumentParser()
    parser.add_argument("input_dir", help="generate JSON averages from these files")
    parser.add_argument("-d", "--date_range", nargs=2, default=None,
                        help="MM/DD/YYYY format, start and end date of range to be considered."
                        " Considers all dates if no range is specified.")
    parser.add_argument("--combine_loose_averages", action='store_true',
                        help="for each build, combine individual "
                             "JSON averages into a single file.")
    parser.add_argument("--supress_graphs", action='store_true',
                        help="don't show graphs, just save them")
    parser.add_argument("-c", "--configurations", nargs="*", help="Get cpu data for these "
                        "configurations. Use same format expected by cpu_performance.py, "
                        "e.g. stereo-sub-S5", default=None)
    parser.add_argument("-r","--resources", nargs="*", help="Get cpu data for these resources. Use"
                        " same format expected by cpu_performance.py, e.g. library-aif",
                        default=None)
    parser.add_argument('--branch', default="unknown", type=str,
                        help="Name of branch which generated CPU results. "
                             "Note that this is only used when the "
                             "--send_metrics' option is set")
    parser.add_argument('--trend_name', default=None, help='Identifying name of the data. Branch'
                        'name will make sense in many cases. This name will be inluded in the title'
                        ' of the graphs and filenames.')
    parser.add_argument('--send_metrics', default=False, action='store_true',
                        help='Post CPU result data to events server')
    return parser.parse_args()



[docs]def main():
    logging.basicConfig(level=logging.DEBUG)
    logger = logging.getLogger('CPU Trending')
    args = parse_args()
    if args.combine_loose_averages:
        cpu_result = combine_loose_averages_to_json(args.input_dir)
        if cpu_result and args.send_metrics:
            transmit_cpu_results_to_event_server(args.branch, cpu_result)
    if args.resources and args.configurations:
        util = CPUResultUtility(args.input_dir, args.configurations, args.resources,
                                args.trend_name, args.date_range, args.supress_graphs)
        util.get_data()
        for resource, config in util.resource_config_generator(args.resources, args.configurations):
                util.graph_and_print_running_averages(resource, config)
    elif args.combine_loose_averages:
        logger.info('No resources or configs specified. No graphs will be created.')
    else:
        logger.error('Resources and configs need to be specified. Use flags -r and -c, e.g. -r '
                     'library-aif -c stereo-sub-S5. Run this script with flag -h for more help.')
