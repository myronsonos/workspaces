[docs]    def test_product_leak(self):
        """
        Run the product leak tests. For each platform,
        go to the build URL and find the artifact matching
        the platform. Download and perform a set of steps
        to unpack files used for product leak testing.

        :param str platform: What platform are we testing?
        """
        self.artifact_name = ARTIFACT_DICT.get(self.PLATFORM)
        self._retrieve_and_unpack_artifacts()
        self._remove_local_artifact()
        file_list = self._get_files_from_package()

        for file_name in file_list:
            self._test_product_leak(file_name)


    @abc.abstractmethod
    def _retrieve_and_unpack_artifacts(self):
        """
        Download and unpack the files related to the platform under
        test.
        """
        self.local_artifact_location = '{}/{}'.format(self.temp_dir,
                                                      self.artifact_name)

        if self.PLATFORM == ICR:
            artifact_url = "{}../icr-build/{}".format(self.build_url,
                                                      self.artifact_name)
        else:
            artifact_url = "{}../{}".format(self.build_url, self.artifact_name)

        self.logger.debug("Checking for artifact at <{}>".format(artifact_url))
        res, _ = urlretrieve(artifact_url, self.local_artifact_location)
        self.verifyEqualOrStop(self.local_artifact_location, res,
            "Expect the file locations to match")

        self.logger.debug("Unpacking files at {}".format(
            self.local_artifact_location))

    def _remove_local_artifact(self):
        """
        Remove the local artifact after we're done unpacking the files
        from it.
        """
        self.logger.debug("Removing artifact file from temp directory")
        os.remove(self.local_artifact_location)

    def _get_files_from_package(self):
        """
        Retrieve all the files in the package

        :return: a list of files found for the package
        :rtype: :obj:`list`
        """
        file_list = []

        for path, _, files in os.walk(self.temp_dir):
            for file_name in files:
                file_list.append("{}/{}".format(path, file_name))
        self.verifyGreaterOrStop(len(file_list), 0,
            "Expect to find at least 1 file. Found {} File(s)".format(
                len(file_list)))

        return file_list

    def _get_leak_strings(self, path):
        """
        Get a list of tuples (<search string>, <string that may include search
        string>) of search strings to grep through the files for product leaks.

        :param str path: path to file with search strings
        :return: a list of tuples of search strings
        :rtype: :obj:`list` of :obj:`tuple`
        """
        strings = []
        # Populate strings_to_search
        with open(path) as sts_fp:
            for line in sts_fp:
                line = line.strip()
                if line and not line.startswith('#'):
                    self.logger.debug("Line: <{}>".format(line))
                    exclude_list = line.split(',')
                    include = exclude_list.pop(0)
                    strings.append((include.strip(), [string.strip() for string in exclude_list]))
        self.verifyGreaterOrStop(len(strings), 0, "Expect to find at least 1 "
            "search string.")

        return strings

    def _clear_tmp_directory(self):
        """
        Remove the tmp directory.
        """
        if os.listdir(self.temp_dir):
            rmtree(self.temp_dir)

    def _test_product_leak(self, file_name):
        """
        Search a file for expected product leak strings. Exclude any
        matching results if the result also matches the "exclude" strings
        (String that contains the search string within it).

        :param str file_name: name of file to check for product leak strings
        """
        for include, exclude_list in self.expected_strings:
            p_strings = Popen(['strings', file_name], stdout=PIPE)
            p_grep_include = Popen(['grep', '-i', '{}'.format(include)],
                                   stdin=p_strings.stdout, stdout=PIPE)

            if exclude_list:
                p_grep_exclude = p_grep_include
                for exclude in exclude_list:
                    p_grep_exclude = Popen(['grep', '-v', '-i', exclude],
                                           stdin=p_grep_exclude.stdout,
                                           stdout=PIPE)
                p_wc = Popen(['wc', '-l'],
                             stdin=p_grep_exclude.stdout, stdout=PIPE)

            else:
                p_wc = Popen(['wc', '-l'],
                             stdin=p_grep_include.stdout, stdout=PIPE)

            output = p_wc.communicate()[0]
            output = int(output.strip())

            if exclude_list:
                self.verifyEqualOrFailCase(0, output,
                    '{0} (excluding {1}) found in {2}'.format(
                        include, exclude_list, file_name.lstrip()))
            else:
                self.verifyEqualOrFailCase(0, output,
                    '{0} found in {1}'.format(
                        include, file_name.lstrip()))
