[docs]def createEndpoint(endpointIPs):
	endpoint = WebObjectBase()
	endpoint.ips = []
	endpointIPsList=endpointIPs.split('/')
	endpointTestIP=endpointIPsList[0]
	endpointMgmtIP=endpointIPsList[1]
	mgmtIp_type="IPV4"
	testIp_type="IPV4"
	if endpointMgmtIP.find(":")!=-1:
		mgmtIp_type="IPV6"
	if endpointTestIP.find(":")!=-1:
		testIp_type="IPV6"
	endpoint.ips.append(createIP(endpointTestIP, testIp_type))
	endpoint.managementIp = createIP(endpointMgmtIP, mgmtIp_type)
	endpoint.name = endpointMgmtIP
	return endpoint
