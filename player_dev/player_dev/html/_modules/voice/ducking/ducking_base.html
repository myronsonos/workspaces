[docs]class DuckingEvents(UDPDiagnosticModuleMonitor):
    """
    Ducking Events management class. Handle ducking events using udp logger.
    """

    DUCKING = '\[(?:default|hta-chsnk|inline)-mixer\]: ducking streams'
    UN_DUCKING = '\[(?:default|hta-chsnk|inline)-mixer\]: unducking streams'

    MIXER_EVENTS = [DUCKING, UN_DUCKING]

    # overall ducking events that we will look for in tests
    DUCKING_EVENTS = {
                      'mixer': {3: MIXER_EVENTS}
                      }

    def __init__(self, zp, logger):
        UDPDiagnosticModuleMonitor.__init__(self, logger)
        self.zp = zp
        self.error_msg = ''

        # keep state of ducking related events as received via udp
        # logger and the failure reason to report in case the event is not
        # received before timeout.
        self.event_states = {
            self.DUCKING: [False, TIMEOUT_REASON.format('ducking', self.zp.friendly_name)],
            self.UN_DUCKING: [False, TIMEOUT_REASON.format('unducking', self.zp.friendly_name)],
        }
        self.error_detected = False
        self.ducking_detected_msg = ''
        self.un_ducking_detected_msg = ''
        self.current_test_name = ''


[docs]    def cb_watch_event(self, msg, zp, (m, expr, l, counters)):
        """
        Callback when a watch string matches
        :param str msg: anacapa log message from udplogger
        :param zp: Sonos zone component
        :type zp: :class:`~sonos.client.SonosZoneComponent`
        :param expr: Watch string used to get matching log msg via udplogger
        """
        if expr in self.event_states:
            self.event_states[expr][EVENT_RECD] = True

        if re.search(self.DUCKING, msg):
            self.ducking_detected_msg = msg
        if re.search(self.UN_DUCKING, msg):
            self.un_ducking_detected_msg = msg

        """

[docs]    def wait_for_event(self, event, timeout_seconds=DEFAULT_TIMEOUT):
        """
        Wait for a given event
        :param str event: Log event aka watch string to wait for
        :param int timeout_seconds: Timeout waiting for the event
        """
        self.logger.info("Waiting for event: {} on ({})".format(event, self.zp.friendly_name))
        try:
            wait_until_true(lambda: self.event_states[event][EVENT_RECD],
                        timeout_seconds=timeout_seconds,
                        iteration_delay=1,
                        reason=self.event_states[event][REASON])
        except TimeoutError as e:
            return False
        # reset state to false to allow detecting next event
        self.event_states[event][EVENT_RECD] = False
        return True

[docs]    def setup_udplogger(self):
        """
        Udp logger set up.
        """
        self.logger.info("Setting up udp logger on {}".format(self.zp))
        self.disable_raise()
        watch_service_states_cb_tuple_list = [
            (self.zp.AVTransport.service, "AVTransportURI", None),
            (self.zp.AVTransport.service, "CurrentTrackURI", None),
            (self.zp.AVTransport.service, "TransportState", None),
            (self.zp.DeviceProperties.service, "ChannelMapSet", None),
            (self.zp.DeviceProperties.service, "HTSatChanMapSet", None)
        ]
        self.start_watch(
            self.zp,
            self.cb_watch_event,
            self.DUCKING_EVENTS,
            reset=True,  # restart_logging for this device
            watch_service_states_cb_tuple_list=watch_service_states_cb_tuple_list
        )
        self.reg_fatal_errors_callbacks(None, self.cb_detected_error)
        self.turn_on_upnp_invocation_callback(self.zp)

[docs]    def start_playback(self, zp, play_mode='NORMAL', qpos=1):
        """
        Start playback on zp
        :param zp: Sonos zone component
        :type zp: :`~sonos.client.SonosZoneComponent`
        :param str source: Source of music queue/share or radio or tv
        :param str play_mode: Play mode
        :param int qpos: Start position in queue for playback
        """
        self.logger.info("Starting playback from share on {}".format(zp))
        zp.RenderingControl.SetVolume(5)
        zp.AVTransport.use_queue()
        uri = zp.ContentDirectory.get_track_uri_for_filename(SHARE_TRACK)
        zp.AVTransport.add_to_queue(uri)
        self.logger.info(
            "Queue position: {} Playmode: {}".format(qpos, play_mode))
        zp.AVTransport.set_queue_position(qpos)
        zp.AVTransport.SetPlayMode(play_mode)
        try:
            zp.AVTransport.Play_and_wait()
        except TimeoutError:
            self.stop("Timed out waiting for <{}> to start play".format(zp))

[docs]    def get_ducking_time(self, deh):
        """
        Calculate the time player ducked
        :return float ducking_time: time ducked
        """
        if not deh.ducking_detected_msg or not deh.un_ducking_detected_msg:
            self.logger.error("No duck or unduck detected. Cannot measure time")
            return -1
        start = self.extract_time_from_log(deh.ducking_detected_msg)
        self.logger.info("Ducking start time: {}".format(start))
        finish = self.extract_time_from_log(deh.un_ducking_detected_msg)
        self.logger.info("Ducking stop time: {}".format(finish))
        ducking_time = (finish-start).total_seconds()
        self.logger.info("Time ducked: {}".format(ducking_time))
        return ducking_time



[docs]    def extract_time_from_log(self, log):
        """
        Extract time from log message
        :param str log: Anacapa Log string containing time
        :return date_time_obj: datetime object
        """
        self.logger.info("Extracting time from log: {}".format(log))
        pattern = '^\[(.*?)\]'
        match = re.search(pattern, log.strip())
        time_string = match.group(1)
        date_time_obj = datetime.strptime(time_string, '%Y-%m-%d %H:%M:%S.%f')
        return date_time_obj

    def _get_value(self, zp, property):
        """
        Get data from rendering control status page for the given property
        param zp: Sonos zone component from which to get value
        :type zp: :class:`~sonos.client.SonosZoneComponent`
        :param str property: Property for which to get info
        :return str value: value of property
        """
        data = zp.http.get_url(RENDERING_CONTROL_STATUS.format(zp.ip))
        data_xml = XMLAccessor(data)

        return data_xml.get('RenderingControl').get(property).get('value')

[docs]    def get_ducking_flag(self, zp):
        """
        Get current ducking flags from the player. Ducking flag is determined
        based on the configuration that the player is in and if the local
        player for an unduck command. See
        # from all/oc/zone/zoneplayer/mixing_sod.h
        // ducking bitmap first 8 bits reserved for local reasons a device
        // would duck. The next 8 bits are used for keeping track of which
        // channels in the bonded zone are actively ducked. Only the primary
        // of a bonded zone keeps track of every channel position.
        #define DUCKING_LOCAL_NULL             (0x00000000)
        #define DUCKING_LOCAL_MUSE             (0x00000001)
        #define DUCKING_LOCAL_VOICE            (0x00000002)
        #define DUCKING_LOCAL_EXTAUDIOSRC      (0x00000004)
        #define DUCKING_LOCAL_TESTPOINT        (0x00000080)
        #define DUCKING_LOCAL_MASK             (0x000000FF)
        // There is one bit reserved for each member of a channel map (see RChannelID)
        // NOTE: if the remote mask size changes, update m_rgRemoteDuckingTimes
        // and review code which uses it.
        #define DUCKING_REMOTE_0               (0x00000100)
        #define DUCKING_REMOTE_1               (0x00000200)
        #define DUCKING_REMOTE_2               (0x00000400)
        #define DUCKING_REMOTE_3               (0x00000800)
        #define DUCKING_REMOTE_4               (0x00001000)
        #define DUCKING_REMOTE_5               (0x00002000)
        #define DUCKING_REMOTE_6               (0x00004000)
        #define DUCKING_REMOTE_7               (0x00008000)
        #define DUCKING_REMOTE_MASK            (0x0000FF00)
        // ducking bitmap slot to track when a duck action
        // was forwarded to the group coordinator
        #define DUCKING_GROUP_GC                (0x80000000)
        #define DUCKING_GROUP_MASK              (0x80000000)

        example:
        in a stereo with bonded sub, if right is ducked, flags on devices in
        config:
            master = 0x00000200
            right =  0x00000101
            sub = 0x00000100

        param zp: Sonos zone component
        :type zp: :class:`~sonos.client.SonosZoneComponent`
        :return str ducking_flag: Value of ducking flag
        """
        return self._get_value(zp, 'DuckingFlags')

        """
        Are players ducked
        :param list players: List of sonos zone component
        :param int timeout_seconds: Timeout seconds for players to duck
        :return:
        """
        try:
            wait_until_true(lambda: all([self.is_player_ducked(zp) for

                            timeout_seconds=timeout_seconds,
                            iteration_delay=1)
            return True
        except TimeoutError:
            return False
