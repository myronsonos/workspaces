    def _generate_pairable_devices(self, voice_capable=None, device=0):
        """
        Generate a pair of stereo pairable devices
        :param voice_capable:
        :param device:
        :return:
        """
        # Filter out all amps since they are considered ht sat capable
        generated_pairs = [x for x in self.generators.generate_testbed_ht_satellite_capable_devices(voice_capable=voice_capable,
                                                                                                    stop_on_fail=False) if
                           not x[0].cli.is_connect_amp() and not x[0].is_s16()]

        if generated_pairs:
            return generated_pairs[device]
        else:
            return None

    def _generate_standalone_device(self, voice_capable=False, ht=False, device=0):
        """
        Generate a standalone playback device
        :param voice_capable:
        :param ht:
        :param device:
        :return:
        """
        # filter out s16s since they need a different helper to make bonded/HT configs then
        # normal HT zps
        generated_standalone = [x for x in self.generators.generate_testbed_playback_devices() if
                                (x.voice_capable is voice_capable and x.ht_master_capable is ht and
                                 not x.is_s16())]

        if generated_standalone:
            return generated_standalone[device]
        else:
            return None

    def _setUp(self, devices):
        for device in devices:
            # Setup the event handler and force unduck the device
            if hasattr(device, 'RenderingControl'):
                device.RenderingControl.SetVolume(0)
                device.RenderingControl.SetMute(False)
            self.verifyTrueOrStop(self._unduck_via_all_and_check(device),
                                  "({}) should not be ducked".format(device.friendly_name))
            self.ducking_event_handler[device.serialNumber].setup_udplogger()
            self._set_debug_module_levels(module_levels={"duck": 5,
                                                         "mixer": 3,
                                                         "rc_impl": 2},
                                          zp=device)

    def _wait_for_duck_unduck_log_message(self, ducked_zone, deh, event, method, timeout=16):
        if not deh.wait_for_event(event, timeout_seconds=timeout):
            self.stop("Did not catch '{}' on ({}) after hitting testpoint on ({}) via source {} after {} secs"
                      .format(event,
                              deh.zp.friendly_name,
                              ducked_zone.friendly_name,
                              method,
                              timeout))
        else:
            self.logger.info("Caught '{}' on ({}) after ducking ({}) via source {}".format(event,
                                                                                           deh.zp.friendly_name,
                                                                                           ducked_zone.friendly_name,
                                                                                           method))

    def _unduck_via_all_and_check(self, zp, retries=5):
        """
        Verifies if a player is ducked and tries to unduck it via the "all" source
        :param zp:
        :param retries:
        :return:
        """
        for x in range(0,retries):
            if self.is_player_ducked(zp):
                self.logger.debug("Force unducking ({}), attempt {}".format(zp.friendly_name, x))
                zp.diag.force_duck_unduck(action="clear", source="all")
                time.sleep(1)
            else:
                return True
        return False

    def _get_ducking_methods(self, zp):
        """
        Return the ducking methods a speaker supports and the expected DUCKING_LOCAL_X value for that method
        :param zp:
        :return:
        """
        if zp.voice_capable:
            return {'voice': 2, 'extaudio': 4}
        else:
            return {'muse': 1}

    def _send_duck_command(self, zp, duck=False, unduck=False, durationMillis=10000, muse=None):
        """
        Send muse duck command to zone player
        :param zp:
        :param duck:
        :param unduck:
        :param durationMillis:
        :param muse:
        :return:
        """
        self.logger.info("Sending MUSE command to player: ({}). Ducking: {}. "
                         "Unducking: {}".format(zp.friendly_name, duck, unduck))

        # make ws connection to the zp if there is none
        if muse is None:
            muse = self.make_connection(zp)
        if duck:
            resp = muse.playerVolume.duck(player_id=zp.raw_udn, hhid=zp.muse_hhid, durationMillis=durationMillis)
        else:
            resp = muse.playerVolume.unduck(player_id=zp.raw_udn, hhid=zp.muse_hhid)
        if isinstance(resp, Success):
            self.logger.info("MUSE command successful.")
            return True
        else:
            self.logger.error("MUSE command failed")
            return False

    def _force_device_to_duck_unduck_and_verify(self, ducked_zone, master=None,
                                                unduck_in_secs=DEFAULT_DUCK_TIME,
                                                ducked_zones_mask=None):
        """
        Verify ducking/unducking initiated by various ducking sources
        :param int unduck_in_secs: Time in seconds in which to issue unduck command
        :param ducked_zone: Sonos zone component on which to perform ducking verification
        :param ducked_zones_mask: dictionary describing the DUCKING_REMOTE value of each component
        :return:
        """
        # Get ducking methods based on ducked player type
        methods = self._get_ducking_methods(ducked_zone)
        for method in methods:
            # special case handling for sats/sub
            # only look on the primary/master zp for the 'ducking streams' log message
            if ducked_zone.is_satellite() or ducked_zone.is_sub():
                deh_target = master
            else:
                deh_target = ducked_zone
            self._setUp((deh_target,))
            deh = self.ducking_event_handler[deh_target.serialNumber]
            # Set ducking DUCKING_LOCAL_X values only if ducked zone is in the mask dict
            if ducked_zone in ducked_zones_mask:
                ducked_zones_mask[ducked_zone] = (ducked_zones_mask[ducked_zone] | methods[method])
            deh_target.RenderingControl.SetVolume(10)
            self.verifyEqualOrFailCase(10, deh_target.RenderingControl.GetVolume(),
                                       "Volume before each test should be set to 10")

            # Duck the player via the chsnk endpoint
            ducked_zone.diag.force_duck_unduck(action="set", source=method)
            # wait for ducking message in logs
            self._wait_for_duck_unduck_log_message(ducked_zone=ducked_zone, deh=deh, event=deh.DUCKING, method=method)
            # sleep the ducking duration
            time.sleep(unduck_in_secs)
            # verify rendering control status page reports correct state
            for device in ducked_zones_mask:
                # Check the "PlaybackDucked" value on the device
                self.verifyTrueOrFailCase(self.is_player_ducked(device),
                                          "({}) should be ducked after ducking ({}) via source {}".format(device.friendly_name,
                                                                                                          ducked_zone.friendly_name,
                                                                                                          method))
                # Talked to Dan and we agreed that checking the ducking bit values is overkill right now
                # I'm leaving in the code to check the bit field
                # in case its ever needed in the future
                # self.verifyEqualOrFailCase(ducked_zones_mask[device],
                #                            int(self.get_ducking_flag(device), 16),
                #                            "({}): Expect ducking flag to match".format(device.friendly_name))

            # Force an unduck via the testpoint
            ducked_zone.diag.force_duck_unduck(action="clear", source=method)
            for _ in range(0, 5):
                self._wait_for_duck_unduck_log_message(ducked_zone=ducked_zone, deh=deh,
                                                       event=deh.UN_DUCKING, method=method)
                ducked_duration_actual = self.get_ducking_time(deh)
                if ducked_duration_actual > 0:
                    self.verifyAlmostEqualOrFailCase(ducked_duration_actual,
                                                     unduck_in_secs,
                                                     "({}): Expect unducking in {} secs".format(ducked_zone.friendly_name,
                                                                                                unduck_in_secs),
                                                     delta=16)
                    self.actual_ducking_durations.append(int(ducked_duration_actual))
                    break
                else:
                    self.logger.warn("Ducking duration was negative")
            else:
                self.logger.error("Ducking duration was still negative after 5 tries")

            # verify rendering control status page reports correct state
            for device in ducked_zones_mask:
                for _ in range(0, 5):
                    if not self.is_player_ducked(device):
                        self.verifyEqualOrFailCase(0,
                                                   int(self.get_ducking_flag(device),
                                                       16),
                                                   "({}) should be unducked after unducking ({}) via source {}".format(device.friendly_name,
                                                                                                                       ducked_zone.friendly_name,
                                                                                                                       method))
                        break
                    else:
                        self.logger.warn("({}): Expected to be unducked".format(device.friendly_name))
                        time.sleep(1)
                else:
                    self.fail("({}) is still ducked after unducking ({}) via {}".format(device.friendly_name,
                                                                                        ducked_zone.friendly_name,
                                                                                        method))
            deh.teardown_udplogger()

            self.verifyTrueOrFailCase(deh_target.AVTransport.is_playing(),
                                      "{} should still be in the playing state".format(deh_target.friendly_name))
            deh_target.RenderingControl.SetVolume(20)
            self.verifyEqualOrFailCase(20, deh_target.RenderingControl.GetVolume(),
                                       "Volume after each test should be set to 0")

    def _start_cloudqueue_playback(self, zp):
        """
        Start playing a CQ playlist off the mock server on the zp
        :param zp:
        :return:
        """
        # make ws connection to the zp
        self.muse = self.make_connection(zp)
        gid = zp.GroupManagement.get_local_group_uuid()
        self.muse.subscribe_playback(group_id=gid,
                                     hhid=zp.muse_hhid)
        self.muse_session_helper.make_muse_session(self.muse, group_id=gid,
                                                   hhid=zp.muse_hhid, test_fixture=self)
        self.muse.load_cloud_queue(session_id=self.muse.session_id,
                                   hhid=zp.muse_hhid,
                                   queue_base_url=self.cq_control.CQ_PLAYBACK_URI,
                                   track_metadata=None,
                                   play_on_completion=True)
        self.verify_event(PlaybackStatusEvent, timeout=1, iterations=10,
                          event_verifier=lambda e: (e.playback_state == PlaybackState.PLAYING))
        time.sleep(PLAYBACK_ALLOW_TIME)

    def _grouped_ducking_testflow(self, gc, voice_capable):
        """
        Common test code for grouped configs. Using the pauseOnDuck policy here since other only the targeted zp will duck.
        :param gc:
        :param voice_capable:
        :return:
        """
        # Get a voice_capable/non-voice member device else skip the test
        member = self._generate_standalone_device(ht=False, voice_capable=voice_capable)
        if not member:
            self.skip("Did not find (voice_capable={}) member device".format(voice_capable))
        else:
            device = 0
            # Make sure the member device is not the same as the GC
            while member is gc:
                device += 1
                member = self._generate_standalone_device(ht=False, voice_capable=voice_capable, device=device)
        self.test_devices = (gc, member)
        for zp in self.test_devices:
            if not zp.diag.has_testpoint_support():
                self.skip("({}) running {} does not allow testpoint access".format(zp.friendly_name,
                                                                                   zp.version))

        member.AVTransport.join_group(gc)

        # Set the policy values to enable automation pause-on-duck when the container is an audiobook
        self.cq_control.set_playback_policy({"pauseOnDuck": True})
        self.cq_control.set_container_metadata({"type": "audiobook"})
        self.cq_control.clear_log()

        self._start_cloudqueue_playback(gc)

        # Wait until context is fetched
        context_log = self.cq_control.wait_for_filtered_log(type='api',
                                                            fn=lambda l: l['path'] == '/context',
                                                            timeout=15,
                                                            expected=1)
        self.verifyGreaterEqualOrFailCase(len(context_log), 1,
                                          "Wait until at least 1 context request has been made")

        for ducking_zone in GROUPED_DUCKING_ZOME:
            if ducking_zone == 'gc':
                ducked_zone = gc
            else:
                ducked_zone = member
            ducked_zone.RenderingControl.SetVolume(10)
            self.verifyEqualOrFailCase(10, ducked_zone.RenderingControl.GetVolume(),
                                       "Volume before each test should be set to 10")

            # Verify playback has pause after the duck command
            if ducked_zone.voice_capable:
                ducked_zone.diag.force_duck_unduck(action="set", source="voice")
            else:
                ducked_zone.diag.force_duck_unduck(action="set", source="muse")
            self.verify_event(PlaybackStatusEvent, timeout=1,
                              event_verifier=lambda e: (e.playback_state == PlaybackState.PAUSED))

            # Verify that playback has resumed after the unduck command
            if ducked_zone.voice_capable:
                ducked_zone.diag.force_duck_unduck(action="clear", source="voice")
            else:
                ducked_zone.diag.force_duck_unduck(action="clear", source="muse")
            self.verify_event(PlaybackStatusEvent, timeout=1, iterations=10,
                              event_verifier=lambda e: (e.playback_state == PlaybackState.PLAYING))

            # Verify that playback can resume if the durationMillis times out
            if not ducked_zone.voice_capable:

    def _staggered_ducking_source_testflow(self):
        """
        Common test code for staggered ducking test cases.
        :return:
        """
        # duck all devices in config with different timeout values
        start = datetime.now()
        duck_duration = 0
        for zp in self.test_devices:
            duck_duration += 10000
            self._send_duck_command(zp, duck=True, durationMillis=duck_duration)

        # Verify all speakers ducked after duck command
        self.verifyTrueOrFailCase(self.are_players_ducked(self.test_devices),
                                  "Expect all players to be ducked after initial muse duck commands")

        # Verify that all speakers stay ducked while highest durationMillis has not expired
        elapsed_time = (datetime.now() - start).seconds
        while elapsed_time < duck_duration/1000:
            for zp in self.test_devices:
                self.verifyTrueOrFailCase(self.is_player_ducked(zp),
                                          "({}) should still be ducked {} seconds "
                                          "after initial ducking commands".format(zp.friendly_name,
                                                                                  elapsed_time))
            # sleep some time here in order to expire the durationMillis of each muse duck command
            time.sleep(5)
            elapsed_time = (datetime.now() - start).seconds

        # All speakers should be unducked after the last durationMillis has expired
        for zp in self.test_devices:
            wait_until_true(lambda: not self.is_player_ducked(zp),
                            timeout_seconds=5, iteration_delay=1,
                            reason="({}) should be unducked after {} seconds".format(zp.friendly_name,
                                                                                     elapsed_time))
            self.verifyTrueOrFailCase(zp.AVTransport.is_playing(),
                                      "{} should still be in the playing state".format(zp.friendly_name))
