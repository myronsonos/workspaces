[docs]    def wait_for_non_stop_directive(self):
        """
        Wait until we get volume or mute directive, ignore stop capture
        directive.
        """
        self.logger.info("Waiting for non stop directive")
        expire = time.time() + 30
        while time.time() < expire:
            self.veh.wait_for_event(self.veh.AVS_DIRECTIVE_RECD)

            self.logger.info("Checking if is stopcapture")
            if self.avs_validator.is_stop_capture_directive(
                    self.veh.avs_directive_msg_queue.get_nowait()):
                self.logger.info("MESSAGE: {}".format(self.veh.avs_directive_msg))
                self.logger.info("***Ignoring stop capture directive***")
            else:
                self.logger.info("***Got speak directive***")
                break



[docs]    @combinatorial("generate_voice_zones")
    def test_voice_end_to_end(self, zp):
        """
        Validate end to end voice events with no music playing on dut
        :param zp: :class:`~sonos.client.internal.SonosZoneComponent`
        """
        file_name = VOICETAP_FILE
        url = self.get_voicetap_url(file_name)
        self.voice_overlay.load_and_start_overlay(url)
        # voice tap plays automatically once mic is turned on.
        # we get a ducking event during voice overlay playback. Reset that
        # before proceeding with unducking, ext audio playback source checks
        exp_avs_trans_seq = self.avs_validator.get_exp_avs_seq_from_spec(
            file_name)
        # handle the directives only first. Handle avs events separately
        for transaction in exp_avs_trans_seq:
            if DIRECTIVE in transaction:
                self.logger.info("Now validating AVS Directive: {}"
                                 .format(exp_avs_trans_seq[transaction]))
                self.veh.wait_for_event(self.veh.AVS_DIRECTIVE_RECD)
                self.avs_validator.validate_avs_transaction(
                    file_name, self.veh.avs_directive_msg_queue.get_nowait(),
                    transaction_type=self.avs_validator.DIRECTIVE)
        self.veh.wait_for_event(self.veh.AVS_SPEECH_STARTED)
        self.avs_validator.validate_avs_transaction(
            file_name, self.veh.avs_event_msg_queue.get_nowait(),
            transaction_type=self.avs_validator.EVENT)
        self.veh.wait_for_event(self.veh.DUCKING)
        self.veh.wait_for_event(self.veh.EXT_AUDIO_SRC_PLAYING)
        self.veh.wait_for_event(self.veh.EXT_AUDIO_SRC_STOPPED)
        self.veh.wait_for_event(self.veh.UN_DUCKING)
        self.veh.wait_for_event(self.veh.AVS_SPEECH_FINISHED)
        self.avs_validator.validate_avs_transaction(
            file_name, self.veh.avs_event_msg_queue.get_nowait(),
            transaction_type=self.avs_validator.EVENT)
