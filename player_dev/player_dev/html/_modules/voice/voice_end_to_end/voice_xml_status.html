[docs]    def setUpFixture(self):
        """ set up fixture from workflow, get a bootleg"""
        super(VoiceStatusValidator, self).setUpFixture()
        self.voice_players = [zp for zp in self.my_devices if zp.voice_capable]
        self.verifyTrueOrStop(len(self.voice_players) > 0,
                              "Expect atleast one voice enabled player in HH")
        local_path = os.path.dirname(os.path.abspath(__file__))
        self.resource_path = local_path + '/resource/'
        if not os.path.exists(self.resource_path):
            os.mkdir(self.resource_path)
        self.websrv = self.setup_webserver(self.resource_path)
        for zp in self.voice_players:
            # If for any reason ext audio src is already playing, stop it.
            try:
                zp.AVTransport.Stop(8008)
            except UPnPError:
                pass

        self.zp = self.voice_players[0]  # hack
        self.zp.bounce_anacapa_and_wait()
        self.get_files_from_nas(VOICETAP_PATH, files_list=[VOICETAP_FILE])
        voice_overlay = VoiceOverlay(self.zp)
        self.zp.RenderingControl.SetVolume(5)
        url = self.get_voicetap_url(VOICETAP_FILE)
        voice_overlay.load_and_start_overlay(url)
        time.sleep(5)
        urllib.urlretrieve("http://{}:1400/status/voice".format(self.zp.ip),
                           filename="voice.xml")
        self.xml_root = ElementTree(file="voice.xml")

        return node[0].get(attrib)
