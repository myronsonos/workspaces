        """
        Play voicetap for the command provided
        :param str command: voice command to play using voiceoverlay
        """
        self.logger.info("Playing voice command: {} using overlay"
                         .format(command))
        command = command.replace(" ", "_")

        # voice tap plays automatically once mic is turned on.

[docs]    def get_my_music_collection_info(self,
                                     level='My Music',
                                     sub_level='Tracks'):
        """
        Return a tuple of the mediaCollection corresponding
        to the sub level and the mediaCollection of the parent
        :param str level: Container from which to get music collection info
        :param sub_level: Sub level in container to get info by
        :return: Tuple of playlist id in Tracks and id of its parent
        collection, My Music.
        :rtype: :obj:`tuple`
        """
        for item in self.amazon_root:
            if item.title == level:
                collection_id = item.id
                sub_items = self.get_root_collections(base=item.id)
                for sub_item in sub_items:
                    if sub_item.title == sub_level:
                        playlist_id = sub_item.id
                        return playlist_id, collection_id
        return None, None

        """
        Create amazon track uri and metadata

        :rtype: :obj:`tuple`
        """
        playlist_id, collection_id = self.get_my_music_collection_info()
        uri = self.AMAZON_MY_MUSIC_URI.format(quote_plus(playlist_id),
                                              self.smapi_id,
                                              self.svc_acct.serial_number)
        metadata = self.AMAZON_MY_MUSIC_METADATA.format(
            quote_plus(playlist_id),
            quote_plus(collection_id),
            self.svc_acct.service_id,

        return uri, metadata



[docs]    def load_amazon_tracks_in_queue(self, zp):
        """
        Load tracks from Amazon Music, My Music folder into queue
        :return: Number of tracks in queue
        :rtype: :obj:`int`
        """
        self.logger.info("Loading tracks in queue")
        zp.AVTransport.clear_queue_and_wait()
        uri, metadata = self.create_amazon_playlist_uri_metadata()
        zp.AVTransport.add_to_queue(uri, metadata=metadata)
        zp.AVTransport.use_queue()
        zp.AVTransport.SetPlayMode('REPEAT_ALL')
        zp.AVTransport.set_queue_position(1)
        return zp.AVTransport.get_number_of_tracks_in_queue()

[docs]    def get_player_to_room_name_map(self):
        """
        Get player to room name map. Gets serial number to room name map from
        environment variable is it exists or build one using names defined in
        VoiceCommandHelper.CONVENTIONAL_TARGET_ROOM_NAMES.
        The environment variable is a space separated serialNumber:<room name>
        list
        :return: Player to target room dictionary
        :rtype: :obj:`dict`
        """
        self.logger.info("Getting player to room name map.")
        if (len(self.voice_command_helper.CONVENTIONAL_TARGET_ROOM_NAMES) >
                len(self.playback_devices)):
            self.stop_suite("Not enough number of playback players available.")
        player_to_target_room_map = {}
        # check if SERIAL_TO_ROOM_NAME_MAP env variable exists. Setting this
        # give more stability since same device<->room names are chosen
        # across tests to avoid device discovery issues. Note: We donot have
        # capability to programaticaly determine while rooms are discovered.
        if SERIAL_TO_ROOM_NAME_MAP is None:
            self.logger.info("No serial to room name map available. Building "
                             "new player<->room name mapping.")
            i = 0
            for name in self.voice_command_helper.CONVENTIONAL_TARGET_ROOM_NAMES:
                zp = self.playback_devices[i]
                zp.DeviceProperties.set_zone_name(name)
                player_to_target_room_map[zp] = name
                i += 1
        else:
            self.logger.info("Serial number to room name map available. Using "
                             "it to build player<->room name map. ")
            testbed_serials = [zp.serialNumber for zp in self.my_devices]
            serial_to_room_names = SERIAL_TO_ROOM_NAME_MAP.split(" ")
            for serial_name_pair in serial_to_room_names:
                temp = serial_name_pair.split(':')
                serial = temp[0]
                room_name = temp[1].replace("_", " ")
                if (room_name in
                    self.voice_command_helper.CONVENTIONAL_TARGET_ROOM_NAMES
                    and serial in testbed_serials):
                    zp = self.subnet.get_device_by_serial(serial)
                    player_to_target_room_map[zp] = room_name
        return player_to_target_room_map

[docs]    def do_test(self,
                func,
                zp,
                command_type,
                command=None,
                noun=None,
                source=None):
        """
        Do playback or control (volume or playback) command
        :param func: Function to call to perform test
        :param zp: :class:`~sonos.client.internal.SonosZoneComponent`
        :param str command_type: Type of command, playback or control
        :param str command: Applicable to volume or playback command to execute
        :param str noun: Applicable to playback, name of station or track or
        artist to play music
        :param str source: Applicable to playback, name of music source
        """
        if command_type == self.voice_command_helper.CONTROL_COMMAND:
            # generate control command
            voice_command = self.voice_command_helper.generate_control_command(
                command=command,
                target_room_name=self.player_to_target_map[zp])
        elif command_type == self.voice_command_helper.PLAY_COMMAND:
            voice_command = self.voice_command_helper.generate_play_command(
                play_noun=noun,
                play_source=source,
                target_room_name=self.player_to_target_map[zp])
        self.logger.info(
            "Testing voice command: {}".format(voice_command))
        log_file_name = voice_command.replace(" ", "_")
        try:
            func(zp, voice_command)
        except TimeoutError:
            self.collect_anacapa_trace(zp, log_file_name=log_file_name)
            self.fail("{}: Voice command  <'{}'> Failed."
                      .format(zp, voice_command))
        except AVSConnectionFailureException as e:
            self.collect_anacapa_trace(
                self.voice_assistant,
                log_file_name="AVS_Connection_Failure_{}".format(
                    log_file_name))
            self.stop("Voice Assistant connection to amazon lost!.")
        except UnSupportedCommandException:
            self.stop("Command: {} not supported in test yet".format(command))
        if self.error_detected:
            self.collect_anacapa_trace(zp, log_file_name=log_file_name)
            self.fail("Error detected on {}. Log line: {}"
                      .format(zp, self.error_msg))
        # Continue to play for few seconds before stopping test
        time.sleep(WAIT_TIME)
