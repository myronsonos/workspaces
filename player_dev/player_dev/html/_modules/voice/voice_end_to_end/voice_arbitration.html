[docs]    def cb_perception_data(self, msg, zp):
        self.perception_data_reports = self.perception_data_reports + 1
        result = re.search(PERCEPTION_DATA_REGEX, msg)
        energy = result.group(1)
        ambient = result.group(2)
        self.logger.info("<{}> reports energy={} ambient={}".format(
            zp, energy, ambient))

[docs]    def wait_for_event(self, event):
        try:
            wait_until_true(
                lambda: event.isSet(),
                timeout_seconds=45,
                iteration_delay=1,
                reason="Failed to receive speech finished event")
        except TimeoutError:
            return False
        return True



[docs]    def start_overlay_and_verify_results(
            self, zp_to_voice_overlay, zp_exp_winner_sn):
        """
        Starts the loaded overlays concurrently on the voice
        players and confirms expected results.
        Confirms:
        1. Only one player is expected to play the response; this
        is confirmed by a player eventing a speech started and speech
        finished message. In tests where a specific voice player
        is expected to respond, confirm the eventing was generated
        by that player.
        2. All voice players are expected to generate perception
        event that captures the energy and ambient levels of the
        raw voice tap file.
        :param dict zp_to_voice_overlay: A dictionary of voice
        mapped using the zone player object
        :param str zp_exp_winner_sn: The serial number of the
        expected voice arbitration "winner". If empty string
        only one of the voice players (any) will be the "winner".
        """
        voice_players = zp_to_voice_overlay.keys()
        [self.start_persistent_log(zp) for zp in voice_players]
        self.watch_speech_started(voice_players, zp_exp_winner_sn)
        self.watch_speech_finished(voice_players, zp_exp_winner_sn)
        self.watch_perception_data(voice_players)
        # Simultaneously (almost),  unmute
        time.sleep(5)
        run_concurrent_zone_player_processes(
            self.start_overlay, voice_players)
        self.verifyTrueOrFailCase(
            self.wait_for_event(self.finished_event),
            "Expect to receive finished event")
        for voice_overlay in zp_to_voice_overlay.values():
            voice_overlay.remove_overlay_file()
        self.verifyEqualOrFailCase(
            1,
            self.zps_with_speech,
            "Expect only one voice player to play speech")
        self.verifyEqualOrFailCase(
            len(voice_players),
            self.perception_data_reports,
            "Expect all voice players to report perception data")



[docs]    def arbitration_setup_and_verify(self, voice_players, choose_winner):
        """
        Confirms that only one voice capable player plays the
        AVS response utterance when different voice tap
        files are used to concurrently start the overlays.
        The "winning" voice tap raw file is loaded on the
        player that is expected to play the voice response.
        The remaining players are loaded with a voice tap raw
        file which has lower energy and ambient characteristics
        than the "winning" voice tap raw file.
        :param list voice_players:
        :class:`sonos.client.internal.SonosZoneComponent` zp:
        :param bool choose_winner: Test is expecting a specific voice
        capable player to utter speech response
        """
        zp_to_voice_overlay = {}
        if choose_winner:
            zp_exp_winner = random.choice(voice_players)
            self.logger.info(
                "Expected voice arbitration winner <{}>".format(zp_exp_winner))
            sn_exp_winner = zp_exp_winner.serialNumber
        else:
            zp_exp_winner = None
            sn_exp_winner = ""
        for zp in voice_players:
            voice_overlay = VoiceOverlay(zp)
            zp_to_voice_overlay[zp] = voice_overlay
            uri = 'http://{}/{}'.format(
                self.websrv.host(), VOICETAP_FILE_LOW_ENERGY)
            if zp == zp_exp_winner:
                uri = 'http://{}/{}'.format(self.websrv.host(), VOICETAP_FILE)
            self.logger.info("Loading URI {} on <{}>".format(uri, zp))
            voice_overlay.load_voicetap(uri)
        self.start_overlay_and_verify_results(
            zp_to_voice_overlay, sn_exp_winner)



[docs]    def arbitration_ht_setup_and_verify(self, ht_config, choose_winner):
        """
        Sets up HT 5.0 with bootleg as satellites and runs
        the voice arbitration test
        :param tuple ht_config: A tuple of HT zones master and
        satellites
        :param bool choose_winner: Test is expecting a specific
        voice capable player to respond
        """
        master, zp_l, zp_r = ht_config
        master.DeviceProperties.create_50_config(zp_l, zp_r)
        self.verifyTrueOrStop(
            master.is_bonded(),
            "<{}> is expected to HT master".format(master))
        for sat in (zp_l, zp_r):
            self.verifyTrueOrStop(
                sat.is_satellite(),
                "<{}> is  expected to be satellite".format(sat))
        self.arbitration_setup_and_verify((zp_l, zp_r), choose_winner)
