[docs]    def get_current_auth_token(self):
        req = urllib2.Request(url=self.GET_TOKEN_URI(self.CQ_URI))
        try:
            data = urllib2.urlopen(req).read().split('"')
            # Grab the longest living token, at the token half life
            # the server wil return both the half expired, and the new
            if len(data) > 12:
                token = data[13]
            else:
                token = data[5]
            return token
        except urllib2.HTTPError, e:
            self.logger.error("Received HTTP {0} error attempting to get auth token".format(e.code))



[docs]    def get_track_metadata(self, item_id):
        req = urllib2.Request(url=(self.GET_METADATA_URI(self.CQ_URI, item_id)))
        try:
            metadata = json.loads(urllib2.urlopen(req).read(), encoding='utf8')
            # normalize the track URL
            k = u'trackUrl'
            if k in metadata:
                if not metadata[k].startswith('http://'):
                    metadata[k] = self.CQ_URI + metadata[k]

                # convert escaped ASCII sequence into UTF-8
                metadata[k] = metadata[k].decode('unicode-escape').encode('utf8')

            return metadata
        except urllib2.HTTPError, e:
            return None

[docs]    def arm_error_script(self, call, error, num_errors=1):
        '''
        :param call: string, cloudQueue response to force the error on (itemWindow, getVersion)
        :param error: int, the HTTP error to force. 401, 404, 503
        :param num_errors: int, number of calls to cause failures on
        '''

        error = error
        if call == 'itemWindow':
            script = 'itemwindowError{}.js'.format(str(error))
        elif call == 'version':
            script = 'versionError{}.js'.format(str(error))
        script_uri = self.ACTIVATE_SCRIPT(self.CQ_URI, script)
        req = urllib2.Request(url=script_uri, data='limit={}'.format(num_errors))
        try:
            urllib2.urlopen(req)
            self.logger.info("Armed {} script on cloudQueue server".format(script))
        except urllib2.HTTPError, e:
            self.logger.error("Attempt to activate {0} script failed, returned HTTP {1}".format(script, e.code))

[docs]    def set_itemwindow_response_delay(self, delay=5, num_errors=1):
        '''
        Arm the itemWindowDelay.js script on the cloud queue test server. This will cause the server to delay [num_error]
        responses to the getItemWindow() call by [delay] amount. The script defaults the delay to 5 on the server.
        :param delay: int, the number of seconds to delay the getItemWindow() responses
        :param num_errors: int, the number of responses to delay
        '''

        params = urllib.urlencode({'delay': delay, 'limit[]': num_errors})
        req = self.ACTIVATE_SCRIPT(self.CQ_URI, 'itemWindowDelay.js')
        try:
            urllib2.urlopen(req, data=params)
            self.logger.info("Armed itemWindowDelay script on cloudQueue server")
        except urllib2.HTTPError, e:
            self.logger.error("Attempt to activate itemWindowDelay script failed, returned {}".format(e.code))

[docs]    def start_cloudqueue_server(self, stdout=None, stderr=None):
        """
        Start the javascript mock cloudqueue server that lives in /test/javascript/cloudqueue-api.
        Default output and error paths pipe to /dev/null but can be overridden by passing in those
        params through your test.
        :param stdout: the standard output for the cloudqueue server
        :param stderr: the standard error for the cloudqueue server
        :return: object; a running cloudqueue
        """
        workspace = os.getenv("WORKSPACE")
        if not workspace:
            self.logger.error("Unable to start local CloudQueue server because the environment variable $WORKSPACE is not set")
            return None

        if not os.path.isdir("/tmp/cloudqueue"):
            self.logger.exception("/tmp/cloudqueue media repo does not exist on local testrunner")
            return None
        else:
            cq_music_path = "/tmp/cloudqueue/"

        cq_path = "{}/test/javascript/cloudqueue-api".format(workspace)
        FNULL = open(os.devnull, 'w')
        if stdout is not None and stderr is not None:
            out, err = stdout, stderr
        else:
            out, err = FNULL, FNULL

        # Check if anything is using the default server port of 3000, if so pick something else
        ports = range(3000, 3400)
        for port in ports:
            port_check = subprocess.Popen('lsof -i -P | grep -i "listen" | grep {}'.format(port),
                                          shell=True,
                                          stdout=subprocess.PIPE)
            port_check_output = port_check.stdout.read()
            port_check.terminate()
            if port_check_output == '':
                # -p:  sets the port number (non-SSL)
                # -s: sets the secure (SSL) port number
                # -l: path to the directory containing the "library", i.e. the mp3 files
                # -f: use "flat" (old-style) JSON for track metadata
                cq_args = [os.path.join(cq_path, 'init'), '-p', str(port), '-s', str(port + 43), '-l', cq_music_path]
                cq = subprocess.Popen(cq_args, cwd=cq_path,
                                      stdout=out, stderr=err, close_fds=True)
                self.CQ_URI = 'http://{}:{}'.format(self.host_ip, port)
                self.CQ_PLAYBACK_URI = '{}/musicqueue/v1.0/'.format(self.CQ_URI)
                self.logger.debug("CQ Server started on port {}".format(port))
                break
        else:
            raise AssertionError("CQ Server did not find an open port and did not start!")

        return cq



[docs]    def stop_cloudqueue_server(self, cq_server):
        cq_server.terminate()
        # Don't try to read the error codes in cases where the CQ is piping output to /dev/null
        if cq_server.wait() != 0 and (cq_server.stdout and cq_server.stderr):
            self.logger.error("CQ server failed [{}]: {}\n{}".format(cq_server.returncode,
                                                                     cq_server.stderr.readlines(),
                                                                     cq_server.stdout.readlines()))
