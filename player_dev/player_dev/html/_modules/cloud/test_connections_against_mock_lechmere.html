[docs]    def get_lechmere_v6_attributes_from_status(self, zp):
        """
        Get the following tags from status page:
            HouseholdControlID
            HouseholdLocationID
            DHCPServerMac
            NetworkIPAddress
            NetworkMask

        Returns an empty dictionary if the status page can't be accessed.

        :param zp: ZonePlayer
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`
        :return: Dictionary of tag name: value
        :rtype: :obj: `dict`
        """
        zp_status_page = get_status_page(zp, StatusRoutes.ZP)
        zp_status = {attr: zp_status_page.find(attr).text for attr in self.ZP_STATUS_FIELDS_V6} if zp_status_page is not None else {}
        device_status_page = get_status_page(zp, StatusRoutes.DEVICE)
        device_status = {attr: device_status_page.find(attr).text for attr in self.DEVICE_STATUS_FIELDS_V6} if device_status_page is not None else {}
        zp_status.update(device_status)
        return zp_status



[docs]    def wait_for_connected_and_registration_message_sent(self,
                                                         zp,
                                                         iteration_delay=.5,
                                                         registration_timeout=15,
                                                         cloud_connected_timeout=65,
                                                         expected_protocol=None,
                                                         verifier=None):
        """
        Wait for the player to connect and send registration message.

        :param zp: ZonePlayer
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`
        :param float iteration_delay: Delay in seconds to poll
        :param int cloud_connected_timeout: Timeout in seconds to wait for
            cloud connection.
        :param int registration_timeout: Timeout in seconds to wait for
            the player to send registration message after connecting.
        :param str expected_protocol: (optional) String to match against
            the protocol to verify the player connected with the correct
            protocol.
        :param func verifier: (optional) verifier to perform additional
            verifications on the registration message.
        """
        # Conditionally wait for the player to connect to the cloud
        if cloud_connected_timeout:
            try:
                zp.diag.wait_for_cloud_connected(iteration_delay=iteration_delay,
                    timeout_seconds=cloud_connected_timeout)
            except TimeoutError:
                self.stop("Timed out waiting for <{}> to connect".format(zp))

        # Conditionally wait for registration message
        if registration_timeout:
            try:
                wait_until_true(lambda: self.mock_lechmere.get_zp_registered(zp),
                                iteration_delay=iteration_delay,
                                timeout_seconds=registration_timeout)
            except TimeoutError:
                self.stop("Timed out waiting for <{}> to send registration message.".format(zp))

        # Verify the registration message.
        self.verify_registration_message_sent(zp,
                                              expected_protocol=expected_protocol,
                                              verifier=verifier)



[docs]    def verify_registration_message_sent(self,
                                         zp,
                                         expected_protocol=supported_protocols,
                                         verifier=None):
        """
        Verify the ZP sent the registration message with the expected fields.

        :param zp: ZonePlayer
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`
        :param str expected_protocol: (optional) String to match against
            the protocol to verify the player connected with the correct
            protocol.
        :param func verifier: (optional) verifier to perform additional
            verifications on the registration message.
        """
        # Get and check registration status
        registration = self.mock_lechmere.get_zp_registered(zp)
        if not registration:
            self.stop("ZP <{}> not registered!".format(zp))

        # Get and check protocol
        protocol = self.mock_lechmere.get_zp_connected_protocol(zp)
        if protocol is None:
            self.stop("ZP <{}> not connected with a protocol!".format(zp))
        elif protocol not in self.PROTOCOLS.keys():
            self.stop("ZP <{}> connected protocol <{}> not supported!".format(zp, protocol))

        # Conditionally verify protocol
        if expected_protocol is not None:
            self.verifyEqualOrStop(expected_protocol, protocol,
                "ZP <{}> connected with expected protocol?".format(zp))

        my_expected_keys = tuple(self.PROTOCOLS[protocol])
        if zp.roomName is None or len(zp.roomName) == 0:
            self.logger.info("ROOM NAME={}".format(zp.roomName))
            # filter out the roomName key for products without a name
            my_expected_keys = filter(lambda x: x != 'roomName', my_expected_keys)

        if zp.color is None or zp.color == "None" or len(zp.color) == 0:
            self.logger.info("COLOR={}".format(zp.color))
            # filter out the color key for old products without color information
            my_expected_keys = filter(lambda x: x != 'color', my_expected_keys)

        # Verify we got a non empty capabilities array
        self.verifyTrueOrStop(len(registration[0]['capabilities']) > 0, "Empty capabilities array on {}".format(zp))

        self.logger.info("Registration message: <{}>, (expected fields: <{}>".format(registration[0],
                                                                                     my_expected_keys))

        self.verifySequenceEqualOrStop(my_expected_keys, sorted(registration[0].keys()),
            "Found all expected fields in registration message (ZP <{}>, protocol <{}>)?".format(zp, protocol))

        if verifier is not None:
            verifier(zp)

[docs]    def check_registration_fields(self,
                                  registration_message,
                                  **expected_registration_fields):
        """

        :param registration_message:
        :param expected_fields:
        :return:
        """
        for field, value in expected_registration_fields.iteritems():
            if field not in registration_message:
                self.logger.debug('Field <{}> not in registration message!'.format(field))
                return False
            elif registration_message[field] != value:
                self.logger.debug('Field <{}> does not match value <{}> in registration message'.format(registration_message[field], value))
                return False
        return True

                                            zps,
                                            expected_hhlocid,
                                            wait_timeout_seconds=15):
        """
        Verify that a list of players (or one player) is registered with the expected
        householdLocationId (which is actually the player's Trial ZP serial number,
        just to confuse you.)

        :param zps: ZonePlayer or list of ZonePlayers
        :param zps: :class:`~sonos.client.internal.SonosZoneComponent` or

        # Handle the case where a single ZP was passed in instead
        # of a list by converting it to a single-item list.
        if not hasattr(zps, '__iter__'):
            zps = [zps]

        # Go through the list and verify that each player is registered with
        # the expected householdLocationId.
        for zp in zps:
            # Wait if specified, but don't raise an exception if the timeout is hit.
            # Subsequent verifications will fail immediately after in either case.
            self.wait_for_connected_and_registration_message_sent(zp,
                                                                  registration_timeout=wait_timeout_seconds,
                                                                  cloud_connected_timeout=0,
                                                                  verifier=lambda zp:
                                                                  self.poll_and_verify_fields(
                                                                      zp,
                                                                      householdLocationId=expected_hhlocid))

        """
        Check the status page against a dictionary of attributes.

        :param zp: ZonePlayer
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`
        :param dict attrs:
        :return: :obj:`bool`
        """
        status_attrs = self.get_lechmere_v6_attributes_from_status(zp)
        info_str = "Matched (status page: <{}: {}> {} expected attribute: <{}: {}>)"
        for k, v in attrs.iteritems():
            if k in status_attrs and status_attrs[k] == v:
                self.logger.info(info_str.format(k, status_attrs[k], "==", k, v))
                continue
            else:
                self.logger.error(info_str.format(k, status_attrs[k], "!=", k, v))
                return False
        return True

                                                           zp,
                                                           iteration_delay=2,

        if expected:
            # If the muse householdId is not present, fail
            if not muse_household_id_present:
                self.fail("Timed out waiting for museHouseholdId to "
                          "be present in localsettings on <{}>".format(zp))
        # Check case where we didn't expect the museHouseholdId
        else:
            # If the muse householdId is present, fail
            if muse_household_id_present:

        """
        Verify that the player re-sends the registration message when the
        muse household ID changes.

        :return:
        """
        zp = self.test_devices[0]
        wait_until_true(lambda: self.mock_lechmere.get_zp_registered(zp) is not None,

                        reason="Looking for <{}>'s registration message".format(zp.friendly_name))
        orig_muse_hhid = self.mock_lechmere.get_zp_registered(zp)[0]['museHouseholdId']

        # Send a setconfig with a new muse household id
        hhid_list = list(orig_muse_hhid)
        if hhid_list[-1] == "Z":
            hhid_list[-1] = "0"
        elif hhid_list[-1] != "Z":
            hhid_list[-1] = "Z"
        new_muse_hhid = "".join(hhid_list)
        self.mock_lechmere.set_config(zp, musehhid=new_muse_hhid)

        # Verify the player reregisters with the new muse household Id within a short window
        def check_muse_hhid():
            try:
                return self.mock_lechmere.get_zp_registered(zp)[0]['museHouseholdId'] == new_muse_hhid
            except Exception:
                return False

        try:
            wait_until_true(check_muse_hhid,
                            iteration_delay=1,
                            timeout_seconds=5)
        except TimeoutError:
