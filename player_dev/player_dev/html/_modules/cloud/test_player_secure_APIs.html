        """
        Verify token that's too old fails

        :return:
        """
        token = self.mock_lechmere.create_user_token(self.user_id, scopes=Scopes.AS_HH_CONFIG_ADMIN.string)
        # Move the token 5 minutes into the past
        self.mock_lechmere.offset_token_time_since_created(token=token,
                                                           offset=self.TOKEN_TOO_OLD_THRESHOLD)

        for verb in self.ALLOWED_VERBS_RESTRICTED:
            self._verify_bad_auth_token_request(token=token,
                                                url=self._url,
                                                expected_rc=codes.forbidden,
                                                data="true")
