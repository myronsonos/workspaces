[docs]class Account(object):
    """
    Object representing an account, see:
        https://confluence.sonos.com/display/PDSW/Music+service+account+replication

    v0 Example:
        <Account Type="3079" SerialNum="3" Flags="2">
            <UN>X_#Svc3079-0-Token</UN>
            ...
        </Account>
    v1 Example:
        <Account Id="123e4567-e89b-12d3-a456-426655440000" Type="3079" SerialNum="3" Flags="2"
         VClockHousehold="RINCON_7828CA00065001400:5503" VClockCloud="2018-01-01T00:00:00Z">
            <UN>X_#Svc3079-0-Token</UN>
             ...
        </Account>
    """
    def __init__(self, xml, version):
        self._version = None
        self._children = []
        self._properties = {}
        try:
            self._parse(xml, version)
        except Exception as e:
            _logger.debug("Failed to parse account XML <{}>, (version: <{}>) with <{}>".format(xml, version, e))
            raise
        assert (self._version is not None)

        for child in self._children:
            self._properties.update({child.tag: child.text})

    def _parse(self, xml, version):
        if isinstance(xml, ET.Element):
            self._parse_xml(xml, version)
        elif isinstance(xml, str):
            self._parse_xmlstr(xml, version)
        else:
            raise TypeError("Unsupported account format: <{}>".format(xml))

    def _parse_xml(self, xml, version):
        """
        Parse account XML element

        :param xml: Account XML
        :type xml: xml.etree.ElementTree.Element
        :param int version: Schema version
        :return:
        """
        if version == 0:
            self._info = ACCOUNT_ATTRIBUTES_V0(**xml.attrib)
            self._version = version
            self._children = list(xml)
        elif version == 1:
            self._info = ACCOUNT_ATTRIBUTES_V1(**xml.attrib)
            self._version = version
            self._children = list(xml)
        else:
            raise UnsupportedAccountVersionException("Unrecognized version <{}>".format(version))

    def _parse_xmlstr(self, xmlstr, version):
        """
        Parse account XML in string format

        :param str xmlstr: Account XML string
        :param int version: Schema version
        :return:
        """
        root = ET.fromstring(xmlstr)
        return self._parse_xml(root, version)

    @property
    def type(self):
        return int(self._info.Type)

    @property
    def serial_num(self):
        return str(self._info.SerialNum)

    @property
    def flags(self):
        return int(self._info.Flags)

    @property
    def id(self):
        if self._info.Id is not None:
            return str(self._info.Id)

    @property
    def vclock_household(self):
        return str(self._info.VClockHousehold)

    @property
    def vclock_household_version(self):
        return int(self._info.VClockHousehold.split(':')[-1])

    @property
    def vclock_cloud(self):
        return str(self._info.VClockCloud)

    @property
    def vclock_cloud_version(self):
        return int(self._info.VClockCloud.split(':')[-1])

    @property
    def deleted(self):
        return bool(self._info.Deleted)

    @property
    def username(self):
        return self._properties.get('UN')

    @property
    def nickname(self):
        return self._properties.get('NN')

    @property
    def metadata(self):
        return self._properties.get('MD')

    @property
    def oauth_device_id(self):
        return self._properties.get('OADevID')

    @property
    def hash(self):
        return self._properties.get('Hash')

    @property
    def token(self):
        return self._properties.get('Token')

    @property
    def key(self):
        return self._properties.get('Key')

    @property
    def is_oauth(self):
        # This is how the player determines if it is an oauth account (RSvcAccount.isAuthTokenSvc)
        return bool('UN' in self._properties and self._properties['UN'].startswith('X_#'))

    def __eq__(self, other):
        return (self.type == other.type and
                self.serial_num == other.serial_num and
                self.flags == other.flags and
                self.deleted == other.deleted)

    def __str__(self):
        return "<Account (v{}), type={}, sn={}, flags={}, deleted={} - properties=({})>".format(
            self._version, self.type, self.serial_num, self.flags, self.deleted, self._properties
        )

[docs]class Accounts(object):
    """
    Object representing an account, see:
        https://confluence.sonos.com/display/PDSW/Music+service+account+replication

    v0 Example:
        <Accounts LastUpdateDevice="RINCON_5CAAFD9646A601400" Version="24" NextSerialNum="4">
            <Account/>
        </Accounts>
    v1 Example:
        <Accounts LastUpdateDevice="RINCON_5CAAFD9646A601400" Version="24" NextSerialNum="4" SchemaVersion="1"
         VClockHousehold="RINCON_7828CA00065001400:5503" VClockCloud="2018-01-01T00:00:00Z"
         MuseHouseholdID="Sonos_kkmYpyiBx3bg1rRNRxS3sx0vMM.ROoxHLq16_nogKUJtCJn">
            <Account/>
        </Accounts>
    """
    def __init__(self, xml, schema_version=None):
        """
        :param xml: XML element or string representing accounts file
        :param version: Schema version, optional - pass None and we will attempt to parse
        """
        self._info = None
        self._schema_version = None
        self._accounts = []

        try:
            self._parse(xml, schema_version)
        except Exception as e:
            _logger.debug("Failed to parse accounts XML <{}>, (version: <{}>) with <{}>".format(xml, schema_version, e))
            raise

        assert (self._schema_version is not None)
        assert (self._info is not None)

    def _parse(self, xml, schema_version=None):
        if isinstance(xml, ET.Element):
            self._parse_xml(xml, schema_version)
        elif isinstance(xml, str):
            self._parse_xmlstr(xml, schema_version)
        else:
            raise TypeError("Unsupported account format: <{}>".format(xml))

    def _parse_xml(self, xml, schema_version):
        # If the client passed in an explicit version, attempt to parse that
        if schema_version is not None:
            if schema_version == 0:
                self._info = ACCOUNTS_ATTRIBUTES_V0(**xml.attrib)
                self._schema_version = 0
            elif schema_version == 1:
                self._info = ACCOUNTS_ATTRIBUTES_V1(**xml.attrib)
                self._schema_version = 1
            else:
                raise UnsupportedAccountsVersionException("Unrecognized version <{}>".format(schema_version))
        # Otherwise, attempt to determine the version through trial and error
        else:
            try:
                self._info = ACCOUNTS_ATTRIBUTES_V0(**xml.attrib)
                self._schema_version = 0
            except Exception:
                try:
                    self._info = ACCOUNTS_ATTRIBUTES_V1(**xml.attrib)
                    self._schema_version = 1
                except Exception:
                    raise UnsupportedAccountsVersionException("Could not determine version of <{}>".format(xml))

        # Now parse the individual accounts
        for child in xml:
            self._accounts.append(Account(child, self._schema_version))

    def _parse_xmlstr(self, xmlstr, schema_version):
        root = ET.fromstring(xmlstr)
        return self._parse_xml(root, schema_version)

    @property
    def last_update_device(self):
        return str(self._info.LastUpdateDevice)

    @property
    def version(self):
        return int(self._info.Version)

    @property
    def next_serial_num(self):
        return int(self._info.NextSerialNum)

    @property
    def schema_version(self):
        return int(self._info.SchemaVersion)

    @property

        return str(self._info.VClockHousehold)

    def vclock_household_version(self):
        return int(self._info.VClockHousehold.split(':')[-1])

    @property
    def vclock_cloud(self):
        return str(self._info.VClockCloud)

    @property
    def vclock_cloud_version(self):
        return int(self._info.VClockCloud.split(':')[-1])

[docs]    def verify_pushed_account(self,
                              serial_number,
                              muse_household_id=None,
                              sid=None,
                              token=None,
                              key=None,
                              nickname=None,
                              deleted=False,
                              delete_all_accounts=True):
        accounts_pushed = Accounts(self.get_pushed_accounts(muse_household_id), schema_version=1)
        self.zp.settings.sync_account_settings()  # wait for accounts to sync
        accounts_local = Accounts(self.zp.cli.get_accounts_file(), schema_version=1)

        try:
            self.verifyTrueOrFailCase(accounts_pushed and accounts_pushed.at_serial_num(serial_number) is not None,
                                      "Account with serial number {} was pushed?".format(serial_number))

            account_pushed = accounts_pushed.at_serial_num(serial_number)
            account_local = accounts_local.at_serial_num(serial_number)

            if deleted:
                self.verifyEqualOrFailCase(None, account_local, "Local account has been removed?")
            else:
                self.verifyEqualOrFailCase(account_local.vclock_household_version,
                                           account_pushed.vclock_household_version,
                                           "Local account vector version clock matches pushed account vector clock version ({} == {})?".format(
                                               account_pushed.vclock_household,
                                               account_local.vclock_household))

            self.verifyGreaterOrFailCase(accounts_local.vclock_household_version,
                                         account_pushed.vclock_household_version,
                                         "File vector clock version is greater than account vector clock version ({} > {})?".format(
                                             accounts_local.vclock_household,
                                             account_pushed.vclock_household))

            self.verifyIsNotNoneOrStop(account_pushed.id,
                                       "Cloud UUID was not populated in the push response")

            if sid is not None:
                self.verifyEqualOrFailCase(sid, account_pushed.type,
                                           "Pushed account has correct serviceId {}?".format(sid))
            if token is not None:
                self.verifyEqualOrFailCase(token, account_pushed.token,
                                           "Pushed account has correct token {}?".format(token))
            if key is not None:
                self.verifyEqualOrFailCase(key, account_pushed.key,
                                           "Pushed account has correct key {}?".format(key))
            if nickname is not None:
                self.verifyEqualOrFailCase(nickname, account_pushed.nickname,
                                           "Pushed account has correct nickname {}?".format(nickname))
            if deleted:
                self.verifyTrueOrFailCase(account_pushed.deleted,
                                          "Pushed account is deleted?")

        finally:
            if delete_all_accounts:
                if muse_household_id is None:
                    muse_household_id = self.zp_designated.muse_hhid
                self.mock_lechmere.delete_service_accounts(muse_household_id)
