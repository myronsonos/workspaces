    def _delta(self, num_sec, leeway=0):
        """
        Calculate the delta programmatically. The backoff/retry-after occurs
        in +- 25%, and we allow for additional leeway to account for latency.

        We do not allow the delta to go negative, so we take the MAX of the
        delta and 0.

        :param int num_sec: Number of seconds
        :param int leeway: Additional leeway to allow as a percentage
            of num_sec (default 0)

        :return float: Delta
        """
        return max(num_sec * leeway, 0)

    def _test_bad_connection(self, zp, wait=True,
        wait_timeout=70, expected_ws_close_code=None, expected_last_http_status=None,
        expected_consecutive_failures=None):
        """
        Change the MLC policy by hitting the appropriate endpoint and verify
        the next bad connection attempt shows:
            1) The expected log messages in anacapa.trace (if specified)
            2) The expected websocket close code on the cloud status page (if specified).
            3) The expected number of auth failures on the cloud status page. This
            will only be checked if expected_ws_close_code is specified (we expect
            the websocket to close)

        :param zp: ZonePlayer to test
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`
        :param str url: URL to get to change the MLC policy.
        :param bool wait: Wait for a bad connection indicated by consecutivefailures?
        :param int wait_timeout: Timeout to wait for expected log msgs.
        :param int expected_ws_close_code: Expected close code for bad connection
            on the cloud status page.
        :param int expected_last_http_status: Expected last http status code
            for connection on the cloud status page.
        :param int expected_consecutive_failures: Number of expected connection
            failures. If None, will not check.
        """
        if wait:
            def wait_func(zp):
                try:
                    cs = zp.diag.get_cloud_as_dict()
                except Exception:
                    return False
                if not cs:
                    return False

                match = True
                if expected_consecutive_failures is None:
                    return True

                # Check consecutive failures
                if expected_consecutive_failures is not None:
                    match = match and cs.ConsecutiveFailures >= expected_consecutive_failures

                return match
            try:
                wait_until_true(lambda: wait_func(zp), timeout_seconds=wait_timeout,
                                iteration_delay=1)
            except TimeoutError:
                self.fail("ZP <{}> timed out waiting for connection failure cnt ({})"
                          "CloudStatus: <{}>".format(zp,
                                                     expected_consecutive_failures,
                                                     zp.diag.get_cloud_as_dict()))
                return

        # Verify the player indicates it saw the appropriate error code
        # on the cloud status page if specified
        if expected_ws_close_code is not None:
            self.verify_auth_failures_values(
                zp,
                expected_last_websocket_code=expected_ws_close_code,
                expected_last_http_status=expected_last_http_status,
                expected_consecutive_failures=expected_consecutive_failures)

        # Verify the player is not connected
        self.verifyTrueOrFailCase(not zp.diag.get_cloud_connected(),
            "ZP <{}> indicates player failed to connect?".format(zp))

    def _verify_event(self, event, expected):
        """
        Verify two LechmereConnectionEvents match (more or less).
        Since the TTL can be unpredictable, allow the user to pass in
        a callable in the expected dictionary for any value. This
        function must take one argument (the value for the field)
        to verify.

        :param event: LMCtuple of event to check
        :type event: :obj:`LechmereConnectionTuple`
        :param expected: LMCtuple of expected event or list of LMCtuple's
            to check. If more than one event is supplied, the event verifier
            will ensure that one of them is present.
        :type expected: :obj:`LechmereConnectionTuple` or
            :obj:`list` of :obj:`LechmereConnectionTuple`

        :return bool: Event matches?
        """
        success = True
        event_dict = event._asdict()

        if isinstance(expected, list):
            expected_dict_list = [e._asdict() for e in expected]
        else:
            expected_dict_list = [expected._asdict()]

        for exp in expected_dict_list:
            for k in event_dict.keys():
                if k in exp.keys():
                    if callable(exp[k]):
                        success = exp[k](event_dict[k])
                    else:
                        success = event_dict[k] == exp[k]
                else:
                    success = False

                # Break out early if we failed somewhere
                if not success:
                    break

            # Break out early if we found one of the expected events
            # that match.
            if success:
                break

        return success

    def _verify_event_report(self, zp, expected_event):
        """
        Verify lechmereConnection event was reported with expected values.

        :param zp: ZonePlayer to test
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`
        :param expected_event: LMCtuple of expected event or list of LMCtuple's
            to check. If more than one event is supplied, the event verifier
            will ensure that one of them is present.
        :type expected_event: :obj:`LechmereConnectionTuple` or
            :obj:`list` of :obj:`LechmereConnectionTuple`
        """
        # Get logs
        logs = self.muse_session_mixin.cq_control.wait_for_filtered_log(
            timeout=30, flush=True, zp=zp, fn=lambda le: le['path'] == '/usageMetrics')

        # Get lechmereConnection events
        events = []
        for i in xrange(len(logs)):
            # Skip the event if it is not a UsageMetrics report. PlayerConfig
            # reports also sometimes appear.
            if 'UsageMetrics' not in logs[i]['body']:
                continue
            event = logs[i]['body']['UsageMetrics']['events']['event']
            # 'event' can either return a dictionary representing a single event
            # or a list of dictionaries for separate events. If we get a dictionary,
            # put it into a list to support handling both cases.
            if isinstance(event, dict):
                event = [event]

            for e in event:
                if 'name' in e:
                    if e['name'] == LECHMERE_CONN_EVENT:
                        # Use a dictionary comprehension to fill in a LMC tuple, ignore the value for
                        # 'retry' as it is for temporary debugging purposes and we do not need to validate it
                        found_event = {
                            entry['key']: entry['value']
                            for entry in e['entry']
                            if entry['key'] != "retry"}
                        events.append(LechmereConnectionTuple(**found_event))
                    else:
                        continue

                    self.logger.info("Found event: {}".format(found_event))

        # Find a matching event
        matching_events = filter(
            lambda e: self._verify_event(e, expected_event), events)

        self.verifyGreaterEqualOrFailCase(len(matching_events), 1,
            "Found at least 1 matching events? (found <{}>, "
            " expected one of <{}>, matching: <{}>)".
            format(events, expected_event, matching_events))

        """
        Get a list containing the standard events for a connection

        This is to handle accepting one or more events given the unpredictability
        of report/connection timings since the decoupling of the token retrieval
        from the cloudrequest thread.

        :param str status: Status of token request
        :param str reason: Reason for token request failed
        :param str code: HTTP response code for connection
        :param str uptimeMs: Expected connection uptime in milliseconds

        :returns: List of tuples containing acceptable events
        :rtype: :obj:`list` of :obj:`LechmereConnectionTuple`
        """
        return [LechmereConnectionTuple(
                    status=status, reason=reason,
                    code=code,
                    TTA=str(0),
                    uptimeMs=uptimeMs),
                LechmereConnectionTuple(
                    status=status, reason=reason,
                    code=code,
                    TTA=str(0),
                    uptimeMs=uptimeMs)]



[docs]    def verify_backoff(self, failure_mode, backoff_seq, expect_fail=True):
        """
        Verify the player follows the backoff specified by backoff seq.

        Typically this will be a list of 1-2 arguments matching the number
        of seconds for the first and second backoff, respectively, but
        the caller can specify as many arguments as they want.

        Should be called right after a failed connection.

        :param str failure_mode: String identified for whether or not
            this was an auth failure or connection failure.
        :param list backoff_seq: Sequence of integers matching the
            expected backoff (delta will be added automatically).
        :return:
        """
        count = 0
        for bc in backoff_seq:
            # Calculate the delta - the backoff is always randomized within
            # a +- 25% range. We allow for a backoff + 2 +- 50% range to account
            # for extra test time as this can be especially sensitive for the
            # first backoff (5 seconds)
            delta = self._delta(bc, leeway=.5) + 2
            # Get the cloud status page
            cs = self.test_zp.diag.get_cloud_as_dict()
            # Get the current number of connection failures
            current_failure_count = cs[failure_mode]
            # Verify the MillisecondsToNextConnect is roughly within the next
            # range for the current backoff.
            if cs['MillisecondsToNextConnect']:
                self.verifyAlmostEqualOrFailCase(bc * 1000,
                     cs['MillisecondsToNextConnect'], "MillisecondsToNextConnect "
                     "within backoff range?", delta=delta * 1000)
            else:
                if cs.State == CloudStatusValues.STATE_OPEN:
                    self.fail('Player <{}> already reconnected when should have backed off, CS: {}!'.
                              format(self.test_zp, cs))
                    break
            try:
                current_time = datetime.now()
                # Wait for the next failed connection attempt within the backoff window
                if expect_fail:
                    wait_until_true(lambda:
                                    self.test_zp.diag.get_cloud_as_dict()[failure_mode] >
                                    current_failure_count,
                                    iteration_delay=1, timeout_seconds=bc + delta)
                else:
                    self.wait_for_connected_and_registration_message_sent(self.test_zp,
                                                                          timeout_seconds=bc + delta)
            except TimeoutError:
                self.fail("<{}> did not attempt reconnection within expected "
                          "backoff window (backoff #: {}, num seconds: {}, type: {})".
                          format(self.test_zp, count, bc, failure_mode))
                break
            else:
                self.logger.info("<{}> attempted reconnection after {} seconds "
                    "(within backoff {} + delta {})".format(
                        self.test_zp,
                        (datetime.now() - current_time).total_seconds(),
                        bc, delta))
            count += 1

    def _close_connection_and_reconnect(self, zp, delay=None):
        """
        Close the connection from the server side with an optional
        delay before reconnecting.

        :param zp: ZonePlayer
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`
        :param int delay: (optional) Delay to force ZP to wait before reconnecting
        """
        # Disconnect ZP
        self.mlc.disconnect_zp(zp, delay=delay)
        # Wait for reconnect
        wait_until_true(lambda:
            zp.diag.get_cloud_as_dict()[
            'MillisecondsToNextConnect'] is not None,
                        iteration_delay=1)

        # Verify status page indicates closed connection
        self.verify_cloud_status_disconnected_values(
            zp, expected_cc=1, expected_last_websocket_close_code=4500)
        if delay:
            time_to_sleep = zp.diag.get_cloud_as_dict()[
                'MillisecondsToNextConnect'] / 1000.0

            self.logger.info("Sleeping <{}> seconds for next connect...".
                format(time_to_sleep))
            guaranteed_sleep(time_to_sleep)

        zp.diag.wait_for_cloud_connected()

        """
        Verify that the player reverts to the backoff schedule after getting

        # Verify LastRetryAfter is correct.
        cs = self.test_zp.diag.get_cloud_as_dict()
        self.verifyEqualOrFailCase(10, cs['LastRetryAfter'],
            "LastRetryAfter field matches expected?")
        consecutive_failures = cs.ConsecutiveFailures

        # Configure mlc to return 400 with no Retry-After
        self.mlc.set_upgrade_response(requests.codes.BAD)  # @UndefinedVariable
        wait_until_true(lambda: self.test_zp.diag.get_cloud_as_dict()['ConsecutiveFailures'] > consecutive_failures,

        cs = self.test_zp.diag.get_cloud_as_dict()
