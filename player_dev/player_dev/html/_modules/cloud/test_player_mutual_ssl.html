    sys.argv = cmdline_args
    return args

[docs]    def restart_mock(self, **opts):
        """
        Restart mock lechmere with different settings

        :param dict opts: Arguments for mock lechmere
        :return:
        """
        if hasattr(self, 'mock_lechmere') and self.mock_lechmere:
            self.mock_lechmere.shutdown()
        args = opts
        self.mock_lechmere = MPCSC(**args)


    @contextmanager
    def _restart_mock_cm(self, **opts):
        """
        Contextmanager for restarting mock lechmere with different settings.
        Will restart the mock with the desired settings, then restart it again
        with default settings when exiting the contextmanager.

        :param dict opts: Arguments for mock lechmere
        :param func mock_func: (optional) Function to call on the mock
            and set options once it starts.
        :return:
        """
        try:
            self.restart_mock(http_port=self.http_port,
                              ssl_port=self.ssl_port,
                              mutual_ssl_port=self.mutual_ssl_port,
                              **opts)
            yield
        finally:
            self.restart_mock(**self.mock_default_args)

    def _wait_and_verify_cloud_connected(self, zp, timeout_seconds=30):
        """
        Wait for ZP to connect

        :param zp: ZP
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`
        :param int timeout_seconds: Time to wait.
        :return:
        """
        try:
            zp.diag.wait_for_cloud_connected(timeout_seconds=timeout_seconds)
        except TimeoutError:
            pass
        # Verify player indicates cloud connected
        self.verifyTrueOrFailCase(zp.diag.get_cloud_connected(),
            "ZP <{}> successfully connected with registration cert?".format(zp))

    def _wait_and_verify_registration_message_sent(self, zp, timeout_seconds=10):
        """
        Wait for the player to send the registration message to the mock
        after connecting.

        :param zp: ZP
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`
        :param int timeout_seconds: Time to wait.
        :return:
        """
        try:
            wait_until_true(
                lambda: self.mock_lechmere.get_zp_registered(zp),
                iteration_delay=1,
                timeout_seconds=timeout_seconds)
        except TimeoutError:
            pass
        # Verify player sent registration message to the service.
        self.verifyTrueOrFailCase(self.mock_lechmere.get_zp_registered(zp),
            "ZP <{}> sent registration message?".format(zp))

    def _wait_and_verify_connection_failed(self,
                                           zp,
                                           verifier=None,
                                           timeout_seconds=45):
        """
        Wait for the player to attempt a connection and verify it fails
        to connect.

        :param zp: ZP
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`
        :param int timeout_seconds: Time to wait for connection attempt.
        :return:
        """
        if verifier is None:
            verifier = lambda: zp.diag.get_cloud_as_dict()['ConsecutiveFailures'] > 0

        verifier_str = ' '.join([l.strip() for l in inspect.getsource(verifier).splitlines()])

        try:
            wait_until_true(
                verifier,
                iteration_delay=5,
                timeout_seconds=timeout_seconds)
        except TimeoutError:
            self.fail("ZP <{}> did not pass verifier <{}> "
                      "within timeout <{}> (Cloud Status: <{}>)".format(zp,
                                                                        verifier_str,
                                                                        timeout_seconds,
                                                                        zp.diag.get_cloud_as_dict()))
        else:
            self.verifyFalseOrFailCase(zp.diag.get_cloud_connected(),
                "ZP <{}> failed to connect?".format(zp))

    def _verify_event_report(self, zp, expected_event, expected_quantity=1):
        """
        Verify event was reported with expected values.

        :param zp: ZonePlayer to test
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`
        :param expected_event: Tuple of expected event or list of Tuples
            to check. If more than one event is supplied, the event verifier
            will ensure that one of them is present.
        :type expected_event: :obj:`MutualSSLEvent` or
            :obj:`list` of :obj:`MutualSSLEvent`
        """
        # Get logs
        logs = self.muse_session_mixin.cq_control.wait_for_filtered_log(
            timeout=60, flush=True, zp=zp, fn=lambda le: le.get('path') == '/usageMetrics',
            expected=expected_quantity)

        # Get events
        events = []
        for i in xrange(len(logs)):
            # Skip the event if it is not a UsageMetrics report. PlayerConfig
            # reports also sometimes appear.
            if 'UsageMetrics' not in logs[i]['body']:
                continue
            event = logs[i]['body']['UsageMetrics']['events']['event']
            # 'event' can either return a dictionary representing a single event
            # or a list of dictionaries for separate events. If we get a dictionary,
            # put it into a list to support handling both cases.
            if isinstance(event, dict):
                event = [event]

            for e in event:
                if 'name' in e and e['name'] == SSL_HANDSHAKE_EVENT:
                    self.logger.info("Processing event: {}".format(e))
                    found_event = {entry['key']: entry.get('value') for entry in e['entry']}
                    events.append(MutualSSLEvent(**found_event))
                    self.logger.info("Found event: {}".format(found_event))

        # Find a matching event
        matching_events = filter(
            lambda e: len([k for k, v in e._asdict().items() if
                          expected_event._asdict().get(k) == v]) ==
                      len(expected_event._asdict().keys()), events)

        self.verifyGreaterEqualOrFailCase(len(matching_events), 1,
            "Found at least 1 matching events? (found <{}>, "
            " expected one of <{}>, matching: <{}>)".
            format(events, expected_event, matching_events))

    def _restart_mock_test(self,
                           zp,
                           verifier,
                           **mock_args):
        """
        Common steps for a test that needs to restart the mock with
        a different configuration. Will restart the mock again with the
        default configuration when returning.

        :param zp: ZP
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`
        :param func verifier: Function to verify pass/fail
        :param dict mock_args: kwargs for mock
        :return:
        """
        with self._restart_mock_cm(**mock_args):
            self.zp.bounce_anacapa_and_wait()
            if verifier is not None:
                verifier()

    def _get_cert_from_p12(self, cert_filename, passwd):
        """
        Dump certificate info.

        :param str cert_file:
        :param str passwd:
        :return:
        """
        p12 = crypto.load_pkcs12(open(cert_filename, 'rb').read(), passwd)
        return p12.get_certificate()

    def _get_player_cert(self, zp):
        """
        Get cert in pem format from zp

        :param zp: ZP
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`
        :return:
        """
        req = requests.get('http://{}:{}/status/regcert'.format(zp.host, zp.port))
        return crypto.load_certificate(crypto.FILETYPE_ASN1,
            b64decode(ET.fromstring(req.text).find("DeviceCertInfo").find("Cert").text))

[docs]    def test_refresh_certificate(self):
        """
        Verify the player schedules a refresh on 403.

        :return:
        """
        self._test_cert_info = None
        self._restart_mock_test(self.zp,
                                lambda: self._wait_and_verify_connection_failed(self.zp),
                                gateway=None,
                                ocsp_code=codes.forbidden)

        job_scheduled = lambda izp: int(ET.fromstring(
            requests.get('http://{}:{}/status/regcert'.format(izp.host, izp.port)).text).
                find("DeviceCertInfo").find("JobLastRun").text)

        try:
            wait_until_true(lambda: job_scheduled(self.zp) > 0,
                            iteration_delay=2, timeout_seconds=60)
        except TimeoutError:
            self.fail("status/regcert indicates cert refresh job scheduled on receiving refresh "
                      "from cloud API? (exp. job last scheduled)")

        self.zp.diag.start_zone_job(JOB_NAME_UPLOAD_EVENTS)
        time.sleep(30)  # wait for the player to post the event reports

        self._verify_event_report(self.zp,
                                  expected_event=MutualSSLEvent(cert=self.formatted_cert_serial,
                                                                uri=self.reg_uri_refresh,
                                                                response=str(codes.forbidden)))

    def test_no_ssl_session_resume_on_error_response(self):
        """
        :return:
        """
        self._test_cert_info = None
        try:
            # Return 404 on upgrade requests
            self.mock_lechmere.set_upgrade_response(404)
            self.zp.bounce_anacapa_and_wait()
            # Wait for the player to fail to connect a few times, this should return immediately
            # with the current implementation, as the player probably tried to connect before
            # advertising itself over SSDP.
            wait_until_true(lambda: self.zp.diag.get_cloud_as_dict().ConsecutiveFailures >= 2,
                            iteration_delay=1, timeout_seconds=20,
                            reason="Waiting for >=2 lechmere connection failures")

            # Get the list of SSL sessions
            logs = self.mock_lechmere.get_log(type="sessions")
            # Make sure at least two attempts were logged
            self.verifyGreaterEqualOrFailCase(len(logs), 2,
                "<{}> should have attempted at least 2 connections?".format(self.zp))

            for le in logs:
                self.verifyEqualOrFailCase("NEW", le.get("sessionStatus"),
                    "SSL session was not resumed on failure? <{}>".format(le))
        finally:
            self.mock_lechmere.reset_upgrade_response()
            # Bounce anacpa to reset backoff timer
            self.zp.bounce_anacapa_and_wait()
