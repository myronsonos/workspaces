def WebsocketMessageRaw(payload, isBinary=False, applyMask=True, doNotCompress=False, hybi=True):
    """
    Prepares the payload to be sent over the wire.

    :param payload: Buffer containing the websocket message to be prepared for
    transmission over the wire.
    :param isBinary: True for binary payload
    :param applyMask: Controls whether the websocket message will be masked
    :param doNotCompress: If True, the message will not be compressed
    :param hybi: Selects between HyBi and hixie-76
    :return: A wire format websocket message.
    """
    msg = PreparedMessage(payload, isBinary, applyMask, doNotCompress)
    return msg.payloadHybi if hybi else msg.payloadHixie

class WebsocketClientProtocol(WebSocketClientProtocol):
    """
    Websocket client protocol base. Override the onOpen() and onClose() methods so we can let
    our WebsocketClient know what state we're in.
    """
    def onOpen(self):
        """
        Override onOpen() (from IWebSocketChannel).
        onOpen() is called when the web socket channel is fully established and ready to send and
        receive websocket messages.
    """
        # Pass ourselves back to our factory's client.
        if self.factory.client:
            self.factory.client.onConnectionOpen(self)

    def onClose(self, wasClean, code, reason):
        """
        Override onClose() (from IWebSocketChannel).
        onClose() is called when the web socket channel is closed either cleanly or due to loss of the
        underlying TCP connection.
        """
        if self.factory.client:
            self.factory.client.onConnectionClosed(self, wasClean, code, reason)

class WebsocketMultiServerProtocol(WebSocketServerProtocol):
    """
    Websocket server protocol base. Handles updating a dictionary of connections
    whenever a new connection is made. This allows us to launch one server
    for all of the ZPs in a household.
    """
    def __init__(self):
        super(WebsocketMultiServerProtocol, self).__init__()
        self.connections = None

    def onConnect(self, request):
        super(WebsocketMultiServerProtocol, self).onConnect(request)
        # Peer comes in as: "ip4:<ip>:<port>" or "<ip>:<port>"
        # We want to work by IP
        self.logger.debug("Request peer <{}>".format(request.peer))
        if request.peer.startswith("ip4"):
            ip = request.peer.split(":")[1]
        else:
            ip = request.peer.split(":")[0]
        self.connections[ip] = self


def make_multi_server_protocol(protocol):
    """
    Builds a protocol and dictionary of websockets connected.

    :param protocol: Base protocol that we need references to
    :return: Tuple of a protocol class that can be used in a websocket server
    factory or sent as the basis for a MockLechmereServer and a dictionary that
    will hold protocol connections indexed by incoming IP address.
    :rtype: tuple
    """
    connections = dict()

    class Protocol(protocol, WebsocketMultiServerProtocol):
        def __init__(self):
            super(Protocol, self).__init__()
            self.connections = connections

    return Protocol, connections


class WebsocketServerProtocol(WebSocketServerProtocol):
    """
    Websocket server protocol base. Handles updating a reference on connect.
    Used with make_server_protocol.
    """
    def __init__(self):
        super(WebsocketServerProtocol, self).__init__()
        self._reference = None

    def onConnect(self, request):
        super(WebsocketServerProtocol, self).onConnect(request)
        if self._reference:
            self._reference(self)


def make_server_protocol(protocol):
    """
    Builds a protocol and reference to the connected protocol.

    :param protocol: Base protocol that we need a reference to.
    :return: Tuple of a protocol class that can be used in a websocket server
    factory or sent as the basis for a MockLechmereServer and a reference to the
    protocol that will be connected.
    :rtype: tuple
    """
    reference = Reference()

    class Protocol(protocol, WebsocketServerProtocol):
        def __init__(self):
            super(Protocol, self).__init__()
            self._reference = reference

    return Protocol, reference

class WebsocketClientFactory(WebSocketClientFactory):
    """
    WebsocketClientFactory overrides autobahn.twisted.websocket.WebSocketClientFactory
    so it can detect failed connections via the clientConnectionFailed() callback.
    """
    def __init__(self, websocketclient, url, **kwargs):
        """
        :param websocketclient is the WebsocketClient that instantiated us
        """
        super(WebsocketClientFactory, self).__init__(url, **kwargs)
        self.client = websocketclient
        self.protocol = websocketclient.PROTOCOL

    def clientConnectionFailed(self, connector, reason):
        WebSocketClientFactory.clientConnectionFailed(self, connector, reason)
        if self.client:
            self.client.onConnectionFailed(connector, reason)


"""
WebsocketClient is the Sonos-defined class that represents a single websocket connection.
It is designed to be used by test scripts, and it hides most (all?) of the twisted / autobahn
details from the test scripts.

Usage:
Subclasses must override the following attributes:
  ENDPOINT - the URL to connect to (TODO: fix this to get the endpoint from the SSDP
             message for players discovered on the LAN)
  PROTOCOL - a subclass of twisted.internet.protocol.ClientProtocol, which will be instantiated
             when the connection is made and which will be available via the 'protocol' attribute
             of the WebsocketClient object.
"""
class WebsocketClient(object):
    ENDPOINT = None
    PROTOCOL = None

    def __init__(self, component, auto_connect=True, wait=True,
                 params=None, is_secure=False, headers=None, protocols=None):
        """
        Initialize the websocket client
        :param component: Component the client connects to
        :type component: SonosZoneComponent
        :param auto_connect: Whether the connection should be connected
            automatically
        :param wait: Whether to wait for the connection to be made.
        :param params: Exposes the autobahn "params" keyword @see autobahn.websocket.util.create_url
        :param is_secure: Exposes the autobahn "isSecure" keyword @see autobahn.websocket.util.create_url
        :param dict headers: Headers for connection
        :param protocols: Exposes the autobahn "protocols" keyword @see \
        autobahn.websocket.protocol.WebsocketClientFactory()
        :return:
        """

        self.logger = logging.getLogger(make_log_name([__name__, self.__class__.__name__]))
        # Note: this class itself no long relies upon self.component.  However, I'm leaving it as is in case
        # any other code relies on this property of WebsocketClient.
        self.component = component
        self.is_secure = is_secure
        if not protocols:
            self.logger.warn("**Attempting to make a Muse v1 connection without a subprotocol**")

        # Cache the host and port here so we can safely access them from the reactor thread
        self._port = int(component.ssl_port if is_secure else component.port)
        self._host = component.host

        url = create_url(self._host, self._port,
                         isSecure=is_secure, path=self.ENDPOINT,
                         params=params)

        if headers is None:
            headers = {}

        if 'X-Sonos-Api-Key' not in headers:
            headers['X-Sonos-Api-Key'] = X_SONOS_API_KEY

        # Rember the api_key so that it can be compared by tests
        self.api_key = headers['X-Sonos-Api-Key']

        # note: creating a new WebsocketClientFactory for each WebsocketClient is non-standard for
        # twisted; however, we leverage this because the factory object is the one that gets notified
        # when a connection fails or is broken, and in our model the factory is 1:1 with the WebsocketClient,
        # so the factory notifies its WebsocketClient when those events occur (connectionFailed(),
        # connectionLost()).
        # Normally, twisted ProtocolFactories are  global singletons, so that all
        # connections (Protocol instances) share the same factory.
        self._factory = WebsocketClientFactory(self, url, headers=headers, protocols=protocols)

        # The _connection knows how to connect() and disconnect().  That's about it.
        # TODO: rename to _connector?
        self._connection = None
        # _deferredConnection is a Deferred that is called back once the
        # connection has been established (or has failed).
        self._deferredConnection = None
        # _deferredDisconnection is a Deferred that is called back once the async
        # disconnect process has completed.
        self._deferredDisconnection = None

        # _protocol is the instance of PROTOCOL associated with this connection.
        # This is set whenever we have an established connection to the server.
        # It is None whenever we do not have an established connection.
        self._protocol = None

        if auto_connect:
            self.connect()


    def _execute(self, func, *args, **kwargs):
        """
        Execute all autobahn- / reactor- / twisted- related calls through this interface.
        Reason:  All network I/O on the websocket must be performed on the
                 reactor thread.
        """
        import twisted.internet.threads
        return twisted.internet.threads.blockingCallFromThread(reactor, func, *args, **kwargs)

    def _connect_once(self):
        # Attempt fresh connect
        # TADC: TODO: can this be simplified by using twisted 'endpoints' (added in Twisted 10.1)?
        if not self._connection:
            if self.is_secure:
                self._connection = reactor.connectSSL(    # @UndefinedVariable (connectSSL())
                    self._host, self._port, self._factory, ssl.CertificateOptions())
            else:
                self._connection = reactor.connectTCP(    # @UndefinedVariable (connectTCP())
                    self._host, self._port, self._factory)
        # Attempt reconnect after disconnect
        elif self.is_disconnected:
            self._connection.connect()
        # Warn about attempting connect while already connected
        else:
            self.logger.warning("Call to connect aborted because in {} state"
                            "".format(self._connection.state))

    def connect(self):
        """
        Connect the websocket and block until the connection either succeeds or there is an
        error or the connection attempt times out.
        """
        # Use _execute() to ensure that the connection is established on the reactor thread.
        self._execute(self._connect)
        return self.is_connected

    def _connect(self):
        if self._deferredConnection is not None:
            raise Exception("_connect: outstanding deferred operation in {}".format(self.__class__))

        self._connect_once()
        # This is the async way to do things in the twisted / reactor model.
        # Create a deferred object and return it.  This is a promise that the connection
        # process will eventually complete; when it does, the deferred object will be
        # used to deliver the result to the caller.
        d = defer.Deferred()
        # Hold onto this deferred so we can call its callback() or errback() method when
        # the async connection completes (successfully or with an error).
        self._deferredConnection = d
        return d

    def disconnect(self):
        """
        Disconnect, and block if necessary until the connection is completely torn down.
        """
        # Use _execute() to ensure that the disconnection is established on the reactor thread.
        r = self._execute(self._disconnect)
        return r

    def _disconnect(self):
        """
        Internal method that performs the disconnection on the reactor thread.
        :return None if already disconnected, or a Deferred instance if the disconnection will
                be asynchronous.
        """
        # IMPORTANT: do NOT set self._protocol to None here; instead, wait for onConnectionClosed()
        # to be called, which will clear out self._protocol.
        if self._deferredConnection is not None:
            raise Exception("_disconnect: outstanding deferred operation in {}".format(self.__class__))

        # If we have a connection, wait for onConnectionClosed() to be called.
        if self.is_connected:
            # Send a close frame over the websocket connection. This will shut down the
            # connection cleanly. Otherwise, the close frame is not transmitted and an
            # error is propogated to onConnectionClosed.
            #
            # We don't currently have any tests that need to tear down a connection uncleanly,
            # but we can add an unclean version of _disconnect in the future if needed.
            self._protocol.sendClose()

            d = defer.Deferred()
            # Hold onto this deferred so we can call its callback() or errback() method when
            # the async disconnection completes.
            self._deferredDisconnection = d
            return d
        else:
            # Just in case we are in the process of trying to connect, call disconnect().  It is safe
            # to call this in any state.
            self._connection.disconnect()
            # If we don't have a connection, we're done.  return nothing (None)
            return

    def onConnectionOpen(self, protocol):
        """
        onConnectionOpen() is called when the websocket is ready for sending messages
        :param r: a WebsocketClientProtocol object that was connected (or a subclass)
        :return:
        """
        self._protocol = protocol

        if self._deferredConnection:
            if isinstance(protocol, WebsocketClientProtocol):
                self._deferredConnection.callback(protocol)
                self._deferredConnection = None
            else:
                raise Exception("unexpected type for 'r' in {}".format(self.__class__))

    def onConnectionFailed(self, connector, reason):
        """
        onConnectionFailed() is called when we fail to establish a websocket connection
        :param reason is an Exception or twisted.python.failure.Failure object (or subclass thereof)
        :return:
        """
        self.logger.warning("connection failed: {}".format(reason))
        if self._deferredConnection:
            # note: don't call errback() here - connect() is not designed to raise an exception, and
            # calling errback() would raise an exception.
            self._deferredConnection.callback(None)
            self._deferredConnection = None

    def onConnectionClosed(self, protocol, wasClean, code, reason):
        """
        onConnectionClosed() is called when a connection closes, for any reason (clean or unclean).
        :param r is an Exception object (or subclass thereof)
        :return:
        """
        hadProtocol = self._protocol != None
        self._protocol = None

        if not hadProtocol:
            errmsg = "websocket closed during connection sequence: " + reason
            self.logger.warning(errmsg)
            if self._deferredConnection:
                # note: don't call errback() here - connect() is not designed to raise an exception, and
                # calling errback() would raise an exception.
                self._deferredConnection.callback(None)
                self._deferredConnection = None
        else:
            if not wasClean:
                self.logger.warning("websocket closed uncleanly: " + reason)

        if self._deferredDisconnection:
            self._deferredDisconnection.callback(None)
            self._deferredDisconnection = None


    def wait_until_disconnected(self, retry_delay=.25, timeout_seconds=30):
        # TADC: TODO: because disconnect() now blocks until the connection is closed,
        # this is now a stub.  In a separate change, we should remove the two calls
        # to this method, and then remove this method.
        pass


    @property
    def protocol(self):
        """  TADC: why is this method necessary?  Who calls it?
        :return:
        :rtype: WebsocketProtocol
        """
        return self._protocol

    @property
    def is_connected(self):
        """
        :return True if a connection has been successfully established to the server.
        """
        return (self._protocol != None)

    @property
    def is_disconnected(self):
        """
        :return True if the connection was never established, or if it was closed either locally
                or by the remote server.
        """
        return not self.is_connected
