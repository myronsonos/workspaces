        """
        Perform GET using 'method', 'ep' (sec. reg. end point) and headers
        :param zp: sonos zone player on which to do GET

        :param str ep: Secure registration endpoint

[docs]    def do_put(self, zp, ep, ssl=True, data=None, verify=False):
        """
        Perform PUT using 'method', 'ep' (sec. reg. end point) and headers
        :param zp: sonos zone player on which to do PUT
        :type zp: :class:``~sonos.client.internal.SonosZoneComponent`
        :param str method: http or ssl  PUT method
        :param str ep: Secure registration endpoint
        :param dict headers: request header
        :param bool verify: Applicable for SSL, do ssl authentication
        """
        url = zp.diag._create_diagnostic_request_url(ep, ssl=ssl)
        if data:
            r = requests.put(url, data=json.dumps(data), verify=verify)
        else:
            r = requests.put(url, verify=verify)
        return r



[docs]    @combinatorial('generate_testbed_unique_devices')
    def test_end_point_account(self, zp):
        """
        Verify a GET using SSL on secure registration base end point is passes
        and returns sec. registration json with state and customer ID info
        :param zp: sonos zone player under test
        :type zp: :class:``~sonos.client.internal.SonosZoneComponent`
        """
        r = self.do_get(zp, EP_SECURE_REG_BASE)
        content = r.content
        self.logger.debug("{}: Content of response for GET on registration EP"
                          ": {}".format(zp, content))
        if not r.ok:
            self.fail("{}: Expect success on GET on registration EP".format(zp))
        else:
            try:
                result_dict = json.loads(content)
            except ValueError:
                self.logger.fail("Return value:{} not a json string"
                                  .format(content))
            else:
                self.verifyTrueOrFailCase(result_dict.has_key("registration"),
                                          "Expect ID in return json")
                value = result_dict["registration"]
                if isinstance(value, dict):
                    self.verifyTrueOrFailCase(value.has_key("state"),
                                              "Expect ID in return json")
                    self.verifyTrueOrFailCase(value.has_key("id"),
                                              "Expect ID in return json")
                else:
                    self.fail("Expect state and id keys in registration info")

    def test_endpoint_prep_transfer_mode(self, zp):
        """
        Verify players enters prep transfer sec. registaration state when
        performing an SSL PUT on registration endpoint with prep transfer
        data. For DOCK, boost, bridge this should fail.
        :param zp: sonos zone player under test
        :type zp: :class:``~sonos.client.internal.SonosZoneComponent`
        """
        prep_transfer_data = build_prep_transfer_data(zp, PREP_TRANSFER_TIMEOUT)
        url = zp.diag._create_diagnostic_request_url(EP_REGISTRATION, ssl=True)
        r = requests.put(url,
                         data = json.dumps(prep_transfer_data),
                         headers=VALIDATE_HEADERS,
                         verify=False
                         )
        if zp.modelNumber in ['BR200', 'WD100', 'ZB100']:
            self.verifyFalseOrFailCase(r.ok==True,
                                  "Expect put command to be successful")
        else:
            self.verifyTrueOrFailCase(r.ok==True,
                                  "Expect put command to be successful")
            try:
                wait_until_true(
                    lambda:zp.DeviceProperties.is_device_in_prep_transfer_mode())
            except TimeoutError:
                self.fail("{} : Not in prep transfer mode.".format(zp))
            else:
                self.logger.log(AssertionLevel.PASS,
                    "{} : in prep transfer mode.".format(zp))
            wait_until_true(
                lambda:not zp.DeviceProperties.is_device_in_prep_transfer_mode())
