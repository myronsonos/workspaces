[docs]    def setUpFixture(self):
        """
        Setup Fixture. Setup udp logger, initialize variables.
        """
        self.logger.info("{}: Setup Fixture.".format(self.__class__.__name__))
        #store list of devices
        self.device_list = [zp for zp in self.my_devices]

        # if any player in hh is in transfer mode, assume all in transfer mode
        # so clear the mode before going further.
        if self.device_list[0].DeviceProperties.is_device_in_account_transfer_mode():
            self.logger.info("HH in account transfer state. Clearing")
            self._clear_transfer_state_in_hh()

        # perform test on a devoce by platform, disable the rest of players to
        # avoid network traffic
        self.test_zones = [zp for zp in
            self.generators.generate_testbed_device_per_platform()]
        self.logger.info("Test Zones : {}".format(self.test_zones))
        self.test_zone = [zp for zp in self.test_zones if zp.playback_device][0]
        self.disable_all_devices_except(self.test_zones)
        self.disabled_devices = [zp for zp in self.device_list if zp not in
                                 self.test_zones]
        self.logger.info("Disabled Zones : {}".format(self.disabled_devices))
        self.non_boost_bridge_dock_devices = [zp for zp in self.test_zones if
            zp.modelNumber not in ["BR200", "ZB100", "WD100"]]
        self.boost_bridge_dock_devices = [zp for zp in self.test_zones if
            zp.modelNumber in ["BR200", "ZB100", "WD100"]]

        # define/initialize udplogger related variables.
        self.audio_fb_disabled = False
        self.enter_diag_mode = {}
        self.exit_diag_mode = {}
        self.enter_prep_mode = {}
        self.exit_prep_mode = {}
        self.enter_transfer_mode = {}
        self.transfer_mode_led_pattern = {}
        for zp in self.test_zones:
            self.enter_diag_mode[zp.serialNumber] = False
            self.exit_diag_mode[zp.serialNumber] = False

        # if device is secure register, unregister it.
        for zp in self.test_zones:
            if zp.DeviceProperties.is_device_secure_registered():
                zp.secure_deregister()
            if zp in self.non_boost_bridge_dock_devices:
                if hasattr(zp, 'AVTransport'):
                    zp.AVTransport.clear_play_state()
                    zp.AVTransport.clear_queue_and_wait()
        self._add_test_share()

    def _is_hh_led_pattern_consistent(self):
        """

    def _watch_udp_logger_for(self, zp, watch_string, callback):
        """
        Start watching for 'watch_string' and call 'callback'

        :param callback: callback
        :type callback: callable
        """
        zp.diag.watch_udp_log_for_string(watch_string,
                                         lambda msg: callback(msg, zp))

[docs]    def test_account_transfer_mode_diag_mode(self):
        """
        Verify diag mode is not disabled when in prep transfer mode
        :param zp: sonos zone player under test
        :type zp: :class:``~sonos.client.internal.SonosZoneComponent
        """
        for zp in self.non_boost_bridge_dock_devices:
            self._watch_udp_logger_for(
                zp, DIAG_MODE_START, self.cb_enter_diag_mode)
            self._watch_udp_logger_for(
                zp, DIAG_MODE_STOP, self.cb_exit_diag_mode)

        # boost, bridge dock don't have udplogger so check in anacapa trace.
        boost_bridge_dock_diag_start = {}
        boost_bridge_dock_diag_stop = {}
        for zp in self.boost_bridge_dock_devices:
            # number of diag mode start log messages in trace
            num_start_instances = len(zp.diag.find_all_in_anacapa_trace(
                                    DIAG_MODE_START))
            boost_bridge_dock_diag_start[zp.serialNumber] = num_start_instances
            # number of diag mode stop log messages in trace
            num_stop_instances = len(zp.diag.find_all_in_anacapa_trace(
                                    DIAG_MODE_STOP))
            boost_bridge_dock_diag_stop[zp.serialNumber] = num_stop_instances
        self.verifyTrueOrStop(self.put_hh_in_transfer_mode(),
                              "Expect HH in account transfer mode")
        # put all devices in diag mode
        run_concurrent_processes(lambda zp: zp.cli.button_diag(),
                                 self.test_zones)
        # allow 2 extra seconds for diag mode led pattern to start
        time.sleep(2)
        for zp in self.non_boost_bridge_dock_devices:
            self.logger.debug("{} : Entered diag mode : {}"
                .format(zp, self.enter_diag_mode[zp.serialNumber]))
        for zp in self.test_zones:
            if zp.modelNumber in ['BR200', 'ZB100', 'WD100']:
                self.verifyGreaterOrFailCase(
                    len(zp.diag.find_all_in_anacapa_trace(DIAG_MODE_START)),
                    boost_bridge_dock_diag_start[zp.serialNumber],
                    "{} Expect in diag mode".format(zp))
            else:
                self.verifyTrueOrFailCase(self.enter_diag_mode[zp.serialNumber],
                                          "{} Expect in diag mode".format(zp))

        #clear diag mode
        run_concurrent_processes(lambda zp: zp.cli.button_press(),
                                 self.test_zones)
        for zp in self.test_zones:
            if zp.modelNumber in ['BR200', 'ZB100', 'WD100']:
                wait_until_true(
                    lambda:len(zp.diag.find_all_in_anacapa_trace(DIAG_MODE_STOP))
                            > boost_bridge_dock_diag_stop[zp.serialNumber])
            else:
                wait_until_true(lambda:self.exit_diag_mode[zp.serialNumber])

    def test_account_transfer_mode_volume_button_disabled(self, zp):
        """
        Verify when in prep transfer mode, on buttoned players, vol- button is
        disabled. (join and sloppy join (vol+ only and play/pause only remain
        enabled to go into button transfer mode). On cap touch players, cap
        zones are disabled.
        :param zp: sonos zone player under test
        :type zp: :class:``~sonos.client.internal.SonosZoneComponent
        """
        if zp.modelNumber in CAP_TOUCH_DEVICES:
            self.audio_fb_disabled = False
            zp.diag.watch_udp_log_for_string(
                AUDIO_FEEDBACK_DISABLED,
                lambda msg: self.cb_audio_fb_disabled(msg, zp))
        # set initial volume on device
        ini_vol = 5
        zp.RenderingControl.SetVolume(ini_vol)
        try:
            wait_until_true(lambda:ini_vol==zp.RenderingControl.GetVolume(),
                            timeout_seconds=5)
        except TimeoutError:
            self.stop("{}: Initial volume not set.")
        self.start_playback(zp)
        if not zp.AVTransport.is_playing():
            self.stop("{}: Expected to be in playing state".format(zp))
        if not self.put_device_in_account_transfer_mode(zp, 'button_press'):
            self.stop_playback(zp)
            self.stop("{}: Expect device in account transfer mode".format(zp))
        zp.cli.volume_down()
        # wait 2 seconds to make sure driver action is completed
        time.sleep(2)
        self.verifyTrueOrFailCase(ini_vol==zp.RenderingControl.GetVolume(),
                                  "{}: Expect no change in volume".format(zp))
        if zp.modelNumber in CAP_TOUCH_DEVICES:
            self.verifyTrueOrFailCase(self.audio_fb_disabled,
                                      "{}: Expect audio feedback disabled"
                                      .format(zp))
            self.audio_fb_disabled = False
        zp.cli.volume_up()
        # wait a second to make sure driver action is completed
        time.sleep(2)
        self.verifyTrueOrFailCase(ini_vol==zp.RenderingControl.GetVolume(),
                                  "{}: Expect no change in volume".format(zp))
        if zp.modelNumber in CAP_TOUCH_DEVICES:
            self.verifyTrueOrFailCase(self.audio_fb_disabled,
                                      "{}: Expect audio feedback disabled"
                                      .format(zp))
            self.audio_fb_disabled = False
        if zp.AVTransport.is_playing():
            zp.cli.volume_mute()
            # wait couple second to make sure driver action is completed
            time.sleep(2)
            self.verifyTrueOrFailCase(zp.AVTransport.is_playing(),
                                      "{}: Expect playing state".format(zp))
            if zp.modelNumber in CAP_TOUCH_DEVICES:
                self.verifyTrueOrFailCase(self.audio_fb_disabled,
                                      "{}: Expect audio feedback disabled"
                                      .format(zp))
                self.audio_fb_disabled = False
        else:
            self.stop("{}: Expect playing state".format(zp))
        self.stop_playback(zp)

    def test_volume_button_restored_on_account_transfer_clear(self, zp):
        """
        Verify volume buttons functionality is restored after going out of
        account transfer mode.
        """
        self.audio_fb_enabled = False
        # set initial volume on device
        ini_vol = 5
        zp.RenderingControl.SetVolume(ini_vol)
        try:
            wait_until_true(lambda:ini_vol==zp.RenderingControl.GetVolume(),
                            timeout_seconds=5)
        except TimeoutError:
            self.stop("{}: Initial volume not set.")
        if not self.put_device_in_account_transfer_mode(zp, 'button_press'):
            self.stop("{}: Expect device in account transfer mode".format(zp))
        self._clear_transfer_state_in_hh()
        if zp.modelNumber in CAP_TOUCH_DEVICES:
            self._setup_udp_logger(zp)
            [zp.diag.set_diag_level(diagc, DIAGC_LVL) for diagc in DIAGCS]
            zp.diag.watch_udp_log_for_string(AUDIO_FEEDBACK_ENABLED,
                lambda msg: self.cb_audio_fb_enabled(msg, zp))
        self.start_playback(zp)
        if not zp.AVTransport.is_playing():
            self.stop("{}: Not playing stoping test".format(zp))
        zp.cli.volume_down()
        # wait a second to make sure driver action is completed
        time.sleep(2)
        self.verifyTrueOrFailCase(zp.RenderingControl.GetVolume() < ini_vol,
                                  "{}: Expect change in volume".format(zp))
        if zp.modelNumber in CAP_TOUCH_DEVICES:
            self.verifyTrueOrFailCase(self.audio_fb_enabled,
                                      "{}: Expect audio feedback enabled"
                                      .format(zp))
            self.audio_fb_enabled = False
        ini_vol = zp.RenderingControl.GetVolume()
        zp.cli.volume_up()
        # wait a second to make sure driver action is completed
        time.sleep(2)
        self.verifyTrueOrFailCase(zp.RenderingControl.GetVolume() > ini_vol,
                                  "{}: Expect change in volume".format(zp))
        if zp.modelNumber in CAP_TOUCH_DEVICES:
            self.verifyTrueOrFailCase(self.audio_fb_enabled,
                                      "{}: Expect audio feedback enabled"
                                      .format(zp))
            self.audio_fb_enabled = False
        if zp.AVTransport.is_playing():
            zp.cli.volume_mute()
            # wait a second to make sure driver action is completed
            time.sleep(2)
            self.verifyTrueOrFailCase(zp.AVTransport.is_paused(),
                                      "{}: Expect playing state".format(zp))
            if zp.modelNumber in CAP_TOUCH_DEVICES:
                self.verifyTrueOrFailCase(self.audio_fb_enabled,
                                      "{}: Expect audio feedback enabled"
                                      .format(zp))
                self.audio_fb_enabled = False
        else:
            self.stop("{}: Expect playing state".format(zp))
        self.stop_playback(zp)

    def test_account_transfer_mode(self, zp):
        """
        Account transfer mode test. While in prep transfer mode on a player
        verify all HH devices enter transfer mode if join button is pressed
        :param zp: sonos zone player under test
        :type zp: :class:``~sonos.client.internal.SonosZoneComponent
        """
        boost_bridge_transfer_led = {}
        for device in self.boost_bridge_dock_devices:
            # WD100 has no logging for LED
            if not device.modelNumber == 'WD100':
                num_instances = len(device.diag.find_all_in_anacapa_trace(
                                    TRANSFER_MODE_LED))
                boost_bridge_transfer_led[device] = num_instances
        self.logger.info("{}: Putting in account transfer mode".format(zp))
        if self.put_device_in_account_transfer_mode(zp):
            try:
                wait_until_true(lambda:self._is_hh_zps_in_transfer_state(),
                                timeout_seconds=10, iteration_delay=1)
            except TimeoutError:
                self.fail("All HH devices not in transfer mode")
            else:
                self.logger.log(AssertionLevel.PASS,
                                "HH devices in transfer mode")
                try:
                    wait_until_true(lambda:self._is_hh_led_pattern_consistent(),
                                timeout_seconds=10, iteration_delay=1)
                except TimeoutError:
                    self.fail("Some/all HH zps not showing LED transfer pattern")
                else:
                    self.logger.log(AssertionLevel.PASS,
                                    "HH zps showing transfer mode LED pattern")
                    for device in boost_bridge_transfer_led:
                        self.verifyGreaterOrFailCase(
                            len(device.diag.find_all_in_anacapa_trace(
                            TRANSFER_MODE_LED)),
                            boost_bridge_transfer_led[device],
                            "{}: Expect transfer mode  LED pattern"
                            .format(device))
        else:
            self.fail("Failed to put {} in account transfer mode".format(zp))

        """

                "{}: Expect 811 error code when setting R_AccountTransferMode"
