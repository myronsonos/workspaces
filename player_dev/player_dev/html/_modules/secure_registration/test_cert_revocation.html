[docs]    def is_sonos_cert_id_consistent(self, zp):
        """
        Check if customer ID on status/registration page and on system settings match
        :param zp: sonos zone player for which to check customer id
        :type zp: :class:``~sonos.client.internal.SonosZoneComponent`
        return bool: Is the information same in both places.
        """
        cust_id = zp.SystemProperties.customer_id
        data = zp.http.getURL("http://{}:1400/status/registration".format(zp.ip))
        data = XMLAccessor(data)
        status_cust_id = data.Registration.CustomerID.value
        self.logger.info("{}: System settings CustomerID: {}"
                         .format(zp, cust_id))
        self.logger.info("{}: Status pg. CustomerID: {}"
                         .format(zp, status_cust_id))
        return cust_id == status_cust_id



[docs]    def get_registration_token_and_sonosID(self, zp,
                                           scope=OAUTH_SCOPE):
        """
        Get registration token and sonos id from server
        :param zp: sonos zone player for which to get registration token
        :type zp: :class:``~sonos.client.internal.SonosZoneComponent`
        :return: Registration token and sonos ID tuple
        :rtype: :`tuple`
        """
        oauth_optuple = OAuth.get_oauth_token(username=OAUTH_USER,
                                              password=OAUTH_PW,
                                              scope=scope)
        self.logger.debug("{}: oauth_optuple: {}".format(zp, oauth_optuple))
        access_token = oauth_optuple.token
        owner = oauth_optuple.owner
        registration_token = PlayerRegistration.get_init_token(access_token,
                                                               owner,
                                                               zp.hhid,
                                                               zp.serialNumber)
        return (registration_token, owner)

[docs]    def secure_register_zp(self, zp, id=None, use_account_transfer=True):
        """
        Secure register zp, put player in transfer mode is set to True
        :param zp: sonos zone player to register
        :type zp: :class:``~sonos.client.internal.SonosZoneComponent`
        :param str id: Sonos id to use when registering else use from the
            server
        :param bool use_account_transfer: Put player in account transfer
            before registering ?
        :return bool: Secure registration successfull.
        """
        self.logger.info("{}: Secure registering. Putting in account transfer"
                         " :{}".format(zp, use_account_transfer))
        # set all diagcs in case they are wiped due to qa_reset or reboots
        [zp.diag.set_diag_level(diagc, DIAGC_LVL) for diagc in DIAGCS]
        self.logger.info("Existing R_Customer ID: {}"
                         .format(zp.SystemProperties.customer_id))
        wait_until_true(lambda:zp.has_services())
        registration_token, owner = self.get_registration_token_and_sonosID(zp)
        if use_account_transfer:
            self.logger.info("{}: Putting in transfer mode".format(zp))
            self.put_device_in_account_transfer_mode(zp)
        if id:
            owner = id
        self.logger.info("{} : Now registering using ID: {}".format(zp, owner))
        zp.diag.secure_register(registration_token, owner)
        self.logger.info("{}: Sec. registration request.HTTP response code {}"
                         .format(zp, zp.http.last_status_code))
        try:
            wait_until_true(lambda:
                zp.DeviceProperties.is_device_secure_registered(),
                timeout_seconds=30)
        except TimeoutError:
            diagid = zp.ZoneGroupTopology.SubmitDiagnostics()
            self.logger.error("{}: Sec. Registration Failed. DIAG : {}"
                .format(zp, diagid))
            return False
        else:
            sonos_cert_id = zp.SystemProperties.customer_id
            self.logger.info("SonosCert ID: {}".format(sonos_cert_id))
            self.logger.info("Owner ID : {}".format(owner))
            if sonos_cert_id == owner:
                return True
            else:
                self.logger.error("{}: SonosCertID and owner ID mismatch"
                                  .format(zp))
                return False

        """
        Get customer_id and RegState from status/registration page

        :type zp: :class:``~sonos.client.internal.SonosZoneComponent`
        :return: customer id and reg_state info

    def _revoke_cert(self, zp):
        """
        Revoke registration certificate
        :param zp: sonos zone player of which to revoke the cert
        :type zp: :class:``~sonos.client.internal.SonosZoneComponent`
        :return bool: Was call to revoke cert successful.
        """
        self.logger.info("Revoking cert for player: {}".format(zp))
        dereg_token = self.get_de_registration_token(zp)
        url = REG_SERVER_URL.format(zp.hhid, zp.serialNumber)
        authorization_value = "Bearer {}".format(dereg_token)
        headers = {"accept": "application/json",
                   "Authorization": authorization_value}
        response = requests.delete(url, headers=headers, verify=False)
        self.logger.info("Cert revoke return code: {}"
                         .format(response.status_code))
        return response.status_code == OK_NO_CONTENT_RESPONSE_CODE


[docs]    def test_cert_revocation_on_player(self):
        """
        Verify secure registration of an unregistered player passes
        if not using account transfer mode
        """
        # secure register device
        if not self.secure_register_zp(self.test_zone):
            self.stop("{}: Secure registration failed".format(self.test_zone))
        else:
            cust_id, _ = self.get_customer_id_and_reg_state_from_status(
                            self.test_zone)
            self.logger.info("{}: Secure registration OK. Customer ID: {}"
                             .format(self.test_zone, cust_id))
        # revoke cert on registration server
        if not self._revoke_cert(self.test_zone):
            self.stop("{}: Cert revocation call registration server failed."
                      .format(self.test_zone))
        self.logger.info("{}: Rebooting..".format(self.test_zone))
        self.test_zone.reboot()
        [self.test_zone.diag.set_diag_level(diagc, DIAGC_LVL)
            for diagc in DIAGCS]
        # verify player gets legacy registered
        try:
            wait_until_true(lambda:
                int(self.test_zone
                 .DeviceProperties
                 .get_secure_registration_state()) == int(LEGACY_REG_STATE),
                iteration_delay=5,
                timeout_seconds=90)
        except TimeoutError:
            self.logger.info("Sec Reg State: {}"
                .format(self.test_zone.DeviceProperties.get_secure_registration_state()))
            diag_id = self.test_zone.ZoneGroupTopology.SubmitDiagnostics()
            self.fail("{}: Device not legacy registered after cert revocation"
                      " Diag ID: {}".format(self.test_zone, diag_id))
        else:
            self.verifyEqualOrFailCase(cust_id,
                self.test_zone.SystemProperties.customer_id,
                "{}: Expect legacy reg. customer ID same as original cust_id:{}"
                .format(self.test_zone, cust_id))
