[docs]def generate_prep_transfer_hmac_digest(zp, timeout=DEFAULT_TIMEOUT):
    """
    Generate base64 encoded hmac digest to go into prep transfer mode
    :param zp: sonos zone player for which to generate hmac digest
    :type zp: :class:``~sonos.client.internal.SonosZoneComponent
    :param int timeout: Prep tranfer timeout
    return: hmac Digest
    rtype: :`str`
    """
    endian_type = NETWORK_ORDER
    data_format = 'I'
    packed_prep_state = struct.pack(endian_type+data_format,
                                    PREP_SEC_REG_STATE)

    packed_timeout = struct.pack(endian_type+data_format, timeout)
    hmac = hashlib.sha256()
    hmac.update(packed_prep_state)
    hmac.update(zp.serialNumber)
    hmac.update(zp.hhid)
    hmac.update(packed_timeout)
    hmac.update(KEY)
    hmac_digest = base64.b64encode(hmac.digest())
    return hmac_digest

    """
    Build prep transfer data
    :param zp: sonos zone player for which to build data
    :type zp: :class:``~sonos.client.internal.SonosZoneComponent
    :param int timeout: Prep tranfer timeout
    return: prep_transfer_data

    prep_transfer_values = []

[docs]class DeviceInPrepTransferMode(object):
    """
    Context manager for putting device in PrepTransfer mode when performing
    secure registration operations. Raises exception when prep transfer mode
    timeout (30 seconds) expires
    """
    def __init__(self, zp, timeout=DEFAULT_TIMEOUT):
        """
        :param zp: Sonos zone component which is put in prep transfer mode
        :type zp: :class:`~sonos.client.internal.SonosZoneComponent`
        :param int timeout: seconds enabled for processing the block under
            our context manager. Defaul timeout = 30sec.
        """
        self._timeout = timeout
        self._zp = zp
        self.prep_transfer_data = build_prep_transfer_data(self._zp,
                                                           self._timeout)
        self.url = EP_REGISTRATION.format("https",
                                          self._zp.ip,
                                          self._zp.ssl_port
                                          )

    def __enter__(self):
        """
        Put the device in prep transfer mode and setup signaling on timeout
        """
        r = requests.put(self.url,
                         data = json.dumps(self.prep_transfer_data),
                         headers=VALIDATE_HEADERS,
                         verify=False
                         )
        self.setup_timeout()

    def __exit__(self, exc_type, value, traceback):
        """
        Raise exception if PrepTransfer timeout occurs before job execution else
        cancel alarm signal on timeout
        """
        if exc_type is PrepTransferTimeout:
            return False
        else:
            self.cancel_timeout()
            return True


[docs]    def handle_timeout(self, signum, frame):
        raise PrepTransferTimeout('Job execution time exceeded PrepTransfer '
                                  '{} seconds timeout'.format(self._timeout))



[docs]    def setup_timeout(self):
        """
        Sets up an alarm signal and a signal handler that raises
        a PrepTransferTimeout after the timeout amount.
        """
        signal.signal(signal.SIGALRM, self.handle_timeout)
        signal.alarm(self._timeout)



[docs]    def cancel_timeout(self):
        """
        Removes signaling handling setup previously if job execution completes
        before PrepTransfer timeout is hit.
        """
        signal.alarm(0)
        signal.signal(signal.SIGALRM, signal.SIG_DFL)

    def _setup_udp_logger(self, zp):
        """
        Setup udp logger and pass in callbacks for strings we are looking for
        :param zp: sonos zone player for which to setup udp logger
        :type zp: :class:``~sonos.client.internal.SonosZoneComponent`
        """
        self.logger.info("Setting up udp logger on : {}".format(zp))
        try:
            zp.diag.start_udp_log()
        except SocketError as e:
            self.logger.critical("UDPLogger port error: {}".format(e))
            self.stop_suite()
        zp.diag.watch_udp_log_for_string(
            PREP_MODE_STATE_CHANGE,
            lambda msg: self.cb_prep_mode_state_change(msg, zp))
        zp.diag.watch_udp_log_for_string(
            ENTER_TRANSFER_MODE, lambda msg: self.cb_enter_transfer_mode(msg, zp))
        zp.diag.watch_udp_log_for_string(
            TRANSFER_MODE_LED, lambda msg: self.cb_transfer_mode_led(msg, zp))


[docs]    def cb_prep_mode_state_change(self, msg, zp):
        """
        Call back when detecting enter prep transfer mode string
        :param str msg: Line containing watch string
        :param zp: sonos zone player for which search string found
        :type zp: :class:``~sonos.client.internal.SonosZoneComponent
        """
        self.logger.info("Detected on {}: {}".format(zp, msg))
        prep_state_pattern = re.compile('(\d)$')
        match = prep_state_pattern.search(msg)
        if match:
            reg_state = int(match.group(1))
        if reg_state == 6:
            self.enter_prep_mode[zp.serialNumber] = True
        else:
            self.exit_prep_mode[zp.serialNumber] = True

    def _is_hh_zps_in_transfer_state(self):
        """
        Is HH zps in transfer mode. Return True on success, false on failure
        :return: Is the hh zps in account transfer mode ?
        :rtype: :obj:`bool`
        """
        for zp in self.my_devices:
            if not zp.DeviceProperties.is_device_in_account_transfer_mode():
                self.logger.debug("{}: not yet in transfer state".format(zp))
                return False
        return True

    def _clear_transfer_state_in_hh(self):
        """
        Clear account transfer mode individually on players in HH
        """
        self.logger.info("Clearing tranfer mode from household devices.")
        zps = self.my_devices[:]
        run_concurrent_processes(lambda zp: zp.cli.stop_anacapa_nicely(), zps)
        run_concurrent_processes(lambda zp: zp.cli.clear_transfer_mode(), zps)
        run_concurrent_zone_player_processes(
            SonosZoneComponent.start_anacapa,
            zps)


[docs]    def put_device_in_account_transfer_mode(self, zp,
                                             button_action='button_press'):
        """
        Put device in account transfer mode
        :param zp: sonos zone player under test
        :type zp: :class:``~sonos.client.internal.SonosZoneComponent
        :param str button_action: Type of button press to be done on device
        :return: Is the device in account transfer mode ?
        :rtype: :obj:`bool`
        """
        with DeviceInPrepTransferMode(zp):
            try:
                wait_until_true(
                    lambda:
                        zp.DeviceProperties.is_device_in_prep_transfer_mode(),
                        timeout_seconds=15)
            except TimeoutError:
                self.stop("{}: Did not enter prep mode. Current state: {}"
                    .format(zp,
                            zp.DeviceProperties.get_secure_registration_state()))
                return False
            else:
                self.logger.info("{}: Entered prep mode".format(zp))
            attr = getattr(zp.cli, button_action, None)
            if attr:
                attr()
            else:
                self.logger.error("Button action {} not available"
                                  .format(button_action))
                return False
            try:
                wait_until_true(
                    lambda:
                        zp.DeviceProperties.is_device_in_account_transfer_mode(),
                        timeout_seconds=10)
            except TimeoutError:
                self.logger.error("{}: Did not enter transfer mode".format(zp))
                return False
            else:
                return True
