    def _get_time_spent_in_prep_mode(self, zp, start_time):
        """
        Meausure time spent in prep mode on zp
        :param zp: sonos zone player on which to measure
        :type zp: :class:``~sonos.client.internal.SonosZoneComponent
        :param start_time: start time of prep transfer mode
        :type start_time: :'float`
        :return: time in seconds
        :rtype: :obj:`int`
        """
        wait_time = 0
        while (zp.DeviceProperties.get_state_variable('SecureRegState')==6):
            time.sleep(1)
            wait_time +=1
            if wait_time > DEFAULT_TIMEOUT+5:
                self.stop("{}: Did not exit prep mode after {} secs"
                          .format(zp, (DEFAULT_TIMEOUT+5)))
        elapsed_time = time.time() - start_time
        return int(elapsed_time)

    def test_prep_transfer_mode(self, zp, vector):
        """
        Prep transfer mode tests
        1. Verify hitting preptransfer url via http results in HTTP _NOT_FOUND
        2. Verify hitting preptransfer url via https->POST results in
            HTTP_NOT_FOUND (we support only PUT)
        3. Verify hitting prep transfer url via https->PUT results in success.
        :param zp: sonos zone player under test
        :type zp: :class:``~sonos.client.internal.SonosZoneComponent
        :param vector: Test vector for above tests
        :type vector: :`tuple`
        """
        proto, method, exp_return_code = vector
        prep_transfer_data = self.get_prep_transfer_data(zp, 
                                                         timeout=MAX_TIMEOUT)
        if proto==SSL:
            url = PREP_TRANSFER_URL.format('https', zp.ip, zp.ssl_port)
        elif proto == HTTP:
            url = PREP_TRANSFER_URL.format('http', zp.ip, zp.port)
        if method==PUT:
            res = requests.put(url,
                               data = json.dumps(prep_transfer_data),
                               headers=VALIDATE_HEADERS,
                               verify=False
                               )
        elif method==POST:
            res = requests.post(url,
                                data = json.dumps(prep_transfer_data),
                                headers=VALIDATE_HEADERS,
                                verify=False
                                )
        self.logger.info("{}: Prep transfer return code: {}"
                          .format(zp, res.status_code))
        self.verifyTrueOrFailCase(res.status_code==exp_return_code,
                                  "{}: Expect {} return code."
                                  .format(zp, exp_return_code))
        if res.ok:
            try:
                wait_until_true(
                    lambda:zp.DeviceProperties.is_device_in_prep_transfer_mode(),
                    timeout_seconds=10)
            except TimeoutError:
                self.fail("{} : Not in prep transfer mode".format(zp))
            else:
                self.logger.log(AssertionLevel.PASS,
                                "{} : in Prep transfer mode".format(zp))
        wait_until_true(
            lambda:not zp.DeviceProperties.is_device_in_prep_transfer_mode())

    def test_prep_transfer_mode_timeout(self, set_timeout, expected_timeout):
        """
        Verify that if a "timeout" value is specified, we honor it as below:
        1. No value supplied, stay in Prep Transfer Mode for 30 seconds
        2. Timeout < 0, stay in Prep Transfer Mode for 30 seconds
        3. Timeout < 5, stay in Prep Transfer Mode for 5 seconds
        4. Timeout > 30, stay in Prep Transfer Mode for 30 seconds
        :param int set_timeout: Timeout to set, except when None
        :param int expected_timeout: Expected timeout set on device
        """
        prep_transfer_data = self.get_prep_transfer_data(self.test_zone,
                                                         timeout=set_timeout)
        url = PREP_TRANSFER_URL.format('https', self.test_zone.ip,
                                       self.test_zone.ssl_port)
        res = requests.put(url,
                           data = json.dumps(prep_transfer_data),
                           headers=VALIDATE_HEADERS,
                           verify=False
                           )
        self.verifyTrueOrStop(res.status_code==SUCCESS_CODE,
                              "Expect HTTPS PUT to be successful")
        # confirm test_zone entered prep transfer mode.
        try:
            wait_until_true(
                lambda:self.test_zone.DeviceProperties.is_device_in_prep_transfer_mode(),
                timeout_seconds=10)
            start_time = time.time()
        except TimeoutError:
            self.stop("{}: Did not enter prep mode".format(self.test_zone))
        #measure time spent in prep transfer mode.
        elapsed_time = self._get_time_spent_in_prep_mode(self.test_zone,
                                                         start_time)
        self.verifyAlmostEqualOrFailCase(elapsed_time, expected_timeout,
            "{}:Expected prep transfer timeout: {}"
            .format(self.test_zone, expected_timeout), delta=5)



[docs]    def test_prep_transfer_mode_reset(self):
        """
        Verify that if we are already in prep transfer mode and we receive a
        new prep transfer request then we remain in Prep Transfer Mode
        until the new timeout.
        """
        prep_transfer_data = self.get_prep_transfer_data(self.test_zone)
        url = PREP_TRANSFER_URL.format('https', self.test_zone.ip,
                                       self.test_zone.ssl_port)
        res = requests.put(url,
                           data = json.dumps(prep_transfer_data),
                           headers=VALIDATE_HEADERS,
                           verify=False
                           )
        self.verifyTrueOrStop(res.status_code==SUCCESS_CODE,
                              "Expect HTTPS PUT to be successful")
        try:
            wait_until_true(
                lambda:self.test_zone.DeviceProperties.is_device_in_prep_transfer_mode(),
                timeout_seconds=10)
        except TimeoutError:
            self.stop("{}: Did not enter prep mode".format(self.test_zone))
        # sleep for 10 seconds and re-issue preptransfer command
        time.sleep(10)
        res = requests.put(url,
                           data = json.dumps(prep_transfer_data),
                           headers=VALIDATE_HEADERS,
                           verify=False
                           )
        self.verifyTrueOrStop(res.status_code==SUCCESS_CODE,
                              "Expect HTTPS PUT to be successful")
        self.verifyTrueOrStop(
            self.test_zone.DeviceProperties.is_device_in_prep_transfer_mode(),
            "{}: Expect prep transfer state".format(self.test_zone))
        start_time = time.time()
        #measure time spent in prep transfer mode.
        elapsed_time = self._get_time_spent_in_prep_mode(self.test_zone,
                                                         start_time)
        self.verifyAlmostEqualOrFailCase(elapsed_time, DEFAULT_TIMEOUT,
                                         "Expected prep transfer timeout: {}"
                                         .format(DEFAULT_TIMEOUT),
                                         delta=5)

    def test_prep_transfer_mode_volume_button_disabled(self, zp):
        """
        Verify when in prep transfer mode, on buttoned players, vol- button is
        disabled. (join and sloppy join (vol+ only and play/pause only remain
        enabled to go into button transfer mode). On cap touch players, cap
        zones are disabled.
        :param zp: sonos zone player under test
        :type zp: :class:``~sonos.client.internal.SonosZoneComponent
        """
        if zp.modelNumber in CAP_TOUCH_DEVICES:
            self.audio_fb_disabled = False
            zp.diag.watch_udp_log_for_string(
                AUDIO_FEEDBACK_DISABLED,
                lambda msg: self.cb_audio_fb_disabled(msg, zp))
        # set initial volume on device
        ini_vol = 5
        zp.RenderingControl.SetVolume(ini_vol)
        try:
            wait_until_true(lambda:ini_vol==zp.RenderingControl.GetVolume(),
                            timeout_seconds=5)
        except TimeoutError:
            self.stop("{}: Initial volume not set.")
        self.start_playback(zp)
        with DeviceInPrepTransferMode(zp, 15):
            zp.cli.volume_down()
            # wait a second to make sure driver action is completed
            time.sleep(1)
            self.verifyTrueOrFailCase(ini_vol==zp.RenderingControl.GetVolume(),
                                      "{}: Expect no change in volume"
                                      .format(zp))
            if zp.modelNumber in CAP_TOUCH_DEVICES:
                self.verifyTrueOrFailCase(self.audio_fb_disabled, 
                                          "{}: Expect audio feedback disabled"
                                          .format(zp))
                self.audio_fb_disabled = False
            zp.cli.volume_up()
            # wait a second to make sure driver action is completed
            time.sleep(1)
            self.verifyTrueOrFailCase(
                ini_vol==zp.RenderingControl.GetVolume(),
                "{}: Expect no change in volume".format(zp))
            if zp.modelNumber in CAP_TOUCH_DEVICES:
                self.verifyTrueOrFailCase(self.audio_fb_disabled, 
                                          "{}: Expect audio feedback disabled"
                                          .format(zp))
                self.audio_fb_disabled = False
            if zp.AVTransport.is_playing():
                zp.cli.volume_mute()
                # wait a second to make sure driver action is completed
                time.sleep(1)
                self.verifyTrueOrFailCase(zp.AVTransport.is_playing(),
                                          "{}: Expect playing state".format(zp))
                if zp.modelNumber in CAP_TOUCH_DEVICES:
                    self.verifyTrueOrFailCase(self.audio_fb_disabled, 
                                            "{}: Expect audio feedback disabled"
                                            .format(zp))
                    self.audio_fb_disabled = False
            else:
                self.stop("{}: Expect playing state".format(zp))
            self.stop_playback(zp)
        # wait for prep transfer mode to timeout
        wait_until_true(
            lambda:not zp.DeviceProperties.is_device_in_prep_transfer_mode())

    def test_volume_button_restored_exit_from_prep_transfer(self, zp):
        """
        Verify volume buttons functionality is restored after going out of
        prep transfer mode.
        """
        self.audio_fb_enabled = False
        zp.diag.watch_udp_log_for_string(AUDIO_FEEDBACK_ENABLED,
                                lambda msg: self.cb_audio_fb_enabled(msg, zp))
        # set initial volume on device
        ini_vol = 5
        zp.RenderingControl.SetVolume(ini_vol)
        try:
            wait_until_true(lambda:ini_vol==zp.RenderingControl.GetVolume(),
                            timeout_seconds=5)
        except TimeoutError:
            self.stop("{}: Initial volume not set.")
        self.start_playback(zp)
        self.verifyTrueOrStop(self.put_device_in_prep_transfer(zp, 5), 
                              "Expect device to be in prep transfer")
        wait_until_true(lambda:self.exit_prep_mode[zp.serialNumber])
        wait_until_true(
            lambda:not zp.DeviceProperties.is_device_in_prep_transfer_mode())
        # wait extra seconds to make sure prep mode state is propagated
        time.sleep(2)
        zp.cli.volume_down()
        # wait a second to make sure driver action is completed
        time.sleep(2)
        
        self.verifyTrueOrFailCase(
            zp.RenderingControl.GetVolume() < ini_vol,
            "{}: Expect volume to go down 1 step".format(zp))
        ini_vol = zp.RenderingControl.GetVolume()
        if zp.modelNumber in CAP_TOUCH_DEVICES:
            self.verifyTrueOrFailCase(
                self.audio_fb_enabled, 
                "{}: Expect audio feedback enabled".format(zp))
            self.audio_fb_enabled = False
        zp.cli.volume_up()
        # wait a second to make sure driver action is completed
        time.sleep(2)
        self.verifyTrueOrFailCase(
            zp.RenderingControl.GetVolume() > ini_vol,
            "{}: Expect volume to go up 1 step".format(zp))
        if zp.modelNumber in CAP_TOUCH_DEVICES:
            self.verifyTrueOrFailCase(
                self.audio_fb_enabled, 
                "{}: Expect audio feedback enabled".format(zp))
            self.audio_fb_enabled = False
        if zp.AVTransport.is_playing():
            zp.cli.volume_mute()
            # wait few seconds to make sure driver action is completed
            time.sleep(3)
            self.verifyTrueOrFailCase(
                zp.AVTransport.is_paused(),
                "{}: Expect paused state".format(zp))
            if zp.modelNumber in CAP_TOUCH_DEVICES:
                self.verifyTrueOrFailCase(
                    self.audio_fb_enabled, 
                    "{}: Expect audio feedback enabled".format(zp))
                self.audio_fb_enabled = False
        else:
            self.stop("{}: Expect initial playing state".format(zp))
        self.stop_playback(zp)

    def test_prep_transfer_mode_diag_mode(self, zp):
        """
        Verify diag mode is not disabled when in prep transfer mode
        :param zp: sonos zone player under test
        :type zp: :class:``~sonos.client.internal.SonosZoneComponent
        """
        zp.diag.watch_udp_log_for_string(
            DIAG_MODE_START, lambda msg: self.cb_enter_diag_mode(msg, zp))
        zp.diag.watch_udp_log_for_string(
            DIAG_MODE_STOP, lambda msg: self.cb_exit_diag_mode(msg, zp))
        with DeviceInPrepTransferMode(zp):
            wait_until_true(lambda:self.enter_prep_mode[zp.serialNumber])
            wait_until_true(
                lambda:zp.DeviceProperties.is_device_in_prep_transfer_mode())
            zp.cli.button_diag()
            try:
                wait_until_true(lambda:self.enter_diag_mode[zp.serialNumber])
            except TimeoutError:
                self.fail("{}:Did not enter diag mode".format(zp))
        wait_until_true(
            lambda:not zp.DeviceProperties.is_device_in_prep_transfer_mode())
        wait_until_true(lambda:self.exit_prep_mode[zp.serialNumber])
        # give extra couple of seconds to make sure zp is out of prep mode
        time.sleep(2)
        zp.cli.button_press()
        wait_until_true(lambda:self.exit_diag_mode[zp.serialNumber])
        zp.diag.stop_watching_udp_log_for_string(DIAG_MODE_START)
        zp.diag.stop_watching_udp_log_for_string(DIAG_MODE_STOP)
