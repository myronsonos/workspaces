    """
    Test registration when adding a player to HH for following cases:

    1. Unregistered HH and add a factory_reset player. Expect-> unregistered HH ( 0 and blank)
    2. Unregistered HH and add a legacy_reg. player. Expect-> unregistered HH
    3. Unregistered HH and add a sec_reg player. Expect-> unregistered HH

    4. Legacy reg. HH and add factory_reset player. Expect-> Legacy reg. HH
    5. Legacy reg. HH and add legacy_reg player. Expect-> Legacy reg. HH
    6. Legacy reg. HH and add secure reg player. Expect-> Legacy reg. HH

    7. Secure reg. HH and add factory_reset player. Expect-> Legacy reg. HH
    8. Secure reg. HH and add legacy_reg player. Expect-> Legacy reg. HH
    9. Secure reg. HH and add secure reg player. Expect-> Legacy reg. HH

    ..note:
    1. The test needs account set up on www.sonos.com depending on which
       environment the test is run on. Example for stage, www-stage.sonos.com
    2. Bridge, Dock and Bridge are not included in this test
    3. The test uses env OAUTH_USER, OAUTH_PW which should be same as the one
       used in 1 above. Else env 'NODE_NAME@sonos.com' is used or auto1137ian
       Set all account and envs first before running the test.
    """

        """
        Setup Fixture
        """
        self.logger.info("Setup Fixture")
        # store copy of devices
        self.device_list = [zp for zp in self.my_devices]
        for zp in self.my_devices:
            if zp.DeviceProperties.is_device_in_account_transfer_mode():

        # pick any 2 devices for test.
        self.test_zones = random.sample(self.non_boost_bridge_dock_devices, 2)
        self.hh_zp = self.test_zones[0]
        self.hhid = self.hh_zp.hhid
        self.new_zp = self.test_zones[1]
        self.logger.info("Test zones selected: {}".format(self.test_zones))
        self.logger.info("HH player: {}".format(self.hh_zp))
        self.logger.info("New player: {}".format(self.new_zp))
        # stop anacapa on all devices other than our test devices
        self.disable_all_devices_except(self.test_zones)
        self.disabled_devices = [zp for zp in self.device_list if zp not in
                                 self.test_zones]
        #download netstart
        self.logger.info("Setting up netstart")
        self.netstart = NetstartTest(self.execution_configuration)

[docs]    def get_registration_token_and_sonosID(self, zp, username, password):
        """
        Get registration token and sonos id from server
        :param zp: sonos zone player for which to get registration token
        :type zp: :class:``~sonos.client.internal.SonosZoneComponent`
        :return: Registration token and sonos ID tuple
        :rtype: :`tuple`
        """
        oauth_optuple = OAuth.get_oauth_token(username=username,
                                              password=password,
                                              scope=OAUTH_SCOPE)
        self.logger.debug("{}: oauth_optuple: {}".format(zp, oauth_optuple))
        access_token = oauth_optuple.token
        owner = oauth_optuple.owner
        registration_token = PlayerRegistration.get_init_token(access_token,
                                                               owner,
                                                               zp.hhid,
                                                               zp.serialNumber)
        return (registration_token, owner)



[docs]    def secure_register_zp(self, zp, id=None,
                           user_acc=OAUTH_USER, user_acc_pw=OAUTH_PW,
                           use_account_transfer=True):
        """
        Secure register zp, put player in transfer mode is set to True
        :param zp: sonos zone player to register
        :type zp: :class:``~sonos.client.internal.SonosZoneComponent`
        :param str id: Sonos id to use when registering else use from the
            server
        :param bool use_account_transfer: Put player in account transfer
            before registering ?
        :return bool: Secure registration successfull.
        """
        self.logger.info("{}: Secure registering. Putting in account transfer"
                         " :{}".format(zp, use_account_transfer))
        # set all diags incase they are wiped due to qa_reset ot reboots
        [zp.diag.set_diag_level(diagc, DIAGC_LVL) for diagc in DIAGCS]
        self.logger.info("Existing R_Customer ID: {}"
                         .format(zp.SystemProperties.customer_id))
        wait_until_true(lambda:zp.has_services())
        try:
            registration_token, owner = self.get_registration_token_and_sonosID(
                                        zp, user_acc, user_acc_pw)
        except Exception as e:
            import traceback
            self.logger.error("{}: Error when getting oauth token or "
                              "registration token.")
            self.logger.error("Error: Args: {} Message: {}"
                              .format(e.args, e.message))
            self.logger.error("Exception traceback: {}"
                              .format(traceback.format_exc()))
            return False
        if use_account_transfer:
            self.logger.info("{}: Putting in transfer mode".format(zp))
            self.put_device_in_account_transfer_mode(zp)
        self.logger.info("{} : Now registering using ID: {}"
                         .format(zp, owner))
        zp.diag.secure_register(registration_token, owner)
        self.logger.info("{}: Sec. registration request.HTTP response code {}"
                         .format(zp, zp.http.last_status_code))
        try:
            wait_until_true(lambda:
                zp.DeviceProperties.is_device_secure_registered(),
                timeout_seconds=30)
        except TimeoutError:
            diagid = zp.ZoneGroupTopology.SubmitDiagnostics()
            self.logger.error("{}: Sec. Registration Failed. DIAG : {}"
                .format(zp, diagid))
            return False
        else:
            sonos_cert_id = zp.SystemProperties.customer_id
            self.logger.info("SonosCert ID: {}".format(sonos_cert_id))
            self.logger.info("Owner ID : {}".format(owner))
            if sonos_cert_id==owner:
                return True
            else:
                self.logger.error("{}: SonosCertID and owner ID mismatch"
                                  .format(zp))
                return False



[docs]    def get_customer_id_and_reg_state_from_status(self, zp):
        """
        Get customer_id and RegState from status/registration page
        :param zp: sonos zone player to query
        :type zp: :class:``~sonos.client.internal.SonosZoneComponent`
        :return: customer id and reg_state info
        :rtype: :`tuple`
        """
        data = zp.http.getURL("http://{}:1400/status/registration".format(zp.ip))
        data = XMLAccessor(data)
        cust_id = data.Registration.CustomerID.value
        reg_state = data.Registration.RegState.value
        return (cust_id, reg_state)

    def test_add_player_to_hh_test(self, hh_state, new_player_state):
        """
        Secure registrations states when adding a new player to hh
        """
        #setup hh_zp
        self.logger.info("Setting up HH ZP: {}".format(self.hh_zp))
        if hh_state in [LEGACY_REGISTERED, SECURE_REGISTERED]:
            self.legacy_register_zp(self.hh_zp)
        if hh_state==SECURE_REGISTERED:
            self.secure_register_zp(self.hh_zp)
        # setup new_zp before adding it to hh
        # even though it seems like we can reduce the if statements and
        # make concise, for clarity, setting up new_app separately for each
        # case
        self.logger.info("Setting up new ZP: {}".format(self.new_zp))
        if new_player_state==FACTORY:
            # no additional setup required as player is already in factory state
            pass

        elif new_player_state==UNREGISTERED:
            self.logger.info("{}: Creating new HH".format(self.new_zp))
            self.netstart.sonosnet_setup(self.new_zp)

        elif new_player_state==LEGACY_REGISTERED:
            self.logger.info("{}: Creating new HH".format(self.new_zp))
            self.netstart.sonosnet_setup(self.new_zp)
            self.logger.info("{}: Legacy registering".format(self.new_zp))
            self.legacy_register_zp(self.new_zp, id_=TEMP_LEGACY_SONOS_ID)

        elif new_player_state==SECURE_REGISTERED:
            self.logger.info("{}: Creating new HH".format(self.new_zp))
            self.netstart.sonosnet_setup(self.new_zp)
            #TODO legacy registration not required. Done here due to a bug
            self.logger.info("{}: Legacy registering".format(self.new_zp))
            self.legacy_register_zp(self.new_zp, id_=TEMP_LEGACY_SONOS_ID)
            # set overrides before secure registering new player
            self.set_string_overrides_and_bounce_anacapa([self.new_zp])
            self.logger.info("{}: Secure registering".format(self.new_zp))
            self.secure_register_zp(self.new_zp,
                                    user_acc=TEMP_OAUTH_USER,
                                    user_acc_pw=TEMP_OAUTH_PW)

        self.verifyTrueOrStop(self.add_new_zp_to_hh(self.hh_zp, self.new_zp),
            "Expect {} in hh".format(self.test_zones))
        self.hh_zp.settings.sync_sys_settings()
        hh_cust_id, hh_reg_state = (
            self.get_customer_id_and_reg_state_from_status(self.hh_zp))
        new_cust_id, new_reg_state = (
            self.get_customer_id_and_reg_state_from_status(self.new_zp))
        self.logger.info(
            "New player status page info : custID: {} RegState: {}".format(
                new_cust_id, new_reg_state))
        # verifications
        self.verifyIsNoneOrFailCase(new_cust_id,
                "{} :Expect customer id to be None".format(self.new_zp))
        # verifications
        if hh_state == UNREGISTERED:
            expected_reg_state = UNREGISTRED_REG_STATE
            expected_sonos_id = UNREGISTERED_SONOS_ID
        elif hh_state == LEGACY_REGISTERED:
            expected_reg_state = LEGACY_REG_STATE
            expected_sonos_id = LEGACY_SONOS_ID
        elif hh_state == SECURE_REGISTERED:
            expected_reg_state = LEGACY_REG_STATE
            expected_sonos_id = hh_cust_id

        try:
            wait_until_true(lambda:
                self.get_reg_state_from_status(self.new_zp) ==
                    expected_reg_state,
                iteration_delay=1, timeout_seconds=15)
        except TimeoutError:
            diagid = self.new_zp.ZoneGroupTopology.SubmitDiagnostics()
            self.logger.info("{}: Sec. Registration DIAG : {}"
                                 .format(self.new_zp, diagid))
            active_r_customer_id = self.hh_zp.SystemProperties.customer_id
            new_r_customer_id = self.new_zp.SystemProperties.customer_id
            self.logger.info("{}: Active player: R_CustomerID: {}"
                             .format(self.hh_zp, active_r_customer_id))
            self.logger.info("{}: New player: R_CustomerID: {}"
                             .format(self.new_zp, new_r_customer_id))
            self.fail("{}: Does not have expected registration state."
                "Current State: {}. Expected: {}".format(
                    self.new_zp,
                    self.get_reg_state_from_status(self.new_zp),
                    expected_reg_state))
        else:
            self.logger.log(AssertionLevel.PASS,
                "{}: Registration state {}. OK:".format(
                    self.new_zp,
                    expected_reg_state))

        self.verifyEqualOrFailCase(
            expected_sonos_id,
            self.new_zp.SystemProperties.customer_id,
            "Check expected syssettings customerid for {}".format(self.new_zp))
        if hh_state == SECURE_REGISTERED:
            # verify we can now securely register new_player without
            # going into ac transfer mode
            self.verifyTrueOrFailCase(
                self.secure_register_zp(self.new_zp,
                    use_account_transfer=False),
                "{}: Expect to securely registered without account transfer"
                .format(self.new_zp))
